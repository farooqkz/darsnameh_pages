ﺩﺭﺱ ﭘﻨﺠﻢ- ﺍﻫﻤﻴﺖ ﭘﺎﺭﺍﮔﺮﺍﻑ ﺑﻨﺪﯼ ﻭ ﻳﮑﺪﺳﺘﯽ ﺍﻓﻌﺎﻝ ﻭ ﺿﻤﺎﻳﺮ

ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﺍﺳﺘﻔﺎﺩﻩ ﺩﺭﺳﺖ ﺍﺯ ﻋﻼﻣﺖ‌ﻫﺎﯼ ﺳﺠﺎﻭﻧﺪﯼ ﺭﺍ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻳﻢ ﻭ ﻳﮏ ﻗﺎﻋﺪﻩ ﮐﻠﯽ ﺭﺍ ﮔﻔﺘﻴﻢ: ﻋﻼﻣﺖ‌ﻫﺎﯼ ﺳﺠﺎﻭﻧﺪﯼ ﺑﻪ ﮐﻠﻤﻪ ﻗﺒﻞ ﺍﺯ ﺧﻮﺩ ﻣﯽ‌ﭼﺴﺒﻨﺪ ﻭ ﺍﺯ ﮐﻠﻤﻪ ﺑﻌﺪ ﺍﺯ ﺧﻮﺩ ﻓﺎﺻﻠﻪ ﻣﯽ‌ﮔﻴﺮﻧﺪ. 

ﻧﮑﺘﻪ ﻣﻬﻢ ﺩﻳﮕﺮ ﺍﻳﻦ ﮐﻪ ﻻﺯﻡ ﺍﺳﺖ ﺩﺭ ﻧﻮﺷﺘﻪ‌ﻫﺎ ﺑﻴﻦ ﺗﻤﺎﻡ ﮐﻠﻤﺎﺕ ﻳﮏ ﻓﺎﺻﻠﻪ (ﻭ ﻧﻪ ﺑﻴﺸﺘﺮ ﻳﺎ ﮐﻤﺘﺮ) ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺑﻴﻦ ﻫﺮ ﺩﻭ ﭘﺎﺭﺍﮔﺮﺍﻑ ﺑﺎﻳﺪ ﻳﮏ ﺧﻂ ﻓﺎﺻﻠﻪ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺩﺭ ﻣﺘﻦ‌ﻫﺎﯼ ﺳﺌﻮﺍﻝ ﻭ ﺟﻮﺍﺑﯽ ﻫﻢ، ﻓﺎﺻﻠﻪ ﻫﺮ ﺳﺌﻮﺍﻝ ﺑﺎ ﺟﻮﺍﺏ ﺑﻌﺪ ﺍﺯ ﺁﻥ ﻳﮏ ﺧﻂ ﻭ ﻓﺎﺻﻠﻪ ﺟﻮﺍﺏ ﺑﺎ ﺳﺌﻮﺍﻝ ﺑﻌﺪ ﺍﺯ ﺁﻥ ﺩﻭ ﺧﻂ ﺍﺳﺖ.

ﺍﺯ ﺟﻤﻼﺕ ﮐﻮﺗﺎﻩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ

ﺯﻳﺒﺎﺗﺮ ﻭ ﻣﻨﺎﺳﺐ‌ﺗﺮ ﺍﺳﺖ ﮐﻪ ﺟﻤﻼﺕ ﻣﺎ، ﺩﺭ ﺣﺪ ﺍﻣﮑﺎﻥ ﮐﻮﺗﺎﻩ ﺑﺎﺷﺪ. ﺑﻌﺪ ﺍﺯ ﻧﻮﺷﺘﻦ ﺟﻤﻼﺕ ﺍﮔﺮ ﺩﺭ ﺑﺎﺯﺧﻮﺍﻧﯽ ﺁﻥ ﺭﺍ ﺑﻠﻨﺪ ﺍﺭﺯﻳﺎﺑﯽ ﮐﺮﺩﻳﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺁﻥ ﺭﺍ ﻣﺨﺘﺼﺮﯼ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ ﻭ ﺑﻪ ﺩﻭ ﻳﺎ ﭼﻨﺪ ﺟﻤﻠﻪ ﮐﻮﺗﺎﻩ ﺗﻘﺴﻴﻢ ﮐﻨﻴﺪ. ﺑﻪ ﻋﻼﻭﻩ، ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﻓﺎﺻﻠﻪ ﻓﺎﻋﻞ ﻳﺎ ﻣﺒﺘﺪﺍﯼ ﺟﻤﻠﻪ ﺑﺎ ﻓﻌﻞ ﺁﻥ، ﮐﻮﺗﺎﻩ ﺑﺎﺷﺪ.

ﻳﮑﯽ ﺍﺯ ﺭﺍﻳﺞ‌ﺗﺮﻳﻦ ﺍﺷﺘﺒﺎﻫﺎﺕ ﺩﺭ ﻧﻮﺷﺘﻪ‌ﻫﺎ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﻓﻌﺎﻝ ﺟﻤﻼﺕ ﻃﻮﻻﻧﯽ ﺑﺎ ﭼﻨﺎﻥ ﻓﺎﺻﻠﻪ ﺯﻳﺎﺩﯼ ﺍﺯ ﺍﺑﺘﺪﺍﯼ ﺟﻤﻠﻪ ﻣﯽ‌ﺁﻳﻨﺪ ﮐﻪ «ﺯﻣﺎﻥ» ﻭ «ﺷﺨﺺ» ﻓﺎﻋﻞ ﻳﺎ ﻣﺒﺘﺪﺍ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﻧﻬﺎﻳﺘﺎ ﻓﻌﻞ ﻧﺎﺩﺭﺳﺖ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺑﺮﻳﻢ.

ﻣﺜﺎﻝ: ﺍﻳﺮﺍﻥ٬ ﺍﻣﺎﺭﺍﺕ ﻣﺘﺤﺪﻩ ﻋﺮﺑﯽ ﺭﺍ ﺑﻪ ﺩﻟﻴﻞ ﺗﮑﺮﺍﺭ ﺍﺩﻋﺎﻫﺎﻳﺶ ﺩﺭﺑﺎﺭﻩ ﺟﺰﺍﻳﺮ ﺳﻪ ﮔﺎﻧﻪ ﺩﺭ ﺧﻠﻴﺞ ﻓﺎﺭﺱ ﮐﻪ ﻣﻄﺎﺑﻖ ﺗﻔﺎﻫﻢ ﻧﺎﻣﻪ‌ﺍﯼ ﻣﻴﺎﻥ ﺍﻳﺮﺍﻥ ﻭ ﺍﻧﮕﻠﻴﺲ ﺗﺤﺖ ﺣﺎﮐﻤﻴﺖ ﺍﻳﺮﺍﻥ ﺍﺳﺖ٬ ﺗﻬﺪﻳﺪ ﺑﻪ ﻗﻄﻊ ﺭﺍﺑﻄﻪ ﮐﺮﺩﻩ ﺍﺳﺖ.

ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﮔﻮﻧﻪ ﺟﻤﻼﺕ ﻣﺮﮐﺐ ﺭﺍ ﺩﺭ ﭼﻨﺪ ﺟﻤﻠﻪ ﺳﺎﺩﻩ‌ﺗﺮ ﺑﻴﺎﻭﺭﻳﺪ ﺗﺎ ﺧﻮﺍﻧﺪﻥ ﻭ ﻓﻬﻤﻴﺪﻥ ﺁﻥ ﺁﺳﺎﻥ‌ﺗﺮ ﺷﻮﺩ.

ﻣﺜﺎﻝ ﻭﻳﺮﺍﻳﺶ ﺷﺪﻩ: ﺍﻳﺮﺍﻥ ﺍﻣﺎﺭﺍﺕ ﻣﺘﺤﺪﻩ ﻋﺮﺑﯽ ﺭﺍ ﺑﻪ ﺩﻟﻴﻞ ﺗﮑﺮﺍﺭ ﺍﺩﻋﺎﻫﺎﻳﺶ ﺩﺭﺑﺎﺭﻩ ﺟﺰﺍﻳﺰ ﺳﻪ ﮔﺎﻧﻪ ﺗﻬﺪﻳﺪ ﺑﻪ ﻗﻄﻊ ﺭﻭﺍﺑﻂ ﮐﺮﺩﻩ ﺍﺳﺖ. ﺍﻳﻦ ﺟﺰﺍﻳﺮ ﻣﻄﺎﺑﻖ ﺗﻔﺎﻫﻢ ﻧﺎﻣﻪ‌ﺍﯼ ﻣﻴﺎﻥ ﺍﻳﺮﺍﻥ ﻭ ﺍﻧﮕﻠﻴﺲ٬ ﺗﺤﺖ ﺣﺎﮐﻤﻴﺖ ﺍﻳﺮﺍﻥ ﻗﺮﺍﺭ ﺩﺍﺭﺩ.

ﺍﺯ ﻣﻴﺎﻥ‌ﺗﻴﺘﺮ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ

ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻣﻄﺎﻟﺐ ﻃﻮﻻﻧﯽ، ﻣﻴﺎﻥ‌ﺗﻴﺘﺮ‌ ﮔﻨﺠﺎﻧﺪﻩ ﺷﻮﻧﺪ ﺗﺎ ﺍﺯ ﻳﮏ ﻧﻮﺍﺧﺘﯽ‌ﺟﻠﻮﮔﻴﺮﯼ ﮐﻨﺪ. ﻣﻴﺎﻥ‌ﺗﻴﺘﺮ ﺑﺎﻳﺪ ﮐﻮﺗﺎﻩ ﺑﺎﺷﺪ ﻭ ﺩﺭ ﺑﺮ ﮔﻴﺮﻧﺪﻩ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻳﺎ ﺟﺬﺍﺏ‌ﺗﺮﻳﻦ ﻧﮑﺘﻪ ﺩﺭ ﻣﻄﺎﻟﺒﯽ ﺑﺎﺷﺪ ﮐﻪ ﺯﻳﺮ ﺁﻥ ﻋﻨﻮﺍﻥ ﻣﯽ‌ﺷﻮﺩ. ﺧﻮﺍﻧﻨﺪﻩ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﺪ ﺑﺎ ﻧﮕﺎﻫﯽ ﺑﻪ ﻣﻴﺎﻥ‌ﺗﻴﺘﺮ، ﺩﺭﮐﯽ ﮐﻠﯽ ﺍﺯ ﺁﻧﭽﻪ ﺩﺭ ﻣﺘﻦ ﺯﻳﺮ ﺁﻥ ﺧﻮﺍﻫﺪ ﺁﻣﺪ ﺑﻪ ﺩﺳﺖ ﺑﻴﺎﻭﺭﺩ ﻭ ﺑﻪ ﺧﻮﺍﻧﺪﻥ ﺁﻥ ﺗﺸﻮﻳﻖ ﺷﻮﺩ.

ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺗﻴﺘﺮ ﻭ ﻣﻴﺎﻥ‌ﺗﻴﺘﺮﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﻣﻘﺎﻟﻪ ﻭ ﮔﺰﺍﺭﺵ ﻭ ﭘﺎﻳﺎﻥ ﻧﺎﻣﻪ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻴﺪ ﺳﺎﺩﻩ ﻭ ﺻﺮﻳﺢ ﻭ ﺷﻔﺎﻑ ﺑﺎﺷﺪ ﻭ ﻣﻨﻈﻮﺭ ﺁﻥ ﺭﻭﺷﻦ ﺑﺎﺷﺪ.

ﺯﻣﺎﻥ ﻓﻌﻞ‌ﻫﺎ ﺑﺎﻳﺪ ﺑﺎ ﻫﻢ ﻫﻤﺎﻫﻨﮓ ﺑﺎﺷﻨﺪ 

ﺑﺎﻳﺪ ﺩﻗﺖ ﺷﻮﺩ ﮐﻪ ﻓﻌﻞ‌ﻫﺎﯼ ﺑﻪ ﮐﺎﺭ ﺭﻓﺘﻪ ﺩﺭ ﻣﻄﻠﺐ ﻫﻤﺎﻫﻨﮕﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﻣﺜﻼً ﺩﺭ ﻳﮏ ﻣﺘﻦ، ﻧﺒﺎﻳﺪ ﺑﻌﻀﯽ ﺟﻤﻼﺕ ﺑﺎ ﻣﺎﺿﯽ ﺳﺎﺩﻩ، ﺑﺮﺧﯽ ﺑﺎ ﻣﺎﺿﯽ ﻧﻘﻠﯽ ﻭ ﺑﺮﺧﯽ ﺑﺎ ﻣﻀﺎﺭﻉ ﺍﺧﺒﺎﺭﯼ ﻧﻮﺷﺘﻪ ﺷﻮﻧﺪ.

ﺑﺮﺍﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﭘﻴﭽﻴﺪﮔﯽ ﻏﻴﺮ ﺿﺮﻭﺭﯼ، ﮔﺎﻫﯽ ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﺗﻤﺎﻡ ﮔﺰﺍﺭﺵ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻣﺎﺿﯽ ﺳﺎﺩﻩ ﺑﻨﻮﻳﺴﻴﺪ ﻭ ﺗﻨﻬﺎ ﻭﻗﺘﯽ ﺩﺭ ﻳﮏ ﺟﻤﻠﻪ ﺑﻪ ﺣﺎﺩﺛﻪ‌ﺍﯼ ﮐﻪ ﻗﺒﻞ ﺍﺯ ﺯﻣﺎﻥ ﻭﻗﻮﻉ ﻣﻮﺿﻮﻉ ﺍﺻﻠﯽ ﺍﺗﻔﺎﻕ ﺍﻓﺘﺎﺩﻩ ﺍﺷﺎﺭﻩ ﻣﯽ‌ﺷﻮﺩ، ﺍﺯ ﻣﺎﺿﯽ ﺑﻌﻴﺪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺍﻳﻦ ﺳﺎﺩﻩ‌ﺗﺮﻳﻦ ﻭ ﻣﺘﺪﺍﻭﻝ‌ﺗﺮﻳﻦ ﺭﺍﻩ ﺍﺳﺖ ﻭ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﮐﻤﺘﺮﻳﻦ ﺿﺮﻳﺐ ﺧﻄﺎ ﺭﺍ ﺩﺍﺭﺩ.

ﻣﺜﺎﻝ: ﺩﺍﺩﮔﺎﻩ ﻣﺘﻬﻤﺎﻥ ﭘﺮﻭﻧﺪﻩ ﺟﻨﺎﻳﺎﺕ ﺟﻨﮕﯽ ﺩﺭ ﺻﺮﺑﺴﺘﺎﻥ ﺩﻳﺮﻭﺯ ﺑﺮﮔﺰﺍﺭ ﺷﺪ. ﺩﺭ ﺟﺮﻳﺎﻥ ﺍﻳﻦ ﺩﺍﺩﮔﺎﻩ، ﻣﺘﻬﻢ ﺭﺩﻳﻒ ﺍﻭﻝ ﮔﻔﺘﻪ ﺍﺳﺖ ﮐﻪ ﺍﺗﻬﺎﻣﺎﺕ ﺧﻮﺩ ﺭﺍ ﻗﺒﻮﻝ ﺩﺍﺭﺩ. ﻭﯼ ﺗﺄﮐﻴﺪ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺩﺭ ﺯﻧﺪﺍﻥ ﺑﺎ ﻭﯼ ﺑﺪ ﺭﻓﺘﺎﺭﯼ ﺷﺪﻩ ﺍﺳﺖ.

ﺑﻌﺪ ﺍﺯ ﻭﻳﺮﺍﻳﺶ: ﺩﺍﺩﮔﺎﻩ ﻣﺘﻬﻤﺎﻥ ﭘﺮﻭﻧﺪﻩ ﺟﻨﺎﻳﺎﺕ ﺟﻨﮕﯽ ﺩﺭ ﺻﺮﺑﺴﺘﺎﻥ ﺩﻳﺮﻭﺯ ﺑﺮﮔﺰﺍﺭ ﺷﺪ. ﺩﺭ ﺟﺮﻳﺎﻥ ﺍﻳﻦ ﺩﺍﺩﮔﺎﻩ، ﻣﺘﻬﻢ ﺭﺩﻳﻒ ﺍﻭﻝ ﺍﺗﻬﺎﻣﺎﺕ ﺧﻮﺩ ﺭﺍ ﻗﺒﻮﻝ ﮐﺮﺩ ﺍﻣﺎ ﺗﺎﮐﻴﺪ ﮐﺮﺩ ﮐﻪ ﭘﻴﺶ ﺍﺯ ﺁﻥ ﺩﺭ ﺯﻧﺪﺍﻥ ﺑﺎ ﻭﯼ ﺑﺪ ﺭﻓﺘﺎﺭﯼ ﺷﺪﻩ ﺑﻮﺩ. 

ﺩﺭ ﻣﺘﻮﻥ ﺧﺒﺮﯼ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺧﺒﺮﯼ ﺍﺯ ﺯﺑﺎﻥ ﺷﺨﺺ ﺩﻳﮕﺮﯼ ﻧﻘﻞ ﻣﯽ‌ﺷﻮﺩ، ﻣﺎﺿﯽ ﻧﻘﻠﯽ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺭﻭﺩ.

ﻣﺜﺎﻝ: ﻣﺘﻬﻢ ﺭﺩﻳﻒ ﺍﻭﻝ ﻣﺪﻋﯽ ﺷﺪ ﮐﻪ ﻫﻤﺴﺮﺵ ﺑﻪ ﺍﻭ ﮔﻔﺖ ﮐﺎﺭﺷﺎﻥ ﻏﻴﺮﻗﺎﻧﻮﻧﯽ ﻧﻴﺴﺖ.

ﺑﻌﺪ ﺍﺯ ﻭﻳﺮﺍﻳﺶ: ﻣﺘﻬﻢ ﺭﺩﻳﻒ ﺍﻭﻝ ﻣﺪﻋﯽ ﺷﺪ ﮐﻪ ﻫﻤﺴﺮﺵ ﺑﻪ ﺍﻭ ﮔﻔﺘﻪ ﺍﺳﺖ ﮐﺎﺭﺷﺎﻥ ﻏﻴﺮﻗﺎﻧﻮﻧﯽ ﻧﻴﺴﺖ.

ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻢ ﮐﻪ ﺩﺭ ﻧﻘﻞ ﻗﻮﻝ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ، «ﺷﺨﺺ» ﺟﻤﻠﻪ (ﺍﻭﻝ، ﺩﻭﻡ ﻳﺎ ﺳﻮﻡ ﺷﺨﺺ) ﺑﺎ ﺷﺨﺼﯽ ﮐﻪ ﻧﻘﻞ ﻗﻮﻝ ﻣﺴﺘﻘﻴﻢ ﺍﺯ ﺍﻭ ﻣﯽ ﺷﻮﺩ ﻳﮑﯽ ﻧﺸﻮﺩ.

ﻣﺜﺎﻝ: ﻣﺘﻬﻢ ﮔﻔﺖ ﮐﻪ ﻧﻤﯽ‌ﺩﺍﻧﺴﺘﻢ ﮐﺎﺭ ﻣﻦ ﺟﺮﻡ ﺍﺳﺖ.

ﺑﻌﺪ ﺍﺯ ﻭﻳﺮﺍﻳﺶ: ﻣﺘﻬﻢ ﮔﻔﺖ ﮐﻪ ﻧﻤﯽ‌ﺩﺍﻧﺴﺘﻪ ﮐﺎﺭ ﺍﻭ ﺟﺮﻡ ﺍﺳﺖ.

ﺍﺯ ﺗﮑﺮﺍﺭ ﮐﻠﻤﺎﺕ ﺩﺭ ﻳﮏ ﺟﻤﻠﻪ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ

ﻣﺜﺎﻝ: ﺑﺮﺍﺑﺮﺕ ﻣﻮﮔﺎﺑﻪ ﺩﺭ ﺣﺎﻟﯽ ﺑﻪ ﻧﻴﻮﻳﻮﺭﮎ ﻣﯽ‌ﺭﻭﺩ ﮐﻪ ﺭﻭﺯﻧﺎﻣﻪ‌ﻫﺎﯼ ﺁﻣﺮﻳﮑﺎﻳﯽ ﺧﺒﺮ ﺩﺍﺩﻩ‌ﺍﻧﺪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻫﺘﻞ‌ﻫﺎﯼ ﺍﻳﻦ ﺷﻬﺮ ﺍﺯ ﭘﺬﻳﺮﻓﺘﻦ ﺭﺍﺑﺮﺕ ﻣﻮﮔﺎﺑﻪ ﺧﻮﺩﺩﺍﺭﯼ ﮐﺮﺩﻩ‌ﺍﻧﺪ.

ﺑﻌﺪ ﺍﺯ ﻭﻳﺮﺍﻳﺶ: ﺭﺍﺑﺮﺕ ﻣﻮﮔﺎﺑﻪ ﺩﺭ ﺣﺎﻟﯽ ﺑﻪ ﻧﻴﻮﻳﻮﺭﮎ ﻣﯽ‌ﺭﻭﺩ ﮐﻪ ﺭﻭﺯﻧﺎﻣﻪ‌ﻫﺎﯼ ﺁﻣﺮﻳﮑﺎﻳﯽ ﺧﺒﺮ ﺩﺍﺩﻩ‌ﺍﻧﺪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻫﺘﻞ‌ﻫﺎﯼ ﺍﻳﻦ ﺷﻬﺮ ﺍﺯ ﭘﺬﻳﺮﻓﺘﻦ ﺍﻭ ﺧﻮﺩﺩﺍﺭﯼ ﮐﺮﺩﻩ‌ﺍﻧﺪ.

ﺩﺭ ﮐﻞ ﻣﺘﻦ ﻫﻢ ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﮐﻠﻤﺎﺕ ﺗﮑﺮﺍﺭﯼ ﮐﻤﺘﺮﯼ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻧﻮﺷﺘﻪ‌ﻫﺎﯼ ﻣﺎ، ﺑﻪ ﻭﻳﮋﻩ ﺩﺭ ﻣﺘﻮﻧﯽ ﮐﻪ ﻣﻀﻤﻮﻥ ﺁﻧﻬﺎ ﻧﻘﻞ ﺍﻇﻬﺎﺭﺍﺕ ﺍﻓﺮﺍﺩ ﺍﺳﺖ، ﺍﻓﻌﺎﻝ ﻭ ﮐﻠﻤﺎﺕ ﺗﮑﺮﺍﺭﯼ ﺑﺴﻴﺎﺭﯼ ﺩﻳﺪﻩ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺑﻪ ﺯﻳﺒﺎﻳﯽ ﻣﺘﻦ ﻟﻄﻤﻪ ﺍﺳﺎﺳﯽ ﻣﯽ‌ﺯﻧﺪ.

ﻣﺮﺯ ﻣﻴﺎﻥ ﻧﻈﺮ ﻧﻮﻳﺴﻨﺪﻩ ﺑﺎ ﻳﮏ ﻭﺍﻗﻌﻴﺖ ﺗﺎﺭﻳﺨﯽ ﻳﺎ ﻋﻠﻤﯽ ﺑﺎﻳﺪ ﻣﺸﺨﺺ ﺑﺎﺷﺪ

ﺩﺭ ﻧﻮﺷﺘﻪ‌ﻫﺎﯼ ﺧﻮﺩ ﺑﺎﻳﺪ ﻣﺮﺯ ﻣﻴﺎﻥ ﮔﺰﺍﺭﻩ‌ﻫﺎﯼ ﺗﺤﻘﻴﻘﯽ ﻭ ﮐﻠﻤﺎﺕ ﻭ ﺗﻌﺎﺑﻴﺮ ﺍﺣﺴﺎﺳﯽ ﺭﺍ ﺑﻪ ﺩﻗﺖ ﺭﻋﺎﻳﺖ ﮐﻨﻴﺪ. ﺩﺭ ﮔﺰﺍﺭﺵ‌ﻫﺎﯼ ﻋﻠﻤﯽ ﻭ ﺗﺤﻘﻴﻘﯽ ﺍﺣﺴﺎﺱ ﺩﺭ ﺑﻴﺸﺘﺮ ﺍﻣﻮﺭ ﺟﺎﻳﯽ ﻧﺪﺍﺭﺩ. ﺩﺭ ﺍﻳﻦ ﻣﺘﻮﻥ ﺑﺎﻳﺪ ﺍﺯ ﻫﺮ ﮔﻮﻧﻪ ﻣﻮﺿﻊ ﮔﻴﺮﯼ، ﺗﻔﺴﻴﺮ، ﺑﻴﺎﻥ ﺍﺣﺴﺎﺱ ﻭ ﺩﺭ ﺣﺪ ﺍﻣﮑﺎﻥ ﺍﺯ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻥ ﮐﻠﻤﺎﺕ ﺗﻮﺻﻴﻔﯽ ﻧﺸﺎﻥ ﺩﻫﻨﺪﻩ ﻋﻘﻴﺪﻩ ﺷﺨﺼﯽ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ.

ﻃﺒﻴﻌﺘﺎً، ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻥ ﺗﻮﺻﻴﻒ ﻭ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺱ ﻭ ﺗﺤﻠﻴﻞ ﻭ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ ﺷﺨﺼﯽ ﺩﺭ ﻧﻮﺷﺘﻪ‌ﻫﺎﯼ ﻣﺎ ﻣﻤﮑﻦ ﺍﺳﺖ، ﺍﻣﺎ ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺩﺭ ﻣﻮﻗﻌﻴﺘﯽ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺮﺍﯼ ﺧﻮﺍﻧﻨﺪﻩ ﻳﺎ ﺷﻨﻮﺩﻩ ﻣﺸﺨﺺ ﺑﺎﺷﺪ ﻭ ﺍﻣﮑﺎﻥ ﺧﻠﻂ ﺁﻥ ﺑﺎ ﻓﮑﺖ‌ﻫﺎ ﻭ ﮔﺰﺍﺭﻩ‌ﻫﺎﯼ ﻏﻴﺮ ﻧﻈﺮﯼ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺑﻪ ﺑﻴﺎﻥ ﺩﻳﮕﺮ ﻣﺮﺯ ﻧﻈﺮ ﻧﻮﻳﺴﻨﺪﻩ ﺑﺎ ﻣﺜﻼ ﻳﮏ ﻭﺍﻗﻌﻴﺖ ﻣﺴﻠﻢ ﺗﺎﺭﻳﺨﯽ ﻳﺎ ﻋﻠﻤﯽ ﻣﺸﺨﺺ ﺑﺎﺷﺪ.

ﺑﻪ ﻫﺮ ﺗﺮﺗﻴﺐ، ﺑﺎ ﻭﺟﻮﺩ ﻣﺠﺎﺯ ﺑﻮﺩﻥ ﺍﺑﺮﺍﺯ ﻋﻘﻴﺪﻩ ﻭ ﺍﺣﺴﺎﺱ ﺷﺨﺼﯽ ﺩﺭ ﻣﺘﻮﻥ ﺿﺮﻭﺭﯼ ﺍﺳﺖ ﮐﻪ ﻫﺮﮔﺰ ﺑﻪ ﻣﺮﺯ ﺗﻮﻫﻴﻦ ﺑﻪ ﺍﻓﺮﺍﺩ ﻭﻟﻮ ﻣﻨﻔﻮﺭﺗﺮﻳﻦ ﺍﻓﺮﺍﺩ ﻧﺰﺩﻳﮏ ﻧﺸﻮﻳﻢ ﻭ ﺑﺮﺍﯼ ﻣﺤﮑﻮﻡ ﮐﺮﺩﻥ ﺣﺘﯽ ﺟﻨﺎﻳﺘﮑﺎﺭﺗﺮﻳﻦ ﺍﺷﺨﺎﺹ ﻧﻴﺰ، ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﺑﻴﺎﻥ ﻣﺴﺘﻨﺪﺍﺕ ﺻﻮﺭﺕ ﺩﻫﻴﻢ ﻭ ﻧﻪ ﺗﻮﻫﻴﻦ ﺁﻣﻴﺰ ﮐﺮﺩﻥ ﻟﺤﻦ ﺧﻮﺩ.

ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﻫﺮﮔﻮﻧﻪ ﻧﻘﻞ ﻗﻮﻝ ﻭ ﺍﺳﺘﻨﺎﺩ ﺑﻪ ﻣﻨﺎﺑﻊ ﺩﺭ ﻧﻮﺷﺘﻪ ﻣﺎ ﺑﺎ ﺫﮐﺮ ﺩﻗﻴﻖ ﻣﻨﺒﻊ ﻭ ﺯﻣﺎﻥ ﺍﻧﺘﺸﺎﺭ ﻣﻮﺟﻮﺩ ﺑﺎﺷﺪ ﺑﻪ ﻋﻼﻭﻩ، ﺿﺮﻭﺭﺕ ﺩﺍﺭﺩ ﮐﻪ ﻫﺮ ﻧﻘﻞ ﻗﻮﻝ ﻣﺴﺘﻘﻴﻢ، ﺩﺭ ﺩﺍﺧﻞ ﻋﻼﻣﺖ ﻣﺨﺼﻮﺹ ﻧﻘﻞ ﻗﻮﻝ ﺑﻴﺎﻳﺪ ﺗﺎ ﻣﺮﺯ ﺁﻥ ﺑﺎ ﻣﻄﺎﻟﺒﯽ ﮐﻪ ﺗﻮﺳﻂ ﻧﻮﻳﺴﻨﺪﻩ ﺑﻴﺎﻥ ﻣﯽ‌ﺷﻮﺩ ﮐﺎﻣﻼً ﺭﻭﺷﻦ ﺑﺎﺷﺪ.

ﺳﻌﯽ ﮐﻨﻴﺪ ﻫﻴﭻ ﮐﺪﺍﻡ ﺍﺯ ﺍﺩﻋﺎﻫﺎﯼ ﻣﻄﺮﺡ ﺷﺪﻩ ﺩﺭ ﻣﺘﻨﯽ ﮐﻪ ﻣﯽ ﻧﻮﻳﺴﻴﺪ ﺑﺪﻭﻥ ﻣﺮﺟﻊ ﻧﺒﺎﺷﺪ. ﺑﺮﺍﯼ ﺍﺭﺟﺎﻉ ﺩﺍﺩﻥ ﺣﺘﯽ ﺍﮔﺮ ﻣﺘﻦ ﻣﺎ ﻣﻘﺎﻟﻪ‌ﺍﯼ ﺩﺍﻧﺸﮕﺎﻫﯽ ﻧﺒﺎﺷﺪ، ﻧﺎﭼﺎﺭ ﺍﺯ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻨﺎﺑﻊ ﻣﻌﺘﺒﺮ ﻭ ﻣﺮﺗﺒﻂ ﻫﺴﺘﻴﻢ.

ﺍﺯ ﺭﺳﻢ‌ﺍﻟﺨﻂ ﻭﺍﺣﺪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ

ﺩﺭ ﻣﻮﺭﺩ ﺭﺳﻢ ﺍﻟﺨﻂ ﻓﺎﺭﺳﯽ ﻫﻤﭽﻮﻥ ﺩﻳﮕﺮ ﺯﺑﺎﻥ‌ﻫﺎ ﺗﻮﺍﻓﻖ ﮐﺎﻣﻞ ﻣﻴﺎﻥ ﺻﺎﺣﺐ ﻧﻈﺮﺍﻥ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺩﺭ ﭼﻨﻴﻦ ﻭﺿﻌﻴﺘﯽ ﺍﺟﻤﺎﻻ ﺍﻧﺘﻈﺎﺭ ﻣﯽ‌ﺭﻭﺩ ﺣﺪﺍﻗﻞ ﺩﺭ ﻳﮏ ﻣﻄﻠﺐ، ﺩﻭ ﺟﻮﺭ ﺭﺳﻢ ﺍﻟﺨﻂ ﺭﺍ ﺑﺮﺍﯼ ﻧﮕﺎﺭﺵ ﮐﻠﻤﺎﺕ ﻭﺍﺣﺪ ﺑﻪ ﮐﺎﺭ ﻧﺒﺮﻳﻢ.

ﺩﺭ ﻳﮏ ﻣﺘﻦ ﻭﺍﺣﺪ، ﻣﺜﻼ ﻧﺒﺎﻳﺪ ﺟﺎﻳﯽ ﻧﻮﺷﺖ«ﺳﻮﺍﻝ» ﻭ ﺩﺭ ﺟﺎﯼ ﺩﻳﮕﺮ ﺩﺭ ﻫﻤﺎﻥ ﻣﺘﻦ ﺍﻳﻦ ﮐﻠﻤﻪ ﺭﺍ ﺑﻪ ﺷﮑﻞ «ﺳﺌﻮﺍﻝ» ﺑﻨﻮﻳﺴﻢ.

ﺑﻪ ﻋﻼﻭﻩ، ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻗﺎﻋﺪﻩ ﺣﺪﺍﻗﻠﯽ، ﻣﯽ‌ﺗﻮﺍﻥ ﺗﻮﺻﻴﻪ ﮐﺮﺩ ﮐﻪ «ﻣﯽ» ﺍﺑﺘﺪﺍﯼ ﻓﻌﻞ‌ﻫﺎ ﻭ ﮐﻠﻴﻪ ﺣﺮﻭﻑ ﺭﺑﻂ ﻣﻴﺎﻥ ﮐﻠﻤﺎﺕ ﺟﺪﺍ ﻧﻮﺷﺘﻪ ﺷﻮﻧﺪ. ﻣﺜﻼ ﺑﻪ ﺟﺎﯼ ﻣﻴﺮﻭﺩ، ﺑﻬﻤﻴﻦ ﺩﻟﻴﻞ، ﮐﺘﺎﺑﺮﺍ، ... ﺑﻨﻮﻳﺴﻴﻢ: ﻣﯽ‌ﺭﻭﺩ، ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ، ﮐﺘﺎﺏ ﺭﺍ ﻭ . . .

ﺗﻮﺻﻴﻪ ﻣﺎ ﺩﺭ ﺍﻳﻦ ﺟﺎ ﺭﻋﺎﻳﺖ ﻗﺎﻋﺪﻩ ﺟﺪﺍﻧﻮﻳﺴﯽ ﮐﻠﻤﺎﺕ ﺍﺳﺖ. ﻣﺜﻼ ﮔﻔﺖ ﻭ ﮔﻮ ﺑﻪ ﺟﺎﯼ ﮔﻔﺘﮕﻮ ﻳﺎ ﻋﻠﯽ ﺭﻏﻢ ﺑﻪ ﺟﺎﯼ ﻋﻠﻴﺮﻏﻢ ؛ ﺍﻣﺎ ﺑﻪ ﻫﺮ ﺩﻟﻴﻞ ﺍﮔﺮ ﺑﺮﺧﻼﻑ ﺍﻳﻦ ﺍﺯ ﻗﺎﻋﺪﻩ ﭘﻴﻮﺳﺘﻪ ﻧﻮﻳﺴﯽ ﭘﻴﺮﻭﯼ ﮐﺮﺩﻳﺪ٬ ﻻﺯﻡ ﺍﺳﺖ ﺍﻳﻦ ﻗﺎﻋﺪﻩ ﺭﺍ ﺩﺭ ﮐﻞ ﻣﺘﻦ ﺑﻪ ﮐﺎﺭ ﺑﮕﻴﺮﻳﺪ ﻭ ﺁﻥ ﺭﺍ ﻳﮏ ﺩﺳﺖ ﺗﺤﻮﻳﻞ ﺧﻮﺍﻧﻨﺪﻩ ﺑﺪﻫﻴﺪ.

ﺩﺭ ﺩﺭﺱ ﺑﻌﺪ ﮐﻪ ﺁﺧﺮﻳﻦ ﻗﺴﻤﺖ ﺍﺯ ﺍﻳﻦ ﻣﺠﻤﻮﻋﻪ ﺍﺳﺖ ﻧﮑﺎﺗﯽ ﺑﺮﺍﯼ ﭘﺮﻫﻴﺰ ﺍﺯ ﺍﺷﺘﺒﺎﻫﺎﺕ ﻣﺘﺪﺍﻭﻝ ﺩﺭ ﻓﺎﺭﺳﯽ ﻧﻮﻳﺴﯽ ﺭﺍ ﻣﺮﻭﺭ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.
