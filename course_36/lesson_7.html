ﺍﻣﺎ ﺑﺮﺍﯼ ﻳﮏ ﻧﻔﺮ ﺩﻳﮕﺮ ﻣﻤﮑﻦ ﺍﺳﺖ ﻟﻴﻨﻮﮐﺲ ﺍﻧﺘﺨﺎﺏ ﺧﻮﺑﯽ ﺑﺎﺷﺪ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﻧﻴﺎﺯﻫﺎﯼ ﻭﺳﻴﻊ ﺗﺮﯼ ﺭﺍ ﭘﺎﺳﺦ ﻣﯽ ﺩﻫﺪ. ﻣﺜﻼ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺎ ﻋﻼﻗﻤﻨﺪ ﺑﻪ ﺗﺤﻠﻴﻞ ﻭﺿﻌﻴﺖ ﺷﺒﮑﻪ ﺍﯼ ﮐﻪ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻴﻢ ﺑﺎﺷﻴﻢ ﻳﺎ ﻳﮏ ﺭﻭﺯ ﺍﻳﺪﻩ ﺍﯼ ﺑﻪ ﻓﮑﺮﻣﺎﻥ ﺑﺮﺳﺪ ﮐﻪ ﭘﻴﺎﺩﻩ ﺳﺎﺯﯼ ﺁﻥ ﻧﻴﺎﺯﻣﻨﺪ ﻧﻮﺷﺘﻦ ﻳﮏ ﺍﺳﮑﺮﻳﭙﺖ ﮐﻮﭼﮏ ﻭ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﻳﮏ ﺑﺎﻧﮏ ﺍﻃﻼﻋﺎﺗﯽ ﺳﺒﮏ ﺑﺎﺷﺪ.

ﻟﻴﻨﻮﮐﺲ ﺩﺭ ﺯﻧﺪﮔﯽ ﺣﺮﻓﻪ ﺍﯼ ﻭ ﺷﻐﻠﯽ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺭ ﻧﻘﺶ ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﻇﺎﻫﺮ ﺷﻮﺩ. ﺩﺭ ﻓﻨﯽ ﺗﺮﻳﻦ ﺳﻄﺢ ﻳﮏ ﻧﻔﺮ ﻣﻤﮑﻦ ﺍﺳﺖ ﺗﻮﺳﻌﻪ ﺩﻫﻨﺪﻩ ﻟﻴﻨﻮﮐﺲ ﺑﺎﺷﺪ. ﺗﺼﻮﺭ ﺍﺷﺘﺒﺎﻫﯽ ﺩﺭ ﻣﻮﺭﺩ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺁﺯﺍﺩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻣﺒﻨﯽ ﺑﺮ ﺍﻳﻨﮑﻪ ﺧﻴﻠﯽ ﻫﺎ ﺑﻪ ﻏﻠﻂ ﻓﮑﺮ ﻣﯽ ﮐﻨﻨﺪ ﮐﻪ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺁﺯﺍﺩ ﺑﺎ ﻣﻔﻬﻮﻡ‌ «ﺑﺪﻭﻥ ﭘﻮﻝ» ﻣﺮﺗﺒﻂ ﺍﺳﺖ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺑﺨﺶ ﻋﻤﺪﻩ ﺗﻮﺳﻌﻪ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎ ﺍﺯ ﻃﺮﻑ ﺷﺮﮐﺖ ﻫﺎﯼ ﺑﺰﺭﮒ ﺍﻧﺠﺎﻡ ﻳﺎ ﭘﺸﺘﻴﺒﺎﻧﯽ ﻣﯽ ﺷﻮﺩ. ﺷﺮﮐﺖ ﻫﺎﻳﯽ ﻣﺜﻞ ﻧﺎﻭﻝ، ﺍﻳﻨﺘﻞ، ﺭﺩﻫﺖ ﻭ ﺣﺘﯽ ﻣﺎﻳﮑﺮﻭﺳﺎﻓﺖ، ﺷﺮﮐﺖ ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﭘﻮﻝ ﺯﻳﺎﺩﯼ ﺧﺮﺝ ﺗﻮﺳﻌﻪ ﻫﺴﺘﻪ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺍﮔﺮ ﺷﻤﺎ ﺗﺴﻠﻂ ﺑﺎﻻ ﺑﻪ ﮐﺮﻧﻞ ﻭ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺑﺪﻭﻥ ﺷﮏ ﻫﻤﻴﺸﻪ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﺍﻳﻦ ﺷﺮﮐﺖ ﻫﺎ ﮐﺎﺭﻫﺎﯼ ﺧﻮﺑﯽ ﭘﻴﺪﺍ ﮐﻨﻴﺪ.

ﺩﺭ ﻳﮏ ﻻﻳﻪ ﭘﺎﻳﻴﻦ ﺗﺮ ﻳﮏ ﻧﻔﺮ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺘﺨﺼﺺ ﻧﺼﺐ، ﺗﻨﻈﻴﻢ ﻳﺎ ﭘﺸﺘﻴﺒﺎﻧﯽ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺎﯼ ﮔﻨﻮ/ﻟﻴﻨﻮﮐﺴﯽ ﺑﺎﺷﺪ. ﺗﻘﺮﻳﺒﺎ ﺗﻤﺎﻡ ﺷﺒﮑﻪ ﻫﺎﯼ ﻣﺨﺎﺑﺮﺍﺗﯽ ﻭ ﺑﺨﺶ ﺑﺰﺭﮔﯽ ﺍﺯ ﺍﻳﻨﺘﺮﻧﺖ ﻭ ﺩﺭﺻﺪ ﺑﺎﻻﻳﯽ ﺍﺯ ﺗﻤﺎﻡ ﺍﻣﻮﺭ ﺟﺪﯼ ﺩﻧﻴﺎ، ﺑﺎ ﺳﻴﺴﺘﻢ ﻫﺎﯼ ﻟﻴﻨﻮﮐﺴﯽ ﺍﺩﺍﺭﻩ ﻣﯽ ﺷﻮﻧﺪ ﻭ ﺩﺭ ﺗﻤﺎﻡ ﮐﺸﻮﺭﻫﺎ ﻫﻤﻴﺸﻪ ﻧﻴﺎﺯ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﮐﺴﯽ ﮐﻪ ﺑﻪ ﺧﻮﺑﯽ ﺑﻪ ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﺴﻠﻂ ﺑﺎﺷﺪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺷﺮﮐﺖ ﻫﺎ ﭘﻮﻝ ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺧﺮﺝ ﻣﯽ ﮐﻨﻨﺪ ﺗﺎ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻨﺪ ﺩﺭ ﺻﻮﺭﺕ ﺧﺮﺍﺑﯽ ﻳﮏ ﺳﺮﻭﺭ ﻟﻴﻨﻮﮐﺴﯽ ﮐﺴﯽ ﻫﺴﺖ ﮐﻪ ﺑﺘﻮﺍﻧﺪ ﺁﻥ ﺭﺍ ﺗﻌﻤﻴﺮ ﮐﻨﺪ ﻭ ﺩﺭ ﻫﻨﮕﺎﻡ ﻧﺼﺐ ﻳﺎ ﺗﻨﻈﻴﻤﺎﺕ ﺍﻭﻟﻴﻪ ﻫﻢ ﺑﺎﺯﺍﺭ ﺗﺠﺎﺭﯼ ﺷﺪﻳﺪﺍ ﻧﻴﺎﺯﻣﻨﺪ ﻣﺘﺨﺼﺼﻴﻨﯽ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺧﻮﺑﯽ ﺍﺯ ﻣﺎﺟﺮﺍ ﺳﺮ ﺩﺭ ﺑﻴﺎﻭﺭﻧﺪ. ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺷﻤﺎ ﺁﻥ ﻓﺮﺩ ﺑﺎﺷﻴﺪ، ﺷﻐﻠﯽ ﺩﺍﺭﻳﺪ ﺳﻄﺢ ﺑﺎﻻ ﻭ ﻣﺴﺘﻘﻴﻤﺎ ﻣﺮﺗﺒﻂ ﺑﺎ ﺧﻮﺩ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﮔﻨﻮ/ﻟﻴﻨﻮﮐﺲ ﻭ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﺁﻥ.

ﺍﻣﺎ ﻟﻴﻨﻮﮐﺲ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﺷﮑﻞ ﻏﻴﺮﻣﺴﺘﻘﻴﻢ ﻫﻢ ﺩﺭ ﺯﻧﺪﮔﯽ ﺷﻐﻠﯽ ﺍﻓﺮﺍﺩ ﻣﻮﺛﺮ ﺑﺎﺷﺪ. ﺩﺍﻧﺸﮕﺎﻩ ﻫﺎ ﻫﺮ ﺳﺎﻝ ﺗﻌﺪﺍﺩ ﺯﻳﺎﺩﯼ ﻣﻬﻨﺪﺱ ﻣﺨﺎﺑﺮﺍﺕ ﻓﺎﺭﻍ ﺍﻟﺘﺤﺼﻴﻞ ﻣﯽ ﮐﻨﻨﺪ ﻭﻟﯽ ﺗﻌﺪﺍﺩ ﻣﻬﻨﺪﺱ ﻫﺎﯼ ﻣﺨﺎﺑﺮﺍﺗﯽ ﮐﻪ ﺩﺍﻧﺶ ﻟﻴﻨﻮﮐﺲ ﺩﺍﺭﻧﺪ ﻭ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺩﺭ ﻳﮏ ﺷﺒﮑﻪ ﻣﺨﺎﺑﺮﺍﺗﯽ ﺍﺳﺘﺨﺪﺍﻡ ﺷﻮﻧﺪ ﺷﺪﻳﺪﺍ ﻣﺤﺪﻭﺩ ﺍﺳﺖ. ﻭﺿﻌﻴﺖ ﺑﺎﺯﺍﺭ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﻧﻴﺰ ﻫﻤﻴﻨﻄﻮﺭ ﺍﺳﺖ. ﺟﻤﻊ ﺑﺰﺭﮔﯽ ﺍﺯ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺲ ﻫﺎ ﻫﺮ ﺳﺎﻝ ﺍﺯ ﺩﺍﻧﺸﮕﺎﻩ ﺑﻴﺮﻭﻥ ﻣﯽ ﺁﻳﻨﺪ ﻭﻟﯽ ﻣﻄﻤﺌﻨﺎ ﺍﮔﺮ ﺷﺮﮐﺘﯽ ﺑﻪ ﺩﻧﺒﺎﻝ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺲ ﺳﯽ ﺑﺎ ﻣﻬﺎﺭﺕ ﻭ ﺗﺠﺮﺑﻪ ﻟﻴﻨﻮﮐﺲ ﺑﺎﺷﺪ، ﺷﺎﻧﺲ ﺍﮐﺜﺮ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺑﺮﺍﯼ ﺍﺳﺘﺨﺪﺍﻡ ﺻﻔﺮ ﺍﺳﺖ!

ﺩﺭ ﺍﻳﻦ ﺑﺨﺶ، ﻟﻴﻨﻮﮐﺲ ﻳﮏ ﻣﻬﺎﺭﺕ ﺟﺎﻧﺒﯽ ﺍﺳﺖ ﺑﺮﺍﯼ ﺟﺪﺍ ﮐﺮﺩﻥ ﺷﻤﺎ ﺍﺯ ﺧﻴﻞ ﺑﺰﺭﮔﯽ ﮐﻪ ﻫﻤﻪ ﻣﻬﺎﺭﺕ ﻫﺎﻳﯽ ﻣﺸﺎﺑﻪ ﺩﺍﺭﻧﺪ. ﺍﻳﻦ ﺩﺍﻧﺶ «ﺑﻴﻦ ﺭﺷﺘﻪ ﺍﯼ» (ﺑﻪ ﻣﻌﻨﺎﯼ ﺩﺍﻧﺴﺘﻦ ﺩﻭ ﭼﻴﺰ ﻣﺨﺘﻠﻒ ﮐﻪ ﺍﮐﺜﺮ ﺍﻓﺮﺍﺩ ﻓﻘﻂ ﻳﮑﯽ ﺍﺯ ﺁﻥ ﻫﺎ ﺭﺍ ﻣﯽ ﺩﺍﻧﻨﺪ) ﺑﺎﻋﺚ ﻣﯽ ﺷﻮﺩ ﻓﺮﺩ ﺗﺒﺪﻳﻞ ﺷﻮﺩ ﺑﻪ ﮔﺰﻳﻨﻪ ﺍﯼ ﺑﺴﻴﺎﺭ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﻳﮏ ﺑﺎﺯﺍﺭ ﮐﺎﺭ ﺧﺎﺹ. ﻧﻴﺎﺯ ﺑﺎﺯﺍﺭ ﺑﺮﺍﯼ ﮐﺴﯽ ﮐﻪ ﻫﺮ ﺩﻭﯼ ﺍﻳﻨﻬﺎ (ﻣﺜﻼ ﻣﺨﺎﺑﺮﺍﺕ ﻭ ﻟﻴﻨﻮﮐﺲ، ﺗﺎﻳﭗ ﺳﺮﻳﻊ ﻭ ﻟﻴﻨﻮﮐﺲ، ﺭﺍﻧﻨﺪﮔﯽ ﻭ ﻟﻴﻨﻮﮐﺲ، ﻫﺮ ﭼﻴﺰﯼ ﻭ ﻟﻴﻨﻮﮐﺲ) ﺭﺍ ﺑﺪﺍﻧﺪ ﻣﻌﻤﻮﻻ ﮐﻢ ﺍﺳﺖ ﻭﻟﯽ ﺩﺭ ﻣﻘﺎﺑﻞ ﮐﺴﺎﻧﯽ ﮐﻪ ﺍﻳﻦ ﺩﻭ ﻣﻬﺎﺭﺕ ﺭﺍ ﺑﺎ ﻫﻢ ﺩﺍﺭﻧﺪ ﺑﺴﻴﺎﺭ ﺑﺴﻴﺎﺭ ﮐﻢ ﻫﺴﺘﻨﺪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺁﻥ ﺑﺎﺯﺍﺭ ﺧﺎﺹ ﻫﻤﻪ ﺗﻼﺷﺶ ﺭﺍ ﻣﯽ ﮐﻨﺪ ﺗﺎ ﮐﺴﯽ ﺭﺍ ﮐﻪ ﻫﺮ ﺩﻭﯼ ﺍﻳﻨﻬﺎ ﺭﺍ ﻣﯽ ﺩﺍﻧﺪ ﻭ ﺑﻪ ﺳﺨﺘﯽ ﭘﻴﺪﺍ ﺷﺪﻩ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﺑﻴﺎﻭﺭﺩ ﻳﺎ ﺣﻔﻆ ﮐﻨﺪ.

ﻣﻮﺳﺴﻪ ﺁﻣﺎﺭﯼ ﻣﺸﺎﻏﻞ ﻟﻴﻨﻮﮐﺴﯽ ﻣﯽ ﮔﻮﻳﺪ ﮐﻪ ۸۱٪ ﺷﺮﮐﺖ ﻫﺎﯼ ﺁﻣﺮﻳﮑﺎﻳﯽ ﺩﺭ ﺳﺎﻝ ۲۰۱۲ ﺑﻪ ﺩﻧﺒﺎﻝ ﻣﺘﺨﺼﺼﻴﻦ ﻟﻴﻨﻮﮐﺲ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ ﻭ ۸۵٪ ﺷﺮﮐﺖ ﻫﺎ ﮔﻔﺘﻪ ﺍﻧﺪ ﮐﻪ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻣﺘﺨﺼﺼﻴﻦ ﺧﻮﺏ ﺩﺭ ﺍﻳﻦ ﺣﻮﺯﻩ ﺑﺴﻴﺎﺭ ﻣﺸﮑﻞ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﻳﮏ ﻧﻤﻮﻧﻪ ﺩﻳﮕﺮ ﺍﺯ ﺍﻫﻤﻴﺖ ﺍﻳﻦ ﺗﺨﺼﺺ ﺩﺭ ﺯﻧﺪﮔﯽ ﺣﺮﻓﻪ ﺍﯼ ﻣﯽ ﺷﻮﺩ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺍﺷﺎﺭﻩ ﮐﺮﺩ ﮐﻪ ۳۷٪ ﺷﺮﮐﺖ ﻫﺎ ﮔﻔﺘﻪ ﺍﻧﺪ ﺩﺭ ﺳﺎﻝ ۲۰۱۲ ﺣﺎﺿﺮﻧﺪ ﺩﺭ ﻣﻮﺭﺩ ﺳﺎﻋﺎﺕ ﮐﺎﺭ ﮐﻨﺎﺭ ﺑﻴﺎﻳﻨﺪ ﻭ ۲۸٪ ﺩﻳﮕﺮ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻩ ﺍﻧﺪ ﮐﻪ ﺣﺎﺿﺮﻧﺪ ﺑﻪ ﻣﺘﺨﺼﺼﻴﻦ ﻟﻴﻨﻮﮐﺲ ﺣﻘﻮﻕ ﺑﻴﺸﺘﺮﯼ ﻧﺴﺒﺖ ﺑﻪ ﺑﻘﻴﻪ ﺷﺮﮐﺖ ﺑﺪﻫﻨﺪ ﺗﺎ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺟﺬﺏ ﺷﺮﮐﺖ ﻫﺎﻳﺸﺎﻥ ﺷﻮﻧﺪ.

ﻻﺯﻡ ﺍﺳﺖ ﺍﺿﺎﻓﻪ ﮐﻨﻢ ﮐﻪ ﺍﻳﻦ ﺣﺮﻑ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﺷﻐﻞ ﻫﺎﯼ ﺩﻳﮕﺮ ﺍﺭﺯﺵ ﮐﻤﺘﺮﯼ ﺍﺯ ﺗﺨﺼﺺ ﻟﻴﻨﻮﮐﺲ ﺩﺍﺭﻧﺪ ﻳﺎ ﺩﺍﺷﺘﻦ ﺗﺨﺼﺺ ﻟﻴﻨﻮﮐﺲ ﺗﻀﻤﻴﻨﯽ ﺑﺮﺍﯼ ﺷﻐﻞ ﺧﻮﺏ ﺍﺳﺖ. ﺑﺨﺶ ﻣﻬﻤﯽ ﺍﺯ ﺯﻧﺪﮔﯽ ﺷﻐﻠﯽ ﺑﻪ ﻣﻬﺎﺭﺕ ﻫﺎﯼ ﻏﻴﺮﻓﻨﯽ ﻣﺮﺑﻮﻁ ﻣﯽ ﺷﻮﺩ ﻭ ﻣﺴﺎﻳﻞ ﻣﺸﺎﺑﻪ ﺍﻣﺎ ﺻﺤﺒﺖ ﺑﺮ ﺳﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺍﻧﺴﺘﻦ ﻟﻴﻨﻮﮐﺲ، ﺯﻳﺮﺳﺎﺧﺖ ﺧﻮﺑﯽ ﺍﺳﺖ ﺑﺮﺍﯼ ﻳﮏ ﺷﻐﻞ ﺧﻮﺏ.

ﺣﺘﯽ ﺍﮔﺮ ﺑﻪ ﺳﺮﺍﻍ ﻟﻴﻨﻮﮐﺲ ﻫﻢ ﻧﺮﻓﺘﻴﺪ، ﻣﺎﺟﺮﺍﯼ ﻣﻬﺎﺭﺕ ﻫﺎﯼ «ﺑﻴﻦ ﺭﺷﺘﻪ ﺍﯼ» ﺭﺍ ﺟﺪﯼ ﺑﮕﻴﺮﻳﺪ ﺗﺎ ﺩﺭ ﺑﻴﻦ ﺟﻤﻊ ﻋﻈﻴﻢ ﺁﺩﻡ ﻫﺎﯼ ﻣﺸﺎﺑﻪ، ﺑﺮﺟﺴﺘﻪ ﺷﻮﻳﺪ. ﺩﺭ ﺩﺭﺱ ﺑﻌﺪﯼ ﮐﻪ ﺩﺭﺱ ﺁﺧﺮ ﺍﻳﻦ ﻣﺠﻤﻮﻋﻪ ﺍﺳﺖ، ﺑﻪ ﺳﺮﺍﻍ ﺟﺎﻣﻌﻪ ﻟﻴﻨﻮﮐﺲ ﺧﻮﺍﻫﻴﻢ ﺭﻓﺖ ﻭ ﺩﺭﺑﺎﺭﻩ ﺭﻭﺵ ﻫﺎﻳﯽ ﺣﺮﻑ ﺧﻮﺍﻫﻴﻢ ﺯﺩ ﮐﻪ ﺍﻓﺮﺍﺩ ﺑﺎ ﺩﺍﻧﺴﺘﻦ ﺁﻧﻬﺎ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺭﺍﻩ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﺳﺮﺯﻣﻴﻦ ﺟﺎﺩﻭﻳﯽ ﭘﻴﺪﺍ ﮐﻨﻨﺪ.
