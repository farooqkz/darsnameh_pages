ﺩﺭﺱ ﭘﻨﺠﻢ- ﺭﻭﻧﺪ ﻭ ﺧﻄﻮﻁ ﺭﻭﻧﺪ

ﻣﻔﻬﻮﻡ ﺭﻭﻧﺪ ﺩﺭ ﺑﺎﺯﺍﺭﻫﺎﯼ ﻣﺎﻟﯽ، ﺍﺷﺎﺭﻩ ﺑﻪ ﺟﻬﺖ ﺑﺎﺯﺍﺭ ﺩﺍﺭﺩ. ﺭﻭﻧﺪﻫﺎ، ﺟﻬﺖ ﻭ ﺷﻴﺐ ﺣﺮﮐﺘﯽ ﻗﻴﻤﺖ ﺍﺯ ﻣﻘﻄﻌﯽ ﺩﺭ ﮔﺬﺷﺘﻪ ﺗﺎ ﺑﻪ ﺣﺎﻝ -ﻭ ﻳﺎ ﺗﺎ ﺁﻳﻨﺪﻩ- ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﻭ ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﺭﻭﻧﺪ ﻧﺸﺎﻧﮕﺮ ﻗﺪﺭﺕ ﺑﺎﺯﺍﺭ ﺩﺭ ﻳﮏ ﺟﻬﺖ ﺍﺳﺖ، ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﺯﻣﺎﻥ ﺷﮑﻞ‌ﮔﻴﺮﯼ ﺁﻥ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺖ ﮐﻪ ﺗﺎ ﻣﺪﺗﯽ ﺑﺎﺯﺍﺭ ﺭﺍ ﺩﺭ ﻳﮏ ﺟﻬﺖ ﭘﻴﺶ ﺑﺒﺮﺩ.

ﻫﺮ ﺭﻭﻧﺪ ﺍﺯ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﻣﻮﺝ ﻧﺰﻭﻟﯽ ﻭ ﺻﻌﻮﺩﯼ ﺗﺸﮑﻴﻞ ﺷﺪﻩ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﻳﮏ ﺩﺭ ﻣﻴﺎﻥ ﺣﺮﮐﺎﺕ ﺑﺎﺯﺍﺭ ﺭﺍ ﺭﻗﻢ ﻣﯽ‌ﺯﻧﻨﺪ.

 

ﺍﻧﻮﺍﻉ ﺭﻭﻧﺪ
ﺭﻭﻧﺪﻫﺎ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺑﻪ ﺳﻪ ﺩﺳﺘﻪ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﺷﻮﻧﺪ:

۱. ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ: ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺭﻭﻧﺪ ﺑﺎﺯﺍﺭ ﺑﻴﺶ ﺍﺯ ﺁﻧﮑﻪ ﻣﻴﻞ ﺑﻪ ﺿﻌﻒ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻣﻴﻞ ﺑﻪ ﺗﻘﻮﻳﺖ ﻭ ﺑﺎﻻ ﺑﺮﺩﻥ ﻗﻴﻤﺖ ﺭﺍ ﺩﺍﺭﺩ. ﺍﻳﻦ ﻣﻔﻬﻮﻡ ﺩﺭ ﺗﺸﮑﻴﻞ ﮐﻒ‌ﻫﺎ ﻭ ﺳﻘﻒ‌ﻫﺎ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﮐﻪ ﺩﺭ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ، ﮐﻒ ﺑﻌﺪﯼ ﺑﺎﻻﺗﺮ ﺍﺯ ﮐﻒ ﻗﺒﻠﯽ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺷﻮﺩ ﻭ ﭼﻨﺎﻧﭽﻪ ﺳﻘﻒ‌ﻫﺎ ﻧﻴﺰ ﺑﻪ ﺻﻮﺭﺕ ﭘﻴﺎﭘﯽ ﺑﺎﻻﺗﺮ ﺗﺸﮑﻴﻞ ﺷﻮﻧﺪ ﻳﮏ ﺭﻭﻧﺪ ﻗﺪﺭﺗﻤﻨﺪ ﺭﺍ ﺩﺍﺭﻳﻢ ﻭﻟﯽ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ، ﺗﺎ ﺯﻣﺎﻧﯽ ﭘﺎ ﺑﺮ ﺟﺎ ﺗﻠﻘﯽ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﮐﻒ‌ﻫﺎ ﺑﻪ ﺻﻮﺭﺕ ﺻﻌﻮﺩﯼ ﺗﺸﮑﻴﻞ ﺷﻮﻧﺪ.

۲. ﺭﻭﻧﺪ ﻧﺰﻭﻟﯽ: ﺭﻭﻧﺪ ﻧﺰﻭﻟﯽ ﻋﮑﺲ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ ﺍﺳﺖ، ﺑﺎﺯﺍﺭ ﻣﻴﻞ ﺑﻪ ﺗﻀﻌﻴﻒ ﺩﺍﺭﺩ ﻭ ﺍﺯ ﺍﻳﻦ ﺭﻭ ﺳﻘﻒ‌ﻫﺎ ﺑﻪ ﺻﻮﺭﺕ ﻧﺰﻭﻟﯽ ﺗﺸﮑﻴﻞ ﺷﺪﻩ ﻭ ﺗﺎ ﺯﻣﺎﻥ ﺑﺮﻗﺮﺍﺭﯼ ﺍﻳﻦ ﭼﻴﺪﻣﺎﻥ ﻧﺰﻭﻟﯽ، ﺭﻭﻧﺪ ﻧﻴﺰ ﻧﺰﻭﻟﯽ ﺗﻠﻘﯽ ﻣﯽ‌ﺷﻮﺩ.

ﻧﻤﻮﻧﻪ‌ﻫﺎﻳﯽ ﺍﺯ ﺭﻭﻧﺪﻫﺎﯼ ﺳﻪ ﮔﺎﻧﻪ ﺭﺍ ﺩﺭ ﺗﺼﻮﻳﺮ ﺯﻳﺮ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ:

ﭼﻨﺎﻧﭽﻪ ﻗﻴﻤﺖ ﺑﻴﻦ ﺩﻭ ﺳﻄﺢ ﺍﻓﻘﯽ ﮔﻴﺮ ﺑﻴﻔﺘﺪ، ﻳﮏ ﺭﻭﻧﺪ ﺧﻨﺜﯽ (ﻧﻪ ﺻﻌﻮﺩﯼ ﻧﻪ ﻧﺰﻭﻟﯽ) ﺭﺍ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺷﺖ. ﺍﻳﻦ ﺣﺎﻟﺖ ﺯﻣﺎﻧﯽ ﺩﺭ ﺑﺎﺯﺍﺭ ﺷﮑﻞ ﻣﯽ‌ﮔﻴﺮﺩ ﮐﻪ ﻧﻴﺮﻭﻫﺎﯼ ﻋﺮﺿﻪ ﻭ ﺗﻘﺎﺿﺎ ﺩﺭ ﺗﻌﺎﺩﻝ ﻧﺴﺒﯽ ﺑﻪ ﺳﺮ ﻣﯽ‌ﺑﺮﻧﺪ ﻭ ﻫﻴﭻ ﻳﮏ ﻗﺪﺭﺕِ ﺟﺎ ﺑﻪ ﺟﺎ ﮐﺮﺩﻥ ﻗﻴﻤﺖ ﻭ ﺷﮑﺴﺘﻦ ﺳﻄﺢ ﮐﻒ ﻭ ﺳﻘﻒ ﺭﺍ ﻧﺪﺍﺭﻧﺪ.

ﻧﮑﺘﻪ ﺩﻳﮕﺮﯼ ﮐﻪ ﺑﺎﻳﺪ ﻣﺪ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﺗﺎﻳﻢ ﻓﺮﻳﻢ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﺧﻄﻮﻁ ﺭﻭﻧﺪ ﻣﺘﻔﺎﻭﺗﯽ ﺭﺳﻢ ﮐﺮﺩ ﻳﺎ ﺑﻪ ﺑﻴﺎﻥ ﺩﻳﮕﺮ، ﺭﻭﻧﺪﻫﺎﯼ ﻣﺨﺘﻠﻒ -ﺑﻠﻨﺪﻣﺪﺕ، ﻣﻴﺎﻥ‌ﻣﺪﺕ ﻭ ﮐﻮﺗﺎﻩ‌ﻣﺪﺕ- ﺭﺍ ﺩﺭ ﻧﻤﻮﺩﺍﺭ ﺷﻨﺎﺳﺎﻳﯽ ﮐﺮﺩ. ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﺧﻂ ﺭﻭﻧﺪﻫﺎﯼ ﺑﻠﻨﺪ ﻣﺪﺕ ﺩﻳﺮ ﺑﻪ ﺩﻳﺮ ﺗﻮﺳﻂ ﻗﻴﻤﺖ ﻟﻤﺲ ﻣﯽ‌ﺷﻮﻧﺪ ﻭﻟﯽ ﺁﮔﺎﻫﯽ ﺍﺯ ﻣﺤﻞ ﻗﺮﺍﺭ ﮔﻴﺮﯼ ﺁﻧﻬﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺷﺮﺍﻳﻂ ﺧﺎﺹ ﺍﺯ ﺗﺤﻠﻴﻞ‌ﻫﺎﯼ ﺍﺷﺘﺒﺎﻩ ﺟﻠﻮﮔﻴﺮﯼ ﻧﻤﺎﻳﺪ.

 

ﻧﺤﻮﻩ ﺭﺳﻢ ﺧﻄﻮﻁ ﺭﻭﻧﺪ
ﺍﻳﻨﮕﻮﻧﻪ ﻧﻴﺴﺖ ﮐﻪ ﺑﺘﻮﺍﻥ ﺑﻪ ﻫﺮ ﺷﮑﻞ ﺩﻟﺨﻮﺍﻫﯽ ﺧﻄﻮﻁ ﺭﻭﻧﺪ ﺭﺍ ﺭﺳﻢ ﮐﺮﺩ. ﺑﻪ ﺑﻴﺎﻥ ﺩﻳﮕﺮ، ﺑﺎﻳﺪ ﺭﻭﻧﺪﯼ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺗﺎ ﺑﺘﻮﺍﻥ ﺑﺮﺍﯼ ﺁﻥ ﺧﻂ ﺭﻭﻧﺪ ﺗﺮﺳﻴﻢ ﮐﺮﺩ.

ﺑﺮﺍﯼ ﺗﺮﺳﻴﻢ ﻳﮏ ﺧﻂ ﺭﻭﻧﺪ ﻭﺍﻗﻌﯽ، ﻧﻴﺎﺯ ﺑﻪ ﺩﻭ ﺳﻘﻒ ﺩﺭ ﺭﻭﻧﺪ ﻧﺰﻭﻟﯽ ﻳﺎ ﺩﻭ ﮐﻒ ﺩﺭ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ ﺩﺍﺭﻳﻢ ﻭﻟﯽ ﺻﺤﺖ ﺧﻂ ﺭﻭﻧﺪ ﻣﻌﻤﻮﻻً ﺑﻌﺪ ﺍﺯ ﻟﻤﺲ ﺳﻮﻡ ﻭ ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﻣﻨﺎﺳﺐ ﻗﻴﻤﺖ ﺑﻪ ﺧﻂ ﺭﺳﻢ ﺷﺪﻩ ﺗﺄﻳﻴﺪ ﻣﯽ‌ﺷﻮﺩ:

ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺩﻗﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﺮ ﭼﻪ ﻳﮏ ﺧﻂ ﺭﻭﻧﺪ، ﺷﻴﺐ ﺗﻨﺪﺗﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺍﺣﺘﻤﺎﻝ ﺷﮑﺴﺘﻪ ﺷﺪﻥ ﺁﻥ ﺩﺭ ﺯﻣﺎﻥ ﻧﺰﺩﻳﮏ ﻧﻴﺰ ﺑﻴﺸﺘﺮ ﺍﺳﺖ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺍﺗﮑﺎ ﺑﻪ ﺁﻥ ﺑﺮﺍﯼ ﻣﻌﺎﻣﻼﺕ ﺍﺣﺘﻤﺎﻟﯽ ﺧﻄﺮﻧﺎﮎ‌ﺗﺮ، ﺯﻳﺮﺍ ﭼﻨﻴﻦ ﺭﻭﻧﺪﻫﺎﻳﯽ ﺩﺭ ﻣﻮﺍﻗﻊ ﻫﻴﺠﺎﻧﯽ ﺑﺎﺯﺍﺭ ﺷﮑﻞ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﻭ ﻣﻌﻤﻮﻻً ﺑﻪ ﺯﻭﺩﯼ ﺷﻴﺐ ﺻﻌﻮﺩ ﻳﺎ ﻧﺰﻭﻝ ﺁﻧﻬﺎ ﺍﺻﻼﺡ ﻣﯽ‌ﺷﻮﺩ.

ﺧﻄﻮﻁ ﺭﻭﻧﺪ ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﺧﻄﻮﻁ ﺣﻤﺎﻳﺖ ﻭ ﻣﻘﺎﻭﻣﺖ ﺑﺎ ﻟﻤﺲ ﭼﻨﺪ ﺑﺎﺭﻩ ﻭ ﭘﺎ ﺑﺮ ﺟﺎ ﻣﺎﻧﺪﻥ ﺩﺭ ﻣﺪﺕ ﺯﻣﺎﻥ ﺑﻴﺸﺘﺮ، ﺍﻫﻤﻴﺖ ﺑﺎﻻﺗﺮﯼ ﻣﯽ‌ﻳﺎﺑﻨﺪ.

ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺩﺭ ﻋﮑﺲ ﻗﺒﻠﯽ ﻗﺎﺑﻞ ﻣﺸﺎﻫﺪﻩ ﺍﺳﺖ، ﺩﺭ ﻭﺍﻗﻊ ﻫﺮ ﺭﻭﻧﺪ ﻣﺘﺸﮑﻞ ﺍﺯ ﻣﻮﺝ‌ﻫﺎﯼ ﮐﻮﭼﮑﺘﺮﯼ ﺩﺭ ﺩﻝ ﺧﻮﺩ ﺍﺳﺖ. ﺩﺭ ﻳﮏ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ، ﻣﻮﺝ‌ﻫﺎﯼ ﺻﻌﻮﺩﯼ ﺩﺍﻣﻨﻪ ﻗﻴﻤﺘﯽ ﺑﻠﻨﺪﺗﺮﯼ ﻧﺴﺒﺖ ﺑﻪ ﻣﻮﺝ‌ﻫﺎﯼ ﻧﺰﻭﻟﯽ ﺩﺍﺭﻧﺪ ﮐﻪ ﺍﻳﻦ ﻧﺸﺎﻧﮕﺮ ﻗﻮﺕ ﻧﻴﺮﻭﻫﺎﯼ ﺻﻌﻮﺩﯼ ﺍﺳﺖ ﻭ ﺑﺎﻋﺚ ﺷﮑﻞ ﮔﻴﺮﯼ ﻳﮏ ﺭﻭﻧﺪ ﺭﻭ ﺑﻪ ﺑﺎﻻ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﻣﻮﺭﺩ ﺭﻭﻧﺪﻫﺎﯼ ﻧﺰﻭﻟﯽ، ﻋﮑﺲ ﺍﻳﻦ ﻣﻮﺭﺩ ﺻﺪﻕ ﻣﯽ‌ﮐﻨﺪ.

ﺍﻣﺎ ﭼﻄﻮﺭ ﺗﺸﺨﻴﺺ ﺩﻫﻴﻢ ﮐﻪ ﻳﮏ ﺭﻭﻧﺪ ﻣﺨﺘﻞ ﺷﺪﻩ؟ ﺑﻪ ﺗﺼﻮﻳﺮ ﺯﻳﺮ ﺩﻗﺖ ﻓﺮﻣﺎﻳﻴﺪ:

ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺩﺭ ﻳﮏ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ ﻗﺮﺍﺭ ﺩﺍﺭﻳﻢ، ﺣﻤﺎﻳﺖ ﺍﻓﻘﯽ ﻗﺒﻠﯽ ﻧﺒﺎﻳﺪ ﺭﻭ ﺑﻪ ﭘﺎﻳﻴﻦ ﺑﺸﮑﻨﺪ ﻭ ﺩﺭ ﻳﮏ ﺭﻭﻧﺪ ﻧﺰﻭﻟﯽ ﻧﻴﺰ ﻣﻘﺎﻭﻣﺖ ﺍﻓﻘﯽ ﻗﺒﻠﯽ ﻧﺒﺎﻳﺪ ﺭﻭ ﺑﻪ ﺑﺎﻻ ﺑﺸﮑﻨﺪ. ﺍﻳﻦ ﻣﻬﻤﺘﺮﻳﻦ ﻧﮑﺘﻪ ﺩﺭ ﺗﺸﺨﻴﺺ ﺑﺮﻗﺮﺍﺭﯼ ﺭﻭﻧﺪ ﺍﺳﺖ. ﮔﺎﻫﯽ ﻳﮏ ﺧﻂ ﺭﻭﻧﺪ ﺷﮑﺴﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﻭﻟﯽ ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﻗﻴﻤﺖ ﺑﻪ ﻭﺭﺍﯼ ﺣﻤﺎﻳﺖ ﻳﺎ ﻣﻘﺎﻭﻣﺖ ﻗﺒﻠﯽ ﻧﺮﻭﺩ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﺍﻃﻤﻴﻨﺎﻥ ﺍﺯ ﻣﺨﺘﻞ ﺷﺪﻥ ﺭﻭﻧﺪ ﺟﺎﺭﯼ ﺑﺎﺯﺍﺭ ﺻﺤﺒﺖ ﮐﺮﺩ، ﺯﻳﺮﺍ ﻓﻘﻂ ﻣﻤﮑﻦ ﺍﺳﺖ ﻗﻴﻤﺖ ﺩﺭ ﺣﺎﻝ ﺗﺼﺤﻴﺢ ﺷﻴﺐ ﺑﺎﺷﺪ.

 

ﺍﻧﻮﺍﻉ ﮐﺎﻧﺎﻝ‌ﻫﺎ ﻭ ﻧﺤﻮﻩ ﺭﺳﻢ ﺁﻧﻬﺎ
ﺍﮔﺮ ﻧﻈﺮﻳﻪ ﺧﻄﻮﻁ ﺭﻭﻧﺪ ﺭﺍ ﻳﮏ ﮔﺎﻡ ﺑﻪ ﺟﻠﻮ ﺑﺒﺮﻳﻢ، ﮐﺎﻧﺎﻝ‌ﻫﺎ ﺭﺍ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺷﺖ. ﻣﻌﻤﻮﻻً ﺭﻭﻧﺪﻫﺎ ﺗﻤﺎﻳﻞ ﺑﻪ ﺣﺮﮐﺖ ﺩﺭ ﺩﺍﺧﻞ ﮐﺎﻧﺎﻝ‌ﻫﺎﯼ ﻗﻴﻤﺘﯽ ﺧﺎﺹ ﺩﺍﺭﻧﺪ ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﺯ ﮐﺎﻧﺎﻝ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮑﯽ ﺍﺯ ﺍﺑﺰﺍﺭﻫﺎﯼ ﺗﺤﻠﻴﻠﯽ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ. ﮐﺎﻧﺎﻝ‌ﻫﺎ ﻧﻴﺰ ﺑﻪ ﺳﻪ ﺩﺳﺘﻪ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﺷﻮﻧﺪ:

۱. ﮐﺎﻧﺎﻝ ﺻﻌﻮﺩﯼ ﮐﻪ ﺩﺭ ﺁﻥ ﻭﻗﻮﻉ ﺳﻘﻒ‌ﻫﺎﯼ ﺑﺎﻻﺗﺮ ﻭ ﮐﻒ‌ﻫﺎﯼ ﺑﺎﻻﺗﺮ ﺭﺍ ﺗﻮﺃﻣﺎﻥ ﺩﺍﺭﻳﻢ.

۲. ﮐﺎﻧﺎﻝ ﻧﺰﻭﻟﯽ ﮐﻪ ﺩﺭ ﺁﻥ ﻭﻗﻮﻉ ﺳﻘﻒ‌ﻫﺎﯼ ﭘﺎﻳﻴﻦ‌ﺗﺮ ﻭ ﮐﻒ‌ﻫﺎﯼ ﭘﺎﻳﻴﻦ‌ﺗﺮ ﺭﺍ ﻫﻤﺰﻣﺎﻥ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﻢ.

۳. ﮐﺎﻧﺎﻝ ﺍﻓﻘﯽ ﮐﻪ ﺩﺭ ﺁﻥ ﻗﻴﻤﺖ ﺑﻴﻦ ﺩﻭ ﺳﻄﺢ ﺍﻓﻘﯽ ﻣﺤﺒﻮﺱ ﻣﯽ‌ﺷﻮﺩ.

ﺩﺭ ﺯﻣﺎﻥ ﺗﺮﺳﻴﻢ ﻳﮏ ﮐﺎﻧﺎﻝ ﺑﺎﻳﺪ ﺩﻗﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺧﻄﻮﻁ ﺁﻥ ﺑﺎ ﻳﮑﺪﻳﮕﺮ ﻣﻮﺍﺯﯼ ﺑﺎﺷﻨﺪ. ﻧﺤﻮﻩ ﺗﺮﺳﻴﻢ ﻳﮏ ﮐﺎﻧﺎﻝ ﺻﻌﻮﺩﯼ ﺑﻪ ﺍﻳﻦ ﺷﮑﻞ ﺍﺳﺖ ﮐﻪ ﻣﺎ ﺍﺑﺘﺪﺍ ﻳﮏ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ ﺭﺍ ﺗﺸﺨﻴﺺ ﻣﯽ‌ﺩﻫﻴﻢ. ﺳﭙﺲ ﺧﻂ ﺭﻭﻧﺪ ﻣﺮﺑﻮﻁ ﺑﻪ ﺁﻥ ﺭﺍ ﺑﺎ ﮐﻤﮏ ﺩﻭ ﮐﻒ ﺻﻌﻮﺩﯼ ﺭﺳﻢ ﻣﯽ‌ﮐﻨﻴﻢ. ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﺮﺍﯼ ﺭﺳﻢ ﺧﻂ ﺑﺎﻻﻳﯽ ﮐﺎﻧﺎﻝ، ﻧﻴﺎﺯﯼ ﺑﻪ ﺩﻭ ﻧﻘﻄﻪ ﻧﺪﺍﺭﻳﻢ ﺑﻠﮑﻪ ﺑﺎ ﺩﺭ ﺩﺳﺖ ﺩﺍﺷﺘﻦ ﺳﻘﻒ ﺗﺸﮑﻴﻞ ﺷﺪﻩ ﺩﺭ ﻓﺎﺻﻠﻪ ﺯﻣﺎﻧﯽ ﻣﻴﺎﻥ ﺩﻭ ﮐﻒ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺧﻄﯽ ﺭﺍ ﺑﺎ ﻋﺒﻮﺭ ﺍﺯ ﺁﻥ ﻭ ﺑﻪ ﻃﻮﺭ ﻣﻮﺍﺯﯼ ﺑﺎ ﺧﻂ ﺭﻭﻧﺪ ﺭﺳﻢ ﮐﻨﻴﻢ:

ﺑﻨﺎﺑﺮﺍﻳﻦ، ﺧﻂ ﮐﺎﻧﺎﻝ ﻫﻤﻴﺸﻪ ﺑﻌﺪ ﺍﺯ ﺗﺸﺨﻴﺺ ﺭﻭﻧﺪ ﻭ ﺗﺮﺳﻴﻢ ﺧﻂ ﺭﻭﻧﺪ ﺑﻪ ﺩﺳﺖ ﻣﯽ‌ﺁﻳﺪ ﻭ ﺗﺎﺑﻌﯽ ﺍﺯ ﺁﻥ ﺍﺳﺖ.
