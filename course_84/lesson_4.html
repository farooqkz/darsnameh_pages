ﺩﺭﺱ ﻫﻔﺘﻢ- ﻣﻴﺎﻧﮕﻴﻦ ﻫﺎﯼ ﻣﺘﺤﺮﮎ ﻭ ﮐﺎﺭﺑﺮﺩ ﺁﻧﻬﺎ ﺩﺭ ﺗﺤﻠﻴﻞ

ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺍﺑﺰﺍﺭﯼ ﺳﺎﺩﻩ ﺍﺳﺖ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺣﺮﮐﺎﺕ ﻗﻴﻤﺖ ﺩﺭ ﻃﻮﻝ ﺯﻣﺎﻥ ﺭﺍ ﺑﻪ ﺷﮑﻠﯽ ﻫﻤﻮﺍﺭ ﺑﺮ ﺭﻭﯼ ﻧﻤﻮﺩﺍﺭ ﭘﻴﺎﺩﻩ ﮐﻨﻴﻢ. ﮔﺮﻓﺘﻦ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺍﺯ «ﻳﮏ ﻗﻴﻤﺖ» (ﻣﺜﻼ ﻗﻴﻤﺖ ﺑﺴﺘﻪ ﺷﺪﻥ ﺩﺭ ﺗﻤﺎﻡ ﺩﻭﺭﻩ‌ﻫﺎ) ﻳﺎ «ﻣﻴﺎﻧﮕﻴﻦ ﭼﻨﺪ ﻗﻴﻤﺖ» (ﻣﺜﻼ ﻣﺠﻤﻮﻉ ﻗﻴﻤﺖ ﺳﻘﻒ، ﻗﻴﻤﺖ ﮐﻒ ﻭ ﻗﻴﻤﺖ ﭘﺎﻳﺎﻧﯽ ﺗﻘﺴﻴﻢ ﺑﺮ ﺳﻪ،) ﺑﺮﺍﯼ N ﺩﻭﺭﻩ ﺍﺧﻴﺮ «ﻣﻴﺎﻧﮕﻴﻦ» ﺑﮕﻴﺮﻳﺪ.

ﻣﺜﻞ ﺗﻤﺎﻡ ﻧﺸﺎﻧﮕﺮﻫﺎﯼ ﮐﻤﮑﯽ، ﻳﮏ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻧﻴﺰ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﺮﺍﯼ ﻓﻬﻢ ﺭﻓﺘﺎﺭ ﻓﻌﻠﯽ ﻭ ﭘﺘﺎﻧﺴﻴﻞ‌ﻫﺎﯼ ﺁﻥ ﺑﻪ ﻣﺎ ﮐﻤﮏ ﮐﻨﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﺍﺳﺘﺨﺮﺍﺝ ﺑﺮﺧﯽ ﺍﻃﻼﻋﺎﺕ ﺍﺯ ﺭﻓﺘﺎﺭ ﻳﮏ ﻧﺸﺎﻧﮕﺮ، «ﭘﺘﺎﻧﺴﻴﻞ ﺣﺮﮐﺖ ﻗﻴﻤﺖ ﺩﺭ ﺁﻳﻨﺪﻩ» ﻳﺎ «ﺳﻄﻮﺡ ﺣﺴﺎﺱ» ﺭﺍ ﺗﺨﻤﻴﻦ ﺯﺩ.

ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﺮ ﺍﺳﺎﺱ ﻗﻴﻤﺖ‌ﻫﺎﯼ ﻗﺒﻠﯽ ﺑﻪ ﺩﺳﺖ ﻣﯽ‌ﺁﻳﺪ ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﻤﻴﺸﻪ ﺍﺯ ﺑﺎﺯﺍﺭ ﻋﻘﺐ ﺍﺳﺖ ﻭ ﻗﺎﺑﻠﻴﺖ ﭘﻴﺶ‌ﺑﻴﻨﯽ ﮐﻨﻨﺪﮔﯽ ﮐﻤﺘﺮﯼ ﺩﺍﺭﺩ ﻭ ﺍﺯ ﺍﻳﻦ ﺭﻭ ﺑﻴﺸﺘﺮ ﻣﻌﺎﻣﻠﻪ‌ﮔﺮﺍﻥ ﺍﺯ ﺁﻥ ﺑﻪ ﻋﻨﻮﺍﻥ ﺣﻤﺎﻳﺖ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﻟﺒﺘﻪ ﺍﺣﺘﻤﺎﻝ ﺍﻳﻔﺎﯼ ﻧﻘﺶ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﻘﺎﻭﻣﺖ ﻧﻴﺰ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻭﻟﯽ ﺗﻨﻬﺎ ﺩﺭ ﺗﺎﻳﻢ ﻓﺮﻳﻢ‌ﻫﺎﯼ ﺑﺎﻻ (ﺭﻭﺯﺍﻧﻪ، ﻫﻔﺘﮕﯽ ﻳﺎ ﻣﺎﻫﺎﻧﻪ).

ﻓﺮﻣﻮﻝ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺑﺮﺍﯼ ﻣﺤﺎﺳﺒﻪ ﺍﻧﻮﺍﻉ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻫﺮ ﭼﻪ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻧﺮﻡ‌ﺗﺮ ﺑﺎﺷﺪ، ﺗﺒﻌﻴﺖ ﺁﻥ ﺍﺯ ﺣﺮﮐﺎﺕ ﺟﺎﺭﯼ ﺑﺎﺯﺍﺭ ﻫﻢ ﺁﺭﺍﻡ‌ﺗﺮ ﺍﺳﺖ ﻭ ﺑﺮﻋﮑﺲ ﻫﺮ ﭼﻪ ﻣﺘﻼﻃﻢ‌ﺗﺮ ﺑﺎﺷﺪ، ﺳﺮﻳﻊ‌ﺗﺮ ﺗﻐﻴﻴﺮﺍﺕ ﻗﻴﻤﺖ ﺭﺍ ﺩﺭ ﺧﻮﺩ ﻣﻨﻌﮑﺲ ﻣﯽ‌ﮐﻨﺪ.

ﺍﺑﺘﺪﺍ ﺑﻪ ﻣﻌﺮﻓﯽ ﺩﻭ ﻧﻮﻉ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﭘﺮ ﮐﺎﺭﺑﺮﺩ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ: ﺳﺎﺩﻩ (Simple) ﻭ ﻧﻤﺎﻳﯽ (Exponential).

 

ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺳﺎﺩﻩ (Simple Moving Average)
ﺍﻳﻦ، ﺳﺎﺩﻩ‌ﺗﺮﻳﻦ ﻧﻮﻉ ﺍﺯ ﻣﻴﺎﻧﮕﻴﻦ‌ﻫﺎﯼ ﻣﺘﺤﺮﮎ ﺍﺳﺖ ﻭ ﻧﺤﻮﻩ ﻣﺤﺎﺳﺒﻪ‌ﺍﺵ ﻧﻴﺰ ﭘﻴﭽﻴﺪﻩ ﻧﻴﺴﺖ. ﮐﺎﻓﯽ ﺍﺳﺖ ﻳﮏ ﻗﻴﻤﺖ ﻣﺸﺨﺺ (ﻗﻴﻤﺖ ﺁﻏﺎﺯﻳﻦ، ﺳﻘﻒ، ﮐﻒ ﻳﺎ ﭘﺎﻳﺎﻧﯽ) ﺭﺍ ﺑﺮﺍﯼ N ﺩﻭﺭﻩ ﺟﻤﻊ ﺯﺩﻩ ﻭ ﺳﭙﺲ ﺗﻘﺴﻴﻢ ﺑﺮ ﺗﻌﺪﺍﺩﺷﺎﻥ (N) ﻧﻤﺎﻳﻴﺪ.

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺗﺼﻮﻳﺮ ﻗﺒﻞ ﻣﻼﺣﻈﻪ ﻣﯽ‌ﮐﻨﻴﺪ، ﻫﺮ ﭼﻪ ﺗﻌﺪﺍﺩ ﺩﻭﺭﻩ‌ﻫﺎﯼ ﻟﺤﺎﻅ ﺷﺪﻩ ﺩﺭ ﻣﺤﺎﺳﺒﻪ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﻴﺸﺘﺮ ﺷﻮﺩ (62 SMA ﻳﻌﻨﯽ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﺮﺍﯼ ۶۲ ﺷﻤﻊ ﺍﺧﻴﺮ) ﺧﻂ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺍﺯ ﻗﻴﻤﺖ ﺩﻭﺭﺗﺮ ﻣﯽ‌ﮔﺮﺩﺩ ﻭ ﺑﺮ ﻋﮑﺲ ﻫﺮ ﭼﻪ ﺗﻌﺪﺍﺩ ﺩﻭﺭﻩ‌ﻫﺎ ﮐﻤﺘﺮ ﺑﺎﺷﺪ (5 SMA) ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﻪ ﻗﻴﻤﺖ ﻧﺰﺩﻳﮏ‌ﺗﺮ ﻣﯽ‌ﺷﻮﺩ. ﺗﻘﺮﻳﺒﺎً ﻫﺮ ﺩﻭﯼ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻤﺘﺮﯼ ﺩﺍﺭﻧﺪ ﻭ ﮐﺎﺭﺑﺮﺩ ﺑﻴﺸﺘﺮ ﺍﺯ ﺁﻥ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮐﯽ ﺑﺎ ﺩﻭﺭﻩ‌ﻫﺎﯼ ﻣﻌﻘﻮﻝ (30 SMA) ﺍﺳﺖ.

 

ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻧﻤﺎﻳﯽ (Exponential Moving Average)
ﻧﻮﻉ ﺩﻳﮕﺮ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﺎ ﮐﺎﺭﺑﺮﺩ ﺯﻳﺎﺩ، ﻣﻴﺎﻧﮕﻴﻦ ﻧﻤﺎﻳﯽ ﺍﺳﺖ. ﮔﺎﻫﯽ ﻣﻮﺍﻗﻊ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺳﺎﺩﻩ، ﺑﻴﺶ ﺍﺯ ﺣﺪ ﺳﺎﺩﻩ ﺍﺳﺖ ﻃﻮﺭﯼ ﮐﻪ ﻧﻮﺳﺎﻧﺎﺕ ﻣﻘﻄﻌﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺭﻭﯼ ﺗﻔﺴﻴﺮﻫﺎ ﺍﺯ ﺁﻥ ﺗﺄﺛﻴﺮ ﺑﮕﺬﺍﺭﺩ ﻭ ﺑﺎﻋﺚ ﺑﺮﺩﺍﺷﺖ‌ﻫﺎﯼ ﺍﺷﺘﺒﺎﻩ ﻧﻴﺰ ﺑﺸﻮﺩ. ﺍﺯ ﺍﻳﻦ ﺭﻭ ﻣﯽ‌ﺗﻮﺍﻥ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻧﻤﺎﻳﯽ (EMA) ﺭﺍ ﺑﻪ ﺟﺎﯼ ﺍﺵ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﺩﺍﺩ ﮐﻪ ﺑﺎ ﺁﻥ ﺗﺤﻠﻴﻠﮕﺮ ﻗﺎﺩﺭ ﺍﺳﺖ ﺑﻪ ﺩﺍﺩﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻭﺯﻥ ﺑﺪﻫﺪ ﻭ ﺍﺛﺮ ﻧﻮﺳﺎﻥ‌ﻫﺎﯼ ﻣﻘﻄﻌﯽ ﺭﺍ ﺧﻨﺜﯽ ﻧﻤﺎﻳﺪ.

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺗﺼﻮﻳﺮ ﺑﺎﻻ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﺷﻮﺩ، ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻧﻤﺎﻳﯽ ﮐﻪ ﺑﺎ ﺧﻂ ﻗﺮﻣﺰ ﻣﺸﺨﺺ ﺷﺪﻩ ﻣﻴﻞ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﻧﺰﺩﻳﮑﯽ ﺑﻪ ﻗﻴﻤﺖ ﺍﺯ ﺧﻮﺩ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺍﻳﻦ ﻳﻌﻨﯽ ﺭﻓﺘﺎﺭ ﻗﻴﻤﺖ ﺭﺍ ﺑﻪ ﺷﮑﻞ ﻭﺍﻗﻌﯽ‌ﺗﺮﯼ ﻧﻤﺎﻳﺎﻥ ﻣﯽ‌ﺳﺎﺯﺩ. ﻋﻠﺖ ﺩﺭ ﺍﻳﻦ ﻧﻬﻔﺘﻪ ﺍﺳﺖ ﮐﻪ ﻧﺤﻮﻩ ﻣﺤﺎﺳﺒﻪ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻧﻤﺎﻳﯽ، ﺗﻤﺮﮐﺰ ﺑﻴﺸﺘﺮ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺩﺍﺩﻩ‌ﻫﺎﯼ ﻗﻴﻤﺘﯽ ﺍﺧﻴﺮ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﺪ ﺯﻳﺮﺍ ﺁﻧﭽﻪ ﺍﺧﻴﺮﺍً ﺭﺥ ﺩﺍﺩﻩ ﻧﺴﺒﺖ ﺑﻪ ﺩﺍﺩﻩ‌ﻫﺎﯼ ﻣﻘﺎﻃﻊ ﺩﻭﺭ ﻳﺎ ﺧﻴﻠﯽ ﺩﻭﺭ ﻣﻬﻢ‌ﺗﺮ ﺍﺳﺖ.

ﻫﺮ ﻳﮏ ﺍﺯ ﺍﻳﻦ ﺩﻭ ﻧﻮﻉ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ، ﺑﺮﺍﯼ ﺧﻮﺩ ﻣﺰﺍﻳﺎ ﻭ ﻣﻌﺎﻳﺒﯽ ﺩﺍﺭﻧﺪ:

- ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺳﺎﺩﻩ، ﺣﺮﮐﺎﺕ ﻧﺮﻣﯽ ﺭﺍ ﺩﺭ ﻧﻤﻮﺩﺍﺭ ﺑﻪ ﻧﻤﺎﻳﺶ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﮐﻪ ﺍﻳﻦ ﺑﻪ ﻣﻌﻨﺎﯼ ﺣﺬﻑ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺳﻴﮕﻨﺎﻝ‌ﻫﺎﯼ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ. ﺍﻣﺎ ﺩﺭ ﻣﻘﺎﺑﻞ، ﻣﻤﮑﻦ ﺍﺳﺖ ﻋﻤﻠﮑﺮﺩ ﮐﻨﺪ ﺁﻥ ﺑﺎﻋﺚ ﻭﺭﻭﺩ ﺩﻳﺮﻫﻨﮕﺎﻡ ﻭ ﺗﺄﺧﻴﺮﯼ ﺑﻪ ﻣﻌﺎﻣﻼﺕ ﺷﻮﺩ.

- ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻧﻤﺎﻳﯽ، ﺳﺮﻳﻊ‌ﺗﺮ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺣﺮﮐﺎﺕ ﻗﻴﻤﺖ ﻫﻤﺎﻫﻨﮓ ﮐﺮﺩﻩ ﻭ ﻧﻮﺳﺎﻥ‌ﻫﺎﯼ ﺍﺧﻴﺮ ﻗﻴﻤﺖ ﺭﺍ ﺑﻬﺘﺮ ﺑﻪ ﻧﻤﺎﻳﺶ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﻭﻟﯽ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﻳﻦ ﻣﺰﻳﺖ، ﺍﺯ ﻳﮏ ﻋﻴﺐ ﺑﺮﺧﻮﺭﺩﺍﺭ ﺍﺳﺖ ﻭ ﺁﻥ ﺍﺣﺘﻤﺎﻝ ﺑﺮﻭﺯ ﺳﻴﮕﻨﺎﻝ‌ﻫﺎﯼ ﺍﺷﺘﺒﺎﻩ ﻣﻌﺎﻣﻼﺗﯽ ﺑﻪ ﺧﺎﻃﺮ ﻧﻮﺳﺎﻥ‌ﻫﺎﯼ ﺳﺮﻳﻌﺶ ﺍﺳﺖ.

 

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻳﮏ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﺮ ﺭﻭﯼ ﻧﻤﻮﺩﺍﺭ
ﻳﮑﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﮐﺎﺭﺑﺮﺩﻫﺎﯼ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺗﺸﺨﻴﺺ ﺭﻭﻧﺪ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ. ﻣﻴﺎﻧﮕﻴﻦ‌ﻫﺎﯼ ﻣﺘﺤﺮﮎ ﺑﻪ ﻣﺎ ﺩﻳﺪ ﻭﺳﻴﻊ‌ﺗﺮﯼ ﻣﯽ‌ﺩﻫﻨﺪ ﻭ ﺟﻬﺖ، ﺷﻴﺐ ﻭ ﻣﻮﻗﻌﻴﺖ ﻗﻴﻤﺖ ﻧﺴﺒﺖ ﺑﻪ ﺁﻧﻬﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻌﻴﺎﺭﯼ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﺗﺸﺨﻴﺺ ﺭﻭﻧﺪ ﺑﺎﺯﺍﺭ ﺑﺎﺷﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ:

- ﺯﻣﺎﻧﯽ ﮐﻪ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺻﻌﻮﺩﯼ ﺍﺳﺖ ﻣﯽ‌ﺗﻮﺍﻥ ﺟﻬﺖ ﺑﺎﺯﺍﺭ ﺩﺭ ﺗﺎﻳﻢ ﻓﺮﻳﻢ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﻧﻴﺰ ﺻﻌﻮﺩﯼ ﺩﺍﻧﺴﺖ. ﻭ ﺑﺎﻟﻌﮑﺲ.

- ﺯﻣﺎﻧﯽ ﮐﻪ ﻗﻴﻤﺖ ﺩﺭ ﺑﺎﻻﯼ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺻﻌﻮﺩﯼ ﺍﺳﺖ، ﺑﺎﺯ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻥ ﺟﻬﺖ ﻏﺎﻟﺐ ﺭﺍ ﺻﻌﻮﺩﯼ ﺩﺍﻧﺴﺖ. ﻭ ﺑﺎﻟﻌﮑﺲ.

- ﺯﻣﺎﻧﯽ ﮐﻪ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺻﻌﻮﺩﯼ ﺍﺳﺖ ﻭ ﻗﻴﻤﺖ ﺩﺭ ﺑﺎﻻﯼ ﺁﻥ ﻗﺮﺍﺭ ﺩﺍﺭﺩ، ﻭ ﺿﻤﻨﺎً ﻓﺎﺻﻠﻪ ﻗﻴﻤﺖ ﺗﺎ ﺧﻂ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻧﻴﺰ ﺯﻳﺎﺩ ﺍﺳﺖ، ﺭﻭﻧﺪ ﺟﺎﺭﯼ ﺑﺎﺯﺍﺭ ﻳﮏ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ ﻗﺪﺭﺗﻤﻨﺪ ﺍﺳﺖ. ﻭ ﺑﺎﻟﻌﮑﺲ.

ﻧﮑﺘﻪ: ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺖ ﮐﻪ ﻗﻴﻤﺖ ﻣﻴﻞ ﺑﻪ ﺩﻭﺭ ﺷﺪﻥ ﺍﺯ (ﻭ ﻧﺰﺩﻳﮏ ﺷﺪﻥ ﺑﻪ) ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺭﺍ ﺩﺍﺭﺩ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﻭﻗﺘﯽ ﺑﺎ ﻳﮏ ﺭﻭﻧﺪ ﻗﺪﺭﺗﻤﻨﺪ ﺭﻭﺑﺮﻭ ﻫﺴﺘﻴﻢ، ﻗﻴﻤﺖ ﺍﺯ ﺧﻂ ﻣﻴﺎﻧﮕﻴﻦ ﺩﻭﺭ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺳﭙﺲ ﺑﺎ ﮐﺎﺳﺘﻪ ﺷﺪﻥ ﺍﺯ ﻓﺸﺎﺭ ﺑﺎﺯﺍﺭ ﺑﻪ ﺳﻤﺖ ﺁﻥ ﺑﺎﺯﮔﺸﺖ ﮐﺮﺩﻩ ﻭ ﺍﮔﺮ ﺭﻭﻧﺪ ﺟﺎﺭﯼ ﻫﻤﭽﻨﺎﻥ ﺑﺨﻮﺍﻫﺪ ﺑﺮﻗﺮﺍﺭ ﺑﻤﺎﻧﺪ، ﺩﺭ ﺣﻮﺍﻟﯽ ﺧﻂ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻭ ﺩﻭﺑﺎﺭﻩ ﺑﺎ ﻗﺪﺭﺕ ﮔﺮﻓﺘﻦ ﺑﺎﺯﺍﺭ، ﺷﺎﻫﺪ ﺩﻭﺭ ﺷﺪﻥ ﻗﻴﻤﺖ ﺍﺯ ﺁﻥ ﻫﺴﺘﻴﻢ.

- ﺯﻣﺎﻧﯽ ﮐﻪ ﺩﺭ ﻳﮏ ﺑﺎﺯﺍﺭ ﺻﻌﻮﺩﯼ، ﻗﻴﻤﺖ ﺍﺯ ﺑﺎﻻ ﺳﻘﻮﻁ ﮐﺮﺩﻩ ﻭ ﺧﻂ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺻﻌﻮﺩﯼ ﺭﺍ ﺭﻭ ﺑﻪ ﭘﺎﻳﻴﻦ ﻣﯽ‌ﺷﮑﻨﺪ، ﺑﺎﺯﺍﺭ (ﺑﻪ ﻟﺤﺎﻅ ﺗﺤﻠﻴﻠﯽ) ﻭﺍﺭﺩ ﺣﺎﻟﺘﯽ ﻧﺎﻣﺸﺨﺺ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺍﻳﻦ ﻭﺿﻌﻴﺖ ﻧﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﻧﺰﻭﻟﯽ ﺷﺪﻥ ﻭ ﻧﻪ ﺑﻪ ﺻﻌﻮﺩﯼ ﺷﺪﻥ ﺑﺎﺯﺍﺭ ﺍﻃﻤﻴﻨﺎﻥ ﺩﺍﺷﺖ ﻭ ﺍﺯ ﺍﻳﻦ ﺭﻭ ﺑﺎﻳﺪ ﺻﺒﺮ ﮐﺮﺩ ﺗﺎ ﻗﻴﻤﺖ، ﻳﮑﯽ ﺍﺯ ﺍﻳﻦ ﺩﻭ ﻭﺿﻌﻴﺖ ﺭﺍ ﺭﻗﻢ ﺑﺰﻧﺪ:

۱. ﻗﻴﻤﺖ ﺑﺎ ﻧﺰﻭﻝ ﺑﻴﺸﺘﺮ ﺁﺭﺍﻳﺶ ﺟﺪﻳﺪ ﻧﺰﻭﻟﯽ ﺭﺍ ﺑﻪ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺗﺤﻤﻴﻞ ﮐﻨﺪ ﻭ ﺧﻮﺩ ﻧﻴﺰ ﺩﺭ ﭘﺎﻳﻴﻦ ﺁﻥ ﻗﺮﺍﺭ ﮔﻴﺮﺩ.

۲. ﻗﻴﻤﺖ ﺩﻭﺑﺎﺭﻩ ﺑﻪ ﺑﺎﻻﯼ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﺎﺯﮔﺮﺩﺩ ﻭ ﻫﺮ ﺩﻭ ﻧﻴﺰ ﺻﻌﻮﺩﯼ ﺑﺎﺷﻨﺪ.

ﺩﺭ ﻫﺮ ﻳﮏ ﺍﺯ ﺩﻭ ﺣﺎﻟﺖ ﻓﻮﻕ، ﻭﺿﻌﻴﺖ ﺑﺎﺯﺍﺭ ﺑﻪ ﻟﺤﺎﻅ ﺭﻭﻧﺪﺩﺍﺭ ﺑﻮﺩﻥ ﻣﺸﺨﺺ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﻓﻬﻢ ﺑﻬﺘﺮ ﺍﻫﻤﻴﺖ ﻫﻢ ﺟﻬﺖ ﻭ ﻫﻢ ﺷﻴﺐ ﺑﻮﺩﻥ ﻗﻴﻤﺖ ﻭ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ، ﺑﻪ ﻣﺜﺎﻝ ﺯﻳﺮ ﺗﻮﺟﻪ ﮐﻨﻴﺪ:

ﭼﻨﺎﻧﭽﻪ ﺩﺭ ﻣﻮﻗﻌﻴﺖ ﻓﻮﻕ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻋﺒﻮﺭ ﻗﻴﻤﺖ ﺍﺯ ﺧﻂ ﺭﻭﻧﺪ ﻧﺰﻭﻟﯽ، ﻗﺼﺪ ﻭﺭﻭﺩ ﺑﻪ ﻣﻌﺎﻣﻠﻪ ﺧﺮﻳﺪ ﺭﺍ ﺩﺍﺷﺘﻴﻢ، ﺩﭼﺎﺭ ﺿﺮﺭ ﻣﯽ‌ﺷﺪﻳﻢ. ﺯﻳﺮﺍ ﺩﺭ ﺍﺩﺍﻣﻪ، ﻗﻴﻤﺖ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺣﺮﮐﺖ ﮐﺮﺩﻩ ﺍﺳﺖ:

ﺍﺳﺘﻔﺎﺩﻩ ﺻﺮﻑ ﺍﺯ ﻳﮏ ﺍﺑﺰﺍﺭ ﺑﺮﺍﯼ ﺗﺸﺨﻴﺺ ﺗﻐﻴﻴﺮ ﺭﻭﻧﺪ (ﺍﺯ ﻧﺰﻭﻟﯽ ﺑﻪ ﺻﻌﻮﺩﯼ ﻳﺎ ﺑﺮ ﻋﮑﺲ) ﮐﺎﺭ ﺩﺭﺳﺘﯽ ﻧﻴﺴﺖ. ﺍﮐﺜﺮ ﻣﻌﺎﻣﻠﻪ‌ﮔﺮﺍﻥ ﺣﺮﻓﻪ‌ﺍﯼ ﺍﺯ ﺩﻭ ﻳﺎ ﭼﻨﺪ ﺍﺑﺰﺍﺭ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ ﺗﺎ ﺗﺸﺨﻴﺺ ﺩﺭﺳﺘﯽ ﻧﺴﺒﺖ ﺑﻪ ﺗﻐﻴﻴﺮ ﻭﺿﻌﻴﺖ ﺑﺎﺯﺍﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ.

ﺍﻳﻦ ﻧﮑﺘﻪ ﻣﻬﻢ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﺎﺯﺍﺭ ﻳﺎ ﺩﺭ ﺣﺎﻝ ﺍﺩﺍﻣﻪ ﺩﺍﺩﻥ ﺑﻪ ﺭﻭﻧﺪ ﻗﺒﻠﯽ ﺧﻮﺩ ﺍﺳﺖ ﻳﺎ ﺩﺭ ﺣﺎﻝ ﺁﻣﺎﺩﻩ ﺷﺪﻥ ﺑﺮﺍﯼ ﺗﻐﻴﻴﺮ ﺭﻭﻧﺪ. ﺩﺭ ﺯﻣﺎﻧﯽ ﮐﻪ ﺑﺎﺯﺍﺭ ﺭﻭﻧﺪ ﻗﺒﻠﯽ ﺭﺍ ﺍﺩﺍﻣﻪ ﻣﯽ‌ﺩﻫﺪ ﺍﮐﺜﺮ ﻣﻌﺎﻣﻠﻪ‌ﮔﺮﺍﻥ ﻣﺸﮑﻠﯽ ﺑﺮﺍﯼ ﻫﻤﺮﺍﻩ ﺷﺪﻥ ﺑﺎ ﺁﻥ ﻧﺪﺍﺭﻧﺪ ﻭ ﺗﻨﻬﺎ ﺑﺤﺚ ﻭﺭﻭﺩ ﺩﺭ ﻣﺤﻞ ﻭ ﺯﻣﺎﻥ ﺩﺭﺳﺖ ﻣﻄﺮﺡ ﺍﺳﺖ. ﺍﻣﺎ ﺩﺭ ﻫﻨﮕﺎﻡ ﺗﻐﻴﻴﺮ ﺭﻭﻧﺪ، ﺍﮐﺜﺮ ﺍﻓﺮﺍﺩ ﺑﻪ ﻣﺸﮑﻞ ﺑﺮ ﻣﯽ‌ﺧﻮﺭﻧﺪ ﺯﻳﺮﺍ ﺷﻨﺎﺳﺎﻳﯽ ﺻﺤﻴﺢ ﻳﮏ ﺗﻐﻴﻴﺮ ﺭﻭﻧﺪ ﻭﺍﻗﻌﯽ ﻭ ﺗﻤﻴﺰ ﺩﺍﺩﻥ ﺁﻥ ﺍﺯ ﺍﺻﻼﺡ‌ﻫﺎﯼ ﻣﻘﻄﻌﯽ، ﺧﻴﻠﯽ ﻣﺸﮑﻞ‌ﺗﺮ ﺍﺳﺖ ﻭ ﻫﻤﻴﺸﻪ ﺻﺒﺮ ﻣﯽ‌ﺧﻮﺍﻫﺪ.

 

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺗﻘﺎﻃﻊ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻫﺎ ﺑﺮ ﺭﻭﯼ ﻧﻤﻮﺩﺍﺭ
ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺍﺷﺎﺭﻩ ﺷﺪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﭼﻨﺪ ﺍﺑﺰﺍﺭ ﻫﻤﺰﻣﺎﻥ ﺑﺮﺍﯼ ﺗﺄﻳﻴﺪ ﺩﻧﺒﺎﻟﻪ‌ﺩﺍﺭ ﺑﻮﺩﻥ ﻳﮏ ﺭﻭﻧﺪ ﻳﺎ ﺗﻐﻴﻴﺮ ﺁﻥ، ﮔﺰﻳﻨﻪ‌ﺍﯼ ﻋﺎﻟﯽ ﺑﺮﺍﯼ ﮐﺎﺳﺘﻦ ﺍﺯ ﺗﺤﻠﻴﻞ‌ﻫﺎﯼ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ. ﻳﮑﯽ ﺍﺯ ﮐﺎﺭﺑﺮﺩﻫﺎﯼ ﺟﺎﻟﺐ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺯﻣﺎﻧﯽ ﺑﻪ ﭼﺸﻢ ﻣﯽ‌ﺁﻳﺪ ﮐﻪ ﺍﺯ ﺩﻭ ﻳﺎ ﺳﻪ ﻣﻴﺎﻧﮕﻴﻦ ﺑﺎ ﺩﻭﺭﻩ‌ﻫﺎﯼ ﺯﻣﺎﻧﯽ ﻣﺘﻔﺎﻭﺕ، ﺩﺭ ﮐﻨﺎﺭ ﻫﻢ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ.

ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ، ﻋﺒﻮﺭ ﻣﻴﺎﻧﮕﻴﻦ‌ﻫﺎﯼ ﻣﺘﺤﺮﮎ ﺍﺯ ﺭﻭﯼ ﻫﻢ ﻭ ﺟﺎ ﺑﻪ ﺟﺎ ﺷﺪﻥ ﺗﺮﺗﻴﺐ ﺁﻧﻬﺎ ﺑﺮ ﺭﻭﯼ ﻧﻤﻮﺩﺍﺭ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺭﺍﻫﻨﻤﺎﯼ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﺗﻐﻴﻴﺮ ﺭﻭﻧﺪ ﺑﺎﺷﺪ ﻭ ﺗﻐﻴﻴﺮ ﻣﻮﻗﻌﻴﺖ ﻗﻴﻤﺖ ﻧﺴﺒﺖ ﺑﻪ ﺁﻧﻬﺎ ﻫﻢ ﻣﻌﻨﺎﯼ ﺑﻬﺘﺮﯼ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ. ﺑﻪ ﻧﻤﻮﺩﺍﺭ ﺯﻳﺮ ﺗﻮﺟﻪ ﻧﻤﺎﻳﻴﺪ:

ﺍﻳﻦ ﻃﻮﺭ ﻧﻴﺴﺖ ﮐﻪ ﺗﻤﺎﻡ ﺗﻘﺎﻃﻊ‌ﻫﺎﯼ ﺍﻳﻦ ﭼﻨﻴﻨﯽ ﺑﻪ ﺷﮑﻞ ﮔﻴﺮﯼ ﻳﮏ ﺭﻭﻧﺪ ﻣﻌﮑﻮﺱ ﻣﻨﺠﺮ ﺷﻮﺩ. ﺑﺎﻳﺪ ﺩﻗﺖ ﺩﺍﺷﺖ ﮐﻪ ﻫﻤﻴﺸﻪ ﻣﻌﻴﺎﺭ ﺍﺻﻠﯽ «ﺭﻓﺘﺎﺭ ﻗﻴﻤﺖ» ﺍﺳﺖ. ﻳﻌﻨﯽ ﭼﻨﺎﻧﭽﻪ ﺭﻓﺘﺎﺭ ﻗﻴﻤﺖ ﺗﺄﻳﻴﺪﯼ ﺑﺮ ﺗﺤﻠﻴﻠﯽ ﻣﺜﻞ ﺗﻘﺎﻃﻊ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻫﺎ ﻧﺒﺎﺷﺪ، ﺗﺤﻠﻴﻞ‌ﻫﺎ ﻧﺒﺎﻳﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻣﻨﺠﺮ ﺑﻪ ﻭﺭﻭﺩ ﺑﻪ ﻣﻌﺎﻣﻠﻪ ﺷﻮﻧﺪ.

ﺩﺭ ﻣﻮﻗﻌﻴﺖ ﻋﮑﺲ ﻗﺒﻠﯽ، ﭘﺲ ﺍﺯ ﻋﺒﻮﺭ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺁﺑﯽ ﺍﺯ ﻧﺎﺭﻧﺠﯽ ﻭ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﺁﻥ ﺩﺭ ﭘﺎﻳﻴﻦ، ﺑﻪ ﻓﺎﺻﻠﻪ‌ﺍﯼ ﻧﺴﺒﺘﺎ ﮐﻮﺗﺎﻩ ﻗﻴﻤﺖ ﻧﻴﺰ ﺩﺭ ﻗﺎﻟﺐ ﻳﮏ ﺷﻤﻊ ﺑﻠﻨﺪ ﺑﻪ ﺯﻳﺮ ﺍﻳﻦ ﺩﻭ ﻧﺰﻭﻝ ﮐﺮﺩﻩ ﻭ ﺷﺮﺍﻳﻂ ﺍﻳﺪﻩ‌ﺁﻝ ﺑﺮﺍﯼ «ﻧﺰﻭﻟﯽ» ﻧﺎﻣﻴﺪﻥ ﺭﻭﻧﺪ ﺍﺯ ﺁﻥ ﻣﻘﻄﻊ ﺑﻪ ﺑﻌﺪ ﺭﺍ ﭘﺪﻳﺪ ﺁﻭﺭﺩﻩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﭼﻨﻴﻦ ﻣﻮﻗﻌﻴﺘﯽ ﻳﮏ ﻭﺿﻌﻴﺖ ﻣﻨﺎﺳﺐ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ.

ﺩﺭ ﺗﺼﻮﻳﺮ ﻧﺨﺴﺖ ﻗﻴﻤﺖ ﺗﻮﺍﻧﺴﺘﻪ ﺷﮑﺴﺖ ﻫﺎﯼ ﮐﻮﭼﮑﯽ ﺩﺭ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﺑﻪ ﻭﺟﻮﺩ ﺁﻭﺭﺩ ﻭﻟﯽ ﻫﺮ ﺑﺎﺭ ﺑﻪ ﻓﺎﺻﻠﻪ‌ﺍﯼ ﮐﻮﺗﺎﻩ ﺑﻪ ﺯﻳﺮ ﺁﻥ ﺑﺮﮔﺸﺘﻪ. ﺿﻤﻦ ﺁﻧﮑﻪ ﺍﻳﻦ ﺷﮑﺴﺖ‌ﻫﺎ ﺑﺮ ﺧﻼﻑ ﺭﻭﻧﺪ ﺟﺎﺭﯼ ﻭ ﺩﺭ ﺣﺎﻟﯽ ﺭﺥ ﺩﺍﺩﻩ ﮐﻪ ﻣﻴﺎﻧﮕﻴﻦ ﻣﺘﺤﺮﮎ ﻫﻤﭽﻨﺎﻥ ﺣﺎﻟﺖ ﻧﺰﻭﻟﯽ ﻭﺍﺿﺤﯽ ﺩﺍﺷﺘﻪ.
