ﺩﺭﺱ ﻫﺸﺘﻢ- ﺍﻟﮕﻮﻫﺎﯼ ﺷﻤﻌﯽ- ﺑﺨﺶ ﺍﻭﻝ

ﺩﺭ ﺩﺭﺱ ﭼﻬﺎﺭﻡ ﺍﺷﺎﺭﻩ‌ﺍﯼ ﺑﻪ ﺷﮑﻞ ﮐﻠﯽ ﺷﻤﻊ‌ﻫﺎ ﮐﺮﺩﻳﻢ ﻭ ﺣﺎﻻ ﻧﻮﺑﺖ ﺑﻪ ﺑﺮﺭﺳﯽ ﺩﻗﻴﻖ‌ﺗﺮ ﺁﻧﻬﺎ ﺭﺳﻴﺪﻩ. ﺩﺭ ﻗﺮﻥ ﻫﺠﺪﻫﻢ، ﻣﻮﻧﻬﻴﺴﺎ ﻫﻮﻣﺎ ﮐﻪ ﺍﻫﻞ ﮊﺍﭘﻦ ﺑﻮﺩ ﺑﺮﺍﯼ ﻣﻌﺎﻣﻠﻪ ﺑﺮﻧﺞ ﺗﺤﻠﻴﻞ ﺗﮑﻨﻴﮑﯽ ﻣﺨﺼﻮﺹ ﺧﻮﺩ ﺩﺭ ﻗﺎﻟﺐ ﺍﻟﮕﻮﻫﺎﯼ ﺷﻤﻌﯽ ﺭﺍ ﺍﺧﺘﺮﺍﻉ ﮐﺮﺩ ﮐﻪ ﺑﻌﺪﻫﺎ ﺗﻮﺳﻂ ﺍﺳﺘﻴﻮ ﻧﻴﺴﻮﻥ ﺑﻪ ﻏﺮﺏ ﺭﺍﻩ ﻳﺎﻓﺖ ﻭ ﺗﮑﻤﻴﻞ ﺷﺪ.

ﺷﻤﻊ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺩﺭ ﻫﺮ ﺗﺎﻳﻢ ﻓﺮﻳﻤﯽ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﺑﮕﻴﺮﻧﺪ، ﺭﻭﺯﺍﻧﻪ، ۱ﺳﺎﻋﺘﻪ، ۳۰ ﺩﻗﻴﻘﻪ‌ﺍﯼ ﻭ ﻏﻴﺮﻩ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ، ﺍﺗﮑﺎﯼ ﺻﺮﻑ ﺑﻪ ﺁﻧﻬﺎ، ﺩﺭ ﺗﺎﻳﻢ ﻓﺮﻳﻢ‌ﻫﺎﯼ ﺯﻳﺮ ﻳﮏ ﺳﺎﻋﺖ ﺗﻮﺻﻴﻪ ﻧﻤﯽ ﺷﻮﺩ ﺯﻳﺮﺍ ﻧﻮﺳﺎﻧﺎﺕ ﺩﺭ ﺗﺎﻳﻢ‌ﻫﺎﯼ ﭘﺎﻳﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﮔﻤﺮﺍﻩ ﮐﻨﻨﺪﻩ، ﻫﻴﺠﺎﻧﯽ ﻭ ﻏﻴﺮ ﻣﻮﺛﻖ ﺑﺎﺷﻨﺪ. ﻭﻟﯽ ﺩﺭ ﺗﺎﻳﻢ ﻓﺮﻳﻢ‌ﻫﺎﯼ ﺑﺎﻻ ﻣﻔﺎﻫﻴﻢ ﺗﺤﻠﻴﻠﯽ ﺁﻧﻬﺎ ﺧﻴﻠﯽ ﻗﺎﺑﻞ ﺍﺗﮑﺎ‌ﺗﺮ ﺍﺳﺖ.

ﺷﻤﻊ‌ﻫﺎ ﺑﺮﺍﯼ ﺗﻮﺻﻴﻒ ﺭﻓﺘﺎﺭ ﻗﻴﻤﺖ ﺩﺭ ﻫﺮ ﻳﮏ ﺍﺯ ﺑﺎﺯﻩ‌ﻫﺎﯼ ﺯﻣﺎﻧﯽ ﮐﺎﺭﺑﺮﺩ ﺩﺍﺭﻧﺪ. ﭼﻨﺎﻧﭽﻪ ﻗﻴﻤﺖ ﺑﺴﺘﻪ ﺷﺪﻥ ﻳﮏ ﺷﻤﻊ ﺑﺎﻻﺗﺮ ﺍﺯ ﻗﻴﻤﺖ ﺁﻏﺎﺯﻳﻦ ﺁﻥ ﺑﺎﺷﺪ، ﺷﻤﻊ ﺻﻌﻮﺩﯼ ﺍﺳﺖ ﻭ ﺩﺭ ﺣﺎﻟﺖ ﺑﺮ ﻋﮑﺲ، ﻧﺰﻭﻟﯽ. ﺑﺮﺧﯽ ﺍﺯ ﺷﻤﻊ‌ﻫﺎ ﻫﻢ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﺪﻧﻪ ﻭﺍﻗﻌﯽ ﺁﻧﻬﺎ ﻳﮏ ﺧﻂ ﺍﻓﻘﯽ ﺑﺎﺭﻳﮏ ﺍﺳﺖ ﺯﻳﺮﺍ ﻗﻴﻤﺖ ﺁﻏﺎﺯﻳﻦ ﻭ ﭘﺎﻳﺎﻧﯽ ﺁﻧﻬﺎ ﺑﺮ ﻫﻢ ﻣﻨﻄﺒﻖ ﺍﺳﺖ.

 

ﺷﻤﻊ‌ﻫﺎﻳﯽ ﺑﺎ ﺑﺪﻧﻪ‌ﻫﺎﯼ ﮐﻮﺗﺎﻩ ﻳﺎ ﺑﻠﻨﺪ
ﺷﻤﻊ‌ﻫﺎ ﻧﻤﺎﻳﻨﺪﻩ ﻧﻮﺳﺎﻥ ﺑﺎﺯﺍﺭ ﺩﺭ ﻣﺤﺪﻭﺩﻩ ﺯﻣﺎﻧﯽ ﻣﺸﺨﺼﯽ ﻫﺴﺘﻨﺪ. ﺷﻤﻊ‌ﻫﺎﯼ ﺑﻠﻨﺪ ﻧﺸﺎﻥ ﺩﻫﻨﺪﻩ ﺻﻌﻮﺩ ﻳﺎ ﻧﺰﻭﻝ ﻗﻮﯼ ﻫﺴﺘﻨﺪ. ﻫﺮ ﭼﻪ ﺑﺪﻧﻪ ﺑﻠﻨﺪﺗﺮ ﺑﺎﺷﺪ، ﻓﺸﺎﺭ ﺑﻴﺸﺘﺮﯼ ﺩﺭ ﺟﻬﺖ ﻣﻮﺭﺩ ﻧﻈﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻭ ﺍﻳﻦ ﺑﻪ ﻣﻌﻨﺎﯼ ﭼﻴﺮﻩ ﺷﺪﻥ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﻳﺎ ﺧﺮﻳﺪﺍﺭﺍﻥ ﺩﺭ ﻳﮏ ﺑﺎﺯﻩ ﺍﺳﺖ. ﺷﻤﻊ‌ﻫﺎﻳﯽ ﺑﺎ ﺑﺪﻧﻪ ﮐﻮﺗﺎﻩ ﻫﻢ ﻧﺸﺎﻧﮕﺮ ﻧﻮﺳﺎﻥ ﺍﻧﺪﮎ ﻫﺴﺘﻨﺪ.

ﻳﮏ ﺷﻤﻊ ﺳﻔﻴﺪ ﺑﻠﻨﺪ ﻧﺸﺎﻥ ﺩﻫﻨﺪﻩ ﻓﺸﺎﺭ ﺍﺯ ﻃﺮﻑ ﺗﻘﺎﺿﺎ (ﺧﺮﻳﺪ ﺯﻳﺎﺩ) ﺍﺳﺖ. ﻫﺮ ﭼﻪ ﺑﻠﻨﺪﺗﺮ ﺑﺎﺷﺪ ﻭ ﻗﻴﻤﺖ ﺧﺎﺗﻤﻪ ﺑﺎﻻﺗﺮ ﺍﺯ ﻗﻴﻤﺖ ﮔﺸﺎﻳﺶ ﺑﺎﺷﺪ، ﺍﻳﻦ ﻓﺸﺎﺭ ﺑﻴﺸﺘﺮ ﺑﻮﺩﻩ ﻭ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺧﺮﻳﺪﺍﺭﺍﻥ ﺭﻭﻳﮑﺮﺩﯼ ﺗﻬﺎﺟﻤﯽ ﺩﺍﺭﻧﺪ.

ﺷﻤﻊ ﺑﻠﻨﺪ ﺳﻴﺎﻩ ﻧﻴﺮ ﻧﺸﺎﻧﮕﺮ ﻓﺸﺎﺭ ﻋﺮﺿﻪ (ﻓﺮﻭﺵ ﺯﻳﺎﺩ) ﺍﺳﺖ ﻭ ﻓﺎﺻﻠﻪ ﺑﻴﺸﺘﺮ ﻗﻴﻤﺖ ﮔﺸﺎﻳﺶ ﻭ ﺧﺎﺗﻤﻪ ﮐﻪ ﻧﺰﻭﻝ ﺑﻴﺸﺘﺮ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ، ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﻧﻤﺎﻳﻨﺪﻩ ﻗﺪﺭﺕ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﻧﻴﺰ ﻫﺴﺖ.

 

ﺷﻤﻊ‌ﻫﺎﻳﯽ ﺑﺎ ﺳﺎﻳﻪ‌ﻫﺎﯼ ﺑﻠﻨﺪ
ﺳﺎﻳﻪ‌ﻫﺎﯼ ﭘﺎﻳﻴﻨﯽ ﻭ ﺑﺎﻻﻳﯽ ﺩﺭ ﺷﻤﻊ‌ﻫﺎ، ﺳﺮ ﻧﺦ‌ﻫﺎﯼ ﻣﻬﻤﯽ ﺭﺍ ﺩﺭﺑﺎﺭﻩ ﻣﻘﺎﻃﻊ ﻣﺨﺘﻠﻒ ﺑﺎﺯﺍﺭ ﺑﻪ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﻨﺪ. ﻧﻮﮎ ﺳﺎﻳﻪ ﺑﺎﻻﻳﯽ ﺭﺍ ﺳﻘﻒ ﻭ ﻧﻮﮎ ﺳﺎﻳﻪ ﭘﺎﻳﻴﻨﯽ ﺭﺍ ﮐﻒ ﻗﻴﻤﺖ ﺩﺭ ﻫﺮ ﺷﻤﻊ ﻣﯽ‌ﻧﺎﻣﻴﻢ. ﺷﻤﻊ‌ﻫﺎﻳﯽ ﺑﺎ ﺳﺎﻳﻪ‌ﻫﺎﯼ ﺑﻠﻨﺪ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﺑﺮﺧﯽ ﻣﻌﺎﻣﻼﺕ، ﺩﺭ ﻓﺎﺻﻠﻪ‌ﺍﯼ ﺯﻳﺎﺩ ﺍﺯ ﻗﻴﻤﺖ ﮔﺸﺎﻳﺶ ﻳﺎ ﺧﺎﺗﻤﻪ ﺁﻥ ﺷﻤﻊ ﺻﻮﺭﺕ ﮔﺮﻓﺘﻪ‌ﺍﻧﺪ ﻭ ﺑﺎﺯﺍﺭ ﻧﻮﺳﺎﻥ ﺷﺪﻳﺪﯼ ﺩﺍﺷﺘﻪ. ﺍﻣﺎ ﺷﻤﻊ‌ﻫﺎﻳﯽ ﺑﺎ ﺳﺎﻳﻪ‌ﻫﺎﯼ ﮐﻮﺗﺎﻩ، ﻧﺸﺎﻧﮕﺮ ﺍﻧﺠﺎﻡ ﺍﮐﺜﺮ ﻣﻌﺎﻣﻼﺕ ﺩﺭ ﻧﺰﺩﻳﮑﯽ ﻗﻴﻤﺖ‌ﻫﺎﯼ ﮔﺸﺎﻳﺶ ﻭ ﺧﺎﺗﻤﻪ ﻫﺴﺘﻨﺪ.

ﺍﮔﺮ ﺷﻤﻌﯽ ﻳﮏ ﺳﺎﻳﻪ‌ﺍﯼ ﺑﻠﻨﺪ ﺩﺭ ﺑﺎﻻ ﻭ ﺳﺎﻳﻪ‌ﺍﯼ ﮐﻮﺗﺎﻩ ﺩﺭ ﭘﺎﻳﻴﻦ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﺎ ﺍﺳﺖ ﮐﻪ ﺧﺮﻳﺪﺍﺭﺍﻥ (ﻃﺮﻑ ﺗﻘﺎﺿﺎ) ﺍﺑﺘﺪﺍ ﺑﺎﻋﺚ ﺭﺍﻧﺪﻩ ﺷﺪﻥ ﻗﻴﻤﺖ ﺑﻪ ﺑﺎﻻ ﺷﺪﻩ‌ﺍﻧﺪ ﻭﻟﯽ ﺑﻨﺎ ﺑﻪ ﺩﻻﻳﻠﯽ ﻃﺮﻑ ﻋﺮﺿﻪ (ﻓﺮﻭﺷﻨﺪﮔﺎﻥ) ﻗﻴﻤﺖ ﺭﺍ ﺩﻭﺑﺎﺭﻩ ﺑﻪ ﭘﺎﻳﻴﻦ ﺑﺮﮔﺮﺩﺍﻧﺪﻩ.

ﺣﺎﻟﺖ ﻋﮑﺲ ﻧﻴﺰ ﺯﻣﺎﻧﯽ ﺭﺥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﻳﮏ ﺷﻤﻊ ﺑﺎ ﺳﺎﻳﻪ ﺑﻠﻨﺪ ﺩﺭ ﭘﺎﻳﻴﻦ ﻭ ﺳﺎﻳﻪ ﮐﻮﺗﺎﻩ ﺩﺭ ﺑﺎﻻ ﺩﺍﺭﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﻣﻴﻞ ﺑﺎﺯﺍﺭ ﺑﺮﺍﯼ ﭘﺎﻳﻴﻦ ﺭﺍﻧﺪﻥ ﻗﻴﻤﺖ، ﺑﻌﺪ ﺍﺯ ﻳﮏ ﻫﻴﺠﺎﻥ ﺍﻭﻟﻴﻪ ﺑﺎ ﻣﺎﻧﻊ ﺭﻭﺑﺮﻭ ﺷﺪﻩ ﻭ ﻗﻴﻤﺖ ﺑﻪ ﺑﺎﻻ ﺑﺎﺯﮔﺸﺖ ﮐﺮﺩﻩ.

ﻧﮑﺘﻪ: ﺷﻤﻊ‌ﻫﺎ ﺩﺭ ﻧﻤﻮﺩﺍﺭ ﻗﻴﻤﺖ، ﺍﻟﮕﻮﻫﺎﯼ ﺗﮑﯽ، ﺩﻭ ﺗﺎﻳﯽ ﻳﺎ ﺳﻪ ﺗﺎﻳﯽ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﺑﺮ ﺍﺳﺎﺱ ﺗﺸﺨﻴﺺ ﺁﻧﻬﺎ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ‌ﻫﺎﯼ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺗﺤﻠﻴﻞ ﻳﺎ ﺍﻧﺠﺎﻡ ﻣﻌﺎﻣﻠﻪ ﺭﺳﻴﺪ. ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﻪ ﺗﻌﺪﺍﺩﯼ ﺍﺯ ﺍﻳﻦ ﺍﻟﮕﻮﻫﺎﯼ ﭘﺎﻳﻪ‌ﺍﯼ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ:

ﺍﻟﮕﻮﻫﺎﯼ ﭘﺎﻳﻪ‌ﺍﯼ
:Spinning Tops ﺷﻤﻊ‌ﻫﺎﻳﯽ ﺑﺎ «ﺳﺎﻳﻪ ﺑﺎﻻﻳﯽ ﻭ ﺳﺎﻳﻪ ﭘﺎﻳﻴﻨﯽ ﺑﻠﻨﺪ، ﻭ ﺑﺪﻧﻪ ﮐﻮﭼﮏ» ﺭﺍ Spinning Tops (ﻓﺮﻓﺮﻩ) ﻣﯽ‌ﻧﺎﻣﻴﻢ. ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﺑﺪﻧﻪ ﺧﻴﻠﯽ ﮐﻮﭼﮏ ﺍﺳﺖ ﺭﻧﮓ ﺁﻥ ﺍﻫﻤﻴﺘﯽ ﻧﺪﺍﺭﺩ ﻭ ﺗﻐﻴﻴﺮﯼ ﺩﺭ ﺑﺮﺩﺍﺷﺖ ﻣﺎ ﺑﻪ ﻭﺟﻮﺩ ﻧﻤﯽ‌ﺁﻭﺭﺩ. ﺍﻳﻦ ﺍﻟﮕﻮ ﻧﺸﺎﻧﮕﺮ ﺗﺮﺩﻳﺪ ﻭ ﺑﯽ ﺗﺼﻤﻴﻤﯽ ﻫﻤﺰﻣﺎﻥ ﺧﺮﻳﺪﺍﺭﺍﻥ ﻭ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﺍﺳﺖ.

ﺑﺪﻧﻪ ﮐﻮﭼﮏ (ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺳﻴﺎﻩ، ﺳﻔﻴﺪ ﻳﺎ ﺣﺘﯽ ﺧﻄﯽ ﺑﺎﺷﺪ) ﺣﺮﮐﺘﯽ ﮐﻮﭼﮏ ﺍﺯ ﻗﻴﻤﺖ ﮔﺸﺎﻳﺶ ﺗﺎ ﺧﺎﺗﻤﻪ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺩﺭ ﮐﻨﺎﺭ ﺳﺎﻳﻪ‌ﻫﺎﯼ ﺑﻠﻨﺪ ﻣﯽ‌ﺗﻮﺍﻥ ﺍﻳﻦ ﺑﺮﺩﺍﺷﺖ ﺭﺍ ﮐﺮﺩ ﮐﻪ ﺩﺭ ﻧﺒﺮﺩ ﻣﻴﺎﻥ ﺧﺮﻳﺪﺍﺭﺍﻥ ﻭ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ، ﻫﻴﭻ ﻳﮏ ﺑﺮﻧﺪﻩ ﻧﺸﺪﻩ ﻭ ﻗﺪﺭﺕ ﭼﻴﺮﮔﯽ ﺑﺮ ﺩﻳﮕﺮﯼ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﻧﻴﺎﻭﺭﺩﻩ‌ﺍﻧﺪ. ﺍﻣﺎ ﺑﺎ ﻭﺟﻮﺩ ﻧﺰﺩﻳﮏ ﺑﻮﺩﻥ ﻗﻴﻤﺖ ﮔﺸﺎﻳﺶ ﻭ ﺧﺎﺗﻤﻪ، ﺩﺍﻣﻨﻪ ﮐﻠﯽ ﻧﻮﺳﺎﻥ ﺯﻳﺎﺩ ﺑﻮﺩﻩ ﻭ ﺍﻳﻦ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺑﺎﺯﺍﺭ ﭘﺘﺎﻧﺴﻴﻞ ﺣﺮﮐﺖ ﺭﺍ ﺩﺍﺭﺩ ﻭﻟﯽ ﺗﺼﻤﻴﻢ ﻗﺪﺭﺕ‌ﻫﺎﯼ ﺩﺧﻴﻞ ﺩﺭ ﺁﻥ، ﻫﻨﻮﺯ ﻳﮏ ﺳﻮ ﻧﺸﺪﻩ.

ﺍﮔﺮ Spinning Tops ﺩﺭ ﻳﮏ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ ﺷﮑﻞ ﺑﮕﻴﺮﺩ، ﻣﻌﻤﻮﻻً ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﻳﮕﺮ ﺧﺮﻳﺪﺍﺭ ﻋﻤﺪﻩ‌ﺍﯼ ﺑﺎﻗﯽ ﻧﻤﺎﻧﺪﻩ ﻭ ﺑﺎﺯﺍﺭ ﺍﺣﺘﻤﺎﻻً ﺑﺎ ﻳﮏ ﺣﺮﮐﺖ ﻧﺰﻭﻟﯽ ﺩﻧﺒﺎﻝ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﻭ ﭼﻨﺎﻧﭽﻪ Spinning Tops ﺩﺭ ﻳﮏ ﺭﻭﻧﺪ ﻧﺰﻭﻟﯽ ﺷﮑﻞ ﺑﮕﻴﺮﺩ، ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﻳﮕﺮ ﻓﺮﻭﺷﻨﺪﻩ ﺑﺰﺭﮔﯽ ﺑﺎﻗﯽ ﻧﻤﺎﻧﺪﻩ ﻭ ﺑﺎﺯﺍﺭ ﺍﺣﺘﻤﺎﻻً ﺑﺎ ﻳﮏ ﺣﺮﮐﺖ ﺻﻌﻮﺩﯼ، ﻧﺰﻭﻝ ﺧﻮﺩ ﺭﺍ ﺍﺻﻼﺡ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

Marubozu ﺑﻪ ﺭﻧﮓ ﺳﻴﺎﻩ، ﺩﺍﺭﺍﯼ ﺑﺪﻧﻪ‌ﺍﯼ ﻧﺰﻭﻟﯽ ﻭ ﮐﺸﻴﺪﻩ ﺍﺳﺖ. ﻗﻴﻤﺖ ﮔﺸﺎﻳﺶ ﺑﺎ ﺳﻘﻒ ﻫﻤﭙﻮﺷﺎﻧﯽ ﺩﺍﺭﺩ ﻭ ﻗﻴﻤﺖ ﺧﺎﺗﻤﻪ ﻧﻴﺰ ﺑﺎ ﻗﻴﻤﺖ ﮐﻒ. ﺍﻳﻦ ﺷﻤﻊ ﻧﺰﻭﻟﯽ "ﻗﻮﯼ" ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﮐﻨﺘﺮﻝ ﮐﺎﻣﻞ ﺑﺎﺯﺍﺭ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺷﺘﻪ‌ﺍﻧﺪ ﻭ ﻣﻌﻤﻮﻻً‌ ﺣﮑﺎﻳﺖ ﺍﺯ «ﺷﺮﻭﻉ ﻳﮏ ﺭﻭﻧﺪ ﻧﺰﻭﻟﯽ ﻗﻮﯼ» ﻳﺎ «ﻳﮏ ﺭﻭﻧﺪ ﻧﺰﻭﻟﯽ ﺍﺻﻼﺣﯽ ﺩﺭ ﺩﻝ ﻳﮏ ﺭﻭﻧﺪ ﺻﻌﻮﺩﯼ» ﺑﺰﺭﮔﺘﺮ ﺩﺍﺭﺩ.

:Doji ﺳﻮﻣﻴﻦ ﺍﻟﮕﻮﯼ ﭘﺎﻳﻪ‌ﺍﯼ ﮐﻪ ﺑﻪ ﺁﻥ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ، Doji ﺍﺳﺖ. ﺷﻤﻊ‌ﻫﺎﯼ Doji ﻳﮏ ﻣﺸﺨﺼﻪ ﮐﻠﯽ ﺩﺍﺭﻧﺪ، ﺍﻧﻄﺒﺎﻕ ﺩﻗﻴﻖ ﻗﻴﻤﺖ ﮔﺸﺎﻳﺶ ﻭ ﻗﻴﻤﺖ ﺧﺎﺗﻤﻪ (ﻳﺎ ﻻﺍﻗﻞ ﻧﺰﺩﻳﮑﯽ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩ ﺁﻥ ﺩﻭ ﺑﻪ ﻫﻢ ﺩﺭ ﺣﺪﯼ ﮐﻪ ﻓﺎﺻﻠﻪ‌ﺷﺎﻥ ﻏﻴﺮ ﻗﺎﺑﻞ ﺗﺸﺨﻴﺺ ﺑﺎﺷﺪ). ﻫﺮ Doji ﺑﺎﻳﺪ ﺑﺪﻧﻪ‌ﺍﯼ ﺑﺴﻴﺎﺭ ﮐﻮﭼﮏ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﮐﻪ ﺑﻪ ﺷﮑﻞ ﺧﻂ ﺑﺎﺭﻳﮏ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﺪ.
