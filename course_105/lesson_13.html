ﺩﺭﺱ ﺳﻮﻡ- ﻣﻌﺎﻳﻨﻪ ﺑﺎﻟﻴﻨﯽ

ﺩﺭ ﺩﻭ ﺩﺭﺱ ﻗﺒﻞ ﺧﻮﺍﻧﺪﻳﻢ ﺩﺭ ﻣﻮﺍﺟﻬﻪ ﺑﺎ ﻣﺼﺪﻭﻣﯽ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺩﺍﺭﺩ ﺑﺎﻳﺪ ﺍﺑﺘﺪﺍ ﺭﺍﻩ‌ﻫﺎﯼ ﺗﻨﻔﺴﯽ ﻭﯼ ﺭﺍ ﺍﺯ ﻟﺤﺎﻅ ﻣﺴﺪﻭﺩ ﺑﻮﺩﻥ ﻳﺎ ﻧﺒﻮﺩﻥ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ ﻭ ﺩﺭ ﺻﻮﺭﺕ ﺑﺴﺘﻪ ﺑﻮﺩﻥ، ﺑﻬﺘﺮﻳﻦ ﻣﺎﻧﻮﺭ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﻣﺎﻧﻮﺭ ﻓﺸﺎﺭ ﺭﻭﯼ ﭘﻴﺸﺎﻧﯽ ﻭ ﺑﺎﻻ ﺁﻭﺭﺩﻥ ﭼﺎﻧﻪ (Head-tilt chin-lift Maneuver) ﺍﺳﺖ. ﭘﺲ ﺍﺯ ﺁﻥ ﺑﺎﻳﺪ ﺑﺒﻴﻨﻴﻢ ﻣﺼﺪﻭﻡ ﺗﻮﺍﻧﺎﻳﯽ ﺗﻨﻔﺲ ﻃﺒﻴﻌﯽ ﺭﺍ ﺩﺍﺭﺩ ﻳﺎ ﻧﻴﺎﺯ ﺑﻪ ﺗﻨﻔﺲ ﻣﺼﻨﻮﻋﯽ ﺩﺍﺭﺩ ﻭ ﺳﭙﺲ ﻭﺿﻌﻴﺖ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﺑﺪﻥ ﻭﯼ ﺭﺍ ﭼﮏ ﮐﻨﻴﻢ ﺗﺎ ﺩﺭ ﺻﻮﺭﺕ ﻧﻴﺎﺯ ﺑﻪ ﺍﻧﺠﺎﻡ ﭘﺎﻧﺴﻤﺎﻥ ﻳﺎ ﺑﻨﺪ ﺁﻭﺭﺩﻥ ﺧﻮﻧﺮﻳﺰﯼ ﻭﯼ ﺑﭙﺮﺩﺍﺯﻳﻢ. ﭘﺲ ﺍﺯ ﻃﯽ ﺍﻳﻦ ﻣﺮﺍﺣﻞ ﮐﻪ ﺑﻪ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ ﺷﻨﺎﺧﺘﻪ ﻣﯽ‌ﺷﻮﺩ، ﺑﺎﻳﺪ ﺑﻪ ﻣﻌﺎﻳﻨﻪ ﺑﺎﻟﻴﻨﯽ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﭘﺮﺩﺍﺧﺖ. ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺑﺎﻳﺪ ﻣﺸﺨﺾ ﮐﻨﻴﺪ ﮐﻪ ﭼﻪ ﻋﻀﻮﯼ ﺍﺯ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺁﺳﻴﺐ ﺩﻳﺪﻩ ﺍﺳﺖ.


ﺩﺭ ﻣﻌﺎﻳﻨﻪ ﺑﺎﻟﻴﻨﯽ ﺁﻧﭽﻪ ﮐﻪ ﺣﺎﺋﺰ ﺍﻫﻤﻴﺖ ﺍﺳﺖ ﺗﻮﺟﻪ ﺑﻪ ﭼﻬﺎﺭ ﻧﺸﺎﻧﻪ ﺗﻮﺭﻡ، ﺑﺪﺷﮑﻠﯽ ﺍﻧﺪﺍﻡ، ﻟﻤﺲ ﺩﺭﺩﻧﺎﮎ ﺑﺪﻥ ﻭ ﺯﺧﻢ ﺑﺎﺯ ﺩﺭ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﺷﺎﻳﺪ ﺍﻳﻦ ﺑﺨﺶ ﺭﺍ ﺑﺘﻮﺍﻥ ﺍﺯ ﺟﻬﺘﯽ ﺍﺩﺍﻣﻪ ﺑﺨﺶ ﺑﺮﺭﺳﯽ ﻭﺿﻌﻴﺖ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺖ.

ﺑﺮﺍﯼ ﻣﻌﺎﻳﻨﻪ ﺑﺎﻟﻴﻨﯽ، ﺑﺎﻳﺪ ﺍﺯ ﺳﺮ ﻣﺼﺪﻭﻡ ﺷﺮﻭﻉ ﮐﻨﻴﺪ، ﺟﻤﺠﻌﻪ ﻭﯼ ﺭﺍ ﺑﻪ ﺁﺭﺍﻣﯽ ﻟﻤﺲ ﮐﻨﻴﺪ ﺗﺎ ﻣﺘﻮﺟﻪ ﺗﻮﺭﻡ ﻳﺎ ﺷﮑﺴﺘﮕﯽ ﺩﺭ ﺁﻥ ﺷﻮﻳﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﻫﻨﮕﺎﻡ ﺑﺮﺭﺳﯽ ﺳﺮ، ﺩﺭ ﮐﻨﺎﺭ ﮔﻮﺵ‌ﻫﺎ ﻣﺎﻳﻌﯽ ﺷﻔﺎﻑ ﺑﺒﻴﻨﻴﺪ ﮐﻪ ﺩﺭ ﺻﻮﺭﺕ ﺷﮑﺴﺘﮕﯽ ﺷﺪﻳﺪ ﺍﺳﺘﺨﻮﺍﻥ ﺟﻤﺠﻤﻪ ﺑﻪ ﺑﻴﺮﻭﻥ ﺍﺯ ﺁﻥ ﺗﺮﻭﺍﺵ ﮐﺮﺩﻩ ﺍﺳﺖ. ﺍﻳﻦ ﻣﺎﻳﻊ ﺷﻔﺎﻑ، ﻣﺎﻳﻊ ﻣﻐﺰﯼ ﻧﺨﺎﻋﯽ ﺍﺳﺖ ﮐﻪ ﺳﺒﺐ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺑﺮﺧﻮﺭﺩ ﻣﻐﺰ ﺑﺎ ﺟﻤﺠﻤﻪ ﺩﺭ ﺑﺮﺍﺑﺮ ﺁﺳﻴﺐ‌ﻫﺎﯼ ﺷﺪﻳﺪ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﺪ ﮐﻪ ﺳﺮ ﻓﺮﺩ ﺑﺎ ﺁﺳﻴﺐ ﺷﺪﻳﺪ ﻣﻮﺍﺟﻪ ﺷﺪﻩ ﺍﺳﺖ. ﮔﺮﺩﻥ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﻧﻴﺰ ﺍﺯ ﺑﺨﺶ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﻫﻤﺮﺍﻩ ﺳﺮ ﻭﯼ ﺑﺎﻳﺪ ﻣﻮﺭﺩ ﺍﺭﺯﻳﺎﺑﯽ ﻗﺮﺍﺭ ﮔﻴﺮﺩ ﻭ ﺗﻮﺭﻡ، ﺑﺪﺷﮑﻠﯽ ﺍﻧﺪﺍﻡ، ﻟﻤﺲ ﺩﺭﺩﻧﺎﮎ ﻳﺎ ﺯﺧﻢ ﺑﺎﺯ ﺩﺭ ﺁﻥ ﺑﺮﺭﺳﯽ ﺷﻮﺩ.

ﺩﺭ ﻫﻨﮕﺎﻡ ﺑﺮﺭﺳﯽ ﺳﺮ ﻣﺼﺪﻭﻡ، ﭼﺸﻤﺎﻥ ﻭﯼ ﺭﺍ ﺑﻪ ﺁﺭﺍﻣﯽ ﺑﺎﺯ ﮐﺮﺩﻩ ﻭ ﺍﻧﺪﺍﺯﻩ ﻣﺮﺩﻣﮏ‌ﻫﺎﯼ ﺩﻭ ﭼﺸﻢ ﻭﯼ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﺪ. ﺩﺭ ﺣﺎﻟﺖ ﻧﺮﻣﺎﻝ، ﺍﻧﺪﺍﺯﻩ ﻣﺮﺩﻣﮏ ﻫﺮ ﺩﻭ ﭼﺸﻢ ﺑﺮﺍﺑﺮ ﻭ ﮔﺮﺩ ﺍﺳﺖ. ﻣﺮﺩﻣﮏ ﺗﻨﮓ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﺸﺎﻧﻪ ﻣﺴﻤﻮﻣﻴﺖ ﺑﺎ ﻣﻮﺍﺩ ﻣﺨﺪﺭ ﻳﺎ ﺿﺮﺑﻪ ﻣﻐﺰﯼ ﺑﺎﺷﺪ. ﻣﺮﺩﻣﮏ ﮔﺸﺎﺩ ﻧﺸﺎﻥ‌ﺩﻫﻨﺪﻩ ﺿﺮﺑﻪ ﻣﻐﺰﯼ ﻳﺎ ﻭﺍﺭﺩ ﺷﺪﻥ ﺷﻮﮎ ﺑﻪ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺍﺳﺖ ﻭ ﻣﺮﺩﻣﮏ‌ﻫﺎﯼ ﻧﺎﺑﺮﺍﺑﺮ ﻧﻴﺰ ﻳﮑﯽ ﺍﺯ ﻧﺸﺎﻧﻪ‌ﻫﺎﯼ ﺳﮑﺘﻪ ﻣﻐﺰﯼ ﺍﺳﺖ. ﻫﻤﺎﻥ‌ﻃﻮﺭ ﮐﻪ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﺍﻧﺪﺍﺯﻩ ﻏﻴﺮﻃﺒﻴﻌﯽ ﻣﺮﺩﻣﮏ ﺑﺎ ﺗﻐﻴﻴﺮ ﺩﺭ ﻣﻐﺰ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﻣﺮﺗﺒﻂ ﺑﻮﺩﻩ ﻭ ﺑﺮﺭﺳﯽ ‌ﺁﻥ‌ﻫﺎ ﺣﺎﺋﺰ ﺍﻫﻤﻴﺖ ﻭﻳﮋﻩ ﺍﺳﺖ.

ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺷﻤﺎ ﻫﻤﭽﻨﻴﻦ ﺑﺎﻳﺪ ﺑﺎ ﺍﻧﺪﺍﺧﺘﻦ ﻧﻮﺭ ﺑﻪ ﭼﺸﻢ ﻭﯼ ﻣﻄﻤﺌﻦ ﺷﻮﻳﺪ ﮐﻪ ﭼﺸﻤﺎﻥ ﻭﯼ ﺑﻪ ﻧﻮﺭ ﻭﺍﮐﻨﺶ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻭ ﺍﻧﺪﺍﺯﻩ ﻣﺮﺩﻣﮏ‌ﻫﺎ ﮐﻮﭼﮏ ﻣﯽ‌ﺷﻮﺩ. ﻭﺍﮐﻨﺶ ﻣﺮﺩﻣﮏ ﺑﻪ ﻧﻮﺭ ﺍﺯ ﺳﺎﻟﻢ ﺑﻮﺩﻥ ﻣﻐﺰ ﻣﺼﺪﻭﻡ ﺣﮑﺎﻳﺖ ﺩﺍﺭﺩ. ﺍﮔﺮ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺍﻧﺪﺍﺧﺘﻦ ﻧﻮﺭ ﺩﺭ ﭼﺸﻤﺎﻥ ﻣﺼﺪﻭﻡ ﻧﺪﺍﺭﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺻﻮﺭﺕ ﺳﺎﺩﻩ، ﺩﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺑﺮﺍﺑﺮ ﭼﺸﻢ ﻭﯼ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﺗﺎ ﭼﺸﻢ ﻣﺼﺪﻭﻡ ﻣﺤﻴﻂ ﺗﺎﺭﻳﮏ ﺭﺍ ﺗﺠﺮﺑﻪ ﮐﻨﺪ ﻭ ﺳﭙﺲ ﺑﺎ ﺑﺮﺩﺍﺷﺘﻦ ﺩﺳﺘﺎﻥ ﺧﻮﺩ، ﭼﺸﻤﺎﻥ ﻣﺼﺪﻭﻡ ﺭﺍ ﺩﺭ ﺣﺎﻟﺖ ﭘﺮ ﻧﻮﺭ ﻗﺮﺍﺭ ﺩﻫﻴﺪ ﻭ ﺑﺴﺘﻪ ﺷﺪﻥ ﻣﺮﺩﻣﮏ ﻭﯼ ﺭﺍ ﺍﺭﺯﻳﺎﺑﯽ ﮐﻨﻴﺪ.

ﭘﺲ ﺍﺯ ﺑﺮﺭﺳﯽ ﺳﺮ، ﺑﻪ ﺳﻤﺖ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺭﻓﺘﻪ ﻭ ﺑﺎ ﻟﻤﺲ ﺍﻳﻦ ﺑﺨﺶ ﺍﺯ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺍﺯ ﻧﺒﻮﺩ ﺗﻮﺭﻡ، ﺑﺪﺷﮑﻠﯽ ﺍﻧﺪﺍﻡ، ﻟﻤﺲ ﺩﺭﺩﻧﺎﮎ ﻳﺎ ﺯﺧﻢ ﺑﺎﺯ ﺩﺭ ﺁﻥ ﻣﻄﻤﺌﻦ ﺷﻮﻳﺪ.

ﭘﺲ ﺍﺯ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺑﺎﻳﺪ ﺷﮑﻢ ﻣﺼﺪﻭﻡ ﺭﺍ ﺑﻪ ﺁﺭﺍﻣﯽ ﺑﺎ ﻟﻤﺲ ﮐﺮﺩﻥ ﻭ ﻓﺸﺎﺭ ﺍﻧﮕﺸﺘﺎﻥ ﻣﻮﺭﺩ ﺑﺮﺭﺳﯽ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﺍﮔﺮ ﻣﺼﺪﻭﻡ ﺩﺭ ﺑﺨﺸﯽ ﺍﺯ ﺷﮑﻢ ﺧﻮﺩ ﺩﺭﺩ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ، ﺍﻳﻦ ﻧﻘﻄﻪ ﺭﺍ ﺩﺭ ﺁﺧﺮ ﺯﻣﺎﻥ ﺑﺮﺭﺳﯽ ﮐﻮﺗﺎﻩ ﺧﻮﺩ ﻓﺸﺎﺭ ﺩﻫﻴﺪ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﻫﻴﭽﮕﺎﻩ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺑﻪ ﺻﻮﺭﺗﯽ ﺑﺪﻥ ﺑﻴﻤﺎﺭ ﺭﺍ ﻟﻤﺲ ﮐﻨﻴﺪ ﻳﺎ ﻓﺸﺎﺭ ﺩﻫﻴﺪ ﮐﻪ ﺳﺒﺐ ﺍﻓﺰﺍﻳﺶ ﺷﺪﺕ ﺁﺳﻴﺐ ﺩﺭ ﺑﺪﻥ ﻭﯼ ﻧﺸﻮﻳﺪ.

ﭘﺲ ﺍﺯ ﺑﺮﺭﺳﯽ ﺷﮑﻢ، ﻟﮕﻦ ﻭ ﺩﻳﮕﺮ ﻧﻮﺍﺣﯽ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﻫﻤﭽﻮﻥ ﺩﺳﺖ ﻭ ﭘﺎﻫﺎﯼ ﻭﯼ ﺭﺍ ﻣﻮﺭﺩ ﺑﺮﺭﺳﯽ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﺩﺭ ﺑﺮﺭﺳﯽ ﻟﮕﻦ ﻭﯼ، ﺑﺎﻳﺪ ﺩﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﮐﻨﺎﺭ ﻟﮕﻦ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﺑﻪ ﺍﺭﺍﻣﯽ ﺩﻭ ﻃﺮﻑ ﻟﮕﻦ ﻭﯼ ﺭﺍ ﺑﻪ ﺩﺍﺧﻞ ﻓﺸﺎﺭ ﺩﻫﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﻫﻤﭽﻨﻴﻦ ﺑﺎﻳﺪ ﻳﮑﺒﺎﺭ ﺩﻳﮕﺮ ﻧﻴﺰ ﺩﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﺮ ﺑﺎﻻﯼ ﻟﮕﻦ ﻣﺼﺪﻭﻡ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ﺁﻥ ﺭﺍ ﺑﻪ ﺁﺭﺍﻣﯽ ﺑﻪ ﺳﻤﺖ ﭘﺎﻳﻴﻦ ﻓﺸﺎﺭ ﺩﻫﻴﺪ ﺗﺎ ﺍﺯ ﻃﺒﻴﻌﯽ ﺑﻮﺩﻥ ﻭﺿﻌﻴﺖ ﻟﮕﻦ ﻭﯼ ﻣﻄﻤﺌﻦ ﺷﻮﻳﺪ. ﺩﺭ ﺑﺮﺭﺳﯽ ﺩﺳﺖ ﻭ ﭘﺎﻫﺎﯼ ﻣﺼﺪﻭﻡ ﻧﻴﺰ ﺟﺮﻳﺎﻥ ﮔﺮﺩﺵ ﺧﻮﻥ ﺭﺍ ﺑﺎ ﮔﺮﻓﺘﻦ ﻧﺒﺾ ﺩﺭ ﺩﺳﺖ ﻭ ﭘﺎﻫﺎ ﺍﺭﺯﻳﺎﺑﯽ ﮐﻨﻴﺪ ﻭ ﺍﮔﺮ ﻣﺼﺪﻭﻡ ﺩﺭ ﻭﺿﻌﻴﺖ ﻫﻮﺷﻴﺎﺭﯼ ﺍﺳﺖ، ﻭﺿﻌﻴﺖ ﺑﯽ‌ﺣﺴﯽ ﺍﻧﮕﺸﺘﺎﻥ ﺩﺳﺖ ﻳﺎ ﭘﺎﻫﺎﯼ ﻭﯼ ﺭﺍ ﺑﺎ ﻓﺸﺎﺭ ﺩﺍﺩﻥ ﺍﻧﮕﺸﺘﺎﻥ ﻭ ﭘﺮﺳﻴﺪﻥ ﺍﺯ ﻭﯼ ﭼﮏ ﮐﻨﻴﺪ. ﺳﭙﺲ ﺍﺯ ﺍﻭ ﺑﺨﻮﺍﻫﻴﺪ ﮐﻪ ﺩﺳﺖ ﻭ ﭘﺎﯼ ﺧﻮﺩ ﺭﺍ ﺣﺮﮐﺖ ﺩﻫﺪ. ﺑﯽ ﺣﺴﯽ ﻳﺎ ﻋﺪﻡ ﺗﻮﺍﻧﺎﻳﯽ ﺣﺮﮐﺖ ﺩﺭ ﻭﺿﻌﻴﺖ ﺷﺪﻳﺪ ﺣﺘﯽ ﻣﯽ ‌ﻭﺍﻧﺪ ﻧﺸﺎﻥ ﺍﺯ ﺁﺳﻴﺐ ﻧﺨﺎﻋﯽ ﺩﺭ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺑﺎﺷﺪ.

ﺍﮔﺮ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﻧﺘﻮﺍﻧﺪ ﺩﺳﺖ ﻳﺎ ﭘﺎﯼ ﺧﻮﺩ ﺭﺍ ﺣﺮﮐﺖ ﺩﻫﺪ، ﺁﺳﻴﺐ ﻧﺨﺎﻋﯽ ﺩﺭ ﮔﺮﺩﻥ ﻭﯼ ﺭﻭﯼ ﺩﺍﺩﻩ ﻭ ﺍﮔﺮ ﺑﺘﻮﺍﻧﺪ ﺗﻨﻬﺎ ﺩﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺣﺮﮐﺖ ﺩﻫﺪ ﻭ ﺍﺯ ﺣﺮﮐﺖ ﺩﺍﺩﻥ ﭘﺎﻫﺎﯼ ﺧﻮﺩ ﻋﺎﺟﺰ ﺑﺎﺷﺪ، ﺁﺳﻴﺐ ﻧﺨﺎﻋﯽ ﺩﺭ ﭘﺎﻳﻴﻦ ﺗﺮ ﺍﺯ ﮔﺮﺩﻥ ﻭﯼ ﺍﺗﻔﺎﻕ ﺍﻓﺘﺎﺩﻩ ﺍﺳﺖ.


ﻧﺒﺾ ﻣﺼﺪﻭﻡ ﺭﺍ ﭼﻄﻮﺭ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ؟
ﺣﺮﮐﺖ ﺧﻮﻥ ﺩﺭ ﺳﺮﺧﺮﮒ‌ﻫﺎﯼ ﺍﻧﺴﺎﻥ ﻣﺘﺎﺛﺮ ﺍﺯ ﺍﻧﻘﺒﺎﺽ ﻗﻠﺐ ﻭ ﺑﻪ ﺻﻮﺭﺕ ﺿﺮﺑﻪ ﺑﻪ ﺩﻳﻮﺍﺭﻩ ﺭﮒ ﻭ ﺗﭙﺸﯽ ﻭﺍﺭ ﺍﺳﺖ. ﺍﮔﺮ ﺳﺮﺧﺮﮒ ﺩﺭ ﻧﺰﺩﻳﮑﯽ ﭘﻮﺳﺖ ﻭ ﺑﻪ ﻭﻳﮋﻩ ﺩﺭ ﺭﻭﯼ ﺍﺳﺘﺨﻮﺍﻥ ﻗﺮﺍﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺍﻳﻦ ﺣﺮﮐﺖ ﺗﭙﺸﯽ ﺑﻪ ﺻﻮﺭﺕ ﺿﺮﺑﺎﻥ ﺑﺎ ﻟﻤﺲ ﮐﺮﺩﻥ ﻗﺎﺑﻞ ﻣﺸﺎﻫﺪﻩ ﺍﺳﺖ.

ﻧﺒﺾ ﺩﺭ ﺑﺰﺭﮔﺴﺎﻻﻥ ﺩﺭ ﺳﻪ ﻧﻘﻄﻪ ﺍﺯ ﺑﺪﻥ ﺑﻪ ﺧﻮﺑﯽ ﻗﺎﺑﻞ ﻣﺸﺎﻫﺪﻩ ﺍﺳﺖ: ﺩﺭ ﻣﭻ ﺩﺳﺖ، ﺩﺭ ﮔﺮﺩﻥ ﻭ ﺩﺭ ﮐﺸﺎﻟﻪ ﺭﺍﻥ.

ﺑﺮﺍﯼ ﮔﺮﻓﺘﻦ ﻧﺒﺾ ﺩﺭ ﻣﭻ ﺩﺳﺖ، ﮐﺎﻓﯽ ﺍﺳﺖ ﺩﻭ ﺍﻧﮕﺸﺖ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻧﺎﺣﻴﻪ ﺯﻳﺮ ﭼﻴﻦ ﻣﭻ ﺩﺳﺖ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ﺍﻥ ﻧﺎﺣﻴﻪ ﺭﺍ ﺑﻪ ﺁﺭﺍﻣﯽ ﻓﺸﺎﺭ ﺩﻫﻴﺪ ﺗﺎ ﺿﺮﺑﺎﻥ ﻧﺒﺾ ﺭﺍ ﺍﺣﺴﺎﺱ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﺍﺣﺴﺎﺱ ﺿﺮﺑﺎﻥ ﺩﺭ ﻧﺎﺣﻴﻪ ﮔﺮﺩﻥ ﻧﻴﺰ ﮐﺎﻓﯽ ﺍﺳﺖ ﺩﻭ ﺍﻧﮕﺸﺖ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺁﺭﺍﻣﯽ ﺩﺭ ﻧﺎﺣﻴﻪ ﺑﻴﻦ ﻧﺎﯼ ﻭ ﻋﻀﻠﻪ ﺑﺰﺭﮒ ﮔﺮﺩﻥ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ﮐﻤﯽ ﺁﻥ ﻧﺎﺣﻴﻪ ﺭﺍ ﻓﺸﺎﺭ ﺩﻫﻴﺪ. ﺍﮔﺮ ﻣﻮﻓﻖ ﺑﻪ ﻣﺸﺎﻫﺪﻩ ﻧﺒﺾ ﺩﺭ ﺍﻳﻦ ﻧﻮﺍﺣﯽ ﻧﺸﺪﻳﺪ ﮐﺎﻓﯽ ﺍﺳﺖ ﮐﻤﯽ ﻣﻮﻗﻌﻴﺖ ﺍﻧﮕﺸﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ ﺗﺎ ﺑﻪ ﺁﺳﺎﻧﯽ ﻧﺒﺾ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﻧﻮﺍﺣﯽ ﺍﺣﺴﺎﺱ ﮐﻨﻴﺪ. ﺩﺭ ﻫﻨﮕﺎﻡ ﮔﺮﻓﺘﻦ ﻧﺒﺾ ﺍﺯ ﺍﻧﮕﺸﺖ ﺷﺴﺖ ﺧﻮﺩ ﺑﺮﺍﯼ ﮔﺮﻓﺘﻦ ﻧﺒﺾ ﺍﺳﺘﻔﺎﺩﻩ ﻧﮑﻨﻴﺪ ﭼﺮﺍ ﮐﻪ ﺍﻳﻦ ﺍﻧﮕﺸﺖ ﺧﻮﺩ ﺩﺍﺭﺍﯼ ﻧﺒﺾ ﺍﺳﺖ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺤﺎﺳﺒﺎﺕ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﻫﻢ ﺑﺮﻳﺰﺩ.

ﻫﺮ ﺍﻧﺴﺎﻥ ﺑﺰﺭﮔﺴﺎﻝ ۶۰ ﺗﺎ ۱۰۰ ﻧﺒﺾ ﺩﺭ ﻳﮏ ﺩﻗﻴﻘﻪ ﺩﺍﺭﺩ ﻭ ﺍﻳﻦ ﺁﻣﺎﺭ ﺑﺮﺍﯼ ﮐﻮﺩﮐﺎﻥ ﺑﻪ ۷۰ ﺗﺎ ۱۰۰ ﻧﺒﺾ ﺩﺭ ﻫﺮ ﺩﻗﻴﻘﻪ ﻭ ﺑﺮﺍﯼ ﻧﻮﺯﺍﺭﺩﺍﻥ ﺷﻴﺮﺧﻮﺍﺭ ﺑﻪ ۱۲۰ ﺗﺎ ۱۸۰ ﻧﺒﺾ ﺩﺭ ﻫﺮ ﺩﻗﻴﻘﻪ ﻣﯽ‌ﺭﺳﺪ. ﺩﺭ ﻋﻤﻠﻴﺎﺕ ﺍﻣﺪﺍﺩ ﺭﺳﺎﻧﯽ ﻻﺯﻡ ﻧﻴﺴﺖ ﮐﻪ ﻭﻗﺖ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﮔﺮﻓﺘﻦ ﻧﺒﺾ ﻣﺼﺪﻭﻡ ﻫﺪﺭ ﺩﻫﻴﺪ ﺗﺎ ﺑﻪ ﺁﻣﺎﺭ ﭼﻨﺪ ﻧﺒﺾ ﺩﺭ ﺩﻗﻴﻘﻪ ﺑﺮﺳﻴﺪ. ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﻴﺪ ﻣﺸﺨﺺ ﮐﻨﻴﺪ ﮐﻪ ﺁﻳﺎ ﻣﺼﺪﻭﻡ ﻧﺒﺾ ﺩﺍﺭﺩ ﻭ ﺍﮔﺮ ﭼﻨﻴﻦ ﺍﺳﺖ ﻗﻠﺐ ﻭﯼ ﺑﻴﺸﺘﺮ ﻳﺎ ﮐﻤﺘﺮ ﺍﺯ ﺣﺎﻟﺖ ﻧﺮﻣﺎﻝ ﻣﯽ‌ﺗﭙﺪ.

ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻣﻌﻤﻮﻻ ﺩﺭ ﺣﺎﻟﺘﯽ ﮐﻪ ﻓﺮﺩ ﺩﭼﺎﺭ ﺑﯽ‌ﺣﺎﻟﯽ ﻳﺎ ﺍﺳﻬﺎﻝ ﻭ ﺍﺳﺘﻔﺮﺍﻍ ﺷﺪﻳﺪ ﺑﻮﺩﻩ ﻳﺎ ﺩﭼﺎﺭ ﺣﻤﻠﻪ ﻗﻠﺒﯽ ﺷﺪﻩ ﺍﺳﺖ ﻧﺒﺾ ﻭﯼ ﮐﻤﺘﺮ ﺍﺯ ﺣﺎﻟﺖ ﻧﺮﻣﺎﻝ ﺍﺳﺖ ﻭ ﻧﺒﺾ ﺩﺭ ﺍﻓﺮﺍﺩ ﺑﺎ ﺟﺜﻪ ﺑﺮﺯﺭﮒ ﻧﻴﺰ ﮐﻤﺘﺮ ﺍﺯ ﺍﻓﺮﺍﺩ ﺑﺎ ﺟﺜﻪ ﮐﻮﭼﮏ ﻳﺎ ﮐﻮﺩﮐﺎﻥ ﺍﺳﺖ. ﺑﺮﺍﯼ ﮔﺮﻓﺘﻦ ﻧﺒﺾ ﺩﺭ ﺷﻴﺮﺧﻮﺍﺭﺍﻥ ﻧﻴﺰ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﻪ ﺟﺎﯼ ﺭﻭﺵ‌ﻫﺎﯼ ﺑﺎﻻ ﺍﺯ ﺷﺮﻳﺎﻥ ﺑﺎﺯﻭﻳﯽ ﺩﺭ ﻭﺳﻂ ﻭ ﺩﺍﺧﻞ ﺑﺎﺯﻭ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ.

ﺩﺭ ﺑﺮﺧﯽ ﺍﻭﻗﺎﺕ ﺭﻧﮓ ﭘﻮﺳﺖ ﻣﺼﺪﻭﻡ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺣﺎﻭﯼ ﻧﺸﺎﻧﻪ‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺷﺨﺺ ﺍﻣﺪﺍﺩﮔﺮ ﺑﺎﺷﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﻪ ﻣﺼﺪﻭﻣﯽ ﺑﺎ ﺭﻧﮓ ﭘﻮﺳﺖ ﺭﻭﺷﻦ ﻳﺎ ﺳﻔﻴﺪ ﺑﺮﺧﻮﺭﺩﻳﺪ، ﺁﮔﺎﻩ ﺑﺎﺷﻴﺪ ﮐﻪ ﻭﯼ ﺑﻪ ﺷﻮﮎ، ﻏﺶ ﻳﺎ ﺣﻤﻠﻪ ﻗﻠﺒﯽ ﺩﭼﺎﺭ ﺷﺪﻩ ﺍﺳﺖ. ﺗﻴﺮﻩ ﺷﺪﻥ ﺭﻧﮓ ﭘﻮﺳﺖ ﻧﺸﺎﻧﻪ ﺧﻔﮕﯽ ﻳﺎ ﻣﺴﻤﻮﻣﻴﺖ ﺍﺳﺖ. ﺍﻳﻦ ﻧﺸﺎﻧﻪ ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﻤﮏ ﺑﺰﺭﮔﯽ ﺑﺮﺍﯼ ﺍﻣﺪﺍﺩﮔﺮ ﺩﺭ ﮐﻨﺘﺮﻝ ﺯﻣﺎﻥ ﺗﺸﺨﻴﺺ ﻣﺼﺪﻭﻣﻴﺖ ﺑﺎﺷﺪ.

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺍﺭﺯﻳﺎﺑﯽ ﺛﺎﻧﻮﻳﻪ ﺑﻴﻤﺎﺭ ﺭﺍ ﺑﺎ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﻣﺮﺍﺣﻞ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻳﺪ، ﻭ ﺍﮔﺮ ﺑﻴﻤﺎﺭ ﺩﺭ ﻭﺿﻌﻴﺖ ﻫﻮﺷﻴﺎﺭﯼ ﺑﻮﺩ، ﻭﻗﺖ ﺭﺍ ﻫﺪﺭ ﻧﺪﻫﻴﺪ ﻭ ﺗﺎ ﺭﺳﻴﺪﻥ ﺍﻭﺭﮊﺍﻧﺲ ﺑﺎ ﭘﺮﺳﻴﺪﻥ ﺳﻮﺍﻟﻬﺎﻳﯽ ﺍﺯ ﻭﯼ “ﺷﺮﺡ ﺣﺎﻝ” ﺑﮕﻴﺮﻳﺪ. ﺷﻤﺎ ﺑﺎﻳﺪ ﺍﺯ ﻭﯼ ﺑﭙﺮﺳﻴﺪ ﮐﻪ ﺁﻳﺎ ﭘﻴﺶ ﺍﺯ ﺍﻳﻦ ﺩﺍﺭﻭﻫﺎﯼ ﺧﺎﺻﯽ ﻣﺼﺮﻑ ﻣﯽ‌ﮐﺮﺩﻩ ﻳﺎ ﺧﻴﺮ، ﻳﺎ ﺑﻪ ﻣﻮﺍﺩ ﻳﺎ ﺩﺍﺭﻭﻫﺎﯼ ﺧﺎﺻﯽ ﺣﺴﺎﺳﻴﺖ ﻳﺎ ﺁﻟﺮﮊﯼ ﺩﺍﺭﺩ؟ ﻫﻤﭽﻨﻴﻦ ﺍﺯ ﺑﻴﻤﺎﺭﯼ‌ﻫﺎﯼ ﻗﺒﻠﯽ ﻣﺼﺪﻭﻡ ﺍﺯ ﺍﻭ ﭘﺮﺳﺶ ﮐﻨﻴﺪ، ﺁﻳﺎ ﺩﺍﺭﺍﯼ ﻧﺎﺭﺍﺣﺘﯽ ﻣﺰﻣﻦ ﻳﺎ ﺧﻄﺮﻧﺎﮐﯽ ﺍﺳﺖ؟ ﺁﻳﺎ ﺑﻪ ﺗﺎﺯﮔﯽ ﺩﭼﺎﺭ ﺑﻴﻤﺎﺭﯼ ﺧﺎﺻﯽ ﺷﺪﻩ ﺍﺳﺖ؟ ﺍﻟﺒﺘﻪ ﻫﻤﻴﺸﻪ ﺳﻌﯽ ﮐﻨﻴﺪ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺍﺯ ﻣﺼﺪﻭﻡ ﺍﻳﻦ ﺳﻮﺍﻝ ﻫﺎ ﺭﺍ ﺑﭙﺮﺳﻴﺪ ﮐﻪ ﺭﻭﺣﻴﻪ ﻭﯼ ﺻﺪﻣﻪ‌ﺍﯼ ﻧﺒﻴﻨﺪ ﻭ ﻫﻤﻮﺍﺭﻩ ﺑﻪ ﺍﻭ ﺭﻭﺣﻴﻪ ﺑﺪﻫﻴﺪ ﻭ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﺭﺍﺣﺘﯽ ﮐﺎﺭ ﺍﻣﺪﺍﺩﮔﺮﺍﻥ ﺍﻭﺭﮊﺍﻧﺲ ﺍﻳﻦ ﺳﻮﺍﻝ‌ﻫﺎ ﺭﺍ ﻣﯽ ﭘﺮﺳﻴﺪ ﺗﺎ ﺁﻥ‌ﻫﺎ ﺑﺘﻮﺍﻧﻨﺪ ﻫﺮ ﭼﻪ ﺳﺮﻳﻊ‌ﺗﺮ ﻧﺤﻮﻩ ﺍﻣﺪﺍﺩﺭﺳﺎﻧﯽ ﺭﺍ ﺗﺸﺨﻴﺺ ﺩﻫﻨﺪ.


ﺍﮔﺮ ﻣﺼﺪﻭﻡ ﺩﺭ ﻭﺿﻌﻴﺖ ﻫﻮﺷﻴﺎﺭﯼ ﺍﺳﺖ ﻭ ﺩﺭ ﻧﺎﺣﻴﻪ‌ﺍﯼ ﺍﺯ ﺑﺪﻥ ﺧﻮﺩ ﺩﺭﺩ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ، ﺍﺯ ﺍﻭ ﺑﺨﻮﺍﻫﻴﺪ ﮐﻪ ﺁﻥ ﻧﺎﺣﻴﻪ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﻧﺸﺎﻥ ﺩﻫﺪ ﻳﺎ ﺍﺳﻢ ﻧﺎﺣﻴﻪ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﺑﮕﻮﻳﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺩﺭﺩ ﻳﮑﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﻧﺸﺎﻧﻪ‌ﻫﺎ ﺑﺮﺍﯼ ﺑﻴﺎﻥ ﻣﺼﺪﻭﻣﻴﺖ ﺩﺭ ﻧﺎﺣﻴﻪ‌ﺍﯼ ﺍﺯ ﺑﺪﻥ ﺍﺳﺖ ﻭ ﺍﮔﺮ ﺩﺭ ﺣﺎﻟﺖ ﮐﻠﯽ، ﻣﺼﺪﻭﻣﻴﺖ‌ﻫﺎﯼ ﻣﺎ ﺑﺪﻭﻥ ﺩﺭﺩ ﺑﻮﺩ ﻣﺎ ﻫﻴﭻ ﮔﺎﻩ ﻣﺘﻮﺟﻪ ﺁﺳﻴﺐ ﺩﺭ ﺑﺪﻥ ﻧﻤﯽ‌ﺷﺪﻳﻢ. ﺩﺭ ﻫﻨﮕﺎﻡ ﭘﺮﺳﺶ ﺍﺯ ﻣﺼﺪﻭﻡ ﺑﭙﺮﺳﻴﺪ ﮐﻪ ﺩﺭﺩ ﻭﯼ ﺑﺮ ﺍﺛﺮ ﺳﺎﻧﺤﻪ ﻣﺼﺪﻭﻣﻴﺖ ﺍﺗﻔﺎﻕ ﺍﻓﺘﺎﺩﻩ، ﻳﺎ ﺩﺭﺩ ﺍﺯ ﺍﻭﻟﻴﻦ ﻟﺤﻈﻪ ﺷﺮﻭﻉ ﺷﺪﻩ ﻳﺎ ﭘﺲ ﺍﺯ ﭼﻨﺪ ﺩﻗﻴﻘﻪ ﺩﺭﺩ ﺭﺍ ﺍﺣﺴﺎﺱ ﮐﺮﺩﻩ ﺍﺳﺖ. ﺷﺪﺕ ﺩﺭﺩ ﻧﻴﺰ ﺍﺯ ﺩﻳﮕﺮ ﻧﮑﺎﺗﯽ ﺍﺳﺖ ﮐﻪ ﻧﺒﺎﻳﺪ ﻓﺮﺍﻣﻮﺵ ﮐﻨﻴﺪ ﻭ ﺁﻥ ﺭﺍ ﺑﺎﻳﺪ ﺍﺯ ﻣﺼﺪﻭﻡ ﺑﭙﺮﺳﻴﺪ.

ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ، ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﺎ ﻫﻢ ﻳﺎﺩ ﮔﺮﻓﺘﻴﻢ ﮐﻪ ﭘﺲ ﺍﺯ ﺍﻧﺠﺎﻡ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ ﺑﺎﻳﺪ ﺗﺎ ﺭﺳﻴﺪﻥ ﺍﻣﺪﺍﺩﮔﺮﺍﻥ ﺍﻭﺭﮊﺍﻧﺲ ﺑﻪ ﻣﺤﻞ ﺳﺎﻧﺤﻪ، ﺑﻪ ﻣﻌﺎﻳﻨﻪ ﺑﺎﻟﻴﻨﯽ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺑﭙﺮﺩﺍﺯﻳﻢ ﻭ ﺭﻭﺵ‌ﻫﺎﯼ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﻣﻌﺎﻳﻨﺎﺕ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﺮﻭﺭ ﮐﺮﺩﻳﻢ. ﺩﺭ ﺩﺭﺱ ﺑﻌﺪ ﺑﻪ ﺳﺮﺍﻍ ﻣﺼﺪﻭﻣﺎﻧﯽ ﻣﯽ‌ﺭﻭﻳﻢ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﺩﺍﺭﻧﺪ ﻭ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﻋﻤﻞ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﺮﻭﺭ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.
