ﺩﺭﺱ ﻫﻔﺘﻢ- ﺷﻮﮎ ﺧﻮﻧﯽ

ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﺍﻧﻮﺍﻉ ﺧﻮﻧﺮﻳﺰﯼ‌ﻫﺎﯼ ﺩﺍﺧﻠﯽ ﻭ ﺧﺎﺭﺟﯽ ﺭﺍ ﺑﺎ ﻫﻢ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ ﻭ ﻳﺎﺩ ﮔﺮﻓﺘﻴﻢ ﮐﻪ ﻫﻨﮕﺎﻡ ﺍﻧﺠﺎﻡ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﻪ ﻣﺼﺪﻭﻡ ﺩﺭ ﺣﺎﻝ ﺧﻮﻧﺮﻳﺰﯼ، ﺑﺎﻳﺪ ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﺍﺯ ﺗﻤﺎﺱ ﻣﺴﺘﻘﻴﻢ ﺩﺳﺖ ﺑﺪﻭﻥ ﭘﻮﺷﺶ ﺑﺎ ﺯﺧﻢ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﻢ ﻭ ﺩﺭ ﺻﻮﺭﺕ ﺿﺮﻭﺭﺕ ﺗﻤﺎﺱ، ﺩﺭ ﺍﻭﻟﻴﻦ ﻓﺮﺻﺖ ﺩﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺁﺏ ﻭ ﺻﺎﺑﻮﻥ ﺑﺸﻮﻳﻴﻢ. ﺳﭙﺲ ﻧﺤﻮﻩ ﮐﻤﮏ‌ﺭﺳﺎﻧﯽ ﺑﻪ ﻣﺼﺪﻭﻡ ﺩﺭ ﺣﺎﻟﺖ‌ﻫﺎﯼ ﺧﻮﻧﺮﻳﺰﯼ ﮐﻢ ﻭ ﺷﺪﻳﺪ ﺭﺍ ﻓﺮﺍ ﮔﺮﻓﺘﻴﻢ ﻭ ﮔﻔﺘﻴﻢ ﺩﺭ ﺣﺎﻟﺖ ﺧﻮﻧﺮﻳﺰﯼ ﺷﺪﻳﺪ ﺩﺭ ﺩﺳﺖ ﻳﺎ ﭘﺎ ﺑﺎﻳﺪ ﺑﻪ ﺩﻭ ﻧﻘﻄﻪ ﺍﺯ ﺑﺪﻥ ﺑﺮﺍﯼ ﮐﺎﻫﺶ ﺧﻮﻧﺮﻳﺰﯼ ﻓﺸﺎﺭ ﻭﺍﺭﺩ ﮐﻨﻴﻢ: ﻣﺤﻞ ﺧﻮﻧﺮﻳﺰﯼ ﻭ ﺷﺮﻳﺎﻥ‌ﻫﺎﯼ ﺍﺻﻠﯽ ﺩﺭ ﺩﺳﺖ ﻳﺎ ﭘﺎ. ﺩﺭ ﺍﻧﺘﻬﺎﯼ ﺩﺭﺱ ﻧﻴﺰ ﺩﺭﺑﺎﺭﻩ ﺣﺎﻟﺖ ﺷﻮﮎ ﻣﺮﺗﺒﻂ ﺑﺎ ﺩﺳﺘﮕﺎﻩ ﺧﻮﻥ‌ﺭﺳﺎﻧﯽ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ ﻭ ﮔﻔﺘﻴﻢ ﮐﻪ ﺑﻪ ﮐﻤﺒﻮﺩ ﺍﮐﺴﻴﮋﻥ ﺩﺭ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺑﻪ ﺩﻻﻳﻠﯽ ﻣﺮﺗﺒﻂ ﺑﺎ ﺩﺳﺘﮕﺎﻩ ﺧﻮﻥ، ﺷﻮﮎ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ. ﺍﮐﻨﻮﻥ ﺍﺩﺍﻣﻪ ﺑﺤﺚ ﺷﻮﮎ ﺭﺍ ﭘﯽ ﻣﯽ‌ﮔﻴﺮﻳﻢ ﻭ ﺩﺭﺑﺎﺭﻩ ﻧﺤﻮﻩ ﺑﺮﺧﻮﺭﺩ ﻫﻨﮕﺎﻡ ﺭﺥ ﺩﺍﺩﻥ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ.

ﺩﺭ ﻫﻨﮕﺎﻡ ﺷﻮﮎ ﭼﻪ ﺑﺎﻳﺪ ﮐﺮﺩ؟
ﻣﻌﻤﻮﻻ ﺗﻮﺻﻴﻪ ﻣﯽ‌ﺷﻮﺩ ﺣﺘﯽ ﺍﮔﺮ ﻣﺼﺪﻭﻡ ﺩﺭ ﻭﺿﻌﻴﺖ ﺷﻮﮎ ﻧﻴﺰ ﻗﺮﺍﺭ ﻧﺪﺍﺭﺩ ﺍﻣﺪﺍﺩﮔﺮ ﺩﺳﺘﻮﺭﺍﻟﻌﻤﻞ ﺍﻣﺪﺍﺩﯼ ﻫﻨﮕﺎﻡ ﺷﻮﮎ ﺭﺍ ﺑﺮﺍﯼ ﻣﺼﺪﻭﻡ ﺑﻪ ﮐﺎﺭ ﮔﻴﺮﺩ ﺗﺎ ﺭﺳﻴﺪﻥ ﺑﻪ ﻭﺿﻌﻴﺖ ﻧﺮﻣﺎﻝ ﺑﺮﺍﯼ ﻣﺼﺪﻭﻡ ﺁﺳﺎﻥ‌ﺗﺮ ﺷﻮﺩ.

ﺩﺭ ﻭﺿﻌﻴﺖ ﻣﻌﻤﻮﻟﯽ ﺩﺭ ﺷﻮﮎ، ﻣﺼﺪﻭﻡ ﺭﺍ ﺑﻪ ﭘﺸﺖ ﺧﻮﺍﺑﺎﻧﺪﻩ ﻭ ﭘﺎﻫﺎ ﺭﺍ ۲۰ ﺗﺎ ۳۰ ﺳﺎﻧﺘﯽ ﻣﺘﺮ ﺑﺎﻻﺗﺮ ﺍﺯ ﺑﺪﻥ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﺗﻨﻬﺎ ﺩﺭ ﺻﻮﺭﺗﯽ ﺍﻳﻦ ﻋﻤﻞ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﮐﻪ ﻣﺼﺪﻭﻡ ﺑﻪ ﺁﺳﻴﺐ ﻧﺨﺎﻋﯽ ﺩﭼﺎﺭ ﻧﺸﺪﻩ ﺑﺎﺷﺪ. ﺑﺎ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ ﺷﻤﺎ ﺑﻪ ﻣﺼﺪﻭﻡ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻴﺪ ﺗﺎ ﺧﻮﻥ ﭘﺎﻫﺎ ﺑﻪ ﺳﻤﺖ ﻗﻠﺐ ﺑﺮﮔﺸﺘﻪ ﻭ ﮐﻤﺒﻮﺩ ﺣﺠﻢ ﺧﻮﻥ ﺍﺣﺘﻤﺎﻟﯽ ﺩﺭ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺟﺒﺮﺍﻥ ﺷﻮﺩ.

ﺩﺭ ﻭﺿﻌﻴﺖ ﺁﺳﻴﺐ ﺑﻪ ﺳﺮ ﻭ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﻧﺨﺎﻉ ﻣﺼﺪﻭﻡ ﺁﺳﻴﺐ ﻧﺪﻳﺪﻩ، ﺗﻮﺻﻴﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺳﺮ ﻣﺼﺪﻭﻡ ﺑﻪ ﭘﺸﺖ ﺧﻮﺍﺑﺎﻧﻴﺪﻩ ﺭﺍ ﮐﻤﯽ ﺑﺎﻻﺗﺮ ﺑﻴﺎﻭﺭﻳﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﮐﺎﻓﯽ ﺍﺳﺖ ﮐﻪ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺭﺍ ﺯﻳﺮ ﺳﺮ ﺍﻭ ﻗﺮﺍﺭ ﺩﻫﻴﺪ.

ﺍﮔﺮ ﻣﺼﺪﻭﻣﯽ ﺩﭼﺎﺭ ﺣﻤﻠﻪ ﻗﻠﺒﯽ ﺷﺪﻩ، ﻳﺎ ﺻﺪﻣﻪ ﺑﻪ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺍﻭ ﻭﺍﺭﺩ ﺁﻣﺪﻩ ﻭ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺩﺭ ﺣﺎﻻﺗﯽ ﮐﻪ ﻣﺼﺪﻭﻡ ﺩﺭ ﻧﻔﺲ ﮐﺸﻴﺪﻥ ﺩﭼﺎﺭ ﻣﺸﮑﻞ ﺷﺪﻩ ﺑﺎﻳﺪ ﺍﻭ ﺭﺍ ﺩﺭ ﺣﺎﻟﺘﯽ ﮐﻪ ﺑﻪ ﺩﻳﻮﺍﺭ ﺗﮑﻴﻪ ﺩﺍﺭﺩ ﺑﺮ ﺭﻭﯼ ﺯﻣﻴﻦ ﺑﻨﺸﺎﻧﻴﺪ ﻭ ﭘﺎﻫﺎﻳﺶ ﺭﺍ ﺩﺭﺍﺯ ﮐﻨﻴﺪ. ﺳﭙﺲ ﭘﺎﻫﺎﯼ ﺍﻭ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﺷﮑﻢ ﺑﺎﻻ ﺑﺎﻭﺭﻳﺪ ﺗﺎ ﺩﺭ ﺣﺎﻟﺖ ﻧﻴﻤﻪ ﻧﺸﺴﺘﻪ ﻗﺮﺍﺭ ﮔﻴﺮﺩ. ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﺍﮔﺮ ﻣﺼﺪﻭﻣﯽ ﺑﻪ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﺑﻴﺮﻭﻧﯽ ﭘﺎﺳﺦ ﻧﻤﯽ‌ﺩﻫﺪ ﻭﻟﯽ ﺗﻨﻔﺲ ﻣﯽ‌ﮐﻨﺪ ﺍﻭ ﺭﺍ ﺩﺭ ﺣﺎﻟﺖ ﺭﻳﮑﺎﻭﺭﯼ ﺑﺨﻮﺍﺑﺎﻧﻴﺪ ﺗﺎ ﺍﻣﺪﺍﺩﮔﺮﺍﻥ ﺍﻭﺭﮊﺍﻧﺲ ﺍﺯ ﺭﺍﻩ ﺑﺮﺳﻨﺪ.

ﺩﺭ ﻣﺼﺪﻭﻡ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﺣﺘﻤﺎﻝ ﻣﯽ‌ﺩﻫﻴﺪ ﺑﻪ ﺁﺳﻴﺐ ﻧﺨﺎﻋﯽ ﺩﭼﺎﺭ ﺷﺪﻩ ﺑﺎﺷﻨﺪ ﻳﺎ ﺷﮑﺴﺘﮕﯽ ﺩﺭ ﻧﺎﺣﻴﻪ ﭘﺎ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ، ﺑﻬﺘﺮ ﺍﺳﺖ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺳﺎﺩﻩ ﺑﻪ ﭘﺸﺖ ﺑﺨﻮﺍﺑﺎﻧﻴﺪ ﻭ ﻣﻨﺘﻈﺮ ﺍﺯ ﺭﺍﻩ ﺭﺳﻴﺪﻥ ﺍﻭﺭﮊﺍﻧﺲ ﺑﻤﺎﻧﻴﺪ.

ﺍﻳﻦ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﺮﺍﯼ ﻣﺼﺪﻭﻣﺎﻥ ﺷﻮﮎ ﺩﻳﺪﻩ ﮐﺎﻓﯽ ﻫﺴﺘﻨﺪ ﻭ ﻣﺒﺤﺚ ﺷﻮﮎ ﺑﺎ ﻓﺮﺍﮔﻴﺮﯼ ﺍﻳﻦ ﻧﮑﺎﺕ ﺑﻪ ﭘﺎﻳﺎﻥ ﻣﯽ‌ﺭﺳﺪ. ﭘﺲ ﺍﺯ ﺍﻳﻦ ﻣﺒﺤﺚ ﻭ ﻗﺒﻞ ﺍﺯ ﺁﻥ ﮐﻪ ﺑﻪ ﺑﺨﺶ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺎﺯ ﻭ ﻣﺮﺍﻗﺒﺖ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺑﭙﺮﺩﺍﺯﻳﻢ، ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺍﺯ ﻳﮏ ﻧﻮﻉ ﺧﻮﻧﺮﻳﺰﯼ ﺩﻳﮕﺮ ﺻﺤﺒﺖ ﮐﻨﻴﻢ ﮐﻪ ﻣﻌﻤﻮﻻ ﺑﺮ ﺍﺛﺮ ﻓﻌﺎﻟﻴﺖ ﺩﺭ ﺯﻳﺮ ﺁﻓﺘﺎﺏ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ، ﻳﺎ ﺩﺭ ﺣﺎﻻﺕ ﺩﻳﮕﺮ ﺩﺭ ﺍﻃﺮﺍﻑ ﺧﻮﺩ ﺩﻳﺪﻩ‌ﺍﻳﺪ. ﺣﺪﺱ ﺷﻤﺎ ﺩﺭﺳﺖ ﺍﺳﺖ، ﻣﺒﺤﺚ ﻣﻮﺭﺩ ﻧﻈﺮ، ﺧﻮﻧﺮﻳﺰﯼ ﺑﻴﻨﯽ ﺍﺳﺖ ﮐﻪ ﻣﻌﻤﻮﻻ ﺭﺥ ﺩﺍﺩﻥ ﺁﻥ ﺩﺭ ﻫﻮﺍﯼ ﮔﺮﻡ ﻣﺘﺪﺍﻭﻝ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ.

ﺧﻮﻧﺮﻳﺰﯼ ﺑﻴﻨﯽ ﺩﻻﻳﻞ ﮔﻮﻧﺎﮔﻮﻧﯽ ﺩﺍﺭﺩ ﻭ ﻋﻼﻭﻩ ﺑﺮ ﻓﻌﺎﻟﻴﺖ ﺩﺭ ﺯﻳﺮ ﺁﻓﺘﺎﺏ، ﺑﻪ ﺩﻻﻳﻠﯽ ﭼﻮﻥ ﺑﺎﻻ ﺭﻓﺘﻦ ﻧﺎﮔﻬﺎﻧﯽ ﻓﺸﺎﺭ ﺧﻮﻥ، ﺍﻳﺠﺎﺩ ﺧﺮﺍﺵ ﻳﺎ ﺧﺸﮑﯽ ﺩﺭ ﻣﺨﺎﻁ ﺑﻴﻨﯽ ﻭ ﻳﺎ ﻭﺟﻮﺩ ﺑﺮﺧﯽ ﺑﻴﻤﺎﺭﯼ‌ﻫﺎﯼ ﺧﻮﻧﯽ ﻭ ﻋﻔﻮﻧﯽ ﺭﺥ ﻣﯽ‌ﺩﻫﺪ.

ﭼﮕﻮﻧﻪ ﺑﻪ ﻣﺼﺪﻭﻡ ﺩﺭ ﻫﻨﮕﺎﻡ ﺧﻮﻧﺮﻳﺰﯼ ﺑﻴﻨﯽ ﮐﻤﮏ ﮐﻨﻴﻢ؟
ﮐﻤﮏ ﮐﺮﺩﻥ ﺑﻪ ﻣﺼﺪﻭﻡ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ ﺍﺳﺖ ﻭ ﮐﺎﻓﯽ ﺍﺳﺖ ﺍﻭ ﺭﺍ ﺑﻪ ﺣﺎﻟﺖ ﻃﺒﻴﻌﯽ ﺑﺮ ﺭﻭﯼ ﺻﻨﺪﻟﯽ ﻳﺎ ﺯﻣﻴﻦ ﻧﺸﺎﻧﺪﻩ ﻭ ﺳﺮ ﺍﻭ ﺭﺍ ﮐﻤﯽ ﺑﻪ ﺟﻠﻮ ﺧﻢ ﮐﻨﻴﺪ. ﺳﭙﺲ ﺑﻴﻨﯽ ﺍﻭ ﺭﺍ ﺑﻪ ﻣﺪﺕ ﭘﻨﺞ ﺩﻗﻴﻘﻪ ﺑﻪ ﺣﺎﻟﺖ ﮔﺎﺯ ﺍﻧﺒﺮﯼ ﻓﺸﺎﺭ ﺩﻫﻴﺪ. ﺑﻴﻤﺎﺭ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺑﺎﻳﺪ ﺍﺯ ﺭﺍﻩ ﺩﻫﺎﻥ ﺗﻨﻔﺲ ﮐﻨﺪ. ﻫﻤﭽﻨﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ، ﺍﺯ ﮐﻤﭙﺮﺱ ﺳﺮﺩ ﻧﻴﺰ ﺑﺮﺍﯼ ﭘﻴﺸﺎﻧﯽ ﻭ ﺟﻠﻮﯼ ﺳﺮ ﺍﻭ ﻧﻴﺰ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺩﺭ ﺧﻮﻧﺮﻳﺰﯼ‌ﻫﺎﯼ ﻣﮑﺮﺭ ﻭ ﺷﺪﻳﺪ ﺑﻴﻨﯽ ﺑﻪ ﺧﺼﻮﺹ ﺩﺭ ﮐﻮﺩﮐﺎﻥ، ﻳﺎ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﭘﺲ ﺍﺯ ۱۰ ﺩﻗﻴﻘﻪ ﻓﺸﺎﺭ ﺩﺍﺩﻥ ﺑﻴﻨﯽ ﻫﻨﻮﺯ ﺧﻮﻧﺮﻳﺰﯼ ﺑﻨﺪ ﻧﻴﺎﻣﺪﻩ ﻭ ﺣﺘﯽ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻣﺼﺪﻭﻡ ﺍﺳﺘﻔﺮﺍﻍ ﺧﻮﻧﯽ ﺩﺍﺭﺩ ﺑﻬﺘﺮ ﺍﺳﺖ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺭﺍ ﺑﻪ ﻣﺮﺍﮐﺰ ﺩﺭﻣﺎﻧﯽ ﻣﻨﺘﻘﻞ ﮐﻨﻴﺪ ﺗﺎ ﺗﺤﺖ ﻣﺪﺍﻭﺍ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ.

ﻋﮑﺲ‌ﻫﺎﯼ ﻳﮏ ﻭ ﺩﻭ ﺣﺎﻻﺕ ﮐﻤﮏ ﺑﻪ ﻣﺼﺪﻭﻡ ﺩﺭ ﺣﺎﻟﺖ ﺧﻮﻧﺮﻳﺰﯼ ﺑﻴﻨﯽ ﻭ ﻣﺤﻞ ﻣﻮﺛﺮ ﻓﺸﺎﺭ ﻧﻴﺮﻭ ﺑﻪ ﺑﻴﻨﯽ ﺍﻭ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ.

ﺑﻌﺪ ﺍﺯ ﻣﺮﻭﺭ ﺍﻳﻦ ﻣﺒﺎﺣﺚ ﺩﺭ ﺑﺨﺶ ﺧﻮﻧﺮﻳﺰﯼ ﺍﮐﻨﻮﻥ ﺩﻳﮕﺮ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﻭﺍﺭﺩ ﻣﺒﺤﺚ ﻣﺮﺍﻗﺒﺖ ﺍﺯ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺎﺯ ﺷﻮﻳﻢ ﻭ ﺍﻳﻦ ﻣﺒﺤﺚ ﺭﺍ ﺑﺎ ﻫﻢ ﻓﺮﺍ ﺑﮕﻴﺮﻳﻢ.

ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺎﺯ ﺑﻪ ﭼﻪ ﺯﺧﻢ‌ﻫﺎﻳﯽ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ؟
 
ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺯﺧﻤﯽ ﺩﺭ ﺑﺪﻥ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﺩ، ﺩﺭ ﺣﻘﻴﻘﺖ ﻳﮏ ﭘﺎﺭﮔﯽ ﺩﺭ ﺳﻄﺢ ﺑﻴﺮﻭﻧﯽ ﭘﻮﺳﺖ ﺭﺥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺑﻨﺎﺑﺮ ﺷﺪﺕ ﺍﻳﻦ ﺯﺧﻢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺳﻄﻮﺡ ﺩﺍﺧﻠﯽ ﭘﻮﺳﺖ ﻭ ﺣﺘﯽ ﻋﻀﻠﻪ ﻭ ﻏﺪﺩ ﻟﻨﻔﺎﻭﯼ ﺭﺍ ﻧﻴﺰ ﺩﺭﮔﻴﺮ ﮐﺮﺩﻩ ﻭ ﺳﺒﺐ ﺍﻳﺠﺎﺩ ﻋﻔﻮﻧﺖ ﺩﺭ ﺗﻤﺎﻡ ﺑﺪﻥ ﻓﺮﺩ ﺷﻮﺩ. ﺯﺧﻢ‌ﻫﺎ ﺑﻪ ﺩﻭ ﮔﻮﻧﻪ ﺯﺧﻢ ﺑﺎﺯ ﻭ ﺯﺧﻢ ﺑﺴﺘﻪ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﻪ ﺯﺧﻢ‌ﻫﺎﻳﯽ ﮐﻪ ﺧﻮﻧﺮﻳﺰﯼ ﻗﺎﺑﻞ ﺭﻭﻳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺎﺯ ﻭ ﺑﻪ ﺯﺧﻢ‌ﻫﺎﻳﯽ ﮐﻪ ﺳﺒﺐ ﺧﻮﻧﺮﻳﺰﯼ ﺩﺭ ﺯﻳﺮ ﭘﻮﺳﺖ ﺷﻮﻧﺪ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺴﺘﻪ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ.

ﺯﺧﻢ ﻫﺎﯼ ﺑﺎﺯ ﺍﻧﻮﺍﻉ ﮔﻮﻧﺎﮔﻮﻧﯽ ﺩﺍﺭﻧﺪ ﮐﻪ ﺑﺴﺘﻪ ﺑﻪ ﻭﺳﻴﻠﻪ ﺑﻪ ﻭﺟﻮﺩ ﺁﻭﺭﻧﺪﻩ ﺁﻥ ﺑﻪ ﺯﺧﻢ‌ﻫﺎﯼ ﺧﺮﺍﺷﻴﺪﮔﯽ، ﺯﺧﻢ‌ﻫﺎﯼ ﺷﮑﺎﻓﺪﺍﺭ، ﭘﺎﺭﻩ‌ﺷﺪﮔﯽ ﻭ ﮐﻨﺪﻩ‌ﺷﺪﮔﯽ، ﺳﻮﺭﺍﺥ ﺷﺪﮔﯽ ﻭ ﺣﺘﯽ ﻗﻄﻊ ﻋﻀﻮ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﺷﻮﻧﺪ.

ﺯﺧﻢ‌ﻫﺎﯼ ﺧﺮﺍﺷﻴﺪﮔﯽ
ﺍﻳﻦ ﻧﻮﻉ ﺯﺧﻢ‌ﻫﺎ ﺳﺎﺩﻩ‌ﺗﺮﻳﻦ ﺯﺧﻢ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺑﺪﻥ ﻳﮏ ﻓﺮﺩ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺧﻮﻧﺮﻳﺰﯼ ﮐﻤﯽ ﺩﺍﺭﻧﺪ ﺍﻣﺎ ﺑﻪ ﻋﻠﺖ ﺁﺳﻴﺐ ﺩﻳﺪﻥ ﺳﻄﺢ ﺑﻴﺮﻭﻧﯽ ﭘﻮﺳﺖ ﺑﻪ ﻣﻘﺪﺍﺭ ﺍﺣﺘﻤﺎﻻ ﺯﻳﺎﺩ، ﺍﻋﺼﺎﺏ ﭘﻮﺳﺘﯽ ﺭﺍ ﺩﺭﮔﻴﺮ ﮐﺮﺩﻩ ﻭ ﺩﺭﺩ ﻓﺮﺍﻭﺍﻧﯽ ﺩﺍﺭﻧﺪ. ﺩﺭ ﺍﻳﻦ ﺯﺧﻢ‌ﻫﺎ ﺑﻬﺘﺮ ﺍﺳﺖ ﺳﻄﺢ ﺯﺧﻢ ﺭﺍ ﺑﺎ ﺁﺏ ﻭ ﺻﺎﺑﻮﻥ ﺑﺸﻮﻳﻴﺪ ﻭ ﺑﻪ ﺁﺭﺍﻣﯽ ﭘﺎﻧﺴﻤﺎﻥ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺍﻳﻦ ﺯﺧﻢ‌ﻫﺎ ﺑﺮ ﺍﺛﺮ ﺯﻣﻴﻦ ﺧﻮﺭﺩﻥ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻩ ﺑﺎﺷﻨﺪ، ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﻓﺮﺍﻭﺍﻥ ﺳﻨﮓ ﺭﻳﺰﻩ‌ﻫﺎﯼ ﮐﻮﭼﮑﯽ ﻧﻴﺰ ﻭﺍﺭﺩ ﺯﺧﻢ ﺷﺪﻩ ﺍﺳﺖ ﻭ ﺑﻬﺘﺮ ﺍﺳﺖ ﺳﻨﮓ ﺭﻳﺰﻩ‌ﻫﺎﯼ ﺩﺭﺷﺖ‌ﺗﺮ ﺭﺍ ﺩﺭ ﺟﻬﺖ ﻣﺨﺎﻟﻒ ﺯﺧﻢ ﺑﻮﺟﻮﺩ ﺁﻣﺪﻩ ﺧﺎﺭﺝ ﮐﻨﻴﺪ. ﻓﺸﺎﺭ ﺩﺍﺩﻥ ﻳﺎ ﺩﺳﺘﮑﺎﺭﯼ ﺯﺧﻢ ﺩﺭ ﺟﻬﺖ ﻣﻮﺍﻓﻖ ﺑﺮﺍﯼ ﺧﺎﺭﺝ ﮐﺮﺩﻥ ﺍﺟﺴﺎﻡ ﺧﺎﺭﺟﯽ ﺳﺒﺐ ﻣﯽ‌ﺷﻮﺩ ﺍﻳﻦ ﺍﺟﺴﺎﻡ ﺩﺭ ﺯﺧﻢ ﻓﺮﻭ ﺭﻓﺘﻪ ﻭ ﺑﻴﺮﻭﻥ ﺍﻭﺭﺩﻥ ﺁﻥ‌ﻫﺎ ﻣﺸﮑﻞ ﺷﻮﺩ.

ﺯﺧﻢ‌ﻫﺎﯼ ﺷﮑﺎﻓﺪﺍﺭ
ﺍﻳﻦ ﻧﻮﻉ ﺯﺧﻢ‌ﻫﺎ ﺑﺮ ﺍﺛﺮ ﺑﺮﻳﺪﮔﯽ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻩ‌ﺍﻧﺪ ﻭ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺮﻳﺪﮔﯽ ﻧﻴﺰ ﺭﺍ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﺍﻳﻦ ﺩﺳﺘﻪ ﺗﻘﺴﻴﻢ‌ﺑﻨﺪﯼ ﮐﺮﺩ. ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺯﺧﻢ‌ﻫﺎ ﺩﻳﻮﺍﺭﻩ ﺯﺧﻢ ﻫﺎ ﻣﺴﺘﻘﻴﻢ ﺍﺳﺖ ﻭ ﺑﺴﺘﻪ ﺑﻪ ﺷﺪﺕ ﺯﺧﻢ ﺍﻳﺠﺎﺩ ﺷﺪﻩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺣﺘﯽ ﺑﻪ ﻋﺮﻭﻕ ﺧﻮﻧﯽ ﻧﻴﺰ ﺻﺪﻣﻪ ﺑﺮﺳﺎﻧﺪ. ﺍﻳﻦ ﻧﻮﻉ ﺯﺧﻢ‌ﻫﺎ ﺭﺍ ﺑﺎ ﻓﺸﺎﺭ ﺩﺍﺩﻥ ﻭ ﭘﺎﻧﺴﻤﺎﻥ ﺑﻪ ﺧﻮﺑﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﮐﻨﺘﺮﻝ ﮐﺮﺩ.

ﺯﺧﻢ‌ﻫﺎﯼ ﭘﺎﺭﻩﺷﺪﮔﯽ ﻭ ﮐﻨﺪﻩ‌ﺷﺪﮔﯽ

ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺯﺧﻢ‌ﻫﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺨﺸﯽ ﺍﺯ ﺳﻄﺢ ﭘﻮﺳﺖ ﭘﺎﺭﻩ ﺷﻮﺩ ﻭ ﺍﺯ ﻣﺤﻞ ﺯﺧﻢ ﺁﻭﺭﻳﺰﺍﻥ ﺷﻮﺩ ﮐﻪ ﺑﺎ ﺩﺭﺩ ﻓﺮﺍﻭﺍﻥ ﻫﻤﺮﺍﻩ ﺍﺳﺖ. ﺍﮔﺮ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﭘﻮﺳﺖ ﺭﺍ ﺑﻪ ﺟﺎﯼ ﺧﻮﺩ ﺑﺮ ﮔﺮﺩﺍﻧﻴﺪ، ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﺗﺎ ﺯﺧﻢ ﺯﻭﺩﺗﺮ ﺟﻮﺵ ﺑﺨﻮﺭﺩ. ﺩﺭ ﮐﻨﺪﻩ‌ﺷﺪﮔﯽ ﻧﻴﺰ، ﻳﮑﯽ ﺍﺯ ﺍﻋﻀﺎﯼ ﺑﺪﻥ ﻣﺎﻧﻨﺪ ﺍﻧﮕﺸﺘﺎﻥ ﻳﺎ ﺩﺳﺖ ﻳﺎ ﭘﺎ ﺍﺯ ﺑﺪﻥ ﺟﺪﺍ ﻣﯽ‌ﺷﻮﺩ. ﺍﺣﺘﻤﺎﻝ ﻋﻔﻮﻧﯽ ﺷﺪﻥ ﺍﻳﻦ ﺯﺧﻢ‌ﻫﺎ ﻭ ﻣﺮﮒ ﻣﺼﺪﻭﻡ ﺑﺴﻴﺎﺭ ﺑﺎﻻﺳﺖ ﻭ ﻣﺼﺪﻡ ﺑﺎﻳﺪ ﺑﻪ ﺳﺮﻋﺖ ﺗﺤﺖ ﺩﺭﻣﺎﻥ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ.

ﺯﺧﻢ‌ﻫﺎﯼ ﺳﻮﺭﺍﺥ‌ﺷﺪﮔﯽ
ﺍﻳﻦ ﻧﻮﻉ ﺯﺧﻢ‌ﻫﺎ ﺑﺮ ﺍﺛﺮ ﻭﺭﻭﺩ ﺍﺟﺴﺎﻡ ﺧﺎﺭﺟﯽ ﻧﻮﮎ ﺗﻴﺰ ﻣﺎﻧﻨﺪ ﭼﺎﻗﻮ، ﻣﻴﺦ ﻳﺎ ﺣﺘﯽ ﮔﻠﻮﻟﻪ ﺑﻪ ﺑﺪﻥ ﺑﻪ‌ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻳﻨﺪ ﻭ ﻣﻌﻤﻮﻻ ﺣﺠﻢ ﺍﻳﻦ ﺯﺧﻢ‌ﻫﺎ ﮐﻮﭼﮏ ﻫﺴﺘﻨﺪ. ﺍﺣﺘﻤﺎﻝ ﻋﻔﻮﻧﯽ ﺷﺪﻥ ﺍﻳﻦ ﺯﺧﻢ‌ﻫﺎ ﻧﻴﺰ ﺑﺴﻴﺎﺭ ﺑﺎﻻ ﺍﺳﺖ ﻭ ﻣﺼﺪﻭﻡ ﺑﺎﻳﺪ ﺑﻪ ﺳﺮﻋﺖ ﺑﻪ ﻣﺮﺍﮐﺰ ﺩﺭﻣﺎﻧﯽ ﻣﻨﺘﻘﻞ ﺷﻮﺩ ﺗﺎ ﺗﺤﺖ ﺩﺭﻣﺎﻥ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ.

ﺩﺭ ﻫﻨﮕﺎﻡ ﻣﻮﺍﺟﻪ ﺷﺪﻥ ﺑﺎ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺎﺯ، ﺭﻭﺣﻴﻪ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻧﺪﻫﻴﺪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺑﺮﺧﯽ ﺍﺯ ﻣﻮﺍﻗﻊ، ﻣﺼﺪﻭﻡ ﺑﻪ ﺻﻮﺭﺕ ﺑﺪﯼ ﺩﭼﺎﺭ ﻣﺼﺪﻭﻣﻴﺖ ﺷﺪﻩ ﻭ ﺯﺧﻢ ﺍﻳﺠﺎﺩ ﺷﺪﻩ، ﺻﺤﻨﻪ‌ﺍﯼ ﻭﺣﺸﺘﻨﺎﮎ ﺭﺍ ﺑﺮﺍﯼ ﺷﻤﺎ ﺭﻗﻢ ﺑﺰﻧﺪ، ﺩﺭ ﺍﻳﻦ ﻫﻨﮕﺎﻡ ﺑﻪ ﺧﻮﺩ ﻣﺴﻠﻂ ﺑﺎﺷﻴﺪ ﻭ ﺗﻨﻬﺎ ﺑﻪ ﮐﻤﮏ ﮐﺮﺩﻥ ﺑﻪ ﻣﺼﺪﻭﻡ ﻓﮑﺮ ﮐﻨﻴﺪ. ﺍﺑﺘﺪﺍ ﺧﻮﻧﺮﻳﺰﯼ ﺭﺍ ﺑﻪ ﺷﻴﻮﻩ‌ﻫﺎﻳﯽ ﮐﻪ ﭘﻴﺶ ﺍﺯ ﺍﻳﻦ ﺁﻣﻮﺧﺘﻴﺪ، ﻣﺘﻮﻗﻒ ﮐﻨﻴﺪ ﻭ ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﺳﻌﯽ ﮐﻨﻴﺪ ﻫﻨﮕﺎﻡ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﺭﺍﻫﮑﺎﺭﻫﺎ، ﺍﺯ ﺗﻤﺎﺱ ﻣﺴﺘﻘﻴﻢ ﺩﺳﺘﺎﻥ ﺑﺪﻭﻥ ﭘﻮﺷﺶ ﺧﻮﺩ ﺑﺎ ﺯﺧﻢ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺰﺭﮒ ﻭ ﮐﺜﻴﻒ ﺭﺍ ﺑﻪ ﻫﻴﭻ ﻋﻨﻮﺍﻥ ﺑﻪ ﺑﻬﺎﻧﻪ ﭘﺎﮎ ﮐﺮﺩﻥ، ﺩﺳﺘﮑﺎﺭﯼ ﻧﮑﻨﻴﺪ ﭼﺮﺍ ﮐﻪ ﺍﺯ ﺗﺨﺼﺺ ﺷﻤﺎ ﺧﺎﺭﺝ ﺍﺳﺖ ﻭ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺩﺭ ﺣﻴﻄﻪ ﮐﺎﺭﯼ ﺍﻓﺮﺍﺩ ﻣﺘﺨﺼﺺ ﻗﺮﺍﺭ ﺩﺍﺭﺩ.

ﻓﺮﺁﻳﻨﺪ ﺑﻬﺒﻮﺩ ﺯﺧﻢ، ﻳﮏ ﻓﺮﺍﻳﻨﺪ ﺧﻮﺩ ﺑﻪ ﺧﻮﺩﯼ ﺍﺳﺖ ﻭ ﺗﻨﻬﺎ ﮐﺎﻓﯽ ﺍﺳﺖ ﻣﺤﻞ ﺍﻳﺠﺎﺩ ﺯﺧﻢ ﮐﻤﯽ ﺗﻤﻴﺰ ﺑﺎﺷﺪ ﺗﺎ ﺩﭼﺎﺭ ﻋﻔﻮﻧﺖ ﻧﺸﻮﺩ ﻭ ﺧﻮﻥ‌ﺭﺳﺎﻧﯽ ﺑﻪ ﺍﻳﻦ ﻣﺤﻞ ﺍﺩﺍﻣﻪ ﻳﺎﺑﺪ ﺗﺎ ﻣﻮﺍﺩ ﻻﺯﻡ ﻭ ﺍﮐﺴﻴﮋﻥ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﭘﻮﺳﺖ ﻭ ﻋﻀﻠﻪ ﻗﺮﺍﺭ ﮔﻴﺮﺩ. ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺯﺧﻢ‌ﻫﺎﻳﯽ ﮐﻪ ﭘﺲ ﺍﺯ ﮔﺬﺷﺖ ۴۸ ﺳﺎﻋﺖ ﺑﺴﺘﻪ ﻧﻤﯽ‌ﺷﻮﻧﺪ، ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﻓﺮﺍﻭﺍﻥ ﺩﭼﺎﺭ ﻋﻔﻮﻧﺖ ﺷﺪﻩ‌ﺍﻧﺪ ﻭ ﺑﺎﻳﺪ ﺗﺤﺖ ﺩﺭﻣﺎﻥ‌ﻫﺎﯼ ﺗﺨﺼﺼﯽ ﻗﺮﺍﺭ ﺑﮕﻴﺮﻧﺪ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﺯ ﺩﺭﻣﺎﻥ ﺧﻮﺩﺳﺮﺍﻧﻪ ﺯﺧﻢ ﭘﺮﻫﻴﺰ ﺷﻮﺩ ﻭ ﻣﺼﺪﻭﻡ ﺑﻪ ﻣﺮﺍﮐﺰ ﺩﺭﻣﺎﻧﯽ ﻣﻨﺘﻘﻞ ﺷﻮﺩ.

ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ، ﺩﺭ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺍﻫﮑﺎﺭﻫﺎﯼ ﮐﻤﮏ ﺑﻪ ﻣﺼﺪﻭﻣﻴﻦ ﺷﻮﮎ‌ﻫﺎﯼ ﺧﻮﻧﯽ ﺭﺍ ﻓﺮﺍ ﮔﺮﻓﺘﻴﻢ ﻭ ﺭﺍﻫﮑﺎﺭ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺧﻮﻧﺮﻳﺰﯼ ﺑﻴﻨﯽ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﺮﻭﺭ ﮐﺮﺩﻳﻢ. ﺳﭙﺲ ﻭﺍﺭﺩ ﻣﺒﺤﺚ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺎﺯ ﺷﺪﻳﻢ ﻭ ﺍﻧﻮﺍﻉ ﺍﻳﻦ ﻧﻮﻉ ﺯﺧﻢ‌ﻫﺎ ﺭﺍ ﻧﻴﺰ ﺑﺎ ﻫﻢ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ. ﺩﺭ ﺟﻠﺴﻪ ﺁﻳﻨﺪﻩ، ﺑﻪ ﺍﺩﺍﻣﻪ ﻣﺒﺤﺚ ﺯﺧﻢ‌ﻫﺎﯼ ﺑﺎﺯ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ ﻭ ﻧﺤﻮﻩ ﻣﺮﺍﻗﺒﺖ ﺍﺯ ﺯﺧﻢ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺣﺎﻟﺖ‌ﻫﺎﯼ ﺧﺎﺹ ﺑﺮﺭﺳﯽ ﻣﯽ‌ﮐﻨﻴﻢ.
