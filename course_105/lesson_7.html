ﺩﺭﺱ ﻧﻬﻢ- ﺳﻮﺧﺘﮕﯽ ﻭ ﺣﻤﻠﻪ ﻗﻠﺒﯽ

ﺩﺭ ﺟﻠﺴﻪ ﻗﺒﻞ ﺯﺧﻢ ﻫﺎ ﻭ ﻣﺮﺍﻗﺒﺖ ﺍﺯ ﺑﺮﺧﯽ ﺯﺧﻢ ﻫﺎﯼ ﺑﺎﺯ ﻫﻤﺎﻧﻨﺪ ﻗﻄﻊ ﻋﻀﻮ ﺭﺍ ﺑﺎ ﻫﻢ ﻓﺮﺍ ﮔﺮﻓﺘﻴﻢ ﻭ ﻣﺮﺍﺣﻞ ﺍﻧﺠﺎﻡ ﻳﮏ ﭘﺎﻧﺴﻤﺎﻥ ﺳﺎﺩﻩ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﺮﻭﺭ ﮐﺮﺩﻳﻢ. ﺩﺭ ﻣﺒﺤﺚ ﭘﺎﻧﺴﻤﺎﻥ ﻫﺎ ﮔﻔﺘﻴﻢ ﮐﻪ ﺯﺧﻢ ﺭﺍ ﺍﺑﺘﺪﺍ ﺑﺎﻳﺪ ﭘﺎﻧﺴﻤﺎﻥ ﮐﺮﺩ ﻭ ﺑﻌﺪ ﺁﻥ ﺭﺍ ﺑﺎﻧﺪ ﭘﻴﺠﯽ ﮐﺮﺩ ﻭ ﻫﻨﮕﺎﻡ ﭘﺎﻧﺴﻤﺎﻥ ﻧﻴﺰ ﺍﺯﺣﺮﮐﺖ ﺩﺍﺩﻥ ﭘﺎﻧﺴﻤﺎﻥ ﺭﻭﯼ ﺯﺧﻢ ﺑﺎﻳﺪ ﺧﻮﺩﺩﺍﺭﯼ ﮐﺮﺩ ﭼﺮﺍ ﮐﻪ ﺑﻪ ﺯﺧﻢ ﺁﺳﻴﺐ ﻣﯽ‌ﺭﺳﺎﻧﺪ. ﺩﺭ ﻫﻨﮕﺎﻡ ﺑﺎﻧﺪ ﭘﻴﺠﯽ ﻧﻴﺰ ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﮐﻨﻴﻢ ﮐﻪ ﺑﺎﻧﺪ ﺭﺍ ﻧﻪ ﺷﻞ ﺑﺒﻨﺪﻳﻢ ﮐﻪ ﭘﺎﻧﺴﻤﺎﻥ ﺣﺮﮐﺖ ﮐﻨﺪ ﻭ ﻧﻪ ﻣﺤﮑﻢ ﮐﻪ ﺳﺒﺐ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﻋﻀﻮ ﺷﻮﺩ. ﺗﺴﺖ ﻓﺸﺎﺭ ﺍﻧﮕﺸﺖ ﺩﺭ ﻋﻀﻮ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﻋﻨﻮﺍﻥ ﺗﺴﺖ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﻋﻀﻮ ﭘﺎﻧﺴﻤﺎﻥ ﺷﺪﻩ ﻣﻌﺮﻓﯽ ﮐﺮﺩﻳﻢ ﻭ ﮔﻔﺘﻴﻢ ﮐﻪ ﺍﮔﺮ ﺳﺮ ﻳﮑﯽ ﺍﺯ ﺍﻧﮕﺸﺘﺎﻥ ﻋﻀﻮ ﺑﺎﻧﺪﭘﻴﺠﯽ ﺷﺪﻩ ﺭﺍ ﻓﺸﺎﺭ ﺩﻫﻴﻢ ﺩﺭ ﮐﻤﺘﺮ ﺍﺯ ﺳﻪ ﺛﺎﻧﻴﻪ ﺑﺎﻳﺪ ﺑﻪ ﺣﺎﻟﺖ ﻭ ﺭﻧﮓ ﻃﺒﻴﻌﯽ ﺑﺎﺯ ﮔﺮﺩﺩ ﻭ ﺍﻓﺰﺍﻳﺶ ﺍﻳﻦ ﻣﺪﺕ ﻧﺸﺎﻥ ﺍﺯ ﺑﺎﻧﺪﭘﻴﺠﯽ ﻣﺤﮑﻢ ﻋﻀﻮ ﺍﺳﺖ.
 
ﺩﺭ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺑﻪ ﻣﺒﺤﺚ ﺳﻮﺧﺘﮕﯽ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ ﻭ ﺑﺎ ﻧﺤﻮﻩ ﮐﻤﮏ ﺑﻪ ﻣﺼﺪﻭﻣﯽ ﺭﺍ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﻳﻢ ﮐﻪ ﺩﭼﺎﺭ ﺳﻮﺧﺘﮕﯽ ﺷﺪﻩ ﺍﺳﺖ. ﺳﻮﺧﺘﮕﯽ‌ﻫﺎ ﻣﻌﻤﻮﻻ ﺑﻪ ﺳﻪ ﻧﻮﻉ ﮔﺮﻣﺎﻳﯽ، ﺷﻴﻤﻴﺎﻳﯽ ﻭ ﺍﻟﮑﺘﺮﻳﮑﯽ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﺷﻮﻧﺪ.
 
ﺳﻮﺧﺘﮕﯽ ﮔﺮﻣﺎﻳﯽ ﻳﺎ ﺣﺮﺍﺭﺗﯽ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺍﺯ ﺍﺳﻤﺶ ﭘﻴﺪﺍﺳﺖ ﺑﻪ ﻋﻠﺖ ﺗﻤﺎﺱ ﺑﺪﻥ ﺑﺎ ﺁﺗﺶ، ﻳﮏ ﺟﺴﻢ ﺩﺍﻍ ﻳﺎ ﺑﺨﺎﺭ ﻣﺎﻳﻌﺎﺕ ﺩﺍﻍ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﺩ. ﺳﻮﺧﺘﮕﯽ ﺷﻴﻤﻴﺎﻳﯽ ﻧﻴﺰ ﺑﻪ ﺩﻟﻴﻞ ﺗﻤﺎﺱ ﺑﺪﻥ ﺑﺎ ﻣﻮﺍﺩ ﺍﺳﻴﺪﯼ، ﻗﻠﻴﺎﻳﯽ ﻳﺎ ﻣﻮﺍﺩ ﺁﻟﯽ ﺍﺗﻔﺎﻕ ﻣﯽ‌ﺍﻓﺘﺪ ﻭ ﻣﻴﺰﺍﻥ ﺗﺨﺮﻳﺐ ﺑﺎﻓﺘﯽ ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﮕﯽ ﺷﺒﻴﻪ ﺳﻮﺧﺘﮕﯽ ﮔﺮﻣﺎﯼ ﺍﺳﺖ. ﺳﻮﺧﺘﮕﯽ ﺍﻟﮑﺘﺮﻳﮑﯽ ﻫﻢ ﺑﺎ ﺗﻤﺎﺱ ﺑﺪﻥ ﺑﺎ ﺟﺮﻳﺎﻥ ﺍﻟﮑﺘﺮﻳﮑﯽ ﺭﺥ ﻣﯽ‌ﺩﻫﺪ.
 
ﺳﻮﺧﺘﮕﯽ‌ﻫﺎ ﻫﻤﭽﻨﻴﻦ ﺑﻪ ﺳﻪ ﺩﺭﺟﻪ ﺍﺯ ﻳﮏ ﺗﺎ ﺳﻪ ﻃﺒﻘﻪ ﺑﻨﺪﯼ ﻣﯽ‌ﺷﻮﻧﺪ. ﺩﺭ ﺳﻮﺧﺘﮕﯽ ﺩﺭﺟﻪ ﻳﮏ ﺗﻨﻬﺎ ﭘﻮﺳﺖ ﺷﻤﺎ ﻗﺮﻣﺰ ﻣﯽ‌ﺷﻮﺩ ﻭﻟﯽ ﺗﺎﻭﻝ ﻧﻤﯽ‌ﺯﻧﺪ ﻭ ﺷﻤﺎ ﻫﻢ ﺩﺭﺩﯼ ﺍﺣﺴﺎﺱ ﻧﻤﯽ‌ﮐﻨﻴﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﮕﯽ ﺧﻔﻴﻒ‌ﺗﺮﻳﻦ ﺳﻮﺧﺘﮕﯽ ﻣﻤﮑﻦ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ ﻭ ﺗﻨﻬﺎ ﻻﻳﻪ ﺧﺎﺭﺟﯽ ﭘﻮﺳﺖ ﺭﺍ ﺩﺭﮔﻴﺮ ﻣﯽ‌ﮐﻨﺪ. ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﮕﯽ ﺩﺭ ﮐﻤﺘﺮ ﺍﺯ ﻳﮏ ﻫﻔﺘﻪ ﺧﻮﺩ ﺑﻪ ﺧﻮﺩ ﺍﺯ ﺑﻴﻦ ﻣﯽ‌ﺭﻭﺩ.
 
ﺩﺭ ﺳﻮﺧﺘﮕﯽ ﺩﺭﺟﻪ ﺩﻭ، ﻗﺴﻤﺘﯽ ﺍﺯ ﺳﻄﺢ ﺩﺍﺧﻠﯽ ﭘﻮﺳﺖ ﻧﻴﺰ ﺩﭼﺎﺭ ﺳﻮﺧﺘﮕﯽ ﻣﯽ‌ﺷﻮﺩ ﺍﻣﺎ ﺑﻪ ﺑﺎﻓﺖ‌ﻫﺎﯼ ﺯﻳﺮﻳﻦ ﭘﻮﺳﺖ ﺁﺳﻴﺒﯽ ﻭﺍﺭﺩ ﻧﻤﯽ‌ﺷﻮﺩ. ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﮕﯽ، ﺗﺎﻭﻝ ﻭ ﺗﺮﺷﺢ ﻣﺎﻳﻌﺎﺕ ﺑﻪ ﻫﻤﺮﺍﻩ ﺩﺭﺩ ﺩﻳﺪﻩ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻣﺪﺕ ﺯﻣﺎﻥ ﺑﻬﺒﻮﺩ ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﯽ ﺣﺪﻭﺩ ﺳﻪ ﻫﻔﺘﻪ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﮕﯽ ﺑﺎﻳﺪ ﺑﻪ ﭘﺰﺷﮏ ﻣﺮﺍﺟﻌﻪ ﺷﻮﺩ.
 
ﺩﺭ ﺳﻮﺧﺘﮕﯽ ﺩﺭﺟﻪ ﺳﻪ، ﮐﻪ ﺷﺪﻳﺪﺗﺮﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﮕﯽ ﺍﺳﺖ، ﻋﻼﻭﻩ ﺑﺮ ﺗﻤﺎﻡ ﺿﺨﺎﻣﺖ ﭘﻮﺳﺖ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎﻓﺖ‌ﻫﺎﯼ ﺯﻳﺮﻳﻦ ﻧﻴﺰ ﺩﭼﺎﺭ ﺳﻮﺧﺘﮕﯽ ﺷﻮﻧﺪ ﻭ ﺑﻪ ﻋﻠﺖ ﺍﺯ ﺑﻴﻦ ﺭﻓﺘﻦ ﻋﺼﺐ ﺯﻳﺮ ﭘﻮﺳﺘﯽ ﺩﺭ ﺍﻳﻦ ﺳﻮﺧﺘﮕﯽ ﺑﻴﻤﺎﺭ ﻣﻌﻤﻮﻻ ﺩﺭﺩﯼ ﺍﺣﺴﺎﺱ ﻧﻤﯽ‌ﮐﻨﺪ. ﺍﮔﺮ ﺩﺭ ﺍﻳﻦ ﺳﻮﺧﺘﮕﯽ ﺑﻴﻤﺎﺭ ﺩﺭﺩ ﺍﺣﺴﺎﺱ ﮐﺮﺩ، ﺑﻪ ﺩﻟﻴﻞ ﻭﺟﻮﺩ ﻧﻮﺍﺣﯽ ﺳﻮﺧﺘﮕﯽ ﺑﺎ ﺩﺭﺟﻪ ﺩﻭ ﺍﺳﺖ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺍﻃﺮﺍﻑ ﻧﻮﺍﺣﯽ ﺳﻮﺧﺘﮕﯽ ﺩﺭﺟﻪ ﺳﻪ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﮕﯽ ﻣﺼﺪﻭﻡ ﺑﺎﻳﺪ ﺑﺪﻭﻥ ﺍﺗﻼﻑ ﻭﻗﺖ ﺑﻪ ﻣﺮﺍﮐﺰ ﺩﺭﻣﺎﻧﯽ ﻣﻨﺘﻘﻞ ﺷﻮﺩ. ﻋﻔﻮﻧﺖ ﻣﻬﻤﺘﺮﻳﻦ ﻣﺸﮑﻞ ﺍﻳﻦ ﻧﻮﻉ ﺳﻮﺧﺘﮕﯽ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ ﻭ ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﺭﺩ ﺩﺭﻣﺎﻧﯽ ﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﻋﻤﻞ ﭘﻴﻮﻧﺪ ﭘﻮﺳﺖ ﻧﻴﺰ ﺑﺮ ﺭﻭﯼ ﻣﺼﺪﻭﻡ ﺍﻧﺠﺎﻡ ﺷﻮﺩ.
 
ﺳﻮﺧﺘﮕﯽ‌ﻫﺎﯼ ﺩﺭﺟﻪ ﻳﮏ ﻭ ﺩﻭ ﺑﻪ ﻋﻨﻮﺍﻥ ﺳﻮﺧﺘﮕﯽ‌ﻫﺎﯼ ﺑﺎ ﺿﺨﺎﻣﺖ ﻧﺴﺒﯽ (ﭘﻮﺳﺖ) ﻭ ﺳﻮﺧﺘﮕﯽ ﺩﺭﺟﻪ ﺳﻮﻡ ﺑﺎ ﻧﺎﻡ ﺳﻮﺧﺘﮕﯽ ﺗﻤﺎﻡ ﺿﺨﺎﻣﺖ (ﭘﻮﺳﺖ) ﻧﻴﺰ ﺷﻨﺎﺧﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ.
 
ﻭﺳﻌﺖ ﺳﻮﺧﺘﮕﯽ ﺭﺍ ﭼﮕﻮﻧﻪ ﺗﺨﻤﻴﻦ ﺑﺰﻧﻴﻢ؟
ﻣﺤﺎﺳﺒﻪ ﻭﺳﻌﺖ ﺳﻮﺧﺘﮕﯽ ﺑﺎ ﻗﺎﻧﻮﻧﯽ ﺑﻪ ﻧﺎﻡ ﻗﺎﻧﻮﻥ ۹ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺍﻳﻦ ﻗﺎﻧﻮﻥ ﮐﻪ ﺑﻪ ﺧﻮﺑﯽ ﺩﺭ ﻋﮑﺲ ﻳﮏ ﺑﻪ ﻧﻤﺎﻳﺶ ﺩﺭﺁﻣﺪﻩ، ﺗﻤﺎﻡ ﺳﺮ ﻳﺎ ﻳﮏ ﺩﺳﺖ ﻳﺎ ﻳﮏ ﭘﺎ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ۹ ﺩﺭﺻﺪ ﻭﺳﻌﺖ ﺳﻮﺧﺘﮕﯽ ﻣﺤﺎﺳﺒﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺗﻤﺎﻡ ﺟﻠﻮﯼ ﺑﺪﻥ ﻭ ﺗﻤﺎﻡ ﭘﺸﺖ ﺑﺪﻥ (ﻫﻤﺎﻧﻨﺪ ﺷﮑﻞ) ۱۸ ﺩﺭﺻﺪ ﻭﺳﻌﻴﺖ ﺳﻮﺧﺘﮕﯽ ﺑﻪ ﺣﺴﺎﺏ ﻣﯽ‌ﺁﻳﻨﺪ. ﺍﻳﻦ ﻗﺎﻧﻮﻥ ﺩﺭ ﮐﻮﺩﮐﺎﻥ ﻭ ﺷﻴﺮﺧﻮﺍﺭﺍﻥ ﺑﻪ ﺍﻳﻨﮕﻮﻧﻪ ﺍﺳﺖ ﮐﻪ ﺗﻤﺎﻡ ﺳﺮ ﻳﺎ ﻫﺮ ﭘﺎ ﻳﺎ ﺩﺳﺖ ۱۸ ﺩﺭﺻﺪ ﻣﺤﺎﺳﺒﻪ ﻣﯽ‌ﺷﻮﺩ. ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﻭﺳﻌﺖ ﺳﻮﺧﺘﮕﯽ ﺩﺭﺟﻪ ﻳﮏ ﺭﺍ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﺍﻳﻦ ﻗﺎﻧﻮﻥ ﻣﺤﺎﺳﺒﻪ ﮐﺮﺩ.  
 
 
ﻋﮑﺲ ۱
 
ﺩﺭ ﻫﻨﮕﺎﻡ ﻣﻮﺍﺟﻪ ﺷﺪﻥ ﺑﺎ ﻣﺼﺪﻭﻡ ﺩﺍﺭﺍﯼ ﺳﻮﺧﺘﮕﯽ ﭼﻪ ﮐﻨﻴﻢ؟
ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﺎ ﻣﺼﺪﻭﻣﯽ ﻣﻮﺍﺟﻪ ﺷﺪﻳﺪ ﮐﻪ ﺑﻪ ﻭﺍﺳﻄﻪ ﺣﺮﺍﺭﺕ ﺳﻮﺧﺘﻪ ﻭ ﺳﻮﺧﺘﮕﯽ ﮔﺮﻣﺎﻳﯽ ﺩﺍﺭﺩ، ﺍﺑﺘﺪﺍ ﻣﺸﺨﺺ ﮐﻨﻴﺪ ﮐﻪ ﺁﻳﺎ ﺍﻳﻦ ﺳﻮﺧﺘﮕﯽ ﺍﺯ ﺩﺭﺟﻪ ﻳﮏ ﺍﺳﺖ ﻳﺎ ﺧﻴﺮ. ﺩﺭ ﺻﻮﺭﺕ ﺩﺭﺟﻪ ﻳﮏ ﺑﻮﺩﻥ ﺍﻳﻦ ﺳﻮﺧﺘﮕﯽ، ﺭﻭﯼ ﻭﺿﻊ ﺳﻮﺧﺘﻪ ﺭﺍ ﺍﺯ ﺣﺪﻭﺩ ۱۰ ﺗﺎ ۴۰ ﺩﻗﻴﻘﻪ ﺳﺮﺩ ﮐﻨﻴﺪ ﻭ ﺳﭙﺲ ﮐﻤﯽ ﻣﺮﻃﻮﺏ ﮐﻨﻨﺪﻩ ﺑﻤﺎﻟﻴﺪ. ﺍﮔﺮ ﺳﻮﺧﺘﮕﯽ ﮐﻤﺘﺮ ﺍﺯ ۵۰ ﺩﺭﺻﺪ ﺍﺳﺖ ﻭ ﺩﺳﺖﻫﺎ ﻭ ﭘﺎﻫﺎ ﻭ ﺍﻋﻀﺎﯼ ﺗﻨﺎﺳﻠﯽ ﻧﺴﻮﺧﺘﻪ ﺑﺎﺷﺪ، ﻧﻴﺎﺯﯼ ﺑﻪ ﻣﺮﺍﺟﻌﻪ ﺑﻪ ﻣﺮﺍﮐﺰ ﺩﺭﻣﺎﻧﯽ ﻧﻴﺴﺖ ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﻪ ﻣﺼﺪﻭﻡ ﺭﺍ ﺑﻪ ﻣﺮﺍﮐﺰﺩﺭﻣﺎﻧﯽ ﺑﺮﺳﺎﻧﻴﺪ.
 
ﺍﮔﺮ ﺳﻮﺧﺘﮕﯽ ﻣﺼﺪﻭﻡ ﺍﺯ ﺩﺭﺟﻪ ﺩﻭ ﺍﺳﺖ، ﺗﺨﻤﻴﻦ ﺑﺰﻧﻴﺪ ﮐﻪ ﺁﻳﺎ ﻭﺳﻌﺖ ﺳﻮﺧﺘﮕﯽ ﺍﻭ ﺯﻳﺎﺩ ﺍﺳﺖ ﻳﺎ ﺧﻴﺮ، ﺍﮔﺮ ﻭﺳﻌﺖ ﺳﻮﺧﺘﮕﯽ ﺯﻳﺎﺩ ﻧﻴﺴﺖ، ﻟﺒﺎﺱ ﻭ ﺯﻳﻮﺭ ﺁﻻﺕ ﻣﺼﺪﻭﻡ ﺭﺍ ﺍﺯ ﺑﺪﻥ ﺍﻭ ﺧﺎﺭﺝ ﮐﻨﻴﺪ. ﺍﮔﺮ ﻟﺒﺎﺱ ﺍﻭ ﺑﻪ ﭘﻮﺳﺖ ﭼﺴﺒﻴﺪﻩ ﺍﺳﺖ، ﺍﺯ ﺩﺳﺘﮑﺎﺭﯼ ﺁﻥ ﺑﭙﺮﻫﻴﺰﻳﺪ ﻭ ﺑﻪ ﺁﻥ ﻧﺎﺣﻴﻪ ﺩﺳﺖ ﻧﺰﻧﻴﺪ ﺯﻳﺮﺍ ﺳﺒﺐ ﺁﺳﻴﺐ ﺑﻪ ﭘﻮﺳﺖ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺯﻳﻮﺭ ﺁﻻﺕ ﻣﺼﺪﻭﻡ ﻫﻤﺎﻧﻨﺪ ﺍﻧﮕﺸﺘﺮ ﻳﺎ ﮔﺮﺩﻧﺒﻨﺪ ﺭﺍ ﺍﺯ ﺩﺳﺘﺎﻥ ﻭ ﻳﺎ ﮔﺮﺩﻥ ﻭﯼ ﺧﺎﺭﺝ ﮐﻨﻴﺪ ﭼﺮﺍ ﮐﻪ ﺍﻟﺘﻬﺎﺏ ﺑﻌﺪﯼ،ﺩﺭ ﺁﻭﺭﺩﻥ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻣﺸﮑﻞ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﭘﺲ ﺍﺯ ﺩﺭﺁﻭﺭﺩﻥ ﻟﺒﺎﺱ ﻣﺼﺪﻭﻡ، ﺭﻭﯼ ﻣﻮﺿﻊ ﺳﻮﺧﺘﮕﯽ ﺭﺍ ۱۰ ﺗﺎ ۴۰ ﺩﻗﻴﻘﻪ ﺳﺮﺩ ﮐﻨﻴﺪ ﺗﺎ ﺩﺭﺩ ﺍﻭ ﺁﺭﺍﻡ ﺷﻮﺩ ﻭ ﺳﭙﺲ ﺍﺯ ﭘﻤﺎﺩ ﺑﺎﺳﻴﺘﺮﺍﺳﻴﻦ ﺑﺮﺍﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﻋﻔﻮﻧﺖ ﺩﺭ ﺳﻮﺧﺘﮕﯽ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﻭ ﺯﺧﻢ ﺭﺍ ﭘﺎﻧﺴﻤﺎﻥ ﮐﻨﻴﺪ. ﺍﮔﺮ ﻭﺳﻌﺖ ﺳﻮﺧﺘﮕﯽ ﺑﻪ ﺑﻴﺶ ﺍﺯ ۱۵ ﺩﺭﺻﺪ ﻣﯽ‌ﺭﺳﺪ ﺑﻴﻤﺎﺭ ﺭﺍ ﺑﻪ ﻣﺮﺍﮐﺰ ﺩﺭﻣﺎﻧﯽ ﻣﻨﺘﻘﻞ ﮐﻨﻴﺪ ﺗﺎ ﻫﺮ ﭼﻪ ﺯﻭﺩﺗﺮ ﺗﺤﺖ ﻣﺪﺍﻭﺍ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ.
 
ﺍﮔﺮ ﺳﻮﺧﺘﮕﯽ ﺍﺯ ﺩﺭﺟﻪ ﺳﻪ ﺑﻮﺩﻩ ﻳﺎ ﻭﺳﻌﺖ ﺳﻮﺧﺘﮕﯽ ﺩﺭﺟﻪ ﺩﻭ ﻣﺼﺪﻭﻡ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩ ﺍﺳﺖ، ﺍﺑﺘﺪﺍ CBA ﺭﺍ ﺩﺭ ﻣﺼﺪﻭﻡ ﭼﮏ ﮐﻨﻴﺪ، ﺳﭙﺲ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺷﻮﮎ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﺯﻳﻮﺭﺁﻻﺕ ﻭ ﻟﺒﺎﺱ ﺍﻭ ﺭﺍ ﺍﺯ ﺑﺪﻧﺶ ﺩﺭ ﺑﻴﺎﻭﺭﻳﺪ. ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﻟﺒﺎﺱ ﺭﺍ ﺑﻪ ﺯﻭﺭ ﻭ ﮐﺸﻴﺪﻥ ﺍﺯ ﻣﺤﻞ ﺳﻮﺧﺘﻪ ﺟﺪﺍ ﻧﮑﻨﻴﺪ ﺑﻠﮑﻪ ﺁﻥ ﻗﺴﻤﺖ ﺭﺍ ﭘﺎﺭﻩ ﮐﻨﻴﺪ ﻭ ﺑﻪ ﺣﺎﻝ ﺧﻮﺩ ﺭﻫﺎ ﮐﻨﻴﺪ. ﺩﺳﺖ ﻳﺎ ﭘﺎﯼ ﺳﻮﺧﺘﻪ ﺭﺍ ﺑﺎﻻﺗﺮ ﺍﺯ ﻗﻠﺐ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ ﻭ ﺑﺎ ﺍﻭﺭﮊﺍﻧﺲ ﺗﻤﺎﺱ ﺑﮕﻴﺮﻳﺪ.
 
ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ
ﺩﺭ ﺳﻮﺧﺘﮕﯽ ﺩﺭﺟﻪ ﺩﻭ ﻭ ﻣﻮﺍﺟﻪ ﺷﺪﻥ ﺑﺎ ﺗﺎﻭﻝ ﺳﻮﺧﺘﻪ ﺭﻭﯼ ﭘﻮﺳﺖ، ﺍﺯ ﭘﺎﺭﻩ ﮐﺮﺩﻥ ﺍﻳﻦ ﺗﺎﻭﻝ‌ﻫﺎ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺍﻳﻦ ﺗﺎﻭﻝ‌ﻫﺎ ﻳﮏ ﻻﻳﻪ ﻣﺤﺎﻓﻈﺖ ﺧﻮﺏ ﺑﺮﺍﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﻋﻔﻮﻧﺖ ﻫﺴﺘﻨﺪ ﻭ ﻫﻤﺎﻧﻨﺪ ﻳﮏ ﭘﺎﻧﺴﻤﺎﻥ ﺍﺳﺘﺮﻳﻞ ﻋﻤﻞ ﻣﯽ‌ﮐﻨﻨﺪ.
 
ﺩﺭ ﻫﻨﮕﺎﻡ ﺳﻮﺧﺘﮕﯽ ﺷﻴﻤﻴﺎﻳﯽ ﭼﻪ ﺑﺎﻳﺪ ﮐﺮﺩ؟
ﺍﮔﺮ ﻣﺎﺩﻩ ﺷﻴﻤﻴﺎﻳﯽ ﮐﻪ ﺳﺒﺐ ﺳﻮﺧﺘﮕﯽ ﺷﺪﻩ، ﻣﺎﺩﻩ‌ﺍﯼ ﻣﺎﻳﻊ ﺍﺳﺖ، ﺑﻼﻓﺎﺻﻠﻪ ﻣﺤﻞ ﺳﻮﺧﺘﮕﯽ ﺭﺍ ﺑﺎ ﺁﺏ ﺑﻪ ﻣﺪﺕ ۲۰ ﺩﻗﻴﻘﻪ ﺑﺸﻮﻳﻴﺪ، ﺑﻪ ﻓﮑﺮ ﺧﻨﺜﯽ ﮐﺮﺩﻥ ﻣﺎﺩﻩ ﺷﻴﻤﻴﺎﻳﯽ ﻧﺒﺎﺷﻴﺪ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺳﻮﺧﺘﮕﯽ ﺭﺍ ﺑﻴﺸﺘﺮ ﮐﻨﺪ. ﻟﺒﺎﺱ ﻭ ﺯﻳﻮﺭﺁﻻﺕ ﻣﺼﺪﻭﻡ ﺭﺍ ﻧﻴﺰ ﺍﺯ ﺑﺪﻧﺶ ﺧﺎﺭﺝ ﮐﺮﺩﻩ ﻭ ﻣﺼﺪﻭﻡ ﺭﺍ ﺑﻪ ﻣﺮﺍﮐﺰ ﺩﺭﻣﺎﻧﯽ ﺑﺮﺳﺎﻧﻴﺪ. ﺍﮔﺮ ﻣﺎﺩﻩ ﺷﻴﻤﺎﻳﯽ ﺳﺒﺐ ﺳﺎﺯ ﺳﻮﺧﺘﮕﯽ، ﻣﺎﺩﻩ‌ﺍﯼ ﺧﺸﮏ ﺍﺳﺖ، ﺍﺑﺘﺪﺍ ﺳﻌﯽ ﮐﻨﻴﺪ ﻣﺎﺩﻩ ﺧﺸﮏ ﺭﺍ ﺍﺯ ﺭﻭﯼ ﭘﻮﺳﺖ ﭘﺎﮎ ﮐﻨﻴﺪ ﭼﺮﺍ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﻭﺍﮐﻨﺶ ﺑﺎ ﺁﺏ ﺷﺪﺕ ﺳﻮﺧﺘﮕﯽ ﺭﺍ ﺑﻴﺸﺘﺮ ﮐﻨﺪ. ﭘﺲ ﺍﺯ ﭘﺎﮎ ﮐﺮﺩﻥ ﻣﺎﺩﻩ ﺧﺸﮏ ﺍﺯ ﻣﺤﻞ ﺳﻮﺧﺘﮕﯽ، ﻫﻤﺎﻧﻨﺪ ﺳﻮﺧﺘﮕﯽ ﻣﺎﺩﻩ ﺷﻴﻤﻴﺎﻳﯽ ﻣﺎﻳﻊ ﻋﻤﻞ ﮐﻨﻴﺪ ﻭ ﻣﺤﻞ ﺳﻮﺧﺘﮕﯽ ﺭﺍ ۲۰ ﺩﻗﻴﻘﻪ ﺯﻳﺮ ﺍﺏ ﺑﺸﻮﻳﻴﺪ ﻭ ﺳﺎﻳﺮ ﮐﺎﺭﻫﺎ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ.
 
ﺩﺭ ﻫﻨﮕﺎﻡ ﺳﻮﺧﺘﮕﯽ ﺍﻟﮑﺘﺮﻳﮑﯽ ﭼﻪ ﺑﺎﻳﺪ ﮐﺮﺩ؟
ﻫﻨﮕﺎﻡ ﺭﺥ ﺩﺍﺩﻥ ﺳﻮﺧﺘﮕﯽ ﺍﻟﮑﺘﺮﻳﮑﯽ، ﻣﻬﻤﺘﺮﻳﻦ ﻣﻮﺿﻮﻉ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺗﻤﺎﺱ ﺟﺮﻳﺎﻥ ﺑﺮﻕ ﻭ ﻣﺼﺪﻭﻡ ﺭﺍ ﻗﻄﻊ ﮐﻨﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﻗﻄﻊ ﮐﺮﺩﻥ ﺟﺮﻳﺎﻥ ﺑﺮﻕ، ﺧﺎﻣﻮﺵ ﮐﺮﺩﻥ ﮐﻠﻴﺪ، ﻗﻄﻊ ﮐﺮﺩﻥ ﮐﻨﺘﻮﺭ ﻳﺎ ﻣﻮﺍﺭﺩﯼ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﺍﻧﺠﺎﻡ ﺷﻮﺩ. ﭘﺲ ﺍﺯ ﺁﻥ ﺑﻪ ﺳﺮﻋﺖ ﺑﺎ ﺍﻭﺭﮊﺍﻧﺲ ﺗﻤﺎﺱ ﺑﮕﻴﺮﻳﺪ ﻭ ﭘﻴﺶ ﺍﺯ ﺭﺳﻴﺪﻥ ﺍﻭﺭﮊﺍﻧﺲ، ﻣﺼﺪﻭﻡ ﺭﺍ ﺍﺯ ﻧﻈﺮ CBA ﺑﺮﺭﺳﯽ ﮐﺮﺩﻩ ﻭ ﺷﻮﮎ ﺩﺭﻣﺎﻧﯽ ﮐﻨﻴﺪ.
 
ﺣﻤﻠﻪ ﻗﻠﺒﯽ
ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﻪ ﭘﺎﻳﺎﻥ ﻣﺒﺤﺚ ﺳﻮﺧﺘﮕﯽ‌ﻫﺎ ﻣﯽ‌ﺭﺳﻴﻢ ﻭ ﺩﺭﺱ ﺭﺍ ﺑﺎ ﻧﮑﺎﺗﯽ ﺍﺯ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﻪ ﻣﺼﺪﻭﻣﯽ ﮐﻪ ﺩﭼﺎﺭ ﺣﻤﻠﻪ ﻗﻠﺒﯽ ﺷﺪﻩ ﺍﺩﺍﻣﻪ ﻣﯽ‌ﺩﻫﻴﻢ. ﺩﺭ ﺣﻤﻠﻪ ﻗﻠﺒﯽ ﻓﺮﺩ ﺩﺭﺩ ﺑﺎ ﻓﺸﺎﺭ ﺷﺪﻳﺪﯼ ﺩﺭ ﻣﺮﮐﺰ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﭼﻨﺪ ﺩﻗﻴﻘﻪ ﺑﻪ ﻃﻮﻝ ﻣﯽ‌ﺍﻧﺠﺎﻣﺪ. ﻣﺼﺪﻭﻡ ﻣﻌﻤﻮﻻ ﺗﻌﺮﻕ ﺩﺍﺭﺩ ﻭ ﺣﺎﻟﺖ ﺗﻬﻮﻉ ﻭ ﺗﻨﮕﯽ ﻧﻔﺲ ﺑﻪ ﺍﻭ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﺪ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﻫﺮ ﭼﻪ ﺳﺮﻳﻌﺘﺮ ﺑﺎ ﺍﻭﺭﮊﺍﻧﺲ ﺗﻤﺎﺱ ﺑﮕﻴﺮﻳﺪ. ﭘﻴﺶ ﺍﺯ ﺭﺳﻴﺪﻥ ﺍﻭﺭﮊﺍﻧﺲ، ﺳﻌﯽ ﮐﻨﻴﺪ ﻣﺼﺪﻭﻡ ﺭﺍ ﺩﺭ ﺣﺎﻟﺖ ﻧﻴﻤﻪ ﻧﺸﺴﺘﻪ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ﺍﺯ ﺧﻮﺍﺑﺎﻧﺪﻥ ﻣﺼﺪﻭﻡ ﭘﺮﻫﻴﺰ ﮐﻨﻴﺪ. ﺩﺭ ﻭﺿﻌﻴﺖ ﻧﻴﻤﻪ ﻧﺸﺴﺘﻪ، ﺯﻳﺮ ﺯﺍﻧﻮﻫﺎ ﺭﺍ ﺑﺎ ﮔﺬﺍﺷﺘﻦ ﺑﺎﻟﺶ ﻳﺎ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺷﺒﻴﻪ ﺑﻪ ﺁﻥ ﺑﺎﻻ ﺑﻴﺎﻭﺭﻳﺪ ﻭ ﭘﺸﺖ ﻣﺼﺪﻭﻡ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﺩﻳﻮﺍﺭ ﺗﮑﻴﻪ ﺩﻫﻴﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﻓﺸﺎﺭ ﻟﺒﺎﺱ ﻣﺼﺪﻭﻡ ﺩﺭ ﺩﻭﺭ ﮐﻤﺮ ﻳﺎ ﮔﺮﺩﻥ ﺭﺍ ﺷﻞ ﮐﻨﻴﺪ ﺗﺎ ﻣﺼﺪﻭﻡ ﺩﺭ ﺣﺎﻟﺘﯽ ﺁﺯﺍﺩ ﻗﺮﺍﺭ ﮔﻴﺮﺩ.
 
 ﺍﮔﺮ ﻣﺼﺪﻭﻡ ﺍﺯ ﻗﺮﺹ ﻧﻴﺘﺮﻭﮔﻠﻴﺴﻴﺮﻳﻦ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﺪ، ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺍﻳﻦ ﻗﺮﺹ ﺭﺍ ﺑﻪ ﺍﻭ ﺑﺪﻫﻴﺪ ﭼﺮﺍ ﮐﻪ ﺍﻳﻦ ﻗﺮﺹ ﺑﻪ ﺭﺍﺣﺖ ﮐﺎﺭ ﮐﺮﺩﻥ ﻗﻠﺐ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ. ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﻧﻴﺘﺮﻭﮔﻠﻴﺴﻴﺮﻳﻦ، ﻓﺸﺎﺭ ﺧﻮﻥ ﺭﺍ ﭘﺎﻳﻴﻦ ﻣﯽ‌ﺁﻭﺭﺩ ﻭ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﭘﺲ ﺍﺯ ﺍﺳﺘﻔﺎﺩﻩ ﺁﻥ، ﺑﺎﻳﺪ ﺩﺭ ﺣﺎﻟﺖ ﻧﺸﺴﺘﻪ ﻳﺎ ﺩﺭﺍﺯﮐﺶ ﻗﺮﺍﺭ ﮔﻴﺮﺩ.
 
ﺍﮔﺮ ﻣﺼﺪﻭﻡ ﭘﺲ ﺍﺯ ﺣﻤﻠﻪ ﻗﻠﺒﯽ ﺩﺭ ﺷﺮﺍﻳﻂ ﺩﺷﻮﺍﺭﯼ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﻭ ﺗﻨﻔﺲ ﻧﻤﯽ‌ﮐﻨﺪ، ﺑﯽ ﺩﺭﻧﮓ ﺍﻭﺭﮊﺍﻧﺲ ﺭﺍ ﺧﺒﺮ ﮐﺮﺩﻩ ﻭ CBA ﺭﺍ ﺭﻭﯼ ﺑﺪﻥ ﻭﯼ ﺁﻏﺎﺯ ﮐﻨﻴﺪ.
 
ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ، ﺩﺭ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﻣﺒﺤﺚ ﺳﻮﺧﺘﮕﯽ‌ﻫﺎ ﻭ ﻧﺤﻮﻩ ﺍﻣﺪﺍﺩ ﺍﺑﺘﺪﺍﻳﯽ ﺑﻪ ﻓﺮﺩ ﺩﭼﺎﺭ ﺳﻮﺧﺘﮕﯽ ﺭﺍ ﺑﺎ ﻫﻢ ﻓﺮﺍ ﮔﺮﻓﺘﻴﻢ ﻭ ﺩﺭ ﺁﺧﺮ ﻧﻴﺰ ﻧﺤﻮﻩ ﮐﻤﮏ ﺍﺑﺘﺪﺍﻳﯽ ﺑﻪ ﻓﺮﺩ ﺩﭼﺎﺭ ﺣﻤﻠﻪ ﻗﻠﺒﯽ ﺭﺍ ﻣﺮﻭﺭ ﮐﺮﺩﻳﻢ. ﻳﺎﺩﮔﻴﺮﯼ ﮐﻤﮏ ﺑﻪ ﻣﺼﺪﻭﻣﺎﻥ ﺩﭼﺎﺭ ﺣﻤﻠﻪ ﻗﻠﺒﯽ ﺍﻳﻦ ﺭﻭﺯﻫﺎ ﺍﺯ ﺍﻫﻤﻴﺘﯽ ﻭﻳﮋﻩ ﺑﺮﺧﻮﺭﺩﺍﺭ ﺍﺳﺖ ﭼﺮﺍ ﮐﻪ ﺁﻣﺎﺭ ﺣﻤﻠﻪ‌ﻫﺎﯼ ﻗﻠﺒﯽ ﺩﺭ ﺍﻳﺮﺍﻥ ﻭ ﺩﻳﮕﺮ ﻧﻘﺎﻁ ﺟﻬﺎﻥ ﺭﻭ ﺑﻪ ﺍﻓﺰﺍﻳﺶ ﺍﺳﺖ.
