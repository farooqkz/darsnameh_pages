ﺍﮔﺮ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺮﺍﻳﺘﺎﻥ ﭘﻴﺶ ﺁﻣﺪﻩ ﻭ ﺧﻮﺩ ﺭﺍ ﺳﺮﺯﻧﺶ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺗﻨﻬﺎ ﻧﻴﺴﺘﻴﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺩﺭ ﻭﺍﻗﻊ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻓﺮﺍﺩﯼ ﺭﺍ ﮐﻪ ﺩﺭ ﺭﺷﺘﻪ‌ﻫﺎﯼ ﻓﻨﯽ ﻳﺎ ﻧﺎﻣﺮﺑﺘﻂ ﺑﺎ ﻋﻠﻮﻡ ﺗﺠﺮﺑﯽ ﺩﺭﺱ ﺧﻮﺍﻧﺪ‌ﻩ‌ﺍﻧﺪ ﺭﺍ ﺩﺭ ﺑﺮ ﻣﯽ‌ﮔﻴﺮﺩ!l

ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﭘﺰﺷﮑﯽ ﻧﺨﻮﺍﻧﺪﻩ‌ﺍﻳﺪ ﻭ ﺑﺎ ﺍﻳﻦ ﺩﺍﻧﺶ ﺣﺘﯽ ﺁﺷﻨﺎﻳﯽ ﺟﺰﻳﯽ ﻫﻢ ﻧﺪﺍﺭﻳﺪ، ﻫﻨﻮﺯ ﻫﻢ ﻭﻗﺖ ﺩﺍﺭﻳﺪ ﮐﻪ ﺑﺎ ﻓﺮﺍﮔﻴﺮﯼ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ، ﺩﺭ ﻫﻨﮕﺎﻡ ﻭﻗﻮﻉ ﺳﺎﻧﺤﻪ ﺍﺯ ﺷﺪﺕ ﺻﺪﻣﺎﺕ ﺁﺳﻴﺐ‌ﺩﻳﺪﮔﺎﻥ ﮐﻢ ﮐﻨﻴﺪ ﻭ ﺩﺭ ﺯﻣﺎﻥ ﻃﻼﻳﯽ ﺍﻣﺪﺍﺩ، ﺑﻪ ﻣﺼﺪﻭﻣﺎﻥ ﮐﻤﮏ ﮐﻨﻴﺪ.

ﺩﺭ ﻭﺍﻗﻊ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﻪ ﻣﺠﻤﻮﻋﻪ ﺍﻗﺪﺍﻣﺎﺕ ﻭ ﻣﺮﺍﻗﺒﺖ‌ﻫﺎﻳﯽ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺍﻣﺪﺍﺩﮔﺮ ﺩﺭ ﺯﻣﺎﻥ ﮐﻮﺗﺎﻫﯽ ﭘﺲ ﺍﺯ ﻭﻗﻮﻉ ﺣﺎﺩﺛﻪ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﺪ ﺗﺎ ﻣﺠﺮﻭﺡ ﺭﺍ ﺍﺯ ﺻﺪﻣﺎﺕ ﺷﺪﻳﺪ ﺣﺎﺩﺛﻪ ﻧﺠﺎﺕ ﺩﻫﺪ. ﺍﻳﻦ ﺣﺎﺩﺛﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﻳﮏ ﺗﺼﺎﺩﻑ ﺷﺪﻳﺪ، ﺳﮑﺘﻪ ﻗﻠﺒﯽ، ﺁﺳﻴﺐ‌ ﻭﺭﺯﺷﯽ، ﮔﺮﻣﺎ ﻳﺎ ﺳﺮﻣﺎ ﺯﺩﮔﯽ ﻭ ﻳﺎ ﻣﻮﺍﺭﺩﯼ ﭼﻮﻥ ﻣﺎﺭ ﮔﺰﻳﺪﮔﯽ ﻭ ﻣﺴﻤﻮﻣﻴﺖ ﺑﺎﺷﺪ.

ﻫﻤﻴﺸﻪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺩﺭ ﻫﻨﮕﺎﻡ ﺭﺥ ﺩﺍﺩﻥ ﻳﮏ ﺳﺎﻧﺤﻪ ﻳﺎ ﺍﺗﻔﺎﻕ ﻧﺎﺧﻮﺷﺎﻳﻨﺪ، ﻣﺪﺕ ﺯﻣﺎﻥ ﮐﻮﺗﺎﻩ ﺑﻌﺪ ﺍﻳﻦ ﺭﺧﺪﺍﺩ ﮐﻪ ﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻤﺘﺮ ﺍﺯ ﭘﻨﺞ ﺩﻗﻴﻘﻪ ﺑﺎﺷﺪ، ﻣﻬﻤﺘﺮﻳﻦ ﺯﻣﺎﻥ ﺍﻣﺪﺍﺩ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ ﮐﻪ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺳﺒﺐ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﻣﺮﮒ ﻣﺼﺪﻭﻡ ﺷﻮﺩ. ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ، ﺍﻳﻦ ﺯﻣﺎﻥ ﺑﻪ ﺯﻣﺎﻥ ﻃﻼﻳﯽ ﺍﻣﺪﺍﺩ ﻣﺸﻬﻮﺭ ﺷﺪﻩ ﺍﺳﺖ. ﺍﻳﻦ ﺯﻣﺎﻥ ﺑﺮﺍﯼ ﺣﺎﻟﺘﯽ ﮐﻪ ﺷﺨﺺ ﺗﻮﺍﻧﺎﻳﯽ ﻧﻔﺲ ﮐﺸﻴﺪﻥ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻩ ﺑﻪ ﺣﺪﻭﺩ ﭼﻬﺎﺭ ﺗﺎ ﺷﺶ ﺩﻗﻴﻘﻪ ﻣﯽ‌ﺭﺳﺪ.

ﻧﻴﺎﺯﯼ ﻧﻴﺴﺖ ﺍﻓﺮﺍﺩ ﺍﻣﺪﺍﺩﮔﺮ ﺣﺎﺿﺮ ﺩﺭ ﺍﻳﻦ ﺻﺤﻨﻪ‌ﻫﺎ ﺩﺍﻧﺶ ﭘﺰﺷﮑﯽ ﮔﺴﺘﺮﺩﻩ‌ﺍﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ، ﺍﻃﻼﻋﺎﺕ ﻣﺨﺘﺼﺮ ﺍﻣﺎ ﻣﻔﻴﺪ ﺩﺭ ﺯﻣﻴﻨﻪ ﭘﺰﺷﮑﯽ ﮐﻪ ﺟﺰﻭ ﻣﻬﺎﺭﺕ‌ﻫﺎﯼ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﻘﺶ ﻣﻬﻤﯽ ﺩﺭ ﻧﺠﺎﺕ ﻣﺠﺮﻭﺣﺎﻥ ﺍﻳﻔﺎ ﮐﻨﺪ.

ﺷﻤﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﻓﺮﺩ ﺍﻣﺪﺍﺩﮔﺮ ﺩﺭ ﺍﻳﻦ ﺻﺤﻨﻪ‌ﻫﺎ ﺍﺑﺘﺪﺍ ﺑﺎﻳﺪ ﺧﻮﻧﺴﺮﺩﯼ ﺧﻮﺩ ﺭﺍ ﺣﻔﻆ ﮐﻨﻴﺪ ﻭ ﺩﺭ ﺍﻧﺠﺎﻡ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺳﺮﻋﺖ ﻋﻤﻞ ﺑﻪ ﺧﺮﺝ ﺩﻫﻴﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺍﺯ ﺍﻧﺠﺎﻡ ﻫﻤﻪ ﮐﻤﮏ‌ﻫﺎ ﺑﺮ ﻧﻴﺎﻳﻴﺪ ﻭ ﺑﺎ ﻓﺮﻳﺎﺩ ﮐﺸﻴﺪﻥ ﺩﺭﺧﻮﺍﺳﺖ ﮐﻤﮏ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﻣﺠﺒﻮﺭ ﺷﻮﻳﺪ ﺑﺎ ﺭﺍﻫﮑﺎﺭﯼ ﺍﺑﺘﮑﺎﺭﯼ، ﺯﻣﻴﻨﻪ ﺍﻧﺘﻘﺎﻝ ﻣﺼﺪﻭﻡ ﺍﺯ ﺻﺤﻨﻪ ﺣﺎﺩﺛﻪ ﺭﺍ ﻓﺮﺍﻫﻢ ﺁﻭﺭﻳﺪ.

ﺩﺭ ﮐﻞ، ﻓﺮﺩ ﺍﻣﺪﺍﺩﮔﺮ ﺑﺎﻳﺪ ﺩﺍﺭﺍﯼ ﺗﻌﺎﺩﻝ ﺭﻭﺣﯽ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﺭﻭﻳﺎﺭﻭﻳﯽ ﺑﺎ ﺻﺤﻨﻪ‌ﻫﺎﯼ ﺍﺣﺘﻤﺎﻟﺂ ﺩﻟﺨﺮﺍﺵ ﺑﺎﺷﺪ ﻭ ﺑﺎ ﺑﻪ ﮐﺎﺭﮔﻴﺮﯼ ﺩﺍﻧﺶ ﺧﻮﺩ ﺍﺯ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﻪ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﮐﻤﮏ ﮐﻨﺪ.

ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ

ﺩﺭ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻣﺼﺪﻭﻣﺎﻧﯽ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺩﺍﺭﻧﺪ، ﺗﺸﺨﻴﺺ ﻋﺎﺭﺿﻪ ﺩﺭ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ ﻣﻬﻤﺘﺮﻳﻦ ﺑﺨﺶ ﮐﻤﮏ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﺪ. ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ ﺷﺎﻳﺪ ﺑﻪ ﺻﻮﺭﺕ ﺳﺎﺩﻩ‌ﯼ ﭘﺮﺳﻴﺪﻥ ﺍﺯ ﻣﺠﺮﻭﺡ ﻭ ﺟﻮﺍﺏ ﻭﯼ ﺁﻏﺎﺯ ﺷﻮﺩ ﻳﺎ ﻧﻴﺎﺯ ﺑﻪ ﺍﻧﺠﺎﻡ ﺗﮑﻨﻴﮏ‌ﻫﺎﯼ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﮔﻔﺘﻪ ﺷﺪ ﺳﺎﺩﻩ‌ﺗﺮﻳﻦ ﺭﺍﻩ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ، ﭘﺮﺳﻴﺪﻥ ﺍﺯ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺍﺳﺖ، ﺍﻳﻨﮑﻪ ﺩﺭ ﮐﺠﺎ ﺍﺣﺴﺎﺱ ﺩﺭﺩ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺣﺎﻝ ﺟﺴﻤﺎﻧﯽ‌ﺍﺵ ﭼﮕﻮﻧﻪ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻳﺪ ﺑﺎ ﺩﺳﺖ ﺑﻪ ﺷﺎﻧﻪ‌ﻫﺎ ﻳﺎ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺑﺰﻧﻴﺪ ﻭ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﺍﺯ ﺍﻭ ﺑﭙﺮﺳﻴﺪ «ﺣﺎﻟﺖ ﺧﻮﺑﻪ؟» ﺍﮔﺮ ﻣﺼﺪﻭﻡ ﺗﻮﺍﻧﺎﻳﯽ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺭﺍ ﺩﺍﺭﺩ، ﺑﺎﻳﺪ ﺑﻪ ﺍﻭ ﺭﻭﺣﻴﻪ ﺩﻫﻴﺪ ﻭ ﺍﺯ ﺍﻭ ﺑﭙﺮﺳﻴﺪ ﮐﻪ ﺩﺭ ﭼﻪ ﺟﺎﻫﺎﻳﯽ ﺍﺣﺴﺎﺱ ﺩﺭﺩ ﻣﯽ‌ﮐﻨﺪ. ﺩﺭ ﺍﻳﻦ ﻫﻨﮕﺎﻡ ﻧﺒﺎﻳﺪ ﺯﻣﺎﻥ ﺭﺍ ﻧﻴﺰ ﺍﺯ ﺩﺳﺖ ﺑﺪﻫﻴﺪ ﻭ ﻓﻮﺭﺍ ﺑﺎ ﺷﻤﺎﺭﻩ ۱۱۵ ﺩﺭ ﺍﻳﺮﺍﻥ ﻭ ۱۱۲ ﺩﺭ ﺍﻓﻐﺎﻧﺴﺘﺎﻥ ﮐﻪ ﻣﺘﻌﻠﻖ ﺑﻪ ﻓﻮﺭﻳﺖ‌ﻫﺎﯼ ﭘﺰﺷﮑﯽ ﺍﺳﺖ ﺗﻤﺎﺱ ﺑﮕﻴﺮﻳﺪ.

ﻣﻤﮑﻦ ﺍﺳﺖ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺑﻪ ﻋﻠﺖ ﻓﺸﺎﺭ ﺷﺪﻳﺪ ﻳﺎ ﺑﻴﻬﻮﺷﯽ ﻧﺘﻮﺍﻧﺪ ﺑﻪ ﺍﻣﺪﺍﺩﮔﺮ ﭘﺎﺳﺦ ﺑﺪﻫﺪ، ﭘﺲ ﺑﺎﻳﺪ ﺑﺎ ﺑﺮﺭﺳﯽ ﺑﺪﻥ ﻓﺮﺩ ﻣﺠﺮﻭﺡ، ﺑﻪ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺩﺳﺘﺮﺳﯽ ﭘﻴﺪﺍ ﮐﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﻫﻨﮕﺎﻡ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﺎ ﻓﺮﻳﺎﺩ ﺯﺩﻥ ﺍﺯ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮ ﺣﺎﺿﺮ ﺩﺭ ﺻﺤﻨﻪ ﮐﻤﮏ ﺑﮕﻴﺮﻳﺪ ﻭ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺑﺨﻮﺍﻫﻴﺪ ﺑﺎ ﻓﻮﺭﻳﺖ‌ﻫﺎﯼ ﭘﺰﺷﮑﯽ ﺗﻤﺎﺱ ﺑﮕﻴﺮﻧﺪ ﻭ ﺧﻮﺩ ﻣﺸﻐﻮﻝ ﺍﻧﺠﺎﻡ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺷﻮﻳﺪ.

ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ
ﻫﺮ ﭼﻨﺪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﺗﻮﺍﻧﺎﻳﯽ ﮐﻤﮏ ﮐﺎﻣﻞ ﺑﻪ ﺑﻴﻤﺎﺭ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺍﻣﺎ ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﺯ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮ ﺣﺎﺿﺮ ﺩﺭ ﺻﺤﻨﻪ ﻧﻴﺰ ﮐﻤﮏ ﺑﺨﻮﺍﻫﻴﺪ ﺗﺎ ﺑﻬﺘﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻣﺪﺍﺩ ﺑﭙﺮﺩﺍﺯﻳﺪ. ﺩﺭ ﻫﻨﮕﺎﻡ ﺍﺟﺮﺍﯼ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺍﮔﺮ ﺑﻴﻤﺎﺭ ﺩﺭ ﺣﺎﻟﺖ ﻫﻮﺷﻴﺎﺭﯼ ﺍﺳﺖ، ﺣﺘﻤﺎ ﺑﺎ ﺻﺤﺒﺖ‌ﻫﺎﻳﺘﺎﻥ ﺑﻪ ﺍﻭ ﺭﻭﺣﻴﻪ ﺑﺪﻫﻴﺪ ﻭ ﺍﻭ ﺭﺍ ﺁﺭﺍﻡ ﮐﻨﻴﺪ.

ﺷﻤﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﺍﻣﺪﺍﺩﮔﺮ، ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﺎ ﻳﮏ ﻣﺼﺪﻭﻡ ﺑﻴﻬﻮﺵ ﻳﺎ ﻧﺎﺗﻮﺍﻥ ﺩﺭ ﭘﺎﺳﺨﮕﻮﻳﯽ ﻣﻮﺍﺟﻪ ﻣﯽ‌ﺷﻮﻳﺪ، ﺑﺮﺍﯼ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ ﺑﺎﻳﺪ ﺑﻪ ﺻﻮﺭﺕ ﺧﻮﺩﮐﺎﺭ ﺳﻪ ﻣﻮﺭﺩ ﺭﺍ ﭼﮏ ﮐﻨﻴﺪ، ﻧﺨﺴﺘﻴﻦ ﻣﺮﺣﻠﻪ (A) ﺍﺯ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ، ﺑﺮﺭﺳﯽ ﺭﺍﻩ ﻫﻮﺍﻳﯽ ﻣﺼﺪﻭﻡ (Airway) ﺍﺳﺖ. ﺩﺭ ﺑﺮﺧﯽ ﺍﺯ ﺳﻮﺍﻧﺢ ﻣﻤﮑﻦ ﺍﺳﺖ ﻓﺮﺩ ﺩﭼﺎﺭ ﺑﻴﻬﻮﺷﯽ ﺷﻮﺩ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﻨﺠﺮ ﺑﻪ ﺷﻞ ﺷﺪﻥ ﻋﻀﻠﻪ‌ﻫﺎﯼ ﮔﻠﻮ ﻭ ﮔﻴﺮﮐﺮﺩﻥ ﺯﺑﺎﻥ ﻣﺼﺪﻭﻡ ﺩﺭ ﺣﻠﻖ ﻭﯼ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺳﺎﺩﮔﯽ ﺭﺍﻩ ﺗﻨﻔﺲ ﻣﺼﺪﻭﻡ ﺭﺍ ﻣﺴﺪﻭﺩ ﮐﺮﺩﻩ ﻭ ﺳﺒﺐ ﻣﺮﮒ ﻭﯼ ﺷﻮﺩ.

ﺩﻭﻣﻴﻦ ﻣﻮﺿﻮﻋﯽ ﮐﻪ ﺩﺭ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ ﺑﺎﻳﺪ ﻣﻮﺭﺩ ﺍﺭﺯﻳﺎﺑﯽ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ (B)، ﺍﻧﺠﺎﻡ ﺗﻨﻔﺲ ﻃﺒﻴﻌﯽ (Breathing) ﺗﻮﺳﻂ ﻣﺼﺪﻭﻡ ﺍﺳﺖ، ﺍﻳﻨﮑﻪ ﺁﻳﺎ ﻣﺼﺪﻭﻡ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺻﻮﺭﺕ ﻃﺒﻴﻌﯽ ﻧﻔﺲ ﺑﮑﺸﺪ ﻳﺎ ﺧﻴﺮ.

ﺳﻮﻣﻴﻦ ﻭ ﺁﺧﺮﻳﻦ ﻣﺮﺣﻠﻪ ﺩﺭ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ (C)، ﺑﺮﺭﺳﯽ (Circulation) ﮔﺮﺩﺵ ﺧﻮﻥ ﻣﺼﺪﻭﻡ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻓﺮﺩ ﺍﻣﺪﺍﺩﮔﺮ ﺑﺎﻳﺪ ﻣﺘﻮﺟﻪ ﺷﻮﺩ ﮐﻪ ﺁﻳﺎ ﺧﻮﻧﺮﻳﺰﯼ ﺷﺪﻳﺪﯼ ﺩﺭ ﺑﺪﻥ ﻓﺮﺩ ﺩﺭ ﺣﺎﻝ ﺭﺥ ﺩﺍﺩﻥ ﺍﺳﺖ ﻳﺎ ﻫﻤﻪ ﭼﻴﺰ ﺩﺭ ﺭﻭﺍﻝ ﻃﺒﻴﻌﯽ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﺍﺭﺩ.

ﭘﺲ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﻢ ﺩﺭ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻣﺼﺪﻭﻡ ﺑﻴﻬﻮﺵ ﻳﺎ ﻧﺎﺗﻮﺍﻥ ﺑﺎﻳﺪ ﺑﺪﻭﻥ ﺩﺭﻧﮓ، ﺳﻪ ﻣﻮﺿﻮﻉ ﺭﺍ ﺑﻪ ﺗﺮﺗﻴﺐ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ:
A- ﺑﺮﺭﺳﯽ ﺭﺍﻩ ﻫﻮﺍﻳﯽ ﻣﺼﺪﻭﻡ
B- ﺗﻨﻔﺲ ﻃﺒﻴﻌﯽ ﻣﺼﺪﻭﻡ
C- ﮔﺮﺩﺵ ﺧﻮﻥ

ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ ﺭﺍ ﭼﮕﻮﻧﻪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ؟

ﺩﺭ ﻣﺮﺣﻠﻪ ﻧﺨﺴﺖ ﺍﺭﺯﻳﺎﺑﯽ ﺍﻭﻟﻴﻪ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﮔﻔﺘﻪ ﺷﺪ ﺑﺎﻳﺪ ﺭﺍﻩ ﻫﻮﺍﻳﯽ ﻣﺼﺪﻭﻡ ﻣﻮﺭﺩ ﺑﺎﺯﺭﺳﯽ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ. ﺍﮔﺮ ﻓﺮﺩ ﺑﻴﻬﻮﺵ ﺑﺎﺷﺪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺩﻻﻳﻠﯽ ﭼﻮﻥ ﺧﻢ ﺷﺪﻥ ﺳﺮ ﺑﻪ ﺟﻠﻮ، ﺑﺮﮔﺸﺖ ﺯﺑﺎﻥ ﺑﻪ ﺣﻠﻖ ﻭ ﻳﺎ ﺑﺮﮔﺸﺖ ﺁﺏ ﺩﻫﺎﻥ ﻳﺎ ﺍﺳﺘﻔﺮﺍﻍ ﺑﻪ ﺣﻠﻖ ﺭﺍﻩ ﺗﻨﻔﺲ ﺑﺴﺘﻪ ﺷﻮﺩ ﮐﻪ ﻫﻤﻪ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻣﺮﮒ ﻣﺼﺪﻭﻡ ﻣﻨﺠﺮ ﺷﻮﺩ.

ﺑﺮﺍﯼ ﺑﺮﺭﺳﯽ ﺑﺎﺯﺑﻮﺩﻥ ﺭﺍﻩ ﻫﻮﺍﻳﯽ ﻣﺼﺪﻭﻡ ﺑﺎﻳﺪ ﺍﺑﺘﺪﺍ ﮔﻮﺵ ﺧﻮﺩ ﺭﺍ ﻧﺰﺩﻳﮏ ﺩﻫﺎﻥ ﻭﯼ ﺑﺮﺩﻩ ﻭ ﺑﻪ ﻣﺪﺕ ۱۰ ﺛﺎﻧﻴﻪ ﺑﻪ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺍﻭ ﻧﮕﺎﻩ ﮐﻨﻴﺪ ﻭ ﺑﺒﻴﻨﻴﺪ ﮐﻪ ﺁﻳﺎ ﺗﻨﻔﺲ ﺩﺭ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺳﺒﺐ ﺑﺎﻻ ﻭ ﭘﺎﻳﻴﻦ ﺭﻓﺘﻦ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﻣﯽ‌ﺷﻮﺩ ﻳﺎ ﺧﻴﺮ.

ﺍﮔﺮ ﻣﺼﺪﻭﻡ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﺪ ﭼﻪ ﺑﻪ ﺻﻮﺭﺕ ﻧﺎﻟﻪ ﻭ ﭼﻪ ﺑﻪ ﺻﻮﺭﺕ ﺍﺩﺍﯼ ﺣﺮﻑ‌ﻫﺎﯼ ﺑﺮﻳﺪﻩ، ﻣﯽ‌ﺗﻮﺍﻥ ﻣﻄﻤﺌﻦ ﺷﺪ ﮐﻪ ﺭﺍﻩ ﻫﻮﺍﻳﯽ ﻭﯼ ﺑﺎﺯ ﺍﺳﺖ ﻭ ﺑﻪ ﺑﺮﺭﺳﯽ ﻣﺮﺣﻠﻪ ﺑﻌﺪ ﺭﻓﺖ. ﺍﮔﺮ ﺻﺪﺍﯼ ﻣﺼﺪﻭﻡ ﺑﻪ ﺻﻮﺭﺕ ﺧﺮﺧﺮ ﻳﺎ ﻏﻠﻐﻞ ﺷﻨﻴﺪﻩ ﻣﯽ‌ﺷﻮﺩ، ﺍﺣﺘﻤﺎﻻ ﺭﺍﻩ ﻣﺠﺎﺭﯼ ﺗﻨﻔﺴﯽ ﻭﯼ ﻧﻴﻤﻪ ﺑﺎﺯ ﺍﺳﺖ ﻭ ﻭﺭﻭﺩ ﻭ ﺧﺮﻭﺝ ﻫﻮﺍ ﺑﻪ ﺭﻳﻪ ﺑﻪ ﺧﻮﺑﯽ ﺻﻮﺭﺕ ﻧﻤﯽ‌ﮔﻴﺮﺩ.

ﺑﻪ ﺻﻮﺭﺕ ﻣﻌﻤﻮﻝ ﻓﺮﺩﯼ ﮐﻪ ﺩﭼﺎﺭ ﮐﻤﺒﻮﺩ ﺍﮐﺴﻴﮋﻥ ﻣﯽ‌ﺷﻮﺩ ﺭﻧﮓ ﺻﻮﺭﺕ، ﭘﻮﺳﺖ ﻳﺎ ﺯﻳﺮ ﻧﺎﺧﻦ ﻭﯼ ﺑﻪ ﺻﻮﺭﺕ ﺁﺑﯽ ﺧﺎﮐﺴﺘﺮﯼ ﺗﻐﻴﻴﺮ ﺭﻧﮓ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺍﻳﻦ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻧﺸﺎﻧﻪ ﺑﺮﺍﯼ ﻣﺘﻮﺟﻪ ﺷﺪﻥ ﺍﻧﺴﺪﺍﺩ ﺭﺍﻩ ﺗﻨﻔﺴﯽ ﻭﯼ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ.

ﺍﮔﺮ ﺑﻪ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﺭﺳﻴﺪﻳﺪ ﮐﻪ ﺭﺍﻩ ﻫﻮﺍﻳﯽ ﻣﺼﺪﻭﻡ ﺑﺴﺘﻪ ﺷﺪﻩ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﺑﺎﺯ ﮐﺮﺩﻥ ﺁﻥ ﺗﻼﺵ ﮐﻨﻴﺪ.

ﺑﺮﺍﯼ ﺣﺎﻟﺘﯽ ﮐﻪ ﺳﺮ ﻣﺼﺪﻭﻡ ﺑﻪ ﺟﻠﻮ ﺧﻢ ﺷﺪﻩ ﻭ ﺑﺎﻋﺖ ﺑﺴﺘﻪ ﺷﺪﻥ ﺭﺍﻩ ﻫﻮﺍﻳﯽ ﺷﺪﻩ، ﺑﺎﻳﺪ ﺳﺮ ﺭﺍ ﺑﻪ ﻋﻘﺐ ﺧﻢ ﮐﺮﺩﻩ ﻭ ﭼﺎﻧﻪ ﺍﻭ ﺭﺍ ﺑﺎﻻ ﺑﮑﺸﻴﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺳﺖ ﺧﻮﺩ ﺭﺍ ﺯﻳﺮ ﮔﺮﺩﻥ ﻣﺼﺪﻭﻡ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ﺁﻥ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﺑﺎﻻ ﺑﻴﺎﻭﺭﻳﺪ ﻭ ﺩﺳﺖ ﺩﻳﮕﺮ ﺭﺍ ﻧﻴﺰ ﺭﻭﯼ ﭘﻴﺸﺎﻧﯽ ﻣﺼﺪﻭﻡ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ﺑﻪ ﮐﻤﮏ ﺁﻥ ﺳﺮ ﻣﺼﺪﻭﻡ ﺭﺍ ﺑﻪ ﻋﻘﺐ ﺣﺮﮐﺖ ﺩﻫﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺳﺒﺐ ﮐﺸﻴﺪﮔﯽ ﺳﺮ ﻭ ﮔﺮﺩﻥ ﻣﺼﺪﻭﻡ ﺷﺪﻩ ﻭ ﺑﻪ ﺑﺎﺯﮔﺸﺎﻳﯽ ﺭﺍﻩ ﻫﻮﺍﻳﯽ ﺍﻭ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ.

ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﻣﺎﻧﻮﺭ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﻳﮏ ﺩﺳﺖ ﺍﻣﺪﺍﺩﮔﺮ ﺑﺮ ﺭﻭﯼ ﭘﻴﺸﺎﻧﯽ ﻣﺼﺪﻭﻡ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﻭ ﺍﻧﮕﺸﺘﺎﻥ ﺩﺳﺖ ﺩﻳﮕﺮ ﻭﯼ ﺯﻳﺮ ﻓﮏ ﻣﺼﺪﻭﻡ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﻭ ﺁﻥ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﺑﺎﻻ ﻓﺸﺎﺭ ﻣﯽ‌ﺩﻫﻨﺪ. ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺍﺟﺎﺯﻩ ﻧﺪﻫﻴﺪ ﺩﻫﺎﻥ ﻣﺼﺪﻭﻡ ﺑﺎﺯ ﺷﻮﺩ ﭼﺮﺍ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺳﺒﺐ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﺯﺑﺎﻥ ﻣﺼﺪﻭﻡ ﺩﺭ ﺣﻠﻖ ﻭﯼ ﺷﻮﺩ ﻭ ﺭﺍﻩ ﺗﻨﻔﺴﯽ ﺍﻭ ﺭﺍ ﻣﺴﺪﻭﺩ ﮐﻨﺪ. ﺍﻳﻦ ﻣﺎﻧﻮﺭ Head-tilt chin-lift Maneuver ﻧﺎﻡ ﺩﺍﺭﺩ. 

ﺍﮔﺮ ﺑﺮﺍﯼ ﻓﺮﺍﮔﻴﺮﯼ ﺍﻳﻦ ﻣﺎﻧﻮﺭ ﻧﻴﺎﺯ ﺑﻪ ﺩﻳﺪﻥ ﻧﺤﻮﻩ ﻋﻤﻠﮑﺮﺩ ﺷﺨﺺ ﺍﻣﺪﺍﺩﮔﺮ ﺩﺍﺭﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻳﻦ ﻭﻳﺪﻳﻮﯼ ۲۸ ﺛﺎﻧﻴﻪ‌ﺍﯼ ﺭﺍ ﺗﻤﺎﺷﺎ ﮐﻨﻴﺪ.

ﺍﮔﺮ ﺳﺮ ﻓﺮﺩ ﺭﻭ ﺑﻪ ﺑﺎﻻ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﻭ ﻫﻢ‌ﭼﻨﺎﻥ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻧﻔﺲ ﺑﮑﺸﺪ، ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﻓﺮﺍﻭﺍﻥ ﺯﺑﺎﻥ ﺩﺭ ﺣﻠﻘﺶ ﮔﻴﺮ ﮐﺮﺩﻩ ﺍﺳﺖ ﻭ ﻣﺴﺘﻘﻴﻢ ﺑﻪ ﻣﺮﺣﻠﻪ ﺭﻫﺎﻳﯽ ﺯﺑﺎﻥ ﺍﺯ ﺣﻠﻖ ﺑﺮﻭﻳﺪ.
