ﺩﺭﺱ ﭘﻨﺠﻢ- ﺑﻴﺮﻭﻥ ﺁﻭﺭﺩﻥ ﺟﺴﻢ ﺧﺎﺭﺟﯽ ﺍﺯ ﮔﻠﻮﯼ ﻣﺼﺪﻭﻡ

ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﺑﺎ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﺁﺷﻨﺎ ﺷﺪﻳﻢ ﻭ ﻧﺤﻮﻩ ﺍﻧﺠﺎﻡ ﺁﻥ ﺭﺍ ﻓﺮﺍ ﮔﺮﻓﺘﻴﻢ. ﺣﺘﻤﺎ ﺑﻪ ﻳﺎﺩ ﺩﺍﺭﻳﺪ ﮐﻪ ﮔﻔﺘﻴﻢ ﻫﻨﮕﺎﻡ ﻣﺸﺎﻫﺪﻩ ﻓﺮﺩ ﺑﻴﻬﻮﺷﯽ ﮐﻪ ﻧﺒﺾ ﻧﺪﺍﺭﺩ ﻭ ﻧﻔﺲ ﻧﻤﯽ‌ﮐﺸﺪ، ﺑﺎﻳﺪ ﺑﺪﻭﻥ ﺩﺭﻧﮓ ﺑﻪ ﺳﺮﺍﻍ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﺭﻓﺘﻪ ﻭ ﺑﻪ ﺻﻮﺭﺕ ﻣﻨﻈﻢ ۳۰ ﺑﺎﺭ ﺑﻪ ﻭﯼ ﻣﺎﺳﺎﮊ ﺑﺪﻫﻴﺪ. ﺳﭙﺲ ﺩﻭ ﺑﺎﺭ ﺗﻨﻔﺲ ﻣﺼﻨﻮﻋﯽ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﻭ ﺍﻳﻦ ﺳﻴﮑﻞ ﺭﺍ ﭼﻬﺎﺭ ﻣﺮﺗﺒﻪ ﺍﺩﺍﻣﻪ ﺩﻫﻴﺪ. ﺳﭙﺲ ﻋﻼﺋﻢ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺭﺍ ﭼﮏ ﮐﻨﻴﺪ. ﺍﮔﺮ ﮔﺮﺩﺵ ﺧﻮﻥ ﻫﻤﭽﻨﺎﻥ ﺩﺭ ﺑﺪﻥ ﺍﻭ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ، ﭼﺮﺧﻪ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﻭ ﺗﻨﻔﺲ ﻣﺼﻨﻮﻋﯽ ﺭﺍ ﺩﻭﺑﺎﺭﻩ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﺍﻣﺎ ﺍﮔﺮ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺍﻳﺠﺎﺩ ﺷﺪ، ﻫﺮ ﭘﻨﺞ ﺛﺎﻧﻴﻪ ﺑﻪ ﺍﻭ ﺗﻨﻔﺲ ﻣﺼﻨﻮﻋﯽ ﺑﺪﻫﻴﺪ ﻭ ﻫﺮ ﻳﮏ ﺩﻗﻴﻘﻪ ﻧﻴﺰ ﻋﻼﺋﻢ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﺑﺪﻧﺶ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﺪ. ﻭ ﺩﺭ ﺣﺎﻟﺖ ﺳﻮﻡ، ﺍﮔﺮ ﮔﺮﺩﺵ ﺧﻮﻥ ﻭ ﺗﻨﻔﺲ ﻣﺼﺪﻭﻡ ﭘﺲ ﺍﺯ ﺍﺟﺮﺍﯼ ﻋﻤﻞ ﺍﺣﻴﺎ ﺑﻪ ﺻﻮﺭﺕ ﻧﺮﻣﺎﻝ ﺑﺎﺯﮔﺸﺖ، ﺍﻭ ﺭﺍ ﺩﺭ ﺣﺎﻟﺖ ﺑﻬﺒﻮﺩﯼ ﻳﺎ ﺭﻳﮑﺎﻭﺭﯼ ﺑﺨﻮﺍﺑﺎﻧﻴﺪ ﻭ ﻫﺮ ﭼﻨﺪ ﺩﻗﻴﻘﻪ ﻭﺿﻌﻴﺖ ﮔﺮﺩﺵ ﺧﻮﻥ ﺍﻭ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﺪ.

ﺑﻴﺸﺘﺮ ﻣﺎﻧﻮﺭﻫﺎﻳﯽ ﮐﻪ ﺗﺎﮐﻨﻮﻥ ﻓﺮﺍ ﮔﺮﻓﺘﻪ‌ﺍﻳﻢ ﻣﻌﻤﻮﻻ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩ ﺑﻴﻬﻮﺵ ﺑﻪ ﮐﺎﺭ ﮔﺮﻓﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺑﺨﺸﯽ ﺍﺯ ﻋﻤﻠﻴﺎﺕ ﺭﻳﮑﺎﻭﺭﯼ ﻣﺼﺪﻭﻡ ﺑﻴﻬﻮﺵ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ. ﺍﻣﺎ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮔﺮﻳﺰﯼ ﺑﻪ ﻣﺎﻧﻮﺭ ﺩﻳﮕﺮﯼ ﺑﺰﻧﻴﻢ ﮐﻪ ﺩﺭ ﻭﺿﻌﻴﺖ ﻣﺴﺪﻭﺩ ﺷﺪﻥ ﻣﺠﺮﺍﯼ ﺗﻨﻔﺴﯽ ﻳﮏ ﻓﺮﺩ ﻫﻮﺷﻴﺎﺭ ﺩﺭ ﻫﻨﮕﺎﻡ ﻏﺬﺍ ﺧﻮﺭﺩﻥ ﺭﺥ ﻣﯽ‌ﺩﻫﺪ. ﺷﺎﻳﺪ ﺑﺮﺍﯼ ﺷﻤﺎ ﭘﻴﺶ ﺁﻣﺪﻩ ﮐﻪ ﻏﺬﺍ ﺩﺭ ﮔﻠﻮﯼ‌ﺗﺎﻥ ﮔﻴﺮ ﮐﻨﺪ، ﺑﺮﺧﯽ ﻣﻮﺍﻗﻊ ﺍﻳﻦ ﮔﻴﺮ ﮐﺮﺩﻥ ﻣﺠﺮﺍﯼ ﺗﻨﻔﺴﯽ ﻓﺮﺩ ﺭﺍ ﺑﺴﺘﻪ ﻭ ﺗﻨﻔﺲ ﺍﻭ ﺭﺍ ﻣﺨﺘﻞ ﻣﯽ‌ﮐﻨﺪ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺳﺎﺩﻩ ﺗﺮﻳﻦ ﺣﺮﮐﺘﯽ ﮐﻪ ﺑﻪ ﺫﻫﻦ ﻣﺎ ﻣﯽ‌ﺭﺳﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﻓﺮﻭ ﮐﺮﺩﻥ ﺍﻧﮕﺸﺘﺎﻥ ﺩﺭ ﮔﻠﻮ، ﺟﺴﻢ ﺧﺎﺭﺟﯽ ﺭﺍ ﺍﺯ ﮔﻠﻮ ﺧﺎﺭﺝ ﮐﻨﻴﻢ ﻳﺎ ﺭﺍﻩ ﺭﺍ ﺑﺮﺍﯼ ﻧﻔﺲ ﮐﺸﻴﺪﻥ ﺑﺎﺯ ﮐﻨﻴﻢ. ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺑﺮﺧﻼﻑ ﺗﺼﻮﺭ ﻋﻤﻮﻣﯽ، ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﺯ ﻓﺮﻭ ﮐﺮﺩﻥ ﺍﻧﮕﺸﺘﺎﻥ ﺑﻪ ﺩﻫﺎﻥ ﺣﻠﻖ ﻣﺼﺪﻭﻡ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﻢ ﭼﺮﺍ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺳﺒﺐ ﻓﺮﻭ ﺭﻓﺘﻦ ﺑﻴﺸﺘﺮ ﺟﺴﻢ ﺧﺎﺭﺟﯽ ﺩﺭ ﮔﻠﻮ ﺷﻮﺩ ﻭ ﺑﻴﺮﻭﻥ ﺁﻣﺪﻥ ﺁﻥ ﺭﺍ ﺳﺨﺖ‌ﺗﺮ ﮐﻨﺪ.

ﺍﻣﺎ ﺭﺍﻩ ﺣﻞ ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﭼﻴﺴﺖ؟ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺑﺮﺍﯼ ﮐﻤﮏ ﺑﻪ ﻣﺼﺪﻭﻡ ﺍﺯ ﺳﻪ ﺭﺍﻩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺳﺎﺩﻩ‌ﺗﺮﻳﻦ ﺍﻗﺪﺍﻡ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺑﺨﻮﺍﻫﻴﺪ ﺑﺎﻳﺴﺘﺪ ﻭ ﺑﺎ ﺳﺮﻓﻪ‌ﻫﺎﯼ ﻣﺪﺍﻭﻡ ﻭ ﻣﺤﮑﻢ، ﺟﺴﻢ ﺧﺎﺭﺟﯽ ﺭﺍ ﺍﺯ ﮔﻠﻮﯼ ﺧﻮﺩ ﺧﺎﺭﺝ ﮐﻨﺪ. ﺍﮔﺮ ﺟﺴﻢ ﺧﺎﺭﺟﯽ ﮐﻮﭼﮏ ﺑﻮﺩﻩ ﻳﺎ ﺩﺭ ﺍﺑﺘﺪﺍﯼ ﮔﻠﻮ ﮔﻴﺮ ﮐﺮﺩﻩ ﺑﺎﺷﺪ، ﺑﺎ ﺍﻳﻦ ﺭﻭﺵ ﻣﯽ‌ﺗﻮﺍﻥ ﻣﺠﺮﺍﯼ ﺗﻨﻔﺴﯽ ﻣﺼﺪﻭﻡ ﺭﺍ ﭘﺎﮐﺴﺎﺯﯼ ﮐﺮﺩ ﺗﺎ ﺑﻪ ﺻﻮﺭﺕ ﻃﺒﻴﻌﯽ ﻧﻔﺲ ﺑﮑﺸﺪ.

ﺍﮔﺮ ﺭﻭﺵ ﺳﺮﻓﻪ ﮐﺮﺩﻥ ﻣﺤﮑﻢ ﻭ ﻣﺘﻮﺍﻟﯽ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﻧﺪﺍﺩ، ﺑﻬﺘﺮ ﺍﺳﺖ ﻣﺼﺪﻭﻡ ﺭﺍ ﺩﺭ ﺣﺎﻟﺘﯽ ﮐﻪ ﺳﺎﻋﺪ ﻳﮏ ﺩﺳﺖ ﺭﺍ ﺟﻠﻮﯼ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺍﻭ ﻗﺮﺍﺭ ﺩﺍﺩﻩ، ﺑﻪ ﺟﻠﻮ ﺧﻢ ﮐﺮﺩﻩ ﻭ ﺑﺎ ﮐﻒ ﺩﺳﺖ ﺩﻳﮕﺮ ﺑﻪ ﭘﺸﺖ ﺍﻭ ﺿﺮﺑﻪ ﺑﺰﻧﻴﺪ. ﺍﺟﺮﺍﯼ ﺍﻳﻦ ﺭﻭﺵ ﺳﺎﺩﻩ ﺭﺍ ﺣﺘﻤﺎ ﺩﺭ ﮐﻤﮏ ﺑﻪ ﺗﻨﻔﺲ ﺷﻴﺮﺧﻮﺍﺭﺍﻥ ﻭ ﻧﻮﺯﺍﺩﺍﻥ ﺩﻳﺪﻩ‌ﺍﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺭﻭﺵ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺰﺭﮔﺴﺎﻻﻥ ﺭﺍ ﺩﺭ ﺣﺎﻟﺖ ﺍﻳﺴﺘﺎﺩﻩ ﻭ ﮐﻤﯽ ﻣﺎﻳﻞ ﻧﮕﺎﻩ ﺩﺍﺷﺖ ﺗﺎ ﻧﻴﺎﺯ ﺑﻪ ﺧﻢ ﮐﺮﺩﻥ ﺁﻥ‌ﻫﺎ ﻧﺒﺎﺷﺪ.

ﺍﮔﺮ ﺭﻭﺵ ﺿﺮﺑﻪ ﺯﺩﻥ ﺑﻪ ﭘﺸﺖ ﻧﻴﺰ ﺟﻮﺍﺏ ﻧﺪﺍﺩ، ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﺪ ﮐﻪ ﺟﺴﻢ ﺧﺎﺭﺟﯽ ﺑﺰﺭﮒ ﺑﻮﺩﻩ ﻳﺎ ﺑﻪ ﻣﻘﺪﺍﺭﯼ ﻧﺴﺒﺘﺎ ﺯﻳﺎﺩ ﺩﺭ ﻣﺠﺮﺍﯼ ﺗﻨﻔﺴﯽ ﻓﺮﺩ ﻓﺮﻭ ﺭﻓﺘﻪ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺑﺎﻳﺪ ﺑﯽ‌ﺩﺭﻧﮓ ﺑﻪ ﺳﺮﺍﻍ ﺭﻭﺵ ﺳﻮﻡ ﺭﻓﺘﻪ ﻭ ﺍﺯ ﻣﺎﻧﻮﺭ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﻓﺸﺎﺭ ﺑﻪ ﺷﮑﻢ ﻳﺎ «ﻫﺎﻳﻢ ﻟﻴﺶ» (Heimlich) ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺟﺴﻢ ﺧﺎﺭﺟﯽ ﺭﺍ ﺍﺯ ﻣﺠﺮﺍﯼ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺑﻴﺮﻭﻥ ﺑﻴﺎﻭﺭﻳﺪ. ﻣﺎﻧﻮﺭ ﻫﺎﻳﻢ ﻟﻴﺶ ﻣﺎﻧﻮﺭﯼ ﺳﺎﺩﻩ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺭﺍﺣﺘﯽ ﺗﻮﺳﻂ ﻳﮏ ﻓﺮﺩ ﻋﺎﺩﯼ ﻗﺎﺑﻞ ﺍﺟﺮﺍ ﺍﺳﺖ.


ﺩﺭ ﺍﻳﻦ ﻣﺎﻧﻮﺭ ﺑﺎﻳﺪ ﺍﺑﺘﺪﺍ ﭘﺸﺖ ﺳﺮ ﻣﺼﺪﻭﻡ ﻗﺮﺍﺭ ﺑﮕﻴﺮﻳﺪ. ﺳﭙﺲ ﻳﮑﯽ ﺍﺯ ﺩﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﻣﺸﺖ ﮐﺮﺩﻩ ﻭ ﺑﺎ ﺩﺳﺖ ﺩﻳﮕﺮ ﺩﺳﺖ ﻣﺸﺖ ﺷﺪﻩ ﺭﺍ ﺑﮕﻴﺮﻳﺪ ﻭ ﺗﺸﮑﻴﻞ ﻳﻪ ﺣﻠﻘﻪ ﺭﺍ ﺑﺪﻫﻴﺪ. ﻧﺤﻮﻩ ﻗﺮﺍﺭﮔﻴﺮﯼ ﺩﺳﺘﺎﻥ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﺭﻭﺵ ﺑﻪ ﮔﻮﻧﻪ‌‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺯﻳﺮ ﺑﻐﻞ ﻣﺼﺪﻭﻡ ﺭﺩ ﺷﺪﻩ ﺍﺳﺖ ﻳﻌﻨﯽ ﺩﺳﺘﺎﻥ ﻣﺼﺪﻭﻡ ﺑﺎﻻﯼ ﺣﻠﻘﻪ ﺍﻳﺠﺎﺩ ﺷﺪﻩ ﺗﻮﺳﻂ ﺩﺳﺘﺎﻥ ﺷﻤﺎ ﻗﺮﺍﺭ ﻣﻴﮕﻴﺮﺩ. ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﻣﺤﻞ ﻗﺮﺍﺭ ﮔﻴﺮﯼ ﻣﺮﮐﺰ ﺣﻠﻘﻪ ﺍﻳﺠﺎﺩ ﺷﺪﻩ ﺗﻮﺳﻂ ﺩﺳﺘﺎﻥ ﺷﻤﺎ ﭘﺎﻳﻴﻦ ﺗﺮ ﺍﺯ ﺯﺍﺋﺪﻩ ﺧﻨﺠﺮﯼ ﺟﻨﺎﻍ ﺳﻴﻨﻪ ﻓﺮﺩ ﻣﺼﺪﻭﻡ ﺑﺎﺷﺪ.

 

ﻋﮑﺲ ﻳﮏ



ﭘﺲ ﺍﺯ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﺻﺤﻴﺢ ﺩﺳﺖ، ﺩﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﺩﺍﺧﻞ ﻭ ﺑﺎﻻ ﻓﺸﺎﺭ ﺩﻫﻴﺪ. ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﻋﻤﻞ ﺑﺎﻳﺪ ﺑﻪ ﺻﻮﺭﺕ ﻓﺸﺎﺭ ﺑﺎﺷﺪ ﻭ ﺍﺯ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﺿﺮﺑﻪ ﻧﺎﮔﻬﺎﻧﯽ ﺑﻪ ﺑﺪﻥ ﻣﺼﺪﻭﻡ ﺑﺎﻳﺪ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺍﺟﺮﺍﯼ ﺍﻳﻦ ﻋﻤﻞ ﺩﺭ ﺑﺎﺭ ﺍﻭﻝ ﺳﻮﺩﻣﻨﺪ ﻧﺒﻮﺩ، ﭼﻨﺪ ﺑﺎﺭ ﺩﻳﮕﺮ ﺍﻳﻦ ﻣﺎﻧﻮﺭ ﺭﺍ ﺗﮑﺮﺍﺭ ﮐﻨﻴﺪ ﺗﺎ ﺑﻪ ﻧﺘﻴﺠﻪ ﺑﺮﺳﻴﺪ. ﺍﮔﺮ ﺍﻳﻦ ﺭﻭﺵ ﻧﻴﺰ ﺑﻪ ﻧﺘﻴﺠﻪ ﻧﺮﺳﻴﺪ ﻭ ﻣﺼﺪﻭﻡ ﺩﻳﮕﺮ ﺑﻪ ﻣﺤﺮﮎ ﻭﺍﮐﻨﺶ ﻧﺸﺎﻥ ﻧﻤﯽ‌ﺩﻫﺪ، ﺑﺎ ﺍﻭﺭﮊﺍﻧﺲ ﺗﻤﺎﺱ ﺑﮕﻴﺮﻳﺪ ﻭ ﺑﻪ ﻋﻤﻠﻴﺎﺕ ﺍﺣﻴﺎ ﺑﭙﺮﺩﺍﺯﻳﺪ.
 

ﻋﮑﺲ ﺩﻭ
 


ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﻣﺎﻧﻮﺭ ﻫﺎﻳﻢ ﻟﻴﺶ، ﺩﺭ ﮐﻮﺩﮐﺎﻥ ﺯﻳﺮ ﻳﮑﺴﺎﻝ ﺑﻪ ﺩﻟﻴﻞ ﺍﺣﺘﻤﺎﻝ ﺑﺎﻻﯼ ﺁﺳﻴﺐ ﺩﻳﺪﮔﯽ ﺷﮑﻢ ﺗﻮﺻﻴﻪ ﻧﻤﯽ‌ﺷﻮﺩ ﻭ ﺩﺭ ﺯﻧﺎﻥ ﺑﺎﺭﺩﺍﺭ ﻳﺎ ﺍﻓﺮﺍﺩ ﭼﺎﻕ ﻧﻴﺰ ﺗﻮﺻﻴﻪ ﺷﺪﻩ ﻓﺸﺎﺭ ﺑﻪ ﺟﺎﯼ ﺷﮑﻢ، ﺑﻪ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﻭﺍﺭﺩ ﺷﻮﺩ.
 

ﻋﮑﺲ ﺳﻪ
 


ﺑﺮﺍﯼ ﻳﺎﺩﮔﻴﺮﯼ ﺑﻬﺘﺮ ﻣﺎﻧﻮﺭ ﻫﺎﻳﻢ ﻟﻴﺶ ﺗﻮﺻﻴﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﺍﻳﻦ ﻭﻳﺪﻳﻮﯼ ﺁﻣﻮﺯﺷﯽ ﺭﺍ ﺑﻪ ﻣﺪﺕ ۱۴۰ ﺛﺎﻧﻴﻪ ﻣﺸﺎﻫﺪﻩ ﮐﻨﻴﺪ.

ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﻢ
ﺑﺮﺍﯼ ﭘﺎﮐﺴﺎﺯﯼ ﻣﺠﺮﺍﯼ ﺗﻨﻔﺴﯽ ﻳﮏ ﻓﺮﺩ ﻫﻮﺷﻴﺎﺭ ﺳﻪ ﺭﺍﻩ ﺣﻞ ﺩﺭ ﭘﻴﺶ ﺭﻭ ﺩﺍﺭﻳﻢ:
۱- ﺳﺮﻓﻪ ﮐﺮﺩﻥ ﻣﺤﮑﻢ ﻭ ﻣﺘﻮﺍﻟﯽ
۲- ﺿﺮﺑﻪ ﺯﺩﻥ ﺑﻪ ﭘﺸﺖ ﻣﺼﺪﻭﻡ
۳- ﻣﺎﻧﻮﺭ ﻫﺎﻳﻢ ﻟﻴﺶ
 

ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﻪ ﻧﻮﺯﺍﺩﺍﻥ
ﺗﺎ ﮐﻨﻮﻥ ﺭﻭﺵ‌ﻫﺎﯼ ﮔﻮﻧﺎﮐﻮﻥ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﺮﺍﯼ ﻧﺠﺎﺕ ﻳﮏ ﻣﺼﺪﻭﻡ ﺑﺰﺭﮔﺴﺎﻝ ﺭﺍ ﻓﺮﺍ ﮔﺮﻓﺘﻴﻢ، ﺍﻣﺎ ﺩﺭ ﺣﻴﻦ ﻳﺎﺩﮔﻴﺮﯼ ﺑﺮﺧﯽ ﺍﺯ ﺭﻭﺵ‌ﻫﺎ ﻧﻴﺰ ﮔﻔﺘﻴﻢ ﮐﻪ ﺍﻳﻦ ﺭﻭﺵ ﺑﺮﺍﯼ ﮐﻮﺩﮐﺎﻥ ﺑﺎﻳﺪ ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﺩﻳﮕﺮ ﺍﺟﺮﺍ ﺷﻮﺩ. ﺩﺭ ﺑﺮﺧﯽ ﺍﻭﻗﺎﺕ ﻣﺼﺪﻭﻡ ﻣﺎ ﺑﻪ ﺟﺎﯼ ﻳﮏ ﻓﺮﺩ ﺑﺰﺭﮔﺴﺎﻝ، ﻳﮏ ﻧﻮﺯﺍﺩ ﺍﺳﺖ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺩﺍﺭﺩ. ﻣﺜﻠﺂ ﻓﺮﺽ ﮐﻨﻴﺪ ﮐﻪ ﺑﻪ ﻧﻮﺯﺍﺩﯼ ﺑﺮﺧﻮﺭﺩﻳﺪ ﮐﻪ ﺑﻪ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﻣﺤﻴﻄﯽ ﭘﺎﺳﺦ ﻧﻤﯽ‌ﺩﻫﺪ ﻭ ﺗﻨﻔﺲ ﻧﺪﺍﺭﺩ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﭼﻪ ﺑﺎﻳﺪ ﺑﮑﻨﻴﻢ؟

ﺗﺮﺗﻴﺐ ﮐﺎﺭﻫﺎﻳﯽ ﮐﻪ ﻗﺮﺍﺭ ﺍﺳﺖ ﺩﺭ ﺷﺮﺍﻳﻂ ﻣﺸﺎﺑﻪ ﺑﺮﺍﯼ ﻣﺼﺪﻭﻣﺎﻥ ﻧﻮﺯﺍﺩ ﻳﺎ ﺍﻓﺮﺍﺩ ﺑﺰﺭﮔﺴﺎﻝ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ ﻳﮑﺴﺎﻥ ﺍﺳﺖ، ﻳﻌﻨﯽ ﻭﻗﺘﯽ ﺑﻪ ﻧﻮﺯﺍﺩﯼ ﺑﺮﺧﻮﺭﺩﻳﺪ ﮐﻪ ﻧﺒﺾ ﻧﺪﺍﺭﺩ ﻭ ﺑﻪ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﻣﺤﻴﻄﯽ ﭘﺎﺳﺦ ﻧﻤﯽ‌ﺩﻫﺪ ﺑﺎﻳﺪ ﺑﯽ‌ﺩﺭﻧﮓ ﻋﻤﻞ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﺭﺍ ﺁﻏﺎﺯ ﮐﻨﻴﺪ ﺍﻣﺎ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﺩﺭ ﻧﻮﺯﺍﺩﺍﻥ ﭼﮕﻮﻧﻪ ﺍﺳﺖ؟

ﺑﺮﺍﯼ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﺩﺭ ﻧﻮﺯﺍﺩﺍﻥ، ﺍﺑﺘﺪﺍ ﺑﺎﻳﺪ ﻣﺤﻞ ﻓﺸﺎﺭ ﻣﺎﺳﺎﮊ ﺭﺍ ﻣﺸﺨﺺ ﮐﺮﺩ. ﻣﺤﻞ ﻭﺍﺭﺩ ﺁﻣﺪﻥ ﻓﺸﺎﺭ ﺩﺭ ﺭﻭﯼ ﺟﻨﺎﻍ ﻧﻮﺯﺍﺩ ﻭ ﻳﮏ ﺑﻨﺪ ﺍﻧﮕﺸﺖ ﭘﺎﻳﻴﻦ‌ﺗﺮ ﺍﺯ ﺧﻄﯽ ﻓﺮﺿﯽ ﺍﺳﺖ ﮐﻪ ﻧﻮﮎ ﺳﻴﻨﻪ ﻧﻮﺯﺍﺩ ﺭﺍ ﺑﻪ ﻫﻢ ﻣﺘﺼﻞ ﻣﯽ‌ﮐﻨﺪ. ﺩﻭ ﺍﻧﮕﺸﺖ ﺧﻮﺩ ﺭﺍ ﺑﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﻧﻘﻄﻪ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ۳۰ ﻣﺮﺗﺒﻪ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺭﺍ ﻓﺸﺎﺭ ﺩﻫﻴﺪ. ﺑﺮ ﺍﺳﺎﺱ ﺩﺳﺘﻮﺭﺍﻟﻌﻤﻞ‌ﻫﺎﯼ ﺟﺪﻳﺪ ﻋﻤﻖ ﻓﺸﺎﺭ ﺑﺮﺍﯼ ﻧﻮﺯﺍﺩﺍﻥ ﮐﻤﺘﺮ ﺍﺯ ﻳﮑﺴﺎﻝ ﺣﺪﻭﺩ ﭼﻬﺎﺭ ﺳﺎﻧﺘﯽ‌ﻣﺘﺮ ﻭ ﺑﺮﺍﯼ ﮐﻮﺩﮐﺎﻥ ﻳﮏ ﺗﺎ ﻫﺸﺖ ﺳﺎﻝ ﺣﺪﻭﺩ ﭘﻨﺞ ﺳﺎﻧﺘﯽ‌ﻣﺘﺮ ﺍﺳﺖ.

ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﻓﺸﺎﺭ ﻧﺒﺎﻳﺪ ﺑﻪ ﺿﺮﺑﻪ ﺗﺒﺪﻳﻞ ﺷﻮﺩ ﻭ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﺗﻌﺪﺍﺩ ﺩﻗﻴﻖ ﺁﻥ ﻧﻴﺰ ﺩﺭ ﺣﻴﻦ ﺍﺟﺮﺍ، ﻓﺸﺎﺭﻫﺎ ﺭﺍ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﺑﺸﻤﺎﺭﻳﺪ. ﭘﺲ ﺍﺯ ﺍﺟﺮﺍﯼ ۳۰ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ، ﺑﻪ ﻧﻮﺯﺍﺩﺍﻥ ﺑﻪ ﺁﺭﺍﻣﯽ ﺩﻭ ﺑﺎﺭ ﺑﺎﻳﺪ ﺗﻨﻔﺲ ﻣﺼﻨﻮﻋﯽ ﺑﺪﻫﻴﺪ. ﺍﻳﻦ ﺳﻴﮑﻞ ﻫﻤﺎﻧﻨﺪ ﺳﻴﮑﻞ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﻭ ﺗﻨﻔﺲ ﻣﺼﻨﻮﻋﯽ ﺩﺭ ﺑﺰﺭﮔﺴﺎﻻﻥ ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﺑﺮ ﺧﻼﻑ ﺑﺰﺭﮔﺴﺎﻻﻥ ﮐﻪ ﺍﻳﻦ ﺳﻴﮑﻞ ﺑﺮﺍﯼ ۴ ﻣﺮﺗﺒﻪ ﺗﮑﺮﺍﺭ ﻣﯽ‌ﺷﺪ، ﺩﺭ ﻧﻮﺯﺍﺭﺩﺍﻥ ﺍﻳﻦ ﺳﻴﮑﻞ ﺭﺍ ﺑﺎﻳﺪ ﭘﻨﺞ ﻣﺮﺗﺒﻪ ﺗﮑﺮﺍﺭ ﮐﺮﺩ ﻭ ﺳﭙﺲ ﻋﻼﺋﻢ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﻧﻮﺯﺍﺩ ﻣﺼﺪﻭﻡ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﺮﺩ. ﺍﮔﺮ ﻣﺸﺎﻫﺪﻩ ﮐﺮﺩﻳﺪ ﮐﻪ ﮔﺮﺩﺵ ﺧﻮﻥ ﻫﻤﭽﻨﺎﻥ ﺩﺭ ﺑﺪﻥ ﺍﻭ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ، ﭼﺮﺧﻪ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﻭ ﺗﻨﻔﺲ ﻣﺼﻨﻮﻋﯽ ﺭﺍ ﺑﺎﻳﺪ ﺩﻭﺑﺎﺭﻩ ﺍﺯ ﺳﺮ ﺑﮕﻴﺮﻳﺪ ﺍﻣﺎ ﺍﮔﺮ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﺑﺪﻥ ﺍﺯ ﺳﺮ ﮔﺮﻓﺘﻪ ﺷﺪ، ﻫﺮ ﺳﻪ ﺛﺎﻧﻴﻪ ﺑﻪ ﺍﻭ ﺗﻨﻔﺲ ﻣﺼﻨﻮﻋﯽ ﺑﺪﻫﻴﺪ ﻭ ﻫﺮ ﭼﻨﺪ ﺩﻗﻴﻘﻪ ﻧﻴﺰ ﻋﻼﺋﻢ ﮔﺮﺩﺵ ﺧﻮﻥ ﺩﺭ ﺑﺪﻧﺶ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﺪ.

ﺩﺭ ﺑﺮﺧﯽ ﺍﻭﻗﺎﺕ ﻣﻤﮑﻦ ﺍﺳﺖ ﺟﺴﻢ ﺧﺎﺭﺟﯽ ﺩﺭ ﮔﻠﻮﯼ ﻧﻮﺯﺍﺩ ﮔﻴﺮ ﮐﺮﺩﻩ ﻭ ﺍﻭ ﺗﻮﺍﻧﺎﻳﯽ ﺗﻨﻔﺲ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺑﺪﻫﺪ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺑﺎﻳﺪ ﻧﻮﺯﺍﺩ ﺭﺍ ﺭﻭ ﺑﻪ ﺯﻣﻴﻦ ﻭ ﺩﺭ ﺣﺎﻟﺖ ﻣﺎﻳﻞ ﻧﮕﺎﻩ ﺩﺍﺷﺖ ﻳﻌﻨﯽ ﺳﺮ ﻭ ﮔﺮﺩﻥ ﺍﻭ ﭘﺎﻳﻴﻦ ﺗﺮ ﺍﺯ ﻗﻔﺴﻪ ﺳﻴﻨﻪ‌ﺍﺵ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻳﺪ ﺑﺎ ﻳﮏ ﺩﺳﺖ، ﻗﺴﻤﺖ ﻓﻮﻗﺎﻧﯽ ﺑﺪﻥ ﻭﯼ، ﻣﺤﻞ ﺳﺮ ﻭ ﮔﺮﺩﻥ ﺭﺍ ﮔﺮﻓﺘﻪ ﻭ ﺑﺪﻥ ﺍﻭ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺳﺎﻋﺪ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ ﻭ ﺑﺎ ﺩﺳﺖ ﺩﻳﮕﺮ ﻧﻴﺰ ﺑﻪ ﻣﻴﺎﻥ ﺩﻭ ﮐﺘﻒ ﻧﻮﺯﺍﺩ ﭘﻨﺞ ﺿﺮﺑﻪ ﺑﺰﻧﻴﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﺿﺮﺑﻪ‌ﻫﺎ ﺑﺎ ﻓﺎﺻﻠﻪ ﺍﺯ ﻫﻢ ﺯﺩﻩ ﺷﻮﻧﺪ.

ﺍﻟﺒﺘﻪ ﺍﺯ ﺭﻭﺵ ﻓﺸﺮﺩﻥ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﺑﻪ ﻧﻮﺯﺍﺩ ﻣﺼﺪﻭﻡ ﮐﻤﮏ ﮐﺮﺩ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ، ﻧﻮﺯﺍﺭﺩ ﺭﺍ ﺑﺎ ﻳﮏ ﺩﺳﺖ ﻃﻮﺭﯼ ﺑﺎﻳﺪ ﻧﮕﻪ ﺩﺍﺭﻳﺪ ﮐﻪ ﺳﺮ ﺷﻴﺮﺧﻮﺍﺭ ﺑﻪ ﺳﻤﺖ ﺁﺳﻤﺎﻥ ﺑﻮﺩﻩ ﻭ ﭘﺎﻳﻴﻦ‌ﺗﺮ ﺍﺯ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺍﻭ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ. ﺳﭙﺲ ﺩﺭ ﻫﻤﺎﻥ ﻧﻘﻄﻪ‌ﺍﯼ ﮐﻪ ﻣﺤﻞ ﻓﺸﺎﺭ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﺑﻮﺩﻩ (ﻳﮏ ﺑﻨﺪ ﺍﻧﮕﺸﺖ ﭘﺎﻳﻴﻦ‌ﺗﺮ ﺍﺯ ﺧﻂ ﻓﺮﺿﯽ ﻣﺘﺼﻞ ﮐﻨﻨﺪﻩ ﻧﻮﮎ ﺳﻴﻨﻪ‌ﻫﺎﯼ ﻧﻮﺯﺍﺩ ﻭ ﺑﺮ ﺭﻭﯼ ﺟﻨﺎﻍ ﺳﻴﻨﻪ‌ﯼ ﻭﯼ) ﺑﺎ ﺩﻭ ﺍﻧﮕﺸﺖ ﭘﻨﺞ ﺑﺎﺭ ﻓﺸﺎﺭ ﻭﺍﺭﺩ ﮐﻨﻴﺪ.

ﺍﮔﺮ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ ﺩﺭ ﺯﻣﻴﻨﻪ ﮐﻤﮏ ﺑﻪ ﻧﻮﺯﺍﺭﺩﺍﻥ ﺑﻴﺸﺘﺮ ﺑﻴﺎﻣﻮﺯﻳﺪ ﻭ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺍ ﻣﺮﻭﺭ ﮐﻨﻴﺪ، ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﻴﻢ ﺍﻳﻦ ﻭﻳﺪﻳﻮﯼ ﺁﻣﻮﺯﺷﯽ  ﺭﺍ ﺑﻪ ﻣﺪﺕ ﺣﺪﻭﺩ ۱۵ ﺩﻗﻴﻘﻪ ﻣﺸﺎﻫﺪﻩ ﮐﻨﻴﺪ.

ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ، ﺩﺭ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﻣﺎﻧﻮﺭﻫﺎﻳﯽ ﺑﺮﺍﯼ ﭘﺎﮐﺴﺎﺯﯼ ﻣﺠﺮﺍﯼ ﺗﻨﻔﺴﯽ ﻳﮏ ﻓﺮﺩ ﻫﻮﺷﻴﺎﺭ ﺭﺍ ﻳﺎﺩ ﮔﺮﻓﺘﻴﻢ ﮐﻪ ﻣﺎﻧﻮﺭ ﻫﺎﻳﻢ ﻟﻴﺶ ﻧﻴﺰ ﺩﺭ ﺍﻳﻦ ﺑﻴﻦ ﻗﺮﺍﺭ ﺩﺍﺷﺖ. ﺳﭙﺲ ﺗﺎ ﺍﻧﺪﺍﺯﻩ‌ﺍﯼ ﺑﻪ ﻣﺒﺤﺚ ﮐﻤﮏ‌ﻫﺎﯼ ﺍﻭﻟﻴﻪ ﺑﻪ ﻧﻮﺯﺍﺩﺍﻥ ﭘﺮﺩﺍﺧﺘﻴﻢ ﻭ ﻣﺎﺳﺎﮊ ﻗﻠﺒﯽ ﻭ ﻣﺎﻧﻮﺭ ﻫﺎﻳﻢ ﻟﻴﺶ ﺑﻪ ﺍﻳﻦ ﮔﺮﻭﻩ ﺭﺍ ﻧﻴﺰ ﻣﺮﻭﺭ ﮐﺮﺩﻳﻢ. ﺩﺭ ﺟﻠﺴﻪ ﺑﻌﺪ ﺑﻪ ﻣﺼﺪﻭﻣﺎﻥ ﺩﺭ ﺣﺎﻝ ﺧﻮﻧﺮﻳﺰﯼ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ ﻭ ﻧﺤﻮﻩ ﮐﻤﮏ ﮐﺮﺩﻥ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﺮﻭﺭ ﻣﯽ‌ﮐﻨﻴﻢ.
