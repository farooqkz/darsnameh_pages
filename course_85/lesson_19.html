ﺩﺭﺱ ﺩﻫﻢ- ﺍﺑﺰﺍﺭ ﻗﻠﻢ ﻣﻮ

ﺍﺑﺰﺍﺭ ﻗﻠﻢ ﻣﻮ ﻳﺎ Brush، ﻳﮑﯽ ﺍﺯ ﭘﺮﮐﺎﺭﺑﺮﺩ‌ﺗﺮﻳﻦ ﺍﺑﺰﺍﺭﻫﺎﯼ ﻓﺘﻮﺷﺎﭖ ﮐﻪ ﺑﻪ ﺩﻟﻴﻞ ﺳﺎﺩﮔﯽ ﮐﺎﺭ ﻭ ﮔﺴﺘﺮﺩﮔﯽ ﺗﻨﻈﻴﻤﺎﺕ، ﻣﺒﻨﺎﯼ ﮐﺎﺭ ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﺑﺰﺍﺭ‌ﻫﺎ ﻭ ﺟﻠﻮﻩ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﻓﺘﻮﺷﺎﭘﯽ ﺍﺳﺖ، ﺍﺑﺰﺍﺭ ﺑﺮﺍﺵ ﺍﺳﺖ.

ﺍﺑﺰﺍﺭ ﺑﺮﺍﺵ ﺩﺭ ﻣﻴﻠﻪ ﺍﺑﺰﺍﺭ (ﺷﮑﻞ ﺯﻳﺮ) ﻗﺎﺑﻞ ﺩﺳﺘﺮﺳﯽ ﺍﺳﺖ ﻭ ﺑﺎ ﮐﻠﻴﺪ ﻣﻴﺎﻧﺒﺮ B ﻧﻴﺰ ﻓﻌﺎﻝ ﻣﯽ‌ﮔﺮﺩﺩ.

‌‌ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﺷﻮﺩ، ﺑﺮﺍﺵ، ﺍﻭﻟﻴﻦ ﺍﺑﺰﺍﺭ ﺩﺭ ﻳﮏ ﺟﻌﺒﻪ ﺍﺑﺰﺍﺭ ۴ ﺗﺎﻳﯽ ﺍﺳﺖ. ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﺁﻥ، ﻧﻤﺎﻳﺸﮕﺮ ﻣﻮﺱ، ﺷﮑﻞ ﺳﺮ ﻗﻠﻢ ﻣﻮ ﺭﺍ ﺑﻪ ﺧﻮﺩ ﻣﯽ‌ﮔﻴﺮﺩ. ﺣﺎﻝ ﺍﮔﺮ ﻣﻮﺱ ﺭﺍ ﺭﻭﯼ ﺗﺼﻮﻳﺮ ﺩﺭﮒ ﮐﻨﻴﻢ، ﺭﻧﮕﯽ ﮐﻪ ﺩﺭ Foreground Color ﺍﻧﺘﺨﺎﺏ ﺷﺪﻩ، ﺭﻭﯼ ﺗﺼﻮﻳﺮ ﮐﺸﻴﺪﻩ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺷﮑﻞ ﺯﻳﺮ ﺑﻪ ﻣﺮﺑﻊ ﺭﻧﮕﯽ ﺑﺎﻻﻳﯽ، Foreground Color ﻭ ﺑﻪ ﻣﺮﺑﻊ ﺭﻧﮕﯽ ﺯﻳﺮﻳﻦ، Background Color ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺷﮑﻞ ﺯﻳﺮ، Foreground Color، ﺯﺭﺩ ﻭ Background Color، ﺳﻴﺎﻩ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﺗﻐﻴﻴﺮ ﺁﻥ‌ﻫﺎ، ﮐﺎﻓﯽ ﺍﺳﺖ ﮐﻪ ﺭﻭﯼ ﻫﺮ ﻳﮏ ﮐﻠﻴﮏ ﮐﺮﺩﻩ ﻭ ﺍﺯ ﭘﻨﺠﺮﻩ ﺍﻧﺘﺨﺎﺏ ﺭﻧﮓ، ﺭﻧﮓ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﻢ. ﺩﺭ ﺩﺭﺱ ﺣﺎﺿﺮ، ﺑﻌﻀﺎ Foreground Color ﺭﺍ ﺭﻧﮓ ﭘﻴﺶ ﺯﻣﻴﻨﻪ ﻭ Background Color ﺭﺍ ﺭﻧﮓ ﭘﺲ ﺯﻣﻴﻨﻪ ﻣﯽ‌ﻧﺎﻣﻴﻢ.)

ﺯﻣﺎﻧﯽ ﮐﻪ ﺍﺑﺰﺍﺭ ﺑﺮﺍﺵ ﻓﻌﺎﻝ ﺍﺳﺖ، ﺍﮔﺮ ﺭﻭﯼ ﺗﺼﻮﻳﺮ، ﺭﺍﺳﺖ ﮐﻠﻴﮏ ﮐﻨﻴﻢ، ﭘﻨﺠﺮﻩ ﻣﺤﺎﻭﺭﻩ‌ﺍﯼ ﻣﻄﺎﺑﻖ ﺷﮑﻞ ﺯﻳﺮ، ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺍﻣﮑﺎﻧﺎﺗﯽ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﮐﺎﺭﺑﺮ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﺪ.

ﺍﻭﻟﻴﻦ ﺍﺳﻼﻳﺪﺭ، ﺩﺭ ﭘﻨﺠﺮﻩ ﻓﻮﻕ، ﺑﺎ ﻋﻨﻮﺍﻥ Size، ﻗﻄﺮ ﻗﻠﻢ ﺭﺍ ﺑﺎ ﻭﺍﺣﺪ ﭘﻴﮑﺴﻞ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﺪ. ﻫﺮﭼﻪ ﻗﻄﺮ ﻗﻠﻢ ﺑﺰﺭﮒ‌ﺗﺮ، ﻗﻄﺮ ﻧﻤﺎﻳﺸﮕﺮ ﻣﻮﺱ، ﮐﻪ ﻗﻄﺮ ﻗﻠﻢ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﺑﺰﺭﮒ‌ﺗﺮ ﻣﯽ‌ﺷﻮﺩ. ﻗﻄﺮ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﻋﺪﺩﯼ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﺑﺨﺶ ﻣﺮﺑﻮﻃﻪ ﺩﺭ ﭘﻨﺠﺮﻩ ﻭﺍﺭﺩ ﮐﺮﺩ. ﻫﻤﭽﻨﻴﻦ ﺑﺮﺍﯼ ﺗﺴﺮﻳﻊ ﻓﺮﺍﻳﻨﺪ ﺗﻨﻈﻴﻢ ﺳﺎﻳﺰ، ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﺩﻭ ﮐﻠﻴﺪ ﻣﯽ‌ﺍﻧﺒﺮ ﺑﺮﺍﮐﺖ ([ﻳﺎ]) ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﻮﺩ.

ﺍﺳﻼﻳﺪﺭ ﺑﻌﺪﯼ Hardness ﻗﻠﻢ ﺍﺳﺖ ﮐﻪ ﺗﺮﺟﻤﻪ ﻟﻐﺖ ﺑﻪ ﻟﻐﺖ ﺁﻥ ﺳﺨﺘﯽ ﻗﻠﻢ ﺍﺳﺖ. ﻫﺮﭼﻪ Hardness ﭘﺎﻳﻴﻦ‌ﺗﺮ ﺑﺎﺷﺪ، ﻟﺒﻪ‌ﻫﺎﯼ ﻗﻠﻢ ﻣﺤﻮ‌ﺗﺮ ﺍﺳﺖ.

ﺍﮔﺮ Hardness ﺩﺭ ﺑﺎﻻ‌ﺗﺮﻳﻦ ﺣﺎﻟﺖ ﺑﺎﺷﺪ، ﻟﺒﻪ‌ﻫﺎﯼ ﺧﻂ ﺭﺳﻢ ﺷﺪﻩ ﺑﺎ ﻗﻠﻢ، ﺷﺎﺭﭖ‌ﺗﺮ ﻭ ﻣﺸﺨﺺ‌ﺗﺮ ﺍﺳﺖ.

ﻻﺯﻡ ﺑﻪ ﺗﺬﮐﺮ ﺍﺳﺖ ﮐﻪ ﺳﺎﻳﺰ ﻗﻠﻢ ﺩﺭ ﻫﺮ ﺩﻭ ﺷﮑﻞ ﺑﺎﻻ ﻳﮑﺴﺎﻥ ﺍﺳﺖ.

ﺩﺭ ﺑﺨﺶ ﭘﺎﻳﻴﻨﯽ ﭘﻨﺠﺮﻩ ﻗﻠﻢ، ﻗﻠﻢ ﻣﻮﻫﺎﻳﯽ ﺑﺎ ﺷﮑﻠﻬﺎﯼ ﻣﺨﺘﻠﻒ ﺑﺼﻮﺭﺕ ﭘﻴﺶ ﻓﺮﺽ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ ﮐﻪ ﮐﺎﺭﺑﺮ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﻧﻴﺎﺯ ﺧﻮﺩ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﺪ.

ﺗﻨﻈﻴﻤﺎﺕ ﺍﺑﺰﺍﺭ ﺑﺮﺍﺵ ﺗﻨﻬﺎ ﺑﻪ ﻫﻤﻴﻦ ﭘﻨﺠﺮﻩ ﺧﺘﻢ ﻧﻤﯽ‌ﺷﻮﺩ. ﺑﻪ ﻟﺤﺎﻅ ﺍﻫﻤﻴﺖ ﺍﻳﻦ ﺍﺑﺰﺍﺭ، ﻳﮏ ﭘﻨﻞ ﻣﺠﺰﺍ ﺑﺎ ﻧﺎﻡ Brush، ﻣﺨﺘﺺ ﺑﻪ ﻗﻠﻢ ﻣﻮ‌ﻫﺎ ﺩﺭ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻓﺘﻮﺷﺎﭖ ﻃﺮﺍﺣﯽ ﺷﺪﻩ ﺍﺳﺖ. ﺍﻳﻦ ﭘﻨﻞ ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﭘﻨﻞ‌ﻫﺎﯼ ﺩﻳﮕﺮ، ﺩﺭ ﻣﻨﻮﯼ Window ﻣﻮﺟﻮﺩ ﺍﺳﺖ.

‌‌ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺷﮑﻞ ﻓﻮﻕ ﻧﻴﺰ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﺷﻮﺩ، ﻣﻌﻤﻮﻻ ﺍﻳﻦ ﭘﻨﻞ ﺩﺭ ﺳﺘﻮﻥ ﮐﻨﺎﺭ ﭘﻨﻞ‌ﻫﺎ ﻧﻴﺰ ﻳﺎﻓﺖ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺍﻳﻦ ﭘﻨﻞ، ﮐﺎﺭﺑﺮ ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﻣﻴﺎﺑﺪ ﮐﻪ ﺧﻮﺩ، ﺍﻗﺪﺍﻡ ﺑﻪ ﻃﺮﺍﺣﯽ ﻗﻠﻢ ﻣﻮ ﻧﻤﺎﻳﺪ. ﺳﻌﯽ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ ﺩﺭ ﺩﻭﺭﻩ‌ﻫﺎﯼ ﺁﺗﯽ، ﺍﻳﻦ ﭘﻨﻞ ﮔﺴﺘﺮﺩﻩ ﺭﺍ ﻣﻮﺭﺩ ﺑﺤﺚ ﻗﺮﺍﺭ ﺩﻫﻴﻢ.

ﺩﺭ Option Bar ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﺑﺰﺍﺭ Brush ﻣﺎﻧﻨﺪ ﺩﻳﮕﺮ ﺍﺑﺰﺍﺭ‌ﻫﺎ، ﺩﮔﻤﻪ ﻣﺮﺑﻮﻁ ﺑﻪ Preset ﺍﺳﺖ ﮐﻪ ﺑﻪ ﮐﺎﺭﺑﺮ ﺍﻣﮑﺎﻥ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﻗﻠﻢ‌ﻫﺎﯼ ﺟﺪﻳﺪ ﺭﺍ ﻣﯽ‌ﺩﻫﺪ.

ﺩﮔﻤﻪ ﺑﻌﺪﯼ،‌‌ ﻫﻤﺎﻥ ﭘﻨﺠﺮﻩ ﺗﻌﻴﻴﻦ ﺷﮑﻞ ﻭ ﺳﺎﻳﺰ ﺭﺍ ﺑﺎﺯ ﻣﯽ‌ﮐﻨﺪ. ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺩﮔﻤﻪ  ، ﭘﻨﻞ ﻃﺮﺍﺣﯽ ﻗﻠﻢ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ. ﻣﻨﻮﯼ Mode ﮐﻪ ﺍﻣﮑﺎﻥ ﺍﻧﺘﺨﺎﺏ Blending Mode‌ﻫﺎ ﺭﺍ ﻓﺮﺍﻫﻢ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺍﺯ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﺑﺮﺧﻮﺭﺩﺍﺭ ﺍﺳﺖ ﺩﺭ ﺯﻣﺎﻥ ﺗﺪﺭﻳﺲ ﻻﻳﻪ‌ﻫﺎ، ﻣﻮﺭﺩ ﺑﺤﺚ ﻗﺮﺍﺭ ﺧﻮﺍﻫﺪ ﮔﺮﻓﺖ.

ﺩﮔﻤﻪ ﺑﻌﺪﯼ ﻳﻌﻨﯽ Opacity، ﮐﻪ ﺩﺭ ﺯﻣﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻗﻠﻢ ﻣﻮ‌ﻫﺎ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎﺭ‌ﻫﺎ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﮔﻴﺮﺩ، ﻣﻴﺰﺍﻥ ﭘﻮﺷﺶ ﺭﻧﮓ ﻗﻠﻢ ﺭﺍ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﺪ. ﻫﺮﭼﻪ ﺍﻳﻦ ﻋﺪﺩ ﺑﺎﻻ‌ﺗﺮ ﺑﺎﺷﺪ، ﭘﻮﺷﺎﻧﻨﺪﮔﯽ ﻗﻠﻢ ﺑﻴﺸﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻭ ﻫﺮ ﭼﻪ ﺍﻳﻦ ﻋﺪﺩ ﭘﺎﻳﻴﻦ‌ﺗﺮ ﺑﺎﺷﺪ، ﭘﻮﺷﺶ ﮐﻤﺘﺮ ﺍﺳﺖ. ﺩﺭ ﺷﮑﻞ ﺯﻳﺮ، Opacity ﺩﺭ ﺧﻂ ﺑﺎﻻﻳﯽ ﺑﺮﺍﺑﺮ ۱۰۰ ﺩﺭﺻﺪ ﻭ ﺩﺭ ﺧﻂ ﭘﺎﻳﻴﻨﯽ ۵۰ ﺩﺭﺻﺪ ﺍﺳﺖ.

ﺑﻪ ﻟﺤﺎﻅ ﮐﺎﺭﺑﺮﺩ ﺯﻳﺎﺩ ﻭ ﻣﮑﺮﺭ ﺍﻳﻦ ﻗﺎﺑﻠﻴﺖ، ﮐﺎﺑﺮ ﺑﻪ ﺭﺍﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻘﺪﺍﺭ ﺁﻥ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﻣﻨﻈﻮﺭ، ﮐﺎﻓﯽ ﺍﺳﺖ ﮐﻪ ﮐﺎﺭﺑﺮ، ﺩﺭ ﺯﻣﺎﻥ ﻓﻌﺎﻝ ﺑﻮﺩﻥ ﺍﺑﺰﺍﺭ ﺑﺮﺍﺵ، ﻋﺪﺩ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﺗﺎﻳﭗ ﮐﻨﺪ. ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ ﺍﮔﺮ opacity ﺑﺮﺍﺑﺮﺑﺎ ۱۰ ﺩﺭﺻﺪ ﻣﺪ ﻧﻈﺮ ﺍﺳﺖ، ﻋﺪﺩ ۱۰ ﻳﺎ ۱ ﺗﺎﻳﭗ ﺷﻮﺩ. ﺍﮔﺮ ﻋﺪﺩ ﻳﮏ ﺭﻗﻤﯽ ﻣﻮﺭﺩ ﻧﻈﺮ ﺍﺳﺖ، ﺑﺎﻳﺪ ﺍﺑﺘﺪﺍ ﺭﻗﻢ ﺻﻔﺮ ﻭ ﺳﭙﺲ ﻋﺪﺩ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﺗﺎﻳﭗ ﮐﻨﻴﻢ.

ﺳﻤﺒﻮﻝ ﺑﻌﺪﯼ ، ﺯﻣﺎﻧﯽ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ ﮐﻪ ﺑﻪ ﺟﺎﯼ ﻣﻮﺱ ﺍﺯ ﻗﻠﻢ ﻧﻮﺭﯼ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﺩ. ﻓﻌﺎﻝ ﮐﺮﺩﻥ ﺍﻳﻦ ﮔﺰﻳﻨﻪ، ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺑﺎ ﺍﻓﺰﺍﻳﺶ ﻓﺸﺎﺭ ﺩﺭ ﻗﻠﻢ ﻧﻮﺭﯼ، Opacity ﺍﻓﺰﺍﻳﺶ ﻳﺎﺑﺪ.

ﻗﺎﺑﻠﻴﺖ ﺑﻌﺪﯼ ﺩﺭ Option Bar ﺍﻣﮑﺎﻥ Flow ﺍﺳﺖ ﮐﻪ ﻣﺎﻧﻨﺪ Opacity، ﻣﻴﺰﺍﻥ ﭘﻮﺷﺶ ﻗﻠﻢ ﺭﺍ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﺪ ﺍﻣﺎ ﺗﻔﺎﻭﺕ‌ﻫﺎﻳﯽ ﺑﺎ Opacity ﺩﺍﺭﺩ.

- ﺍﻭﻻ، ﺣﺪﺍﮐﺜﺮ ﭘﻮﺷﺶ ﺭﺍ Opacity  ﺗﻌﻴﻴﻦ ﻣﯽ‌ﮐﻨﺪ.

- ﺛﺎﻧﻴﺎ، ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﮐﻠﻴﺪ ﻣﻮﺱ ﺭﺍ ﻓﺸﺎﺭ ﺩﺍﺩﻩ‌ﺍﻳﻢ، ﻫﺮ ﺑﺎﺭ ﮐﻪ ﻗﻠﻢ ﺭﺍ ﺭﻭﯼ ﻳﮏ ﺑﺨﺶ ﺣﺮﮐﺖ ﺑﺪﻫﻴﻢ، ﺑﻪ ﻣﻘﺪﺍﺭ Flow، ﭘﻮﺷﺶ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﺩ (ﺣﺪ ﺍﮐﺜﺮ ﺗﺎ ﻣﻘﺪﺍﺭ Opacity). ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ، ﺍﮔﺮ Flow، ﺑﻪ ﺍﻧﺪﺍﺯﻩ ۳۰ ﺩﺭﺻﺪ ﺗﻨﻈﻴﻢ ﺷﻮﺩ، ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﮐﻠﻴﺪ ﻣﻮﺱ ﺭﺍ ﭘﺎﻳﻴﻦ ﻧﮕﻪ ﺩﺍﺷﺘﻪ‌ﺍﻳﻢ، ﻫﺮ ﺑﺎﺭ ﮐﻪ ﻗﻠﻢ ﺭﺍ ﺭﻭﯼ ﻳﮏ ﻗﺴﻤﺖ ﺣﺮﮐﺖ ﺑﺪﻫﻴﻢ ۳۰ ﺩﺭﺻﺪ ﭘﻮﺷﺶ ﺭﻧﮕﯽ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﺩ. ﻭﻳﺎ ﺍﮔﺮ ﻗﻠﻢ ﺭﺍ ﺩﻭﺑﺎﺭ ﺭﻭﯼ ﻳﮏ ﻗﺴﻤﺖ ﺑﮑﺸﻴﻢ، ۶۰ ﺩﺭﺻﺪ ﭘﻮﺷﺶ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ. ﺣﺎﻝ ﺁﻧﮑﻪ ﺍﮔﺮ Opacity، ﺭﺍ ﺭﻭﯼ ۳۰ ﺩﺭﺻﺪ ﺗﻨﻈﻴﻢ ﮐﻨﻴﻢ ﻭ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺩﺳﺖ ﺭﺍ ﺍﺯ ﺩﮔﻤﻪ ﻣﻮﺱ ﺑﺮﺩﺍﺭﻳﻢ، ﭼﻨﺪ ﺑﺎﺭ ﺭﻭﯼ ﻳﮏ ﻧﻘﻄﻪ ﺣﺮﮐﺖ ﺩﻫﻴﻢ، ﺗﻨﻬﺎ ۳۰ ﺩﺭﺻﺪ ﭘﻮﺷﺶ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺷﺖ.

Opacity ﻭ Flow ﻳﺎ ﺗﺮﮐﻴﺐ ﺁﻥ‌ﻫﺎ، ﺩﺭ ﺯﻣﺎﻥ ﺭﻭﺗﻮﺵ ﻋﮑﺲ ﻳﺎ ﻃﺮﺍﺣﯽ ﮔﺮﺍﻓﻴﮑﯽ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﻘﺸﯽ ﺑﺴﻴﺎﺭ ﺍﺳﺎﺳﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺩﮔﻤﻪ ﺑﻌﺪﯼ ﺩﺭ Option Bar، ﻗﻠﻢ ﺭﻧﮓ ﺍﻓﺸﺎﻥ ﻳﺎ Airbrush ﺍﺳﺖ ﮐﻪ ﻗﻠﻢ ﺭﻧﮓ ﺍﻓﺸﺎﻥ ﺭﺍ ﺷﺒﻴﻪ ﺳﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ. ﺍﮔﺮ ﺍﻳﻦ ﻗﻠﻢ ﺭﺍ ﺭﻭﯼ ﻧﻘﻄﻪ‌ﺍﯼ ﻧﮕﺎﻩ ﺩﺍﺭﻳﻢ ﻭ ﮐﻠﻴﺪ ﻣﻮﺱ ﺭﺍ ﻓﺸﺎﺭ ﺩﻫﻴﻢ، ﻣﺎﻧﻨﺪ ﺭﻧﮓ ﺍﻓﺸﺎﻥ‌ﻫﺎ، ﺩﺍﺋﻤﺎ ﺑﺮ ﭘﻮﺷﺶ ﺭﻧﮓ ﺍﻓﺰﻭﺩﻩ ﻣﯽ‌ﺷﻮﺩ.

ﺳﻤﺒﻞ ﺁﺧﺮ ﺩﺭ Option Bar ﺍﺑﺰﺍﺭ ﻗﻠﻢ ﻣﻮ، ﻣﺮﺑﻮﻁ ﺑﻪ ﻗﻠﻢ ﻧﻮﺭﯼ ﺍﺳﺖ. ﺩﺭ ﺻﻮﺭﺕ ﻓﻌﺎﻝ ﺑﻮﺩﻥ، ﺍﻓﺰﺍﻳﺶ ﻓﺸﺎﺭ ﺩﺭ ﻗﻠﻢ ﻧﻮﺭﯼ، ﺑﺎﻋﺚ ﺍﻓﺰﺍﻳﺶ ﺳﺎﻳﺰ ﻗﻠﻢ ﻣﯽ‌ﺷﻮﺩ.

 

History Brush
ﺍﮐﻨﻮﻥ ﮐﻪ ﺍﺑﺰﺍﺭ ﺑﺮﺍﺵ ﻣﻮﺭﺩ ﺑﺮﺭﺳﯽ ﻗﺮﺍﺭ ﮔﺮﻓﺖ ﺩﺭ ﻣﻮﺭﺩ ﻳﮑﯽ ﺍﺯ ﺍﺑﺰﺍﺭﻫﺎﻳﯽ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﻭﻇﻴﻔﻪ ﺁﻥ ﺑﺎﺯﮔﺮﺩﺍﻧﺪﻥ ﺑﺨﺸﯽ ﺍﺯ ﺗﺼﻮﻳﺮ ﺑﻪ ﻣﺮﺍﺣﻞ ﻗﺒﻠﯽ ﺍﺳﺖ. ﺍﻳﻦ ﺍﺑﺰﺍﺭ،‌‌ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺷﮑﻞ ﺯﻳﺮ ﻧﻴﺰ ﺩﻳﺪﻩ ﻣﯽ‌ﺷﻮﺩ ﺩﺭ ﻣﻴﻠﻪ ﺍﺑﺰﺍﺭ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﻭ ﻣﺘﻌﻠﻖ ﺑﻪ ﻳﮏ ﺟﻌﺒﻪ ﺍﺑﺰﺍﺭ ﺩﻭ ﺗﺎﻳﯽ ﺍﺳﺖ.

ﺗﻔﺎﻭﺕ ﺍﻳﻦ ﺍﺑﺰﺍﺭ ﺑﺎ ﮐﺎﺭﺑﺮﺩ ﭘﻨﻞ ﻫﻴﺴﺘﻮﺭﯼ، ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺗﻮﺳﻂ ﺍﻳﻦ ﺍﺑﺰﺍﺭ، ﻣﯽ‌ﺗﻮﺍﻥ ﺗﻨﻬﺎ ﻗﺴﻤﺘﯽ ﺍﺯ ﺗﺼﻮﻳﺮ ﺭﺍ ﺑﻪ ﻣﺮﺍﺣﻞ ﻗﺒﻠﯽ ﺑﺮﮔﺮﺩﺍﻧﺪ ﺩﺭ ﺣﺎﻟﻴﮑﻪ ﺳﺎﻳﺮ ﺑﺨﺸﻬﺎﯼ ﺗﺼﻮﻳﺮ، ﺩﺭ‌‌ ﻫﻤﺎﻥ ﻣﺮﺣﻠﻪ ﺑﺎﻗﯽ ﻣﯽ‌ﻣﺎﻧﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﻣﻨﻈﻮﺭ، ﺍﺑﺘﺪﺍ ﺑﺎﻳﺪ، ﻣﺸﺨﺺ ﮐﺮﺩ ﮐﻪ ﮐﺪﺍﻡ ﻣﺮﺣﻠﻪ ﺍﺯ ﭘﻨﻞ ﻫﻴﺴﺘﻮﺭﯼ، ﻣﺪ ﻧﻈﺮ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺗﻌﻴﻴﻦ ﻣﺮﺣﻠﻪ ﻣﻮﺭﺩ ﻧﻈﺮ، ﺑﺎﻳﺪ ﺩﺭ ﺧﺎﻧﻪ ﺳﻤﺖ ﭼﭗ ﺁﻥ ﻣﺮﺣﻠﻪ ﻣﺮﺑﻮﻃﻪ ﺩﺭ ﭘﻨﻞ ﻫﻴﺴﺘﻮﺭﯼ، ﮐﻠﻴﮏ ﮐﻨﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺳﻤﺒﻞ History Brush، ﺩﺭ ﮐﻨﺎﺭ ﺁﻥ ﻣﺮﺣﻠﻪ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﺩ.

ﺣﺎﻝ ﺍﮔﺮ ﻧﺸﺎﻧﮕﺮ ﻣﻮﺱ ﺭﺍ ﮐﻪ ﻣﺎﻧﻨﺪ ﻳﻪ ﺻﻮﺭﺕ ﻳﮏ ﺑﺮﺍﺵ ﺍﺳﺖ ﻭ ﺍﺯ‌‌ ﻫﻤﺎﻥ ﻭﻳﮋﮔﯽ‌ﻫﺎ ﻭ ﻗﻮﺍﺋﺪ ﭘﻴﺮﻭﯼ ﻣﯽ‌ﮐﻨﺪ، ﺭﻭﯼ ﻗﺴﻤﺘﻬﺎﯼ ﻣﻮﺭﺩ ﻧﻈﺮ ﺑﮑﺸﻴﻢ، ﺁﻥ ﻗﺴﻤﺖ‌ﻫﺎ ﺑﻪ‌‌ ﻫﻤﺎﻥ ﻣﺮﺣﻠﻪ ﺑﺎﺯ ﻣﯽ‌ﮔﺮﺩﺩ ﮐﻪ ﺩﺭ ﭘﻨﻞ ﻫﻴﺴﺘﻮﺭﯼ ﻋﻼﻣﺖ ﮔﺬﺍﺭﯼ ﺷﺪﻩ ﺍﺳﺖ.

ﺩﺭ ﻣﺜﺎﻝ ﺯﻳﺮ، ﺗﺼﻮﻳﺮ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺗﻴﺮﻩ‌ﺗﺮ ﺷﺪﻩ ﺍﺳﺖ. ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﻳﮑﯽ ﺍﺯ ﮔﻞ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺭﻧﮓ ﻭ ﺗﻮﻥ ﺍﻭﻟﻴﻪ ﺑﺎﺯﮔﺮﺩﺍﻧﻴﻢ. ﺍﺑﺘﺪﺍ ﺩﺭ ﭘﻨﻞ ﻫﻴﺴﺘﻮﺭﯼ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺭﺍ ﮐﻪ ﺩﺭ ﺁﻥ، ﮔﻞ‌ﻫﺎ ﺩﺍﺭﺍﯼ ﺭﻧﮓ ﺍﻭﻟﻴﻪ ﻫﺴﺘﻨﺪ، ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻴﻢ. ﺳﭙﺲ ﻗﻠﻢ ﺭﺍ ﺗﻨﻈﻴﻢ ﮐﺮﺩﻩ ﻭ ﺭﻭﯼ ﮔﻞ ﻣﻮﺭﺩ ﻧﻈﺮ ﻣﯽ‌ﮐﺸﻴﻢ.

‌‌ ﻫﻤﺎﻥ ﮔﻮﻧﻪ ﮐﻪ ﺩﺭ ﺷﮑﻞ ﺩﻳﺪﻩ ﻣﯽ‌ﺷﻮﺩ، ﺭﻧﮓ ﮔﻞ ﺳﻤﺖ ﭼﭗ ﺑﻪ ﺣﺎﻟﺖ ﺍﻭﻟﻴﻪ ﺑﺎﺯ ﮔﺸﺘﻪ ﺍﺳﺖ ﺍﻣﺎ ﺑﻪ ﻋﻠﺖ ﻋﺪﻡ ﺍﻧﺘﺨﺎﺏ ﻗﻠﻢ، ﺗﻨﻈﻴﻤﺎﺕ ﻭ ﺍﺟﺮﺍﯼ ﻣﻨﺎﺳﺐ، ﻧﺘﻴﺠﻪ ﭼﻨﺪﺍﻥ ﻣﻄﻠﻮﺏ ﻧﻴﺴﺖ. ﺑﺮﺍﯼ ﺍﺧﺬ ﻧﺘﻴﺠﻪ ﻣﻄﻠﻮﺏ، ﻣﯽ‌ﺗﻮﺍﻥ ﻗﻠﻢ ﺭﺍ ﮐﻮﭼﮏ‌ﺗﺮ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ، Hardness ﺭﺍ ﻣﺘﻨﺎﺳﺐ ﺑﺎ ﻟﺒﻪ‌ﻫﺎ ﺩﺭ ﻋﮑﺲ ﺗﻌﻴﻴﻦ ﮐﻨﻴﻢ ﻭ ﻋﮑﺲ ﺭﺍ ﺑﻪ ﺍﻧﺪﺍﺯﻩ‌ﺍﯼ ﺑﺰﺭﮔﻨﻤﺎﻳﯽ ﮐﻨﻴﻢ، ﮐﻪ ﺑﻪ ﺳﺎﺩﮔﯽ ﺑﺘﻮﺍﻧﻴﻢ ﻓﻘﻂ ﮔﻞ ﺭﺍ ﻭ ﻧﻪ ﺍﻃﺮﺍﻑ ﺁﻥ ﺭﺍ ﭘﻮﺷﺶ ﺩﻫﻴﻢ.
