ﺩﺭﺱ ﺍﻭﻝ- ﺍﺯ ﺧﺸﻢ ﭼﯽ ﻣﯽ ﺩﻭﻧﻴﻢ؟

ﺑﺎﺯﻡ ﻻﺯﻣﻪ ﻳﺎﺩﺁﻭﺭﯼ ﮐﻨﻢ ﺧﺸﻢ ﻳﻪ ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﺣﺴﯽ ﻃﺒﻴﻌﻴﻪ ﺑﻪ ﻓﺸﺎﺭﻫﺎ ﻭ ﻧﺎﺭﺿﺎﻳﺘﯽ ﻫﺎﻳﯽ ﮐﻪ ﺗﻮﯼ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺑﺮﺍﻣﻮﻥ ﭘﻴﺶ ﻣﻴﺎﺩ. ﻃﺒﻴﻌﻴﻪ ﮐﻪ ﻭﻗﺘﯽ ﻣﻴﺨﻮﺍﻳﻢ ﺑﻪ ﭼﻴﺰﯼ ﺑﺮﺳﻴﻢ ﻭ ﻣﺎﻧﻌﯽ ﺳﺮ ﺭﺍﻫﻤﻮﻥ ﺳﺒﺰ ﻣﻴﺸﻪ ﻋﺼﺒﺎﻧﯽ ﺷﻴﻢ.

- ﺑﻴﺸﺘﺮ ﻭﻗﺘﻬﺎ ﺍﻳﻦ ﻣﺎ ﻧﻴﺴﺘﻴﻢ ﮐﻪ ﺗﺼﻤﻴﻢ ﻣﻴﮕﻴﺮﻳﻢ ﻋﺼﺒﯽ ﺷﻴﻢ، ﻭﻟﯽ ﺑﻪ ﻫﺮ ﺣﺎﻝ ﻳﮑﻬﻮ ﺣﺲ ﻣﻴﮑﻨﻴﻢ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻳﻢ. ﺑﻌﻀﯽ ﻭﻗﺘﻬﺎ ﺗﺎ ﺟﻮﺵ ﻧﻴﺎﻭﺭﺩﻳﻢ ﺣﺘﯽ ﻧﻤﻴﻔﻬﻤﻴﻢ ﻋﺼﺒﯽ ﺷﺪﻳﻢ.

- ﺍﮔﻪ ﺍﺣﺴﺎﺱ ﻋﺼﺒﺎﻧﻴﺖ ﺑﺮﺍﯼ ﻃﻮﻻﻧﯽ ﻣﺪﺕ ﺗﮑﺮﺍﺭ ﺷﻪ، ﺑﺪﻭﻥ ﺍﻳﻦ ﮐﻪ ﺍﺗﻔﺎﻗﯽ ﻳﺎ ﻣﺤﺮﮐﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﺣﺲ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻪ، ﺑﻬﺶ ﻣﻴﮕﻴﻢ «ﻣﻮﻭﺩ». ﺣﺎﻟﺘﻬﺎ (ﻣﻮﻭﺩﻫﺎ)ﯼ ﻋﺼﺒﯽ ﺳﺮﻳﻌﺘﺮ ﺑﻪ ﺍﺣﺴﺎﺱ ﻋﺼﺒﺎﻧﻴﺖ ﺍﻭﻥ ﻫﻢ ﺍﺯ ﻧﻮﻉ ﺷﺪﻳﺪ ﻣﻨﺠﺮ ﻣﻴﺸﻦ.

- ﻭﻗﺘﯽ ﺧﺸﻢ ﺷﺮﻭﻉ ﻣﻴﺸﻪ، ﺭﻭﯼ ﺻﺪﺍ، ﺣﺮﮐﺎﺕ ﻭ ﻭﺿﻌﻴﺖ ﺻﻮﺭﺕ ﻭ ﺟﻮﺭﯼ ﮐﻪ ﻓﮑﺮ ﻣﻴﮑﻨﻴﻢ ﺗﺎﺛﻴﺮ ﻣﻴﺬﺍﺭﻩ. ﺗﺎﺯﻩ ﺁﺩﻡ ﺭﻭ ﺗﺤﺮﻳﮏ ﻣﻴﮑﻨﻪ ﮐﻪ ﻳﻪ ﮐﺎﺭﻫﺎﻳﯽ ﻫﻢ ﺑﮑﻨﻪ. ﺑﻌﻀﯽ ﻭﻗﺘﻬﺎ ﻓﮑﺮﻫﺎﻳﯽ ﺗﻮﯼ ﺳﺮﻣﻮﻥ ﻫﺴﺖ ﮐﻪ ﻋﺼﺒﺎﻧﻴﻤﻮﻥ ﻣﻴﮑﻨﻪ. ﺑﻌﻀﯽ ﻭﻗﺘﻬﺎ ﻫﻢ ﺍﻭﻥ ﻓﮑﺮﻫﺎ ﻭ ﻫﻢ ﺍﺗﻔﺎﻗﯽ ﮐﻪ ﻣﻴﺎﻓﺘﻪ ﺑﺎﻋﺚ ﻣﻴﺸﻪ ﻋﺼﺒﺎﻧﯽ ﺷﻴﻢ.

- ﺩﺍﻧﺸﻤﻨﺪﻫﺎ ﻣﻴﮕﻦ ﻫﺪﻑ ﺣﺲ ﻫﺎﻳﯽ ﻣﺜﻞ ﻋﺼﺒﺎﻧﻴﺖ، ﻫﻤﺎﻫﻨﮕﯽ ﻭ ﺑﺴﻴﺞ ﮐﺮﺩﻥ ﮐﻞ ﺳﻴﺴﺘﻢ ﺑﺪﻥ ﻣﺎﺳﺖ ﮐﻪ ﻣﯽ ﺧﻮﺍﺩ ﺑﻪ ﻣﺤﻴﻂ ﺍﻃﺮﺍﻓﻤﻮﻥ ﺟﻮﺍﺏ ﺑﺪﻩ. ﺍﻳﻦ ﺍﺗﻔﺎﻕ ﻫﻢ ﻭﻗﺘﯽ ﻣﯽ ﺍﻓﺘﻪ ﮐﻪ ﻣﺎ ﻫﻴﭻ ﺷﻨﺎﺧﺖ  ﻭ ﺁﮔﺎﻫﯽ ﻧﺴﺒﺖ ﺑﻪ ﺍﻳﻦ ﻗﻀﻴﻪ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ. ﻭﻟﯽ، ﻣﺎ ﺑﻪ ﺣﺲ ﻫﺎﻣﻮﻥ ﺁﮔﺎﻫﯽ ﺩﺍﺭﻳﻢ ﻭ ﺍﻭﻧﻬﺎ ﺭﻭ ﻳﮏ ﺟﻮﺭﯼ ﺗﻮﯼ ﺑﺪﻧﻤﻮﻥ ﺩﺭﮎ ﻣﯽ ﮐﻨﻴﻢ.

- ﺩﺭ ﺍﺻﻞ، ﺣﺴﻬﺎﻳﯽ ﮐﻪ ﺑﺎ ﺧﺸﻢ ﺍﺯ ﻳﮏ ﺧﻮﻧﻮﺍﺩﻩ ﻫﺴﺘﻨﺪ، ﺍﺣﺘﻤﺎﻻ ﺑﺨﺸﯽ ﺍﺯ ﺳﻴﺴﺘﻢ ﺩﻓﺎﻋﯽ ﻣﻐﺰ ﻣﺎﺳﺖ ﮐﻪ ﺍﺯ ﻣﻴﻠﻴﻮﻧﻬﺎ ﺳﺎﻝ ﭘﻴﺶ ﻣﺎ ﺭﻭ ﺩﺭ ﺑﺮﺍﺑﺮ ﻣﺤﻴﻂ ﻭ ﺑﺮﺍﯼ ﺑﻘﺎ ﻭ ﺯﻧﺪﻩ ﻣﻮﻧﺪﻥ ﻣﺤﺎﻓﻈﺖ ﮐﺮﺩﻩ.

ﺍﮔﺮ ﻳﻪ ﮐﻢ ﻋﻠﻤﻴﺘﺮ ﺑﺨﻮﺍﻳﻢ ﺑﺪﻭﻧﻴﻢ، ﺑﻴﻦ ﺑﺨﺸﻬﺎﯼ ﻣﺨﺘﻠﻒ ﻣﻐﺰ ﺑﺨﺸﯽ ﻫﻢ ﺑﻪ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺗﻔﮑﺮ ﺍﺧﺘﺼﺎﺹ ﺩﺍﺭﻩ. ﺑﺨﺶ ﻣﺮﺑﻮﻁ ﺑﻪ ﺣﺴﻬﺎ ﺧﺸﻢ ﺭﻭ ﻫﻢ ﻫﺪﺍﻳﺖ ﻣﻴﮑﻨﻪ. ﻏﻴﺮ ﺍﺯ ﺍﻳﻦ٬ ﻣﺎ ﺑﺨﺶ ﺩﻳﮕﻪ‌ ﺍﯼ ﻫﻢ ﺗﻮﯼ ﻣﻐﺰﻣﻮﻥ ﺩﺍﺭﻳﻢ ﮐﻪ ﻣﺨﺼﻮﺹ ﻧﮕﻬﺪﺍﺭﯼ ﺧﺎﻃﺮﺍﺗﻤﻮﻧﻪ ﻭ ﻣﺜﻞ ﻳﻪ ﺣﺎﻓﻈﻪ ﺟﺎﻧﺒﯽ ﻣﻴﻤﻮﻧﻪ. ﺑﻌﻀﯽ ﻭﻗﺘﻬﺎ ﮐﻪ ﻣﺎ ﺑﻪ ﻳﻪ ﻣﺤﺮﮐﯽ ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﻋﺼﺒﯽ ﻧﺸﻮﻥ ﻣﻴﺪﻳﻢ، ﺍﻳﻦ ﻋﮑﺲ ﺍﻟﻌﻤﻞ ﻋﺼﺒﯽ ﺗﺮﮐﻴﺒﻴﻪ ﺍﺯ ﺗﺎﺛﻴﺮ ﻫﻤﻮﻥ ﻣﻮﻗﻌﻴﺖ ﻭ ﻳﻪ ﺗﺠﺮﺑﻪ ﺫﻫﻨﯽ ﻗﺪﻳﻤﯽ. ﺍﻳﻦ ﺧﺎﻃﺮﻩ ﺑﻪ ﻭﺳﻴﻠﻪ ﻣﻐﺰ ﺑﺮﺍﯼ ﺣﻔﺎﻇﺖ ﺍﺯ ﻣﺎ ﺩﺭ ﺑﺮﺍﺑﺮ ﺧﻄﺮ ﺑﻪ ﮐﺎﺭ ﻣﻴﺎﻓﺘﻪ ﻭ ﺣﺘﯽ ﻣﻤﮑﻨﻪ ﻣﺎ ﺧﻮﺩﻣﻮﻥ ﻫﻢ ﺍﻭﻟﺶ ﻣﺘﻮﺟﻪ ﺍﺵ ﻧﺸﻴﻢ.

ﺣﺎﻻ ﺩﻳﮕﻪ ﻧﻮﺑﺖ ﺑﺨﺶ ﺗﻔﮑﺮ ﻣﻐﺰﻩ ﮐﻪ ﻳﻪ ﺭﺍﻫﯽ ﺑﺮﺍﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺍﻳﻦ ﺣﺲ ﺧﺸﻢ ﮐﻪ ﺗﻮ ﺑﺨﺶ ﺍﺣﺴﺎﺱ ﻣﻐﺰ ﺑﻪ ﻭﺟﻮﺩ ﺍﻭﻣﺪﻩ ﭘﻴﺪﺍ ﮐﻨﻪ٬ ﺧﺸﻤﯽ ﮐﻪ ﺷﺎﻳﺪ ﺣﺎﻓﻈﻪ ﺟﺎﻧﺒﯽ ﻣﺎ ﻫﻢ ﺗﺸﺪﻳﺪﺵ ﮐﺮﺩﻩ ﺑﺎﺷﻪ.

ﻳﺎﺩﻣﻮﻥ ﻧﺮﻩ:
ﺑﻪ ﻋﻨﻮﺍﻥ ﺍﻧﺴﺎﻥ ﻭ ﺗﻨﻬﺎ ﻣﻮﺟﻮﺩ ﻣﺘﻔﮑﺮ، ﻣﺎ ﺍﻳﻦ ﺗﻮﺍﻧﺎﻳﯽ ﺭﻭ ﺩﺍﺭﻳﻢ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﻢ ﭼﻪ ﻃﻮﺭ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺗﻤﻮﻥ ﮐﻨﺎﺭ ﺑﻴﺎﻳﻢ.

 

ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﻣﻴﺘﻮﻧﻪ ﻣﺜﺒﺖ ﺑﺎﺷﻪ
ﺧﺸﻢ ﻳﻪ ﺣﺲ ﻃﺒﻴﻌﻴﻪ ﺑﺮﺍﯼ ﻫﻤﻪ ﻣﺎ ﺁﺩﻣﻬﺎ. ﺧﺸﻢ ﻧﺸﻮﻧﻪ ﺍﻳﻨﻪ ﮐﻪ ﻣﺎ ﺁﺳﻴﺐ ﺩﻳﺪﻡ ﻳﺎ ﺍﺣﺴﺎﺱ ﺧﻄﺮ ﻭ ﺗﻬﺪﻳﺪ ﻣﻴﮑﻨﻴﻢ. ﻳﻪ ﺯﻧﮓ ﺧﻄﺮﻩ ﮐﻪ ﻣﻴﮕﻪ ﻳﻪ ﺍﻳﺮﺍﺩﯼ ﻫﺴﺖ، ﻳﻪ ﭼﻴﺰﯼ ﺩﺭﺳﺖ ﭘﻴﺶ ﻧﻤﻴﺮﻩ. ﺷﺎﻳﺪ ﺯﻧﮓ ﺧﻄﺮ ﺧﻮﺑﯽ ﺑﺎﺷﻪ ﻭﺍﺳﻪ ﺍﻳﻦ ﮐﻪ ﺑﺎ ﻳﻪ ﺗﺼﻤﻴﻢ ﺩﺭﺳﺖ ﺍﺯ ﺧﻮﺩﻣﻮﻥ ﻣﺤﺎﻓﻈﺖ ﮐﻨﻴﻢ.

ﺧﺸﻢ ﻭ ﻋﺼﻴﺎﻧﻴﺖ ﻣﻴﺘﻮﻧﻪ ﺩﺭﺳﺖ ﻭ ﺳﺎﻟﻢ ﺍﺯ ﻭﺟﻮﺩ ﺁﺩﻡ ﺗﺨﻠﻴﻪ ﺷﻪ. ﻧﮕﻪ ﺩﺍﺷﺘﻦ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﻗﻮﺭﺕ ﺩﺍﺩﻥ ﺣﺮﺹ ﺍﻧﺮﮊﯼ ﺯﻳﺎﺩﯼ ﺍﺯ ﺁﺩﻡ ﻣﻴﺒﺮﻩ ﻭ ﺑﺮﺍﯼ ﺳﻼﻣﺘﯽ ﻫﻢ ﺧﻮﺏ ﻧﻴﺴﺖ. ﺍﮔﺮ ﺧﺸﻤﻤﻮﻥ ﺭﻭ ﺩﺭﺳﺖ ﺁﺯﺍﺩ ﮐﻨﻴﻢ، ﻣﻴﺘﻮﻧﻴﻢ ﺍﺭﺗﺒﺎﻁ‌ ﺑﻬﺘﺮ ﻭ ﺳﺎﻟﻤﺘﺮﯼ ﺑﺎ ﺁﺩﻣﻬﺎﯼ ﺩﻭﺭ ﻭ ﺑﺮﻣﻮﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ.

ﻳﻪ ﻣﺜﺎﻝ ﺳﺎﺩﻩ: ﻳﻪ ﺩﻭﺳﺖ ﺧﻮﺏ ﺩﺍﺭﯼ ﮐﻪ ﻫﻤﻴﺸﻪ ﺩﻳﺮ ﻣﻴﺎﺩ ﺳﺮ ﻗﺮﺍﺭ. ﺧﺐ، ﻣﻌﻠﻮﻣﻪ ﮐﻪ ﺁﺩﻡ ﻧﺎﺭﺍﺣﺖ ﻣﻴﺸﻪ. ﺣﺎﻻ ﭼﯽ ﮐﺎﺭ ﻣﻴﮑﻨﯽ؟ ﻫﻴﭽﯽ ﺑﻬﺶ ﻧﻤﻴﮕﯽ ﻭ ﻓﻘﻂ ﺳﻌﯽ ﻣﻴﮑﻨﯽ ﺑﻬﺎﻧﻪ ﺑﻴﺎﺭﯼ ﻭ ﻧﺒﻴﻨﻴﺶ؟ ﺑﻪ ﻫﻤﻴﻦ ﺭﺍﺣﺘﯽ ﺭﻓﻴﻖ ﺧﻮﺑﺖ ﺭﻭ ﺍﺯ ﺩﺳﺖ ﻣﻴﺪﯼ؟ 

ﺧﻮﺏ ﮐﺎﺭ ﺩﻳﮕﻪ‌ ﺍﯼ ﻫﻢ ﻣﻴﺸﻪ ﮐﺮﺩ. ﻣﻴﺘﻮﻧﯽ ﺑﻬﺶ ﺑﮕﯽ ﭼﻪ ﻗﺪﺭ ﻧﺎﺭﺍﺣﺖ ﻣﻴﺸﯽ ﻭﻗﺘﯽ ﺍﻭﻥ ﻫﻤﻴﺸﻪ ﺩﻳﺮ ﻣﻴﺮﺳﻪ ﺳﺮ ﻗﺮﺍﺭ. ﺣﺲ ﻣﻴﮑﻨﯽ ﺗﻮ ﺍﺻﻼ ﻭﺍﺳﻪ‌ ﺍﺵ ﺍﻫﻤﻴﺖ ﻧﺪﺍﺭﯼ، ﺑﻬﺖ ﺑﺮﻣﻴﺨﻮﺭﻩ.

ﺩﻭﺳﺘﺖ ﺍﻭﻥ ﻭﻗﺖ ﻣﻤﮑﻨﻪ ﺑﻪ ﺧﺎﻃﺮ ﺩﻭﺳﺘﻴﺘﻮﻥ ﻭ ﺭﻭﺭﺍﺳﺘﯽ ﺗﻮ ﺧﻮﺩﺵ ﺭﻭ ﻋﻮﺽ ﮐﻨﻪ،ﺩﻭﺳﺘﻴﺘﻮﻥ ﻫﻢ ﺑﻬﺘﺮ ﻣﯽ‌ﺷﻪ.

ﺍﻳﻦ ﻣﻴﺸﻪ ﻳﻪ ﺍﺳﺘﻔﺎﺩﻩ ﻣﺜﺒﺖ ﺍﺯ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﮐﻪ ﺗﺎﺯﻩ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻪ ﺁﺩﻡ ﺍﺯ ﺩﺳﺖ ﺧﻮﺩﺵ ﺭﺍﺿﯽ ﻭ ﺧﻮﺷﺤﺎﻝ ﻫﻢ ﺑﺸﻪ. ﭼﻮﻥ ﺍﮔﺮ ﺑﺘﻮﻧﯽ ﺭﺍﺟﻊ ﺑﻪ ﺍﺣﺴﺎﺳﺎﺗﺖ ﺑﺎ ﺑﻘﻴﻪ ﺣﺮﻑ ﺑﺰﻧﯽ، ﻳﻌﻨﯽ ﺩﺍﺭﯼ ﺑﻪ ﻫﻤﻪ ﺩﻧﻴﺎ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﻣﻴﮕﯽ: «ﺑﺎﺑﺎ ﻣﻦ ﻭﺍﺳﻪ ﺧﻮﺩﻡ ﺍﻋﺘﺒﺎﺭ ﻭ ﺍﺭﺯﺵ ﻗﺎﺋﻠﻢ ﻭ ﻣﻴﺨﻮﺍﻡ ﻫﻤﻪ ﻫﻢ ﺑﺎ ﻣﻦ ﺍﻳﻦ ﺟﻮﺭﯼ ﺭﻓﺘﺎﺭ ﮐﻨﻦ.»

ﺑﻌﻀﯽ ﻭﻗﺘﻬﺎ ﻫﻢ ﺷﮑﻠﻬﺎﯼ ﺷﺪﻳﺪﯼ ﺍﺯ ﺧﺸﻢ ﻭ ﻧﻔﺮﺕ ﻣﻴﺘﻮﻧﻪ ﺑﻪ ﻳﻪ ﺗﺼﻤﻴﻢ ﺧﻮﺏ ﺍﺟﺘﻤﺎﻋﯽ ﺗﺒﺪﻳﻞ ﺷﻪ.

ﻣﺜﺎﻝ ﺧﻮﺑﺶ ﺍﻳﻦ ﮐﻪ ﻳﻪ ﻣﻮﺳﺴﻪ ﻫﺴﺖ ﺗﻮﯼ ﺁﻣﺮﻳﮑﺎ ﺑﻪ ﺍﺳﻢ «ﻣﺎﺩﺭﺍﻥ ﻣﺨﺎﻟﻒ ﺭﺍﻧﻨﺪﮔﯽ ﺩﺭ ﺯﻣﺎﻥ ﻣﺴﺘﯽ». ﺍﻳﻦ ﻣﻮﺳﺴﻪ ﺭﻭ ﺩﻭ ﺗﺎ ﻣﺎﺩﺭ ﺭﺍﻩ ﺍﻧﺪﺍﺧﺘﻦ ﮐﻪ ﺑﭽﻪ ﻫﺎﺷﻮﻥ ﺭﻭ ﺩﻭ ﺗﺎ ﺭﺍﻧﻨﺪﻩ ﻣﺴﺖ ﺗﻮﯼ ﺗﺼﺎﺩﻑ ﮐﺸﺘﻪ ﺑﻮﺩﻧﺪ. ﺩﻭﺭ ﻭ ﺑﺮﺧﻮﺩﻣﻮﻥ ﻫﻢ ﻣﻤﮑﻨﻪ ﺍﺯ ﺍﻳﻦ ﻣﻮﺳﺴﻪ‌ ﻫﺎﯼ ﺧﻴﺮﻳﻪ ﺯﻳﺎﺩ ﺑﺎﺷﻪ ﮐﻪ ﺷﺮﻭﻋﺶ ﺍﺯ ﻳﻪ ﺣﺲ ﺧﺸﻢ ﻭ ﻧﻔﺮﺕ ﺑﻮﺩﻩ.

 

ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﮎِﯼ ﻣﺸﮑﻞ ﺳﺎﺯ ﻣﻴﺸﻪ؟
ﻣﺎ ﺍﺯ ﻧﺸﻮﻥ ﺩﺍﺩﻥ ﺧﺸﻢ ﻣﺜﻞ ﻫﺮ ﺭﻓﺘﺎﺭ ﺩﻳﮕﻪ ﺍﯼ ﮐﻪ ﺍﺯﻣﻮﻥ ﺳﺮ ﻣﻴﺰﻧﻪ ﻳﻪ ﻣﻨﻈﻮﺭﯼ ﺩﺍﺭﻳﻢ. ﻭﻗﺘﯽ ﻋﺼﺒﺎﻧﻴﺘﻤﻮﻥ ﺭﻭ ﻧﺸﻮﻥ ﻣﻴﺪﻳﻢ ﻳﻌﻨﯽ ﻣﻴﺨﻮﺍﻳﻢ ﺩﻳﮕﺮﺍﻥ ﺭﻭ ﻣﺘﻮﺟﻪ ﺧﻮﺩﻣﻮﻥ ﮐﻨﻴﻢ ﮐﻪ ﺻﺪﺍﻣﻮﻥ ﺭﻭ ﺑﺸﻨﻮﻥ، ﺟﻮﺍﺏ‌ﻣﻮﻥ ﺭﻭ ﺑﺪﻥ، ﻣﺎ ﺭﻭ ﺑﺒﻴﻨﻦ ﻳﺎ ﻣﺸﮑﻠﻤﻮﻥ ﺭﻭ ﺣﻞ ﮐﻨﻦ.

ﻓﻘﻂ، ﻣﺸﮑﻞ ﺍﻳﻦ ﺟﺎﺳﺖ ﮐﻪ ﺗﻮﯼ ۹۵% ﺍﺯ ﻣﻮﺍﺭﺩ، ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﻭﺿﻊ ﺭﻭ ﺧﺮﺍﺑﺘﺮ ﻣﻴﮑﻨﻪ.

ﻳﻌﻨﯽ ﺧﺸﻢ ﻫﻤﻮﻥ ﻃﻮﺭ ﮐﻪ ﻣﻴﺘﻮﻧﻪ ﻧﺘﻴﺠﻪ ﻣﺜﺒﺖ ﻫﻢ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻪ، ﻣﻴﺘﻮﻧﻪ ﺑﻪ ﺍﺗﻔﺎﻗﻬﺎﯼ ﺑﺪﯼ ﻫﻢ ﻣﻨﺠﺮ ﺷﻪ ﻭ ﺧﺐ ﺍﻭﻥ ﻭﻗﺖ ﺍﮔﻪ ﺑﻬﺎﺵ ﺍﺯ ﺳﻮﺩﯼ ﮐﻪ ﻣﻴﺒﺮﻳﻢ ﺧﻴﻠﯽ ﺑﻴﺸﺘﺮ ﺑﺎﺷﻪ ﭼﯽ؟

 

ﺑﻬﺎﻳﯽ ﮐﻪ ﺑﺮﺍﯼ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﺑﺎﻳﺪ ﭘﺲ ﺑﺪﻳﻢ
- ﺳﻼﻣﺘﻴﻤﻮﻥ: ﮐﺴﺎﻧﯽ ﮐﻪ ﻓﺸﺎﺭ ﺧﻮﻧﺸﻮﻥ ﺑﻪ ﺧﺎﻃﺮ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺧﺸﻢ ﺑﺎﻻ ﻣﻴﺮﻩ، ﺳﻪ ﺑﺮﺍﺑﺮ ﺑﻴﺸﺘﺮ ﺍﺯ ﺁﺩﻣﻬﺎﯼ ﺩﻳﮕﻪ ﺩﺭ ﻣﻌﺮﺽ ﺑﻴﻤﺎﺭﯼ ﻗﻠﺐ ﻫﺴﺘﻦ. 

- ﺣﺲ ﺭﺿﺎﻳﺘﻤﻮﻥ ﺍﺯ ﺧﻮﺩﻣﻮﻥ ﻭ ﺍﺣﺘﺮﺍﻣﯽ ﺭﻭ ﮐﻪ ﭘﻴﺶ ﺑﻘﻴﻪ ﺩﺍﺭﻳﻢ ﺍﺯ ﺩﺳﺖ ﻣﻴﺪﻳﻢ: ﺩﺭﺳﺘﻪ ﮐﻪ ﺑﻌﻀﯽ ﺁﺩﻣﻬﺎ ﻫﻤﻴﺸﻪ ﺑﻪ ﺧﻮﺩﺷﻮﻥ ﺣﻖ ﻣﻴﺪﻥ ﮐﻪ ﻋﺼﺒﺎﻧﯽ ﺑﺎﺷﻦ٬ ﻭﻟﯽ ﺑﻴﺸﺘﺮ ﺁﺩﻣﻬﺎ ﺑﻌﺪ ﺍﺯ ﺍﻳﻦ ﮐﻪ ﻋﺼﺒﺎﻧﯽ ﻣﻴﺸﻦ ﻭ ﺭﻓﺘﺎﺭﻫﺎﯼ ﻋﺼﺒﯽ ﺍﺯ ﺧﻮﺩﺷﻮﻥ ﻧﺸﻮﻥ ﻣﻴﺪﻥ ﭘﺸﻴﻤﻮﻥ ﻣﻴﺸﻦ ﻭ ﭘﻴﺶ ﺧﻮﺩﺷﻮﻥ ﮐﻮﭼﮏ ﻣﻴﺸﻦ. ﻃﺮﻑ ﺩﻳﮕﻪ ﻗﻀﻴﻪ ﻫﻢ ﺍﻋﺘﺒﺎﺭ ﻭ ﺍﺣﺘﺮﺍﻣﻴﻪ ﮐﻪ ﭘﻴﺶ ﺑﻘﻴﻪ ﺩﻭﺳﺘﻬﺎﻣﻮﻥ، ﻫﻤﮑﻼﺳﻴﻬﺎ، ﻫﻤﺴﺎﻳﻪ ﻫﺎ ﻭ ﺧﻮﻧﻮﺍﺩﻩ ﺿﺎﻳﻊ ﻣﻴﺸﻪ ﻭ ﺍﺯ ﺩﺳﺖ ﻣﻴﺪﻳﻢ.

- ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻥ ﺭﻭﺍﺑﻄﻤﻮﻥ: ﺍﻳﻦ ﺷﺎﻳﺪ ﻳﮑﯽ ﺍﺯ ﺑﺪﺗﺮﻳﻦ ﻫﺰﻳﻨﻪ ﻫﺎﻳﯽ ﺑﺎﺷﻪ ﮐﻪ ﺑﺎﺑﺖ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﮐﻨﺘﺮﻝ ﻧﺸﺪﻩ ﻣﺠﺒﻮﺭ ﺷﻴﻢ ﺑﺪﻳﻢ، ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻥ ﺩﻭﺳﺘﻬﺎ ﻭ ﺭﻭﺍﺑﻂ ﺧﻮﺑﻤﻮﻥ .

- ﭘﻮﻝ، ﺁﺯﺍﺩﯼ ﻭ ﻋﻤﺮﻣﻮﻥ: ﺷﺎﻳﺪ ﺑﻪ ﻧﻈﺮ ﺧﻨﺪﻩ ﺩﺍﺭ ﺑﻴﺎﺩ٬ ﻭﻟﯽ ﮐﻠﯽ ﺍﺯ ﻧﻮﺟﻮﻭﻧﻬﺎﻳﯽ ﮐﻪ ﺍﻻﻥ ﻣﺠﺮﻣﻨﺪ٬ ﻭﻗﺘﯽ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻥ ﻭ ﻧﺘﻮﻧﺴﺘﻦ ﺧﻮﺩﺷﻮﻥ ﺭﻭ ﮐﻨﺘﺮﻝ ﮐﻨﻦ ﻳﺎ ﺩﻋﻮﺍ ﮐﺮﺩﻥ ﻳﺎ ﮐﺘﮏ‌ﮐﺎﺭﯼ ﻭ ﮐﻢِ ﮐﻢِﺵ ﺑﻪ ﺧﺎﻃﺮﺵ ﻳﺎ ﺟﺮﻳﻤﻪ ﺷﺪﻥ ﻭ ﺧﻮﻧﻮﺍﺩﻩ ﺷﻮﻥ ﻣﺠﺒﻮﺭ ﺷﺪﻥ ﺧﺴﺎﺭﺕ ﺑﺪﻥ ﻳﺎ ﺑﺪﺗﺮ ﻣﺠﺒﻮﺭ ﺷﺪﻥ ﺑﺮﻥ ﺯﻧﺪﺍﻥ ﻭ ﺑﻪ ﺧﺎﻃﺮﺵ ﺣﺘﯽ ﺗﺎ ﭘﺎﯼ ﺍﺯ ﺩﺳﺖ ﺩﺍﺩﻥ ﺟﻮﻧﺸﻮﻥ ﻫﻢ ﺭﻓﺘﻦ ﮐﻪ ﺩﻳﮕﻪ ﻫﻴﭻ ﺟﺒﺮﺍﻧﯽ ﻧﺪﺍﺭﻩ.

ﺗﻮ ﻗﺴﻤﺘﻬﺎﯼ ﺑﻌﺪﯼ ﺑﺎ ﺗﺠﺮﺑﻪ ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﺍﻳﻦ ﺑﭽﻪ‌ﻫﺎ ﺑﻴﺸﺘﺮ ﺁﺷﻨﺎ ﻣﻴﺸﻴﻢ.

ﺧﻮﺏ ﻓﮑﺮﺍﺕ ﺭﻭ ﺑﮑﻦ ﻭ ﺑﺒﻴﻦ ﺗﺎ ﺣﺎﻻ ﭼﻪ ﻗﺪﺭ ﻭ ﮐﺠﺎ ﺑﺮﺍﯼ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻥ ﻫﺰﻳﻨﻪ ﺩﺍﺩﯼ؟ ﻓﮑﺮ ﻣﻴﮑﻨﯽ ﺍﺭﺯﺷﺶ ﺭﻭ ﺩﺍﺷﺘﻪ؟ ﺧﻮﺏ ﺑﻬﺶ ﻓﮑﺮ ﮐﻦ. ﻓﮑﺮ ﮐﺮﺩﻥ ﺍﻭﻟﻴﻦ ﻗﺪﻣﻪ.

 

ﺗﻮ ﺩﺭﺱ ﻫﺎﯼ ﺑﻌﺪﯼ ﺷﮕﺮﺩﻫﺎﻳﯽ ﺑﺮﺍﯼ ﮐﻨﺘﺮﻝ ﻭ ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ ﮔﻔﺘﻪ ﻣﻴﺸﻪ ﻭ ﻗﺪﻡ ﺑﻪ ﻗﺪﻡ ﺍﻭﻥ ﻗﺪﺭ ﻣﯽ‌ﺭﻳﻢ ﺟﻠﻮ ﮐﻪ ﺑﺘﻮﻧﻴﻢ ﺁﺧﺮ ﺳﺮ ﺧﺸﻤﻤﻮﻥ ﺭﻭ ﺑﮕﻴﺮﻳﻢ ﺗﻮ ﻣﺸﺘﻤﻮﻥ ﻭ ﻣﻬﺎﺭﺵ ﮐﻨﻴﻢ.
