ﺩﺭﺱ ﻫﻔﺘﻢ- ﻋﺮﺿﻪ ﻭ ﺗﻘﺎﺿﺎ

ﻋﺮﺿﻪ ﻭ ﺗﻘﺎﺿﺎ ﺩﻭ ﻭﺍﮊﻩ ﺍﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﺍﻗﺘﺼﺎﺩﺩﺍﻧﺎﻥ ﺑﺴﻴﺎﺭ ﺑﻪ ﮐﺎﺭ ﻣﯽ ﺑﺮﻧﺪ. ﻋﺮﺿﻪ ﻭ ﺗﻘﺎﺿﺎ ﺩﻭ ﻧﻴﺮﻭﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺍﻗﺘﺼﺎﺩ ﺭﺍ ﻣﯽ ﺭﺍﻧﻨﺪ ﻭ ﺑﺎﺯﺍﺭ ﺭﺍ ﺷﮑﻞ ﻣﯽ ﺩﻫﻨﺪ.

 

ﺑﺎﺯﺍﺭ ﭼﻴﺴﺖ؟
ﺑﺎﺯﺍﺭ ﻣﺠﻤﻮﻋﻪ ﺍﯼ ﺍﺯ ﺧﺮﻳﺪﺍﺭﺍﻥ ﻭ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﻳﮏ ﮐﺎﻻ ﻳﺎ ﺧﺪﻣﺖ ﺭﺍ ﻣﯽ ﮔﻮﻳﻨﺪ. ﺩﺭ ﺑﺎﺯﺍﺭ ﺧﺮﻳﺪﺍﺭﺍﻥ ﻣﻴﺰﺍﻥ ﺗﻘﺎﺿﺎ (Demand)  ﻭ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﻣﻴﺰﺍﻥ ﻋﺮﺿﻪ (Supply)  ﺭﺍ ﻣﻌﻴﻦ ﻣﯽ ﮐﻨﻨﺪ. ﺑﺎﺯﺍﺭ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﺷﮑﺎﻝ ﻣﺨﺘﻠﻔﯽ ﺑﻪ ﺧﻮﺩ ﺑﮕﻴﺮﺩ. ﮔﺎﻫﯽ ﺍﻳﻦ ﺑﺎﺯﺍﺭ ﺑﺴﻴﺎﺭ ﺳﺎﻣﺎﻥ ﺩﺍﺩﻩ ﺷﺪﻩ ﺍﺳﺖ ﻭ ﮔﺎﻩ ﮐﻤﺘﺮ. 

 

ﺑﺎﺯﺍﺭ ﺭﻗﺎﺑﺘﯽ ﭼﻴﺴﺖ؟
ﺑﺎﺯﺍﺭ ﭘﻔﮏ ﺭﺍ ﺩﺭ ﻧﻄﺮ ﺑﮕﻴﺮﻳﺪ. ﺑﺎﺯﺍﺭ ﭘﻔﮏ ﺑﺎﺯﺍﺭﯼ ﺍﺳﺖ ﺑﻪ ﺷﺪﺕ ﺭﻗﺎﺑﺘﯽ. ﺗﻌﺪﺍﺩ ﺑﺴﻴﺎﺭﯼ ﺗﻮﻟﻴﺪﮐﻨﻨﺪﻩ ﻭ ﻓﺮﻭﺷﻨﺪﻩ ﺩﺭ ﺍﻳﻦ ﺑﺎﺯﺍﺭ ﺣﻀﻮﺭ ﺩﺍﺭﻧﺪ. ﻳﮏ ﺧﺮﻳﺪﺍﺭ ﻣﯽ ﺩﺍﻧﺪ، ﻓﺮﻭﺷﻨﺪﻩ ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺣﻀﻮﺭ ﺩﺍﺭﻧﺪ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﺪ ﮐﺎﻻﯼ ﺁﻧﻬﺎ ﺭﺍ ﺑﺨﺮﺩ ﻭ ﺧﺮﻳﺪﺍﺭﺍﻥ ﺑﺴﻴﺎﺭ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﻫﺴﺘﻨﺪ ﮐﻪ ﮐﺎﻻﯼ ﺁﻥ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﺭﺍ ﺑﺨﺮﻧﺪ.

ﻓﺮﻭﺷﻨﺪﻩ ﻧﻴﺰ ﻣﯽ ﺩﺍﻧﺪ ﮐﻪ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﺑﺴﻴﺎﺭ ﺩﻳﮕﺮﯼ ﺑﺎ ﮐﺎﻻﻳﯽ ﮐﻢ ﻭ ﺑﻴﺶ ﻣﺸﺎﺑﻪ ﮐﺎﻻﯼ ﺍﻭ ﺩﺭ ﺑﺎﺯﺍﺭ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﻭ ﺧﺮﻳﺪﺍﺭﺍﻥ ﺯﻳﺎﺩﯼ ﻧﻴﺰ ﻫﺴﺘﻨﺪ ﮐﻪ ﮐﺎﻻﯼ ﺍﻭ ﺭﺍ ﺑﺨﺮﻧﺪ. ﺩﺭ ﻧﺘﻴﺠﻪ ﻗﻴﻤﺖ ﻭ ﮐﻤﻴﺖ ﭘﻔﮏ ﻣﻮﺟﻮﺩ ﺩﺭ ﺑﺎﺯﺍﺭ ﺗﻮﺳﻂ ﻳﮏ ﻓﺮﻭﺷﻨﺪﻩ ﻳﺎ ﺧﺮﻳﺪﺍﺭ ﻣﻌﻴﻦ ﻧﻤﯽ ﺷﻮﺩ. ﻗﻴﻤﺖ ﻭ ﮐﻤﻴﺖ ﭘﻔﮏ ﻣﻮﺟﻮﺩ ﺩﺭ ﺑﺎﺯﺍﺭ ﺗﻮﺳﻂ ﺭﻓﺘﺎﺭ ﺗﻤﺎﻣﯽ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﻭ ﺧﺮﻳﺪﺍﺭﺍﻥ ﺗﻌﻴﻴﻦ ﻣﯽ ﺷﻮﺩ. ﭼﻨﻴﻦ ﺑﺎﺯﺍﺭﯼ ﻳﮏ ﺑﺎﺯﺍﺭ ﺭﻗﺎﺑﺘﯽ (Competitive Market) ﺍﺳﺖ.

ﺑﺎﺯﺍﺭ ﺭﻗﺎﺑﺘﯽ ﺑﻪ ﺑﺎﺯﺍﺭﯼ ﮔﻔﺘﻪ ﻣﯽ ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺁﻥ ﺗﻌﺪﺍﺩ ﺯﻳﺎﺩﯼ ﻓﺮﻭﺷﻨﺪﻩ ﻭ ﺧﺮﻳﺪﺍﺭ ﺣﻀﻮﺭ ﺩﺍﺭﻧﺪ ﻭ ﻫﺮ ﻳﮏ ﺍﺯ ﺁﻧﻬﺎ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺍﺛﺮ ﮐﻤﯽ ﺩﺭ ﺗﻌﻴﻴﻦ ﻗﻴﻤﺖ ﺑﺎﺯﺍﺭ ﺩﺍﺭﻧﺪ. ﺑﺎﺯﺍﺭ ﮐﺎﻣﻼ ﺭﻗﺎﺑﺘﯽ (Perfectly Competitive Market) ﺑﺎﺯﺍﺭﯼ ﺍﺳﺖ ﮐﻪ ﺩﻭ ﺷﺮﻁ ﺩﺭ ﺁﻥ ﺑﺮﻗﺮﺍﺭ ﺑﺎﺷﺪ. ﺍﻭﻝ ﮐﺎﻻﯼ ﻋﺮﺿﻪ ﺷﺪﻩ ﺩﺭ ﺑﺎﺯﺍﺭ ﮐﺎﻣﻼ ﻳﮑﺴﺎﻥ ﺍﺳﺖ ﻭ ﺩﻭﻡ ﺍﻳﻨﮑﻪ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﻭ ﺧﺮﻳﺪﺍﺭﺍﻥ ﭼﻨﺎﻥ ﺯﻳﺎﺩﻧﺪ ﮐﻪ ﻫﻴﭻ ﻳﮏ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻧﻤﯽ ﺗﻮﺍﻧﻨﺪ ﺩﺭ ﺗﻌﻴﻴﻦ ﻗﻴﻤﺖ ﺑﺎﺯﺍﺭ ﻣﻮﺛﺮ ﺑﺎﺷﻨﺪ. ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﺑﺎﺯﺍﺭ ﮐﺎﻣﻼ ﺭﻗﺎﺑﺘﯽ ﺍﺳﺖ ﻭ ﺧﺮﻳﺪﺍﺭﺍﻥ ﻭ ﻓﺮﻭﺷﻨﺪﮔﺎﻥ ﺑﺎﻳﺪ ﭘﺬﻳﺮﺍﯼ ﻗﻴﻤﺖ ﺑﺎﺷﻨﺪ ﺑﻪ ﺁﻧﻬﺎ ﻗﻴﻤﺖ ﭘﺬﻳﺮ (Price Taker) ﻣﯽ ﮔﻮﻳﻨﺪ.

ﺑﺮﺧﯽ ﺑﺎﺯﺍﺭﻫﺎ ﻫﺴﺘﻨﺪ ﮐﻪ ﺷﺮﺍﻳﻂ ﺑﺎﺯﺍﺭ ﮐﺎﻣﻼ ﺭﻗﺎﺑﺘﯽ ﺩﺭ ﺁﻧﻬﺎ ﺗﻘﺮﻳﺒﺎ ﺻﺪﻕ ﻣﯽ ﮐﻨﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ  ﺑﺎﺯﺍﺭ ﮔﻨﺪﻡ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ. ﺗﻌﺪﺍﺩ ﺯﻳﺎﺩﯼ ﺧﺮﻳﺪﺍﺭ ﻭ ﻓﺮﻭﺷﻨﺪﻩ ﺩﺭ ﺁﻥ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﻭ ﻫﻴﭻ ﮐﺲ ﻧﻤﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺗﻌﻴﻴﻦ ﮐﻨﻨﺪﻩ ﻗﻴﻤﺖ ﺑﺎﺷﺪ ﻳﺎ ﺑﺮ ﺁﻥ ﺍﺛﺮ ﭼﻨﺪﺍﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﻫﻤﻪ ﮐﺎﻻﻫﺎ ﺩﺭ ﺑﺎﺯﺍﺭ ﺭﻗﺎﺑﺘﯽ ﮐﺎﻣﻞ ﻓﺮﻭﺧﺘﻪ ﻧﻤﯽ ﺷﻮﻧﺪ. ﺑﺮﺧﯽ ﺑﺎﺯﺍﺭﻫﺎ ﺗﻨﻬﺎ ﻳﮏ ﻓﺮﻭﺷﻨﺪﻩ ﺩﺍﺭﻧﺪ ﮐﻪ ﺑﻪ ﺁﻧﻬﺎ ﻣﻮﻧﻮﭘﻮﻟﯽ (Monopoly) ﻣﯽ ﮔﻮﻳﻨﺪ. ﻋﻤﺪﻩ ﺑﺎﺯﺍﺭﻫﺎ ﺩﺭ ﻣﻴﺎﻧﻪ ﺍﻳﻦ ﺩﻭ ﺳﺮ ﻳﻌﻨﯽ ﺑﺎﺯﺍﺭ ﺭﻗﺎﺑﺘﯽ ﮐﺎﻣﻞ ﻭ ﻣﻮﻧﻮﭘﻮﻟﯽ ﻗﺮﺍﺭ ﺩﺍﺭﻧﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﺑﺎﺯﺍﺭ ﺩﺭ ﺷﺮﺍﻳﻂ  ﺩﻭ ﺳﺮ ﻃﻴﻒ ﺑﻪ ﻣﺎ ﮐﻤﮏ ﻣﯽ ﮐﻨﺪ ﺩﺭ ﺁﻏﺎﺯ ﺑﺘﻮﺍﻧﻴﻢ ﺗﺤﻠﻴﻞ ﺑﻬﺘﺮﯼ ﺍﺯ ﺑﺎﺯﺍﺭ ﺍﺭﺍﺋﻪ ﺑﺪﻫﻴﻢ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﺑﺤﺚ ﻫﺎﻳﻤﺎﻥ، ﻣﺎ ﺑﺎﺯﺍﺭ ﺭﺍ ﺑﺎﺯﺍﺭ ﺭﻗﺎﺑﺘﯽ ﮐﺎﻣﻞ ﺩﺭ ﻧﻈﺮ ﻣﯽ ﮔﻴﺮﻳﻢ.

 

ﺗﻘﺎﺿﺎ
ﻣﻴﺰﺍﻥ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﻳﮏ ﮐﺎﻻ ﺣﺠﻤﯽ ﺍﺯ ﺁﻥ ﮐﺎﻻ ﺍﺳﺖ ﮐﻪ ﺧﺮﻳﺪﺍﺭﺍﻥ ﻃﻠﺐ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺗﻮﺍﻧﺎﻳﯽ ﺧﺮﻳﺪﺵ ﺭﺍ ﺩﺍﺭﻧﺪ. ﻋﻮﺍﻣﻞ ﺑﺴﻴﺎﺭﯼ ﺑﺮ ﻣﻴﺰﺍﻥ ﺗﻘﺎﺿﺎ ﻣﻮﺛﺮﻧﺪ. ﻗﺎﻧﻮﻥ ﺗﻘﺎﺿﺎ ﻣﯽ ﮔﻮﻳﺪ ﺩﺭ ﻭﺿﻌﻴﺘﯽ ﮐﻪ ﻫﻤﻪ ﻋﻮﺍﻣﻞ ﺩﻳﮕﺮ ﺛﺎﺑﺖ ﺑﺎﺷﻨﺪ ﻭﻗﺘﯽ ﻗﻴﻤﺖ ﮐﺎﻻ ﺑﺎﻻ ﺑﺮﻭﺩ ﻣﻴﺰﺍﻥ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺁﻥ ﮐﺎﻫﺶ ﻣﯽ ﻳﺎﺑﺪ. ﻧﻤﻮﺩﺍﺭﯼ ﮐﻪ ﺭﺍﺑﻄﻪ ﻣﻴﺎﻥ ﻗﻴﻤﺖ ﮐﺎﻻ ﻭ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺁﻥ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ ﺭﺍ ﻧﻤﻮﺩﺍﺭ ﺗﻘﺎﺿﺎ ﻣﯽ ﮔﻮﻳﻨﺪ. ﻧﻤﻮﺩﺍﺭ ﺗﻘﺎﺿﺎ ﺩﺭ ﺑﺎﺯﺍﺭ ﺩﺭ ﻭﺍﻗﻊ ﺑﺮﺁﻳﻨﺪ ﻧﻤﻮﺩﺍﺭﻫﺎﯼ ﺗﻘﺎﺿﺎﯼ ﺧﺮﻳﺪﺍﺭﺍﻥ ﻣﺨﺘﻠﻒ ﺩﺭ ﺑﺎﺯﺍﺭ ﺍﺳﺖ. ﻧﻤﻮﺩﺍﺭ ﺗﻘﺎﺿﺎ ﻧﻤﻮﺩﺍﺭﯼ ﻧﺰﻭﻟﯽ ﺍﺳﺖ. 

ﮔﻔﺘﻴﻢ ﮐﻪ ﻋﻮﺍﻣﻞ ﻣﺘﻌﺪﺩﯼ ﺑﺮ ﺗﻘﺎﺿﺎ ﻣﻮﺛﺮﻧﺪ. ﺩﺭ ﺍﺩﺍﻣﻪ ﺗﻐﻴﻴﺮ ﺩﺭ ﻫﺮ ﻳﮏ ﺍﺯ ﺍﻳﻦ ﻋﻮﺍﻣﻞ ﺑﺎ ﺛﺎﺑﺖ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﺩﻳﮕﺮ ﻋﻮﺍﻣﻞ ﺭﺍ ﺑﺮﺭﺳﯽ ﻣﯽ ﮐﻨﻴﻢ.

ﻗﻴﻤﺖ: ﮔﻔﺘﻴﻢ ﮐﻪ ﻭﻗﺘﯽ ﻗﻴﻤﺖ ﺑﺎﻻ ﻣﯽ ﺭﻭﺩ، ﺗﻘﺎﺿﺎ ﮐﻢ ﻣﯽ ﺷﻮﺩ ﻭ ﻭﻗﺘﯽ ﻗﻴﻤﺖ ﮐﺎﻫﺶ ﻳﺎﺑﺪ، ﺗﻘﺎﺿﺎ ﺍﻓﺰﺍﻳﺶ ﻣﯽ ﻳﺎﺑﺪ. ﺑﻪ ﻋﺒﺎﺭﺗﯽ ﺩﻳﮕﺮ ﺩﺭ ﺍﻳﻦ ﺷﺮﺍﻳﻂ ﻣﺎ ﺑﺮ ﺭﻭﯼ ﻧﻤﻮﺩﺍﺭ ﺗﻘﺎﺿﺎ ﺣﺮﮐﺖ ﮐﺮﺩﻩ ﻭ ﺑﺎﻻ ﻭ ﭘﺎﻳﻴﻦ ﻣﯽ ﺭﻭﻳﻢ.

ﺩﺭﺁﻣﺪ: ﭼﻪ ﺍﺗﻔﺎﻗﯽ ﻣﯽ ﺍﻓﺘﺪ ﻭﻗﺘﯽ ﺩﺭﺁﻣﺪ ﺗﻐﻴﻴﺮ ﮐﻨﺪ؟ ﺑﻪ ﻋﺒﺎﺭﺗﯽ ﻭﻗﺘﯽ ﮐﻪ ﻗﺪﺭﺕ ﺧﺮﻳﺪ ﺍﻓﺰﺍﻳﺶ ﻳﺎ ﮐﺎﻫﺶ ﻳﺎﺑﺪ.

ﻧﻮﺷﺎﺑﻪ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ. ﻓﺮﺽ ﮐﻨﻴﺪ ﺩﺭ ﺁﻣﺪ ﺷﻤﺎ ﮐﺎﻫﺶ ﻳﺎﺑﺪ، ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﻧﻮﺷﺎﺑﻪ ﮐﺎﻫﺶ ﻳﺎﺑﺪ. ﺑﻪ ﭼﻨﻴﻦ ﮐﺎﻻﻳﯽ ﮐﺎﻻﯼ ﻋﺎﺩﯼ ﮔﻔﺘﻪ ﻣﯽ ﺷﻮﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﮐﺎﻻﯼ ﻋﺎﺩﯼ ﮐﺎﻻﻳﯽ ﺍﺳﺖ ﮐﻪ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺁﻥ ﺑﺎ ﺍﻓﺰﺍﻳﺶ ﺩﺭ ﺁﻣﺪ ﺍﻓﺰﺍﻳﺶ ﻣﯽ ﻳﺎﺑﺪ ﻳﺎ ﺑﻪ ﻋﺒﺎﺭﺗﯽ ﺗﻐﻴﻴﺮﺍﺕ ﺩﺭ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺁﻥ ﺭﺍﺑﻄﻪ ﻣﺴﺘﻘﻴﻢ ﺑﺎ ﺗﻐﻴﻴﺮﺍﺕ ﺩﺭ ﺩﺭﺁﻣﺪ ﺧﺮﻳﺪﺍﺭ ﺩﺍﺭﺩ.

ﺍﮐﺜﺮ ﮐﺎﻻﻫﺎ ﭼﻨﻴﻦ ﻫﺴﺘﻨﺪ ﻭﻟﯽ ﺁﻳﺎ ﮐﺎﻻﻳﯽ ﻫﺴﺖ ﮐﻪ ﭼﻨﻴﻦ ﻧﺒﺎﺷﺪ؟ ﻳﻌﻨﯽ ﻭﻗﺘﯽ ﺩﺭﺁﻣﺪ ﺷﻤﺎ ﺍﻓﺰﺍﻳﺶ ﻣﯽ ﻳﺎﺑﺪ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﻣﺼﺮﻑ ﺁﻥ ﮐﺎﻫﺶ ﻳﺎﺑﺪ؟ ﺑﻠﻪ، ﭼﻨﻴﻦ ﮐﺎﻻﻫﺎﻳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ. ﻳﮏ ﻣﺜﺎﻝ ﺧﻮﺏ ﺑﺮﺍﯼ ﭼﻨﻴﻦ ﮐﺎﻻﻳﯽ ﺑﻠﻴﻂ ﺍﺗﻮﺑﻮﺱ ﺍﺳﺖ. . ﻭﻗﺘﯽ ﻭﺿﻊ ﺷﻤﺎ ﺧﻮﺏ ﻣﯽ ﺷﻮﺩ، ﺍﺣﺘﻤﺎﻝ ﺍﻳﻨﮑﻪ ﺍﺗﻮﺑﻮﺱ ﺳﻮﺍﺭ ﺷﻮﻳﺪ ﮐﺎﻫﺶ ﻣﯽ ﻳﺎﺑﺪ ﻭ ﺍﺣﺘﻤﺎﻻ ﺗﺎﮐﺴﯽ ﻳﺎ ﺍﺗﻮﻣﺒﻴﻞ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺳﻮﺍﺭ ﺧﻮﺍﻫﻴﺪ ﺷﺪ. ﺑﻪ ﺍﻳﻦ ﮐﺎﻻﻫﺎ ﮐﺎﻻﯼ ﭘﺴﺖ ﮔﻔﺘﻪ ﻣﯽ ﺷﻮﺩ.

ﺗﻐﻴﻴﺮ ﺩﺭ ﺩﺭﺁﻣﺪ ﺍﻣﺎ ﻣﻨﺠﺮ ﺑﻪ ﺣﺮﮐﺖ ﺑﺮ ﺭﻭﯼ ﻧﻤﻮﺩﺍﺭ ﺗﻘﺎﺿﺎ ﻧﻤﯽ ﺷﻮﺩ ﺑﻠﮑﻪ ﺁﻥ ﺭﺍ ﺟﺎﺑﺠﺎ ﻣﯽ ﮐﻨﺪ. ﺍﻓﺰﺍﻳﺶ ﺩﺭﺁﻣﺪ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﺭﺍﺳﺖ ﻭ ﮐﺎﻫﺶ ﺁﻥ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﭼﭗ ﺟﺎﺑﺠﺎ ﻣﯽ ﮐﻨﺪ. 

ﻗﻴﻤﺖ ﮐﺎﻻﻫﺎﯼ ﻣﺮﺗﺒﻂ: ﻓﺮﺽ ﮐﻨﻴﺪ ﻗﻴﻤﺖ ﻳﺨﻤﮏ ﭘﺎﻳﻴﻦ ﺑﻴﺎﻳﺪ. ﺑﻨﺎﺑﺮ ﻗﺎﻧﻮﻥ ﺗﻘﺎﺿﺎ، ﻣﻴﺰﺍﻥ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﻳﺨﻤﮏ ﺑﺎﻻ ﺧﻮﺍﻫﺪ ﺭﻓﺖ. ﺍﻳﻦ ﭼﻪ ﺍﺛﺮﯼ ﺑﺮ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺑﺴﺘﻨﯽ ﺧﻮﺍﻫﺪ ﮔﺬﺍﺷﺖ؟ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺑﺴﺘﻨﯽ ﭘﺎﻳﻴﻦ ﺧﻮﺍﻫﺪ ﺁﻣﺪ. ﻳﺨﻤﮏ ﻭ ﺑﺴﺘﻨﯽ ﺭﺍ ﮐﺎﻻﻫﺎﯼ ﺟﺎﻳﮕﺰﻳﻦ (Substitute Goods) ﻣﯽ ﻧﺎﻣﻨﺪ. ﻫﺮ ﮔﺎﻩ ﺑﺎﻻ ﺭﻓﺘﻦ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﻳﮏ ﮐﺎﻻ ﻣﻨﺠﺮ ﺑﻪ ﮐﺎﻫﺶ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﮐﺎﻻﯼ ﺩﻳﮕﺮﯼ ﺷﻮﺩ، ﺁﻥ ﮐﺎﻻﻫﺎ ﺭﺍ ﮐﺎﻻﯼ ﺟﺎﻳﮕﺰﻳﻦ ﻣﯽ ﻧﺎﻣﻨﺪ. ﮐﺎﻻﻫﺎﯼ ﺟﺎﻳﮕﺰﻳﻦ ﻣﻌﻤﻮﻻ ﮐﺎﻻﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﻪ ﺟﺎﯼ ﻳﮑﺪﻳﮕﺮ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﺷﻮﻧﺪ. ﻣﺜﻞ ﻫﺎﺕ ﺩﺍﮒ ﻭ ﻫﻤﺒﺮﮔﺮ ﻳﺎ ﺑﻠﻴﻂ ﺳﻴﻨﻤﺎ ﻭ ﺍﺟﺎﺭﻩ ﻭﻳﺪﺋﻮ.

ﺣﺎﻻ ﻓﺮﺽ ﮐﻨﻴﺪ ﻗﻴﻤﺖ ﺑﻨﺰﻳﻦ ﺭﺷﺪ ﭼﺸﻤﮕﻴﺮﯼ ﺑﮑﻨﺪ، ﭼﻪ ﺍﺗﻔﺎﻗﯽ ﻣﯽ ﺍﻓﺘﺪ؟ ﻃﺒﻴﻌﺘﺎ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺑﻨﺰﻳﻦ ﮐﺎﻫﺶ ﻣﯽ ﻳﺎﺑﺪ. ﺍﻳﻦ ﺍﺗﻔﺎﻕ ﭼﻪ ﺗﺎﺛﻴﺮﯼ ﺑﺮ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺍﺗﻮﻣﺒﻴﻞ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ؟ ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﺑﺎ ﺑﺎﻻ ﺭﻓﺘﻦ ﻗﻴﻤﺖ ﺑﻨﺰﻳﻦ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺧﺮﻳﺪ ﺍﺗﻮﻣﺒﻴﻞ ﻫﻢ ﮐﺎﻫﺶ ﻳﺎﺑﺪ ﻭ ﺗﻌﺪﺍﺩ ﺑﻴﺸﺘﺮﯼ ﺑﻪ ﻓﮑﺮ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻭﺳﺎﻳﻞ ﻧﻘﻠﻴﻪ ﻋﻤﻮﻣﯽ ﺑﻴﻔﺘﻨﺪ. ﺑﻨﺰﻳﻦ ﻭ ﺍﺗﻮﻣﺒﻴﻞ ﮐﺎﻻﻫﺎﯼ ﻣﮑﻤﻞ ﻳﮑﺪﻳﮕﺮﻧﺪ. ﻫﺮ ﮔﺎﻩ ﺑﺎﻻ ﺭﻓﺘﻦ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﻳﮏ ﮐﺎﻻ ﻣﻨﺠﺮ ﺑﻪ ﺑﺎﻻ ﺭﻓﺘﻦ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﮐﺎﻻﯼ ﺩﻳﮕﺮﯼ ﺷﻮﺩ، ﺁﻥ ﺩﻭ ﮐﺎﻻ ﺭﺍ ﮐﺎﻻﯼ ﻣﮑﻤﻞ (Complement Goods) ﻣﯽ ﻧﺎﻣﻨﺪ.

ﮔﻔﺘﻴﻢ ﮐﻪ ﺍﻓﺰﺍﻳﺶ ﻗﻴﻤﺖ ﮐﺎﻻﯼ ﺟﺎﻳﮕﺰﻳﻦ ﻣﻨﺠﺮ ﺑﻪ ﺍﻓﺰﺍﻳﺶ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺧﻮﺩ ﮐﺎﻻ ﻣﯽ ﮔﺮﺩﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﻓﺰﺍﻳﺶ ﻗﻴﻤﺖ ﮐﺎﻻﯼ ﺟﺎﻳﮕﺰﻳﻦ، ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﺭﺍﺳﺖ ﻭ ﮐﺎﻫﺶ ﻗﻴﻤﺖ ﺁﻥ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﭼﭗ ﺟﺎﺑﺠﺎ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﮔﻔﺘﻴﻢ ﮐﻪ ﺍﻓﺰﺍﻳﺶ ﻗﻴﻤﺖ ﮐﺎﻻﯼ ﻣﮑﻤﻞ ﻣﻨﺠﺮ ﺑﻪ ﮐﺎﻫﺶ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺧﻮﺩ ﮐﺎﻻ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﻓﺰﺍﻳﺶ ﻗﻴﻤﺖ ﮐﺎﻻﯼ ﻣﮑﻤﻞ ﻣﻮﺟﺐ ﺟﺎﺑﺠﺎﻳﯽ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺑﻪ ﺳﻤﺖ ﭼﭗ ﻭ ﮐﺎﻫﺶ ﻗﻴﻤﺖ ﮐﺎﻻﯼ ﻣﮑﻤﻞ ﻣﻮﺟﺐ ﺟﺎﺑﺠﺎﻳﯽ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺑﻪ ﺳﻤﺖ ﺭﺍﺳﺖ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﭘﻴﺶ ﺑﻴﻨﯽ ﻭﺿﻊ ﺁﻳﻨﺪﻩ: ﭘﻴﺶ ﺑﻴﻨﯽ ﺷﻤﺎ ﺍﺯ ﻭﺿﻌﻴﺖ ﺗﺎﻥ ﺩﺭ ﺁﻳﻨﺪﻩ ﺩﺭ ﻣﻴﺰﺍﻥ ﺗﻘﺎﺿﺎ ﻣﻮﺛﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺍﮔﺮ ﮔﻤﺎﻥ ﺑﺒﺮﻳﺪ ﺁﻳﻨﺪﻩ ﺧﻮﺑﯽ ﭘﻴﺶ ﺭﻭ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ، ﺗﻘﺎﺿﺎﯼ ﺗﺎﻥ ﺍﻓﺰﺍﻳﺶ ﺧﻮﺍﻫﺪ ﻳﺎﻓﺖ. ﺍﮔﺮ ﮔﻤﺎﻥ ﺑﺒﺮﻳﺪ ﺁﻳﻨﺪﻩ ﺧﻮﺑﯽ ﻧﺨﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ، ﺗﻘﺎﺿﺎﯼ ﺗﺎﻥ ﮐﺎﻫﺶ ﺧﻮﺍﻫﺪ ﻳﺎﻓﺖ.

ﺳﻴﺎﺳﺘﮕﺰﺍﺭﺍﻥ ﻫﻤﻮﺍﺭﻩ ﻣﺎﻳﻠﻨﺪ ﻣﺼﺮﻑ ﺳﻴﮕﺎﺭ ﺭﺍ ﮐﺎﻫﺶ ﺑﺪﻫﻨﺪ ﭼﺮﺍ ﮐﻪ ﻣﺼﺮﻑ ﺳﻴﮕﺎﺭ ﺍﺛﺮﺍﺕ ﺯﻳﺎﻥ ﺑﺎﺭ ﺑﺴﻴﺎﺭﯼ ﺑﺮ ﺍﻓﺮﺍﺩ ﻭ ﺟﺎﻣﻌﻪ ﻣﯽ ﮔﺬﺍﺭﺩ. ﺩﻭ ﺭﺍﻩ ﺑﺮﺍﯼ ﮐﺎﻫﺶ ﺗﻘﺎﺿﺎﯼ ﺳﻴﮕﺎﺭ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻳﮏ ﺭﺍﻩ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺭﺍ ﺟﺎﺑﻪ ﺟﺎ ﮐﺮﺩﻩ، ﺑﻪ ﺳﻤﺖ ﭼﭗ ﻣﻨﺘﻘﻞ ﮐﻨﻴﻢ. ﺭﺍﻩ ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻣﺘﺪﺍﺩ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺑﻪ ﺳﻤﺖ ﺑﺎﻻ ﺣﺮﮐﺖ ﮐﻨﻴﻢ.

ﺑﺮﺍﯼ ﺟﺎﺑﺠﺎ ﮐﺮﺩﻥ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺳﻴﮕﺎﺭ ﻣﯽ ﺗﻮﺍﻥ ﺍﺯ ﺳﻴﺎﺳﺖ ﻫﺎﻳﯽ ﻫﻤﭽﻮﻥ ﺗﺒﻠﻴﻐﺎﺕ ﻫﺸﺪﺍﺭ ﺩﻫﻨﺪﻩ ﺍﺟﺒﺎﺭﯼ ﺑﺮ ﺭﻭﯼ ﭘﺎﮐﺖ ﻫﺎﯼ ﺳﻴﮕﺎﺭ، ﻣﻤﻨﻮﻉ ﮐﺮﺩﻥ ﺗﺒﻠﻴﻐﺎﺕ ﺳﻴﮕﺎﺭ ﺩﺭ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻭ... ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ. ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﺳﻴﺎﺳﺖ ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﻣﻮﺟﺐ ﮐﺎﻫﺶ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺳﻴﮕﺎﺭ ﺩﺭ ﻫﺮ ﻗﻴﻤﺘﯽ ﺍﺯ ﻃﺮﻳﻖ ﺟﺎﺑﻪ ﺟﺎ ﮐﺮﺩﻥ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺷﻮﻧﺪ.

ﺭﺍﻩ ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻗﻴﻤﺖ ﺳﻴﮕﺎﺭ ﺍﻓﺰﺍﻳﺶ ﻳﺎﺑﺪ ﻳﺎ ﺑﻪ ﻋﺒﺎﺭﺗﯽ ﺑﺮ ﺭﻭﯼ ﻣﻨﺤﻨﯽ ﺗﻘﺎﺿﺎ ﺑﻪ ﺳﻤﺖ ﺑﺎﻻ ﺣﺮﮐﺖ ﮐﻨﻴﻢ. ﺩﻭﻟﺖ ﻫﺎ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺑﺎ ﺍﻓﺰﻭﺩﻥ ﺑﺮ ﻣﺎﻟﻴﺎﺕ، ﻣﻮﺟﺐ ﺍﻓﺰﺍﻳﺶ ﻗﻴﻤﺖ ﺳﻴﮕﺎﺭ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﮐﺎﻫﺶ ﺗﻘﺎﺿﺎ ﺑﺮﺍﯼ ﺳﻴﮕﺎﺭ ﺷﻮﻧﺪ. 

ﺩﺭ ﺑﺮﺭﺳﯽ ﻫﺎﻳﯽ ﮐﻪ ﺍﻧﺠﺎﻡ ﺷﺪﻩ ﺍﺳﺖ، ﺍﻗﺘﺼﺎﺩﺩﺍﻧﺎﻥ ﺩﺭﻳﺎﻓﺘﻪ ﺍﻧﺪ ﮐﻪ ﺍﻓﺰﺍﻳﺶ ﺩﻩ ﺩﺭﺻﺪﯼ ﻗﻴﻤﺖ ﺳﻴﮕﺎﺭ ﻣﻮﺟﺐ ﮐﺎﻫﺶ ۴ ﺩﺭﺻﺪﯼ ﺗﻘﺎﺿﺎ ﻣﯽ ﺷﻮﺩ. ﻋﻤﺪﻩ ﮐﺴﺎﻧﯽ ﮐﻪ ﻧﺴﺒﺖ ﺑﻪ ﺍﻳﻦ ﺍﻓﺰﺍﻳﺶ ﺣﺴﺎﺳﻨﺪ ﻭ ﺑﺎ ﺍﻓﺰﺍﻳﺶ ﻗﻴﻤﺖ ﻣﺼﺮﻑ ﺷﺎﻥ ﺭﺍ ﮐﻢ ﻣﯽ ﮐﻨﻨﺪ، ﻧﻮﺟﻮﺍﻧﺎﻥ ﻫﺴﺘﻨﺪ.

ﺑﮕﺬﺍﺭﻳﺪ ﮐﻤﯽ ﻣﺴﺌﻠﻪ ﺭﺍ ﺑﻴﺸﺘﺮ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ ﻭ ﻣﻄﺎﻟﺒﯽ ﮐﻪ ﺧﻮﺍﻧﺪﻩ ﺍﻳﻢ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﮐﺎﺭ ﺑﺒﻨﺪﻳﻢ. ﺑﺮﺧﯽ ﺍﺩﻋﺎ ﻣﯽ ﮐﻨﻨﺪ ﮐﻪ ﺍﻓﺰﺍﻳﺶ ﻗﻴﻤﺖ ﺳﻴﮕﺎﺭ ﻣﻮﺟﺐ ﻣﯽ ﺷﻮﺩ ﮐﻪ ﻣﺼﺮﻑ ﮐﻨﻨﺪﮔﺎﻥ ﺑﻪ ﺳﻤﺖ ﻣﺼﺮﻑ ﺣﺸﻴﺶ ﺑﺮﻭﻧﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺁﻧﻬﺎ ﻣﺪﻋﯽ ﺍﻧﺪ ﮐﻪ ﺍﻳﻦ ﺩﻭ ﮐﺎﻻ ﺟﺎﻳﮕﺰﻳﻦ ﻳﮑﺪﻳﮕﺮ ﻫﺴﺘﻨﺪ. ﻣﻄﺎﻟﻌﺎﺕ ﺑﺴﻴﺎﺭﯼ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﻨﺪ ﮐﻪ ﺳﻴﮕﺎﺭ ﺩﺭﻭﺍﺯﻩ ﺍﯼ ﺑﻪ ﺳﻤﺖ ﻣﺼﺮﻑ ﺩﻳﮕﺮ ﻣﻮﺍﺩ ﻣﺨﺪﺭ ﺍﺳﺖ ﻭ ﮐﺎﻫﺶ ﻗﻴﻤﺖ ﺳﻴﮕﺎﺭ ﺭﺍﺑﻄﻪ ﻣﻌﻨﺎﺩﺍﺭﯼ ﺑﺎ ﺍﻓﺰﺍﻳﺶ ﻣﺼﺮﻑ ﺣﺸﻴﺶ ﺩﺍﺭﺩ. ﺑﻪ ﻋﺒﺎﺭﺗﯽ ﺣﺸﻴﺶ ﻭ ﺳﻴﮕﺎﺭ ﮐﺎﻻﻫﺎﻳﯽ ﻣﮑﻤﻞ ﻫﺴﺘﻨﺪ ﻭ ﻧﻪ ﮐﺎﻻﻫﺎﯼ ﺟﺎﻳﮕﺰﻳﻦ.
