ﻣﻘﺪﻣﻪ

OS X
ﻣﮏ (Mac OS X) ﮐﻪ ﺍﺯ ﻣﻴﺎﻧﻪ ﺳﺎﻝ ۲۰۱۲ ﻣﻴﻼﺩﯼ ﻭ ﭘﺲ ﺍﺯ ﻋﺮﺿﻪ ﺁﺧﺮﻳﻦ ﻧﺴﺨﻪ ﺍﺵ (ﺷﻴﺮ ﮐﻮﻫﯽ) ﺑﺎ ﻧﺎﻡ OS X ﺷﻨﺎﺧﺘﻪ ﻣﯽ ﺷﻮﺩ، ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺍﺻﻠﯽ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺗﻮﻟﻴﺪﯼ ﺷﺮﮐﺖ ﺍﭘﻞ ﺍﺳﺖ. ﺳﻴﺴﺘﻢ ﻋﺎﻣﻠﯽ ﺑﺮ ﭘﺎﻳﻪ ﻳﻮﻧﻴﮑﺲ ﺑﺎ ﻫﺴﺘﻪ ﺍﯼ ﻣﺘﺸﮑﻞ ﺍﺯ ﮐﺪﻫﺎﯼ ﺑﺎﺯ ﻭ ﺑﺴﺘﻪ.

ﺍﭘﻞ ﺍﻭﻟﻴﻦ ﻧﺴﺨﻪ ﺩﺳﮑﺘﺎﭘﺶ ﺑﺎ ﻧﺎﻡ ﭼﻴﺘﺎ (Mac OS X 10.0) ﺭﺍ ﺩﺭ ۲۴ ﻣﺎﺭﺱ ۲۰۰۱ ﺭﺳﻤﺎ ﻋﺮﺿﻪ ﮐﺮﺩ. ﺍﻟﺒﺘﻪ ﭘﻴﺶ ﺍﺯ ﺁﻥ ﻧﻴﺰ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺍﺯ ﺳﺎﻝ ۱۹۸۴ ﺗﻮﺳﻂ ﺍﭘﻞ ﻋﺮﺿﻪ ﺷﺪﻩ ﺑﻮﺩ ﺍﻣﺎ ﺍﻭﻟﻴﻦ ﻧﺴﺨﻬﺎﯼ ﮐﻪ ﻧﺎﻡ ﻣﮏ ﺭﺍ ﺑﻪ ﺧﻮﺩ ﮔﺮﻓﺖ ﻭ ﺑﻪ ﺷﮑﻠﯽ ﮐﻪ ﺍﻣﺮﻭﺯ «ﻣﮏ» ﺭﺍ ﻣﯽ ﺷﻨﺎﺳﻴﻢ، ﭘﺲ ﺍﺯ ﺑﺎﺯﮔﺸﺖ ﻣﺠﺪﺩ ﺍﺳﺘﻴﻮ ﺟﺎﺑﺰ ﺑﻪ ﺍﭘﻞ ﺩﺭ ﺳﺎﻝ ۲۰۰۱ ﻋﺮﺿﻪ ﺷﺪ. ﺁﺧﺮﻳﻦ ﻧﺴﺨﻪ ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻧﻴﺰ ﺑﺎ ﻧﺎﻡ ﺷﻴﺮﮐﻮﻫﯽ (OS X 10.8)، ﺍﻭﺍﺧﺮ ﺟﻮﻻﯼ ۲۰۱۲ ﻣﻴﻼﺩﯼ ﻋﺮﺿﻪ ﺷﺪﻩ ﺍﺳﺖ.

ﺍﻟﺒﺘﻪ ﻧﺴﺨﻪ ﺑﻌﺪﯼ ﺁﻥ ﺑﺎ ﻧﺎﻡ ﻣﺎﻭﺭﻳﮑﺲ (OS X 10.9) ﻫﻢ ﻃﯽ ﮐﻨﻔﺮﺍﻧﺲ ﺗﻮﺳﻌﻪ ﺩﻫﻨﺪﮔﺎﻥ ﺍﭘﻞ ﺩﺭ ﺗﺎﺭﻳﺦ ۱۰ ﮊﻭﺋﻦ ۲۰۱۳ ﺭﻭﻧﻤﺎﻳﯽ ﺷﺪﻩ ﺍﺳﺖ ﻭ ﻗﺮﺍﺭ ﺍﺳﺖ ﺗﺎ ﭘﺎﻳﻴﺰ ۲۰۱۳ ﻧﺴﺨﻪ ﻧﻬﺎﻳﯽ ﺁﻥ ﻭﺍﺭﺩ ﺑﺎﺯﺍﺭ ﺷﻮﺩ.

ﻧﮑﺘﻪ: ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻨﮑﻪ ﺍﻳﻦ ﺩﻭﺭﻩ ﭘﻴﺶ ﺍﺯ ﻋﺮﺿﻪ ﺭﺳﻤﯽ OS X 10.9 ﺗﻬﻴﻪ ﺷﺪﻩ، ﺗﻤﺎﻣﯽ ﺩﺭﺱ ﻫﺎ ﺑﺮﺍﺳﺎﺱ ﺁﺧﺮﻳﻦ ﻧﺴﺨﻪ ﺩﺭ ﺩﺳﺘﺮﺱ ﺍﺯ ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ، ﻳﻌﻨﯽ ﺷﻴﺮ ﮐﻮﻫﯽ (OS X 10.8) ﻧﮕﺎﺭﺵ ﻭ ﺁﻣﺎﺩﻩ ﺷﺪﻩ ﺍﻧﺪ ﻭ ﺗﺼﺎﻭﻳﺮ ﻧﻴﺰ ﻣﺘﻌﻠﻖ ﺑﻪ ﻫﻤﻴﻦ ﻧﺴﺨﻪ ﻫﺴﺘﻨﺪ.

 

ﺍﻣﻨﻴﺖ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﮏ
ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﮏ ﺑﻪ ﻭﺍﺳﻄﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻫﺴﺘﻪ ﻳﻮﻧﻴﮑﺲ ﻭ ﻗﺎﺑﻠﻴﺖ ﻫﺎﯼ ﺗﻨﻈﻴﻢ ﺩﺳﺘﺮﺳﯽ ﺧﺎﺹ ﺑﻪ ﻓﺎﻳﻞ ﻫﺎ، ﺍﺯ ﻟﺤﺎﻅ ﺍﻣﻨﻴﺘﯽ ﺑﻪ ﻣﺮﺍﺗﺐ ﺑﻬﺘﺮ ﺍﺯ ﻭﻳﻨﺪﻭﺯ ﺍﺳﺖ. ﺍﺯ ﺩﻳﮕﺮ ﺩﻻﻳﻞ ﺍﻓﺰﺍﻳﺶ ﺿﺮﻳﺐ ﺍﻣﻨﻴﺘﯽ ﺩﺭ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﮏ، ﺗﻌﺪﺍﺩ ﮐﻢ ﮐﺎﺭﺑﺮﺍﻥ ﺍﺳﺖ. ﺑﺎ ﺍﻳﻦ ﻭﺟﻮﺩ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﻧﻤﯽ ﺗﻮﺍﻥ ﻣﮏ ﻭ ﻫﻴﭻ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺩﻳﮕﺮﯼ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﮐﺎﻣﻞ ﺍﻣﻦ ﺷﻤﺮﺩ. ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻨﮑﻪ ﺗﻨﻬﺎ ۷ ﺗﺎ ۱۰ ﺩﺭﺻﺪ ﮐﺎﺭﺑﺮﺍﻥ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺷﺨﺼﯽ ﺟﻬﺎﻥ ﺍﺯ ﻣﮏ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻨﺪ، ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺗﺎ ﺳﺎﻝ ﻫﺎﯼ ﮔﺬﺷﺘﻪ ﭼﻨﺪﺍﻥ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻫﮑﺮﻫﺎﯼ ﮐﺎﻣﭙﻴﻮﺗﺮﯼ ﻧﺒﻮﺩ. ﻫﮑﺮﻫﺎ ﺗﺮﺟﻴﺢ ﻣﯽ ﺩﺍﺩﻧﺪ ﻭﻗﺖ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﻫﮏ ﮐﺮﺩﻥ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ۹۰ ﺩﺭﺻﺪ ﮐﺎﺭﺑﺮﺍﻥ ﺑﺎﻗﻴﻤﺎﻧﺪﻩ ﺻﺮﻑ ﮐﻨﻨﺪ. ﺍﻣﺎ ﺩﺭ ﺳﺎﻝ ﻫﺎﯼ ﺍﺧﻴﺮ ﺗﻌﺪﺍﺩ ﺑﺪﺍﻓﺰﺍﺭﻫﺎﯼ ﺗﻮﻟﻴﺪ ﺷﺪﻩ ﺑﺮﺍﯼ ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺍﻓﺰﺍﻳﺶ ﺩﺍﺷﺘﻪ ﻭ ﮐﺎﺭﺑﺮﺍﻥ ﻣﮏ ﺑﺎﻳﺪ ﻫﺸﻴﺎﺭﺗﺮ ﻋﻤﻞ ﮐﻨﻨﺪ. ﻫﻤﭽﻨﻴﻦ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﯼ ﺍﻣﻨﻴﺘﯽ ﻣﺎﻧﻨﺪ ﺁﻧﺘﯽ ﻭﻳﺮﻭﺱ ﻫﺎ ﻭ ﻓﺎﻳﺮﻭﺍﻝ ﻫﺎ ﻧﻴﺰ ﺑﻪ ﺗﻮﺍﻥ ﺍﻣﻨﻴﺘﯽ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﻣﮏ ﻣﯽ ﺍﻓﺰﺍﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﻪ ﺑﺮﺧﯽ ﺍﺯ ﺁﻧﻬﺎ ﺍﺷﺎﺭﻩ ﺧﻮﺍﻫﺪ ﺷﺪ.

 

ﺳﻴﺎﺳﺖ ﻫﺎﯼ ﺍﭘﻞ
ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺍﺷﺎﺭﻩ ﺷﺪ، ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ، ﺍﭘﻞ ﺳﻬﻤﯽ ۷ ﺗﺎ ۱۰ ﺩﺭﺻﺪﯼ ﺩﺭ ﺑﺎﺯﺍﺭ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺷﺨﺼﯽ (ﺭﻭﻣﻴﺰﯼ ﻭ ﻟﭗ ﺗﺎﭖ) ﺩﺍﺭﺩ. ﺍﭘﻞ ﺩﺭ ﺑﺎﺯﺍﺭ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺷﺨﺼﯽ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻠﺶ ﺭﺍ ﺑﻪ ﻫﻤﺮﺍﻩ ﺳﺨﺖ ﺍﻓﺰﺍﺭ ﺳﺎﺯﮔﺎﺭ ﻭ ﺍﺧﺘﺼﺎﺻﯽ ﺧﻮﺩ ﻋﺮﺿﻪ ﻣﯽ ﮐﻨﺪ؛ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﻣﺎﻳﮑﺮﻭﺳﺎﻓﺖ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻭﻳﻨﺪﻭﺯ ﺭﺍ ﺑﺎ ﺳﺨﺖ ﺍﻓﺰﺍﺭ ﺳﺎﺧﺘﻪ ﺷﺪﻩ ﺗﻮﺳﻂ ﺷﺮﮐﺖ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺳﺎﺯﮔﺎﺭ ﮐﺮﺩﻩ ﻭ ﺁﻥ ﺭﺍ ﺑﺮﺍﯼ ﻋﺮﺿﻪ ﻣﺴﺘﻘﻴﻢ ﺑﻪ ﺑﺎﺯﺍﺭ ﻋﺮﺿﻪ ﻣﯽ ﻧﻤﺎﻳﺪ.

ﺳﻴﺎﺳﺖ ﺍﭘﻞ ﺗﻘﺮﻳﺒﺎ ﺍﺯ ﺍﺑﺘﺪﺍ ﺑﺮﺍﺳﺎﺱ ﻋﺮﺿﻪ ﻧﺮﻡ ﺍﻓﺰﺍﺭ  ﺑﻪ ﻫﻤﺮﺍﻩ ﺳﺨﺖ ﺍﻓﺰﺍﺭ ﺑﺎ ﮐﻴﻔﻴﺖ، ﺳﺎﺯﮔﺎﺭ ﻭ ﺑﺎ ﺛﺒﺎﺕ ﺑﻮﺩﻩ ﺍﺳﺖ. ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺍﭘﻞ ﻫﻤﻮﺍﺭﻩ ﺍﺯ ﻟﺤﺎﻅ ﻃﺮﺍﺣﯽ ﺍﻏﻮﺍﮔﺮ ﻭ ﺯﻳﺒﺎﻳﯽ ﺩﺭ ﮐﻨﺎﺭ ﻗﺎﺑﻠﻴﺖ ﻫﺎﯼ ﺳﺨﺖ ﺍﻓﺰﺍﺭﯼ ﻣﻨﺎﺳﺐ ﻭ ﮐﺎﻓﯽ، ﺳﺮﺁﻣﺪ ﺑﻮﺩﻩ ﺍﻧﺪ. ﻫﺮ ﭼﻨﺪ ﮐﻪ ﺍﻳﻦ ﺳﻴﺎﺳﺖ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﻧﺘﻮﺍﻧﺴﺘﻪ ﺍﭘﻞ ﺭﺍ ﺑﻪ ﺭﻗﻴﺒﯽ ﺟﺪﯼ ﺑﺮﺍﯼ ﻣﺎﻳﮑﺮﻭﺳﺎﻓﺖ ﺑﺪﻝ ﮐﻨﺪ، ﺍﻣﺎ ﻣﺤﺼﻮﻻﺕ ﺍﻳﻦ ﺷﺮﮐﺖ ﻋﻼﻗﻤﻨﺪﺍﻥ ﻭ ﺩﻭﺳﺘﺪﺍﺭﺍﻥ ﺧﻮﺩ ﺭﺍ ﺩﺍﺭﺩ.

 

ﻫﮑﻴﻨﺘﺎﺵ
ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺳﻴﺎﺳﺖ ﻓﺮﻭﺵ ﺍﭘﻞ ﺩﺭ ﻋﺮﺿﻪ ﻫﻤﺰﻣﺎﻥ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻭ ﺳﺨﺖ ﺍﻓﺰﺍﺭ، ﻫﮑﺮﻫﺎ ﺗﻮﺍﻧﺴﺘﻪ ﺍﻧﺪ ﺑﺎ ﺭﻭﺵ ﻫﺎﯼ ﺧﺎﺻﯽ ﻫﻤﭽﻮﻥ ﻣﺎﺷﻴﻦ ﻣﺠﺎﺯﯼ ﻭ... ﻧﺼﺐ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﮏ ﺭﺍ ﺭﻭﯼ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﻳﯽ ﮐﻪ ﺑﺎ ﻗﻄﻌﺎﺕ ﺳﺎﻳﺮ ﺷﺮﮐﺖ ﻫﺎ ﺍﺳﻤﺒﻞ ﺷﺪﻩ ﺍﻧﺪ، ﺑﻮﺕ ﮐﻨﻨﺪ. ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺑﻪ ﻧﺼﺐ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﮏ (ﻳﺎ ﻣﮑﻴﻨﺘﺎﺵ) ﺑﺮ ﺭﻭﯼ ﺳﺎﻳﺮ ﺳﺨﺖ ﺍﻓﺰﺍﺭﻫﺎﯼ ﻣﻮﺟﻮﺩ ﺩﺭ ﺑﺎﺯﺍﺭ «ﻫﮑﻴﻨﺘﺎﺵ» ﮔﻔﺘﻪ ﻣﯽ ﺷﻮﺩ. ﺍﻣﺮﻭﺯﻩ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﺑﺴﻴﺎﺭ ﺭﻭﺍﺝ ﭘﻴﺪﺍ ﮐﺮﺩﻩ ﻭ ﺍﻓﺮﺍﺩ ﻋﻼﻗﻤﻨﺪ ﺑﻪ ﺍﺟﺮﺍﯼ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﮏ ﺑﺮ ﺭﻭﯼ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﻣﺘﻔﺮﻗﻪ ﻫﺮ ﺭﻭﺯ ﺑﻴﺸﺘﺮ ﻣﯽ ﺷﻮﻧﺪ.

ﺁﻳﺎ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻳﺎ ﺳﺨﺖ ﺍﻓﺰﺍﺭﻫﺎﯼ ﻣﮏ ﻣﺰﻳﺖ ﺧﺎﺻﯽ ﻧﺴﺒﺖ ﺑﻪ ﺳﺎﻳﺮ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺷﺨﺼﯽ ﺩﺍﺭﻧﺪ؟

ﭘﺎﺳﺦ ﺍﻳﻦ ﺳﻮﺍﻝ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩ ﻣﺨﺘﻠﻒ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﮐﺎﺭﺑﺮﯼ، ﻧﻘﺸﯽ ﺍﺳﺎﺳﯽ ﺩﺭ ﺍﻧﺘﺨﺎﺏ ﺩﺭﺳﺖ ﻭ ﻣﻨﺎﺳﺐ ﺑﻪ ﻫﻨﮕﺎﻡ ﺧﺮﻳﺪ ﺩﺍﺭﺩ. ﺩﺭ ﺍﻳﺮﺍﻥ ﺗﻌﺪﺍﺩ ﮐﺎﺭﺑﺮﺍﻥ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﮏ ﺩﺭ ﻣﻘﺎﻳﺴﻪ ﺑﺎ ﮐﺎﺭﺑﺮﺍﻥ ﻭﻳﻨﺪﻭﺯ ﺑﺴﻴﺎﺭ ﮐﻢ ﺍﺳﺖ. ﺷﺎﻳﺪ ﻣﻬﻤﺘﺮﻳﻦ ﺩﻟﻴﻞ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﻗﻴﻤﺖ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺍﭘﻞ ﻭ ﻋﺪﻡ ﺍﻣﮑﺎﻥ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺳﺨﺖ ﺍﻓﺰﺍﺭﻫﺎﯼ ﺟﺎﻧﺒﯽ ﺳﺎﺯﮔﺎﺭ ﺑﺎ ﺁﻥ، ﺩﺭ ﻣﻘﺎﻳﺴﻪ ﺑﺎ ﺳﺎﻳﺮ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﻣﻮﺟﻮﺩ ﺩﺭ ﺑﺎﺯﺍﺭ ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﻋﺪﻡ ﻭﺟﻮﺩ ﭘﺸﺘﻴﺒﺎﻧﯽ ﺭﺳﻤﯽ ﺍﺯ ﺳﻤﺖ ﺷﺮﮐﺖ ﺍﭘﻞ ﻧﻴﺰ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﻟﻴﻠﯽ ﺩﻳﮕﺮ ﺑﻪ ﺣﺴﺎﺏ ﺁﻳﺪ.

ﻫﺮ ﭼﻨﺪ ﮐﻪ ﻗﻴﻤﺖ ﺑﺎﻻﺗﺮ ﺍﻟﺰﺍﻣﺎ ﺑﻪ ﻣﻌﻨﯽ ﮐﻴﻔﻴﺖ ﻳﺎ ﻗﺎﺑﻠﻴﺖ ﻫﺎﯼ ﺑﻴﺸﺘﺮ ﻧﻴﺴﺖ، ﺍﻣﺎ ﺧﻴﻞ ﻭﺳﻴﻌﯽ ﺍﺯ ﮐﺎﺭﺑﺮﺍﻥ ﺩﺭ ﺳﻄﻮﺡ ﮐﺎﺭﯼ ﻣﻘﺪﻣﺎﺗﯽ ﺗﺎ ﭘﻴﺸﺮﻓﺘﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﺎﺭﺑﺮ ﻣﮏ ﺑﺎﺷﻨﺪ. ﺍﻟﺒﺘﻪ ﮐﻴﻔﻴﺖ ﺳﺨﺖ ﺍﻓﺰﺍﺭ ﻣﺤﺼﻮﻻﺕ ﺍﭘﻞ ﺩﺭ ﻣﻘﺎﻳﺴﻪ ﺑﺎ ﻣﺤﺼﻮﻻﺕ ﺳﺎﻳﺮ ﺷﺮﮐﺖ ﻫﺎ ﺑﻪ ﻭﺿﻮﺡ ﻗﺎﺑﻞ ﻟﻤﺲ ﺍﺳﺖ، ﺍﻣﺎ ﺍﻳﻦ ﺷﻴﻮﻩ ﮐﺎﺭﺑﺮﺩ ﻭ ﻣﻴﺰﺍﻥ ﻫﺰﻳﻨﻪ ﺷﻤﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺼﺮﻑ ﮐﻨﻨﺪﻩ ﺍﺳﺖ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﺍﻧﺘﺨﺎﺏ ﻧﻬﺎﻳﯽ ﻣﯽ ﺷﻮﺩ. ﺑﻪ ﻫﺮ ﺷﮑﻞ ﺍﮔﺮ ﺑﻪ ﺗﺎﺯﮔﯽ ﮐﺎﺭﺑﺮ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﮏ ﺷﺪﻩ ﺍﻳﺪ ﻳﺎ ﻗﺼﺪ ﻣﻬﺎﺟﺮﺕ ﺑﻪ ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺭﺍ ﺩﺍﺭﻳﺪ، ﮔﺬﺭﺍﻧﺪﻥ ﺩﺭﺱ ﻫﺎﯼ ﺍﻳﻦ ﺩﻭﺭﻩ ﺁﻣﻮﺯﺷﯽ ﺑﻪ ﺷﻤﺎ ﺗﻮﺻﻴﻪ ﻣﯽ ﺷﻮﺩ.
