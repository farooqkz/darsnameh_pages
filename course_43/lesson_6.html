ﺩﺭﺱ ﻫﻔﺘﻢ- ﺑﻴﺎﻥ ﻣﻔﺎﻫﻴﻢ ﺍﺑﺘﺪﺍﻳﯽ ﺩﺭ ﺗﺮﮐﻴﺐ ﺑﻨﺪﯼ ﺗﺼﻮﻳﺮ

ﻣﻔﻬﻮﻡ ﺭﻳﺘﻢ ﻭ ﺗﮑﺮﺍﺭ
ﺗﮑﺮﺍﺭ ﻣﻨﻈﻢ ﻳﮏ ﻋﻨﺼﺮ، ﺍﺻﻄﻼﺣﺎ ﺭﻳﺘﻢ ﺧﻮﺍﻧﺪﻩ ﻣﯽ ﺷﻮﺩ. ﺭﻳﺘﻢ ﺩﺭ ﺑﻌﻀﯽ ﻫﻨﺮﻫﺎ ﻣﺎﻧﻨﺪ ﻣﻮﺳﻴﻘﯽ ﻋﻨﺼﺮﯼ ﺫﺍﺗﯽ ﺍﺳﺖ ﮐﻪ ﺑﺪﻭﻥ ﻭﺟﻮﺩ ﺁﻥ ﻫﻴﭻ ﺁﻫﻨﮕﯽ ﺳﺎﺧﺘﻪ ﻧﻤﯽ ﺷﻮﺩ. ﺍﻣﺎ ﺩﺭ ﻫﻨﺮﻫﺎﯼ ﺗﺠﺴﻤﯽ ﺍﻳﺠﺎﺩ ﺭﻳﺘﻢ ﻳﮑﯽ ﺍﺯ ﺗﮑﻨﻴﮏ ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﻃﺮﺍﺡ ﺩﺭ ﺍﻳﺠﺎﺩ ﻣﻌﻨﺎﻫﺎﻳﯽ ﺧﺎﺹ ﺩﺭ ﺗﺼﻮﻳﺮ ﮐﻤﮏ ﮐﻨﺪ. ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺭﻳﺘﻢ ﺑﺎﻳﺪ ﻳﮏ ﻋﻨﺼﺮ ﭘﺎﻳﻪ ﺍﯼ ﺍﻳﺠﺎﺩ ﮐﺮﺩ. ﺍﻳﻦ ﻋﻨﺼﺮ ﭘﺎﻳﻪ ﺍﯼ (ﻳﺎ ﻡُﻭﺗﻴﻒ) ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﺯ ﺷﮑﻠﯽ ﺳﺎﺩﻩ ﻳﺎ ﭘﻴﭽﻴﺪﻩ ﺗﺮ ﺩﺭﺳﺖ ﺷﺪﻩ ﺑﺎﺷﺪ. ﺗﮑﺮﺍﺭ ﻣﻨﻈﻢ ﺍﻳﻦ ﻋﻨﺼﺮ ﭘﺎﻳﻪ ﺍﯼ ﺩﺭ ﺗﺼﻮﻳﺮ ﻳﺎ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﮐﻠﻴﺖ ﺗﺼﻮﻳﺮ ﺭﺍ ﺷﮑﻞ ﻣﯽ ﺩﻫﺪ ﻭ ﻳﺎ ﺯﻣﻴﻨﻪ ﺍﯼ ﺑﺮﺍﯼ ﺣﻀﻮﺭ ﺩﻳﮕﺮ ﻋﻨﺎﺻﺮ ﺗﺼﻮﻳﺮﯼ ﺭﺍ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ ﺁﻭﺭﺩ.

ﺑﺮﺍﯼ ﺭﻭﺷﻦ ﺷﺪﻥ ﻣﻄﻠﺐ ﻭ ﺍﻫﻤﻴﺖ ﻣﻌﻨﺎﯼ ﺭﻳﺘﻢ ﺩﺭ ﺗﺼﻮﻳﺮﺳﺎﺯﯼ ﺩﻭ ﻣﺜﺎﻝ ﺩﺭ ﺍﻳﻨﺠﺎ ﺍﺭﺍﻳﻪ ﻣﯽ ﺷﻮﺩ.

ﻧﻤﻮﻧﻪ ﺍﻭﻝ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﺩﺭ ﻫﻨﺮﻫﺎﯼ ﺳﻨﺘﯽ ﺍﻳﺮﺍﻧﯽ ﭘﻴﺪﺍ ﮐﺮﺩ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﮐﺎﺷﯽ ﮐﺎﺭﯼ ﻫﺎ، ﮔﭻ ﺑﺮﯼ ﻫﺎ ﻭ ﺍﻣﺜﺎﻟﻬﻢ ﺍﺯ ﺗﮑﺮﺍﺭ ﻫﻨﺪﺳﯽ ﻳﮏ ﻋﻨﺼﺮ ﭘﺎﻳﻪ ﺍﯼ ﺷﮑﻞ ﮔﺮﻓﺘﻪ ﺍﻧﺪ. ﺩﺭ ﻫﻨﺮﻫﺎﯼ ﺳﻨﺘﯽ ﻣﻌﻤﻮﻻ ﻳﮏ ﻋﻨﺼﺮ (ﻧﻘﺶ) ﭘﺎﻳﻪ ﺍﯼ ﺑﺮﺍﺳﺎﺱ ﻳﮏ ﻫﻨﺪﺳﻪ ﺩﻗﻴﻖ ﻭ ﻣﻨﻈﻢ ﺑﻪ ﺗﻨﺎﻭﺏ ﺗﮑﺮﺍﺭ ﻣﯽ ﺷﻮﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺗﺼﻮﻳﺮ (ﻧﻘﺶ) ﺯﻳﺒﺎﯼ ﺩﺍﺧﻞ ﮔﻨﺒﺪ ﻣﺴﺠﺪ ﺷﻴﺦ ﻟﻄﻒ ﺍﻟﻠﻪ ﺣﺎﺻﻞ ﺗﮑﺮﺍﺭ ﺭﻳﺘﻤﻴﮏ ﻳﮑﯽ ﺍﺯ ﻧﻘﻮﺵ ﺳﻨﺘﯽ ﺍﺳﺖ. ﺍﻳﻦ ﻧﻘﺶ ﭘﺎﻳﻪ ﺍﯼ ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻨﮑﻪ ﺑﺎ ﭼﺮﺧﺶ ﺣﻮﻝ ﻣﺮﮐﺰ ﺗﮑﺮﺍﺭ ﻣﯽ ﺷﻮﺩ، ﺑﺎ ﻧﺰﺩﻳﮏ ﺗﺮ ﺷﺪﻥ ﺑﻪ ﻣﺮﮐﺰ ﮐﻮﭼﮏ ﺗﺮ ﻭ ﮐﻮﭼﮏ ﺗﺮ ﻣﯽ ﺷﻮﺩ. ﺍﻳﻦ ﻧﻤﻮﻧﻪ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ ﺩﺭ ﮐﺎﺭﺑﺮﺩ ﺭﻳﺘﻢ ﻣﯽ ﺗﻮﺍﻥ ﺍﺑﺪﺍﻋﺎﺕ ﺯﻳﺎﺩﯼ ﺑﻪ ﮐﺎﺭ ﺑﺴﺖ.

ﻧﻤﻮﻧﻪ ﺩﻭﻡ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺭﻳﺘﻢ ﺩﺭ ﻃﺮﺍﺣﯽ ﭘﺎﺭﭼﻪ ﺍﺳﺖ. ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﻳﮏ ﻋﻨﺼﺮ ﭘﺎﻳﻪ ﺍﯼ ﻭ ﺗﮑﺮﺍﺭ ﺁﻥ ﻣﯽ ﺗﻮﺍﻥ ﺗﺼﺎﻭﻳﺮ ﺑﺪﻳﻊ ﻭ ﺯﻳﺒﺎﻳﯽ ﺭﺍ ﺑﺮ ﭘﺎﺭﭼﻪ ﻫﺎ ﺍﻳﺠﺎﺩ ﮐﺮﺩ. ﺩﺭ ﭼﺎﭖ ﭘﺎﺭﭼﻪ ﻫﺎ ﻣﻌﻤﻮﻻ ﻳﮏ ﻧﻘﺶ ﭘﺎﻳﻪ ﺍﯼ ﺑﻪ ﺻﻮﺭﺕ ﺭﻳﺘﻤﻴﮏ (ﻫﻨﺪﺳﯽ ﻳﺎ ﻏﻴﺮ ﻫﻨﺪﺳﯽ) ﻣﺮﺗﺒﺎ ﺗﮑﺮﺍﺭ ﻣﯽ ﺷﻮﺩ.

ﻧﮑﺘﻪ ﺩﻳﮕﺮﯼ ﮐﻪ ﻫﻨﮕﺎﻡ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺭﻳﺘﻢ ﻣﯽ ﺗﻮﺍﻥ ﻣﺪ ﻧﻈﺮ ﺩﺍﺷﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻋﻨﺼﺮ ﭘﺎﻳﻪ ﺍﯼ ﻫﻨﮕﺎﻡ ﺗﮑﺮﺍﺭ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﻧﺪﮐﯽ ﺗﻐﻴﻴﺮ ﮐﻨﺪ، ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﻋﻨﺼﺮ ﭘﺎﻳﻪ ﺍﯼ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﺗﺪﺭﻳﺞ ﮐﻮﭼﮏ ﺗﺮ ﻳﺎ ﺑﺰﺭﮒ ﺗﺮ ﺷﻮﺩ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺎﻋﺚ ﺍﻳﺠﺎﺩ ﺗﻨﻮﻉ ﺩﺭ ﮐﻠﻴﺖ ﺗﺼﻮﻳﺮ ﺧﻮﺍﻫﺪ ﺷﺪ. ﻧﮑﺘﻪ ﻗﺎﺑﻞ ﺗﻮﺟﻪ ﺩﺭ ﺭﻳﺘﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻋﻨﺼﺮ ﭘﺎﻳﻪ ﺍﯼ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻣﯽ ﺗﻮﺍﻧﺪ ﮐﻠﻴﺖ ﺑﺰﺭﮒ ﻳﮏ ﺗﺼﻮﻳﺮ ﺭﺍ ﺍﻳﺠﺎﺩ ﮐﻨﺪ.

 

ﻣﻔﻬﻮﻡ ﻓﻀﺎﯼ ﻣﺜﺒﺖ ﻭ ﻣﻨﻔﯽ
ﻣﻨﻈﻮﺭ ﺍﺯ ﻓﻀﺎﯼ ﻣﻨﻔﯽ، ﻓﻀﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺑﻴﻦ ﻋﻨﺎﺻﺮ ﻣﻮﺟﻮﺩ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﺑﻮﺟﻮﺩ ﻣﯽ ﺁﻳﺪ ﻭ ﺍﻳﻦ ﻓﻀﺎ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻋﻨﺼﺮ ﻭﺍﺟﺪ ﻣﻌﻨﺎ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﺩﺭ ﻣﻮﺍﺟﻪ ﺑﺎ ﻳﮏ ﺗﺼﻮﻳﺮ ﺁﻧﭽﻪ ﺗﺎﺛﻴﺮ ﻋﻤﺪﯼ ﺑﺮ ﺑﻴﻨﻨﺪﻩ ﻣﯽ ﮔﺬﺍﺭﺩ ﻋﻨﺼﺮ ﻣﺜﺒﺖ ﺍﺳﺖ ﻭ ﻣﺎﺑﻘﯽ ﻋﻨﺼﺮ ﻣﻨﻔﯽ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺭﻭﺩ. ﺍﻣﺎ ﺩﺭ ﻣﻮﺍﺭﺩﯼ ﻓﻀﺎﯼ ﻣﻨﻔﯽ ﻣﯽ ﺗﻮﺍﻧﺪ ﺧﻮﺩ ﻭﺍﺟﺪ ﻣﻌﻨﺎ ﺑﺎﺷﺪ. ﻧﻤﻮﻧﻪ ﺧﻮﺏ ﺑﺮﺍﯼ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺍﻫﻤﻴﺖ ﻓﻀﺎﯼ ﻣﺜﺒﺖ ﻭ ﻣﻨﻔﯽ ﺗﺼﻮﻳﺮ ﺩﻭ ﻧﻴﻤﺮﺥ ﺭﻭﺑﺮﻭﯼ ﻫﻢ ﺍﺳﺖ ﮐﻪ ﻓﻀﺎﯼ ﻣﻨﻔﯽ ﺁﻧﻬﺎ ﺩﺭ ﻭﺍﻗﻊ ﺷﮑﻞ ﻳﮏ ﮔﻠﺪﺍﻥ ﺭﺍ ﻣﯽ ﺳﺎﺯﺩ.

ﻣﻌﻨﺎ ﻭ ﺍﻫﻤﻴﺖ ﻓﻀﺎﯼ ﻣﺜﺒﺖ ﻭ ﻣﻨﻔﯽ ﺩﺭ ﺗﺼﺎﻭﻳﺮ ﺍﻳﻦ ﻣﻔﻬﻮﻡ ﺭﺍ ﺁﺷﮑﺎﺭ ﻣﯽ ﮐﻨﺪ ﮐﻪ ﺗﻤﺎﻡ ﻋﻨﺎﺻﺮ ﺑﺼﺮﯼ ﻳﮏ ﺗﺼﻮﻳﺮ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﻇﺎﻫﺮ ﻣﺴﺘﻘﻞ ﻭ ﺟﺪﺍ ﺍﺯ ﻫﻢ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﻨﺪ، ﺍﻣﺎ ﺍﮔﺮ ﻓﻀﺎﯼ ﻣﻨﻔﯽ ﻣﻌﻨﺎﺩﺍﺭﯼ ﺑﻴﻦ ﺁﻧﻬﺎ ﺍﻳﺠﺎﺩ ﺷﺪﻩ ﺑﺎﺷﺪ، ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﻳﮏ ﮐﻠﻴﺖ ﺭﺍ ﺑﻮﺟﻮﺩ ﺁﻭﺭﻧﺪ. ﻭ ﻳﺎ ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﺗﺮﮐﻴﺐ ﺑﻨﺪﯼ ﻋﻨﺎﺻﺮ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺭ ﺧﺪﻣﺖ ﺍﻳﺠﺎﺩ ﺷﮑﻠﯽ ﺑﺎﺷﺪ ﮐﻪ ﺍﺯ ﻓﻀﺎﻫﺎﯼ ﻣﻨﻔﯽ ﺑﻮﺟﻮﺩ ﻣﯽ ﺁﻳﺪ. ﮐﻮﺩﮐﺎﻥ ﮐﻪ ﺫﻫﻦ ﺁﻧﻬﺎ ﺑﺎ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭘﻴﺶ ﻓﺮﺽ ﻫﺎ ﺁﺷﻨﺎ ﻧﻴﺴﺖ، ﺩﺭ ﻣﻮﺍﺟﻪ ﺑﺎ ﺗﺼﺎﻭﻳﺮ ﻣﺨﺘﻠﻒ ﺗﻮﺟﻪ ﺑﻴﺸﺘﺮﯼ ﺑﻪ ﻓﻀﺎﻫﺎﯼ ﻣﻨﻔﯽ ﻣﯽ ﮐﻨﻨﺪ.

ﺩﺭ ﻓﺮﻫﻨﮓ ﭼﻴﻨﯽ ﻋﻼﻣﺘﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺑﻬﻨﺎﻡ «ﻳﻴﻦ ﻭ ﻳﺎﻧﮓ (Yin Yang)» ﮐﻪ ﺩﺭ ﺁﻥ ﺍﺯ ﺍﺭﺯﺵ ﻓﻀﺎﯼ ﻣﺜﺒﺖ ﻭ ﻣﻨﻔﯽ ﺑﻪ ﻳﮏ ﺍﻧﺪﺍﺯﻩ ﺍﺳﺘﻔﺎﺩﻩ ﺷﺪﻩ ﺍﺳﺖ. ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺷﮑﻠﯽ ﮐﻪ ﺍﺯ ﻓﻀﺎﯼ ﻣﻨﻔﯽ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ ﺁﻳﺪ، ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﮐﻪ ﺫﻫﻦ ﺑﻴﻨﻨﺪﻩ ﺭﺍ ﺑﻪ ﺟﺴﺘﺠﻮﯼ ﺑﻴﺸﺘﺮﯼ ﻓﺮﺍ ﻣﯽ ﺧﻮﺍﻧﺪ، ﻣﻌﻤﻮﻻ ﺍﺯ ﻗﺪﺭﺕ ﺗﺎﺛﻴﺮﮔﺬﺍﺭﯼ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺧﻮﺭﺩﺍﺭ ﺍﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﺍﻳﺠﺎﺩ ﻳﮏ ﺗﺼﻮﻳﺮ ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻓﻀﺎﯼ ﻣﻨﻔﯽ ﺑﻴﻦ ﻋﻨﺎﺻﺮ ﻧﻴﺰ ﻣﺘﻤﺮﮐﺰ ﮐﻨﻴﻢ ﺗﺎ ﺍﺯ ﺁﻧﭽﻪ ﺑﻮﺟﻮﺩ ﻣﯽ ﺁﻭﺭﻳﻢ، ﺑﺘﻮﺍﻧﻴﻢ ﺑﻬﺮﻩ ﺑﻴﺸﺘﺮﯼ ﺑﺒﺮﻳﻢ.

ﻣﻮﺭﻳﺲ ﺍﺷﺮ (Morris Esher) ﻫﻨﺮﻣﻨﺪ ﮔﺮﺍﻓﻴﺴﺖ ﻭ ﺗﺼﻮﻳﺮﺳﺎﺯ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻗﺎﺑﻠﻴﺖ ﺗﺼﺎﻭﻳﺮ ﻣﻨﻔﯽ ﺗﻮﺍﻧﺴﺘﻪ ﺍﺳﺖ ﺗﺼﺎﻭﻳﺮ ﺑﺴﻴﺎﺭ ﺧﻼﻗﺎﻧﻪ ﺍﯼ ﺭﺍ ﺗﺮﺳﻴﻢ ﻧﻤﺎﻳﺪ.

 

ﻣﻔﺎﻫﻴﻢ ﭘﺎﻳﻪ ﺍﯼ ﻣﺘﻀﺎﺩ: ﺳﮑﻮﻥ ﻭ ﺣﺮﮐﺖ، ﻣﻨﻈﻢ ﻭ ﻧﺎﻣﻨﻈﻢ، ﭘﺮﺍﮐﻨﺪﻩ ﻭ ﻣﺘﻤﺮﮐﺰ...
ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻗﺒﻼ ﺑﺤﺚ ﺷﺪ، ﻫﺮ ﺗﺼﻮﻳﺮ ﺣﺎﻭﯼ ﭘﻴﺎﻣﯽ ﺍﺳﺖ ﮐﻪ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﻪ ﺑﻴﻨﻨﺪﻩ ﺁﻥ ﻣﻨﺘﻘﻞ ﺷﻮﺩ. ﭘﻴﺎﻡ ﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﭼﻪ ﺳﺎﺩﻩ ﺑﺎﺷﻨﺪ ﻭ ﭼﻪ ﭘﻴﭽﻴﺪﻩ، ﻋﻤﺪﺗﺎ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺑﺮﺍﺳﺎﺱ ﻳﮏ ﺳﺮﯼ ﻣﻔﺎﻫﻴﻢ ﭘﺎﻳﻪ ﺍﯼ ﺍﺳﺘﻮﺍﺭ ﮔﺮﺩﻧﺪ.

ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺍﮔﺮ ﻫﺪﻑ ﺍﺯ ﻳﮏ ﺗﺎﺑﻠﻮﯼ ﺗﺒﻠﻴﻐﺎﺗﯽ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺗﻮﺍﻧﺎﻳﯽ ﻫﺎﯼ ﻳﮏ ﺧﻮﺩﺭﻭ ﺑﺎﺷﺪ، ﺩﺭ ﺁﻥ ﺗﺼﻮﻳﺮ ﺑﺎﻳﺪ ﺑﺮ ﻣﻔﻬﻮﻡ ﺣﺮﮐﺖ (ﺳﺮﻋﺖ) ﺗﺎﮐﻴﺪ ﺷﻮﺩ. ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﮔﺮ ﻳﮏ ﺗﺼﻮﻳﺮ ﺑﺨﻮﺍﻫﺪ ﻭﺿﻌﻴﺖ ﺗﺮﺍﻓﻴﮏ ﺷﻬﺮ ﺭﺍ ﻧﻤﺎﻳﺶ ﺩﻫﺪ، ﺑﺎﻳﺪ ﺍﺯ ﻋﻨﺎﺻﺮﯼ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﺪ ﮐﻪ ﺑﺮ ﺳﮑﻮﻥ ﺗﺎﮐﻴﺪ ﻣﯽ ﮐﻨﻨﺪ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﺍﻧﻮﺍﻉ ﻭ ﺍﻗﺴﺎﻡ ﺗﺼﺎﻭﻳﺮ ﮔﻮﻧﺎﮔﻮﻥ ﺭﺍ ﺍﻳﺠﺎﺩ ﮐﻨﻴﻢ، ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﺍﺯ ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﭘﺎﻳﻪ ﺍﯼ ﺩﺭ ﺑﻪ ﻭﺟﻮﺩ ﺁﻭﺭﺩﻥ ﺁﻧﻬﺎ ﮐﻤﮏ ﺑﮕﻴﺮﻳﻢ. ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﻋﺒﺎﺭﺗﻨﺪ ﺍﺯ ﺳﮑﻮﻥ ﻭ ﺣﺮﮐﺖ، ﻣﻨﻈﻢ ﻭ ﻧﺎﻣﻨﻈﻢ، ﭘﺮﺍﮐﻨﺪﻩ ﻭ ﻣﺘﻤﺮﮐﺰ، ﻣﺘﻌﺎﺩﻝ ﻭ ﻧﺎﭘﺎﻳﺪﺍﺭ، ﻣﺘﻘﺎﺭﻥ ﻭ ﻧﺎﻣﺘﻘﺎﺭﻥ، ﺳﺎﺩﻩ ﻭ ﭘﻴﭽﻴﺪﻩ (ﭘﺮﻧﻘﺶ)، ﺷﻔﺎﻑ ﻭ ﻣﺎﺕ، ﻣﺘﻨﻮﻉ ﻭ ﻳﮑﺪﺳﺖ ﻭ... .

ﺩﺭ ﻧﺘﻴﺠﻪ ﻗﺪﻡ ﺍﻭﻝ ﺩﺭ ﺍﻳﺠﺎﺩ ﻳﮏ ﺗﺼﻮﻳﺮ ﺑﺼﺮﯼ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺪﺍﻧﻴﻢ ﭘﻴﺎﻡ ﻣﻮﺭﺩ ﻧﻈﺮ ﺗﺼﻮﻳﺮ ﺑﺮ ﭼﻪ ﻣﻔﻬﻮﻡ ﻳﺎ ﻣﻔﺎﻫﻴﻢ ﭘﺎﻳﻪ ﺍﯼ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﺳﺘﻮﺍﺭ ﺷﻮﺩ. ﺳﭙﺲ ﺑﺮﺍﺳﺎﺱ ﺗﻤﺎﻡ ﺁﻣﻮﺧﺘﻪ ﻫﺎﻳﯽ ﮐﻪ ﺩﺍﺭﻳﻢ، ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﻢ ﺁﻥ ﻣﻔﻬﻮﻡ ﺭﺍ ﭼﮕﻮﻧﻪ ﻣﯽ ﺗﻮﺍﻥ ﺍﺯ ﻃﺮﻳﻖ ﺗﺮﮐﻴﺐ ﻋﻨﺎﺻﺮ ﻣﺨﺘﻠﻒ ﺍﻳﺠﺎﺩ ﮐﺮﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺑﺮﺍﯼ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﻣﻔﻬﻮﻡ ﺳﺮﻋﺖ ﻭ ﺣﺮﮐﺖ (ﺑﺮﺍﯼ ﺗﺒﻠﻴﻎ ﻳﮏ ﺧﻮﺩﺭﻭ) ﺗﺮﮐﻴﺒﯽ ﺍﺯ ﺧﻄﻮﻁ ﺗﻴﺰ ﻣﺘﻤﺮﮐﺰ ﺑﺮ ﻳﮏ ﻧﻘﻄﻪ، ﺭﻧﮓ ﻫﺎﯼ ﮔﺮﻡ، ﺗﻀﺎﺩ ﺭﻭﺷﻨﺎﻳﯽ ﺯﻳﺎﺩ ﻭ... ﻣﯽ ﺗﻮﺍﻧﺪ ﻣﻔﻬﻮﻣﯽ ﺍﺯ ﺳﺮﻋﺖ ﺭﺍ ﺍﻟﻘﺎﺀ ﮐﻨﺪ. ﭘﺲ ﺑﺎﻳﺪ ﺩﺭ ﺗﺼﻮﻳﺮ ﻣﻮﺭﺩ ﻧﻈﺮﻣﺎﻥ ﺗﺎ ﺟﺎﻳﯽ ﮐﻪ ﺍﻣﮑﺎﻥ ﺩﺍﺭﺩ ﺍﺯ ﺍﻳﻦ ﺗﻤﻬﻴﺪﺍﺕ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ.

ﻫﺮ ﻣﻔﻬﻮﻣﯽ ﺩﺭ ﻣﻘﺎﺑﻞ ﺧﻮﺩ ﻣﻔﻬﻮﻡ ﻣﺘﻀﺎﺩ ﻭ ﻣﺘﻘﺎﺑﻠﯽ ﻫﻢ ﺩﺍﺭﺩ (ﭘﺎﻳﺪﺍﺭ ﺩﺭ ﻣﻘﺎﺑﻞ ﻧﺎﭘﺎﻳﺪﺍﺭ). ﺩﺭ ﻭﺍﻗﻊ ﻫﺮ ﺗﻤﻬﻴﺪﯼ ﮐﻪ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﻳﮏ ﻣﻔﻬﻮﻡ ﺑﻪ ﮐﺎﺭ ﺑﺴﺘﻪ ﻣﯽ ﺷﻮﺩ، ﻋﻤﻞ ﺧﻼﻑ ﻭ ﻣﻘﺎﺑﻞ ﺁﻥ ﻣﻔﻬﻮﻡ ﻣﻘﺎﺑﻞ ﺭﺍ ﺍﻳﺠﺎﺩ ﻣﯽ ﮐﻨﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﻣﻔﻬﻮﻡ ﺍﻳﺴﺘﺎﻳﯽ (ﭘﺎﻳﺪﺍﺭﯼ) ﺭﺍ ﺍﻳﺠﺎﺩ ﮐﻨﻴﻢ، ﺗﺮﮐﻴﺐ ﺑﻨﺪﯼ ﻣﺘﻌﺎﺩﻝ ﻭ ﺣﺘﯽ ﻣﺘﻘﺎﺭﻥ (ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﮐﻪ ﻣﺮﮐﺰ ﺛﻘﻞ ﺗﺮﮐﻴﺐ ﺑﻨﺪﯼ ﺩﺭ ﻣﻴﺎﻧﻪ ﭘﺎﻳﻴﻦ ﺗﺼﻮﻳﺮ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ)، ﺧﻄﻮﻁ ﺍﻓﻘﯽ ﻭ ﺿﺨﻴﻢ، ﺭﻧﮓ ﻫﺎﯼ ﺳﺮﺩ ﻭ ﺑﺎ ﺩﺭﺟﻪ ﺍﺷﺒﺎﻉ ﺑﺎﻻ ﻭ ﮐﻨﺘﺮﺍﺳﺖ ﺳﺎﻳﻪ- ﺭﻭﺷﻦ ﮐﻢ ﻭ... ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﻣﺎ ﮐﻤﮏ ﮐﻨﺪ. ﺩﺭ ﻣﻘﺎﺑﻞ ﻫﺮ ﭼﻘﺪﺭ ﺧﻼﻑ ﺍﻳﻦ ﺗﻤﻬﻴﺪﺍﺕ ﻋﻤﻞ ﮐﻨﻴﻢ ﺩﺭ ﻭﺍﻗﻊ ﺑﻪ ﻣﻔﻬﻮﻡ ﻧﺎﭘﺎﻳﺪﺍﺭﯼ ﻧﺰﺩﻳﮏ ﺷﺪﻩ ﺍﻳﻢ.

 

ﺍﻳﺠﺎﺩ ﺑﻌﺪ ﺳﻮﻡ
ﺩﻧﻴﺎﯼ ﻭﺍﻗﻌﯽ ﻣﺎ ﺩﻧﻴﺎﯼ ﺳﻪ ﺑﻌﺪﯼ ﺍﺳﺖ. ﻣﺎ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﺷﺶ ﺟﻬﺖ ﻣﺨﺘﻠﻒ، ﺑﺎﻻ- ﭘﺎﻳﻴﻦ، ﭼﭗ- ﺭﺍﺳﺖ ﻭ ﺟﻠﻮ- ﻋﻘﺐ ﺭﺍ ﺩﺭ ﺗﺠﺮﺑﻪ ﻫﺎﯼ ﺭﻭﺯﻣﺮﻩ ﺧﻮﺩ ﺩﺭﮎ ﮐﻨﻴﻢ. ﺍﻣﺎ ﺩﻧﻴﺎﯼ ﺗﺼﻮﻳﺮ ﺩﻭ ﺑﻌﺪﯼ ﺍﺳﺖ، ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﮐﻪ ﺑﺮﺍﯼ ﻧﻤﺎﻳﺶ ﻋﻤﻖ (ﺟﻠﻮ ﻳﺎ ﻋﻘﺐ ﺑﻮﺩﻥ ﻳﮏ ﻋﻨﺼﺮ) ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﺑﺎﻳﺪ ﺗﻤﻬﻴﺪﺍﺕ ﺧﺎﺻﯽ ﺭﺍ ﺑﻪ ﮐﺎﺭ ﺑﮕﻴﺮﻳﻢ.

ﺑﺮﺍﺳﺎﺱ ﻣﺸﺎﻫﺪﺍﺕ ﻭ ﺗﺠﺮﺑﻴﺎﺕ ﻓﺮﺩﯼ، ﻫﻤﻪ ﻣﺎ ﺩﺭﮎ ﮐﺮﺩﻩ ﺍﻳﻢ ﮐﻪ ﻫﺮ ﭼﻪ ﻓﺎﺻﻠﻪ ﻳﮏ ﺟﺴﻢ ﺍﺯ ﻣﺎ ﺑﻴﺸﺘﺮ ﻣﯽ ﺷﻮﺩ، ﺁﻥ ﺟﺴﻢ ﮐﻮﭼﮏ ﺗﺮ، ﺭﻧﮓ ﭘﺮﻳﺪﻩ ﺗﺮ ﻭ ﺟﺰﻳﻴﺎﺕ ﻗﺎﺑﻞ ﻣﺸﺎﻫﺪﻩ ﺍﺵ ﮐﻤﺘﺮ ﻣﯽ ﺷﻮﺩ. ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻳﻦ ﺳﻪ ﺗﻤﻬﻴﺪ ﺑﺮﺍﯼ ﺑﺎﺯﻧﻤﺎﻳﯽ ﻋﻤﻖ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﮐﺎﺭ ﮔﺮﻓﺘﻪ ﺷﻮﺩ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﺑﺎ ﺑﺰﺭﮒ ﻭ ﮐﻮﭼﮏ ﮐﺮﺩﻥ ﻋﻨﺎﺻﺮ، ﭘﺮﺭﻧﮓ ﻭ ﮐﻤﺮﻧﮓ ﮐﺮﺩﻥ ﺁﻧﻬﺎ ﻭ ﺍﻓﺰﺍﻳﺶ ﻳﺎ ﮐﺎﻫﺶ ﺟﺰﻳﻴﺎﺕ ﺁﻥ ﻣﯽ ﺗﻮﺍﻥ ﺩﻭﺭﯼ ﻭ ﻧﺰﺩﻳﮑﯽ ﺁﻧﻬﺎ ﺭﺍ ﺍﺯ ﻫﻢ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﻧﻤﺎﻳﺶ ﺩﺍﺩ.

ﺭﻭﺵ ﺩﻳﮕﺮ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺳﻪ ﺑﻌﺪﯼ ﺑﻮﺩﻥ ﺍﺟﺴﺎﻡ، ﻧﻤﺎﻳﺶ ﻭﺟﻮﻩ ﻣﺨﺘﻠﻒ ﻳﮏ ﺣﺠﻢ ﺍﺳﺖ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﻫﻤﻴﺸﻪ ﺩﻭ ﻳﺎ ﺳﻪ ﻭﺟﻪ ﻳﮏ ﻣﮑﻌﺐ ﺑﺮﺍﯼ ﻣﺎ ﻗﺎﺑﻞ ﻣﺸﺎﻫﺪﻩ ﺍﺳﺖ (ﺍﮔﺮ ﺣﺠﻢ ﺑﻴﺶ ﺍﺯ ﺍﻧﺪﺍﺯﻩ ﺑﺰﺭﮒ ﻧﺒﺎﺷﺪ). ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﮔﺮ ﺳﻌﯽ ﮐﻨﻴﻢ ﻭﺟﻮﻩ ﻣﺨﺘﻠﻒ ﺭﺍ ﺑﻪ ﻳﮏ ﺷﮑﻞ ﺍﺿﺎﻓﻪ ﮐﻨﻴﻢ، ﺩﺭ ﻭﺍﻗﻊ ﺣﺴﯽ ﺍﺯ ﺳﻪ ﺑﻌﺪﯼ ﺑﻮﺩﻥ ﺑﻪ ﺁﻥ ﺩﺍﺩﻩ ﺍﻳﻢ. ﺍﻟﺒﺘﻪ ﻣﺎ ﻫﻤﻴﺸﻪ ﻭﺟﻮﻩ ﻳﮏ ﻣﮑﻌﺐ ﺭﺍ ﮐﻪ ﻣﯽ ﺩﺍﻧﻴﻢ ﻣﺮﺑﻊ ﺷﮑﻞ ﻫﺴﺘﻨﺪ ﺑﻪ ﺻﻮﺭﺕ ﻓﺮﻡ ﻫﺎﻳﯽ ﺍﺯ ﻟﻮﺯﯼ ﻣﺸﺎﻫﺪﻩ ﻣﯽ ﮐﻨﻴﻢ. ﺗﺮﺳﻴﻢ ﺩﻗﻴﻖ ﺁﻧﭽﻪ ﻣﺸﺎﻫﺪﻩ ﻣﯽ ﮐﻨﻴﻢ ﺑﺮ ﺻﻔﺤﻪ ﺗﺼﻮﻳﺮ ﮐﻤﮏ ﺯﻳﺎﺩﯼ ﺩﺭ ﻭﺍﻗﻊ ﻧﻤﺎﻳﯽ ﻧﻤﺎﻳﺶ ﺳﻪ ﺑﻌﺪﯼ ﺗﺼﺎﻭﻳﺮ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﺗﻤﻬﻴﺪﺍﺕ ﮐﻪ ﻗﺒﻼ ﻧﻴﺰ ﺑﻪ ﺁﻥ ﺍﺷﺎﺭﻩ ﺷﺪ، ﻧﻤﺎﻳﺶ ﺳﺎﻳﻪ ﻫﺎﯼ ﻳﮏ ﺟﺴﻢ ﺍﺳﺖ. ﺣﺠﻢ ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﺩﺭ ﻣﻘﺎﺑﻞ ﻧﻮﺭ ﺳﺎﻳﻪ ﺍﻳﺠﺎﺩ ﻣﯽ ﮐﻨﻨﺪ ﮐﻪ ﻧﻤﺎﻳﺶ ﺍﻳﻦ ﺳﺎﻳﻪ ﺭﻭﺷﻦ ﻳﻪ ﺣﺠﻢ ﻫﺎﯼ ﺩﻭ ﺑﻌﺪﯼ، ﺣﺲ ﺳﻪ ﺑﻌﺪﯼ ﺑﻮﺩﻥ ﻣﯽ ﺑﺨﺸﺪ.

ﺗﻤﺮﻳﻦ ۱: ﺩﻭ ﻣﺮﺑﻊ ﺑﻪ ﺍﺑﻌﺎﺩ ۱۰x۱۰ ﺳﺎﻧﺘﻴﻤﺘﺮ ﺩﺭ ﮐﻨﺎﺭ ﻫﻢ ﺗﺮﺳﻴﻢ ﮐﻨﻴﺪ ﻭ ﺗﻨﻬﺎ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻳﮑﺴﺮﯼ ﺧﻄﻮﻁ ﺳﻌﯽ ﮐﻨﻴﺪ ﺩﺭ ﻳﮑﯽ ﻣﻔﻬﻮﻣﯽ ﺍﺯ ﺍﻳﺴﺘﺎﻳﯽ ﻭ ﺩﺭ ﺩﻳﮕﺮﯼ ﻣﻔﻬﻮﻡ ﺟﻨﺒﺶ ﻭ ﺣﺮﮐﺖ ﺭﺍ ﺍﻳﺠﺎﺩ ﮐﻨﻴﺪ. ﺍﻧﺘﺨﺎﺏ ﺍﻧﺪﺍﺯﻩ، ﺿﺨﺎﻣﺖ، ﻧﻮﻉ ﺍﻧﺤﻨﺎﺀ ﻭ... ﺧﻄﻮﻁ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺷﻤﺎ ﺍﺳﺖ.
