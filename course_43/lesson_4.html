ﺩﺭﺱ ﺷﺸﻢ- ﺍﻳﺠﺎﺩ ﮐﻴﻔﺖ ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺩﺭ ﺗﺼﻮﻳﺮ

ﺗﻀﺎﺩ ﻭ ﻫﻤﺎﻫﻨﮕﯽ ﺩﺭ ﺗﺼﻮﻳﺮ
ﻫﻤﺎﻧﮕﻮﻧﻪ ﮐﻪ ﺩﺭ ﺑﺤﺚ ﺭﻧﮓ ﺍﺷﺎﺭﻩ ﺷﺪ، ﺍﻳﺠﺎﺩ ﺗﻀﺎﺩ ﻣﻴﺎﻥ ﻳﮏ ﻋﻨﺼﺮ ﻳﺎ ﻣﺠﻤﻮﻋﻪ ﺍﯼ ﺍﺯ ﻋﻨﺎﺻﺮ ﺗﺼﻮﻳﺮ ﺑﺎ ﺑﻘﻴﻪ ﺗﺼﻮﻳﺮ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺭ ﺑﺮﺟﺴﺘﻪ ﺳﺎﺧﺘﻦ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺩﺭﮎ ﺑﻬﺘﺮ ﺁﻥ ﻋﻨﺼﺮ ﮐﻤﮏ ﮐﻨﺪ. ﺩﺭ ﻭﺍﻗﻊ ﻋﻨﺼﺮﯼ ﮐﻪ ﻗﺮﺍﺭ ﺍﺳﺖ ﭘﻴﺎﻡ ﻳﮏ ﺗﺼﻮﻳﺮ ﺭﺍ ﺑﻪ ﻣﺨﺎﻃﺐ ﺧﻮﺩ ﺑﺮﺳﺎﻧﺪ، ﺍﮔﺮ ﺩﺭ ﻣﺘﻦ ﻳﺎ ﺯﻣﻴﻨﻪ ﺗﺼﻮﻳﺮ ﻣﺤﻮ ﺷﺪﻩ ﺑﺎﺷﺪ، ﺩﺭﮐﺶ ﺑﺮﺍﯼ ﻣﺨﺎﻃﺐ ﺑﺴﻴﺎﺭ ﺳﺨﺖ ﺗﺮ ﺍﺳﺖ ﺗﺎ ﻳﮏ ﻋﻨﺼﺮ ﻣﺘﻀﺎﺩ ﻭ ﺑﺮﺟﺴﺘﻪ. ﺍﮔﺮ ﻫﻤﺎﻫﻨﮕﯽ ﻧﻮﻋﯽ ﺁﺭﺍﻣﺶ ﻭ ﺳﮑﻮﻥ ﺭﺍ ﺑﻪ ﻭﺟﻮﺩ ﺑﻴﺎﻭﺭﺩ، ﺗﻀﺎﺩ (ﮐﻨﺘﺮﺍﺳﺖ) ﺗﻌﺎﺩﻝ ﺭﺍ ﺑﺮ ﻫﻢ ﻣﯽ ﺯﻧﺪ ﻭ ﺫﻫﻦ ﻣﺨﺎﻃﺐ ﺭﺍ ﺗﺤﺮﻳﮏ ﮐﺮﺩﻩ ﻭ ﺗﻮﺟﻪ ﺍﺵ ﺭﺍ ﺟﻠﺐ ﻣﯽ ﮐﻨﺪ.

ﻋﻠﺖ ﺍﻳﻦ ﺍﻣﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺯﻧﺪﮔﯽ ﻭﺍﻗﻌﯽ، ﻣﺎ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭘﺪﻳﺪﻩ ﻫﺎ ﺭﺍ ﺩﺭ ﻣﻘﺎﺑﻞ ﻣﺘﻀﺎﺩ ﺁﻥ ﻣﻌﻨﺎ ﻣﯽ ﮐﻨﻴﻢ، ﺳﺮﻣﺎ ﺩﺭ ﺑﺮﺍﺑﺮ ﮔﺮﻣﺎ، ﺑﻠﻨﺪﯼ ﺩﺭ ﺑﺮﺍﺑﺮ ﮐﻮﺗﺎﻫﯽ ﻭ... . ﺩﺭ ﺯﺑﺎﻥ ﺗﺼﻮﻳﺮ ﻫﻢ ﻫﻤﻴﻦ ﻣﻮﺿﻮﻉ ﻣﺼﺪﺍﻕ ﺩﺍﺭﺩ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ، ﺑﺰﺭﮔﯽ ﺍﻧﺪﺍﺯﻩ ﻳﮏ ﺳﺎﺧﺘﻤﺎﻥ ﺑﻠﻨﺪ ﺩﺭ ﮐﻨﺎﺭ ﻣﺠﻤﻮﻋﻪ ﺍﯼ ﺍﺯ ﺧﺎﻧﻪ ﻫﺎﯼ ﮐﻮﭼﮏ ﺑﺮﺟﺴﺘﻪ ﺗﺮ ﻧﻤﺎﻳﺎﻥ ﻣﯽ ﺷﻮﺩ.

ﻋﻼﻭﻩ ﺑﺮ ﮐﻨﺘﺮﺍﺳﺖ ﺭﻧﮕﯽ ﮐﻪ ﭘﻴﺸﺘﺮ ﺩﺭﺑﺎﺭﻩ ﺁﻥ ﺻﺤﺒﺖ ﺷﺪ، ﺗﻀﺎﺩ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﺎ ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ، ﻓﺮﻡ ﻭ ﻳﺎ ﺩﺭﺟﻪ ﺭﻭﺷﻨﺎﻳﯽ ﻣﻴﺎﻥ ﻋﻨﺎﺻﺮ ﻣﺨﺘﻠﻒ ﻧﺴﺒﺖ ﺑﻪ ﻫﻢ ﺣﺎﺻﻞ ﺷﻮﺩ.

 

ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ ﺩﺭ ﺗﺼﻮﻳﺮ
ﻣﺎ ﺗﻔﺎﻭﺕ ﻣﻴﺎﻥ ﺑﺎﻓﺖ ﻣﻮﺍﺩ ﻭ ﻣﺼﺎﻟﺢ ﻣﺨﺘﻠﻒ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﻟﻤﺲ ﮐﺮﺩﻥ ﺁﻧﻬﺎ ﺩﺭﮎ ﻣﯽ ﮐﻨﻴﻢ. ﻣﻔﺎﻫﻴﻤﯽ ﻣﺎﻧﻨﺪ ﺯﺑﺮﯼ ﻳﮏ ﮔﻮﻧﯽ، ﻧﺮﻣﯽ ﭘﻮﺳﺖ ﮔﺮﺑﻪ، ﻟﻄﺎﻓﺖ ﺩﺳﺘﻤﺎﻝ ﺍﺑﺮﻳﺸﻤﯽ ﻳﺎ ﺧﺸﻦ ﺑﻮﺩﻥ ﺳﻤﺒﺎﺩﻩ ﺍﺯ ﻃﺮﻳﻖ ﺗﺠﺮﺑﻪ ﺣﺲ ﻻﻣﺴﻪ ﺩﺭ ﺫﻫﻦ ﻣﺎ ﺛﺒﺖ ﺷﺪﻩ ﺍﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﺮ ﺑﺎﺭ ﮐﻪ ﻣﺜﻼ ﭘﻮﺳﺘﻪ ﻳﮏ ﺗﻨﻪ ﺩﺭﺧﺖ ﺭﺍ ﺑﺒﻴﻨﻴﻢ، ﺑﻼﻓﺎﺻﻠﻪ ﺩﺭ ﺫﻫﻦ ﻣﺎﻥ ﺍﺣﺴﺎﺳﯽ ﺭﺍ ﮐﻪ ﺍﺯ ﻟﻤﺲ ﮐﺮﺩﻥ ﺁﻥ ﺳﻄﺢ ﺗﺠﺮﺑﻪ ﮐﺮﺩﻳﻢ، ﺑﻴﺎﺩ ﻣﯽ ﺁﻭﺭﻳﻢ (ﻣﺘﺎﺳﻔﺎﻧﻪ ﻭﺍﮊﻩ ﻫﺎﻳﯽ ﮐﻪ ﺗﻮﺻﻴﻒ ﮐﻨﻨﺪﻩ ﺟﻨﺴﻴﺖ ﺑﺎﻓﺖ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﺎﺷﻨﺪ، ﺩﺭ ﺯﺑﺎﻥ ﺑﺴﻴﺎﺭ ﻧﺎﺩﺭﻧﺪ).

ﺩﺭ ﺯﺑﺎﻥ ﺗﺼﻮﻳﺮ ﻧﻴﺰ ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ ﺑﺮﺍﺳﺎﺱ ﻫﻤﻴﻦ ﺗﺠﺮﺑﻪ ﺑﻨﺎ ﺷﺪﻩ ﺍﺳﺖ. ﻭﺍﮐﻨﺶ ﻣﺨﺎﻃﺐ ﺑﺎ ﺩﻳﺪﻥ ﻋﻨﺎﺻﺮﯼ ﮐﻪ ﻣﺜﻼ ﺑﺎﻓﺘﯽ ﺷﺒﻴﻪ ﻓﻠﺰ، ﺻﻴﻘﻠﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺑﺎ ﭼﻴﺰﯼ ﮐﻪ ﺷﺒﻴﻪ ﺑﻪ ﭼﻮﺏ ﺍﺳﺖ، ﺣﺘﯽ ﺍﮔﺮ ﺭﻧﮓ ﻭ ﺩﺭﺟﻪ ﺭﻭﺷﻨﺎﻳﯽ ﺁﻧﻬﺎ ﻳﮑﺴﺎﻥ ﺑﺎﺷﺪ، ﮐﺎﻣﻼ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﺩﻳﺪﻥ ﺳﻄﺤﯽ ﺑﺎ ﺑﺎﻓﺖ ﺧﺎﺹ، ﺑﺎﻋﺚ ﻣﯽ ﺷﻮﺩ ﻓﺮﺩ ﻣﺨﺎﻃﺐ ﺗﻤﺎﻡ ﻣﻔﺎﻫﻴﻢ ﻣﺮﺗﺒﻂ ﺑﺎ ﺁﻥ ﺭﺍ ﺩﺭ ﺫﻫﻨﺶ ﺗﺪﺍﻋﯽ ﮐﻨﺪ.

ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺳﻄﻮﺡ ﺑﺎ ﺑﺎﻓﺖ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﻃﻮﺭ ﻣﺴﺘﻘﻴﻢ ﺍﺯ ﺳﻄﻮﺣﯽ ﺑﺎ ﺟﻨﺴﻴﺖ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ، ﻣﺜﻼ ﺍﺯ ﺗﮑﻪ ﻫﺎﯼ ﭘﺎﺭﭼﻪ ﻫﺎﯼ ﻣﺨﺘﻠﻒ (ﺗﻮﺭﯼ، ﮔﻮﻧﯽ ﻳﺎ ﺍﺑﺮﻳﺸﻤﯽ) ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺳﻄﻮﺡ ﺗﺼﻮﻳﺮﯼ ﺑﻬﺮﻩ ﮔﺮﻓﺖ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﻣﺴﺘﻘﻴﻤﺎ ﺑﺮ ﺭﻭﯼ ﺗﺼﻮﻳﺮ ﻣﺘﺼﻞ ﮐﺮﺩ. ﺭﻭﺵ ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻧﻘﻄﻪ ﻭ ﺧﻂ، ﺳﻄﻮﺣﯽ ﺍﻳﺠﺎﺩ ﮐﺮﺩ ﮐﻪ ﺑﺎﻓﺖ ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺭﺍ ﺍﻟﻘﺎﺀ ﮐﻨﻨﺪ. ﻣﺜﻼ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ ﺻﺎﻑ ﻣﯽ ﺗﻮﺍﻥ ﺍﺯ ﺧﻄﻮﻁ ﻣﻨﺤﻨﯽ ﻭ ﻣﻮﺍﺯﯼ ﺑﺎ ﺿﺨﺎﻣﺖ ﮐﻢ ﻭ ﺍﻧﺤﻨﺎﺀ ﺁﺭﺍﻡ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ ﻳﺎ ﺗﻌﺪﺍﺩﯼ ﻧﻘﺎﻁ ﮐﻮﭼﮏ، ﮐﻤﺮﻧﮓ ﻭ ﭘﺮﺍﮐﻨﺪﻩ ﺭﺍ ﺩﺭ ﮐﻨﺎﺭ ﻫﻢ ﻗﺮﺍﺭ ﺩﺍﺩ. ﺑﺮﻋﮑﺲ ﺧﻄﻮﻁ ﺷﮑﺴﺘﻪ ﻭ ﺑﺎ ﺯﺍﻭﻳﻪ ﻫﺎﯼ ﺗﻴﺰ، ﻧﺎﻣﻨﻈﻢ ﻭ ﺩﺭﻫﻢ ﻭ ﺑﺮﻫﻢ ﺣﺴﯽ ﺍﺯ ﺳﻄﺢ ﺯﺑﺮ ﺭﺍ ﻣﻨﺘﻘﻞ ﻣﯽ ﮐﻨﻨﺪ.

ﺟﻨﺲ ﺯﻣﻴﻨﻪ ﺗﺼﻮﻳﺮ (ﮐﺎﻏﺬ ﻳﺎ ﭘﺎﺭﭼﻪ) ﺩﺭ ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﻮﺛﺮ ﺍﺳﺖ. ﻣﺜﻼ ﺍﮔﺮ ﺑﺎ ﻗﻠﻢ ﻣﻮ ﻭ ﺟﻮﻫﺮ، ﺧﻂ ﺻﺎﻓﯽ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﻳﮏ ﺻﻔﺤﻪ ﺳﻔﻴﺪ ﻣﻘﻮﺍ ﺑﮑﺸﻴﺪ ﻭ ﻫﻤﻴﻦ ﮐﺎﺭ ﺭﺍ ﺑﺎﺭ ﺩﻳﮕﺮ ﺭﻭﯼ ﻣﻘﻮﺍﻳﯽ ﮐﻪ ﮐﻤﯽ ﻣﺮﻃﻮﺏ ﺍﺳﺖ ﺗﮑﺮﺍﺭ ﮐﻨﻴﺪ ﻭ ﺑﺎﺭ ﺳﻮﻡ ﺁﻥ ﺭﺍ ﺭﻭﯼ ﻣﻘﻮﺍﻳﯽ ﮐﻪ ﺳﻄﺢ ﺁﻥ ﺑﺎ ﺳﻤﺒﺎﺩﻩ ﺯﺑﺮ ﺷﺪﻩ ﺍﻣﺘﺤﺎﻥ ﮐﻨﻴﺪ، ﺑﻪ ﻭﺿﻮﺡ ﺩﺭﺧﻮﺍﻫﻴﺪ ﻳﺎﻓﺖ ﺗﺎ ﭼﻪ ﻣﻴﺰﺍﻥ ﻣﺎﻫﻴﺖ ﻳﮏ ﺧﻂ ﺑﺮ ﺭﻭﯼ ﺑﺎﻓﺖ ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺗﻐﻴﻴﺮ ﻣﯽ ﮐﻨﺪ. ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺩﺭ ﺗﺼﻮﻳﺮ، ﻭ ﺩﺭ ﻧﻬﺎﻳﺖ ﺍﻓﺰﺍﻳﺶ ﺗﻀﺎﺩ ﺑﻴﻦ ﻋﻨﺎﺻﺮ ﺗﺼﻮﻳﺮﯼ، ﺍﻣﺮﻭﺯﻩ ﺍﻣﮑﺎﻧﺎﺕ ﻭ ﻣﺼﺎﻟﺢ ﺯﻳﺎﺩﯼ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﺳﺖ.

ﺩﺭ ﺩﻧﻴﺎﯼ ﻣﺠﺎﺯﯼ ﻭ ﺑﺮﻧﺎﻣﻪ ﻫﺎﻳﯽ ﮐﻪ ﺑﺎ ﭘﺮﺩﺍﺯﺵ ﺗﺼﺎﻭﻳﺮ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﻧﺪ، ﺍﻣﮑﺎﻧﺎﺕ ﻓﺮﺍﻭﺍﻧﯽ ﺟﻬﺖ ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﻓﺮﺍﻫﻢ ﺁﻣﺪﻩ ﺍﺳﺖ.

ﺩﺭ ﻧﻬﺎﻳﺖ، ﺑﺎ ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﻋﻤﻖ ﻣﻌﻨﺎ ﻭ ﻗﺪﺭﺕ ﺗﺎﺛﻴﺮﮔﺬﺍﺭﯼ ﺁﻥ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﺩﻫﻴﻢ.

 

 

ﺑﻪ ﮐﺎﺭﮔﻴﺮﯼ ﻓﺮﻡ ﺩﺭ ﺗﺼﻮﻳﺮ
ﺳﻄﻮﺡ ﻣﺨﺘﻠﻒ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺩﺍﺭﺍﯼ ﻓﺮﻡ ﻫﺎﯼ ﻣﺘﻔﺎﻭﺗﯽ ﺑﺎﺷﻨﺪ. ﺳﻪ ﻓﺮﻡ ﺍﺻﻠﯽ ﻭ ﭘﺎﻳﻪ ﻋﺒﺎﺭﺗﻨﺪ ﺍﺯ: ﺩﺍﻳﺮﻩ، ﻣﺜﻠﺚ ﻭ ﻣﺮﺑﻊ- ﻣﺴﺘﻄﻴﻞ.

ﺩﺭ ﻭﺍﻗﻊ ﺍﻳﻦ ﺳﻪ ﻓﺮﻡ ﻣﺎﻧﻨﺪ ﺳﻪ ﺭﻧﮓ ﺍﺻﻠﯽ ﻫﺴﺘﻨﺪ، ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﮐﻪ ﻣﯽ ﺗﻮﺍﻥ ﺗﻤﺎﻣﯽ ﻓﺮﻡ ﻫﺎ ﻭ ﻋﻨﺎﺻﺮ ﺗﺼﻮﻳﺮﯼ ﺩﻳﮕﺮ ﺭﺍ ﺑﻪ ﻧﻮﻋﯽ ﺍﺯ ﺗﺮﮐﻴﺐ ﻳﺎ ﺗﻐﻴﻴﺮ ﺷﮑﻞ ﺍﻳﻦ ﺳﻪ ﻓﺮﻡ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩ. ﺑﻪ ﻋﺒﺎﺭﺗﯽ ﻓﺮﻡ ﻫﺎﻳﯽ ﮐﻪ ﺣﺎﻟﺘﯽ ﺍﻓﻘﯽ ﻳﺎ ﻋﻤﻮﺩﯼ ﺩﺍﺭﻧﺪ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﻣﺴﺘﻄﻴﻞ، ﻓﺮﻡ ﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺍﺯ ﺧﻄﻮﻁ ﻣﻮﺭﺏ، ﻗﻄﺮﯼ ﻭ ﺯﺍﻭﻳﻪ ﺩﺍﺭ ﺗﺸﮑﻴﻞ ﺷﺪﻩ ﺑﺎﺷﻨﺪ ﺑﻪ ﻣﺜﻠﺚ ﻭ ﻓﺮﻡ ﻫﺎﯼ ﻣﺪﻭﺭ ﻭ ﻣﻨﺤﻨﯽ ﺭﺍ ﺑﻪ ﺩﺍﻳﺮﻩ ﻧﺴﺒﺖ ﺩﺍﺩ.

ﻗﺮﺍﺭ ﺩﺍﺩﻥ ﻫﺮ ﻳﮏ ﺍﺯ ﺍﻳﻦ ﻓﺮﻡ ﻫﺎ (ﻳﺎ ﻓﺮﻡ ﻫﺎﻳﯽ ﺷﺒﻴﻪ ﺑﻪ ﺁﻧﻬﺎ) ﺩﺭ ﮐﻨﺎﺭ ﺩﻳﮕﺮﯼ ﺍﻳﺠﺎﺩ ﺗﻀﺎﺩ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﻣﺜﻼ ﺍﮔﺮ ﺩﺭ ﮐﻨﺎﺭ ﺷﮑﻞ ﺳﺎﺩﻩ ﻣﺴﺘﻄﻴﻠﯽ، ﺷﮑﻞ ﺩﻳﮕﺮﯼ ﺑﺎ ﺧﻄﻮﻁ ﻧﺎﻣﻨﻈﻢ ﻭ ﺷﮑﺴﺘﻪ (ﺣﺎﺻﻞ ﺍﺯ ﺗﺮﮐﻴﺐ ﭼﻨﺪ ﻣﺜﻠﺚ) ﺭﺍ ﻗﺮﺍﺭ ﺩﺍﺩ، ﺑﻼﻓﺎﺻﻠﻪ ﺷﮑﻞ ﺩﻭﻡ ﺑﺎﻋﺚ ﺗﺤﺮﮎ ﺑﺨﺸﻴﺪﻥ ﺑﻪ ﺗﺼﻮﻳﺮ ﻣﯽ ﺷﻮﺩ ﻭ ﻧﻈﺮ ﻣﺨﺎﻃﺐ ﺭﺍ ﺑﻪ ﺧﻮﺩ ﺟﻠﺐ ﻣﯽ ﮐﻨﺪ. ﺍﻳﻦ ﺗﻀﺎﺩ ﻓﺮﻣﯽ ﮐﻤﮏ ﺯﻳﺎﺩﯼ ﺑﻪ ﺍﻧﺘﻘﺎﻝ ﭘﻴﺎﻡ ﺗﺼﻮﻳﺮ ﻣﯽ ﮐﻨﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻧﻤﻮﻧﻪ، ﺩﺭ ﻣﺜﺎﻝ ﺑﺎﻻ ﺍﮔﺮ ﺗﺼﻮﻳﺮ ﺍﺻﻠﯽ ﻳﮏ ﺁﮔﻬﯽ ﺗﺒﻠﻴﻐﺎﺗﯽ ﺑﺎﺷﺪ، ﭘﻴﺎﻣﯽ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﭘﻴﺸﻨﻬﺎﺩ ﻭﻳﮋﻩ ﺑﻪ ﻣﺨﺎﻃﺐ ﻣﯽ ﺧﻮﺍﻫﺪ ﺭﺳﺎﻧﺪﻩ ﺷﻮﺩ، ﺑﺎﻳﺪ ﺩﺭﻭﻥ ﺷﮑﻞ ﻣﺜﻠﺜﯽ ﮐﻪ ﺑﺎ ﺑﻘﻴﻪ ﺗﺼﻮﻳﺮ ﮐﺎﻣﻼ ﺩﺭ ﺗﻀﺎﺩ ﺍﺳﺖ ﻭ ﺑﺮﺟﺴﺘﻪ ﻭ ﺑﻪ ﻧﺤﻮ ﻗﺎﺑﻞ ﺗﻮﺟﻪ ﺟﻠﺐ ﺗﻮﺟﻪ ﻣﯽ ﮐﻨﺪ، ﮔﻨﺠﺎﻧﺪﻩ ﺷﻮﺩ.

ﻳﮏ ﻧﮑﺘﻪ ﻗﺎﺑﻞ ﺗﻮﺟﻪ ﺩﺭ ﺑﻪ ﮐﺎﺭ ﮔﻴﺮﯼ ﻓﺮﻡ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻧﺘﺨﺎﺏ ﻣﮑﺎﻥ ﻓﺮﻡ ﻫﺎﯼ ﻣﺘﻀﺎﺩ، ﺗﻌﺎﺩﻝ ﻳﺎ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﺗﺮﮐﻴﺐ ﺑﻨﺪﯼ ﮐﻠﯽ ﺗﺼﻮﻳﺮ ﺣﺘﻤﺎ ﺑﺎﻳﺪ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ. ﺯﻳﺮﺍ ﻳﮏ ﻓﺮﻡ ﻣﺘﻀﺎﺩ، ﺣﺘﯽ ﺧﻴﻠﯽ ﮐﻮﭼﮏ، ﻣﯽ ﺗﻮﺍﻧﺪ ﺗﻌﺎﺩﻝ ﺗﺮﮐﻴﺐ ﺑﻨﺪﯼ ﺭﺍ ﺑﻪ ﻫﻢ ﺑﺮﻳﺰﺩ ﮐﻪ ﺍﻳﻦ ﻋﺪﻡ ﺗﻌﺎﺩﻝ ﻣﯽ ﺗﻮﺍﻧﺪ ﻋﻤﺪﯼ ﻭ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻃﺮﺍﺡ ﺑﺎﺷﺪ.

ﺩﺭ ﺗﻀﺎﺩ ﻓﺮﻣﯽ، ﻋﻼﻭﻩ ﺑﺮ ﺷﮑﻞ، ﺍﻧﺪﺍﺯﻩ ﻳﺎ ﺗﻨﺎﺳﺐ ﻓﺮﻡ ﻫﺎ ﻧﺴﺒﺖ ﺑﻪ ﻳﮑﺪﻳﮕﺮ ﻧﻴﺰ ﻣﻬﻢ ﺍﺳﺖ. ﺗﻐﻴﻴﺮ ﺩﺭ ﻣﻘﻴﺎﺱ ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﻓﺮﻡ ﻫﺎ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﺎﻋﺚ ﺍﻳﺠﺎﺩ ﺗﻀﺎﺩ ﻭ ﺟﻠﺐ ﺗﻮﺟﻪ ﻣﺨﺎﻃﺐ ﺷﻮﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺍﮔﺮ ﺩﺭ ﮐﻨﺎﺭ ﺗﺼﻮﻳﺮ ﻳﮏ ﺩﺭﺧﺖ ﺑﺰﺭﮒ، ﻣﻴﻮﻩ ﺁﻥ ﺩﺭﺧﺖ ﺑﻪ ﺑﺰﺭﮔﯽ ﺍﺑﻌﺎﺩ ﻫﻤﺎﻥ ﺩﺭﺧﺖ ﺁﻭﺭﺩﻩ ﺷﻮﺩ، ﺍﻳﻦ ﺗﺎﮐﻴﺪ ﻣﻮﺟﺐ ﺑﺮﺟﺴﺘﻪ ﻭ ﺑﺎﺭﺯ ﺷﺪﻥ ﺍﻫﻤﻴﺖ ﻓﺮﻡ ﻣﻴﻮﻩ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺗﻀﺎﺩ ﺍﻧﺪﺍﺯﻩ ﺑﻴﻦ ﻓﺮﻡ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻳﮑﯽ ﺍﺯ ﺍﺑﺘﺪﺍﻳﯽ ﺗﺮﻳﻦ ﻭ ﻣﻮﺛﺮﺗﺮﻳﻦ ﺭﻭﺵ ﻫﺎﯼ ﺑﺮﺟﺴﺘﻪ ﻭ ﻣﺘﻤﺎﻳﺰ ﮐﺮﺩﻥ ﻋﻨﺎﺻﺮ ﺩﺭ ﺗﺼﻮﻳﺮ ﺍﺳﺖ. ﺍﮔﺮ ﺑﻪ ﻧﻘﺎﺷﯽ ﮐﻮﺩﮐﺎﻥ ﺩﻗﺖ ﮐﻨﻴﺪ، ﻧﻤﻮﻧﻪ ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺍﺯ ﮐﺎﺭﺑﺮﺩ ﺍﻳﻦ ﺭﻭﺵ ﺭﺍ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﮐﺎﺭﻫﺎﯼ ﺷﺎﻥ ﺑﺒﻴﻨﻴﺪ. ﮐﻮﺩﮐﺎﻥ ﺑﺎ ﺭﻭﺵ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﻳﺠﺎﺩ ﺗﻀﺎﺩ ﺁﺷﻨﺎ ﻧﻴﺴﺘﻨﺪ، ﺍﻣﺎ ﺑﻪ ﻃﻮﺭ ﻏﺮﻳﺰﯼ ﻓﺮﻡ ﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺑﺮﺍﻳﺸﺎﻥ ﺍﻫﻤﻴﺖ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﺩ، ﺑﺎ ﺍﻧﺪﺍﺯﻩ ﻭ ﺗﻨﺎﺳﺐ ﺑﺰﺭﮒ ﺗﺮﯼ ﺗﺮﺳﻴﻢ ﻣﯽ ﮐﻨﻨﺪ.

 

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺳﺎﻳﻪ- ﺭﻭﺷﻦ ﺩﺭ ﺗﺼﻮﻳﺮ
ﺩﺭﺑﺎﺭﻩ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺗﻀﺎﺩ ﺗﻴﺮﮔﯽ ﻭ ﺭﻭﺷﻨﺎﻳﯽ ﭘﻴﺶ ﺗﺮ ﺩﺭ ﻣﺒﺤﺚ ﺗﻀﺎﺩﻫﺎﯼ ﺭﻧﮕﯽ ﺑﺤﺚ ﺷﺪ. ﺑﻪ ﻃﻮﺭ ﺧﻼﺻﻪ، ﺑﻴﻦ ﻧﻬﺎﻳﺖ ﺭﻭﺷﻨﺎﻳﯽ، ﻳﻌﻨﯽ ﺳﻔﻴﺪ ﻭ ﻧﻬﺎﻳﺖ ﺗﻴﺮﮔﯽ، ﻳﻌﻨﯽ ﺳﻴﺎﻩ، ﺩﺭﺟﺎﺕ ﻣﺨﺘﻠﻔﯽ ﺍﺯ ﺧﺎﮐﺴﺘﺮﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻣﻌﻤﻮﻻ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﺣﺮﻓﻪ ﺍﯼ ﺑﺎ ﺗﺼﻮﻳﺮﺳﺎﺯﯼ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﻧﺪ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺗﻌﺪﺍﺩ ﺭﻧﮓ ﻫﺎﯼ ﺧﺎﮐﺴﺘﺮﯼ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﺑﻴﻦ ﺳﻴﺎﻩ ﻭ ﺳﻔﻴﺪ ﺗﺸﺨﻴﺺ ﺩﻫﻨﺪ. ﺩﺭ ﻭﺍﻗﻊ ﻧﻬﺎﻳﺖ ﺗﻀﺎﺩ ﺳﺎﻳﻪ ﻭ ﺭﻭﺷﻦ ﻭﻗﺘﯽ ﺍﻳﺠﺎﺩ ﻣﯽ ﺷﻮﺩ ﮐﻪ ﺍﻳﻦ ﺩﻭ ﺭﻧﮓ ﺩﺭ ﮐﻨﺎﺭ ﻳﮑﺪﻳﮕﺮ ﻗﺮﺍﺭ ﺑﮕﻴﺮﻧﺪ. ﺭﻭﺷﻨﯽ ﻭ ﺗﺎﺭﻳﮑﯽ ﻧﺴﺒﯽ، ﻣﻴﺰﺍﻥ ﺷﺪﺕ ﮐﻨﺘﺮﺍﺳﺖ ﺭﺍ ﺗﻌﻴﻴﻦ ﻣﯽ ﮐﻨﺪ. ﺩﺭ ﺍﻳﻨﺠﺎ ﻋﻼﻭﻩ ﺑﺮ ﻣﻄﺎﻟﺐ ﺫﮐﺮ ﺷﺪﻩ، ﭼﻨﺪ ﻧﮑﺘﻪ ﺭﺍ ﺑﺎﻳﺪ ﻣﺘﺬﮐﺮ ﮐﺮﺩ.

ﺍﻭﻝ ﺍﻳﻨﮑﻪ ﻣﻴﺰﺍﻥ ﻗﺪﺭﺕ ﺭﻧﮓ ﻫﺎﯼ ﺳﻴﺎﻩ ﻭ ﺳﻔﻴﺪ ﺑﺎ ﻫﻢ ﺑﺮﺍﺑﺮ ﻧﻴﺴﺘﻨﺪ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﮐﻪ ﺍﮔﺮ ﺳﻄﺤﯽ ﺭﺍ ﺑﻪ ﺩﻭ ﻗﺴﻤﺖ ﻣﺴﺎﻭﯼ ﺳﻴﺎﻩ ﻭ ﺳﻔﻴﺪ ﺗﻘﺴﻴﻢ ﮐﻨﻴﻢ، ﺑﻪ ﻃﻮﺭ ﻣﺤﺴﻮﺱ ﻗﺴﻤﺖ ﺗﻴﺮﻩ ﺑﺮ ﻗﺴﻤﺖ ﺭﻭﺷﻦ ﻏﻠﺒﻪ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺗﻌﺎﺩﻝ ﺑﻴﻦ ﺳﻴﺎﻩ ﻭ ﺳﻔﻴﺪ ﺑﺎﻳﺪ ﺗﻨﺎﺳﺒﯽ ﺩﺭ ﺣﺪﻭﺩ ﻳﮏ ﺳﻮﻡ ﻳﺎ ﻳﮏ ﭼﻬﺎﺭﻡ ﺭﺍ ﺑﻴﻦ ﺍﻳﻦ ﺩﻭ ﺭﻧﮓ ﺭﻋﺎﻳﺖ ﮐﺮﺩ.

ﺩﻭﻡ ﺁﻧﮑﻪ ﺍﮔﺮ ﺑﺮﺍﯼ ﺭﺳﻢ ﻳﮏ ﺩﺍﻳﺮﻩ ﮐﺎﻓﯽ ﺍﺳﺖ ﺧﻂ ﻣﺤﻴﻄﯽ ﺁﻥ ﺭﺍ ﺭﺳﻢ ﮐﻨﻴﻢ، ﺍﻣﺎ ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﺍﻳﻦ ﺩﺍﻳﺮﻩ ﺳﻔﻴﺪ ﺭﻧﮓ ﺑﺮﺟﺴﺘﻪ ﺗﺮ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﺪ، ﮐﺎﻓﯽ ﺍﺳﺖ ﺑﺨﺸﯽ ﺍﺯ ﻣﺤﻴﻂ ﺍﻃﺮﺍﻑ ﺁﻥ ﺭﺍ ﺳﻴﺎﻩ ﻧﻤﺎﻳﻴﻢ. ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﺩﺍﻳﺮﻩ ﺳﻔﻴﺪ ﮐﺎﻣﻼ ﺑﺮﺟﺴﺘﻪ ﻭ ﻣﺘﻤﺎﻳﺰ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻫﺮ ﭼﻪ ﺧﻄﻮﻁ ﻣﺤﻴﻄﯽ ﻳﮏ ﺷﮑﻞ ﺗﻀﺎﺩ ﺗﻴﺮﻩ ﻭ ﺭﻭﺷﻦ ﺑﻴﺸﺘﺮﯼ ﺑﻴﻦ ﺩﺭﻭﻥ ﻭ ﺑﻴﺮﻭﻥ ﺁﻥ ﺍﻳﺠﺎﺩ ﮐﻨﻨﺪ، ﺁﻥ ﺷﮑﻞ ﺑﺮﺟﺴﺘﻪ ﺗﺮ ﺑﻪ ﻧﻈﺮ ﺧﻮﺍﻫﺪ ﺭﺳﻴﺪ.

ﻭ ﺩﻳﮕﺮ ﺍﻳﻨﮑﻪ ﺑﻬﺮﻩ ﮔﻴﺮﯼ ﺍﺯ ﺳﺎﻳﻪ ﻭ ﺭﻭﺷﻦ ﮐﺎﺭﺑﺮﺩ ﺯﻳﺎﺩﯼ ﺩﺭ ﻧﻤﺎﻳﺶ ﺩﺍﺩﻥ ﺣﺠﻢ ﻫﺎﯼ ﺳﻪ ﺑﻌﺪﯼ ﺩﺭ ﺗﺼﺎﻭﻳﺮ ﺩﺍﺭﺩ. ﺩﺭ ﺩﻧﻴﺎﯼ ﻭﺍﻗﻌﯽ، ﺗﻤﺎﻣﯽ ﺍﺟﺴﺎﻡ ﺳﻪ ﺑﻌﺪﯼ ﺯﻳﺮ ﻧﻮﺭ ﻃﺒﻴﻌﯽ ﻳﺎ ﻣﺼﻨﻮﻋﯽ ﺩﺍﺭﺍﯼ ﺳﺎﻳﻪ ﻫﺴﺘﻨﺪ. ﻧﻤﺎﻳﺶ ﺳﺎﻳﻪ- ﺭﻭﺷﻦ ﻋﻨﺎﺻﺮ ﺗﺼﺎﻭﻳﺮ ﺑﻪ ﺁﻧﻬﺎ ﺗﺼﻮﺭ ﺣﺠﻢ ﺧﻮﺍﻫﺪ ﺩﺍﺩ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﺩﺭ ﺩﺭﻭﻥ ﮐﺎﺩﺭ ﺩﻭ ﺑﻌﺪﯼ ﻳﮏ ﺗﺼﻮﻳﺮ، ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺑﻌﺪ ﺳﻮﻡ ﺍﺯ ﻃﺮﻳﻖ ﺳﺎﻳﻪ ﻫﺎﯼ ﺍﺟﺴﺎﻡ (ﻫﺮ ﭼﻨﺪ ﺳﺎﺩﻩ ﻭ ﺍﺑﺘﺪﺍﻳﯽ) ﻗﺎﺑﻞ ﺗﺼﻮﺭ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺗﻤﺮﻳﻦ ۱: ﺑﻪ ﻋﻨﻮﺍﻥ ﺗﻤﺮﻳﻦ ﺳﻌﯽ ﮐﻨﻴﺪ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺧﻂ ﻭ ﻧﻘﻄﻪ، ﺑﺎﻓﺖ ﻫﺎﻳﯽ ﺷﺒﻴﻪ ﺷﻴﺸﻪ، ﻓﻠﺰ ﻭ ﭼﻮﺏ ﺍﻳﺠﺎﺩ ﮐﻨﻴﺪ. ﺩﺭ ﺻﻮﺭﺕ ﺍﻣﮑﺎﻥ ﺍﺯ ﺑﺎﻓﺖ ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﻋﮑﺎﺳﯽ ﮐﻨﻴﺪ (ﺳﻴﺎﻩ ﻭ ﺳﻔﻴﺪ) ﻭ ﻧﺘﻴﺠﻪ ﮐﺎﺭ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺁﻥ ﺗﺼﺎﻭﻳﺮ ﻣﻘﺎﻳﺴﻪ ﮐﻨﻴﺪ.

ﺗﻤﺮﻳﻦ ۲: ﻳﮏ ﻋﮑﺲ ﺩﺳﺘﻪ ﺟﻤﻌﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﺑﺎ ﺑﺰﺭﮒ ﻭ ﮐﻮﭼﮏ ﮐﺮﺩﻥ ﺍﻧﺪﺍﺯﻩ ﺍﻓﺮﺍﺩ ﻣﺨﺘﻠﻒ ﺩﺭ ﻋﮑﺲ (ﺑﻪ ﻭﺳﻴﻠﻪ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻳﺎ ﻓﺘﻮﮐﭙﯽ) ﺑﺮﺍﺳﺎﺱ ﺍﻫﻤﻴﺖ ﻫﺮ ﻓﺮﺩ، ﻫﻤﺎﻥ ﺗﺼﻮﻳﺮ ﺭﺍ ﺩﻭﺑﺎﺭﻩ ﺑﺎﺯﺳﺎﺯﯼ ﮐﻨﻴﺪ. ﭼﻪ ﺗﻐﻴﻴﺮﯼ ﺩﺭ ﻣﺤﺘﻮﺍﯼ ﺗﺼﻮﻳﺮ ﺑﻮﺟﻮﺩ ﺁﻣﺪﻩ ﺍﺳﺖ؟
