ﺩﺭﺱ ﺍﻭﻝ- ﺁﺷﻨﺎﻳﯽ ﺑﺎ ﺳﻮﺍﺩ ﺑﺼﺮﯼ

ﺍﺭﺗﺒﺎﻁ ﺑﺼﺮﯼ ﻭ ﮐﺎﺭﺑﺮﺩ ﺁﻥ ﺩﺭ ﻫﻨﺮ ﻭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ
ﻫﻤﻪ ﻣﺎ ﺑﺎ ﺩﻳﺪﻥ ﺳﺮﺩﺭ ﭼﺮﺍﻏﺎﻧﯽ ﺷﺪﻩ ﺧﺎﻧﻪ ﺍﯼ ﻣﺘﻮﺟﻪ ﻣﯽ ﺷﻮﻳﻢ ﺩﺭ ﺁﻧﺠﺎ ﻣﺤﻔﻞ ﺷﺎﺩﯼ ﺑﺮﭘﺎﺳﺖ ﻭ ﺑﺎﻟﻌﮑﺲ، ﺑﻴﺮﻕ ﺳﻴﺎﻩ ﺑﺎﻻﯼ ﺩﺭ ﺧﺎﻧﻪ ﺍﯼ ﺩﻳﮕﺮ ﺑﻪ ﻣﺎ ﻣﯽ ﮔﻮﻳﺪ ﺻﺎﺣﺒﺎﻥ ﺁﻥ ﻣﻨﺰﻝ ﻋﺰﺍﺩﺍﺭﺍﻧﺪ... ﺍﻳﻦ ﻣﺜﺎﻝ ﺍﺯ ﺁﻥ ﺟﻬﺖ ﺁﻭﺭﺩﻩ ﺷﺪ ﺗﺎ ﻧﺸﺎﻥ ﺩﻫﺪ ﻫﺮ ﺭﻭﺯ ﻣﺎ ﺑﺎ ﺩﻳﺪﻥ ﺗﺎﺑﻠﻮﻫﺎ، ﺗﺼﺎﻭﻳﺮ، ﻋﻼﻣﺖ ﻫﺎ ﻭ ﻧﺸﺎﻧﻪ ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ، ﺧﻮﺩﺁﮔﺎﻩ ﻳﺎ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ، ﺍﻃﻼﻋﺎﺕ ﻭ ﭘﻴﺎﻡ ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺭﺍ ﺩﺭﻳﺎﻓﺖ ﻭ ﺩﺭﮎ ﻣﯽ ﮐﻨﻴﻢ. ﻓﻬﻤﻴﺪﻥ ﻣﻌﻨﺎﯼ ﺑﻌﻀﯽ ﺍﺯ ﺍﻳﻦ ﻋﻼﻣﺖ ﻫﺎ، ﻣﺜﻞ ﺩﻳﺪﻥ ﭼﺮﺍﻍ ﻗﺮﻣﺰ ﺭﺍﻫﻨﻤﺎﻳﯽ، ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ ﺍﺳﺖ. ﺍﻣﺎ ﺷﺎﻳﺪ ﺩﺭ ﺑﻌﻀﯽ ﺍﺯ ﻣﻮﺍﺭﺩ ﻣﺎ ﻗﺎﺩﺭ ﺑﻪ ﺩﺭﮎ ﻣﻔﻬﻮﻡ ﻳﮏ ﺗﺼﻮﻳﺮ ﻧﺒﺎﺷﻴﻢ. ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﺭﺍ ﻣﻤﮑﻦ ﺍﺳﺖ ﻫﻨﮕﺎﻡ ﺩﻳﺪﻥ ﻳﮏ ﺗﺎﺑﻠﻮﯼ ﻧﻘﺎﺷﯽ ﺗﺠﺮﺑﻪ ﮐﺮﺩﻩ ﺑﺎﺷﻴﺪ.

ﮔﺎﻫﯽ ﺍﻭﻗﺎﺕ ﺩﻳﺪﻥ ﻳﮏ ﺗﺼﻮﻳﺮ ﻣﺎ ﺭﺍ ﺧﻮﺷﺤﺎﻝ ﻳﺎ ﻏﻤﮕﻴﻦ، ﻣﻀﻄﺮﺏ ﻳﺎ ﺁﺭﺍﻡ ﻣﯽ ﮐﻨﺪ، ﺍﻣﺎ ﺍﮔﺮ ﺍﺯ ﻣﺎ ﺑﭙﺮﺳﻨﺪ ﭼﻪ ﭼﻴﺰﯼ ﺩﺭ ﺁﻥ ﺗﺼﻮﻳﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﭼﻨﻴﻦ ﺍﺣﺴﺎﺳﯽ ﺭﺍ ﺩﺭ ﻣﺎ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ، ﻳﺎ ﭼﻪ ﻋﺎﻣﻠﯽ ﺩﺭ ﺁﻥ ﺗﺼﻮﻳﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺍﻳﻦ ﭼﻨﻴﻦ ﺣﺲ ﺧﻮﺷﺎﻳﻨﺪ ﻳﺎ ﻧﺎﺧﻮﺷﺎﻳﻨﺪﯼ ﺭﺍ ﺍﻳﺠﺎﺩ ﻣﯽ ﮐﻨﺪ، ﻧﻤﯽ ﺩﺍﻧﻴﻢ ﺁﻥ ﭼﻴﺰ ﭼﻴﺴﺖ.

ﺑﺎ ﺍﻳﻦ ﻣﻘﺪﻣﻪ ﻣﺸﺨﺺ ﻣﯽ ﺷﻮﺩ ﮐﻪ ﺗﺎﺑﻠﻮﻫﺎ، ﻧﺸﺎﻧﻪ ﻫﺎ، ﻋﻼﻣﺖ ﻫﺎ ﻭ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻫﺮ ﭘﺪﻳﺪﻩ ﺑﺼﺮﯼ ﮐﻪ ﻣﺎ ﺑﻪ ﺁﻥ ﺗﺼﻮﻳﺮ ﻣﯽ ﮔﻮﻳﻴﻢ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺣﺎﻭﯼ ﭘﻴﺎﻡ ﻳﺎ ﻣﻌﻨﺎﻳﯽ ﺑﺎﺷﻨﺪ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻫﺮ ﺗﺼﻮﻳﺮ ﻣﺎﻧﻨﺪ ﻳﮏ ﺟﻤﻠﻪ ﻳﺎ ﻣﺘﻦ ﻧﻮﺷﺘﺎﺭﯼ ﺣﺎﻭﯼ ﭘﻴﺎﻣﯽ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺩﻳﺪﻥ ﺁﻥ ﻣﯽ ﺗﻮﺍﻧﻴﻢ ﻣﻔﻬﻮﻣﺶ ﺭﺍ ﺩﺭﮎ ﮐﻨﻴﻢ؛ ﻫﻤﺎﻧﮕﻮﻧﻪ ﮐﻪ ﺑﺎ ﺷﻨﻴﺪﻥ ﻳﮏ ﺟﻤﻠﻪ، ﺧﺒﺮﯼ ﻳﺎ ﺩﺳﺘﻮﺭﯼ ﺑﻪ ﻣﺎ ﻣﯽ ﺭﺳﺪ، ﻫﺮ ﺗﺼﻮﻳﺮ ﻫﻢ ﭘﻴﺎﻣﯽ ﺑﺮﺍﯼ ﻣﺎ ﺩﺍﺭﺩ.

ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﺩﺍﺷﺖ ﮐﻪ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﮐﺎﺭﺑﺮﺩ ﺯﺑﺎﻥ ﺗﻨﻬﺎ ﻣﺨﺘﺺ ﺑﻪ ﺍﺩﻳﺒﺎﻥ ﻭ ﻧﻮﻳﺴﻨﺪﮔﺎﻥ ﻭ ﺷﺎﻋﺮﺍﻥ ﻧﻴﺴﺖ، ﮐﺎﺭﺑﺮﺩ ﺯﺑﺎﻥ ﺗﺼﻮﻳﺮﯼ ﻫﻢ ﺑﻪ ﻧﻘﺎﺷﺎﻥ ﻭ ﺗﺼﻮﻳﺮﮔﺮﺍﻥ ﻭ ﺳﺎﻳﺮ ﻫﻨﺮﻣﻨﺪﺍﻥ ﻣﻨﺤﺼﺮ ﻧﻤﯽ ﺷﻮﺩ. ﻫﻤﻪ ﺍﻓﺮﺍﺩ، ﮐﻤﺘﺮ ﻳﺎ ﺑﻴﺸﺘﺮ، ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺧﻮﺩ ﺑﺎ ﺁﻥ ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﻧﺪ.

 

ﺍﺭﺗﺒﺎﻁ ﮐﻼﻣﯽ ﻭ ﺍﺭﺗﺒﺎﻁ ﺗﺼﻮﻳﺮﯼ
ﻣﺎ ﺍﻧﺴﺎﻥ ﻫﺎ ﺍﺯ ﺩﻭ ﺭﻭﺵ ﺑﺮﺍﯼ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻴﻢ، ﺍﻭﻝ ﺍﺭﺗﺒﺎﻁ ﮐﻼﻣﯽ (ﺷﺎﻣﻞ ﺣﺮﻑ ﺯﺩﻥ، ﺧﻮﺍﻧﺪﻥ ﻭ ﻧﻮﺷﺘﻦ) ﻭ ﺩﻭﻡ ﺍﺭﺗﺒﺎﻁ ﺗﺼﻮﻳﺮﯼ. ﮐﻼﻡ (ﺯﺑﺎﻥ) ﻣﻬﻤﺘﺮﻳﻦ ﻋﺎﻣﻞ ﺍﺭﺗﺒﺎﻃﯽ ﻣﺎ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺍﺳﺖ. ﻣﺎ ﺣﺮﻑ ﺯﺩﻥ ﺭﺍ ﺩﺭ ﺧﺎﻧﻪ ﻓﺮﺍ ﻣﯽ ﮔﻴﺮﻳﻢ ﻭ ﺧﻮﺍﻧﺪﻥ ﻭ ﻧﻮﺷﺘﻦ ﺩﺭ ﻣﺪﺭﺳﻪ ﺑﻪ ﻣﺎ ﺁﻣﻮﺯﺵ ﺩﺍﺩﻩ ﻣﯽ ﺷﻮﺩ. ﺑﺮﺍﯼ ﺑﺮﻗﺮﺍﺭﯼ ﺍﺭﺗﺒﺎﻁ ﮐﻼﻣﯽ ﺑﺎﻳﺪ ﻗﻮﺍﻋﺪ ﻭ ﺍﺻﻮﻟﯽ ﻣﺸﺨﺺ ﺑﻪ ﮐﺎﺭ ﮔﺮﻓﺘﻪ ﺷﻮﺩ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻳﮏ ﺟﻤﻠﻪ ﻣﻌﻨﺎﺩﺍﺭ ﺑﺎﺷﺪ ﻫﻢ ﺑﺎﻳﺪ ﻫﻢ ﺍﺯ ﮐﻠﻤﺎﺕ ﻭ ﻫﻢ ﺍﺯ ﺳﺎﺧﺘﺎﺭ ﻣﻨﺎﺳﺐ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ. ﭘﺸﺖ ﺳﺮ ﻫﻢ ﭼﻴﺪﻥ ﺗﺼﺎﺩﻓﯽ ﮐﻠﻤﺎﺕ ﺑﯽ ﺭﺑﻂ ﻫﻴﭻ ﻭﻗﺖ ﻣﻌﻨﺎﺩﺍﺭ ﻧﻴﺴﺖ.

ﺍﻣﺎ ﺩﺭ ﻣﻮﺭﺩ ﺍﺭﺗﺒﺎﻁ ﺗﺼﻮﻳﺮﯼ، ﺍﮐﺜﺮ ﺍﻓﺮﺍﺩ ﺍﺯ ﺁﻥ ﺑﻪ ﺻﻮﺭﺗﯽ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺍﺯ ﻗﻮﺍﻋﺪ ﻭ ﺍﺻﻮﻝ ﺁﻥ ﺁﮔﺎﻩ ﻧﻴﺴﺘﻨﺪ. ﻣﯽ ﺗﻮﺍﻥ ﺑﺎ ﻗﻴﺎﺱ ﺑﺎ ﺍﺭﺗﺒﺎﻁ ﮐﻼﻣﯽ ﺑﻪ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﺭﺳﻴﺪ ﮐﻪ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﺘﻮﺍﻧﻴﻢ ﺩﺭ ﺍﺭﺗﺒﺎﻃﺎﺕ ﺗﺼﻮﻳﺮﯼ ﺍﺯ ﺍﻳﻦ ﺳﻄﺢ ﺍﺑﺘﺪﺍﻳﯽ ﻓﺮﺍﺗﺮ ﺑﺮﻭﻳﻢ ﻭ ﺍﺯ ﺁﻥ ﺑﻬﺘﺮ ﻭ ﺑﻴﺸﺘﺮ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ، ﻻﺯﻡ ﺍﺳﺖ ﺗﺎ ﺍﺻﻮﻝ ﻭ ﻗﻮﺍﻋﺪ ﺍﻭﻟﻴﻪ ﺍﺭﺗﺒﺎﻁ ﺑﺼﺮﯼ ﺭﺍ ﺑﻴﺎﻣﻮﺯﻳﻢ، ﻫﻤﺎﻧﮕﻮﻧﻪ ﮐﻪ ﺩﺳﺘﻮﺭ ﺯﺑﺎﻥ ﺭﺍ ﻳﺎﺩ ﮔﺮﻓﺘﻪ ﺍﻳﻢ. ﺍﻳﻦ ﺗﻮﺍﻧﺎﻳﯽ ﺑﺮﺍﯼ ﻧﻘﺎﺵ، ﻋﮑﺎﺱ ﻳﺎ ﮔﺮﺍﻓﻴﺴﺖ ﺷﺪﻥ ﻣﺎ ﻧﻴﺴﺖ. ﺑﻠﮑﻪ ﻣﺎ ﺭﺍ ﻗﺎﺩﺭ ﻣﯽ ﺳﺎﺯﺩ ﻋﻤﻖ ﻭ ﻣﻴﺰﺍﻥ ﺍﺭﺗﺒﺎﻃﻤﺎﻥ ﺭﺍ ﺑﺎ ﻣﺤﻴﻂ ﺍﻃﺮﺍﻑ ﺧﻮﺩ ﺑﻴﺸﺘﺮ ﮐﻨﻴﻢ.

 

ﺩﻟﻴﻞ ﺍﻫﻤﻴﺖ ﺍﺭﺗﺒﺎﻁ ﺗﺼﻮﻳﺮﯼ ﻭ ﮔﺴﺘﺮﺵ ﺁﻥ ﺩﺭ ﺩﻭﺭﺍﻥ ﻣﻌﺎﺻﺮ
ﺣﺲ ﺑﻴﻨﺎﻳﯽ ﻗﻮﻳﺘﺮﻳﻦ ﺣﺲ ﺍﺩﺭﺍﮐﯽ ﺍﻧﺴﺎﻥ ﻫﺎ ﺍﺳﺖ. ﻣﺎ ﺑﺮﺍﯼ ﺩﺭﻳﺎﻓﺖ ﺍﻃﻼﻋﺎﺕ ﺩﺭﺑﺎﺭﻩ ﻫﺮ ﭘﺪﻳﺪﻩ ﺍﯼ، ﻣﺸﺘﺎﻕ ﺗﺮ ﺑﻪ ﺩﻳﺪﻥ ﺗﺼﻮﻳﺮﯼ ﺍﺯ ﺁﻥ ﻫﺴﺘﻴﻢ ﺗﺎ ﺷﻨﻴﺪﻥ ﻳﺎ ﺧﻮﺍﻧﺪﻥ ﺩﺭﺑﺎﺭﻩ ﺁﻥ. ﺯﻳﺮﺍ ﺗﺼﻮﻳﺮ ﻫﺮ ﺷﯽﺀ ﻧﺰﺩﻳﮏ ﺗﺮﻳﻦ ﭼﻴﺰ ﺑﻪ ﺁﻥ ﻭﺍﻗﻌﻴﺖ ﺍﺳﺖ ﻭ ﺑﻪ ﺗﺠﺮﺑﻪ ﻣﺴﺘﻘﻴﻢ ﻣﺎ ﺷﺒﺎﻫﺖ ﺩﺍﺭﺩ ﻭ ﻳﮏ ﺗﺼﻮﻳﺮ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﺎ ﺻﺮﻑ ﺍﻧﺮﮊﯼ ﻧﺴﺒﺘﺎ ﮐﻢ، ﺑﺴﻴﺎﺭ ﺳﺮﻳﻊ ﻭ ﺁﺳﺎﻥ ﺍﻃﻼﻋﺎﺕ ﺯﻳﺎﺩﯼ ﺭﺍ ﺑﻪ ﻣﺎ ﻣﻨﺘﻘﻞ ﮐﻨﺪ (ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﺍﮐﺜﺮ ﺍﻓﺮﺍﺩ ﺩﻳﺪﻥ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺭﺍ ﺑﻪ ﺧﻮﺍﻧﺪﻥ ﺭﻭﺯﻧﺎﻣﻪ ﺗﺮﺟﻴﺢ ﻣﯽ ﺩﻫﻨﺪ).

ﭘﻴﺶ ﺍﺯ ﺍﺑﺪﺍﻉ ﺯﺑﺎﻥ ﻭ ﺑﻌﺪﻫﺎ ﺧﻂ، ﺗﺼﺎﻭﻳﺮ ﺗﻨﻬﺎ ﺭﺍﻩ ﺍﺭﺗﺒﺎﻁ ﻣﻴﺎﻥ ﺍﻧﺴﺎﻧﻬﺎ ﺑﻮﺩ. ﺑﻪ ﺗﺪﺭﻳﺞ ﺯﺑﺎﻥ ﻧﻘﺶ ﻣﻬﻤﯽ ﺩﺭ ﺍﺭﺗﺒﺎﻃﺎﺕ ﻭ ﮔﺴﺘﺮﺵ ﺗﻤﺪﻥ ﺑﺸﺮﯼ ﺍﻳﻔﺎﺀ ﮐﺮﺩ. ﺑﺎ ﺍﺧﺘﺮﺍﻉ ﺩﺳﺘﮕﺎﻩ ﭼﺎﭖ ﺍﻳﻦ ﻧﻘﺶ ﺑﺎﺯ ﻫﻢ ﺑﻴﺸﺘﺮ ﺷﺪ. ﺍﻣﺎ ﺩﺭ ﺩﻭﺭﺍﻥ ﻣﻌﺎﺻﺮ ﻭ ﺑﻌﺪ ﺍﺯ ﺍﺧﺘﺮﺍﻉ ﺩﻭﺭﺑﻴﻦ ﻋﮑﺎﺳﯽ ﻭ ﺑﻌﺪﻫﺎ ﺳﻴﻨﻤﺎ ﻭ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻭ ﺭﻭﺍﺝ ﺑﻴﺶ ﺍﺯ ﺑﻴﺶ ﺗﺒﻠﻴﻐﺎﺕ، ﺩﻭﺑﺎﺭﻩ ﺯﺑﺎﻥ ﺗﺼﻮﻳﺮ ﺍﻫﻤﻴﺖ ﭘﻴﺪﺍ ﮐﺮﺩ ﻭ ﺭﻭﺯ ﺑﻪ ﺭﻭﺯ ﮐﺎﺭﺑﺮﺩ ﺁﻥ ﺑﻴﺸﺘﺮ ﺷﺪﻩ ﺍﺳﺖ.

ﺯﻣﺎﻧﯽ ﮐﻪ ﺩﺳﺘﮕﺎﻩ ﭼﺎﭖ ﺍﺧﺘﺮﺍﻉ ﺷﺪ ﻭ ﮐﺎﺭﺑﺮﺩ ﺯﺑﺎﻥ ﻧﻮﺷﺘﺎﺭﯼ ﻫﻢ ﻓﺮﺍﮔﻴﺮ ﺷﺪ، ﻣﺘﻌﺎﻗﺐ ﺁﻥ ﺩﻭ ﺍﺗﻔﺎﻕ ﻣﻬﻢ ﺍﻓﺘﺎﺩ. ﺍﻭﻝ ﺗﻌﺪﺍﺩ ﺑﺎﺳﻮﺍﺩﺍﻥ ﺭﻭﺯ ﺑﻪ ﺭﻭﺯ ﺑﻴﺸﺘﺮ ﺷﺪﻧﺪ ﻭ ﺩﻭﻡ ﺑﻪ ﺗﺪﺭﻳﺞ ﺟﺎﯼ ﮐﻠﻤﺎﺕ ﺩﺭ ﺟﻤﻠﻪ ﺍﺯ ﻧﻈﺮ ﺩﺳﺘﻮﺭﯼ ﺍﻫﻤﻴﺘﯽ ﺑﻴﺶ ﺍﺯ ﺻﺮﻑ ﮐﻠﻤﺎﺕ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﭘﻴﺪﺍ ﮐﺮﺩ ﻭ ﺩﺳﺘﻮﺭ ﺯﺑﺎﻥ ﺷﮑﻞ ﺗﮑﺎﻣﻞ ﻳﺎﻓﺘﻪ ﺗﺮﯼ ﺑﻪ ﺧﻮﺩ ﮔﺮﻓﺖ. ﺩﺭ ﻣﻮﺭﺩ ﺍﺭﺗﺒﺎﻁ ﺑﺼﺮﯼ ﻫﻢ ﻫﻤﻴﻦ ﺍﺗﻔﺎﻕ ﺑﺎ ﺭﻭﺍﺝ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺩﻭﺭﺑﻴﻦ ﻋﮑﺎﺳﯽ ﺩﺭ ﻗﺮﻥ ﺑﻴﺴﺘﻢ ﻭ ﭘﺲ ﺍﺯ ﺁﻥ ﺑﻪ ﺑﺎﺯﺍﺭ ﺁﻣﺪﻥ ﺩﻭﺭﺑﻴﻦ ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺍﻓﺘﺎﺩﻩ ﺍﺳﺖ. ﺍﻣﺮﻭﺯ ﻫﺮ ﮐﺴﯽ ﺑﺎ ﻳﮏ ﺩﻭﺭﺑﻴﻦ ﮐﻮﭼﮏ ﻭ ﻧﺮﻡ ﺍﻓﺰﺍﺭﯼ ﺳﺎﺩﻩ ﺗﻮﺍﻧﺎﻳﯽ ﺧﻠﻖ ﻳﮏ ﺗﺼﻮﻳﺮ ﺭﺍ ﺩﺍﺭﺩ. ﮐﺎﺭﯼ ﮐﻪ ﺩﺭ ﺳﺎﻝ ﻫﺎﯼ ﺩﻭﺭ ﺗﻨﻬﺎ ﺍﺯ ﻋﻬﺪﻩ ﻋﺪﻩ ﺍﯼ ﻫﻨﺮﻣﻨﺪ ﻭ ﺍﻓﺮﺍﺩ ﺧﺎﺹ ﺑﺮﻣﯽ ﺁﻣﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﻣﺮﻭﺯﻩ ﺑﻪ ﺟﺎﺳﺖ ﺍﺯ ﺍﺻﻮﻝ ﻭ ﻗﻮﺍﻋﺪﯼ ﺣﺮﻑ ﺑﺰﻧﻴﻢ ﮐﻪ ﺑﻪ ﺧﻠﻖ ﻭ ﺩﺭﮎ ﭘﻴﺎﻡ ﻫﺎﯼ ﺑﺼﺮﯼ ﮐﻤﮏ ﮐﻨﺪ.

ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺑﻪ ﻳﮏ ﻧﮑﺘﻪ ﺗﻮﺟﻪ ﺩﺍﺷﺖ ﮐﻪ ﺩﺭ ﺟﻬﺎﻥ ﺣﺪﻭﺩ ﺳﻪ ﻫﺰﺍﺭ ﺯﺑﺎﻥ ﻭ ﮔﻮﻳﺶ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺍﮐﺜﺮ ﺁﻧﻬﺎ ﺩﺍﺭﺍﯼ ﻗﻮﺍﻋﺪ ﻭ ﺍﺻﻮﻝ ﻧﺴﺒﺘﺎ ﻣﻨﻀﺒﻂ ﻭ ﻣﻨﻄﻘﯽ ﻫﺴﺘﻨﺪ. ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﺭﺗﺒﺎﻁ ﺗﺼﻮﻳﺮﯼ ﺑﻪ ﻣﺮﺍﺗﺐ ﺟﻬﺎﻥ ﺷﻤﻮﻝ ﺗﺮ ﺍﺳﺖ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﻧﺴﺒﺖ ﻗﻮﺍﻋﺪ ﻭ ﺍﺻﻮﻝ ﺁﻥ ﺍﻧﻌﻄﺎﻑ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﺩ.

 

ﻓﺮﺍﮔﻴﺮ ﺑﻮﺩﻥ ﺍﺻﻮﻝ ﻭ ﻗﻮﺍﻋﺪ ﺍﺭﺗﺒﺎﻁ ﺑﺼﺮﯼ
ﺍﺩﺭﺍﮎ ﺑﺼﺮﯼ ﻣﺎ ﺍﺯ ﻣﺤﻴﻂ ﺍﻃﺮﺍﻑ ﻣﺎﻥ ﺑﺴﺘﮕﯽ ﺑﻪ ﻋﻮﺍﻣﻞ ﻣﺨﺘﻠﻔﯽ ﺩﺍﺭﺩ. ﻓﻴﺰﻳﻮﻟﻮﮊﯼ ﺩﺳﺘﮕﺎﻩ ﺑﻴﻨﺎﻳﯽ ﻭ ﻣﺎﻫﻴﺖ ﮐﺎﺭ ﺳﻴﺴﺘﻢ ﺑﺪﻥ ﺍﻧﺴﺎﻥ ﺗﺎﺛﻴﺮ ﺯﻳﺎﺩﯼ ﺩﺭ ﺍﺩﺭﺍﮎ ﻣﺎ ﺍﺯ ﭘﺪﻳﺪﻩ ﻫﺎﯼ ﺑﺼﺮﯼ ﺩﺍﺭﺩ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺩﺭﻳﺎﻓﺖ ﻭ ﺗﻔﺴﻴﺮ ﻣﺎ ﺍﺯ ﻳﮏ ﺗﺼﻮﻳﺮ ﺩﺭ ﺣﺎﻟﺖ ﺳﺎﮐﻦ ﻭ ﻧﺸﺴﺘﻪ ﺑﺎ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺩﺭ ﺣﺎﻝ ﺣﺮﮐﺖ ﻫﺴﺘﻴﻢ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﻫﻤﭽﻨﻴﻦ ﭘﻴﺮ ﻳﺎ ﺟﻮﺍﻥ ﺑﻮﺩﻥ، ﻣﺮﺩ ﻳﺎ ﺯﻥ ﺑﻮﺩﻥ، ﺑﭽﻪ ﻳﺎ ﺑﺰﺭﮔﺴﺎﻝ ﺑﻮﺩﻥ ﻫﻢ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺭ ﻧﺤﻮﻩ ﺍﺩﺭﺍﮎ ﻣﺎ ﺍﺯ ﺗﺼﺎﻭﻳﺮ ﺗﺎﺛﻴﺮﮔﺬﺍﺭ ﺑﺎﺷﺪ.

ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻗﺎﺏ ﻋﮑﺴﯽ ﺭﺍ ﺩﺭ ﺧﺎﻧﻪ ﺑﻴﺎﺩ ﺁﻭﺭﻳﺪ ﮐﻪ ﺩﺭ ﮐﻮﺩﮐﯽ ﭼﻘﺪﺭ ﺑﺰﺭﮒ ﺑﻪ ﻧﻈﺮ ﻣﯽ ﺭﺳﻴﺪ ﻭ ﺩﺭ ﺑﺰﺭﮔﺴﺎﻟﯽ ﺍﺯ ﺍﺑﻬﺖ ﺍﻓﺘﺎﺩ. ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻨﻬﺎ ﻭﺿﻌﻴﺖ ﺭﻭﺣﯽ، ﺧﺼﻮﺻﻴﺎﺕ ﻓﺮﻫﻨﮕﯽ ﻭ ﺑﻌﻀﯽ ﺿﺮﻭﺭﻳﺎﺕ ﻣﺤﻴﻂ ﺩﺭ ﺍﺩﺭﺍﮎ ﻣﺎ ﻣﻮﺛﺮ ﻫﺴﺘﻨﺪ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻧﺤﻮﻩ ﻧﮕﺮﺵ ﻣﺎ ﺑﻪ ﺟﻬﺎﻥ ﺩﺭ ﭼﮕﻮﻧﮕﯽ ﺁﻧﭽﻪ ﻣﯽ ﺑﻴﻨﻴﻢ ﺑﯽ ﺗﺎﺛﻴﺮ ﻧﻴﺴﺖ. ﺩﺭ ﻧﻬﺎﻳﺖ ﻣﺤﻴﻂ ﺯﻧﺪﮔﯽ ﻣﺎ ﺗﺎﺛﻴﺮ ﺯﻳﺎﺩﯼ ﺑﺮ ﻧﺤﻮﻩ ﺍﺩﺭﺍﮎ ﻣﺎ ﺍﺯ ﺗﺼﺎﻭﻳﺮ ﺩﺍﺭﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﮐﺴﯽ ﮐﻪ ﺳﺎﻝ ﻫﺎ ﺩﺭ ﮐﻮﻳﺮ ﺯﻧﺪﮔﯽ ﮐﺮﺩﻩ، ﻭﻗﺘﯽ ﺑﻪ ﻣﻨﻄﻘﻪ ﺍﯼ ﮐﻮﻫﺴﺘﺎﻧﯽ ﻣﯽ ﺭﻭﺩ ﺩﺭ ﺍﺩﺭﺍﮎ ﺗﺼﺎﻭﻳﺮ ﻣﺤﻴﻂ ﺧﻮﺩ ﺗﺎ ﻣﺪﺗﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﭼﺎﺭ ﺳﺮﺩﺭﮔﻤﯽ ﺷﻮﺩ. ﻳﺎ ﺗﺼﺎﻭﻳﺮ ﻗﺎﺑﻞ ﻓﻬﻢ ﺑﺮﺍﯼ ﺍﺳﮑﻴﻤﻮﻫﺎ ﺑﺎ ﺁﻧﭽﻪ ﻣﺎ ﺩﺭﮎ ﻣﯽ ﮐﻨﻴﻢ ﺑﺴﻴﺎﺭ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ.

ﺍﻣﺎ ﻋﻠﻴﺮﻏﻢ ﺗﻤﺎﻡ ﺍﻳﻦ ﺍﺧﺘﻼﻓﺎﺕ ﻧﺎﺷﯽ ﺍﺯ ﻓﻴﺰﻳﻮﻟﻮﮊﯼ ﻭ ﻓﺮﻫﻨﮓ ﻭ ﻣﺤﻴﻂ ﮐﻪ ﺭﻭﯼ ﺍﺩﺭﺍﮎ ﻣﺎ ﺗﺎﺛﻴﺮ ﻣﯽ ﮔﺬﺍﺭﺩ، ﺍﺳﺎﺱ ﻧﺤﻮﻩ ﺩﺭﻳﺎﻓﺖ ﺑﺼﺮﯼ ﺩﺭ ﺑﻴﻦ ﻫﻤﻪ ﺍﻓﺮﺍﺩ ﺗﻘﺮﻳﺒﺎ ﻳﮑﺴﺎﻥ ﺍﺳﺖ ﻭ ﺍﻏﻠﺐ ﺍﻧﺴﺎﻥ ﻫﺎ ﺩﺭﻳﺎﻓﺖ ﺗﻘﺮﻳﺒﺎ ﻣﺸﺎﺑﻬﯽ ﺩﺭ ﮐﻠﻴﺎﺕ ﺍﺯ ﺗﺼﺎﻭﻳﺮ ﺩﺍﺭﻧﺪ. ﺗﻘﺮﻳﺒﺎ ﻫﻤﻪ ﺍﻓﺮﺍﺩ ﺑﺮ ﺗﻌﺮﻳﻒ ﻣﻔﺎﻫﻴﻤﯽ ﻣﺎﻧﻨﺪ ﺣﺮﮐﺖ، ﺍﻳﺴﺘﺎﻳﯽ، ﺗﻌﺎﺩﻝ ﻭ ﻣﺎﻧﻨﺪ ﺍﻳﻨﻬﺎ ﮐﻪ ﮐﺎﺭﺑﺮﺩ ﺯﻳﺎﺩﯼ ﺩﺭ ﺍﺭﺗﺒﺎﻁ ﺑﺼﺮﯼ ﺩﺍﺭﺩ، ﺍﺗﻔﺎﻕ ﻧﻈﺮ ﺩﺍﺭﻧﺪ. ﻗﻮﺍﻋﺪ ﻭ ﺍﺻﻮﻝ ﺳﻮﺍﺩ ﺑﺼﺮﯼ ﺑﺮﺍﺳﺎﺱ ﻫﻤﻴﻦ ﺍﺩﺭﺍﮎ ﻣﺸﺘﺮﮎ ﺑﻴﻦ ﺍﻓﺮﺍﺩ ﻣﺨﺘﻠﻒ ﺗﺪﻭﻳﻦ ﺷﺪﻩ ﺍﺳﺖ.

 

ﺍﺳﺎﺱ ﻭ ﻣﺤﺘﻮﯼ ﺳﻮﺍﺩ ﺑﺼﺮﯼ
ﻫﺮ ﺗﺼﻮﻳﺮ ﺷﺎﻣﻞ ﻳﮏ ﺳﺮﯼ ﺍﺟﺰﺍﺀ ﻭ ﻋﻨﺎﺻﺮ ﻣﺨﺘﻠﻒ ﻣﺎﻧﻨﺪ ﻧﻘﻄﻪ، ﺧﻂ، ﺷﮑﻞ ﻭ ﺭﻧﮓ ﺍﺳﺖ ﮐﻪ ﺩﺭﻭﻥ ﻳﮏ ﻗﺎﺏ ﺩﺭ ﮐﻨﺎﺭ ﻫﻢ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﻧﺪ. ﺍﻳﻦ ﻋﻨﺎﺻﺮ ﻭ ﺍﺟﺰﺍﺀ ﺩﺭﺳﺖ ﻣﺎﻧﻨﺪ ﮐﻠﻤﺎﺕ ﻫﺴﺘﻨﺪ ﮐﻪ ﻧﺤﻮﻩ ﺗﺮﮐﻴﺐ ﮐﺮﺩﻥ ﺁﻧﻬﺎ ﺑﺎ ﻫﻢ ﺟﻤﻠﻪ ﺍﯼ ﻣﻌﻨﺎﺩﺍﺭ ﺭﺍ ﻣﯽ ﺳﺎﺯﺩ. ﭘﺲ ﺑﺮﺍﯼ ﺑﻴﺎﻥ ﻣﻔﻬﻮﻣﯽ ﺑﺼﺮﯼ، ﺑﺎﻳﺪ ﻫﻢ ﺑﺎ ﻋﻨﺎﺻﺮ ﻣﺨﺘﻠﻒ ﺷﮑﻞ ﺩﻫﻨﺪﻩ ﺗﺼﻮﻳﺮ ﺁﺷﻨﺎ ﺷﺪ ﻭ ﻫﻢ ﻧﺤﻮﻩ ﺗﺮﮐﻴﺐ ﺁﻧﻬﺎ ﺭﺍ ﺑﺎ ﻫﻢ ﻳﺎﺩ ﮔﺮﻓﺖ. ﻣﻌﻨﺎ ﻭ ﻣﻔﻬﻮﻡ ﻳﮏ ﺗﺼﻮﻳﺮ ﺍﺯ ﺍﻧﺘﺨﺎﺏ ﻋﻨﺎﺻﺮ ﻣﺨﺘﻠﻒ ﻭ ﻧﺤﻮﻩ ﭼﻴﺪﻩ ﺷﺪﻥ (ﮐﻤﭙﻮﺯﻳﺴﻴﻮﻥ) ﺍﻳﻦ ﻋﻨﺎﺻﺮ ﮐﻨﺎﺭ ﻫﻢ ﺑﻪ ﺩﺳﺖ ﻣﯽ ﺁﻳﺪ.

ﺗﺎ ﭘﻴﺶ ﺍﺯ ﺍﺧﺘﺮﺍﻉ ﺩﺳﺘﮕﺎﻩ ﻋﮑﺎﺳﯽ، ﺗﺼﺎﻭﻳﺮ ﻭ ﻧﻘﺎﺷﯽ ﻫﺎ، ﻣﺤﻴﻂ ﺧﺎﺭﺝ ﻭ ﻭﺍﻗﻌﻴﺖ ﻫﺎﯼ ﻋﻴﻨﯽ ﺭﺍ ﺑﻴﺸﺘﺮ ﺷﺒﻴﻪ ﺳﺎﺯﯼ ﻣﯽ ﮐﺮﺩﻧﺪ. ﻣﺜﻼ ﺑﺮﺍﯼ ﻧﻤﺎﻳﺶ ﺭﻭﺣﺎﻧﻴﺖ ﺩﺭ ﺩﻧﻴﺎﯼ ﻣﺴﻴﺤﯽ، ﭼﻬﺮﻩ ﻣﺴﻴﺢ ﻧﻮﺭﺍﻧﯽ ﺗﺼﻮﻳﺮ ﻣﯽ ﺷﺪ. ﻳﺎ ﭼﺸﻢ ﺍﻧﺪﺍﺯﯼ ﺍﺯ ﭼﻤﻨﺰﺍﺭﯼ ﺳﺒﺰ ﺍﺣﺴﺎﺳﯽ ﺍﺯ ﺁﺭﺍﻣﺶ ﺭﺍ ﻣﻨﺘﻘﻞ ﻣﯽ ﮐﺮﺩ. ﺍﻟﺒﺘﻪ ﻫﻨﺮﻣﻨﺪ ﺁﻥ ﺯﻣﺎﻥ ﺍﺯ ﺍﺻﻮﻝ ﺗﺮﮐﻴﺐ ﺍﺟﺰﺍﺀ ﻭ ﻋﻨﺎﺻﺮ ﻭ ﺑﻪ ﮐﺎﺭﮔﻴﺮﯼ ﻣﻨﺎﺳﺐ ﺭﻧﮓ ﺁﮔﺎﻩ ﺑﻮﺩ. ﺍﻣﺎ ﺑﺎ ﭘﻴﺪﺍﻳﺶ ﻋﮑﺎﺳﯽ، ﻧﻘﺎﺷﯽ ﻭ ﺗﺼﻮﻳﺮﺳﺎﺯﯼ ﺑﻪ ﺗﺪﺭﻳﺞ ﺷﮑﻠﯽ ﺍﻧﺘﺰﺍﻋﯽ ﺗﺮ ﺑﻪ ﺧﻮﺩ ﮔﺮﻓﺖ. ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﮐﻪ ﺑﻪ ﺟﺎﯼ ﺍﻧﺘﻘﺎﻝ ﻳﮏ ﺍﺣﺴﺎﺱ ﻳﺎ ﭘﻴﺎﻡ ﺍﺯ ﻃﺮﻳﻖ ﺑﺎﺯﻧﻤﺎﻳﯽ ﻭﺍﻗﻌﻴﺖ، ﺳﻌﯽ ﺷﺪ ﺗﺎ ﺍﺯ ﻋﻨﺎﺻﺮ ﻭ ﺍﺟﺰﺍﻳﯽ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﻫﻤﺎﻥ ﺍﺣﺴﺎﺱ ﺭﺍ ﺩﺭ ﻣﺨﺎﻃﺐ ﺍﻳﺠﺎﺩ ﮐﻨﻨﺪ ﺍﺳﺘﻔﺎﺩﻩ ﺷﻮﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺧﻄﻮﻃﯽ ﺁﺑﯽ- ﻓﻴﺮﻭﺯﻩ ﺍﯼ ﺑﺎ ﺍﻧﺤﻨﺎﻳﯽ ﺁﺭﺍﻡ ﻭ ﺍﻓﻘﯽ ﮐﻪ ﺑﻪ ﻣﻮﺍﺯﺍﺕ ﻳﮑﺪﻳﮕﺮ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﻧﺪ ﻣﯽ ﺗﻮﺍﻧﺪ ﻫﻤﺎﻥ ﺍﺣﺴﺎﺱ ﺁﺭﺍﻣﺶ ﻭ ﻣﻌﻨﻮﻳﺖ ﺭﺍ ﺑﻪ ﻣﺨﺎﻃﺐ ﻣﻨﺘﻘﻞ ﮐﻨﺪ. ﻳﺎ ﺣﺘﯽ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺑﻴﺸﺘﺮ ﺍﻫﻤﻴﺖ ﺩﺍﺩﻩ ﺷﻮﺩ ﮐﻪ ﺩﺭ ﻧﻤﺎﻳﺶ ﻭﺍﻗﻌﻴﺖ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺧﻞ، ﺗﺼﺮﻑ ﻭ ﺍﻏﺮﺍﻕ ﺻﻮﺭﺕ ﺑﮕﻴﺮﺩ ﺗﺎ ﺗﺎﺛﻴﺮ ﭘﻴﺎﻡ ﺗﺼﻮﻳﺮ ﺑﻴﺸﺘﺮ ﺷﻮﺩ. ﺍﺯ ﺁﻥ ﺯﻣﺎﻥ ﺑﻮﺩ ﮐﻪ ﻣﻄﺎﻟﻌﻪ ﺑﺮ ﺭﻭﯼ ﻗﻮﺍﻋﺪ ﻭ ﺍﺻﻮﻝ ﺯﺑﺎﻥ ﺗﺼﻮﻳﺮ ﺭﻭﺍﺝ ﭘﻴﺪﺍ ﮐﺮﺩ.

ﻣﻄﻠﺐ ﻣﻬﻢ ﻗﺎﺑﻞ ﺫﮐﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺁﺛﺎﺭ ﻫﻨﺮﯼ ﻣﻌﺎﺻﺮ ﺑﺮ ﭘﺎﻳﻪ ﻫﻤﻴﻦ ﻗﻮﺍﻋﺪ ﻭ ﺍﺻﻮﻝ ﺧﻠﻖ ﺷﺪﻩ ﺍﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺮﺍﯼ ﺩﺭﮎ ﺍﻳﻦ ﺁﺛﺎﺭ ﻧﻤﯽ ﺗﻮﺍﻥ ﺗﻨﻬﺎ ﺑﻪ ﻧﻴﺮﻭﯼ ﮐﺸﻒ ﻭ ﺷﻬﻮﺩ ﻣﺘﮑﯽ ﺑﻮﺩ، ﺑﻠﮑﻪ ﻻﺯﻡ ﺍﺳﺖ ﺗﺎ ﺑﺎ ﺷﻨﺎﺧﺖ ﻣﻘﻮﻻﺗﯽ ﻣﺎﻧﻨﺪ ﺗﺮﮐﻴﺐ ﺑﻨﺪﯼ، ﻋﻨﺎﺻﺮ ﭘﺎﻳﻪ ﺍﯼ، ﺷﻴﻮﻩ ﻫﺎﯼ ﺍﻧﺘﻘﺎﻝ ﭘﻴﺎﻡ ﻭ ﺑﻌﻀﯽ ﺍﺯ ﻓﻨﻮﻥ ﻭ ﺗﮑﻨﻴﮏ ﻫﺎ ﺑﺎ ﺍﻳﻦ ﺁﺛﺎﺭ ﺭﻭﺑﺮﻭ ﺷﺪ ﻭ ﺑﺎ ﺁﻧﻬﺎ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩ.

 

ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﺩﺍﺷﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﺭﺍﻩ ﺣﻠﯽ ﺳﺎﺩﻩ ﻭ ﺟﺎﻣﻊ ﺑﺮﺍﯼ ﻣﺴﻠﻂ ﺷﺪﻥ ﺑﻪ ﺯﺑﺎﻥ ﺗﺼﻮﻳﺮ ﺍﺭﺍﺋﻪ ﻧﻤﯽ ﺷﻮﺩ، ﺑﻠﮑﻪ ﻃﯽ ﻣﺒﺎﺣﺚ ﻣﺨﺘﻠﻒ ﺑﻪ ﻣﺨﺎﻃﺐ ﮐﻤﮏ ﻣﯽ ﺷﻮﺩ ﺗﺎ ﺑﺎ ﻣﻘﻮﻻﺕ ﺫﮐﺮ ﺷﺪﻩ ﺁﺷﻨﺎ ﺷﻮﺩ ﻭ ﭘﺲ ﺍﺯ ﺁﻥ ﺑﺎ ﺗﻤﺮﻳﻦ ﻭ ﮐﺎﺭﺑﺮﺩ ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﺩﺭ ﺯﻧﺪﮔﯽ ﺧﻮﺩ، ﺗﺴﻠﻂ ﺧﻮﺩ ﺭﺍ ﺑﺮ ﺍﻳﻦ ﺯﺑﺎﻥ ﺍﻓﺰﺍﻳﺶ ﺩﻫﺪ.

ﺗﻤﺮﻳﻦ ۱: ﺍﺯ ﻣﻴﺎﻥ ﻣﺠﻼﺕ، ﺭﻭﺯﻧﺎﻣﻪ ﻫﺎ ﻭ ﻳﺎ ﮐﺘﺎﺏ ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻳﺪ ﻳﮏ ﻳﺎ ﺩﻭ ﺗﺼﻮﻳﺮ ﺭﺍ ﮐﻪ ﺑﻪ ﻧﻈﺮ ﺷﻤﺎ ﺍﺯ ﻧﻈﺮ ﺑﺼﺮﯼ ﺩﺍﺭﺍﯼ ﺍﺭﺯﺵ ﻫﺴﺘﻨﺪ، ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﺑﻪ ﻧﻈﺮ ﺷﻤﺎ ﭼﻪ ﻋﻮﺍﻣﻠﯽ ﺩﺭ ﺍﻳﻦ ﺗﺼﺎﻭﻳﺮ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﮐﻪ ﺑﺎﻋﺚ ﺷﺪﻩ ﺷﻤﺎ ﺁﻧﻬﺎ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﻣﻮﺍﺭﺩ ﺭﺍ ﻳﺎﺩﺩﺍﺷﺖ ﮐﻨﻴﺪ ﻭ ﭘﺲ ﺍﺯ ﺍﺗﻤﺎﻡ ﺍﻳﻦ ﺩﺭﺳﻨﺎﻣﻪ ﺁﻧﻬﺎ ﺭﺍ ﻣﺮﻭﺭ ﮐﻨﻴﺪ.

ﺗﻤﺮﻳﻦ ۲: ﺑﺎ ﻣﺮﻭﺭ ﻣﻘﺪﻣﻪ ﺍﻳﻦ ﺩﺭﺱ، ﻟﻴﺴﺘﯽ ﺗﻬﻴﻪ ﮐﻨﻴﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﺩﻩ ﻣﻮﺭﺩ ﺍﺯ ﮐﺎﺭﺑﺮﺩ ﺗﺼﻮﻳﺮ ﻭ ﻧﺸﺎﻧﻪ ﻫﺎﯼ ﺑﺼﺮﯼ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﻭ ﺍﺗﻔﺎﻗﺎﺕ ﺩﻭﺭ ﻭ ﺍﻃﺮﺍﻑ ﺧﻮﺩ ﮔﻨﺠﺎﻧﺪﻩ ﺷﺪﻩ ﺑﺎﺷﺪ. ﺑﻪ ﻧﻈﺮ ﺷﻤﺎ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﭼﻪ ﭘﻴﺎﻣﯽ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﻣﯽ ﺭﺳﺎﻧﺪ.
