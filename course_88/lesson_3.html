ﺩﺭﺱ ﭘﻨﺠﻢ- ﺗﻨﻈﻴﻤﺎﺕ ﻧﻤﺎﻳﻪ (ﭘﺮﻭﻓﺎﻳﻞ) ﮔﻮﮔﻞ+- ﺑﺨﺶ ﺳﻮﻡ

ﭘﻴﺸﮕﻔﺘﺎﺭ
ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﺗﻨﻈﻴﻤﺎﺕ ﻧﻤﺎﻳﻪ ﮔﻮﮔﻞ+ ﺭﺍ ﺍﺩﺍﻣﻪ ﺩﺍﺩﻳﻢ ﻭ ﺩﺭ ﻣﻮﺭﺩ ﺗﻨﻈﻴﻤﺎﺕ ﻋﮑﺲ ﻧﻤﺎﻳﻪ، ﻋﮑﺲ ﺟﻠﺪ ﻧﻤﺎﻳﻪ، ﺑﺨﺶ ﺍﻭﻝ ﺳﺮﺑﺮﮒ ﺩﺭﺑﺎﺭﻩ (ﻗﺴﻤﺖ ﺩﺍﺳﺘﺎﻥ ﻭ ﺍﻓﺮﺍﺩ) ﻭ ﺣﻠﻘﻪ‌ﻫﺎ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻳﻢ.

ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﺎﻗﯽ ﺑﺨﺶ‌ﻫﺎﯼ ﺳﺮﺑﺮﮒ «ﺩﺭﺑﺎﺭﻩ» ﮐﻪ ﺷﺎﻣﻞ ﻣﺤﻞ ﺗﺤﺼﻴﻞ، ﮐﺎﺭ، ﺍﻃﻼﻋﺎﺕ ﺍﺳﺎﺳﯽ، ﻣﮑﺎﻥ، ﭘﻴﻮﻧﺪ ﻭ ﺍﻃﻼﻋﺎﺕ ﺗﻤﺎﺱ ﻣﯽ‌ﺷﻮﺩ ﺭﺍ ﺗﻮﺿﻴﺢ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺩ.

 

ﺳﺮﺑﺮﮒ ﺩﺭﺑﺎﺭﻩ

ﻣﺤﻞ ﺗﺤﺼﻴﻞ
ﺩﺭ ﻋﮑﺲ ﺯﻳﺮ ﮐﺎﺩﺭ ﻣﺤﻞ ﺗﺤﺼﻴﻞ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ. ﺩﺭ ﭘﺎﻳﻴﻦ ﮐﺎﺩﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻭﻳﺮﺍﻳﺶ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﺻﻔﺤﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻭﻳﺮﺍﻳﺶ ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﻣﺤﻞ ﺗﺤﺼﻴﻞ ﺑﺎﺯ ﺷﻮﺩ.

ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ، ﮐﺎﺩﺭ ﻣﺤﻞ ﺗﺤﺼﻴﻞ

ﺩﺭ ﺍﻳﻦ ﮐﺎﺩﺭ ﻣﺤﻞ ﺗﺤﺼﻴﻞ ﺧﻮﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ. ﻭ ﺍﻃﻼﻋﺎﺗﯽ ﺩﺭ ﻣﻮﺭﺩ ﺗﺤﺼﻴﻼﺕ ﺧﻮﺩ ﺑﻨﻮﻳﺴﻴﺪ ﺗﺎ ﺩﻳﮕﺮﺍﻥ ﺑﺎ ﺗﺤﺼﻴﻼﺕ ﺷﻤﺎ ﺁﺷﻨﺎ ﺷﻮﻧﺪ ﺗﺎ ﻫﻢ ﺭﺷﺘﻪ‌ﺍﻳﯽ ﻫﺎ ﻭ ﻋﻼﻗﻪ‌ﻣﻨﺪﺍﻥ ﺑﻪ ﺭﺷﺘﻪ ﺧﻮﺩ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﻴﺪ.

ﻗﺴﻤﺖ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﻣﺤﻞ ﺗﺤﺼﻴﻞ

ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﻧﺎﻡ ﻣﺤﻞ ﺗﺤﺼﻴﻞ، ﺭﺷﺘﻪ ﺗﺤﺼﻴﻠﯽ ﻭ ﺷﺮﻭﻉ ﻭ ﭘﺎﻳﺎﻥ ﺗﺤﺼﻴﻞ ﺭﺍ ﺩﺭ ﮐﺎﺩﺭ ﺍﻭﻝ ﻭﺍﺭﺩ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﺗﻴﮏ ﻣﺮﺑﻊ ﮐﻨﺎﺭ ﻓﻌﻠﯽ ﺭﺍ ﻣﯽ‌ﺯﻧﻴﺪ. ﻭ ﺩﺭ ﮐﺎﺩﺭ ﭘﺎﻳﻴﻨﯽ ﺗﺤﺼﻴﻼﺕ ﻗﺒﻠﯽ، ﺟﺰﻳﻴﺎﺕ ﺗﺤﺼﻴﻼﺕ ﻓﻌﻠﯽ ﻭ ﻏﻴﺮﻩ ﺭﺍ ﻣﯽ‌ﻧﻮﻳﺴﻴﺪ. ﭘﺲ ﺍﺯ ﭘﺎﻳﺎﻥ ﮐﺎﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻋﻤﻮﻣﯽ ﮐﻠﻴﮏ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﻴﺪ ﭼﻪ ﮔﺮﻭﻩ‌ﻫﺎﻳﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺁﻥ ﺭﺍ ﺑﺒﻴﻨﻨﺪ. ﭘﺲ ﺍﺯ ﺍﻧﺠﺎﻡ ﺗﻐﻴﻴﺮﺍﺕ ﺑﺎ ﮐﻠﻴﮏ ﺑﺮ ﺭﻭﯼ ﺫﺧﻴﺮﻩ ﺁﻥ ﺭﺍ ﺛﺒﺖ ﮐﻨﻴﺪ.

 

ﮐﺎﺭ
ﺩﺭ ﻋﮑﺲ ﺯﻳﺮ ﮐﺎﺩﺭ ﮐﺎﺭ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ. ﺩﺭ ﭘﺎﻳﻴﻦ ﮐﺎﺩﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻭﻳﺮﺍﻳﺶ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﺻﻔﺤﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻭﻳﺮﺍﻳﺶ ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﮐﺎﺭ ﺑﺎﺯ ﺷﻮﺩ.

ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ، ﮐﺎﺩﺭ ﮐﺎﺭ

ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﺑﺨﺶ ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺷﻐﻠﺘﺎﻥ ﺭﺍ ﻭﺍﺭﺩ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﻫﻢ ﺷﻤﺎ  ﻫﻤﮑﺎﺭﺍﻥ ﺧﻮﺩ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﻴﺪ ﻭ ﻫﻢ، ﻫﻤﮑﺎﺭﺍﻥ ﺷﻤﺎ ﻭ ﻋﻼﻗﻪ‌ﻣﻨﺪﺍﻥ ﺑﻪ ﺷﻐﻞ ﺷﻤﺎ، ﺷﻤﺎ ﺭﺍ ﭘﻴﺪﺍ ﻭ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻨﺪ.

ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺷﻐﻞ ﺧﻮﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ

ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﺩﺭ ﻣﻮﺭﺩ ﺷﻐﻞ ﺧﻮﺩ ﻣﯽ‌ﻧﻮﻳﺴﻴﺪ: ﮐﻪ ﺷﻐﻞ ﺷﻤﺎ ﭼﻴﺴﺖ؟ ﺩﺭ ﮐﺠﺎ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻴﺪ؟ ﺩﺭ ﮐﻨﺎﺭ ﺷﻐﻠﺘﺎﻥ ﭼﻪ ﻣﻬﺎﺭﺕ‌ﻫﺎﻳﯽ ﺩﺍﺭﻳﺪ؟ ﺩﺭ ﻗﺴﻤﺖ ﮐﺎﺭﻓﺮﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻨﻮﻳﺴﻴﺪ ﺯﻳﺮ ﻧﻈﺮ ﭼﻪ ﮐﺴﯽ ﻳﺎ ﭼﻪ ﺍﺭﮔﺎﻧﯽ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻴﺪ. ﮐﯽ ﻣﺸﻐﻮﻝ ﺑﻪ ﮐﺎﺭ ﺷﺪﻳﺪ؟ ﺑﺎ ﺯﺩﻥ ﺗﻴﮏ ﻣﺮﺑﻊ ﮐﻨﺎﺭ ﻓﻌﻠﯽ ﺗﺎﺭﻳﺦ ﺣﺎﻝ ﺣﺎﺿﺮ ﺭﺍ ﺛﺒﺖ ﮐﻨﻴﺪ. ﺩﺭ ﻗﺴﻤﺖ ﺷﺮﺡ ﺷﻐﻞ ﻫﺮ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺩﺭ ﻣﻮﺭﺩ ﺷﻐﻠﺘﺎﻥ ﺭﺍ ﻭﺍﺭﺩ  ﮐﻨﻴﺪ. ﭘﺲ ﺍﺯ ﭘﺎﻳﺎﻥ ﮐﺎﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻋﻤﻮﻣﯽ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﻭ ﻣﺸﺨﺺ ﮐﻨﻴﺪ ﭼﻪ ﮔﺮﻭﻩ‌ﻫﺎﻳﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺁﻥ ﺭﺍ ﺑﺒﻴﻨﻨﺪ، ﭘﺲ ﺍﺯ ﺍﻧﺠﺎﻡ ﺗﻐﻴﻴﺮﺍﺕ ﺑﺎ ﮐﻠﻴﮏ ﺑﺮ ﺭﻭﯼ ﺫﺧﻴﺮﻩ ﺁﻥ ﺭﺍ ﺛﺒﺖ ﮐﻨﻴﺪ.

 

ﺍﻃﻼﻋﺎﺕ ﺍﺳﺎﺳﯽ
ﺩﺭ ﻋﮑﺲ ﺯﻳﺮ ﮐﺎﺩﺭ ﺍﻃﻼﻋﺎﺕ ﺍﺳﺎﺳﯽ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ. ﺩﺭ ﭘﺎﻳﻴﻦ ﮐﺎﺩﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻭﻳﺮﺍﻳﺶ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﺻﻔﺤﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻭﻳﺮﺍﻳﺶ ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻃﻼﻋﺎﺕ ﺍﺳﺎﺳﯽ ﺑﺎﺯ ﺷﻮﺩ.

ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ، ﮐﺎﺩﺭ ﺍﻃﻼﻋﺎﺕ ﺍﺻﻠﯽ

ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﻭﺍﺭﺩ ﻣﯽ‌ﮐﻨﻴﺪ ﻳﮑﺴﺮﯼ ﺍﻃﻼﻋﺎﺕ ﺷﺨﺼﯽ ﺍﺳﺖ. ﺷﻤﺎ ﺟﻨﺴﻴﺖ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﻭﺍﺭﺩ ﻣﯽ‌ﮐﻨﻴﺪ. ﺩﺭ ﺟﺴﺘﺠﻮﯼ ﭼﻪ ﮐﺴﯽ ﻫﺴﺘﻴﺪ؟ ﻳﻌﻨﯽ ﻫﺪﻓﺘﺎﻥ ﺩﺭ ﻣﻮﺭﺩ ﺭﺍﺑﻄﻪ ﻭ ﺍﺯ ﺣﻀﻮﺭ ﺩﺭ ﮔﻮﮔﻞ+ ﭼﻴﺴﺖ؟ ﺩﻧﺒﺎﻝ ﺩﻭﺳﺖ ﻫﺴﺘﻴﺪ؟ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺑﺎ ﮐﺴﯽ ﻗﺮﺍﺭ ﺑﺰﺍﺭﻳﺪ؟ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﻭﺍﺭﺩ ﺭﺍﺑﻄﻪ ﺑﺸﻮﻳﺪ؟ ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﺭﺍﺑﻄﻪ‌ﺍﻳﯽ ﺩﺍﺭﻳﺪ ﻳﺎ ﺧﻴﺮ؟ ﻭ ﻏﻴﺮﻩ.

ﻣﺤﻞ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﺍﻃﻼﻋﺎﺕ ﺍﺳﺎﺳﯽ

ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺭﻭﺍﺑﻄﯽ ﮐﻪ ﺩﺍﺭﻳﺪ ﻭ ﺭﻭﺍﺑﻄﯽ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺭﺍ ﻭﺍﺭﺩ ﻣﯽ‌ﮐﻨﻴﺪ. ﺗﺎﺭﻳﺦ ﺗﻮﻟﺪﺗﺎﻥ ﺭﺍ ﻗﺒﻼ ﻭﺍﺭﺩ ﮐﺮﺩﻩ ﺍﻳﺪ ﺩﺭ ﺻﻮﺭﺕ ﺗﻤﺎﻳﻞ ﺁﻥ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ. ﺟﻨﺴﻴﺖ ﻫﻢ ﺩﺭ ﻣﺮﺍﺣﻞ ﻗﺒﻠﯽ ﺗﮑﻤﻴﻞ ﻧﻤﺎﻳﻪ ﻭﺍﺭﺩ ﺷﺪﻩ ﮐﻪ ﺩﺭ ﺻﻮﺭﺕ ﺗﻤﺎﻳﻞ ﺁﻥ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ. ﺍﮔﺮ ﺩﺭ ﺭﺍﺑﻄﻪ‌ ﺍﻳﯽ ﻫﺴﺘﻴﺪ ﺁﻥ ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﻳﮑﯽ ﺍﺯ ﮔﺰﻳﻨﻪ‌ﻫﺎ ﺑﻴﺎﻥ ﮐﻨﻴﺪ. ﺩﺭ ﺻﻮﺭﺕ ﺗﻤﺎﻳﻞ ﻧﺪﺍﺷﺘﻦ ﺑﻪ ﮔﻔﺘﻦ ﻭﺿﻌﻴﺖ ﺭﺍﺑﻄﻪ ﺁﻥ ﺭﺍ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ «ﻧﻤﯽ‌ﺧﻮﺍﻫﻢ ﺑﮕﻮﻳﻢ» ﻣﺸﺨﺺ ﮐﻨﻴﺪ. ﺩﺭ ﮐﺎﺩﺭ ﺁﺧﺮ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﮔﺮ ﻧﺎﻡ ﺩﻭﻡ ﺩﺍﺭﻳﺪ ﻭﺍﺭﺩ ﮐﻨﻴﺪ ﻳﺎ ﻧﺎﻡ ﺩﻭﻣﯽ ﺑﺮﺍﯼ ﺧﻮﺩ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﭘﺲ ﺍﺯ ﭘﺎﻳﺎﻥ ﮐﺎﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻋﻤﻮﻣﯽ ﮐﻠﻴﮏ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﻴﺪ ﭼﻪ ﮔﺮﻭﻩ‌ﻫﺎﯼ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺁﻥ ﺭﺍ ﺑﺒﻴﻨﻨﺪ. ﭘﺲ ﺍﺯ ﺍﻧﺠﺎﻡ ﺗﻐﻴﻴﺮﺍﺕ ﺑﺎ ﮐﻠﻴﮏ ﺑﺮ ﺭﻭﯼ ﺫﺧﻴﺮﻩ ﺁﻥ ﺭﺍ ﺛﺒﺖ ﮐﻨﻴﺪ.

 

ﻣﮑﺎﻥ‌ﻫﺎ
ﺩﺭ ﻋﮑﺲ ﺯﻳﺮ ﮐﺎﺩﺭ ﻣﮑﺎﻥ‌ﻫﺎ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻣﯽ ﮐﻨﻴﺪ. ﺩﺭ ﭘﺎﻳﻴﻦ ﮐﺎﺩﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻭﻳﺮﺍﻳﺶ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﺻﻔﺤﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻭﻳﺮﺍﻳﺶ ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﻣﮑﺎﻥ‌ﻫﺎ ﺑﺎﺯ ﺷﻮﺩ.

ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ، ﮐﺎﺩﺭ ﻣﮑﺎﻥ‌ﻫﺎ
