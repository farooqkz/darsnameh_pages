ﺑﺎ ﺭﻭﻳﺪﺍﺩ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺳﺎﺩﮔﯽ ﻳﮏ ﮔﺮﺩﻫﻤﺎﻳﯽ ﻳﺎ ﻳﮏ ﻭﺍﻗﻊ ﺭﺍ ﺳﺎﺯﻣﺎﻧﺪﻫﯽ ﮐﻨﻴﺪ ﺧﻮﺍﻩ ﺍﻳﻦ ﮔﺮﺩﻫﻤﺎﻳﯽ ﺑﻪ ‌ﺻﻮﺭﺕ ﺣﻀﻮﺭﯼ ﺑﺎﺷﺪ ﻳﺎ ﺁﻧﻼﻳﻦ. ﻫﻤﻪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺩﺭ ﺭﻭﻳﺪﺍﺩ ﻋﻀﻮ ﻫﺴﺘﻨﺪ ﻣﯽ‌‌ﺗﻮﺍﻧﻨﺪ ﻧﻈﺮﺍﺕ ﺭﺍ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻧﺪ ﻋﮑﺲ‌‌ﻫﺎ ﺭﺍ ﺑﺼﻮﺭﺕ ﺧﻮﺩﮐﺎﺭ ﻫﻨﮕﺎﻡ ﺷﺮﮐﺖ ﺩﺭ ﺭﻭﻳﺪﺍﺩ ﺁﭘﻠﻮﺩ ﮐﻨﻨﺪ. ﺍﻓﺮﺍﺩ ﺭﺍ ﺩﻭﺭ ﻫﻢ ﺟﻤﻊ ﮐﻨﻨﺪ. ﺧﺎﻃﺮﺍﺕ ﻭ ﻣﻮﻗﻌﻴﺖ‌ﻫﺎﯼ ﻣﻬﻢ ﻳﺎ ﮔﺮﺩﻫﻤﺎﻳﯽ‌‌ﻫﺎﯼ ﺑﺪﻭﻥ ﺑﺮﻧﺎﻣﻪ ﻗﺒﻠﯽ ﺭﺍ ﺑﺎ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﮔﻮﮔﻞ+ ﺟﻤﻊ‌ﺁﻭﺭﯼ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻧﺪ. ﻣﺎﻧﻨﺪ ﺭﻭﻳﺪﺍﺩ ﭼﻬﺎﺭﺷﻨﺒﻪ ﺳﻮﺭﯼ ﻭ ﻏﻴﺮﻩ. ﺷﻤﺎ ﺩﻭﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻳﮏ ﻣﮑﺎﻥ ﻣﺸﺨﺺ ﺩﻋﻮﺕ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺑﻪ ﻓﻀﺎﯼ ﻣﺠﺎﺯﯼ ﺩﻋﻮﺕ ﻣﯽ‌ﮐﻨﻴﺪ.

 

ﺍﻳﺠﺎﺩ ﺭﻭﻳﺪﺍﺩ ﺷﺨﺼﯽ ﺧﻮﺩ
ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﻳﮏ ﺭﻭﻳﺪﺍﺩ ﺑﺮﺍﯼ ﺧﻮﺩ، ﺩﺭ ﺻﻔﺤﻪ «ﺭﻭﻳﺪﺍﺩﻫﺎ» ﮐﻪ ﺁﻥ ﺭﺍ ﺑﺎﺯ ﮐﺮﺩﻳﺪ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﺍﻳﺠﺎﺩ ﺭﻭﻳﺪﺍﺩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﺻﻔﺤﻪ ﺭﻭﻳﺪﺍﺩ ﻫﺎ، ﺍﻳﺠﺎﺩ ﺭﻭﻳﺪﺍﺩ ﺷﺨﺼﯽ

ﭘﺲ ﺍﺯ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻳﺠﺎﺩ ﺭﻭﻳﺪﺍﺩ» ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺳﺎﺧﺖ ﺭﻭﻳﺪﺍﺩ ﻣﯽ‌ﺷﻮﻳﺪ.

ﺻﻔﺤﻪ ﺳﺎﺧﺖ ﺭﻭﻳﺪﺍﺩ

ﺍﻭﻟﻴﻦ ﮐﺎﺭﯼ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻃﺮﺡ ﺯﻣﻴﻨﻪ‌‌ﺍﻳﯽ (ﺟﻠﺪ ﺭﻭﻳﺪﺍﺩ) ﺑﺮﺍﯼ ﺭﻭﻳﺪﺍﺩ ﺧﻮﺩ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ.  ﺗﺼﻮﻳﺮﯼ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌‌ﮐﻨﻴﺪ ﻫﻤﺮﺍﻩ ﺑﺎ ﺭﺍﻳﺎﻧﺎﻣﻪ‌ﻫﺎﯼ ﺩﻋﻮﺕ، ﭘﺴﺖ‌‌ﻫﺎﯼ ﻣﺮﺑﻮﻁ ﺑﻪ ﺭﻭﻳﺪﺍﺩ ﻭ ﺻﻔﺤﻪ ﺟﺰﺋﻴﺎﺕ ﺭﻭﻳﺪﺍﺩ‌ﺗﺎﻥ ﻧﻤﺎﻳﺶ ﺩﺍﺩﻩ ﻣﯽ‌‌ﺷﻮﺩ. ﺑﺮﺍﯼ ﻣﺸﺎﻫﺪﻩ ﮐﻠﻴﻪ ﮔﺰﻳﻨﻪ‌‌ﻫﺎﯼ ﻃﺮﺡ ﺯﻣﻴﻨﻪ ﺑﺎ ﻧﻤﺎﺩﻫﺎﯼ < ﻭ > ﺩﺭ ﻃﺮﺡ‌‌ﻫﺎﯼ ﺯﻣﻴﻨﻪ ﮔﺮﺩﺵ ﮐﻨﻴﺪ ﻳﺎ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺗﻐﻴﻴﺮ ﻃﺮﺡ ﺯﻣﻴﻨﻪ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﭘﺲ ﺍﺯ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺗﻐﻴﻴﺮ ﻃﺮﺡ ﺯﻣﻴﻨﻪ» ﺻﻔﺤﻪ‌ﺍﯼ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺮ ﺍﺳﺎﺱ ﺭﻭﻳﺪﺍﺩﯼ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﺴﺎﺯﻳﺪ ﻃﺮﺣﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ.

ﺗﻐﻴﻴﺮ ﻃﺮﺡ ﺯﻣﻴﻨﻪ ﺭﻭﻳﺪﺍﺩ ﺧﻮﺩ

ﺩﺭ ﺯﻳﺮ ﺟﻠﺪ ﺭﻭﻳﺪﺍﺩ ﮐﺎﺩﺭﯼ ﮐﻪ ﺩﺭ ﺁﻥ «ﻋﻨﻮﺍﻥ ﺭﻭﻳﺪﺍﺩ» ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺍﺳﺖ ﻗﺮﺍﺭ ﺩﺍﺭﺩ. ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﺁﻥ ﺑﺮﺍﯼ ﺭﻭﻳﺪﺍﺩ‌ﺗﺎﻥ، ﻣﺘﻨﺎﺳﺐ ﺑﺎ ﻣﻮﺿﻮﻉ ﺭﻭﻳﺪﺍﺩ ﻧﺎﻣﯽ ﻭﺍﺭﺩ ﮐﻨﻴﺪ. ﻣﺎﻧﻨﺪ ﭼﻬﺎﺭﺷﻨﺒﻪ ﺳﻮﺭﯼ.

ﺩﺭ ﺭﻭ ﺑﻪ ﺭﻭﯼ ﮐﺎﺩﺭ ﺍﻧﺘﺨﺎﺏ ﮔﺰﻳﻨﻪ «ﮔﺰﻳﻨﻪ‌ﻫﺎﯼ ﺭﻭﻳﺪﺍﺩ» ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻳﮏ ﻣﻨﻮﯼ ﮐﺸﻮﻳﯽ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﮔﺰﻳﻨﻪ‌ﻫﺎﯼ ﺭﻭﻳﺪﺍﺩ

ﺍﻭﻟﻴﻦ ﮔﺰﻳﻨﻪ «ﻣﻬﻤﺎﻧﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮ ﺭﺍ ﺩﻋﻮﺕ ﮐﻨﻨﺪ» ﺍﺳﺖ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻭ ﻇﺎﻫﺮ ﺷﺪﻥ ﺗﻴﮏ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﮐﻪ ﻣﻬﻤﺎﻧﺎﻧﯽ ﮐﻪ ﺷﻤﺎ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺩﻋﻮﺕ ﮐﺮﺩﻳﺪ ﺑﺘﻮﺍﻧﻨﺪ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮ ﺭﺍ ﺩﻋﻮﺕ ﮐﻨﻨﺪ ﻳﺎ ﺧﻴﺮ. ﻣﺜﺎﻝ ﺷﻤﺎ ﺩﻭﺳﺖ‌ﺗﺎﻥ ﻋﻠﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﻳﮑﯽ ﺍﺯ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎﺳﺖ ﺩﻋﻮﺕ ﻣﯽ‌ﮐﻨﻴﺪ، ﺣﺎﻝ ﻋﻠﯽ ﺑﺘﻮﺍﻧﺪ ﺩﻭﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺩﻋﻮﺕ ﮐﻨﺪ ﻳﺎ ﺧﻴﺮ.

ﺩﻭﻣﻴﻦ ﮔﺰﻳﻨﻪ «ﻣﻬﻤﺎﻧﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻋﮑﺲ ﺍﺿﺎﻓﻪ ﮐﻨﻨﺪ» ﺍﺳﺖ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻭ ﻇﺎﻫﺮ ﺷﺪﻥ ﺗﻴﮏ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﮐﻪ ﻣﻬﻤﺎﻧﺎﻧﯽ ﮐﻪ ﺷﻤﺎ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺩﻋﻮﺕ ﮐﺮﺩﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻋﮑﺴﯽ ﺍﺿﺎﻓﻪ ﮐﻨﻨﺪ ﻳﺎ ﺧﻴﺮ.

ﺳﻮﻣﻴﻦ ﮔﺰﻳﻨﻪ «ﭘﻨﻬﺎﻥ ﮐﺮﺩﻥ ﻟﻴﺴﺖ ﻣﻬﻤﺎﻧﺎﻥ» ﺍﺳﺖ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻭ ﻇﺎﻫﺮ ﺷﺪﻥ ﺗﻴﮏ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﮐﻪ ﺍﻋﻀﺎﯼ ﺣﺎﺿﺮ ﺩﺭ ﺭﻭﻳﺪﺍﺩ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺯ ﺣﻀﻮﺭ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮ ﺑﺎ ﺧﺒﺮ ﺑﺎﺷﻨﺪ ﻳﺎ ﺧﻴﺮ.

ﭼﻬﺎﺭﻣﻴﻦ ﮔﺰﻳﻨﻪ «ﭘﻴﺸﺮﻓﺘﻪ» ﺍﺳﺖ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻳﮏ ﻣﻨﻮﯼ ﮐﺸﻮﻳﯽ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﻧﻤﺎﻳﺶ ﮔﺰﻳﻨﻪ‌ﻫﺎﯼ ﺑﻴﺸﺘﺮ» ﻳﮑﺴﺮﯼ ﮐﺎﺩﺭ‌ﻫﺎ ﺩﺭ ﺻﻔﺤﻪ ﺍﻳﺤﺎﺩ ﺭﻭﻳﺪﺍﺩ ﺷﻤﺎ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﺍﻃﻼﻋﺎﺕ ﺩﺭ ﺁﻥ‌ﻫﺎ ﺍﺧﺘﻴﺎﺭﯼ ﺍﺳﺖ. ﻣﺎﻧﻨﺪ ﺁﺩﺭﺱ ﺳﺎﻳﺖ، ﺁﺩﺭﺱ ﻣﺤﻞ ﻓﺮﻭﺵ ﺑﻠﻴﻂ (ﺑﻪ ﻓﺮﺽ ﺍﻳﻨﮑﻪ ﺭﻭﻳﺪﺍﺩ ﺷﻤﺎ ﺩﻋﻮﺕ ﺑﻪ ﺩﻳﺪﻥ ﻳﮏ ﺗﺌﺎﺗﺮ ﺑﺎﺷﺪ) ﻭ ﻏﻴﺮﻩ.

ﺩﺭ ﺭﺩﻳﻒ ﺩﻭﻡ، ﺯﻳﺮ ﮐﺎﺩﺭ «ﺍﻧﺘﺨﺎﺏ ﻋﻨﻮﺍﻥ ﺭﻭﻳﺪﺍﺩ» ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﺎﺭﻳﺦ ﻭ ﺳﺎﻋﺖ ﺷﺮﻭﻉ ﻭ ﭘﺎﻳﺎﻥ ﺭﻭﻳﺪﺍﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ.

ﺩﺭ ﺭﺩﻳﻒ ﺳﻮﻡ، ﺯﻳﺮ ﮐﺎﺩﺭ «ﺍﻧﺘﺨﺎﺏ ﺗﺎﺭﻳﺦ ﺭﻭﻳﺪﺍﺩ» ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺤﻞ ﻳﺎ ﻣﮑﺎﻥ ﺭﻭﻳﺪﺍﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ. ﻣﺎﻧﻨﺪ ﺑﻠﻮﺍﺭ ﮐﺸﺎﻭﺭﺯ ﻣﺤﻠﻪ ﻗﺪﻳﻤﯽ. ﺍﻟﺒﺘﻪ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﻣﮑﺎﻥ ﺭﻭﻳﺪﺍﺩ ﺍﺧﺘﻴﺎﺭﯼ ﺍﺳﺖ.

ﺩﺭ ﺭﺩﻳﻒ ﭼﻬﺎﺭﻡ، ﺩﺭ ﺯﻳﺮ ﮐﺎﺩﺭ «ﺍﻧﺘﺨﺎﺏ ﻣﮑﺎﻥ ﺭﻭﻳﺪﺍﺩ» ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﻣﻮﺭﺩ ﺟﺰﻳﻴﺎﺕ ﺭﻭﻳﺪﺍﺩ ﺧﻮﺩ ﺑﻴﺸﺘﺮ ﺗﻮﺿﻴﺢ ﺩﻫﻴﺪ.

ﺩﺭ ﺭﺩﻳﻒ ﭘﻨﺠﻢ، ﺩﺭ ﺯﻳﺮ ﮐﺎﺩﺭ «ﺟﺰﻳﺌﺎﺕ» ﮐﺎﺩﺭ ﮔﻴﺮﻧﺪﻩ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻫﺮ ﮐﺴﯽ ﺭﺍ ﮐﻪ ﺗﻤﺎﻳﻞ ﺩﺍﺭﻳﺪ ﺩﺭ ﺭﻭﻳﺪﺍﺩ ﺷﻤﺎ ﺑﺎﺷﺪ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﻧﺤﻮﻩ ﺍﻧﺘﺨﺎﺏ ﻣﺎﻧﻨﺪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺍﺳﺖ ﮐﻪ ﻗﺒﻼ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻳﻢ.

ﭘﺲ ﺍﺯ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﺍﻃﻼﻋﺎﺕ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺩﻋﻮﺕ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﺷﻤﺎ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺭﻭﻳﺪﺍﺩ ﺧﻮﺩ ﺷﺪﻩ‌ﺍﻳﺪ.

ﺻﻔﺤﻪ ﺭﻭﻳﺪﺍﺩ ﺷﻤﺎ

ﺩﺭ ﮐﺎﺩﺭ ﮐﻨﺎﺭ ﻋﮑﺲ ﻧﻤﺎﻳﻪ‌ﺗﺎﻥ ﮐﻪ ﺩﺭ ﺁﻥ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺍﺳﺖ «ﻧﻈﺮﺗﺎﻥ ﺭﺍ ﺑﮕﻮﻳﻴﺪ» ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭘﺴﺘﯽ ﺭﺍ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ. ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺷﮑﻞ ﺩﻭﺭﺑﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻋﮑﺴﯽ ﺭﺍ ﺍﺯ ﺭﺍﻳﺎﻧﻪ ﺧﻮﺩ ﺑﺮﺍﯼ ﻧﻮﺷﺘﻪ ﺑﺎﺭﮔﺬﺍﺭﯼ ﮐﻨﻴﺪ. ﻧﺤﻮﻩ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻭ ﺁﭘﻠﻮﺩ ﺭﺍ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻠﯽ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ‌ﺍﻳﻢ.

ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﻭﻳﺮﺍﻳﺶ ﺭﻭﻳﺪﺍﺩ» ﮐﻪ ﺩﺭ ﺳﻤﺖ ﭼﭗ ﺻﻔﺤﻪ ﺑﺎﻻ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺩﺭ ﺍﺑﺘﺪﺍ ﺩﺭ ﻣﻮﺭﺩ ﺭﻭﻳﺪﺍﺩ ﺧﻮﺩ ﻭﺍﺭﺩ ﮐﺮﺩﻳﺪ ﻭﻳﺮﺍﻳﺶ ﮐﻨﻴﺪ.

ﮔﺰﻳﻨﻪ ﺑﻌﺪﯼ «1+» ﺍﺳﺖ، ﺷﻤﺎ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺻﻔﺤﻪ ﺭﻭﻳﺪﺍﺩ ﺧﻮﺩ ﺭﺍ +1 ﮐﻨﻴﺪ.

ﮔﺰﻳﻨﻪ ﮐﻨﺎﺭ ﺁﻥ «ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺭﻭﻳﺪﺍﺩ» ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﺭﻭﻳﺪﺍﺩ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺍﻓﺮﺍﺩ ﻣﻮﺭﺩ ﻧﻈﺮ ﺧﻮﺩ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ.

 

ﻋﻨﻮﺍﻥ ﺷﻤﺎ ﺷﺮﮐﺖ ﻣﯽ‌ﮐﻨﻴﺪ؟
ﺩﺭ ﺻﻔﺤﻪ ﺭﻭﻳﺪﺍﺩ ﺷﺨﺼﯽ ﺧﻮﺩ، ۲ ﮐﺎﺩﺭ ﺩﺭ ﺯﻳﺮ ﻋﻨﻮﺍﻥ «ﺷﻤﺎ ﺷﺮﮐﺖ ﻣﯽ‌ﮐﻨﻴﺪ» ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﺍﻭﻟﯽ ﺟﻮﺍﺏ ﺑﻪ ﺳﻮﺍﻝ ﻋﻨﻮﺍﻥ ﺍﺳﺖ، ﺩﺭ ﺍﻳﻦ ﺭﻭﻳﺪﺍﺩ ﺷﺮﮐﺖ ﻣﯽ‌ﮐﻨﻴﺪ ﻳﺎ ﺧﻴﺮ؟ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮐﺎﺩﺭ ﺍﻭﻝ 3 ﮔﺰﻳﻨﻪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ (ﺑﻠﻪ، ﺧﻴﺮ، ﺷﺎﻳﺪ) ﻫﺮ ﮐﺪﺍﻡ ﺭﺍ ﮐﻪ ﺗﻤﺎﻳﻞ ﺩﺍﺭﻳﺪ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻴﺪ.

ﺍﮔﺮ ﺷﺨﺼﯽ ﻫﻢ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﺭﻭﻳﺪﺍﺩﯼ ﺩﻋﻮﺕ ﮐﺮﺩ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺭﻓﺘﻦ ﺑﻪ ﺍﻳﻦ ﻗﺴﻤﺖ ﺗﻤﺎﻳﻞ ﺩﺍﺷﺘﻦ ﻳﺎ ﻧﺪﺍﺷﺘﻦ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺣﻀﻮﺭ ﺩﺭ ﺭﻭﻳﺪﺍﺩ ﺍﻋﻼﻡ ﮐﻨﻴﺪ.

ﮐﺎﺩﺭ ﺑﻌﺪﯼ ﻫﻢ ﺗﻌﺪﺍﺩ ﻣﻬﻤﺎﻥ‌ﻫﺎ ﺭﺍ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﺪ.

 

ﻋﻨﻮﺍﻥ ﺟﺰﺋﻴﺎﺕ
ﻋﻨﻮﺍﻥ «ﺟﺰﺋﻴﺎﺕ» ﺩﺭ ﺯﻳﺮ ﻋﻨﻮﺍﻥ «ﺷﻤﺎ ﺷﺮﮐﺖ ﻣﯽ‌ﮐﻨﻴﺪ» ﻗﺮﺍﺭ ﺩﺍﺭﺩ. ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﭘﻴﮑﺎﻥ ﺭﻭ ﺑﻪ ﺭﻭﯼ ﻋﻨﻮﺍﻥ ﻳﮏ ﻣﻨﻮﯼ ﮐﺸﻮﻳﯽ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ.

ﺗﻨﻈﻴﻢ ﺟﺰﺋﻴﺎﺕ ﺭﻭﻳﺪﺍﺩ ﺷﻤﺎ

ﺩﺭ ﻫﻤﻴﻦ ﺩﺭﺱ ﺩﺭ ﻣﻮﺭﺩ ﺧﻴﻠﯽ ﺍﺯ ﮔﺰﻳﻨﻪ‌ﻫﺎﯼ ﺍﻳﻦ ﻣﻨﻮﯼ ﮐﺸﻮﻳﯽ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ. ﺩﺭ ﻣﻮﺭﺩ ﻣﻮﺍﺭﺩﯼ ﮐﻪ ﺻﺤﺒﺖ ﻧﮑﺮﺩﻳﻢ ﺗﻮﺿﻴﺢ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺩ. ﺍﻭﻟﻴﻦ ﻣﻮﺭﺩ «ﻣﺪﻳﺮﻳﺖ ﻟﻴﺴﺖ ﻣﻬﻤﺎﻥ» ﺍﺳﺖ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﺻﻔﺤﻪ ﺟﺪﻳﺪﯼ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺁﻥ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﻣﻬﻤﺎﻥ‌ﻫﺎﻳﯽ ﮐﻪ ﺗﻤﺎﻳﻞ ﺩﺍﺭﻳﺪ ﺣﺬﻑ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺭﻭﯼ X ﻭﺍﻗﻊ ﺩﺭ ﮐﻨﺎﺭ ﻧﺎﻡ ﻓﺮﺩﯼ ﮐﻪ ﻣﯽ‌‌ﺧﻮﺍﻫﻴﺪ ﺣﺬﻑ ﻧﻤﺎﻳﻴﺪ ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﺩﻭﻣﻴﻦ ﻣﻮﺭﺩ «ﺗﮑﺮﺍﺭ ﺭﻭﻳﺪﺍﺩ» ﺍﺳﺖ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﺻﻔﺤﻪ ﺟﺪﻳﺪ ﻣﺎﻧﻨﺪ ﺻﻔﺤﻪ ﺍﻳﺠﺎﺩ ﺭﻭﻳﺪﺍﺩ ﺷﺨﺼﯽ ﺑﺎ ﻫﻤﺎﻥ ﮐﺎﺩﺭ‌ﻫﺎ ﻭ ﺟﺰﺋﻴﺎﺕ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺎﻧﻨﺪ ﻗﺒﻞ ﺍﻃﻼﻋﺎﺕ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ ﻭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺩﻋﻮﺕ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﻣﻮﺭﺩ ﺁﺧﺮ «ﺍﻳﻦ ﺭﻭﻳﺪﺍﺩ ﺣﺬﻑ ﺷﻮﺩ» ﺍﺳﺖ. ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺭﻭﻳﺪﺍﺩ ﺭﺍ ﺣﺬﻑ ﮐﻨﻴﺪ. ﺑﺎ ﺣﺬﻑ ﻳﮏ ﺭﻭﻳﺪﺍﺩ، ﺍﻳﻦ ﺭﻭﻳﺪﺍﺩ ﺍﺯ ﺻﻔﺤﻪ ﺭﻭﻳﺪﺍﺩ‌ﻫﺎﯼ ﻫﻤﻪ ﺍﻓﺮﺍﺩ ﺟﺮﻳﺎﻥ ﻭ ﺗﻘﻮﻳﻢ ﺁﻥ‌ﻫﺎ ﺣﺬﻑ ﻣﯽ‌‌ﺷﻮﺩ ﻭ ﻫﻴﭽﮑﺲ ﻧﻤﯽ‌‌ﺗﻮﺍﻧﺪ ﺭﻭﻳﺪﺍﺩ ﺭﺍ ﺑﺎﺯ ﮐﻨﺪ ﻳﺎ ﻧﻈﺮﺍﺕ، ﻋﮑﺲ‌ﻫﺎ ﻳﺎ ﺩﻳﮕﺮ ﻣﺤﺘﻮﺍ ﻫﺎﯼ ﺁﻥ ﺭﺍ ﺑﺒﻴﻨﺪ.

 

ﻋﻨﻮﺍﻥ ﻣﻬﻤﺎﻥ‌ﻫﺎ
ﺩﺭ ﮐﺎﺩﺭ ﺁﺧﺮ ﺻﻔﺤﻪ ﺭﻭﻳﺪﺍﺩ‌ﻫﺎ، ﺭﻭ ﺑﻪ ﺭﻭﯼ ﻋﻨﻮﺍﻥ ﻣﻬﻤﺎﻥ‌ﻫﺎ، ۲ ﮔﺰﻳﻨﻪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﻭﻟﯽ ﮔﺰﻳﻨﻪ «ﺍﺭﺳﺎﻝ ﭘﻴﺎﻡ» ﺍﺳﺖ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﺻﻔﺤﻪ ﺟﺪﻳﺪﯼ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﺍﺭﺳﺎﻝ ﭘﻴﺎﻡ ﺍﺯ ﺭﻭﻳﺪﺍﺩ ﺑﻪ ﻣﻬﻤﺎﻥ‌ﻫﺎ

ﻣﻬﻤﺎﻥ‌ﻫﺎﯼ ﺷﻤﺎ ﺍﻋﻼﻥ ﭘﻴﺎﻡ ﺷﻤﺎ ﺭﺍ ﺩﺭﻳﺎﻓﺖ ﺧﻮﺍﻫﻨﺪ ﮐﺮﺩ. ﺩﺭ ﮐﺎﺩﺭ ﺍﻭﻝ ﭘﻴﺎﻡ ﺧﻮﺩ ﺭﺍ ﺑﻨﻮﻳﺴﻴﺪ ﻭ ﺩﺭ ﮐﺎﺩﺭ ﮔﻴﺮﻧﺪﻩ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﮐﻪ ﭘﻴﺎﻡ ﺑﺮﺍﯼ ﭼﻪ ﮐﺴﺎﻧﯽ ﺍﺭﺳﺎﻝ ﺷﻮﺩ.

ﻓﺮﺽ ﺷﻤﺎ ﻳﺎ ﻣﻬﻤﺎﻥ ﺷﻤﺎ ﻋﮑﺴﯽ ﺑﻪ ﺭﻭﻳﺪﺍﺩ ﺷﺨﺼﯽ ﺷﻤﺎ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻩ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺩﺭ ﺻﻔﺤﻪ ﺭﻭﻳﺪﺍﺩ ﺷﻤﺎ ﮐﺎﺩﺭﯼ ﺑﺎ ﻧﺎﻡ «ﻋﮑﺲ‌ﻫﺎﯼ ﺭﻭﻳﺪﺍﺩ» ﺩﺭ ﺯﻳﺮ ﮐﺎﺩﺭ «ﺷﻤﺎ ﺷﺮﮐﺖ ﻣﯽ‌ﮐﻨﻴﺪ؟» ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ. ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﻣﺸﺎﻫﺪﻩ ﻋﮑﺲ‌ﻫﺎ» ﺻﻔﺤﻪ ﺟﺪﻳﺪﯼ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ.

 ﺩﺭ ﻣﻮﺭﺩ ﮔﺰﻳﻨﻪ‌ﻫﺎﯼ ﺩﻳﮕﺮﯼ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ.
