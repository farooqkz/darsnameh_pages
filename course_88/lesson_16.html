ﺩﺭﺱ ﻳﺎﺯﺩﻫﻢ- ﺻﻔﺤﻪ ﻋﮑﺲ ﻫﺎ

ﭘﻴﺸﮕﻔﺘﺎﺭ
ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺗﻨﻈﻴﻤﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﻋﮑﺲ‌ﻫﺎ ﻭ ﺁﻟﺒﻮﻡ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺑﻪ ﺷﻤﺎ ﮔﻔﺘﻪ ﺧﻮﺍﻫﺪ ﺷﺪ ﭼﮕﻮﻧﻪ ﺍﺯ ﺍﻓﮑﺖ‌ﻫﺎﯼ ﻣﺘﺤﺮﮎ ﮔﻮﮔﻞ ﺑﺮﺍﯼ ﻋﮑﺲ‌ﻫﺎﯼ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ.

 

ﻋﮑﺲ
ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺍﺻﻠﯽ‌ﺗﺎﻥ ﺩﺭ ﮔﻮﮔﻞ+ ﺷﻮﻳﺪ. ﺩﺭ ﺳﻤﺖ ﺭﺍﺳﺖ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﮔﻮﮔﻞ+ ﻣﻮﺷﻮﺍﺭﻩ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺻﻔﺤﻪ ﺍﺻﻠﯽ» ﻧﮕﻪ ﺩﺍﺭﻳﺪ. ﻳﮏ ﻣﻨﻮﯼ ﮐﺸﻮﻳﯽ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ، ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﭼﻬﺎﺭﻡ ﻳﻌﻨﯽ «ﻋﮑﺲ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺍﻓﺮﺍﺩ ﺷﻮﻳﻢ.

ﺻﻔﺤﻪ ﻋﮑﺲ‌ﻫﺎ

ﭘﺲ ﺍﺯ ﻭﺍﺭﺩ ﺷﺪﻥ ﺑﻪ ﺻﻔﺤﻪ ﻋﮑﺲ‌ﻫﺎ ﺩﺭ ﺑﺎﻻﯼ ﺻﻔﺤﻪ ۴ ﺳﺮﺑﺮﮒ (ﻋﮑﺲ‌ﻫﺎﯼ ﺑﺮﮔﺰﻳﺪﻩ، ﻫﻤﻪ ﻋﮑﺲ‌ﻫﺎ، ﺑﻴﺸﺘﺮ، ﺁﭘﻠﻮﺩ ﻋﮑﺲ) ﻣﯽ‌ﺑﻴﻨﻴﺪ. ﺑﺮﺍﯼ ﺁﻣﻮﺯﺵ ﺑﻬﺘﺮ ﻭ ﻳﺎﺩ ﮔﻴﺮﯼ ﺑﻬﺘﺮ ﺷﻤﺎ ﺍﺯ ﺳﺮﺑﺮﮒ ﭼﻬﺎﺭﻡ ﺷﺮﻭﻉ ﻣﯽ‌ﮐﻨﻴﻢ.

 

ﭼﻬﺎﺭﻣﻴﻦ ﺳﺮﺑﺮﮒ: ﺁﭘﻠﻮﺩ ﻋﮑﺲ
ﺩﺭ ﺍﻳﻦ ﺳﺮ ﺑﺮﮒ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻋﮑﺴﯽ ﺭﺍ ﺍﺯ ﺭﺍﻳﺎﻧﻪ ﺷﺨﺼﯽ ﺧﻮﺩ ﺁﭘﻠﻮﺩ ﮐﻨﻴﺪ. ﻃﺮﻳﻘﻪ ﺁﭘﻠﻮﺩ ﮐﺮﺩﻥ ﻋﮑﺲ ﺭﺍ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﺑﻪ ﻃﻮﺭ ﮐﺎﻣﻞ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻳﻢ. ﻋﮑﺴﯽ ﺭﺍ  ﺍﺯ ﺳﻴﺴﺘﻢ ﺧﻮﺩ ﺁﭘﻠﻮﺩ ﮐﻨﻴﺪ. ﭘﺲ ﺍﺯ ﺍﻳﻦ ﮐﻪ ﺷﻤﺎ ﻋﮑﺴﯽ ﺭﺍ ﺁﭘﻠﻮﺩ ﮐﺮﺩﻳﺪ ﺍﮔﺮ ﺁﻥ ﺭﺍ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺭﻓﺘﻦ ﺑﻪ ﺳﺮﺑﺮﮒ ﺍﻭﻟﯽ ﮐﻪ «ﻋﮑﺲ‌ﻫﺎﯼ ﺑﺮﮔﺰﻳﺪﻩ» ﺍﺳﺖ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﮐﻨﻴﺪ ﻭ ﺗﻐﻴﻴﺮﺍﺕ ﻻﺯﻡ ﺭﺍ ﺩﺭ ﺁﻥ ﺑﺪﻫﻴﺪ. ﺍﻣﺎ ﺍﮔﺮ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﻧﮕﺬﺍﺭﻳﺪ ﺑﺮﺍﯼ ﻣﺸﺎﻫﺪﻩ ﺁﻥ ﻋﮑﺲ ﺑﻪ ﺳﺮﺑﺮﮒ ﺳﻮﻡ «ﻫﻤﻪ ﻋﮑﺲ‌ﻫﺎ» ﺑﺮﻭﻳﺪ. ﺩﺭ ﺳﺮﺑﺮﮒ ﻫﻤﻪ ﻋﮑﺲ‌ﻫﺎﯼ ﻣﻦ، ﻫﻢ ﻋﮑﺲ‌ﻫﺎﯼ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ ﺷﺪﻩ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﻫﻢ ﻋﮑﺲ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﻧﮕﺬﺍﺷﺘﻪ‌ﺍﻳﺪ.

 

ﺍﻭﻟﻴﻦ ﺳﺮﺑﺮﮒ: ﻋﮑﺲ‌ﻫﺎﯼ ﺑﺮﮔﺰﻳﺪﻩ
ﭘﺲ ﺍﺯ ﺁﭘﻠﻮﺩ ﻋﮑﺲ ﻭ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻦ ﺁﻥ، ﻋﮑﺲ‌ﻫﺎ ﻭﺍﺭﺩ ﺍﻳﻦ ﺳﺮﺑﺮﮒ ﻣﯽ‌ﺷﻮﻧﺪ.

ﺳﺮﺑﺮﮒ ﻋﮑﺲ‌ﻫﺎﯼ ﺑﺮﮔﺰﻳﺪﻩ ﺩﺭ ﺻﻔﺤﻪ ﻋﮑﺲ‌ﻫﺎ

ﺩﺭ ﺳﻤﺖ ﭼﭗ ﺻﻔﺤﻪ ﺩﻭ ﮔﺰﻳﻨﻪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﻭﻟﯽ ﻧﺸﺎﻥ ﮐﺮﺩﻥ ﺍﻓﺮﺍﺩ ﻭ ﺩﻭﻣﯽ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺍﺳﺖ. ﺍﮔﺮ ﺭﻭﯼ ﭘﻴﮑﺎﻥ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﻟﻴﺴﺘﯽ ﺍﺯ ﮔﺰﻳﻨﻪ‌ﻫﺎ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﺁﻥ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

ﺍﻭﻟﻴﻦ ﮔﺰﻳﻨﻪ: ﻧﺸﺎﻥ ﮐﺮﺩﻥ ﺍﻓﺮﺍﺩ
ﺷﻤﺎ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺯﻳﺮ ﻣﯽ‌ﺷﻮﻳﺪ ﮐﻪ ﺩﺭ ﺑﺎﻻﯼ ﺻﻔﺤﻪ، ﺑﺎﻻﯼ ﻋﮑﺲ، ﻧﺎﻡ ﺻﺎﺣﺐ ﻋﮑﺲ ﻳﺎ ﻫﺮ ﮐﺴﯽ ﺭﺍ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺑﻨﻮﻳﺴﻴﺪ. ﺳﭙﺲ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﻧﺸﺎﻥ ﮔﺬﺍﺭﯼ ﺍﻧﺠﺎﻡ ﺷﺪ» ﮐﻪ ﺩﺭ ﭘﺎﻳﻴﻦ ﺻﻔﺤﻪ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ. 

ﻧﺸﺎﻥ ﮐﺮﺩﻥ ﺍﻓﺮﺍﺩ

ﺩﻭﻣﻴﻦ ﮔﺰﻳﻨﻪ: ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ

ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻋﮑﺲ

ﺩﺭ ﺑﺎﻻﯼ ﻋﮑﺲ ﮐﺎﺩﺭﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﺁﻥ ﻧﻈﺮ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﻋﮑﺲ ﺑﻨﻮﻳﺴﻴﺪ.

ﺩﺭ ﭘﺎﻳﻴﻦ ﻋﮑﺲ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻓﺰﻭﺩﻥ ﺁﻟﺒﻮﻡ» ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻋﮑﺲ ﺭﺍ ﺑﻪ ﺁﻟﺒﻮﻡ‌ﻫﺎﯼ ﻗﺒﻠﯽ‌ﺗﺎﻥ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺑﺮﺍﯼ ﺁﻥ ﺁﻟﺒﻮﻡ ﺟﺪﻳﺪ ﺑﺴﺎﺯﻳﺪ.

ﺗﻐﻴﻴﺮﺍﺕ ﻋﮑﺲ

ﺍﻭﻟﻴﻦ ﺷﮑﻞ ﺍﺯ ﺳﻤﺖ ﺭﺍﺳﺖ، ﺧﺎﻣﻮﺵ ﻭ ﺭﻭﺷﻦ ﮐﺮﺩﻥ ﺑﻬﺒﻮﺩ ﺧﻮﺩ ﮐﺎﺭ ﺍﺳﺖ.

ﺩﻭﻣﻴﻦ ﮔﺰﻳﻨﻪ «ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﻧﻮﺷﺘﻪ ﺑﻪ ﻋﮑﺲ» ﺍﺳﺖ. ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﻧﻤﺎﺩ ﺷﻤﺎ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺯﻳﺮ ﺧﻮﺍﻫﻴﺪ ﺷﺪ.

ﺍﻓﺰﻭﺩﻥ ﻧﻮﺷﺘﺎﺭ ﺑﻪ ﻣﺘﻦ

ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﺗﺼﻮﻳﺮ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ، ﺳﻤﺖ ﺭﺍﺳﺖ ﺻﻔﺤﻪ «ﺍﻓﺰﻭﺩﻥ ﻧﻮﺷﺘﺎﺭ ﺑﻪ ﻣﺘﻦ» ﺳﻪ ﮐﺎﺩﺭ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﻭﻟﻴﻦ ﮐﺎﺩﺭ «ﻧﻮﺷﺘﺎﺭ ﺑﺎﻻ» ﺩﻭﻣﻴﻦ ﮐﺎﺩﺭ «ﻧﻮﺷﺘﺎﺭ ﻭﺳﻂ» ﻭ ﺳﻮﻣﻴﻦ ﮐﺎﺩﺭ «ﻧﻮﺷﺘﺎﺭ ﭘﺎﻳﻴﻦ» ﺍﺳﺖ. ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺩﺭ ﻋﮑﺲ ۶ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﺩﺭ ﮐﺎﺩﺭ ﺍﻭﻝ ﻫﺮ ﻣﺘﻨﯽ ﺭﺍ ﺑﻨﻮﺳﻴﺪ ﺑﻪ ﺑﺎﻻﯼ ﻋﮑﺲ، ﺩﺭ ﮐﺎﺩﺭ ﻭﺳﻂ ﻫﺮ ﻣﺘﻨﯽ ﺭﺍ ﺑﻨﻮﻳﺴﻴﺪ ﺑﻪ ﻭﺳﻂ ﻋﮑﺲ ﻭ ﺩﺭ ﮐﺎﺩﺭ ﭘﺎﻳﻴﻦ ﻫﺮ ﻣﺘﻨﯽ ﺭﺍ ﺑﻨﻮﻳﺴﻴﺪ ﺑﻪ ﭘﺎﻳﻴﻦ ﻋﮑﺲ ﺍﺿﺎﻓﻪ ﻣﯽ‌ﺷﻮﺩ. ﭘﺲ ﭘﺎﻳﺎﻥ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﻣﺘﻦ ﺑﻪ ﻋﮑﺲ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﺫﺧﻴﺮﻩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﺳﻮﻣﻴﻦ ﮔﺰﻳﻨﻪ ﭼﺮﺧﺎﻧﺪﻥ ﻋﮑﺲ ﻭ ﭼﻬﺎﺭﻣﻴﻦ ﮔﺰﻳﻨﻪ ﺣﺬﻑ ﻋﮑﺲ ﺍﺳﺖ.

ﺍﻓﺰﻭﺩﻥ ﻋﮑﺲ: ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺭﺍﻳﺎﻧﻪ‌ﺗﺎﻥ ﻳﺎ ﮔﻮﮔﻞ+ ﻋﮑﺴﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﻭ ﺩﺭ ﮐﻨﺎﺭ ﺍﻳﻦ ﻋﮑﺲ ﺑﻪ ﺻﻮﺭﺕ ﺩﻭﺗﺎﻳﯽ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ. ﺑﻌﺪ ﺍﺯ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ ﻋﮑﺲ ﺷﻤﺎ ﺑﻪ ﺻﻮﺭﺕ ﺯﻳﺮ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﺩ.

ﺍﻓﺰﻭﺩﻥ ﻋﮑﺲ‌ﻫﺎ ﺑﻪ ﻫﻢ

ﺩﺭ ﺑﺨﺶ ﮔﻴﺮﻧﺪﻩ: ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺸﺨﺺ ﮐﻨﻴﺪ ﮐﻪ ﭼﻪ ﮐﺴﺎﻧﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﻳﻦ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺷﻤﺎ ﺭﺍ ﺑﺒﻴﻨﻨﺪ. ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺩﺭ ﮐﺎﺩﺭ ﺭﻭ ﺑﻪ ﺭﻭﯼ ﮔﻴﺮﻧﺪﻩ ﺑﺮ ﺭﻭﯼ ﺣﻠﻘﻪ‌ﻫﺎﻳﯽ ﮐﻪ ﺗﻤﺎﻳﻞ ﺩﺍﺭﻳﺪ ﺑﺎ ﺁﻥ‌ﻫﺎ ﺑﻪ «ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ» ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ.
