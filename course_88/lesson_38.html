ﻧﮑﺘﻪ: ﺑﺎ ﺗﻨﻈﻴﻤﺎﺗﯽ ﮐﻪ ﺩﺭ ﭘﺎﻳﻴﻦ ﺗﻮﺿﻴﺢ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺩ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﮐﺎﺭﯼ ﮐﻨﻴﺪ ﮐﻪ ﺍﻓﺮﺍﺩ ﺩﻟﺨﻮﺍﻩ ﺷﻤﺎ، ﻋﻤﻮﻡ ﻳﺎ ﻓﻘﻂ ﺷﻤﺎ ﻗﺎﺩﺭ ﺑﻪ ﺩﻳﺪﻥ ﺁﻥ ﺑﺎﺷﻴﺪ. ﭘﺲ ﻧﮕﺮﺍﻥ ﺩﻳﺪﻩ ﺷﺪﻥ ﻣﻄﺎﻟﺐ‌ﺗﺎﻥ ﺗﻮﺳﻂ ﺩﻳﮕﺮﺍﻥ ﻧﺒﺎﺷﻴﺪ ﺍﻧﺘﺨﺎﺏ ﺍﻳﻦ ﮐﻪ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺑﺮﺍﯼ ﭼﻪ ﮐﺴﺎﻧﯽ ﻗﺎﺑﻞ ﺩﻳﺪﻥ ﺑﺎﺷﺪ ﺑﺎ ﺷﻤﺎ ﺍﺳﺖ.

 

ﺍﻓﺮﺍﺩ
ﺩﺭ ﻋﮑﺲ ﺯﻳﺮ ﮐﺎﺩﺭ ﺍﻓﺮﺍﺩ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ. ﺩﺭ ﭘﺎﻳﻴﻦ ﮐﺎﺩﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻭﻳﺮﺍﻳﺶ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﺻﻔﺤﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻭﻳﺮﺍﻳﺶ ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻓﺮﺍﺩ ﺑﺎﺯ ﺷﻮﺩ.

ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ، ﮐﺎﺩﺭ ﺍﻓﺮﺍﺩ

ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﺷﻤﺎ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺩﺭ ﮔﻮﮔﻞ+ ﺣﻀﻮﺭ ﺩﺍﺭﻧﺪ ﺑﺪﺍﻧﻨﺪ ﮐﻪ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ ﭼﻪ ﮐﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ ﻭ ﺷﻤﺎ ﺩﺭ ﭼﻪ ﺣﻠﻘﻪ‌ﻫﺎﻳﯽ ﻫﺴﺘﻴﺪ ﻳﺎ ﻧﺪﺍﻧﻨﺪ؟ ﻳﺎ ﭼﻪ ﮐﺴﺎﻧﯽ ﺑﺪﺍﻧﻨﺪ؟

ﺗﻨﻈﻴﻤﺎﺕ ﺍﻓﺮﺍﺩ ﺩﺭ ﻧﻤﺎﻳﻪ

ﺍﻭﻟﻴﻦ ﮔﺰﻳﻨﻪ «ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ » ﺍﺳﺖ. ﺍﮔﺮ ﺗﻤﺎﻳﻞ ﻧﺪﺍﺭﻳﺪ ﮐﻪ ﮐﺴﯽ ﺑﺪﺍﻧﺪ ﮐﻪ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ ﭼﻪ ﮐﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ ﻣﺮﺑﻌﯽ ﮐﻪ ﺩﺭ ﮐﻨﺎﺭ ﮔﺰﻳﻨﻪ «ﻧﻤﺎﻳﺶ ﺍﻓﺮﺍﺩ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ» ﻫﺴﺖ ﺭﺍ ﺧﺎﻟﯽ ﺑﮕﺬﺍﺭﻳﺪ ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺗﻴﮏ ﻣﺮﺑﻊ ﺭﺍ ﺑﺎ ﮐﻠﻴﮏ ﺩﺭ ﺁﻥ ﺑﺰﻧﻴﺪ.

ﺑﻌﺪ ﺍﺯ ﮐﻠﻴﮏ ﺩﺭ ﻣﺮﺑﻊ ﺩﺭ ﮐﻨﺎﺭ ﮔﺰﻳﻨﻪ «ﻧﻤﺎﻳﺶ ﺍﻓﺮﺍﺩ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ» ﺟﻌﺒﻪ‌ﺍﻳﯽ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺁﻥ «ﺣﻠﻘﻪ‌ﻫﺎ» ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺍﺳﺖ. ﺑﺮ ﺭﻭﯼ ﺁﻥ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﮐﻪ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﭘﻴﺶ ﻓﺮﺽ ﮔﻮﮔﻞ ﻧﻤﺎﻳﺎﻥ ﺷﻮﻧﺪ ﺷﻤﺎ ﺑﺎ ﺗﻴﮏ ﺯﺩﻥ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺍﻳﻦ ﺣﻠﻘﻪ‌ﻫﺎ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﺍﻓﺮﺍﺩ ﺣﺎﺿﺮ ﺩﺭ ﺍﻳﻦ ﺣﻠﻘﻪ‌ﻫﺎ ﺭﺍ ﺩﻳﮕﺮﺍﻥ ﺑﺒﻴﻨﻨﺪ.

ﭘﺲ ﺍﺯ ﺍﻧﺘﺨﺎﺏ ﺣﻠﻘﻪ‌ﻫﺎ «ﺳﻮﺍﻝ ﭼﻪ ﮐﺴﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﺭﺍ ﺑﺒﻴﻨﺪ؟» ﺭﺍ ﺟﻮﺍﺏ ﺩﻫﻴﺪ. ﺩﺭ ﺻﻮﺭﺕ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺩﺭ ﺩﺍﻳﺮﻩ ﮐﻨﺎﺭ ﻋﻤﻮﻡ، ﻫﻤﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺒﻴﻨﻨﺪ ﻭ ﺩﺭ ﺻﻮﺭﺕ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺩﺭ ﺩﺍﻳﺮﻩ ﮐﻨﺎﺭ ﻓﻘﻂ ﺣﻠﻘﻪ‌ﻫﺎ، ﺍﻓﺮﺍﺩ ﺣﺎﺿﺮ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺒﻴﻨﻨﺪ ﮐﻪ ﭼﻪ ﮐﺴﺎﻧﯽ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ ﻫﺴﺘﻨﺪ.

ﺩﻭﻣﻴﻦ ﮔﺰﻳﻨﻪ «ﺷﻤﺎ ﺭﺍ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﻳﺸﺎﻥ ﺩﺍﺭﻧﺪ» ﺍﺳﺖ. ﺍﮔﺮ ﺗﻤﺎﻳﻞ ﻧﺪﺍﺭﻳﺪ ﮐﻪ ﻋﻤﻮﻡ ﺑﺪﺍﻧﻨﺪ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﭼﻪ ﺣﻠﻘﻪ‌ﻫﺎﻳﯽ ﻫﺴﺘﻴﺪ ﺗﻴﮏ ﮐﻨﺎﺭ ﮔﺰﻳﻨﻪ «ﻧﻤﺎﻳﺶ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺧﻮﺩ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻩ‌ﺍﻧﺪ» ﺭﺍ ﻧﺰﻧﻴﺪ ﻭ ﺍﮔﺮ ﺗﻤﺎﻳﻞ ﺩﺍﺭﻳﺪ ﮐﻪ ﺩﻳﮕﺮﺍﻥ ﺑﺪﺍﻧﻨﺪ ﺷﻤﺎ ﺩﺭ ﭼﻪ ﺣﻠﻘﻪ‌ﻫﺎﻳﯽ ﻫﺴﺘﻴﺪ ﺗﻴﮏ ﺭﺍ ﺑﺰﻧﻴﺪ. ﭘﺲ ﺍﺯ ﺍﺗﻤﺎﻡ ﮐﺎﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺫﺧﻴﺮﻩ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﺗﻐﻴﻴﺮﺍﺕ ﺛﺒﺖ ﺷﻮﺩ. ﺳﭙﺲ ﺷﻤﺎ ﺩﻭﺑﺎﺭﻩ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﻧﻤﺎﻳﻪ ﻣﯽ‌ﺷﻮﻳﺪ.

 

ﺩﺍﺳﺘﺎﻥ
ﺩﺭ ﻋﮑﺲ ﺯﻳﺮ ﮐﺎﺩﺭ ﺩﺍﺳﺘﺎﻥ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ، ﺩﺭ ﭘﺎﻳﻴﻦ ﮐﺎﺩﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻭﻳﺮﺍﻳﺶ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﺻﻔﺤﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻭﻳﺮﺍﻳﺶ ﺍﻃﻼﻋﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺩﺍﺳﺘﺎﻥ ﺑﺎﺯ ﺷﻮﺩ.

ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ، ﮐﺎﺩﺭ ﺩﺍﺳﺘﺎﻥ

ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﺩﺭ ﻣﻮﺭﺩ ﺧﻮﺩﺗﺎﻥ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﺪ ﺗﺎ ﺩﻳﮕﺮﺍﻥ ﮐﻪ ﺻﻔﺤﻪ ﺷﻤﺎ ﺭﺍ ﺑﺎﺯ ﻣﯽ‌ﮐﻨﻨﺪ ﺍﺯ ﺷﻤﺎ ﺍﻃﻼﻋﺎﺗﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺷﻤﺎ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﺪ ﺗﺎ ﺩﻳﮕﺮﺍﻥ ﺷﻤﺎ ﺭﺍ ﺑﻴﺸﺘﺮ ﺑﺸﻨﺎﺳﻨﺪ.

ﺗﻨﻈﻴﻤﺎﺕ ﺩﺍﺳﺘﺎﻥ ﺩﺭ ﻧﻤﺎﻳﻪ

ﺑﺨﺶ ﺍﻭﻝ ﮐﻪ ﺷﻌﺎﺭ ﺍﺳﺖ. ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﻳﮏ ﻣﻄﻠﺐ ﺧﺎﺻﯽ ﺩﺭ ﻣﻮﺭﺩ ﺧﻮﺩ ﺑﻨﻮﺳﻴﺪ ﮐﻪ ﺑﺎ ﺁﻥ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﻣﻌﺮﻓﯽ ﮐﻨﻴﺪ، ﻣﺎﻧﻨﺪ «ﺳﻪ ﺗﺎ ﮐﺎﺭ ﻫﺴﺖ ﺩﻭﺳﺖ ﺩﺍﺭﻡ ﺍﻧﺠﺎﻡ ﺑﺪﻡ. ﮐﺎﺭﮐﺮﺩﻥ، ﻓﻴﻠﻢ ﺩﻳﺪﻥ، ﺧﻮﺍﺑﻴﺪﻥ» ﺩﻗﺖ ﮐﻨﻴﺪ ﮐﻪ «ﺍﻳﻦ ﺑﺨﺶ ﻏﻴﺮ ﻗﺎﺑﻞ ﺗﻐﻴﻴﺮ ﺍﺳﺖ»، ﻳﻌﻨﯽ ﺩﻳﺪﻥ ﺁﻥ ﺑﺮﺍﯼ ﻋﻤﻮﻡ ﺁﺯﺍﺩ ﺍﺳﺖ ﭘﺲ ﺷﻌﺎﺭ ﺧﻮﺑﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ.

ﺑﺨﺶ ﺩﻭﻡ ﮐﻪ ﻣﻌﺮﻓﯽ ﺍﺳﺖ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﻫﺮ ﻃﻮﺭ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﻣﻌﺮﻓﯽ ﮐﻨﻴﺪ ﺗﺎ ﺩﻳﮕﺮﺍﻥ ﺑﺎ ﺷﻤﺎ، ﺭﻭﺣﻴﻪ‌ﺗﺎﻥ، ﻋﻼﻳﻖ‌ﺗﺎﻥ، ﻣﺬﻫﺐ‌ﺗﺎﻥ ﻭ ﻏﻴﺮﻩ ﺁﺷﻨﺎ ﺷﻮﻧﺪ.

ﻫﻤﺎﻥ ﻃﻮﺭ ﺩﺭ ﻋﮑﺲ ﺑﺎﻻ ﻣﺸﺨﺺ ﺍﺳﺖ ﺳﻤﺖ ﺭﺍﺳﺖ ﮔﺰﻳﻨﻪ ﻣﻌﺮﻓﯽ ﮔﺰﻳﻨﻪ‌ﺍﻳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺑﺎ ﻧﺎﻡ ﮔﺰﻳﻨﻪ ﻋﻤﻮﻣﯽ، ﭘﺲ ﺍﺯ ﺍﻳﻦ، ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﺭﺍ ﺩﺭ ﺟﺎﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ ﭘﺲ ﺑﺎ ﺩﻗﺖ ﺍﻳﻦ ﺑﺨﺶ ﺭﺍ ﻣﻄﺎﻟﻌﻪ ﮐﻨﻴﺪ ﺗﺎ ﺑﻪ ﻣﺸﮑﻠﯽ ﺑﺮ ﻧﺨﻮﺭﻳﺪ.

ﻧﮑﺘﻪ: ﭘﺲ ﺍﺯ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﻋﻤﻮﻣﯽ ﻳﮏ ﻣﺴﺘﻄﻴﻞ ﮐﻮﭼﮏ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺁﻥ ﭼﻨﺪ ﮔﺰﻳﻨﻪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻋﻤﻮﻣﯽ، ﻓﻘﻂ ﺷﻤﺎ، ﺳﻔﺎﺭﺷﯽ، ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ ﻭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﮔﺴﺘﺮﺩﻩ. ﺩﺭ ﺯﻳﺮ ﺗﻮﺿﻴﺢ ﺍﻳﻦ ﺑﺨﺶ ﻫﺎ ﺁﻭﺭﺩﻩ ﺷﺪﻩ ﺍﺳﺖ.

- ﻋﻤﻮﻣﯽ: ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﻋﻤﻮﻣﯽ ﺍﻳﻦ ﺗﻐﻴﻴﺮ ﻳﺎ ﺍﻳﻦ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺷﻤﺎ ﺭﺍ ﻫﻤﻪ ﻣﯽ‌ﺑﻴﻨﻨﺪ، ﭼﻪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺩﺭ ﻟﻴﺴﺖ ﺩﻭﺳﺘﺎﻥ ﺷﻤﺎ (ﻫﻤﺎﻥ ﺣﻠﻘﻪ‌ﻫﺎ) ﺣﻀﻮﺭ ﺩﺍﺭﻧﺪ ﻭ ﭼﻪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺁﻥ‌ﻫﺎ ﻗﺮﺍﺭ ﺩﺍﺭﻳﺪ ﻭ ﭼﻪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺷﻤﺎ ﺍﺻﻼ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻧﺪﻳﺪﻩ‌ﺍﻳﺪ ﮐﻪ ﺑﺨﻮﺍﻫﻴﺪ ﺑﺸﻨﺎﺳﻴﺪ ﻳﺎ ﻧﻪ.

- ﻓﻘﻂ ﺷﻤﺎ: ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﻓﻘﻂ ﺷﻤﺎ ﻫﻴﭻ ﮐﺲ ﺍﻳﻦ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺭﺍ ﻧﺨﻮﺍﻫﺪ ﺩﻳﺪ. ﭼﻪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ ﻫﺴﺘﻨﺪ ﻭ ﭼﻪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺷﻤﺎ ﻧﻴﺴﺘﻨﺪ ﻭ ﭼﻪ ﮐﺴﺎﻧﯽ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﺁﻥ‌ﻫﺎ ﻫﺴﺘﻴﺪ.
