ﺩﺭﺱ ﺷﺸﻢ- ﺗﻨﻈﻴﻤﺎﺕ ﻧﻤﺎﻳﻪ (ﭘﺮﻭﻓﺎﻳﻞ) ﮔﻮﮔﻞ+- ﺑﺨﺶ ﭼﻬﺎﺭﻡ

ﭘﻴﺸﮕﻔﺘﺎﺭ
ﺩﺭ ﺩﺭﺱ ﺷﺸﻢ ﺑﺨﺶ‌ﻫﺎﯼ ﺳﺮﺑﺮﮒ «ﺩﺭﺑﺎﺭﻩ» ﮐﻪ ﺷﺎﻣﻞ ﻣﺤﻞ ﺗﺤﺼﻴﻞ، ﮐﺎﺭ، ﺍﻃﻼﻋﺎﺕ ﺍﺳﺎﺳﯽ، ﻣﮑﺎﻥ، ﭘﻴﻮﻧﺪ ﻭ ﺍﻃﻼﻋﺎﺕ ﺗﻤﺎﺱ ﻣﯽ‌ﺷﻮﺩ ﺭﺍ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻳﻢ ﻭ ﺁﻥ ﺭﺍ ﺑﻪ ﭘﺎﻳﺎﻥ ﺭﺳﺎﻧﺪﻳﻢ.

ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺩﺭﺑﺎﺭﻩ ﺳﺮﺑﺮﮒ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ ﮐﻪ ﺷﺎﻣﻞ ﭘﺴﺖ‌ﻫﺎ، ﻧﻈﺮﺍﺕ، ﻋﮑﺲ‌ﻫﺎ ﻭ +1 ﻣﯽ‌ﺷﻮﺩ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

 

ﺳﺮﺑﺮﮒ ﭘﺴﺖ‌ﻫﺎ
ﺩﺭ ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ ﺩﺭ ﮐﻨﺎﺭ ﺳﺮﺑﺮﮒ «ﺩﺭﺑﺎﺭﻩ» ﺑﺮ ﺭﻭﯼ ﺳﺮﺑﺮﮒ «ﭘﺴﺖ‌ﻫﺎ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﻭ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﭘﺴﺖ‌ﻫﺎ ﺷﻮﻳﺪ. ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﻄﺎﻟﺐ‌ﺗﺎﻥ ﻳﺎ ﭘﺴﺖ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ. ﭘﺴﺖ‌ﻫﺎﯼ ﻧﻮﺷﺘﺎﺭﯼ، ﭘﻴﻮﻧﺪ، ﻋﮑﺲ، ﻓﻴﻠﻢ ﻭ ﺭﻭﻳﺪﺍﺩ‌ﻫﺎ. ﺷﻤﺎ ﺍﻳﻦ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺭﺍ ﺩﺭ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﺧﻮﺩ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ، ﺩﺭ ﻭﺍﻗﻊ ﻫﻤﭽﻴﻦ ﮐﺎﺩﺭﯼ ﮐﻪ ﺩﺭ ﻋﮑﺲ ﺯﻳﺮ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﺩﺭ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﻫﻢ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﭘﺲ ﺗﻮﺿﻴﺤﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﭘﺴﺖ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﺑﻌﺪ ﺍﺩﺍﻣﻪ ﻣﯽ‌ﺩﻫﻴﻢ. ﻓﻌﻼ ﺑﻬﺘﺮ ﺍﺳﺖ ﻧﻤﺎﻳﻪ ﺧﻮﺩ ﺭﺍ ﺗﮑﻤﻴﻞ ﮐﻨﻴﻢ.

ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ، ﺻﻔﺤﻪ ﭘﺴﺖ‌ﻫﺎ

ﺑﻪ ﺍﺩﺍﻣﻪ ﺗﮑﻤﻴﻞ ﻧﻤﺎﻳﻪ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ. ﺩﺭ ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ ﺑﺮ ﺭﻭﯼ ﺳﺮﺑﺮﮒ «ﻋﮑﺲ» ﮐﻪ ﺩﺭ ﮐﻨﺎﺭ ﺳﺮﺑﺮﮒ «ﭘﺴﺖ‌ﻫﺎ» ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﻋﮑﺲ ﺷﻮﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻋﮑﺲ ﺧﻮﺩﺗﺎﻥ ﻭ ﻋﮑﺲ‌ﻫﺎﯼ ﻣﻮﺭﺩ ﻋﻼﻗﻪ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ.

ﺻﻔﺤﻪ ﻧﻤﺎﻳﻪ، ﺳﺮﺑﺮﮒ ﭘﺴﺖ‌ﻫﺎ

ﺍﻭﻟﻴﻦ ﮔﺰﻳﻨﻪ‌ﺍﻳﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﮔﺰﻳﻨﻪ «ﺍﻓﺰﻭﺩﻥ ﻋﮑﺲ‌ﻫﺎ» ﺍﺳﺖ ﮐﻪ ﺩﺭﻭﻥ ﮐﺎﺩﺭﯼ ﻗﺮﻣﺰ ﺭﻧﮓ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺑﺎﺭﮔﺬﺍﺭﯼ ﻋﮑﺲ ﺍﺯ ﺭﺍﻳﺎﻧﻪ ﻣﯽ‌ﺷﻮﻳﺪ. ﻃﺮﻳﻘﻪ ﺑﺎﺭﮔﺬﺍﺭﯼ ﻋﮑﺲ ﺍﺯ ﺭﺍﻳﺎﻧﻪ ﺭﺍ ﺩﺭ ﻫﻤﻴﻦ ﺩﺭﺱ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻳﻢ.

ﺗﻨﻈﻴﻤﺎﺕ ﻋﮑﺲ

ﭘﺲ ﺍﺯ ﺑﺎﺭﮔﺬﺍﺭﯼ ﻋﮑﺲ  ﺍﺯ ﺭﺍﻳﺎﻧﻪ ﺻﻔﺤﻪ‌ﺍﻳﯽ ﮐﻪ ﺩﺭ ﺁﻥ ﺣﻀﻮﺭ ﺩﺍﺭﻳﺪ ﺗﻐﻴﻴﺮ ﻣﯽ‌ﮐﻨﺪ ﻭ ﻋﮑﺲ ﺑﺎﺭﮔﺬﺍﺭﯼ ﺷﺪﻩ ﺩﺭ ﮔﻮﺷﻪ ﺳﻤﺖ ﭼﭗ ﻧﻤﺎﻳﺎﻥ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺑﺎﻻﯼ ﺻﻔﺤﻪ ﮔﻮﺷﻪ ﺳﻤﺖ ﺭﺍﺳﺖ ﮔﺰﻳﻨﻪ «ﺍﻓﺰﻭﺩﻥ ﺑﻪ ﺁﻟﺒﻮﻡ» ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ، ﺑﺮ ﺭﻭﯼ ﺁﻥ ﮐﻠﻴﮏ ﮐﻨﻴﺪ، ﻣﺴﺘﻄﻴﻞ ﮐﻮﭼﮑﯽ ﺩﺭ ﺯﻳﺮ ﺁﻥ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﺩﺭ ﮐﺎﺩﺭ ﺍﻭﻝ ﮐﻪ ﺩﺭ ﺁﻥ «ﻧﺎﻡ ﺁﻟﺒﻮﻡ ﺟﺪﻳﺪ» ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺍﺳﺖ ﺑﺮﺍﯼ ﺁﻟﺒﻮﻣﯽ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﺴﺎﺯﻳﺪ ﻧﺎﻣﯽ ﺗﻌﻴﻴﻦ ﮐﻨﻴﺪ ﻣﺎﻧﻨﺪ ﻋﮑﺲ ﻫﻨﺮﻣﻨﺪﺍﻥ ﻳﺎ ﻋﮑﺲ‌ﻫﺎﯼ ﺧﺎﻧﻮﺍﺩﮔﯽ ﻭ ﻏﻴﺮﻩ. ﭘﺲ ﺍﺯ ﻧﻮﺷﺘﻦ ﻧﺎﻡ ﺑﺎ ﺯﺩﻥ ﺩﮐﻤﻪ ﺍﻳﻨﺘﺮ ﺑﺮ ﺭﻭﯼ ﺻﻔﺤﻪ ﮐﻠﻴﺪ ﺁﻟﺒﻮﻡ ﺳﺎﺧﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻋﮑﺲ ﺑﺎﺭﮔﺬﺍﺭﯼ ﺷﺪﻩ ﻫﻢ ﻭﺍﺭﺩ ﺁﻥ ﻣﯽ‌ﺷﻮﺩ.

ﺩﺭ ﮔﻮﺷﻪ ﺳﻤﺖ ﭼﭗ ﺑﺎﻻﯼ ﻋﮑﺲ ﮔﺰﻳﻨﻪ ﺑﺎﺭﮔﺬﺍﺭﯼ ﺍﺯ ﺭﺍﻳﺎﻧﻪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﻫﻤﺎﻥ ﺭﻭﺵ ﻗﺒﻠﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻫﺮ ﭼﻨﺪ ﺗﻌﺪﺍﺩ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﻋﮑﺲ ﺑﺎﺭﮔﺬﺍﺭﯼ ﮐﻨﻴﺪ.

ﺩﺭ ﮐﻨﺎﺭ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﺩﻭ ﺟﻌﺒﻪ ﮐﻮﭼﮏ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ‌ﻫﺎ ﺍﻧﺪﺍﺯﻩ ﻋﮑﺲ ﮐﻮﭼﮏ ﻭ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﮐﻨﺎﺭ ﺍﻳﻦ ﺟﻌﺒﻪ‌ﻫﺎ، ﺟﻌﺒﻪ ﺩﻳﮕﺮﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺭﻭﯼ ﺁﻥ ﻓﻠﺶ ﮐﻮﭼﮑﯽ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﺷﻤﺎ ﺗﺮﺗﻴﺐ ﭼﻴﺪﻣﺎﻥ ﻋﮑﺲ‌ﻫﺎﯼ ﺧﻮﺩ ﺩﺭ ﺁﻟﺒﻮﻡ ﺭﺍ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﻴﺪ، ﺑﺮﺍﺳﺎﺱ ﺗﺎﺭﻳﺦ ﻳﺎ ﻧﺎﻡ ﻳﺎ ﺑﺮﻋﮑﺲ.

ﺣﺎﻝ ﻓﻠﺶ ﻣﻮﺷﻮﺍﺭﻩ ﺧﻮﺩ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﻋﮑﺲ ﻗﺮﺍﺭ ﺩﻫﻴﺪ، ﺩﺍﻳﺮﻩ‌ﺍﻳﯽ ﻧﻘﻄﻪ ﭼﻴﻦ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﺩ ﺍﮔﺮ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﮐﻠﻴﮏ ﮐﻨﻴﺪ، ﮐﺎﺩﺭ ﮐﻮﭼﮑﯽ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺑﺮﺍﯼ ﻧﺸﺎﻧﻪ‌ﮔﺬﺍﺭﯼ ﻳﺎ ﺗﮓ ﮐﺮﺩﻥ ﻳﺎ ﭘﻴﻮﻧﺪ ﺯﺩﻥ ﺍﺳﺖ.

ﻧﮑﺘﻪ: ﻧﺸﺎﻧﻪ ﮔﺬﺍﺭﯼ ﭼﻴﺴﺖ؟ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻓﺮﺩ ﺧﺎﺻﯽ ﺭﺍ ﺑﻪ ﺍﻳﻦ ﻋﮑﺲ ﭘﻴﻮﻧﺪ ﺑﺰﻧﻴﺪ ﺑﺎ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ ﻓﺮﺩ ﻣﻮﺭﺩ ﻧﻈﺮ ﻣﻄﻠﻊ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻣﯽ ﺗﻮﺍﻧﺪ ﻋﮑﺲ ﻭ ﺁﻟﺒﻮﻡ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﮐﻨﺪ.

ﭘﺎﻳﻴﻦ ﻋﮑﺲ، ﮔﻮﺷﻪ ﺭﺍﺳﺖ ﺁﻥ ﮔﺰﻳﻨﻪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺑﺎ ﻧﺎﻡ «ﺟﻠﺪ ﺁﻟﺒﻮﻡ» ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻋﮑﺲ ﻣﻮﺭﺩ ﻧﻈﺮ ﺑﻪ ﻋﻨﻮﺍﻥ ﻋﮑﺲ ﺟﻠﺪ ﺁﻟﺒﻮﻡ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﺷﻮﺩ.

ﭘﺲ ﺍﺯ ﺍﺗﻤﺎﻡ ﮐﺎﺭ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻧﺠﺎﻡ ﺷﺪ» ﮐﻪ ﺩﺭ ﭘﺎﻳﻴﻦ ﺳﻤﺖ ﺭﺍﺳﺖ ﺻﻔﺤﻪ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﺑﻪ ﻣﺮﺣﻠﻪ ﺑﻌﺪﯼ ﺑﺮﻭﻳﺪ.

ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺷﻤﺎ ﺗﻌﻴﻴﻦ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﻋﮑﺲ ﺷﻤﺎ ﺑﺮﺍﯼ ﭼﻪ ﮐﺴﺎﻧﯽ ﻳﺎ ﭼﻪ ﺣﻠﻘﻪ‌ﻫﺎﻳﯽ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ ﺷﻮﺩ. ﺩﺭ ﭘﺎﻳﻴﻦ ﻋﮑﺲ ﮐﺎﺩﺭﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺑﺎ ﻧﺎﻡ «ﮔﻴﺮﻧﺪﻩ» ﮐﻪ ﺩﺭ ﺍﻳﻦ ﮐﺎﺩﺭ ﺷﻤﺎ ﺣﻠﻘﻪ‌ﻫﺎﯼ ﻣﻮﺭﺩ ﻧﻈﺮ ﺧﻮﺩ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻴﺪ.

ﻧﺤﻮﻩ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ

ﺑﺮﺍﯼ ﺍﻧﺘﺨﺎﺏ ﺣﻠﻘﻪ‌ﻫﺎ ﻳﺎ ﺍﻓﺮﺍﺩ ﺑﺮ ﺭﻭﯼ ﺁﺩﻣﮑﯽ ﮐﻪ ﺩﺭ ﮐﻨﺎﺭ ﮐﺎﺩﺭ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﺍﻧﺘﺨﺎﺏ ﺍﻓﺮﺍﺩ ﻣﻮﺭﺩ ﻧﻈﺮ ﺑﺮﺍﯼ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ
