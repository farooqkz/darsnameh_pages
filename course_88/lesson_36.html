ﺩﺭﺱ ﻧﻮﺯﺩﻫﻢ- ﺗﻨﻈﻴﻤﺎﺕ ﺣﺴﺎﺏ

ﭘﻴﺸﮕﻔﺘﺎﺭ
ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﺩﺭ ﻣﻮﺭﺩ ﺻﻔﺤﺎﺕ ﻣﺤﻠﯽ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ ﻭ ﺗﻮﺿﻴﺢ ﺍﻣﮑﺎﻧﺎﺕ ﮔﻮﮔﻞ+ ﺭﺍ ﺑﻪ ﭘﺎﻳﺎﻥ ﺭﺳﺎﻧﺪﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺩﺭ ﻣﻮﺭﺩ ﺍﻣﻨﻴﺖ ﻭ ﺗﻨﻈﻴﻢ ﺣﺴﺎﺏ ﮔﻮﮔﻞ+ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

 

ﺗﻨﻈﻴﻤﺎﺕ ﺣﺴﺎﺏ
ﺑﺮﺍﯼ ﻭﺭﻭﺩ ﺑﻪ ﺗﻨﻈﻴﻤﺎﺕ ﺣﺴﺎﺏ ﺧﻮﺩ، ﺩﺭ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﺑﺮ ﺭﻭﯼ ﻋﮑﺲ ﻧﻤﺎﻳﻪ‌ﺗﺎﻥ ﮐﻪ ﺩﺭ ﮔﻮﺷﻪ ﺻﻔﺤﻪ ﺳﻤﺖ ﭼﭗ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﻭﺭﻭﺩ ﺑﻪ ﺗﻨﻈﻴﻤﺎﺕ ﺣﺴﺎﺏ

ﺩﺭ ﺻﻔﺤﻪ ﮐﻪ ﺩﺭ ﺯﻳﺮ ﻋﮑﺲ ﻧﻤﺎﻳﻪ‌ﺗﺎﻥ ﺑﺎﺯ ﺷﺪﻩ ﺍﺳﺖ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺣﺴﺎﺏ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺯﻳﺮ ﺧﻮﺍﻫﻴﺪ ﺷﺪ.

ﺗﻨﻈﻴﻤﺎﺕ ﺣﺴﺎﺏ

ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﭼﻬﺎﺭ ﺳﺮﺑﺮﮒ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺳﺮﺑﺮﮒ ﺍﻭﻝ «ﺍﻃﻼﻋﺎﺕ ﺷﺨﺼﯽ» ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﺑﻴﺸﺘﺮ ﺗﻨﻈﻴﻤﺎﺕ ﺍﻳﻦ ﺻﻔﺤﻪ (ﻭﻳﺮﺍﻳﺶ ﻧﻤﺎﻳﻪ، ﻣﺸﺎﻫﺪﻩ ﻧﻤﺎﻳﻪ، ﻭﻳﺮﺍﻳﺶ ﺗﻨﻈﻴﻤﺎﺕ، ﻣﺸﺎﻫﺪﻩ ﺣﻠﻘﻪ‌ﻫﺎﻳﺘﺎﻥ) ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ‌ﺍﻳﻢ.

ﻭﻳﺮﺍﻳﺶ ﺍﻃﻼﻋﺎﺕ ﻭﺭﻭﺩ ﺑﻪ ﺳﻴﺴﺘﻢ

ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﻭﻳﺮﺍﻳﺶ» ﮐﻪ ﺩﺭ ﺯﻳﺮ ﻋﻨﻮﺍﻥ «ﺍﻳﻤﻴﻞ ﻭ ﺷﻤﺎﺭﻩ ﺗﻠﻔﻦ» ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﺍﻳﻤﻴﻞ ﻭ ﺷﻤﺎﺭﻩ ﺗﻠﻔﻦ

ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﺷﻤﺎ ﺭﺍﻳﺎﻧﺎﻣﻪ ﻭ ﺗﻨﻈﻴﻤﺎﺕ ﺗﻠﻔﻦ ﺧﻮﺩ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺁﻧﻬﺎ ﺭﺍ ﻭﻳﺮﺍﻳﺶ ﮐﻨﻴﺪ. ﺷﻤﺎﺭﻩ ﺗﻠﻔﻦ‌‌ﻫﺎ ﻭ ﺳﺮﻭﻳﺲ‌‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﻟﻴﺴﺖ ﺷﺪﻩ‌ﺍﻧﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺳﺮﻭﻳﺲ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﮔﻮﮔﻞ ﺍﺯ ﺍﻳﻦ ﺷﻤﺎﺭﻩ‌ﻫﺎ ﮐﻪ ﺍﺭﺍﺋﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪ ﭘﺲ ﻭﻳﺮﺍﻳﺶ ﻳﺎ ﺣﺬﻑ ﺷﻤﺎﺭﻩ‌‌‌ﺗﺎﻥ ﺩﺭ ﺍﻳﻨﺠﺎ ﺁﻥ ﺳﺮﻭﻳﺲ‌‌ﻫﺎ ﺭﺍ ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﻗﺮﺍﺭ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺩ.

 

ﻋﻨﻮﺍﻥ ﺍﻳﻤﻴﻞ ﺑﺎﺯﻳﺎﺑﯽ
ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﻳﮏ ﺍﻳﻤﻴﻞ ﺑﺎﺯﻳﺎﺑﯽ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ» ﺷﻤﺎ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺟﺪﻳﺪﯼ ﺧﻮﺍﻫﻴﺪ ﺷﺪ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﺑﺎﻳﺪ ﺩﻭﺑﺎﺭﻩ ﮔﺬﺭﻭﺍﮊﻩ ﺣﺴﺎﺏ ﮐﺎﺭﺑﺮﯼ ﺧﻮﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ. ﺳﭙﺲ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺯﻳﺮ ﺧﻮﺍﻫﻴﺪ ﺷﺪ.

ﮔﺰﻳﻨﻪ‌ﻫﺎﯼ ﺑﺎﺯﻳﺎﺑﯽ ﺣﺴﺎﺏ

«ﮔﺰﻳﻨﻪ‌‌ﻫﺎﯼ ﺑﺎﺯﻳﺎﺑﯽ» ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌‌ﮐﻨﻨﺪ ﺗﺎ ﺍﺯ ﺣﺴﺎﺏ ﺧﻮﺩ ﺩﺭ ﺑﺮﺍﺑﺮ ﻫﮑﺮ‌ﻫﺎ ﻣﺤﺎﻓﻈﺖ ﮐﺮﺩﻩ ﻭ ﻫﻤﭽﻨﻴﻦ ﺭﻭﺷﯽ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺷﻤﺎ ﻗﺮﺍﺭ ﻣﯽ‌‌ﺩﻫﻨﺪ ﺗﺎ ﺩﺭ ﺻﻮﺭﺕ ﻓﺮﺍﻣﻮﺵ ﮐﺮﺩﻥ ﮔﺬﺭﻭﺍﮊﻩ ﺑﻪ ﺣﺴﺎﺏ ﺧﻮﺩ ﺩﺳﺘﺮﺳﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﺯﻳﺮ ﻋﻨﻮﺍﻥ «ﺗﻠﻔﻦ ﻫﻤﺮﺍﻩ» ﺑﺮ ﺭﻭﯼ ﻭﻳﺮﺍﻳﺶ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﻭ ﺩﺭ ﺻﻮﺭﺕ ﺗﻤﺎﻳﻞ ﺷﻤﺎﺭﻩ ﺟﺪﻳﺪ ﺧﻮﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ.

ﻋﻨﻮﺍﻥ «ﺍﻳﻤﻴﻞ» ﺑﻪ ﺩﻭ ﻗﺴﻤﺖ «ﺑﺎﺯﻳﺎﺑﯽ ﺁﺩﺭﺱ ﺍﻳﻤﻴﻞ» ﻭ «ﺁﺩﺭﺱ‌ﻫﺎﯼ ﺍﻳﻤﻴﻞ ﺟﺎﻳﮕﺰﻳﻦ» ﺗﻘﺴﻴﻢ ﻣﯽ‌ﺷﻮﺩ.

 

ﺑﺎﺯﻳﺎﺑﯽ ﺁﺩﺭﺱ ﺍﻳﻤﻴﻞ
ﺁﺩﺭﺱ ﺍﻳﻤﻴﻞ ﺑﺎﺯﻳﺎﺑﯽ ﺭﻭﺷﯽ ﺭﺍ ﻓﺮﺍﻫﻢ ﻣﯽ‌ﺁﻭﺭﺩ ﺗﺎ ﺩﺭ ﺻﻮﺭﺕ ﺍﻳﺠﺎﺩ ﺗﻐﻴﻴﺮﺍﺕ ﻣﻬﻢ ﺩﺭ ﺍﺭﺗﺒﺎﻁ ﺑﺎ ﺣﺴﺎﺏ ﮐﺎﺭﺑﺮﯼ ﺷﻤﺎ، ﮔﻮﮔﻞ ﺑﺘﻮﺍﻧﺪ ﺑﺎ ﺷﻤﺎ ﺩﺭ ﺗﻤﺎﺱ ﺑﺎﺷﺪ. ﺍﺯ ﺁﺩﺭﺱ ﺭﺍﻳﺎﻧﺎﻣﻪ ﺑﺎﺯﻳﺎﺑﯽ ﺷﻤﺎ ﮔﻮﮔﻞ ﺑﺮﺍﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﻋﻤﻠﮑﺮﺩ ﻫﮑﺮ‌ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﺪ، ﻫﻤﭽﻨﻴﻦ ﺑﺮﺍﯼ ﺷﻤﺎ ﺭﺍﻳﺎﻧﺎﻣﻪ‌ﺍﯼ ﺍﺭﺳﺎﻝ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺩﺭ ﺻﻮﺭﺕ ﻋﺪﻡ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﺣﺴﺎﺏ ﮐﺎﺭﺑﺮﯼ ﺧﻮﺩ ﺑﺘﻮﺍﻧﻴﺪ ﮔﺬﺭﻭﺍﮊﻩ ﺧﻮﺩ ﺭﺍ ﺑﺎﺯﻳﺎﺑﯽ ﮐﻨﻴﺪ.

ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻓﺰﻭﺩﻥ ﺁﺩﺭﺱ ﺍﻳﻤﻴﻞ ﺑﺎﺯﻳﺎﺑﯽ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﻭ ﺭﺍﻳﺎﻧﺎﻣﻪ ﺩﻳﮕﺮﯼ ﺑﻪ ﺟﺰ ﺭﺍﻳﺎﻧﺎﻣﻪ ﺍﺻﻠﯽ ﺧﻮﺩ ﻭﺍﺭﺩ ﮐﻨﻴﺪ.

 

ﺁﺩﺭﺱ‌ﻫﺎﯼ ﺍﻳﻤﻴﻞ ﺟﺎﻳﮕﺰﻳﻦ
ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺁﺩﺭﺱ ﺍﻳﻤﻴﻞ ﺟﺎﻳﮕﺰﻳﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻭﺍﺭﺩ ﺑﺮﻧﺎﻣﻪ ﺷﻮﻳﺪ ﻭ ﮐﺎﺭ‌ﻫﺎﯼ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﺍﺯ ﺁﺩﺭﺱ ﺍﻳﻤﻴﻞ ﺟﺎﻳﮕﺰﻳﻦ ﻣﯽ‌‌ﺗﻮﺍﻧﻴﺪ ﺑﺮﺍﯼ ﻭﺭﻭﺩ ﺑﻪ ﺣﺴﺎﺏ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺩﺭ ﺩﺭﺱ ﺻﻔﺤﺎﺕ ﻳﮏ ﺭﺍﻳﺎﻧﺎﻣﻪ ﺟﺎﻳﮕﺰﻳﻦ ﺳﺎﺧﺘﻴﻢ. ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﺭﺩ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﻣﻮﺍﺭﺩﯼ ﺭﺍ ﺑﺎ ﺁﻥ‌ﻫﺎ ﺑﻪ‌ ﺍﺷﺘﺮﺍﮎ ﻣﯽ‌‌ﮔﺬﺍﺭﻳﺪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺁﺩﺭﺱ ﺍﻳﻤﻴﻞ ﺟﺎﻳﮕﺰﻳﻦ ﺷﻤﺎ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﮐﻨﻨﺪ. ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻓﺰﻭﺩﻥ ﺁﺩﺭﺱ‌ﻫﺎﯼ ﺍﻳﻤﻴﻞ ﺟﺎﻳﮕﺰﻳﻦ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﻭ ﺭﺍﻳﺎﻧﺎﻣﻪ ﺩﻳﮕﺮﯼ ﺑﻪ ﺟﺰ ﺭﺍﻳﺎﻧﺎﻣﻪ ﺍﺻﻠﯽ ﺧﻮﺩ ﻭﺍﺭﺩ ﮐﻨﻴﺪ. ﺳﭙﺲ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﺫﺧﻴﺮﻩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﭘﺲ ﺍﺯ ﺍﻳﻦ ﺗﻐﻴﺮﺍﺕ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻧﺠﺎﻡ ﺷﺪ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

 

ﺳﺮﺑﺮﮒ ﺍﻣﻨﻴﺖ

ﺷﻤﺎ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺗﻐﻴﻴﺮ ﮔﺬﺭﻭﺍﮊﻩ» ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺯﻳﺮ ﻣﯽ‌ﺷﻮﻳﺪ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﮔﺬﺭﻭﺍﮊﻩ ﺧﻮﺩ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ.

ﺗﻐﻴﻴﺮ ﮔﺬﺭﻭﺍﮊﻩ

ﺩﺭ ﮐﺎﺩﺭ ﺍﻭﻝ ﮔﺬﺭﻭﺍﮊﻩ ﻓﻌﻠﯽ ﺧﻮﺩ ﺭﺍ ﺑﻨﻮﻳﺴﻴﺪ ﻭ ﺩﺭ ﮐﺎﺩﺭ ﺩﻭﻡ ﻭ ﺳﻮﻡ ﮔﺬﺭﻭﺍﮊﻩ ﺟﺪﻳﺪ ﺧﻮﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ ﺳﭙﺲ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺗﻐﻴﻴﺮ ﮔﺬﺭﻭﺍﮊﻩ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﻣﺮﺍﺣﻞ ﺗﺎﻳﻴﺪ ۲ ﻣﺮﺣﻠﻪ‌ﺍﻳﯽ

ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺷﺮﻭﻉ ﺗﻨﻈﻴﻢ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﺩﺭ ﺻﻔﺤﻪ ﺑﻌﺪ ﮔﺬﺭﻭﺍﮊﻩ ﺧﻮﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ. ﭘﺲ ﺩﺭ ﺻﻔﺤﻪ ﺯﻳﺮ (ﺩﺭ ﻋﮑﺲ ﻣﯽ‌ﺑﻴﻨﻴﺪ):

ﻣﺮﺍﺣﻞ ﺗﺎﻳﻴﺪ ۲ ﻣﺮﺣﻠﻪ‌ﺍﻳﯽ

ﺳﺮﺑﺮﮒ ﺯﺑﺎﻥ

ﺳﺮﺑﺮﮒ ﺍﺑﺰﺍﺭﻫﺎﯼ ﻣﺮﺑﻮﻁ ﺑﻪ ﺩﺍﺩﻩ‌ﻫﺎ

ﺳﺮﺑﺮﮒ ﺍﺑﺰﺍﺭﻫﺎﯼ ﻣﺮﺑﻮﻁ ﺑﻪ ﺩﺍﺩﻩ‌ﻫﺎ

- ﻣﺪﻳﺮﻳﺖ ﺣﺴﺎﺏ ﻏﻴﺮ ﻓﻌﺎﻝ
ﭘﺲ ﺍﺯ ﺗﻮﻗﻒ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺣﺴﺎﺏ ﺧﻮﺩ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺨﻮﺍﻫﻴﺪ ﺩﺍﺩﻩ‌‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻳﮏ ﺩﻭﺳﺖ ﻣﻮﺭﺩ ﺍﻋﺘﻤﺎﺩ ﻳﺎ ﻳﮑﯽ ﺍﺯ ﺍﻋﻀﺎﯼ ﺧﺎﻧﻮﺍﺩﻩ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ، ﻳﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺨﻮﺍﻫﻴﺪ ﺣﺴﺎﺏ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻃﻮﺭ ﮐﺎﻣﻞ ﺣﺬﻑ ﮐﻨﻴﺪ. ﺷﺮﺍﻳﻂ ﺯﻳﺎﺩﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﺭﺍ ﺍﺯ ﺩﺳﺘﺮﺳﯽ ﻳﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺣﺴﺎﺏ ﮔﻮﮔﻞ ﺧﻮﺩ ﺑﺎﺯ ﺩﺍﺭﺩ. ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﺪﻳﺮ ﺣﺴﺎﺏ ﻏﻴﺮﻓﻌﺎﻝ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺸﺨﺺ ﮐﻨﻴﺪ ﺁﻳﺎ ﺣﺴﺎﺏ ﺷﻤﺎ ﻏﻴﺮ ﻓﻌﺎﻝ ﺑﺎﺷﺪ ﻭ ﭼﻪ ﺯﻣﺎﻧﯽ ﺍﻳﻦ ﺣﺎﻟﺖ ﺭﻭﯼ ﺩﻫﺪ، ﭼﻪ ﺍﺗﻔﺎﻗﯽ ﺑﺮﺍﯼ ﺩﺍﺩﻩ‌ﻫﺎﯼ ﺷﻤﺎ ﺑﻴﻔﺘﺪ ﻭ ﭼﻪ ﮐﺴﯽ ﺍﺯ ﺁﻥ ﻣﻄﻠﻊ ﺷﻮﺩ.

ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺗﻨﻈﻴﻢ ﻣﺪﻳﺮﻳﺖ ﺣﺴﺎﺏ ﻏﻴﺮ ﻓﻌﺎﻝ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﺩﺭ ﺻﻔﺤﻪ ﺑﻌﺪﯼ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺭﺍﻩ ﺍﻧﺪﺍﺯﯼ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﺯﻳﺮ ﻋﻨﻮﺍﻥ «ﺑﻪ ﻣﻦ ﺍﻃﻼﻉ ﺩﺍﺩﻩ ﺷﻮﺩ» ﻣﺪﻳﺮ ﺣﺴﺎﺏ ﻏﻴﺮ ﻓﻌﺎﻝ ﻗﺒﻞ ﺍﺯ ﭘﺎﻳﺎﻥ ﻣﻬﻠﺖ ﺯﻣﺎﻧﯽ ﺷﻤﺎ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﭘﻴﺎﻣﮏ ﻭ ﺩﺭ ﺻﻮﺭﺕ ﺗﻤﺎﻳﻞ ﺑﺎ ﺭﺍﻳﺎﻧﺎﻣﻪ ﺑﻪ ﺷﻤﺎ ﻫﺸﺪﺍﺭ ﻣﯽ‌ﺩﻫﺪ. ﺷﻤﺎﺭﻩ ﺗﻠﻔﻦ ﻭ ﺭﺍﻳﺎﻧﺎﻣﻪ ﻣﻮﺭﺩ ﻧﻈﺮ ﺧﻮﺩ ﺭﺍ ﻭﺍﺭﺩ ﮐﻨﻴﺪ.

ﺯﻳﺮ ﻋﻨﻮﺍﻥ «ﺩﻭﺭﻩ ﺍﺗﻤﺎﻡ ﻣﻬﻠﺖ ﺯﻣﺎﻧﯽ» ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻳﮏ ﻣﻬﻠﺖ ﺯﻣﺎﻧﯽ ﺗﻨﻈﻴﻢ ﮐﻨﻴﺪ ﮐﻪ ﺑﻌﺪ ﺍﺯ ﺁﻥ ﺣﺴﺎﺏ ﺷﻤﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﺣﺴﺎﺏ ﻏﻴﺮ ﻓﻌﺎﻝ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻪ ﻣﯽ‌‌ﺷﻮﺩ. ﺍﻳﻦ ﻣﻬﻠﺖ ﺯﻣﺎﻧﯽ ﺍﺯ ﺁﺧﺮﻳﻦ ﻭﺭﻭﺩ ﺑﻪ ﺳﻴﺴﺘﻢ ﺷﻤﺎ ﺩﺭ ﺣﺴﺎﺏ ﮔﻮﮔﻞ ﺷﺮﻭﻉ ﻣﯽ‌‌ﺷﻮﺩ. ﺑﺮ ﺭﻭﯼ ﺟﻌﺒﻪ ﺯﻳﺮ ﺍﻳﻦ ﻋﻨﻮﺍﻥ ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﻭ ﻣﻬﻠﺖ ﺯﻣﺎﻧﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ.

ﺯﻳﺮ ﻋﻨﻮﺍﻥ «ﺍﻃﻼﻉ ﺑﻪ ﻣﺨﺎﻃﺒﻴﻦ ﻭ ﺍﺷﺘﺮﺍﮎ ‌ﮔﺬﺍﺭﯼ ﺩﺍﺩﻩ‌‌ﻫﺎ» ﻣﺨﺎﻃﺒﻴﻦ ﻣﻮﺭﺩ ﺍﻋﺘﻤﺎﺩﯼ ﮐﻪ ﻣﯽ‌‌ﺧﻮﺍﻫﻴﺪ ﺑﻪ ﺁﻧﻬﺎ ﺍﻃﻼﻉ ﺩﻫﻴﺪ ﮐﻪ ﺩﻳﮕﺮ ﺍﺯ ﺣﺴﺎﺏ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﯽ‌‌ﮐﻨﻴﺪ ﺭﺍ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ. ﻫﻤﭽﻨﻴﻦ ﻣﯽ‌‌ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﺻﻮﺭﺕ ﺗﻤﺎﻳﻞ ﺩﺍﺩﻩ‌‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺁﻧﻬﺎ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ.

ﺣﺎﻝ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻓﺰﻭﺩﻥ ﻣﺨﺎﻃﺐ ﻣﻮﺭﺩ ﺍﻋﺘﻤﺎﺩ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ.
