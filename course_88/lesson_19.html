ﺍﻧﺘﺨﺎﺏ ﻧﻮﻉ ﺍﻧﺠﻤﻦ

ﺍﮔﺮ ﺣﺎﻟﺖ ﺧﺼﻮﺻﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻳﺪ ﺑﺎ ﺍﻳﻦ ﺳﻮﺍﻝ ﺭﻭ ﺑﻪ ﺭﻭ ﻣﯽ‌ﺷﻮﻳﺪ ﮐﻪ ﺁﻳﺎ ﺩﻳﮕﺮﺍﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﻧﺠﻤﻦ ﺷﻤﺎ ﺭﺍ ﺟﺴﺘﺠﻮ ﮐﻨﻨﺪ؟ ﺷﻤﺎ ۲ ﮔﺰﻳﻨﻪ ﺑﺮﺍﯼ ﺍﻧﺘﺨﺎﺏ ﺩﺍﺭﻳﺪ:

ﺍﻭﻟﯽ ﺧﻴﺮ ﺑﻮﺩﻩ (ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﻳﻦ ﮐﻪ ﺍﺯ ﺟﺴﺘﺠﻮ‌ﻫﺎ ﻣﺨﻔﯽ ﺑﺎﺷﺪ) ﺩﻭﻣﯽ ﺑﻠﻪ ﺍﺳﺖ (ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﻳﻦ ﮐﻪ ﺍﻓﺮﺍﺩ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﻧﺠﻤﻦ ﺷﻤﺎ ﺭﺍ ﺩﺭ ﺟﺴﺘﺠﻮ‌ﻫﺎ ﭘﻴﺪﺍ ﮐﻨﻨﺪ ﻭ ﺩﺭ ﺧﻮﺍﺳﺖ ﭘﻴﻮﺳﺘﻦ ﺑﻔﺮﺳﺘﻨﺪ.‌)

ﭘﺲ ﺍﺯ ﺍﻧﺘﺨﺎﺏ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻳﺠﺎﺩ ﺍﻧﺠﻤﻦ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﺑﺨﺶ «ﺩﺭﺑﺎﺭﻩ» ﺍﻭﻟﻴﻦ ﻣﮑﺎﻧﯽ ﺍﺳﺖ ﮐﻪ ﺍﻋﻀﺎﯼ ﺟﺪﻳﺪ ﻣﯽ‌‌ﺗﻮﺍﻧﻨﺪ ﺑﺎ ﻣﺮﺍﺟﻌﻪ ﺑﻪ ﺁﻥ ﺩﺭﺑﺎﺭﻩ ﺍﻧﺠﻤﻦ ﺷﻤﺎ ﺍﻃﻼﻋﺎﺕ ﮐﺴﺐ ﮐﻨﻨﺪ ﭘﺲ ﺍﻃﻼﻋﺎﺕ ﮐﺎﻣﻠﯽ ﺩﺭ ﻣﻮﺭﺩ ﺍﻧﺠﻤﻦ ﺧﻮﺩ ﻭﺍﺭﺩ ﮐﻨﻴﺪ. ﻫﺪﻑ ﺍﻧﺠﻤﻦ ﺭﺍ ﺗﻮﺿﻴﺢ ﺩﻫﻴﺪ، ﺍﻧﺘﻈﺎﺭﺍﺕ ﺧﻮﺩ ﺭﺍ ﺩﺭﺑﺎﺭﻩ ﭘﺴﺖ‌ﻫﺎ ﻭ ﺑﺤﺚ‌ﻫﺎ ﻣﺸﺨﺺ ﮐﻨﻴﺪ ﻭ ﺳﺎﻳﺮ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﻻﺯﻡ ﺍﺳﺖ ﺩﺭ ﻣﻮﺭﺩ ﺍﻧﺠﻤﻦ ﺧﻮﺩ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ.

 

ﭼﻬﺎﺭﻣﻴﻦ ﺑﺨﺶ: ﭘﻴﻮﻧﺪ‌ﻫﺎ
ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﻓﺰﻭﺩﻥ ﭘﻴﻮﻧﺪ» ﺁﺩﺭﺱ‌ﻫﺎﯼ ﻣﻮﺭﺩ ﻧﻈﺮ ﺧﻮﺩ ﻣﺎﻧﻨﺪ (ﺁﺩﺭﺱ ﻓﻴﺴﺒﻮﮎ، ﺻﻔﺤﻪ ﺧﻮﺩ ﺩﺭ ﮔﻮﮔﻞ+، ﺁﺩﺭﺱ ﻭﺑﻼﮒ ﺧﻮﺩ ﻭ ﻏﻴﺮﻩ) ﺭﺍ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ.

ﭘﻨﺠﻤﻴﻦ ﺑﺨﺶ: ﻣﻮﻗﻴﻌﺖ ﻣﮑﺎﻧﯽ
ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﻣﻮﻗﻴﻌﺖ ﻣﮑﺎﻧﯽ (ﺷﻬﺮ، ﮐﺸﻮﺭ ﻭ ﻏﻴﺮﻩ) ﺍﺧﺘﻴﺎﺭﯼ ﺍﺳﺖ.

ﭘﺲ ﺍﺯ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﺍﻃﻼﻋﺎﺕ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﺍﻧﺠﺎﻡ ﺷﺪ ﺑﺎﻻﯼ ﺻﻔﺤﻪ ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﺳﭙﺲ ﺷﻤﺎ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﺍﻧﺠﻤﻦ ﺧﻮﺩ ﻣﯽ‌ﺷﻮﻳﺪ.

ﺻﻔﺤﻪ ﺍﻧﺠﻤﻦ ﺷﻤﺎ

ﺯﻳﺮ ﻧﺎﻡ ﺍﻧﺠﻤﻦ ﻭ ﺑﺎﻻﯼ ﻋﮑﺲ ﺧﻮﺩ ﺑﺮ ﺭﻭﯼ ﻋﻼﻣﺖ ﭼﺮﺥ ﺩﻧﺪﻩ ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﻣﻨﻮﯼ ﮐﺸﻮﻳﯽ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ.

ﺗﻨﻈﻴﻤﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻧﺠﻤﻦ

 

ﮔﺰﻳﻨﻪ ﺍﻭﻝ: ﺩﻋﻮﺕ ﺍﺯ ﺍﻓﺮﺍﺩ
ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﻳﮏ ﮐﺎﺩﺭ ﺩﻋﻮﺕ ﺑﺎ ﻧﻮﺷﺘﻪ «ﺩﻋﻮﺕ ﺍﻓﺮﺍﺩ ﺑﻪ ﺍﻳﻦ ﺍﻧﺠﻤﻦ» ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ. ﻣﯽ‌‌ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﭘﻴﺎﻣﯽ ﺑﻨﻮﻳﺴﻴﺪ ﮐﻪ ﻫﻤﺮﺍﻩ ﺑﺎ ﺩﻋﻮﺕ ﺷﻤﺎ ﺍﺭﺳﺎﻝ ﺷﻮﺩ. ﻧﺎﻡ‌‌ﻫﺎ، ﺣﻠﻘﻪ‌ﻫﺎ ﻳﺎ ﺁﺩﺭﺱ‌‌ﻫﺎﯼ ﺍﻳﻤﻴﻞ ﺍﻓﺮﺍﺩﯼ ﺭﺍ ﮐﻪ ﻣﯽ‌‌ﺧﻮﺍﻫﻴﺪ ﺩﻋﻮﺕ ﮐﻨﻴﺪ ﺭﺍ ﺍﺿﺎﻓﻪ ﻧﻤﺎﻳﻴﺪ. ﺳﭙﺲ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺍﺭﺳﺎﻝ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﮔﺰﻳﻨﻪ ﺩﻭﻡ: ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺍﻧﺠﻤﻦ
ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﻫﻢ ﻣﺎﻧﻨﺪ ﮔﺰﻳﻨﻪ «ﺩﻋﻮﺕ ﺍﺯ ﺍﻓﺮﺍﺩ» ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﺯﻣﺎﻧﯽ ﭘﺪﻳﺪﺍﺭ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺍﻧﺠﻤﻦ ﺷﻤﺎ ﻋﻤﻮﻣﯽ ﺑﺎﺷﺪ. ﺩﺭ ﺻﻮﺭﺕ ﺳﺎﺧﺖ ﺍﻧﺠﻤﻦ ﺧﺼﻮﺻﯽ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﻭﺟﻮﺩ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﮔﺰﻳﻨﻪ ﺳﻮﻡ: ﻭﻳﺮﺍﻳﺶ ﺍﻧﺠﻤﻦ
ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺩﺭ ﺍﺑﺘﺪﺍ ﻭﺍﺭﺩ ﮐﺮﺩﻳﺪ ﺭﺍ ﻭﻳﺮﺍﻳﺶ ﮐﻨﻴﺪ. ﭘﺲ ﺍﺯ ﭘﺎﻳﺎﻥ ﻭﻳﺮﺍﻳﺶ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺫﺧﻴﺮﻩ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ.

ﮔﺰﻳﻨﻪ ﭼﻬﺎﺭﻡ: ﻣﺪﻳﺮﻳﺖ ﺍﻋﻀﺎ
ﺍﻳﻦ ﺳﺮﺑﺮﮒ ﺑﺮﺍﯼ ﻭﻳﺮﺍﻳﺶ ﺍﻋﻀﺎ ﻭ ﻧﺎﻇﺮﺍﻥ ﺍﻧﺠﻤﻦ ﺍﺳﺖ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻋﻀﺎﯼ ﺍﻧﺠﻤﻦ ﺭﺍ ﺍﺿﺎﻓﻪ ﻳﺎ ﺣﺬﻑ ﮐﺮﺩ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺮﺍﯼ ﺍﻧﺠﻤﻦ ﻧﺎﻇﺮ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﻭﻗﺘﯽ ﻭﺍﺭﺩ ﺍﻳﻦ ﺻﻔﺤﻪ ﻣﯽ‌ﺷﻮﻳﺪ ﺗﻤﺎﻡ ﺍﻋﻀﺎﯼ ﺍﻧﺠﻤﻦ ﺭﺍ ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﭘﻴﮑﺎﻥ ﮐﻨﺎﺭ ﻋﮑﺲ ﻫﺮ ﻓﺮﺩ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﻐﻴﻴﺮﺍﺕ ﮔﻔﺘﻪ ﺷﺪﻩ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ.

ﮔﺰﻳﻨﻪ ﭘﻨﺠﻢ: ﺧﺮﻭﺝ ﺍﺯ ﺍﻧﺠﻤﻦ
ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﺁﺧﺮﻳﻦ ﻋﻀﻮ ﺍﻳﻦ ﺍﻧﺠﻤﻦ ﻫﺴﺘﻴﺪ ﺑﺮﺍﯼ ﺗﺮﮎ ﺁﻥ ﺑﺎﻳﺪ ﺍﻧﺠﻤﻦ ﺭﺍ ﺣﺬﻑ ﮐﻨﻴﺪ. ﻋﻤﻠﮑﺮﺩ ﺣﺬﻑ ﺍﻧﺠﻤﻦ ﻗﺎﺑﻞ ﻭﺍﮔﺮﺩ ﻧﻴﺴﺖ. ﺑﺎ ﮐﻠﻴﮏ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻋﻀﻮﻳﺖ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺍﻳﻦ ﺍﻧﺠﻤﻦ ﺣﺬﻑ ﮐﻨﻴﺪ ﻭ ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﭘﺴﺖ‌ﻫﺎﯼ ﺷﻤﺎ ﺍﻳﻦ ﺍﻧﺠﻤﻦ ﺭﺍ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﺷﻤﺎ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻧﻤﯽ‌ﺷﻮﺩ.

ﺩﺭ ﻣﻮﺭﺩ ﮐﺎﺭﺑﺮﺩ ﮔﺰﻳﻨﻪ ﺷﺸﻢ ﻭ ﻫﻔﺘﻢ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﮐﺎﻣﻞ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ ﺍﻳﻢ.

ﺯﻳﺮ ﻧﺎﻡ ﺍﻧﺠﻤﻦ ﻭ ﺑﺎﻻﯼ ﻋﮑﺲ ﺧﻮﺩ ﺑﺮ ﺭﻭﯼ ﻋﻼﻣﺖ ﺯﻧﮓ  ﻧﻤﺎﺩ ﺯﻧﮓ ﻣﯽ‌‌ﺗﻮﺍﻧﻴﺪ ﺗﻨﻈﻴﻢ ﺍﻋﻼﻥ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ. ﺍﻋﻼﻥ ﺭﻭﺷﻦ ﻳﺎ ﺍﻋﻼﻥ ﺧﺎﻣﻮﺵ.

 ﺯﻣﺎﻧﯽ ﮐﻪ ﭘﺴﺘﯽ ﺭﺍ ﺩﺭ ﺍﻧﺠﻤﻦ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻴﺪ ﺩﺭ ﮐﻨﺎﺭ ﭘﺴﺖ ﺷﻤﺎ ﭘﻴﮑﺎﻧﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻳﮏ ﻣﻨﻮﯼ ﮐﺸﻮﻳﯽ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﺗﻨﻈﻴﻤﺎﺕ ﭘﺴﺖ‌ﻫﺎ ﺩﺭ ﺍﻧﺠﻤﻦ

ﭘﻴﻦ ﮐﺮﺩﻥ ﭘﺴﺖ: ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﭘﺴﺖ ﺷﻤﺎ ﺑﻪ ﺑﺎﻻﯼ ﺻﻔﺤﻪ ﺍﻧﺠﻤﻦ ﻓﺮﺳﺘﺎﺩﻩ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﻭﺍﻗﻊ ﺷﻤﺎ ﭘﺴﺖ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺭﻭﺯﺭﺳﺎﻧﯽ ﻣﯽ‌ﮐﻨﻴﺪ.

ﻭﻳﺮﺍﻳﺶ ﭘﺴﺖ: ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭘﺴﺖ ﺭﺍ ﻭﻳﺮﺍﻳﺶ ﮐﻨﻴﺪ.

ﺣﺬﻑ ﭘﺴﺖ: ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭘﺴﺖ ﺭﺍ ﺣﺬﻑ ﮐﻨﻴﺪ.

ﭘﻴﻮﻧﺪ ﭘﺴﺖ: ﺑﺎ ﮐﻠﻴﮏ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮔﺰﻳﻨﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻟﻴﻨﮏ ﺍﻳﻦ ﭘﺴﺖ ﺭﺍ ﺭﻭﻧﻮﺷﺖ ﻭ ﺩﺭ ﺻﻔﺤﻪ ﺧﻮﺩ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ.

ﺩﺭ ﻣﻮﺭﺩ ﺩﻭ ﮔﺰﻳﻨﻪ ﺁﺧﺮ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ.

 

ﺧﻼﺻﻪ
ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻧﺤﻮﻩ ﭘﻴﻮﺳﺘﻦ ﺑﻪ ﻳﮏ ﺍﻧﺠﻤﻦ ﻭ ﻧﺤﻮﻩ ﺳﺎﺧﺖ ﻳﮏ ﺍﻧﺠﻤﻦ ﺑﺮﺍﯼ ﺧﻮﺩ ﺭﺍ ﻳﺎﺩ ﮔﺮﻓﺘﻴﺪ ﻭ ﺑﺎ ﺗﻨﻈﻴﻤﺎﺕ ﺍﻧﺠﻤﻦ ﺧﻮﺩ ﺁﺷﻨﺎ ﺷﺪﻳﺪ.

ﺁﻧﭽﻪ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﺑﻌﺪ ﺧﻮﺍﻫﻴﺪ ﺧﻮﺍﻧﺪ:
ﺩﺭ ﺩﺭﺱ ﺑﻌﺪ ﺑﺎ ﺭﻭﻳﺪﺍﺩ‌ﻫﺎ ﺁﺷﻨﺎ ﻣﯽ‌ﺷﻮﻳﺪ ﻭ ﻧﺤﻮﻩ ﺳﺎﺧﺖ ﻳﮏ ﺭﻭﻳﺪﺍﺩ ﻭ ﺗﻨﻈﻴﻤﺎﺕ ﺁﻥ ﺭﺍ ﻳﺎﺩ ﺧﻮﺍﻫﻴﺪ ﮔﺮﻓﺖ.
