ﭘﻴﺸﮕﻔﺘﺎﺭ
ﺩﺭ ﺩﺭﺱ ﭘﻴﺸﻴﻦ ﺗﻨﻈﻴﻤﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﻋﮑﺲ‌ﻫﺎ ﻭ ﺁﻟﺒﻮﻡ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻩ ﺷﺪ ﻭ ﮔﻔﺘﻪ ﺷﺪ ﭼﮕﻮﻧﻪ ﺍﺯ ﺍﻓﮑﺖ‌ﻫﺎﯼ ﻣﺘﺤﺮﮎ ﮔﻮﮔﻞ ﺑﺮﺍﯼ ﻋﮑﺲ‌ﻫﺎﯼ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﺑﺤﺜﯽ ﺑﺎ ﻧﺎﻡ ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﺑﭙﺮﺩﺍﺯﻳﻢ.

 

ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ
ﺗﻮﺿﻴﺤﯽ ﮐﻪ ﮔﻮﮔﻞ ﺑﺮﺍﯼ ﺍﻳﻦ ﻗﺴﻤﺖ ﻧﻮﺷﺘﻪ: ﮐﺎﻭﺵ ﭘﺴﺖ‌ﻫﺎﯼ ﭘﺮﻃﺮﻓﺪﺍﺭ ﻭ ﺟﺎﻟﺒﯽ ﮐﻪ ﺩﺭ ‪Google+‏ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ، ﺍﺳﺖ ﮐﻪ ﺧﻮﺩ ﻧﻤﺎﻳﺎﻧﮕﺮ ﮐﺎﺭﺑﺮﺩ ﺍﻳﻦ ﻗﺴﻤﺖ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻳﻦ ﺍﺑﺰﺍﺭ ﮐﺎﻓﯽ‌ﺳﺖ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺍﺻﻠﯽ‌ﺗﺎﻥ ﺩﺭ ﮔﻮﮔﻞ+ ﺷﻮﻳﺪ. ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﻃﺮﻳﻘﻪ ﻭﺍﺭﺩ ﺷﺪﻥ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﺩﺭﺱ ﺩﺍﺩﻳﻢ. ﺩﺭ ﺳﻤﺖ ﺭﺍﺳﺖ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﮔﻮﮔﻞ+ ﻣﻮﺷﻮﺍﺭﻩ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ «ﺻﻔﺤﻪ ﺍﺻﻠﯽ» ﻧﮕﻪ ﺩﺍﺭﻳﺪ، ﻳﮏ ﻣﻨﻮﯼ ﮐﺸﻮﻳﯽ ﺑﺎﺯ ﺧﻮﺍﻫﺪ ﺷﺪ، ﺑﺮ ﺭﻭﯼ ﮔﺰﻳﻨﻪ ﭘﻨﺠﻢ ﻳﻌﻨﯽ «ﺧﺒﺮ ﻫﺎﯼ ﺩﺍﻍ ﭼﻴﺴﺖ» ﮐﻠﻴﮏ ﮐﻨﻴﺪ ﺗﺎ ﻭﺍﺭﺩ ﺻﻔﺤﻪ ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﭼﻴﺴﺖ ﺷﻮﻳﻢ.

ﺻﻔﺤﻪ ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﭼﻴﺴﺖ؟

ﺻﻔﺤﻪ ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﻣﺤﺘﻮﺍﯼ ﺍﻧﺘﺨﺎﺑﯽ ﺟﺎﻟﺐ ﻭ ﻣﻨﺎﺳﺒﯽ ﮐﻪ ﺩﺭ ﮔﻮﮔﻞ+ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﺭﺍ ﺑﺮﺟﺴﺘﻪ ﻣﯽ‌‌ﮐﻨﺪ ﻭ ﺍﻃﻼﻋﺎﺕ ﻣﻔﻴﺪ ﻭ ﻣﺘﻨﻮﻋﯽ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺧﺒﺮ‌ﻫﺎﯼ ﺩﺍﻍ ﻭ ﭘﺴﺖ‌‌ﻫﺎﯼ ﺗﻮﺻﻴﻪ ﺷﺪﻩ ﺑﺎ ﺑﻨﺮﯼ ﺩﺭ ﺑﺎﻻﯼ ﭘﺴﺖ ﺗﺤﺖ ﻋﻨﻮﺍﻥ «ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﻭ ﻣﻮﺍﺭﺩ ﺗﻮﺻﻴﻪ‌ ﺷﺪﻩ ﺩﺭ  Google+‎» ﺩﺭ ﺟﺮﻳﺎﻥ «ﻫﻤﻪ ﺣﻠﻘﻪ‌‌ﻫﺎ» ﺩﺭ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﺷﻤﺎ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﻧﺪ.

ﻭﺍﺭﺩ ﺻﻔﺤﻪ «ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﭼﻴﺴﺖ» ﺷﻮﻳﺪ. ﮐﻨﺎﺭ ﺟﻤﻠﻪ «ﺧﺒﺮ ﻫﺎﯼ ﺩﺍﻍ ﭼﻴﺴﺖ ﻭ ﭼﻪ ﭼﻴﺰﯼ ﺗﻮﺻﻴﻪ ﻣﯽ‌ﺷﻮﺩ» ﺑﺮ ﺭﻭﯼ ﮐﻠﻴﮏ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺗﻤﺎﻳﻞ ﺩﺍﺭﻳﺪ ﮐﻪ ﺧﺒﺮ‌ﻫﺎﯼ ﺩﺍﻍ ﺩﺭ ﺟﺮﻳﺎﻥ ﺍﺻﻠﯽ ﺷﻤﺎ (‌ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﮔﻮﮔﻞ+) ﻧﻤﺎﻳﺶ ﺩﺍﺩﻩ ﺷﻮﻧﺪ ﮔﺰﻳﻨﻪ «ﻧﻤﺎﻳﺶ ﭘﺴﺖ‌ﻫﺎ ﺩﺭ ﺟﺮﻳﺎﻥ ﺍﺻﻠﯽ» ﺭﺍ ﺗﻴﮏ ﺩﺍﺭ ﮐﻨﻴﺪ. ﺳﭙﺲ ﻧﺸﺎﻧﮕﺮﺗﺎﻥ ﺭﺍ ﺭﻭﯼ ﺗﻌﺪﺍﺩ ﻧﮕﻪ‌ ﺩﺍﺭﻳﺪ ﻭ ﻳﮑﯽ ﺍﺯ ﮔﺰﻳﻨﻪ‌ﻫﺎﯼ (‏ﺑﻴﺸﺘﺮ، ﺍﺳﺘﺎﻧﺪﺍﺭﺩ ﻳﺎ ﮐﻤﺘﺮ) ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺷﻤﺎ ﻣﻴﺰﺍﻥ ﺗﻌﺪﺍﺩ ﺩﺭﻳﺎﻓﺖ ﺍﻳﻦ ﺧﺒﺮﻫﺎ ﺭﺍ ﺑﺮﺍﯼ ﺧﻮﺩ ﺗﻨﻈﻴﻢ ﻣﯽ‌ﮐﻨﻴﺪ.

 ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻧﻤﯽ‌ﺧﻮﺍﻫﻴﺪ ﭘﺴﺘﯽ ﺭﺍ ﺍﺯ «ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﻭ ﻣﻮﺍﺭﺩ ﺗﻮﺻﻴﻪ ﺷﺪﻩ» ﺑﺒﻴﻨﻴﺪ ﻋﻼﻣﺖ ﮔﺰﻳﻨﻪ «ﻧﻤﺎﻳﺶ ﭘﺴﺖ‌‌ﻫﺎ ﺩﺭ ﺟﺮﻳﺎﻥ ﺻﻔﺤﻪ ﺍﺻﻠﯽ» ﺭﺍ ﺑﺮﺩﺍﺭﻳﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺗﻤﺎﻡ ﻣﻄﺎﻟﺐ ﻣﺮﺑﻮﻁ ﺑﻪ ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﺭﺍ ﺍﺯ ﺻﻔﺤﻪ ﺷﻤﺎ ﺑﺮ ﻣﯽ‌ﺩﺍﺭﺩ ﻭ ﺩﺭ ﺁﻳﻨﺪﻩ ﻫﻢ ﺁﻧﻬﺎ ﺭﺍ ﻧﺸﺎﻥ ﻧﻤﯽ‌ﺩﻫﺪ.

ﺩﺭ ﺯﻳﺮ ﻋﻨﻮﺍﻥ «ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ ﭼﻴﺴﺖ ﻭ ﭼﻪ ﭼﻴﺰﯼ ﺗﻮﺻﻴﻪ ﻣﯽ‌ﺷﻮﺩ» ﮐﺎﺩﺭ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭﻭﻥ ﺁﻥ ﻋﻼﻣﺖ # ﻫﺸﺘﮓ (ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻞ ﺩﺭ ﻣﻮﺭﺩ ﺁﻥ ﺗﻮﺿﻴﺢ ﺩﺍﺩﻳﻢ) ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎ ﻭﺍﺭﺩ ﮐﺮﺩﻥ ﮐﻠﻤﻪ ﻣﻮﺭﺩ ﻧﻈﺮ ﺷﻤﺎ ﺻﻔﺤﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺁﻥ ﻧﻤﺎﻳﺎﻥ ﻣﯽ‌ﺷﻮﺩ. ﻓﻘﻂ ﮐﻠﻤﻪ‌ﺍﻳﯽ ﮐﻪ ﻭﺍﺭﺩ ﻣﯽ‌ﮐﻨﻴﺪ ﺑﺎﻳﺪ ﻗﺒﻼ ﺑﻪ ﺻﻮﺭﺕ ﻫﺸﺘﮓ ﺩﺭ ﺁﻣﺪﻩ ﺑﺎﺷﺪ ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺟﺴﺘﺠﻮ ﻧﺎﻣﻮﻓﻖ ﺍﺳﺖ.

ﺩﺭ ﺍﻳﻦ ﺑﺨﺶ (ﺧﺒﺮﻫﺎﯼ ﺩﺍﻍ) ﻣﻮﺍﺭﺩ ﺩﻳﮕﺮﯼ ﺑﺮﺍﯼ ﺑﻴﺸﺘﺮ ﺟﺬﺍﺏ ﮐﺮﺩﻥ ﺁﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﺯ ﺟﻤﻠﻪ ﺁﻥ «ﺍﻧﺠﻤﻦ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﭙﺴﻨﺪﻳﺪ»، «ﺩﻧﺒﺎﻝ ﮐﺮﺩﻥ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ» ﻭ «ﻣﻤﮑﻦ ﺍﺳﺖ ﻋﻼﻗﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ» ﺍﺳﺖ ﮐﻪ ﻫﺮ ﮐﺪﺍﻡ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺷﻤﺎ ﺭﺍ ﺩﺭ ﺑﻬﺒﻮﺩ ﮐﺎﺭﺑﺮﺩ ﺍﺯ ﮔﻮﮔﻞ+ ﮐﻤﮏ ﮐﻨﻨﺪ.

 

ﺍﻧﺠﻤﻦ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﭙﺴﻨﺪﻳﺪ
ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﺻﻔﺤﻪ ﺑﻪ ﺷﻤﺎ ﺍﻧﺠﻤﻦ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺯ ﺁﻧﻬﺎ ﺧﻮﺷﺘﺎﻥ ﺑﻴﺎﻳﺪ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﺪ. ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﮔﺰﻳﻨﻪ «ﻣﺸﺎﻫﺪﻩ ﻫﻤﻪ» ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺠﻤﻦ‌ﻫﺎﯼ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﻳﺎ ﺟﺴﺘﺠﻮ ﮐﻨﻴﺪ.

 

ﺩﻧﺒﺎﻝ ﮐﺮﺩﻥ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ
ﺍﻳﻦ ﻗﺴﻤﺖ ﺑﻪ ﺷﻤﺎ ﺩﻧﺒﺎﻝ ﮐﺮﺩﻥ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﻭﺳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﺪ. ﺍﻳﻦ ﺑﺨﺶ ﻫﻢ ﻣﺎﻧﻨﺪ ﺑﺨﺶ ﭘﻴﺸﻴﻦ ﺩﺍﺭﺍﯼ ﮔﺰﻳﻨﻪ «ﻣﺸﺎﻫﺪﻩ ﻫﻤﻪ» ﺍﺳﺖ.

 

ﻣﻤﮑﻦ ﺍﺳﺖ ﻋﻼﻗﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ

ﺑﻄﻮﺭ ﮐﻞ ﺍﻳﻦ ﺑﺨﺶ‌ﻫﺎ ﻫﻤﻪ ﺑﺮﺍﯼ ﭘﻴﺸﻨﻬﺎﺩ ﻣﻄﺎﻟﺐ ﻭ ﻣﺤﺘﻮﺍﻫﺎﯼ ﺑﺮﺟﺴﺘﻪ ﻭ ﺧﻮﺍﻧﺪﻧﯽ ﺑﺮﺍﯼ ﮐﺎﺭﺑﺮﺍﻥ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﮐﻤﯽ ﮔﺮﺩﺵ ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ‌ﻫﺎ ﺑﻪ ﻫﺮ ﺁﻧﭽﻪ ﻣﻮﺭﺩ ﻋﻼﻗﻪ ﺷﻤﺎ ﺍﺳﺖ ﺩﺳﺘﺮﺳﯽ ﭘﻴﺪﺍ ﮐﺮﺩﻩ ﻭ ﺑﺎ ﺩﻧﺒﺎﻝ ﮐﺮﺩﻥ ﺁﻧﻬﺎ ﻣﻄﺎﻟﺐ ﺩﻭﺳﺖ ﺩﺍﺷﺘﻨﯽ ﺭﺍ ﺑﻪ ﺻﻔﺤﻪ ﮔﻮﮔﻞ+ ﺧﻮﺩ ﺑﻴﺎﻭﺭﻳﺪ.

 

ﺧﻼﺻﻪ
ﺻﻔﺤﻪ ﺧﺒﺮ‌ﻫﺎﯼ ﺩﺍﻍ ﻣﺤﺘﻮﺍﯼ ﺍﻧﺘﺨﺎﺑﯽ ﺟﺎﻟﺐ ﻭ ﻣﻨﺎﺳﺒﯽ ﮐﻪ ﺩﺭ ﮔﻮﮔﻞ+ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﺭﺍ ﺑﺮﺟﺴﺘﻪ ﻣﯽ‌‌ﮐﻨﺪ ﻭ ﺍﻃﻼﻋﺎﺕ ﻣﻔﻴﺪ ﻭ ﻣﺘﻨﻮﻋﯽ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﻨﻈﻴﻢ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﻣﺤﺘﻮﺍ ﺩﺭ ﺟﺮﻳﺎﻥ ﺍﺻﻠﯽ ﺷﻤﺎ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﺷﻮﺩ ﻳﺎ ﺧﻴﺮ.
