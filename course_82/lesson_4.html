ﺩﺭﺱ ۴- ﺭﻭﺵ ﻫﺎﯼ ﺑﻬﺒﻮﺩ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ ﺩﺭ ﻣﻄﺎﻟﻌﻪ

ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﮔﺬﺷﺘﻪ ﺑﺎ ﺩﻻﻳﻞ ﻋﺪﻡ ﺗﻤﺮﮐﺰ، ﺍﺯ ﺟﻤﻠﻪ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﻭ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﻭ ﻫﻤﭽﻨﻴﻦ ﺷﻴﻮﻩ‌ﻫﺎﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺁﻥ ﺁﺷﻨﺎ ﺷﺪﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺭﻭﺵ‌ﻫﺎﯼ ﺑﻬﺒﻮﺩ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ ﺩﺭ ﻣﻄﺎﻟﻌﻪ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ.

 

ﻳﺎﺩﺩﺍﺷﺖ ﺑﺮﺩﺍﺭﯼ
ﻳﺎﺩﺩﺍﺷﺖ‌ﺑﺮﺩﺍﺭﯼ ﻳﮑﯽ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﻮﺛﺮ ﻭ ﮐﺎﺭﺁﻣﺪ ﺑﺮﺍﯼ ﺍﻓﺰﺍﻳﺶ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ ﺍﺳﺖ ﮐﻪ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺗﻤﺎﻡ ﺍﻓﮑﺎﺭ ﺑﺮ ﺭﻭﯼ ﻳﮏ ﻧﻘﻄﻪ ﻣﺘﻤﺮﮐﺰ ﺷﻮﻧﺪ.

ﻳﺎﺩﺩﺍﺷﺖ‌ﺑﺮﺩﺍﺭﯼ ﻓﻘﻂ ﺑﻪ ﻣﻌﻨﺎﯼ ﺁﻥ ﻧﻴﺴﺖ ﮐﻪ ﺧﻼﺻﻪ‌ﺍﯼ ﺍﺯ ﺍﺗﻔﺎﻗﺎﺕ ﻭ ﺭﻭﻳﺪﺍﺩﻫﺎ ﺭﺍ ﻧﻮﺷﺖ، ﺑﻠﮑﻪ ﺧﻮﺩ ﻳﮏ ﻓﺮﺁﻳﻨﺪﯼ ﺍﺳﺖ ﮐﻪ ﻗﺎﺑﻠﻴﺖ‌ ﺑﻬﺒﻮﺩ ﺗﮑﻨﻴﮏ‌ﻫﺎﯼ ﺗﻤﺮﮐﺰ ﺑﺮ ﻣﻄﺎﻟﻌﻪ ﺭﺍ ﺩﺍﺭﺩ ﻭ ﺑﻪ ﺩﻟﻴﻞ ﺁﻧﮑﻪ ﻳﮏ ﻧﻮﻉ «ﺗﮑﺮﺍﺭ ﮐﺮﺩﻥ» ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ، ﺳﺒﺐ ﻣﺘﻤﺮﮐﺰ ﺷﺪﻥ ﺣﻮﺍﺱ ﻭ ﺑﻪ ﺧﺎﻃﺮ ﺳﭙﺮﺩﻥ ﻣﻄﺎﻟﺐ ﻣﯽ‌ﺷﻮﺩ. ﻫﻤﭽﻨﻴﻦ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺷﻤﺎ ﺍﺯ ﺣﺎﻟﺖ ﻏﻴﺮﻓﻌﺎﻝ ﺩﺭ ﮐﻼﺱ (ﻭ ﻳﺎ ﻫﺮ ﮐﺎﺭﯼ- ﻣﺜﻞ ﻣﻄﺎﻟﻌﻪ) ﺧﺎﺭﺝ ﺷﻮﻳﺪ ﻭ ﺷﺮﻭﻉ ﺑﻪ ﻓﻌﺎﻟﻴﺖ ﻧﻤﺎﻳﻴﺪ.

ﻳﺎﺩﺩﺍﺷﺖ‌ﺑﺮﺩﺍﺭﯼ ﻣﺮﺍﺣﻞ ﻣﺨﺘﻠﻔﯽ ﺩﺍﺭﺩ. ﻣﺎ ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﻪ ﻳﺎﺩﺩﺍﺷﺖ‌ﺑﺮﺩﺍﺭﯼ ﺩﺭ ﮐﻼﺱ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ. ﺍﻳﻦ ﻣﺮﺍﺣﻞ ﻗﺒﻞ ﺍﺯ ﺷﺮﻭﻉ ﮐﻼﺱ ﺩﺭﺳﯽ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ.

۱- ﻣﺮﺣﻠﻪ ﺁﻣﺎﺩﻩ‌ﺳﺎﺯﯼ
ﺟﺰﻭﻩ ﻭ ﮐﺘﺎﺏ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺧﻼﺻﻪ ﺑﺨﻮﺍﻧﻴﺪ ﺗﺎ ﺑﺮﺍﯼ ﻣﻮﺿﻮﻉ ﮐﻼﺱ ﺁﻣﺎﺩﮔﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﻳﺎﺩﺩﺍﺷﺖ‌ﻫﺎﯼ ﻗﺒﻠﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﺟﻠﺴﺎﺕ ﻗﺒﻞ ﺑﺮﺩﺍﺷﺘﻪ‌ﺍﻳﺪ ﻣﺮﻭﺭ ﮐﻨﻴﺪ ﺗﺎ ﺫﻫﻦ ﺷﻤﺎ ﺑﺮﺍﯼ ﻳﺎﺩﮔﻴﺮﯼ ﻣﻄﺎﻟﺐ ﺟﺪﻳﺪ ﺁﻣﺎﺩﻩ ﺷﻮﺩ. ﻣﺮﻭﺭ ﺍﻃﻼﻋﺎﺕ ﮔﺬﺷﺘﻪ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺍﻃﻼﻋﺎﺕ ﺟﺪﻳﺪ، ﺳﺮﻳﻊ‌ﺗﺮ ﻭ ﺑﻬﺘﺮ ﺩﺭ ﻣﻐﺰ ﺷﻤﺎ ﺫﺧﻴﺮﻩ ﺷﻮﺩ.

ﺳﭙﺲ ﺍﻃﻤﻴﻨﺎﻥ ﺣﺎﺻﻞ ﮐﻨﻴﺪ ﮐﻪ ﺗﻤﺎﻡ ﻭﺳﺎﻳﻞ ﻣﻮﺭﺩ ﻧﻴﺎﺯ ﮐﻼﺱ ﺭﺍ ﺑﻪ ﻫﻤﺮﺍﻩ ﺩﺍﺭﻳﺪ ﺗﺎ ﺍﺿﻄﺮﺍﺏ ﺑﯽ‌ﻣﻮﺭﺩ ﺑﻪ ﺷﻤﺎ ﻭﺍﺭﺩ ﻧﺸﻮﺩ. ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﻤﻪ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﺍﻋﻢ ﺍﺯ ﺁﻣﺎﺩﮔﯽ ﺫﻫﻨﯽ ﻭ ﺁﻣﺎﺩﮔﯽ ﻓﻴﺰﻳﮑﯽ ﺑﻪ ﺍﻳﻦ ﻣﻨﻈﻮﺭ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺫﻫﻦ ﺷﻤﺎ ﺭﺍ ﺑﺮﺍﯼ ﻓﺮﺁﻳﻨﺪ ﻳﺎﺩﮔﻴﺮﯼ ﺁﻣﺎﺩﻩ ﺳﺎﺯﺩ ﻭ ﺍﻧﮕﻴﺰﻩ ﻭ ﻋﻼﻗﻪ ﺩﺭ ﻓﮑﺮﺗﺎﻥ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﺩﻫﺪ.

٢- ﻣﺮﺣﻠﻪ ﻳﺎﺩﺩﺍﺷﺖ‌ﺑﺮﺩﺍﺭﯼ
ﭼﻨﺪﻳﻦ ﻧﮑﺘﻪ ﺭﺍ ﻫﻨﮕﺎﻡ ﻳﺎﺩﺩﺍﺷﺖ‌ﺑﺮﺩﺍﺭﯼ ﺑﺎﻳﺪ ﺭﻋﺎﻳﺖ ﮐﻨﻴﺪ. ﺣﺘﻤﺎ ﺩﻗﻴﻖ ﻭ ﺧﻮﺍﻧﺎ ﺑﻨﻮﻳﺴﻴﺪ. ﺳﻌﯽ‌ ﮐﻨﻴﺪ ﮐﻪ ﻧﮑﺎﺕ ﺭﺍ ﻫﻤﺰﻣﺎﻥ ﺑﺎ ﺗﺪﺭﻳﺲ ﻣﻌﻠﻢ، ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﻣﺮﺗﺐ ﺑﻨﻮﻳﺴﻴﺪ ﮐﻪ ﺑﻌﺪﺍ ﺩﺭ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻧﮑﺎﺕ ﻣﺮﺗﺒﻂ ﺩﭼﺎﺭ ﻣﺸﮑﻞ ﻧﺸﻮﻳﺪ. ﻫﻤﻪ ﺍﻃﻼﻋﺎﺕ ﻣﻬﻢ ﺭﺍ ﻳﺎﺩﺩﺍﺷﺖ ﮐﻨﻴﺪ ﻭ ﺩﻗﺖ ﮐﻨﻴﺪ ﮐﻪ ﻧﮑﺘﻪ ﻣﻬﻤﯽ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻧﺪﻫﻴﺪ.

٣- ﺑﺎﺯﺑﻴﻨﯽ ﻳﺎﺩﺩﺍﺷﺖ‌ﻫﺎ
ﺑﻼﻓﺎﺻﻠﻪ ﺑﻌﺪ ﺍﺯ ﺍﺗﻤﺎﻡ ﮐﻼﺱ ﻳﺎﺩﺩﺍﺷﺖ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﻣﻮﺭﺩ ﺑﺎﺯﺑﻴﻨﯽ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ﻣﺸﮑﻼﺕ ﺁﻥ ﺭﺍ ﺭﻓﻊ ﮐﻨﻴﺪ. ﺍﻃﻤﻴﻨﺎﻥ ﺣﺎﺻﻞ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﻳﺎﺩﺩﺍﺷﺖ‌ﻫﺎ ﻗﺎﺑﻞ ﻓﻬﻢ ﻭ ﮐﺎﻣﻞ ﻫﺴﺘﻨﺪ. ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻗﺴﻤﺘﯽ ﺍﺯ ﻳﺎﺩﺩﺍﺷﺖ ﺧﻮﺩ ﺭﺍ ﻣﺘﻮﺟﻪ ﻧﻤﯽ‌ﺷﻮﻳﺪ ﺳﻌﯽ ﮐﻨﻴﺪ ﺑﺎ ﮐﻤﮏ ﺍﺳﺘﺎﺩ ﻭ ﻳﺎ ﭘﺮﺳﺶ ﺍﺯ ﺩﻭﺳﺘﺎﻥ‌ﺗﺎﻥ ﺁﻥ ﺭﺍ ﺭﻓﻊ ﮐﻨﻴﺪ.

۴- ﻃﺮﺡ ﭘﺮﺳﺶ
ﺍﺯ ﺭﻭﯼ ﻳﺎﺩﺩﺍﺷﺖ‌ﻫﺎﯼ ﺧﻮﺩﺗﺎﻥ ﺷﺮﻭﻉ ﺑﻪ ﻃﺮﺡ ﭘﺮﺳﺶ ﮐﻨﻴﺪ. ﭘﺮﺳﺶ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﮐﺎﻣﻞ ﺑﻨﻮﻳﺴﻴﺪ. ﺍﻳﻦ ﭘﺮﺳﺶ‌ﻫﺎ ﺑﻌﺪﺍ ﺑﺮﺍﯼ ﺑﻬﺘﺮ ﻓﻬﻤﻴﺪﻥ ﻣﻄﻠﺐ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﺧﻮﺍﻫﻨﺪ ﮐﺮﺩ. ﺑﻪ ﮐﻴﻔﻴﺖ ﭘﺮﺳﺶ‌ﻫﺎ ﺑﻴﺸﺘﺮ ﺍﺯ ﮐﻤﻴﺖ ﺁﻧﻬﺎ ﺑﻬﺎ ﺩﻫﻴﺪ. ﭘﺮﺳﺶ‌ﻫﺎ ﺭﺍ ﻃﻮﺭﯼ ﻃﺮﺍﺣﯽ ﮐﻨﻴﺪ ﮐﻪ ﺗﻤﺎﻡ ﻣﻄﺎﻟﺐ ﺩﺭﺳﯽ ﺭﺍ ﺩﺭ ﺑﺮ ﺑﮕﻴﺮﺩ.

۵- ﺁﺯﻣﻮﻥ
ﺍﮐﻨﻮﻥ ﻧﻮﺑﺖ ﻓﺮﺳﺘﺎﺩﻥ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺑﻪ ﺣﺎﻓﻈﻪ ﺑﻠﻨﺪﻣﺪﺕ ﺍﺳﺖ. ﺑﺎ ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﭘﺮﺳﺶ‌ﻫﺎﻳﯽ ﮐﻪ ﻃﺮﺡ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺑﻪ ﺫﻫﻦ‌ﺗﺎﻥ ﮐﻤﮏ ﮐﻨﻴﺪ ﺗﺎ ﻣﻄﺎﻟﺐ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺑﺴﭙﺎﺭﺩ.

ﺩﻭ ﺭﻭﺵ ﻣﺘﺪﺍﻭﻝ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﻭﻝ ﺍﻳﻨﮑﻪ ﺍﺯ ﻫﻤﮑﻼﺳﯽ‌ﻫﺎﻳﺘﺎﻥ ﮐﻤﮏ ﺑﮕﻴﺮﻳﺪ. ﻳﺎﺩﺩﺍﺷﺖ‌ﻫﺎ ﺭﺍ ﺑﻪ ﻳﮑﯽ ﺍﺯ ﺁﻧﻬﺎ ﺩﺍﺩﻩ ﻭ ﺍﺯ ﺍﻭ ﺑﺨﻮﺍﻫﻴﺪ ﮐﻪ ﺍﺯ ﺷﻤﺎ ﺳﻮﺍﻝ ﮐﻨﺪ.

ﺩﺭ ﺭﻭﺵ ﺩﻭﻡ ﻳﮏ ﺗﮑﻪ ﮐﺎﻏﺬ ﺭﻭﯼ ﻳﺎﺩﺩﺍﺷﺖ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ ﻭ ﺑﻪ ﭘﺮﺳﺶ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺣﺎﺷﻴﻪ ﻳﺎﺩﺩﺍﺷﺖ ﺧﻮﺩ ﻧﻮﺷﺘﻪ ﺍﻳﺪ ﭘﺎﺳﺦ ﺩﻫﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﭼﻨﺪ ﺑﺎﺭ ﺑﺮﺍﯼ ﻫﺮ ﭘﺮﺳﺶ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﺗﺎ ﻣﻄﻤﺌﻦ ﺷﻮﻳﺪ ﮐﻪ ﺗﺴﻠﻂ ﮐﺎﻣﻞ ﺭﻭﯼ ﻣﻄﻠﺐ ﭘﻴﺪﺍ ﮐﺮﺩﻩ‌ﺍﻳﺪ.

ﻧﮑﺎﺗﯽ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﻳﺎﺩﺩﺍﺷﺖ ﺑﺮﺩﺍﺭﯼ ﺑﻬﺘﺮ

  • ﺍﻧﺘﻬﺎﯼ ﻫﺮ ﺻﻔﺤﻪ ﺭﺍ ﺑﺎ ﻳﮏ ﺟﺎﯼ ﺧﺎﻟﯽ ﺑﺎﻗﯽ ﺑﮕﺬﺍﺭﻳﺪ ﺗﺎ ﺩﺭ ﺻﻮﺭﺕ ﻧﻴﺎﺯ ﺩﺭ ﺁﻳﻨﺪﻩ ﺁﻥ ﺭﺍ ﺑﺎ ﻣﻄﺎﻟﺐ ﺗﮑﻤﻴﻞ ﺷﺪﻩ ﭘﺮ ﮐﻨﻴﺪ. ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﺪﻑ ﺷﻤﺎ ﻳﺎﺩﺩﺍﺷﺖ‌ﺑﺮﺩﺍﺭﯼ ﻣﻔﻴﺪ ﺍﺳﺖ ﻭ ﻧﻪ ﺻﺮﻓﺎ ﺳﻴﺎﻩ ﮐﺮﺩﻥ ﮐﺎﻏﺬ ﻭ ﺗﻠﻒ ﮐﺮﺩﻥ ﻭﻗﺖ.
  • ﺩﺭ ﮐﻼﺱ ﺩﺭﺱ ﺗﺮﺟﻴﺤﺎ ﺩﺭ ﺭﺩﻳﻒ‌ﻫﺎﯼ ﺟﻠﻮ ﺑﻨﺸﻴﻨﻴﺪ ﺯﻳﺮﺍ ﻣﻌﻤﻮﻻ ﺩﺭ ﺭﺩﻳﻒ‌ﻫﺎﯼ ﺟﻠﻮ ﮐﻤﺘﺮ ﺣﻮﺍﺱ‌ﺗﺎﻥ ﭘﺮﺕ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺷﻨﻴﺪﻥ، ﺩﻳﺪﻥ ﻭ ﺗﻮﺟﻪ ﺑﻪ ﻣﻄﺎﻟﺐ ﻣﻬﻢ ﺁﺳﺎﻥ‌ﺗﺮ ﺍﺳﺖ ﻭ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ ﺑﺎ ﺳﻬﻮﻟﺖ ﺑﻴﺸﺘﺮﯼ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ.
  • ﻧﮑﺎﺕ ﻣﻬﻢ ﻳﺎﺩﺩﺍﺷﺖ‌ﻫﺎﻳﺘﺎﻥ ﺭﺍ ﺑﺎ ﺷﮑﻞ، ﺑﺎ ﺧﻂ ﮐﺸﻴﺪﻥ ﺩﻭﺭ ﺁﻧﻬﺎ، ﺑﺎ ﺗﻐﻴﻴﺮ ﻧﺤﻮﻩ ﻧﮕﺎﺭﺵ ﺑﺮﺟﺴﺘﻪ ﮐﻨﻴﺪ ﺗﺎ ﺩﺭ ﻣﻮﻗﻊ ﻧﻴﺎﺯ ﺭﺍﺣﺖ‌ﺗﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺁﻥ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﻴﺪ.
  • ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺩﺭ ﻗﺴﻤﺘﯽ ﺍﺯ ﺩﺭﺱ ﻋﻘﺐ ﺍﻓﺘﺎﺩﻩ‌ﺍﻳﺪ ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﺁﻥ ﻗﺴﻤﺖ ﺭﺍ ﺑﺎ ﻋﺠﻠﻪ ﺑﻨﻮﻳﺴﻴﺪ. ﻗﺴﻤﺘﯽ ﺍﺯ ﺻﻔﺤﻪ ﺭﺍ ﺧﺎﻟﯽ ﮔﺬﺍﺷﺘﻪ ﻭ ﺑﻌﺪ ﺍﺯ ﮐﻼﺱ ﺍﺯ ﻣﻌﻠﻢ ﺧﻮﺩ ﻳﺎ ﺩﻭﺳﺘﺎﻥ‌ﺗﺎﻥ ﺑﺮﺍﯼ ﻧﻮﺷﺘﻦ ﺁﻥ ﮐﻤﮏ ﺑﮕﻴﺮﻳﺪ.
  • ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﺩﻗﻴﻘﺎ ﮐﻠﻤﺎﺕ ﻣﻌﻠﻢ‌ﺗﺎﻥ ﺭﺍ ﺣﻔﻆ ﮐﻨﻴﺪ، ﺳﻌﯽ ﮐﻨﻴﺪ ﺁﻥ ﺭﺍ ﺑﻔﻬﻤﻴﺪ ﻭ ﺳﭙﺲ ﺁﻥ ﺭﺍ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﮐﻠﻤﺎﺕ ﺧﻮﺩﺗﺎﻥ ﺑﻪ ﺧﺎﻃﺮ ﺑﺴﭙﺎﺭﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺭﻭﻧﺪ ﺑﻪ ﺧﺎﻃﺮ ﺳﭙﺮﺩﻥ ﺑﺴﻴﺎﺭ ﺁﺳﺎﻥ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

 

ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ
ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﺗﮑﻨﻴﮏ‌ﻫﺎﯼ ﺑﻬﺒﻮﺩ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ، ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺍﺳﺖ. ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﺑﻪ ﻓﻌﺎﻟﻴﺘﯽ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﻛﻪ ﺍﻧﺴﺎﻥ ﻗﺒﻞ ﺍﺯ ﺁﻧﻜﻪ ﺑﺨﻮﺍﻫﺪ ﻛﺎﺭﯼ ﺭﺍ ﺷﺮﻭﻉ ﻧﻤﺎﻳﺪ، ﺩﻗﻴﻘﺎ ﺑﺪﺍﻧﺪ ﭼﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻜﻨﺪ ﻭ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻫﺪﺍﻓﺶ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﻣﺪﻭﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻟﺬﺍ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﺻﺤﻴﺢ ﻭ ﻋﻠﻤﯽ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﻛﻪ ﺍﻧﺴﺎﻥ ﺑﺎ ﺗﻼﺵ ﻛﻤﺘﺮ ﻭ ﺣﺘﯽ ﺻﺮﻑ ﻭﻗﺖ ﮐﻤﺘﺮ ﻣﻄﺎﻟﺐ ﺑﻴﺸﺘﺮﯼ ﻳﺎﺩ ﺑﮕﻴﺮﺩ، ﮐﻤﺘﺮ ﺧﺴﺘﻪ ﺷﻮﺩ ﻭ ﺣﺘﯽ ﺍﺯ ﺩﺭﺱ ﺧﻮﺍﻧﺪﻥ ﻭ ﻣﻄﺎﻟﻌﻪ ﻟﺬﺕ ﺑﺒﺮﺩ.

ﺩﺭ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ‌ﺭﻳﺰﯼ ﺧﻮﺏ ﻭ ﺩﻗﻴﻖ ﺍﺑﺘﺪﺍ ﺑﺎﻳﺪ ﺍﻭﻟﻮﻳﺖ‌ﻫﺎﯼ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﺭﺍ ﻣﺸﺨﺺ ﻭ ﻭﻗﺖ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﺳﺎﺱ ﺍﻳﻦ ﺍﻭﻟﻮﻳﺖ‌ﻫﺎ ﺗﻘﺴﻴﻢ ﻭ ﺑﺮ ﭘﺎﻳﻪ ﺍﻫﻤﻴﺖ ﺍﻳﻦ ﺍﻭﻟﻮﻳﺖ‌ﻫﺎ، ﻭﻗﺖ ﺧﻮﺩ ﺭﺍ ﺗﻮﺯﻳﻊ ﮐﻨﻴﺪ. ﻳﮏ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﺩﻗﻴﻖ ﺍﻧﺴﺎﻥ ﺭﺍ ﺍﺯ ﺑﻼﺗﻜﻠﻴﻔﯽ ﻭ ﺳﺮﺩﺭﮔﻤﯽ ﺩﻭﺭ ﻣﯽ‌ﮐﻨﺪ. ﻋﺪﻡ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﺻﺤﻴﺢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺯﻣﻴﻨﻪ‌ﺳﺎﺯ ﺍﺿﻄﺮﺍﺏ ﻭ ﺗﺸﻮﻳﺶ ﻧﺎﺷﯽ ﺍﺯ ﻋﺪﻡ ﺍﻧﺠﺎﻡ ﻭﻇﺎﻳﻒ ﺷﻮﺩ ﻭ ﻣﻮﺟﺐ ﺗﺸﺪﻳﺪ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ ﺷﻮﺩ. ﻭﻟﯽ ﺑﺎ ﺍﻧﺠﺎﻡ ﺑﻪ‌ ﻣﻮﻗﻊ ﻭ ﻣﻮﺛﺮ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﺭﻭﺯﺍﻧﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﻣﻮﺟﺐ ﺑﺎﻻ ﺭﻓﺘﻦ ﺑﻬﺪﺍﺷﺖ ﺭﻭﺍﻧﯽ ﺧﻮﺩ ﺷﺪ ﻭ ﺍﺯ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺟﻠﻮﮔﻴﺮﯼ ﮐﺮﺩ.

ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻣﯽ‌ﺩﺍﻧﻴﻢ ﻣﻬﻤﺘﺮﻳﻦ ﻋﻮﺍﻣﻞ ﻳﺎﺩﮔﻴﺮﯼ ﻋﺒﺎﺭﺗﻨﺪ ﺍﺯ:

  • ﺩﺍﺷﺘﻦ ﻫﺪﻑ ﻭ ﺍﻧﮕﻴﺰﻩ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﮐﺎﺭ ﻣﻮﺭﺩ ﻧﻈﺮ
  • ﻓﻬﻤﻴﺪﻥ ﻭ ﺩﺭﮎ ﻣﻄﻠﺐ
  • ﺗﻮﺟﻪ ﮐﺎﻣﻞ ﻭ ﺩﻗﺖ ﮐﺎﻓﯽ
  • ﻫﻤﺖ ﻭ ﭘﺸﺘﻜﺎﺭ ﻭ ﻧﺎﺍﻣﻴﺪ ﻧﺸﺪﻥ
  • ﺍﻣﻴﺪﻭﺍﺭﯼ ﻭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ

ﺑﺮﺍﯼ ﺩﺍﺷﺘﻦ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﻣﻮﻓﻖ ﻭ ﮐﺎﻣﻞ ﻧﮑﺎﺕ ﺯﻳﺮ ﺭﺍ ﺭﻋﺎﻳﺖ ﮐﻨﻴﺪ. ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺑﻪ ﺑﺮﻧﺎﻣﻪ ﺧﻮﺩ ﭘﺎﻳﺒﻨﺪ ﺑﺎﺷﻴﺪ ﺍﻣﮑﺎﻥ ﻣﻮﻓﻘﻴﺖ ﺷﻤﺎ ﺑﻴﺸﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

۱- ﺑﺮﻧﺎﻣﻪ ﺩﺭﺳﯽ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﺳﺎﺱ ﺑﺮﻧﺎﻣﻪ ﻫﻔﺘﮕﯽ ﻛﻼﺱ ﺗﻨﻈﻴﻢ ﻧﻤﺎﺋﻴﺪ. ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﺍﺯ ﮐﻼﺱ ﺟﻠﻮﺗﺮ ﺑﺎﺷﻴﺪ. ﻫﻤﺎﻫﻨﮓ ﺑﺎ ﮐﻼﺱ ﭘﻴﺶ ﺑﺮﻭﻳﺪ.

۲- ﻣﻴﺰﺍﻥ ﺳﺎﻋﺎﺕ ﻣﻄﺎﻟﻌﻪ ﺭﻭﺯﺍﻧﻪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ‌ﺍﯼ ﺑﺎﺷﺪ ﮐﻪ ﺑﺎﻋﺚ ﺧﺴﺘﮕﯽ ﻣﻔﺮﻁ ﺷﻤﺎ ﻧﺸﻮﺩ. ﻫﻨﮕﺎﻡ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ ﺑﺮﺍﯼ ﺗﻌﻴﻴﻦ ﻣﻘﺪﺍﺭ ﻣﻄﺎﻟﻌﻪ ﺭﻭﺯﺍﻧﻪ ﺗﻮﺍﻧﺎﻳﯽ ﺧﻮﺩ ﺭﺍ ﻫﻢ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ. ﺳﺎﻋﺎﺕ ﻣﻄﺎﻟﻌﻪ ﻣﻮﺛﺮ ﻭ ﻣﻔﻴﺪ ﺩﺭ ﻃﯽ ﺷﺒﺎﻧﻪ ﺭﻭﺯ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩ ﻣﺨﺘﻠﻒ ﻣﺘﻔﺎﻭﺕ ﻭ ﺑﺮﺍﺳﺎﺱ ﺗﺠﺮﺑﻪ ﺷﺨﺼﯽ ﻣﺸﺨﺺ ﻣﯽ‌ﺷﻮﺩ. ﻣﺪﺕ ﺯﻣﺎﻥ ﺁﻥ ﻧﺒﺎﻳﺪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ‌ﺍﯼ ﺑﺎﺷﺪ ﮐﻪ ﺍﺯ ﺗﻮﺍﻥ ﺟﺴﻤﯽ ﻭ ﺫﻫﻨﯽ ﺷﻤﺎ ﺑﻴﺸﺘﺮ ﺑﺎﺷﺪ ﭼﻮﻥ ﺑﺎﻋﺚ ﻓﺮﺳﻮﺩﮔﯽ ﺫﻫﻦ‌ﺗﺎﻥ ﺷﺪﻩ ﻭ ﮐﻴﻔﻴﺖ ﻳﺎﺩﮔﻴﺮﯼ ﺭﺍ ﮐﺎﻫﺶ ﻣﯽ‌ﺩﻫﺪ.

۳- ﺑﺮﺍﯼ ﻓﻬﻢ ﺑﻬﺘﺮ ﻭ ﻳﺎﺩﮔﻴﺮﯼ ﺑﻴﺸﺘﺮ ﺭﻭﺍﻧﺨﻮﺍﻧﯽ ﺧﻮﺩ ﺗﻘﻮﻳﺖ ﮐﻨﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﻛﻪ ﻓﻬﻢ ﺟﻤﻼﺕ ﺑﺮﺍﯼ ﺷﻤﺎ ﺁﺳﺎﻥ ﮔﺮﺩﺩ ﻭ ﺳﺮﻋﺖ ﺷﻤﺎ ﺩﺭ ﺧﻮﺍﻧﺪﻥ ﻭ ﻓﻬﻤﻴﺪﻥ ﻣﻄﻠﺐ ﺑﺎﻻ ﺑﺮﻭﺩ.

۴- ﺳﻌﯽ ﮐﻨﻴﺪ ﻫﻨﮕﺎﻡ ﻣﻄﺎﻟﻌﻪ ﺑﻪ ﻃﻮﺭ ﮐﺎﻣﻞ ﺗﻤﺮﮐﺰ ﮐﻨﻴﺪ ﻭ ﻣﻔﻬﻮﻡ ﮐﻠﻤﺎﺕ ﻭ ﺟﻤﻼﺕ ﺭﺍ ﺑﻪ ﺧﻮﺑﯽ ﺑﻔﻬﻤﻴﺪ. ﻫﻴﭽﮕﺎﻩ ﮐﻴﻔﻴﺖ ﻣﻄﺎﻟﻌﻪ ﺭﺍ ﻓﺪﺍﯼ ﮐﻤﻴﺖ ﺁﻥ ﻧﮑﻨﻴﺪ. ﻓﻬﻤﻴﺪﻥ ﻳﮏ ﺻﻔﺤﻪ ﺑﻬﺘﺮ ﺍﺯ ﺧﻮﺍﻧﺪﻥ ﺩﻩ ﺻﻔﺤﻪ ﺑﺪﻭﻥ ﻓﻬﻤﻴﺪﻥ ﻣﻄﻠﺐ ﺁﻥ ﺍﺳﺖ.

۵- ﭼﻨﺪ ﺳﺎﻋﺖ ﺍﺯ ﻫﻔﺘﻪ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻭﻗﺖ ﺁﺯﺍﺩ ﺍﺧﺘﺼﺎﺹ ﺩﻫﻴﺪ ﺗﺎ ﻋﻘﺐ ﻣﺎﻧﺪﮔﯽ‌ﻫﺎ ﺭﺍ ﺟﺒﺮﺍﻥ ﮐﻨﻴﺪ ﻭ ﻫﻤﭽﻨﻴﻦ ﺑﻪ ﻣﺮﻭﺭ ﻗﺴﻤﺖ‌ﻫﺎﻳﯽ ﺑﭙﺮﺩﺍﺯﻳﺪ ﮐﻪ ﺑﻪ ﺧﻮﺑﯽ ﻣﺘﻮﺟﻪ ﻧﺸﺪﻩ‌ﺍﻳﺪ. ﺗﻮﺻﻴﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺳﺎﻋﺎﺗﯽ ﺭﺍ ﺑﻪ ﻃﻮﺭ ﺍﺧﺘﺼﺎﺻﯽ ﺑﻪ ﻣﺮﻭﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﮔﺬﺷﺘﻪ ﺧﻮﺩ ﺍﺧﺘﺼﺎﺹ ﺩﻫﻴﺪ ﻭ ﺩﺭﻭﺱ ﺭﺍ ﺩﻭﺭﻩ ﮐﻨﻴﺪ.

۶- ﻣﻄﺎﻟﻌﻪ ﺩﺭﺳﯽ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺳﻪ ﻗﺴﻤﺖ ﺗﻘﺴﻴﻢ ﮐﻨﻴﺪ: ﻗﺒﻞ ﺍﺯ ﺗﺪﺭﻳﺲ ﻣﻌﻠﻢ، ﻣﺮﻭﺭ ﮐﻠﯽ ﺑﻌﺪ ﺍﺯ ﺍﺗﻤﺎﻡ ﺗﺪﺭﻳﺲ ﺩﺭ ﻫﻤﺎﻥ ﺭﻭﺯ ﻭ ﻣﻄﺎﻟﻌﻪ ﺩﻗﻴﻖ ﻭ ﺑﻴﺸﺘﺮ. ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻣﻄﺎﻟﻌﻪ ﺑﻴﺸﺘﺮ ﻭ ﺩﻗﻴﻖ ﺭﺍ ﺣﺘﻤﺎ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺑﺮﻧﺎﻣﻪ ﺩﺭﺳﯽ ﺭﻭﺯ ﺑﻌﺪ ﺧﻮﺩ ﺍﻧﺘﺨﺎﺏ ﻧﻤﺎﺋﻴﺪ.

۷- ﻣﺤﻴﻄﯽ ﮐﻪ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﻴﺪ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ. ﻣﺤﻴﻂ ﻣﻄﺎﻟﻌﻪ ﺑﺎﻳﺪ ﺁﺭﺍﻡ ﻭ ﺑﺪﻭﺭ ﺍﺯ ﻫﺮﮔﻮﻧﻪ ﻫﻴﺎﻫﻮ ﺑﺎﺷﺪ ﺗﺎ ﻣﻮﺟﺐ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﺷﻤﺎ ﻧﺸﻮﺩ. ﺳﻌﯽ ﮐﻨﻴﺪ ﻣﺤﻴﻂ ﻣﻄﺎﻟﻌﻪ ﺷﻤﺎ ﺛﺎﺑﺖ ﺑﺎﺷﺪ ﺗﺎ ﺍﺯ ﺍﺳﺘﺮﺱ‌ﻫﺎﯼ ﻣﺤﻴﻂ ﺟﺪﻳﺪ ﺩﻭﺭ ﺑﺎﺷﻴﺪ.

ﺗﺎ ﺟﺎﻳﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺮﻧﺎﻣﻪ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺗﻨﻈﻴﻢ ﮐﻨﻴﺪ ﮐﻪ ﺳﺎﻋﺎﺗﯽ ﺍﺯ ﺭﻭﺯ ﺭﺍ ﺩﺭ ﮐﻨﺎﺭ ﺧﺎﻧﻮﺍﺩﻩ ﻭ ﺩﻭﺳﺘﺎﻥ ﺑﮕﺬﺭﺍﻧﻴﺪ. ﺩﺭ ﺻﻮﺭﺕ ﻓﺸﺮﺩﻩ ﺑﻮﺩﻥ ﺑﺮﻧﺎﻣﻪ ﺩﺭﺳﯽ ﺧﻮﺩ ﺳﻌﯽ ﮐﻨﻴﺪ ﺣﺪﺍﻗﻞ ﻫﻨﮕﺎﻡ ﺧﻮﺭﺩﻥ ﻏﺬﺍ ﺩﺭ ﮐﻨﺎﺭ ﺁﻧﻬﺎ ﺑﺎﺷﻴﺪ. ﺩﺭ ﮐﻨﺎﺭ ﺧﺎﻧﻮﺍﺩﻩ ﺑﻮﺩﻥ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻣﻘﺪﺍﺭ ﺯﻳﺎﺩﯼ ﺍﺯ ﺍﺳﺘﺮﺱ ﺷﻤﺎ ﮐﻢ ﮐﻨﺪ.

۹- ﺑﻌﺪ ﺍﺯ ﻏﺬﺍ ﻣﺨﺼﻮﺻﺎ ﺑﻌﺪ ﺍﺯ ﻧﻬﺎﺭ ﺍﺳﺘﺮﺍﺣﺖ ﮐﻨﻴﺪ. ﮐﻴﻔﻴﺖ ﻳﺎﺩﮔﻴﺮﯼ ﺷﻤﺎ ﺑﻌﺪ ﺍﺯ ﺧﻮﺭﺩﻥ ﻏﺬﺍ ﺑﻪ ﻣﻔﺪﺍﺭ ﺯﻳﺎﺩﯼ ﮐﺎﻫﺶ ﻣﯽ‌ﻳﺎﺑﺪ. ﺗﻮﺻﻴﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺣﺪﺍﻗﻞ ﺗﺎ ﻳﮏ ﺳﺎﻋﺖ ﺑﻌﺪ ﺍﺯ ﻧﻬﺎﺭ ﺍﺳﺘﺮﺍﺣﺖ ﮐﻨﻴﺪ ﻭ ﺑﻪ ﻣﻐﺰ ﺧﻮﺩ ﺍﺳﺘﺮﺍﺣﺖ ﺑﺪﻫﻴﺪ.

۱۰- ﺩﺭ ﻫﻨﮕﺎﻡ ﺍﺳﺘﺮﺍﺣﺖ ﺑﻌﺪ ﺍﺯ ﻏﺬﺍ ﻳﺎ ﺍﺳﺘﺮﺍﺣﺖ ﺑﻴﻦ ﻣﻄﺎﻟﻌﻪ ﺟﺪﺍ ﺍﺯ ﺗﻤﺎﺷﺎﯼ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻭ ﮐﺎﺭ ﺑﺎ ﺭﺍﻳﺎﻧﻪ ﻳﺎ ﺍﻧﺠﺎﻡ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺭﺍﻳﺎﻧﻪ‌ﺍﯼ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﻣﺪﺗﯽ ﭼﺸﻢ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺑﺒﻨﺪﻳﺪ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻳﮏ ﻣﻮﺯﻳﮏ ﺁﺭﺍﻡ ﺑﺎ ﺻﺪﺍﯼ ﮐﻢ ﮔﻮﺵ ﺩﻫﻴﺪ.
