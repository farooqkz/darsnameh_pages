ﺩﺭﺱ ۲- ﺣﻮﺍﺱ ﭘﺮﺗﯽ ﻭ ﺭﻭﺵ ﻫﺎﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺁﻥ

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻧﺴﺎﻥ‌ﻫﺎ ﺍﺯ ﻧﺒﻮﺩ ﺗﻤﺮﮐﺰ ﻫﻨﮕﺎﻡ ﻣﻄﺎﻟﻌﻪ ﺭﻧﺞ ﻣﯽ‌ﺑﺮﻧﺪ. ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﻳﮑﯽ ﺍﺯ ﺷﺎﻳﻊ‌ﺗﺮﻳﻦ ﻣﺸﮑﻼﺕ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﻭ ﺩﺍﻧﺸﺠﻮﻳﺎﻥ ﺍﺳﺖ.

ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﻋﻮﺍﻣﻠﯽ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ ﮐﻪ ﺑﺎﻋﺚ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻥ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﻣﯽ‌ﺷﻮﻧﺪ.

 

ﭼﻨﺪﮐﺎﺭﯼ (Multitasking)
ﮐﺴﺎﻧﯽ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﭼﻨﺪ ﮐﺎﺭ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻨﺪ، ﺷﺎﻳﺪ ﺍﺣﺴﺎﺱ ﮐﻨﻨﺪ ﮐﻪ ﮐﺎﺭﻫﺎﻳﺸﺎﻥ ﺭﺍ ﺳﺮﻳﻊ‌ﺗﺮ ﺗﻤﺎﻡ ﻣﯽ‌ﮐﻨﻨﺪ، ﺍﻣﺎ ﺗﻘﺮﻳﺒﺎ ﻫﻤﻴﺸﻪ ﺍﻧﺠﺎﻡ ﭼﻨﺪ ﮐﺎﺭ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﺍﺯ ﺍﻧﺠﺎﻡ ﻳﮏ ﮐﺎﺭﯼ ﺩﺭ ﻳﮏ ﺯﻣﺎﻥ ﺑﻴﺸﺘﺮ ﻃﻮﻝ ﻣﯽ‌‌ﮐﺸﺪ.

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﭼﻨﺪ ﮐﺎﺭ ﺭﺍ ﺑﺎ ﻫﻢ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ، ﺟﺎﺑﻪ‌ﺟﺎﻳﯽ ﻣﻴﺎﻥ ﮐﺎﺭﻫﺎ ﺑﺎﻋﺚ ﻫﺪﺭ ﺭﻓﺘﻦ ﻭﻗﺖ ﻣﯽ‌ﺷﻮﺩ. ﻫﻤﭽﻨﻴﻦ ﺗﻤﺮﮐﺰﻫﺎﯼ ﭘﯽ‌ﺩﺭ‌ﭘﯽ ﻭ ﻣﺠﺪﺩ ﺑﺮ ﺭﻭﯼ ﭼﻨﺪ ﮐﺎﺭ، ﻣﻮﺟﺐ ﺧﺴﺘﮕﯽ ﻣﻐﺰﺗﺎﻥ ﺷﺪﻩ ﻭ ﻣﺸﮑﻞ ﺭﺍ ﺑﻴﺸﺘﺮ ﻣﯽ‌ﮐﻨﺪ.

ﻓﺮﺽ ﮐﻨﻴﺪ ﮐﻪ ﺑﺎﻳﺪ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﻳﮏ ﻣﺴﺌﻠﻪ ﺭﻳﺎﺿﯽ ﺭﺍ ﺣﻞ ﮐﻨﻴﺪ ﻭ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﺑﻪ ﻣﻮﺳﻴﻘﯽ ﮔﻮﺵ ﺩﻫﻴﺪ ﻭ ﺑﺎ ﻫﻤﺴﺮﺗﺎﻥ ﻫﻢ ﺻﺤﺒﺖ ﮐﻨﻴﺪ. ﺑﻪ ﻃﻮﺭ ﻗﻄﻊ، ﻣﺪﺕ ﺯﻣﺎﻧﯽ ﮐﻪ ﻃﻮﻝ ﻣﯽ‌ﮐﺸﺪ ﺗﺎ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺭﺍ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﭘﻴﺶ ﺑﺒﺮﻳﺪ ﺑﻴﺸﺘﺮ ﺍﺯ ﺯﻣﺎﻧﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻃﻮﺭ ﺟﺪﺍﮔﺎﻧﻪ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ.

 

ﺧﺴﺘﮕﯽ
ﺑﺮﺭﺳﯽ‌ﻫﺎﯼ ﺑﺴﻴﺎﺭﯼ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﮐﻢ ﺧﻮﺍﺑﯽ، ﻧﻪ ﺗﻨﻬﺎ ﺗﻤﺮﮐﺰ،‌ ﮐﻪ ﺣﺎﻓﻈﻪ ﮐﻮﺗﺎﻩ ﻣﺪﺕ ﻭ ﺑﺴﻴﺎﺭﯼ ﺩﻳﮕﺮ ﺍﺯ ﮐﺎﺭﮐﺮﺩﻫﺎﯼ ﺫﻫﻨﯽ ﺭﺍ ﻣﺨﺘﻞ ﻣﯽ‌ﮐﻨﺪ. ﻫﻨﮕﺎﻣﯽ ﺍﺯ ﺧﻮﺍﺏ ﻣﺤﺮﻭﻡ ﻣﯽ‌ﺷﻮﻳﺪ ﻳﺎ ﺑﻪ ﻣﻘﺪﺍﺭ ﮐﻤﯽ ﻣﯽ‌ﺧﻮﺍﺑﻴﺪ، ﻣﻴﺰﺍﻥ ﺗﻤﺮﮐﺰ ﻭ ﺗﻮﺟﻪ‌ﺗﺎﻥ ﺍﻓﺖ ﻗﺎﺑﻞ ﺗﻮﺟﻬﯽ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ.

ﮐﺎﺭ ﺯﻳﺎﺩ ﻭ ﻋﺪﻡ ﺍﺳﺘﺮﺍﺣﺖ ﮐﺎﻓﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺷﺪﺕ ﺩﺭ ﻣﻴﺰﺍﻥ ﺗﻮﺍﻥ ﺷﻤﺎ ﺑﺮﺍﯼ ﺗﻤﺮﮐﺰ ﺗﺎﺛﻴﺮ ﺑﮕﺬﺍﺭﺩ.

 

ﻣﻼﻟﺖ ﻭ ﮐﺴﺎﻟﺖ
ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎﯼ ﻣﻼﻝ‌ﺁﻭﺭ ﻭ ﮐﺴﺎﻟﺖ ﺑﺎﺭ ﺗﻮﺍﻧﺎﻳﯽ ﺷﻤﺎ ﺭﺍ ﺑﺮﺍﯼ ﺗﻤﺮﮐﺰ ﮐﺎﻫﺶ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﺳﺎﺩﮔﯽ ﺩﺭ ﻣﻌﺮﺽ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﺪ.

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮐﺎﺭﯼ ﮐﺴﺎﻟﺖ‌ﺑﺎﺭ ﺭﺍ ﮐﻪ ﺑﻪ ﺁﻥ ﻋﻼﻗﻪ‌ﺍﯼ ﻧﺪﺍﺭﻳﺪ ﺭﺍ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ، ﺗﻘﺮﻳﺒﺎ ﻫﺮ ﭼﻴﺰ ﺩﻳﮕﺮﯼ ﺭﺍ ﺑﺮﺍﯼ‌ﺗﺎﻥ ﺟﺬﺍﺏ‌ﺗﺮ ﺍﺯ ﺍﻧﺠﺎﻡ ﺁﻥ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺷﻤﺎ ﺑﻪ ﻃﻮﺭ ﻣﺪﺍﻭﻡ ﺑﻪ ﮐﺎﺭﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﺟﺬﺍﺏ‌ﺗﺮ ﻫﺴﺘﻨﺪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ.

 

ﻣﺰﺍﺣﻤﺖ‌ﻫﺎﯼ ﺍﻟﮑﺘﺮﻭﻧﻴﮏ
ﻭﺭﻭﺩ ﮐﺎﻣﭙﻴﻮﺗﺮ، ﺍﻳﻨﺘﺮﻧﺖ ﻭ ﻣﻮﺑﺎﻳﻞ‌ﻫﺎ ﺑﻪ ﺯﻧﺪﮔﯽ ﺍﻧﺴﺎﻥ ﻣﻮﺟﺐ ﺳﺮﻋﺖ ﺑﺨﺸﻴﺪﻥ ﺑﻪ ﺯﻧﺪﮔﯽ ﻭ ﺁﺳﺎﻥ ﺷﺪﻥ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﮐﺎﺭﻫﺎﯼ ﺭﻭﺯﻣﺮﻩ ﺷﺪﻩ ﺍﺳﺖ. ﻭﻟﯽ ﻣﺸﮑﻼﺗﯽ ﻫﻢ ﻫﻤﺮﺍﻩ ﺧﻮﺩ ﺁﻭﺭﺩﻩ ﮐﻪ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﻳﮑﯽ ﺍﺯ ﺷﺎﻳﻊ ﺗﺮﻳﻦ ﺁﻧﻬﺎﺳﺖ. ﺍﮔﺮ ﺑﻪ ﻃﻮﺭ ﻣﺪﺍﻭﻡ ﺑﻪ ﺩﻧﺒﺎﻝ ﭼﮏ ﮐﺮﺩﻥ ﺍﻳﻤﻴﻞ ﺑﺎﺷﻴﺪ ﻳﺎ ﻫﺮ ﭼﻨﺪ ﺩﻗﻴﻘﻪ ﻳﮏ ﺑﺎﺭ ﻓﻴﺲ‌ﺑﻮﮎ ﻭ ﺗﻮﻳﻴﺘﺮ ﺧﻮﺩ ﺭﺍ ﭼﮏ ﮐﻨﻴﺪ، ﺑﻪ ﺳﺎﺩﮔﯽ ﺣﻮﺍﺱ‌ﺗﺎﻥ ﭘﺮﺕ ﻣﯽ‌ﺷﻮﺩ. ﺍﮔﺮ ﺩﺭ ﺣﺎﻝ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﻫﺴﺘﻴﺪ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﺗﻤﺮﮐﺰ ﺩﺍﺭﺩ، ﻫﺮ ﺑﺎﺭ ﮐﻪ ﺻﺪﺍﻳﯽ ﺍﺯ ﮔﻮﺷﯽ ﺧﻮﺩ ﻣﯽ‌ﺷﻨﻮﻳﺪ، ﺭﺷﺘﻪ ﺍﻓﮑﺎﺭﺗﺎﻥ ﺑﻪ ﺭﺍﺣﺘﯽ ﺍﺯ ﻫﻢ ﮔﺴﻴﺨﺘﻪ ﺧﻮﺍﻫﺪ ﺷﺪ.

 

ﻋﻮﺍﻣﻞ ﺫﻫﻨﯽ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ
ﻋﻮﺍﻣﻞ ﺫﻫﻨﯽ ﺍﻳﺠﺎﺩ ﮐﻨﻨﺪﻩ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺍﻓﮑﺎﺭﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﺭ ﺫﻫﻦ ﺷﻤﺎ ﺟﺮﻳﺎﻥ ﺩﺍﺭﻧﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺗﺎﺛﻴﺮﯼ ﻗﻮﯼ ﺑﺮ ﺭﻭﯼ ﺫﻫﻦ ﻭ ﺍﻓﮑﺎﺭ ﺷﻤﺎ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﻧﺎﮐﺎﻣﯽ‌ﻫﺎ ﻳﺎ ﺷﺎﺩﮐﺎﻣﯽ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﻃﻮﻝ ﺭﻭﺯ ﻳﺎ ﺣﺘﯽ ﺩﺭ ﻃﻮﻝ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﺩﺍﺷﺘﻪ‌ﺍﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺳﺎﺩﮔﯽ ﺫﻫﻦ ﺷﻤﺎ ﺭﺍ ﻣﺸﻐﻮﻝ ﮐﻨﺪ ﻭ ﺑﺎﻋﺚ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﺷﻮﺩ. ﻧﮕﺮﺍﻧﯽ ﻭ ﺩﻟﻮﺍﭘﺴﯽ ﺩﺭ ﻣﻮﺭﺩ ﺁﻳﻨﺪﻩ، ﻣﺸﮑﻼﺕ ﺍﻃﺮﺍﻓﻴﺎﻥ، ﺗﺮﺱ ﻭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻓﮑﺎﺭ ﭘﺮﻳﺸﺎﻥ ﺩﻳﮕﺮ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺳﺎﺩﮔﯽ ﻋﺎﻣﻠﯽ ﺑﺎﺷﺪ ﺑﺮﺍﯼ ﮐﻤﮏ ﺑﻪ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﺷﻤﺎ ﺑﺮ ﺭﻭﯼ ﻣﻮﺿﻮﻋﯽ ﺧﺎﺹ.

 

ﺍﺛﺮﺍﺕ ﺩﺍﺭﻭﻫﺎ ﻭ ﻣﺴﺎﺋﻞ ﭘﺰﺷﮑﯽ
ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﺮ ﺭﻭﯼ ﻗﺪﺭﺕ ﺗﻤﺮﮐﺰ ﺷﻤﺎ ﺗﺎﺛﻴﺮ ﻣﺴﺘﻘﻴﻢ ﻣﯽ‌ﮔﺬﺍﺭﻧﺪ. ﺍﺧﺘﻼﻝ ﺗﻤﺮﮐﺰ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺩﻟﻴﻞ ﺑﻴﻤﺎﺭﯼ‌ﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ ﺍﺧﺘﻼﻝ ﺑﻴﺶ‌ﻓﻌﺎﻟﯽ ﻭ ﮐﻢ‌ﺗﻮﺟﻬﯽ (ADHD)، ﻗﻄﻊ ﺗﻨﻔﺲ ﺣﻴﻦ ﺧﻮﺍﺏ، ﺍﻓﺴﺮﺩﮔﯽ، ﮐﻢ‌ﺧﻮﻧﯽ ﻳﺎ ﺑﻴﻤﺎﺭﯼ ﺗﻴﺮﻭﺋﻴﺪ ﺑﺎﺷﺪ.

ﺩﺍﺭﻭﻫﺎﻳﯽ ﻫﻢ ﮐﻪ ﻣﺼﺮﻑ ﻣﯽ‌ﮐﻨﻴﺪ ﻣﻤﮑﻦ ﺍﺳﺖ ﻋﺎﺭﺿﻪ ﺟﺎﻧﺒﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ ﻭ ﺑﺎﻋﺚ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﺷﻮﻧﺪ. ﺑﺮﺧﯽ ﺍﺯ ﺩﺍﺭﻭﻫﺎ ﻣﺎﻧﻨﺪ ﺩﺍﺭﻭﻫﺎﻳﯽ ﮐﻪ ﺑﺮﺍﯼ ﺩﺭﻣﺎﻥ ﺍﻓﺴﺮﺩﮔﯽ، ﺻﺮﻉ ﻳﺎ ﻋﻔﻮﻧﺖ‌ﻫﺎﯼ ﺁﻧﻔﻠﻮﺁﻧﺰﺍ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺭﻭﻧﺪ،‌ ﻋﻮﺍﺭﺽ ﺟﺎﻧﺒﯽ ﺯﻳﺎﺩﯼ ﺩﺍﺭﻧﺪ ﮐﻪ ﻋﺪﻡ ﺩﻗﺖ ﻭ ﺗﻤﺮﮐﺰ ﻳﮑﯽ ﺍﺯ ﺍﻳﻦ ﻋﻮﺍﺭﺽ ﺍﺳﺖ.

 

ﻭﺭﻭﺩ ﺗﮑﻨﻮﻟﻮﮊﯼ ﺑﻪ ﺯﻧﺪﮔﯽ ﺍﻧﺴﺎﻥ ﻣﻮﺟﺐ ﺗﻐﻴﻴﺮﺍﺕ ﺯﻳﺎﺩﯼ ﺩﺭ ﻧﺤﻮﻩ ﺗﻌﺎﻣﻞ ﺍﻧﺴﺎﻥ ﺑﺎ ﻣﺤﻴﻂ ﺍﻃﺮﺍﻑ ﺷﺪﻩ ﺍﺳﺖ. ﻣﺨﺼﻮﺻﺎ ﺑﺎ ﮔﺴﺘﺮﺵ ﺍﻳﻨﺘﺮﻧﺖ ﻫﺮ ﺭﻭﺯ ﺣﺠﻢ ﺯﻳﺎﺩﯼ ﺍﺯ ﺍﻃﻼﻋﺎﺕ ﺑﻪ ﻣﻐﺰ ﻭﺍﺭﺩ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﭘﺮﺩﺍﺯﺵ ﺩﺍﺭﺩ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺧﻴﻠﯽ ﺍﺯ ﺍﻓﺮﺍﺩ ﺑﺮﺍﯼ ﻋﻘﺐ ﻧﻤﺎﻧﺪﻥ ﺍﺯ ﻋﻠﻢ ﺭﻭﺯ ﻣﻘﺪﺍﺭ ﺯﻣﺎﻥ ﻣﻄﺎﻟﻌﻪ ﺧﻮﺩ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﻣﯽ‌ﺩﻫﻨﺪ، ﺍﻣﺎ ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺑﻪ ﺩﻟﻴﻞ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﻭ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺩﺭ ﻫﻨﮕﺎﻡ ﻣﻄﺎﻟﻌﻪ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﻧﺘﻴﺠﻪ ﺩﻟﺨﻮﺍﻩ ﺧﻮﺩ ﺭﺍ ﺑﮕﻴﺮﻧﺪ.

ﺣﺘﻤﺎ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﺑﺮﺍﻳﺘﺎﻥ ﭘﻴﺶ ﺁﻣﺪﻩ ﺍﺳﺖ ﮐﻪ ﻣﻘﺎﻟﻪ‌ﺍﯼ ﺭﺍ ﭼﻨﺪﺑﺎﺭ ﺧﻮﺍﻧﺪﻩ‌ﺍﻳﺪ ﻭﻟﯽ ﺑﻪ ﺧﺎﻃﺮ ﻣﺸﻐﻮﻝ ﺑﻮﺩﻥ ﺫﻫﻨﺘﺎﻥ ﻣﺘﻮﺟﻪ ﻣﻘﺎﻟﻪ ﻧﺸﺪﻩ‌ﺍﻳﺪ ﻭ ﺑﺪﻭﻥ ﺭﺳﻴﺪﻥ ﺑﻪ ﻧﺘﻴﺠﻪ ﻣﻮﺭﺩ ﻧﻈﺮﺗﺎﻥ ﻓﻘﻂ ﺑﺎﻋﺚ ﮐﺎﻫﺶ ﻳﺎﺩﮔﻴﺮﯼ ﻭ ﺧﺴﺘﻪ ﺷﺪﻥ ﻣﻐﺰﺗﺎﻥ ﺷﺪﻩ‌ﺍﻳﺪ.

ﺩﺭ ﺍﺩﺍﻣﻪ ﺍﺑﺘﺪﺍ ﺑﺎ ﺩﻻﻳﻞ ﺍﻳﺠﺎﺩ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺁﺷﻨﺎ ﻣﯽ‌ﺷﻮﻳﻢ ﻭ ﺳﭙﺲ ﺑﻪ ﺭﺍﻩ‌ﻫﺎﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺁﻥ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ.

 

ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﭼﻴﺴﺖ؟
ﻣﯽ‌ﺗﻮﺍﻥ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺭﺍ ﺍﻳﻨﮕﻮﻧﻪ ﺗﻌﺮﻳﻒ ﮐﺮﺩ: ﻋﺪﻡ ﺗﻮﺍﻧﺎﻳﯽ ﮐﻨﺘﺮﻝ ﻭ ﻫﺪﺍﻳﺖ ﺍﻓﮑﺎﺭ ﺩﺭ ﺟﻬﺖ ﺧﺎﺹ ﺭﺍ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﮔﻮﻳﻨﺪ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻫﺮ ﮔﻮﻧﻪ ﺩﻭﺭ ﺷﺪﻥ ﺍﺯ ﺭﻭﻧﺪ ﻣﻄﺎﻟﻌﻪ (ﻭ ﻳﺎ ﮐﺎﺭﯼ ﮐﻪ ﺑﻨﺎﺳﺖ ﺩﺭ ﺯﻣﺎﻥ ﻣﺸﺨﺼﯽ ﺑﻪ ﭘﺎﻳﺎﻥ ﺑﺮﺳﺪ) ﺑﻪ ﺩﻟﻴﻞ ﺩﺭﮔﻴﺮ ﺷﺪﻥ ﻓﮑﺮ ﺑﺎ ﺍﻓﮑﺎﺭ ﻳﺎ ﺗﺨﻴﻼﺕ ﺩﻳﮕﺮ ﺭﺍ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﮔﻮﻳﻨﺪ.

ﺑﺮﺍﯼ ﻏﻠﺒﻪ ﺑﺮ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﻭ ﺗﻤﺮﮐﺰ، ﺍﻭﻟﻴﻦ ﮐﺎﺭ ﺷﻨﺎﺳﺎﻳﯽ ﻣﻨﺸﺎ ﻭ ﺩﻻﻳﻞ ﺍﻳﺠﺎﺩ ﺁﻥ ﺍﺳﺖ. ﻣﺮﺣﻠﻪ ﺑﻌﺪ ﻳﺎﺩﮔﻴﺮﯼ ﻣﻬﺎﺭﺕ‌ﻫﺎ ﻭ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﺮﺑﻮﻁ ﺑﻪ ﺗﻤﺮﮐﺰ ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﮐﺴﺐ ﺍﻳﻦ ﻣﻬﺎﺭﺕ‌ﻫﺎ ﺍﺣﺘﻴﺎﺡ ﺑﻪ ﺗﻤﺮﻳﻦ ﻭ ﻣﻤﺎﺭﺳﺖ ﺩﺍﺭﺩ. ﮔﺬﺭ ﺯﻣﺎﻥ ﻭ ﺗﻤﺮﻳﻦ ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺗﻮﺍﻧﺎﻳﯽ ﮐﻨﺘﺮﻝ ﺍﻓﮑﺎﺭ ﺧﻮﺩ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﻴﺪ ﻭ ﺗﻤﺮﮐﺰ ﺧﻮﺩ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﺩﻫﻴﺪ.

 

ﺩﻻﻳﻞ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ
ﻋﻮﺍﻣﻞ ﮔﻮﻧﺎﮔﻮﻧﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﻮﺟﺐ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺭﺍ ﺑﻪ ﺩﻭ ﮔﺮﻭﻩ ﺍﺻﻠﯽ ﺗﻘﺴﻴﻢ ﮐﺮﺩ.

- ﻋﻮﺍﻣﻞ ﺑﻴﺮﻭﻧﯽ: ﻋﻮﺍﻣﻞ ﺑﻴﺮﻭﻧﯽ ﻣﺎﻧﻨﺪ ﺳﺮ ﻭ ﺻﺪﺍ، ﮔﺮﻣﺎ ﻳﺎ ﺳﺮﻣﺎﯼ ﻫﻮﺍ، ﻧﻮﺭ ﺷﺪﻳﺪ ﻭ... ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻓﺮﺁﻳﻨﺪ ﺗﻤﺮﮐﺰ ﺭﺍ ﺩﭼﺎﺭ ﺍﺧﺘﻼﻝ ﮐﺮﺩﻩ ﻭ ﺑﺎﻋﺚ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺷﻮﻧﺪ.

- ﻋﻮﺍﻣﻞ ﺩﺭﻭﻧﯽ: ﻋﻮﺍﻣﻞ ﺩﺭﻭﻧﯽ ﺑﻪ ﻋﻮﺍﻣﻠﯽ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻧﺴﺎﻥ ﺍﺳﺖ. ﺍﻳﻦ ﻋﻮﺍﻣﻞ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺟﺴﻤﯽ ﻳﺎ ﺫﻫﻨﯽ ﺑﺎﺷﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﻏﻢ ﻭ ﻏﺼﻪ ﻭ ﺧﻴﺎﻟﺒﺎﻓﯽ ﺑﻪ ﻋﻨﻮﺍﻥ ﻋﻮﺍﻣﻞ ﺩﺭﻭﻧﯽ ﺫﻫﻨﯽ ﻭ ﻫﻤﭽﻨﻴﻦ ﺍﺯ ﺗﺸﻨﮕﯽ ﻭ ﮔﺮﺳﻨﮕﯽ ﺑﻪ ﻋﻨﻮﺍﻥ ﻋﻮﺍﻣﻞ ﺩﺭﻭﻧﯽ ﺟﺴﻤﯽ ﻧﺎﻡ ﺑﺮﺩ.

ﺑﺮﺍﯼ ﻏﻠﺒﻪ ﺑﺮ ﻋﻮﺍﻣﻞ ﺩﺭﻭﻧﯽ ﺑﺎﻳﺪ ﻣﻬﺎﺭﺕ‌ﻫﺎﯼ ﮐﻨﺘﺮﻝ ﻓﮑﺮ ﺧﻮﺩ ﺭﺍ ﮔﺴﺘﺮﺵ ﺩﺍﺩﻩ ﻭ ﺳﻌﯽ ﮐﻨﻴﺪ ﻓﮑﺮ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻣﺤﺪﻭﺩﻩ ﻣﻮﺭﺩ ﻧﻈﺮ ﺧﻮﺩ ﻫﺪﺍﻳﺖ ﮐﻨﻴﺪ.

ﻏﻠﺒﻪ ﺑﺮ ﻋﻮﺍﻣﻞ ﺑﻴﺮﻭﻧﯽ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ‌ﺗﺮ ﺍﺯ ﻏﻠﺒﻪ ﺑﺮ ﻋﻮﺍﻣﻞ ﺩﺭﻭﻧﯽ ﺍﺳﺖ. ﺣﺘﯽ ﺑﺮﺧﯽ ﺍﻓﺮﺍﺩ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺪﻭﻥ ﺗﻮﺟﻪ ﺑﻪ ﻋﻮﺍﻣﻞ ﺑﻴﺮﻭﻧﯽ ﺑﺮ ﺭﻭﯼ ﻣﻮﺿﻮﻉ ﺧﺎﺹ ﺗﻤﺮﮐﺰ ﮐﻨﻨﺪ. (ﻣﺜﻞ ﮐﺴﺎﻧﯽ ﮐﻪ ﺩﺭ ﺍﺗﻮﺑﻮﺱ ﻭ ﻳﺎ ﻣﺮﺍﮐﺰ ﺷﻠﻮﻍ ﺑﻪ ﺳﺎﺩﮔﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻣﻄﺎﻟﻌﻪ ﮐﻨﻨﺪ.) ﻭﻟﯽ ﻣﻌﻤﻮﻻ ﺗﻤﺮﮐﺰ ﺩﺭ ﻣﻘﺎﺑﻞ ﻋﻮﺍﻣﻞ ﺩﺭﻭﻧﯽ ﻣﺜﻞ ﻧﮕﺮﺍﻧﯽ ﻭ ﻧﺎﺭﺍﺣﺘﯽ ﺩﺷﻮﺍﺭ ﺍﺳﺖ.

 

ﭼﮕﻮﻧﻪ ﺑﺎ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﻣﻘﺎﺑﻠﻪ ﮐﻨﻴﻢ؟
ﺍﻭﻟﻴﻦ ﮐﺎﺭ ﺑﺮﺍﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺷﻨﺎﺳﺎﻳﯽ ﻋﺎﻣﻞ ﻭ ﻣﻨﺸﺎ ﺁﻥ ﺍﺳﺖ. ﺍﮔﺮ ﻋﺎﻣﻞ ﺑﻴﺮﻭﻧﯽ ﺑﺎﺷﺪ ﺭﻓﻊ ﺁﻥ ﺁﺳﺎﻥ ﺍﺳﺖ. ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ ﺍﮔﺮ ﻋﺎﻣﻞ ﻋﺪﻡ ﺗﻤﺮﮐﺰ، ﻣﺤﻞ ﺷﻠﻮﻍ، ﻧﻮﺭ ﮐﻢ ﻭ ﻳﺎ ﺳﺮﻣﺎ ﺍﺳﺖ، ﻣﯽ‌ﺗﻮﺍﻥ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺍﻳﻦ‌ﻫﺎ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﺍﺩ ﺗﺎ ﺑﻪ ﻋﺎﻣﻞ ﺑﻴﺮﻭﻧﯽ ﻏﻠﺒﻪ ﮐﺮﺩ.

ﺍﻣﺎ ﺍﮔﺮ ﻋﺎﻣﻞ ﻋﺪﻡ ﺗﻤﺮﮐﺰ، ﺩﺭﻭﻧﯽ ﺑﺎﺷﺪ ﻧﻴﺎﺯ ﺑﻪ ﮐﺴﺐ ﻣﻬﺎﺭﺕ‌ﻫﺎﯼ ﺑﻴﺸﺘﺮﯼ ﺍﺳﺖ. ﺍﺑﺘﺪﺍ ﺑﺎﻳﺪ ﻋﺎﻣﻞ ﺩﺭﻭﻧﯽ ﺑﻪ ﻃﻮﺭ ﺩﻗﻴﻖ ﺷﻨﺎﺳﺎﻳﯽ ﺷﻮﺩ. ﺁﻳﺎ ﻋﻠﺖ ﺁﻥ ﻓﻘﺪﺍﻥ ﻋﻼﻗﻪ ﺑﻪ ﻣﻮﺿﻮﻉ ﻣﻄﺎﻟﻌﻪ ﺍﺳﺖ؟ ﺁﻳﺎ ﺭﻭﺵ ﻣﻄﺎﻟﻌﻪ ﻧﺎﺻﺤﻴﺢ ﺍﺳﺖ؟ ﺁﻳﺎ ﻣﻮﺿﻮﻉ ﺧﺎﺻﯽ ﺑﺎﻋﺚ ﻧﮕﺮﺍﻧﯽ ﻣﺎ ﺷﺪﻩ ﺍﺳﺖ؟

ﺷﻨﺎﺧﺖ ﺩﻗﻴﻖ ﻋﺎﻣﻞ، ﮐﻤﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﮐﻪ ﺑﺘﻮﺍﻧﻴﻢ ﺭﺍﺣﺖ‌ﺗﺮ ﺑﺮﺍﯼ ﺣﻞ ﺁﻥ ﭼﺎﺭﻩ‌ﺍﻧﺪﻳﺸﯽ ﮐﻨﻴﻢ.

ﺭﻭﺵ‌ﻫﺎﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺑﻪ ﺷﺮﺡ ﺯﻳﺮ ﺍﺳﺖ:

۱- ﺍﺧﺘﺼﺎﺹ ﺩﺍﺩﻥ ﺯﻣﺎﻧﯽ ﺑﺮﺍﯼ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﻧﮕﺮﺍﻧﯽ‌ﻫﺎ
ﺯﻣﺎﻥ ﻣﻌﻴﻨﻰ ﺭﺍ ﺩﺭ ﻃﻮﻝ ﺭﻭﺯ، ﺑﺮﺍﻯ ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺭﺑﺎﺭﻩ ﻣﻮﺿﻮﻋﺎﺕ ﻣﺨﺘﻠﻔﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺫﻫﻦ ﺷﻤﺎ ﺭﺍ ﻣﺸﻐﻮﻝ ﮐﻨﺪ ﺍﺧﺘﺼﺎﺹ ﺩﻫﻴﺪ. ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ  ﻧﻈﻢ ﻭ ﺍﻧﻀﺒﺎﻁ ﺩﺭ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﺭﻭﺯﺍﻧﻪ ﺑﺴﻴﺎﺭ ﺍﺭﺯﺷﻤﻨﺪ ﻭ ﻣﻔﻴﺪ ﺍﺳﺖ، ﺍﻳﻦ ﻧﻈﻢ ﺭﺍ ﺑﺎﻳﺪ ﺩﺭ ﺍﻧﺪﻳﺸﻪ‌ﺗﺎﻥ ﻧﻴﺰ ﺣﺎﮐﻢ ﮐﻨﻴﺪ.

ﺍﮔﺮ ﺍﻓﻜﺎﺭ ﻣﺰﺍﺣﻢ ﻭ ﭘﺮﻳﺸﺎﻥ ﺩﺭ ﺯﻣﺎﻧﯽ ﺑﻪ ﺳﺮﺍﻍ‌ﺗﺎﻥ ﺁﻣﺪ ﮐﻪ ﻧﺒﺎﻳﺪ ﺑﻴﺎﻳﺪ  ﻭ ﺑﺎﻋﺚ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﻭ ﻋﺪﻡ ﺗﻤﺮﮐﺰ ﺷﻤﺎ ﺷﻮﺩ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺧﻮﺩ ﺗﻠﻘﻴﻦ ﮐﻨﻴﺪ ﻛﻪ ﺯﻣﺎﻥ ﺍﻧﺪﻳﺸﻴﺪﻥ ﺩﺭﺑﺎﺭﻩ ﺍﻳﻦ ﺍﻓﮑﺎﺭ، ﻓﻼﻥ ﺳﺎﻋﺖ ﺧﺎﺹ ﻭ ﻣﺸﺨﺺ ﺍﺳﺖ. ﭘﺲ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺑﺮﺍﻯ ﻫﺮ ﻛﺎﺭ ﻭ ﻓﻌﺎﻟﻴﺘﻰ ﺯﻣﺎﻥ ﻣﺸﺨﺼﯽ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻪ‌ﺍﻳﺪ، ﺑﺮﺍﯼ ﺗﻔﮑﺮﺍﺕ ﻭ ﻧﮕﺮﺍﻧﯽ‌ﻫﺎﻳﺘﺎﻥ ﺩﺭﺑﺎﺭﻩ ﺁﻳﻨﺪﻩ ﻭ ﻫﻤﭽﻨﻴﻦ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﮔﺬﺷﺘﻪ، ﺯﻣﺎﻧﯽ ﺭﺍ ﻣﺸﺨﺺ ﮐﻨﻴﺪ ﻭ ﺍﺟﺎﺯﻩ‌ ﻧﺪﻫﻴﺪ ﮐﻪ ﺍﻳﻦ ﺍﻓﮑﺎﺭ ﺯﻣﺎﻥ ﺣﺎﻝ ﺷﻤﺎ ﺭﺍ ﻗﺮﺑﺎﻧﯽ ﮐﻨﺪ.

۲- ﻋﻼﻗﻪ ﺩﺍﺷﺘﻦ ﺑﻪ ﻣﻮﺿﻮﻉ ﻣﻄﺎﻟﻌﻪ
ﺍﻳﻦ ﻧﮑﺘﻪ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﻣﻮﺿﻮﻉ ﻣﻄﺎﻟﻌﻪ (ﻳﺎ ﮐﺎﺭﯼ ﮐﻪ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ) ﻋﻼﻗﻪ‌ﻣﻨﺪ ﺑﺎﺷﻴﺪ. ﺣﺘﯽ ﺍﮔﺮ ﻧﺴﺒﺖ ﺑﻪ ﻣﻮﺿﻮﻉ ﻣﻄﺎﻟﻌﻪ ﻋﻼﻗﻪ ﻧﺪﺍﺭﻳﺪ (ﻣﺜﻼ ﻳﮏ ﻣﻮﺿﻮﻉ ﺩﺭﺳﯽ) ﺳﻌﯽ ﮐﻨﻴﺪ ﻧﺴﺒﺖ ﺑﻪ ﺁﻥ ﻣﻮﺿﻮﻉ ﻋﻼﻗﻪ ﭘﻴﺪﺍ ﮐﻨﻴﺪ. ﻋﻼﻗﻪ ﺑﻪ ﻣﻮﺿﻮﻉ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻴﻞ ﺑﻪ ﻓﺮﺍﮔﻴﺮﯼ ﺭﺍ ﺩﺭ ﺷﻤﺎ ﺍﻓﺰﺍﻳﺶ ﺩﺍﺩﻩ ﻭ ﺑﺎﻋﺚ ﺷﻮﺩ ﺗﻤﺮﮐﺰ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺗﻠﻘﻴﻦ ﮐﻨﻴﺪ ﮐﻪ ﺑﻪ ﻣﻮﺿﻮﻉ ﻣﻮﺭﺩ ﻣﻄﺎﻟﻌﻪ ﻋﻼﻗﻪ ﺩﺍﺭﻳﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﺍﺭﺗﺒﺎﻃﯽ ﺑﻴﻦ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﻭ ﻣﻮﺿﻮﻉ ﺩﺭ ﺣﺎﻝ ﻣﻄﺎﻟﻌﻪ ﭘﻴﺪﺍ ﮐﻨﻴﺪ ﺗﺎ ﻋﻼﻗﻪ‌ﻳﺘﺎﻥ ﺑﺮﺍﯼ ﺍﺩﺍﻣﻪ ﺑﻴﺸﺘﺮ ﺷﻮﺩ. ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﻫﺮ ﭼﻪ ﺟﻠﻮﺗﺮ ﺑﺮﻭﻳﺪ ﺍﺷﺘﻴﺎﻕ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﻓﺮﺍﮔﻴﺮﯼ ﭘﻴﺪﺍ ﺧﻮﺍﻫﻴﺪ ﮐﺮﺩ ﻭ ﺗﻤﺮﮐﺰ ﺭﺍﺣﺖ‌ﺗﺮ ﺍﻧﺠﺎﻡ ﺧﻮﺍﻫﺪ ﺷﺪ.

۳- ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﻭ ﺍﻧﮕﻴﺰﻩ
ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺩﻗﻴﻖ ﻭ ﺩﺍﺷﺘﻦ ﺍﻧﮕﻴﺰﻩ ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﺭﺍﻩ‌ﻫﺎﯼ ﺑﺎﻻ ﺑﺮﺩﻥ ﺗﻤﺮﮐﺰ ﻫﻨﮕﺎﻡ ﻣﻄﺎﻟﻌﻪ ﺍﺳﺖ.

ﺩﺭ ﺍﺑﺘﺪﺍ ﻭ ﻗﺒﻞ ﺍﺯ ﻫﺮ ﮐﺎﺭﯼ، ﺑﺮﺍﻯ ﺗﻤﺎﻡ ﻓﻌﺎﻟﻴﺖ‏‌ﻫﺎﻯ ﺩﺭﺳﻰ ﻭ ﻏﻴﺮﺩﺭﺳﻰ، ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﻛﻨﻴﺪ. ﺍﻭﻗﺎﺕ ﺷﺒﺎﻧﻪ‌ﺭﻭﺯ ﺭﺍ ﺩﺭ ﻳﻚ ﺟﺪﻭﻝ ﺯﻣﺎﻥ‏‌ﺑﻨﺪﻯ ﺷﺪﻩ ﻧﻮﺷﺘﻪ ﻭ ﺑﺮﺍﻯ ﻫﺮ ﻛﺎﺭﻯ، ﺯﻣﺎﻥ ﺧﺎﺹ ﻭ ﻣﺪﺕ ﺯﻣﺎﻥ ﻣﻌﻴﻦ ﻣﺸﺨﺺ ﻛﻨﻴﺪ. ﺑﮑﻮﺷﻴﺪ ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﺑﺮﻧﺎﻣﻪ ﺧﻮﺩ ﺭﺍ ﺍﺟﺮﺍ ﮐﻨﻴﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺑﻪ ﺫﻫﻦ ﺧﻮﺩ ﻧﻈﻢ ﺑﺪﻫﻴﺪ.

ﻋﻼﻗﻪ ﺑﻪ ﻣﻮﺿﻮﻉ ﺩﺭ ﺣﺎﻝ ﻣﻈﺎﻟﻌﻪ ﻭ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺩﻗﻴﻖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻮﺟﺐ ﺍﻓﺰﺍﻳﺶ ﺍﻧﮕﻴﺰﻩ ﺷﻤﺎ ﺑﺸﻮﺩ. ﻫﺪﻑ‌ﻫﺎﯼ ﮐﻮﭼﮏ ﺑﺮﺍﯼ ﺧﻮﺩ ﺗﻌﺮﻳﻒ ﮐﻨﻴﺪ ﺗﺎ ﺍﻧﮕﻴﺰﻩ ﺧﻮﺩ ﺭﺍ ﺑﺎﻻ ﺑﺒﺮﻳﺪ. ﺍﻧﮕﻴﺰﻩ ﺑﺎﻻ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺸﮑﻼﺕ ﺷﻤﺎ ﺩﺭ ﺯﻣﻴﻨﻪ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺭﺍ ﺭﻓﻊ ﮐﻨﺪ.

۴- ﻳﺎﺩﺩﺍﺷﺖ ﮐﺮﺩﻥ ﻋﻮﺍﻣﻞ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ
ﻫﻤﻴﺸﻪ ﻳﮏ ﺩﻓﺘﺮﭼﻪ ﻳﺎﺩﺩﺍﺷﺖ ﺑﺮ ﺭﻭﯼ ﻣﻴﺰ ﻣﻄﺎﻟﻌﻪ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ ﻭ ﺍﮔﺮ ﭼﻴﺰﯼ ﺫﻫﻦ ﺷﻤﺎ ﺭﺍ ﺩﺭ ﻫﻨﮕﺎﻡ ﻣﻄﺎﻟﻌﻪ ﻣﺸﻐﻮﻝ ﮐﺮﺩ ﺁﻥ ﺭﺍ ﻳﺎﺩﺩﺍﺷﺖ ﮐﻨﻴﺪ. ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ ﺍﮔﺮ ﺩﺭ ﻣﻮﺭﺩ ﻣﺸﮑﻞ ﻣﺎﻟﯽ ﺧﻮﺩ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﻧﮕﺮﺍﻥ ﺳﻼﻣﺘﯽ ﮐﺴﯽ ﻫﺴﺘﻴﺪ، ﻫﻤﻪ ﺁﻧﻬﺎ ﺭﺍ ﻳﺎﺩﺩﺍﺷﺖ ﮐﻨﻴﺪ. ﺑﻪ ﺧﻮﺩ ﺗﻠﻘﻴﻦ ﮐﻨﻴﺪ ﮐﻪ ﭘﺲ ﺍﺯ ﺍﺗﻤﺎﻡ ﻣﻄﺎﻟﻌﻪ ﻭ ﺩﺭ ﻓﺮﺻﺖ ﺩﻳﮕﺮ ﺑﻪ ﺍﻳﻦ ﻣﺴﺎﺋﻞ ﺧﻮﺍﻫﻴﺪ ﭘﺮﺩﺍﺧﺖ. ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺍﺯ ﻧﮕﺮﺍﻧﯽ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﻣﺴﺎﺋﻞ ﮐﺎﺳﺘﻪ ﺷﻮﺩ ﻭ ﺑﺎ ﺧﻴﺎﻝ ﺭﺍﺣﺖ‌ﺗﺮﯼ ﺑﻪ ﻣﻄﺎﻟﻌﻪ ﺑﭙﺮﺩﺍﺯﻳﺪ.

۵- ﺩﻭﺭ ﮐﺮﺩﻥ ﺗﻔﮑﺮﺍﺕ ﻣﻨﻔﯽ ﻭ ﺗﻼﺵ ﺑﺮﺍﯼ ﺟﺬﺏ ﺍﻓﮑﺎﺭ ﻣﺜﺒﺖ
ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺘﻦ ﺍﻳﻦ ﻧﮑﺘﻪ ﺿﺮﻭﺭﯼ ﺍﺳﺖ ﺗﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﺍﻓﮑﺎﺭ ﻣﻨﻔﯽ ﺑﻬﺎ ﻣﯽ‌ﺩﻫﻴﺪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺩﺍﻡ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺑﻴﺎﻓﺘﻴﺪ! ﭘﻴﺶ ﺍﺯ ﺷﺮﻭﻉ ﻣﻄﺎﻟﻌﻪ ﺍﺯ ﮔﻔﺘﻦ ﻭﺍﮊﻩ‌ﻫﺎﻳﯽ ﻫﻤﭽﻮﻥ: ﻧﻤﯽ‌ﺧﻮﺍﻫﻢ، ﻧﻤﯽ‌ﺩﺍﻧﻢ، ﻧﻤﯽ‌ﺗﻮﺍﻧﻢ ﻭ... ﺑﭙﺮﻫﻴﺰﻳﺪ. ﺗﺮﺱ، ﺑﺪﮔﻤﺎﻧﯽ ﻭ ﺑﯽ‌ﺍﺭﺍﺩﮔﯽ ﺭﺍ ﺍﺯ ﺫﻫﻦ ﺧﻮﺩ ﺩﻭﺭ ﮐﻨﻴﺪ.

ﻫﺮﮔﺰ ﺑﻪ ﺧﻮﺩ ﺍﺟﺎﺯﻩ ﻧﺪﻫﻴﺪ ﮐﻪ ﻫﻴﺠﺎﻥ‌ﻫﺎ ﻭ ﺗﻔﮑﺮﺍﺕ ﻣﻨﻔﯽ، ﺷﻤﺎ ﺭﺍ ﺩﺭ ﺧﻮﺩ ﻏﺮﻕ ﮐﻨﺪ ﻭ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ ﺭﺍ ﺩﺭ ﺷﻤﺎ ﻣﺨﺘﻞ ﮐﻨﺪ. ﭘﻴﻮﺳﺘﻪ ﺑﺎ ﺧﻮﺩ ﺗﮑﺮﺍﺭ ﮐﻨﻴﺪ ﮐﻪ ﻗﺎﺩﺭ ﺑﻪ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ (ﻣﻄﺎﻟﻌﻪ، ﻧﻮﺷﺘﻦ، ﮐﺎﺭ ﮐﺮﺩﻥ ﻭ ...) ﻫﺴﺘﻴﺪ ﻭ ﺑﻪ ﺑﻬﺘﺮﻳﻦ ﻧﺤﻮ ﺍﺯ ﭘﺲ ﺁﻥ ﺑﺮ ﻣﯽ‌ﺁﻳﻴﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﺗﻔﮑﺮ ﻣﺜﺒﺖ ﺭﺍ ﻣﺪﺍﻡ ﺩﺭ ﺧﻮﺩ ﭘﺮﻭﺭﺵ ﺩﻫﻴﺪ. ﮐﺴﯽ ﮐﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺧﻮﺩ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﺩﻫﺪ، ﺍﺣﺴﺎﺱ ﺷﻌﻒ ﻭ ﺷﺎﺩﯼ ﺭﺍ ﺩﺭ ﺧﻮﺩ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺩﺭ ﺣﺎﻟﺖ ﺷﺎﺩﻣﺎﻧﯽ ﻧﻴﺰ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ ﺑﻬﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.
