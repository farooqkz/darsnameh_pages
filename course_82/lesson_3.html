ﺩﺭﺱ ۳- ﺧﻴﺎﻝ ﺑﺎﻓﯽ ﻭ ﺭﻭﺵ ﻫﺎﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺁﻥ

ﻳﮑﯽ ﺍﺯ ﻣﺸﮑﻼﺕ ﻣﻬﻢ ﻭ ﺍﺳﺎﺳﯽ ﻫﻨﮕﺎﻡ ﻣﻄﺎﻟﻌﻪ ﻳﺎ ﻫﻨﮕﺎﻡ ﺣﻀﻮﺭ ﺩﺭ ﮐﻼﺱ (ﻭ ﻳﺎ ﻫﺮ ﮐﺎﺭ ﺩﻳﮕﺮ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﺗﻤﺮﮐﺰ ﺩﺍﺭﺩ) ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺍﺳﺖ. ﺣﺘﻤﺎ ﺑﺮﺍﯼ ﺷﻤﺎ ﻫﻢ ﭘﻴﺶ ﺁﻣﺪﻩ ﮐﻪ ﺩﺭ ﮐﻼﺱ ﺩﺭﺱ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺗﻮﺟﻪ‌ﺗﺎﻥ ﺩﺭ ﺟﺎﯼ ﺩﻳﮕﺮﯼ ﺳﻴﺮ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺗﺨﻴﻼﺕ ﺷﺨﺼﯽ ﺍﺟﺎﺯﻩ ﺩﻧﺒﺎﻝ ﮐﺮﺩﻥ ﻣﻄﺎﻟﺐ ﺩﺍﺧﻞ ﮐﻼﺱ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﻧﻤﯽ‌ﺩﻫﺪ.

ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻳﮏ ﻣﺸﮑﻞ ﻫﻤﻴﺸﮕﯽ ﺑﺎﺷﺪ ﺑﺎﻋﺚ ﻣﺸﮑﻼﺕ ﺯﻳﺎﺩﯼ ﺩﺭ ﺭﻭﻧﺪ ﻳﺎﺩﮔﻴﺮﯼ ﻓﺮﺩ ﻣﯽ‌ﺷﻮﺩ. ﺑﺨﺼﻮﺹ ﺩﺭ ﻣﻴﺎﻥ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺑﺎﻋﺚ ﺍﻓﺖ ﺗﺤﺼﻴﻠﯽ ﺷﺪﻩ ﻭ ﺑﺎﻋﺚ ﺗﻠﻒ ﺷﺪﻥ ﺍﺳﺘﻌﺪﺍﺩﻫﺎﯼ ﺍﻧﺴﺎﻥ ﻣﯽ‌ﺷﻮﺩ.

ﺩﺭ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺗﻮﺟﻪ ﻓﺮﺩ ﺧﻴﺎﻟﺒﺎﻑ ﺍﺯ ﺍﻃﺮﺍﻑ ﺧﻮﺩ ﭘﺮﺕ ﺷﺪﻩ ﻭ ﻣﻌﻄﻮﻑ ﺑﻪ ﺍﻓﮑﺎﺭ ﺑﺮﺁﻣﺪﻩ ﺍﺯ ﺗﺼﻮﺭﺍﺕ ﻭ ﺣﺎﻓﻈﻪ ﻣﯽ‌ﺷﻮﺩ. ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﻣﻌﻤﻮﻻ ﺩﺭ ﻧﻮﺟﻮﺍﻧﺎﻥ ﻓﺮﺍﮔﻴﺮﺗﺮ ﺍﺯ ﺩﻳﮕﺮ ﮔﺮﻭﻩ‌ﻫﺎﯼ ﺳﻨﯽ ﺍﺳﺖ. ﺩﻟﻴﻞ ﺁﻥ ﻫﻢ ﺩﺭﮔﻴﺮﯼ‌ﻫﺎﯼ ﺍﺣﺴﺎﺳﯽ ﺍﺳﺖ ﮐﻪ ﻧﻮﺟﻮﺍﻧﺎﻥ ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺑﺎ ﺁﻥ ﺑﺮﺧﻮﺭﺩ ﻣﯽ‌ﮐﻨﻨﺪ. ﻣﻌﻤﻮﻻ ﺍﻳﻦ ﻣﺸﮑﻞ ﺩﺭ ﮐﻼﺱ ﺩﺭﺱ ﻣﺸﻬﻮﺩﺗﺮ ﺍﺳﺖ.

ﻫﺮ ﭼﻨﺪ ﮐﻪ ﻣﺨﺎﻃﺒﺎﻥ ﺩﺭﺳﻨﺎﻣﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺯ ﮔﺮﻭﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺳﻨﯽ ﺑﺎﺷﻨﺪ، ﺍﻣﺎ ﻣﺎ ﺑﺮﺍﯼ ﻧﻤﻮﻧﻪ، ﮔﺮﻭﻫﯽ ﺍﺯ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻭ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺑﻪ ﻋﻠﺖ‌ﻫﺎﯼ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺭ ﮐﻼﺱ ﺩﺭﺱ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ. ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﻫﺮ ﺟﺎﻳﯽ (ﺍﺯ ﻣﻄﺎﻟﻌﻪ ﮔﺮﻓﺘﻪ ﺗﺎ ﮐﺎﺭ ﻳﮏ ﻣﮑﺎﻧﻴﮏ ﺑﺮ ﺭﻭﯼ ﺧﻮﺩﺭﻭ) ﺻﻮﺭﺕ ﺑﮕﻴﺮﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻣﯽ‌ﺗﻮﺍﻥ ﺍﻳﻦ ﻋﻠﻞ ﺭﺍ ﺑﻪ ﮐﺎﺭﻫﺎﯼ ﺩﻳﮕﺮ ﻧﻴﺰ ﺑﺴﻂ ﺩﺍﺩ.

 

ﻋﻠﻞ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺭ ﮐﻼﺱ ﺩﺭﺱ
- ﺭﻓﺘﺎﺭ ﻭ ﺷﺨﺼﻴﺖ ﻣﻌﻠﻢ ﺩﺭ ﮐﻼﺱ ﻭ ﺭﻭﺵ ﺗﺪﺭﻳﺲ ﺍﻭ: ﻧﻮﻉ ﺑﺮﺧﻮﺭﺩ ﻣﻌﻠﻢ ﺑﺎ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ، ﻧﺤﻮﻩ ﺍﺩﺍﺭﻩ ﮐﻼﺱ ﻭ ﺗﺪﺭﻳﺲ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﺎﺛﻴﺮ ﺯﻳﺎﺩﯼ ﺩﺭ ﺧﻴﺎﻟﺒﺎﻑ ﺑﻮﺩﻥ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻳﮑﻨﻮﺍﺧﺘﯽ ﻭ ﺧﺸﮏ ﺑﻮﺩﻥ ﮐﻼﺱ، ﺩﺭﺻﺪ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺭﺍ ﺑﻪ ﻧﺤﻮ ﻗﺎﺑﻞ ﻣﻼﺣﻈﻪ‌ﺍﯼ ﺍﻓﺰﺍﻳﺶ ﻣﯽ‌ﺩﻫﺪ.

- ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﺷﺨﺼﻴﺘﯽ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﻭ ﻣﺸﮑﻼﺕ ﻭ ﺩﺭﮔﻴﺮﯼ‌ﻫﺎﯼ ﺫﻫﻨﯽ ﺍﻭ: ﺗﻌﺪﺍﺩ ﻗﺎﺑﻞ ﺗﻮﺟﻬﯽ ﺍﺯ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻧﯽ ﮐﻪ ﺩﭼﺎﺭ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﻫﺴﺘﻨﺪ ﮐﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﭼﺎﺭ ﻣﻌﻀﻼﺕ ﺷﺨﺼﯽ ﺩﺭ ﺭﻭﺍﺑﻂ ﺍﺣﺴﺎﺳﯽ ﺧﻮﺩ ﻫﺴﺘﻨﺪ. ﺍﻳﻦ ﺭﻭﺍﺑﻂ ﺍﺣﺴﺎﺳﯽ ﻗﺪﺭﺕ ﺑﺎﻻﻳﯽ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﮔﺮﻓﺘﻦ ﺗﻤﺎﻡ ﻇﺮﻓﻴﺖ‌ﻫﺎﯼ ﻓﮑﺮﯼ ﺍﻧﺴﺎﻥ ﺩﺍﺭﻧﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﺭﺍﺣﺘﯽ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺭﺍ ﺍﺯ ﻣﺤﻴﻂ ﺩﺭﺳﯽ ﻭ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﻳﺎﺩﮔﻴﺮﯼ ﻋﻠﻴﺮﻏﻢ ﺣﻀﻮﺭ ﺩﺭ ﮐﻼﺱ ﺩﻭﺭ ﮐﻨﻨﺪ.

- ﻓﻀﺎﯼ ﮐﻼﺱ ﺩﺭﺱ: ﻧﻮﻉ ﭼﻴﺪﻣﺎﻥ ﻭ ﺁﺭﺍﻳﺶ ﮐﻼﺱ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﺎﺛﻴﺮ ﺑﺴﺰﺍﻳﯽ ﺩﺭ ﮐﺎﻫﺶ ﻳﺎ ﺍﻓﺰﺍﻳﺶ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺭ ﻣﻴﺎﻥ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻋﮑﺲ‌ﻫﺎﯼ ﺭﻭﯼ ﺩﻳﻮﺍﺭ ﮐﻼﺱ، ﺩﺳﺖ‌ﻧﻮﺷﺘﻪ‌ﻫﺎ ﻭ ﺳﺎﻳﺮ ﻟﻮﺍﺯﻡ ﮐﻼﺱ ﺑﺎﻳﺪ ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﺩﺭ ﻣﻮﺭﺩ ﺩﺭﺱ ﺑﺎﺷﺪ ﺗﺎ ﺣﻮﺍﺱ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺭﺍ ﭘﺮﺕ ﻧﮑﻨﺪ.

 

ﺭﺍﻩ‌ﻫﺎﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺭ ﮐﻼﺱ
ﺑﺨﺸﯽ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺭ ﮐﻼﺱ ﺩﺭﺱ ﺑﺮ ﻋﻬﺪﻩ ﻣﻌﻠﻢ ﺍﺳﺖ. ﻣﺎ ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﻪ ﻃﻮﺭ ﮔﺬﺭﺍ ﺑﻪ ﺍﻳﻦ ﻧﮑﺎﺕ ﺍﺷﺎﺭﻩ ﻣﯽ‌ﮐﻨﻴﻢ ﺍﻣﺎ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺭﻭﺵ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺭ ﻫﺮ ﮐﺎﺭﯼ ﭘﻴﺪﺍ ﮐﻨﻴﺪ. ﺑﻪ ﻃﻮﺭ ﻧﻤﻮﻧﻪ، ﺍﮔﺮ ﻓﻀﺎﯼ ﮐﻼﺱ ﺩﺭﺱ ﺑﺮﺍﯼ ﻳﮏ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﻣﻨﺎﺳﺐ ﻧﻴﺴﺖ ﻭ ﺷﻤﺎ ﻳﮏ ﻣﮑﺎﻧﻴﮏ ﻫﺴﺘﻴﺪ ﻭ ﮔﺎﻫﯽ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺑﻪ ﺩﻟﻴﻞ ﭼﻴﺪﻣﺎﻥ ﺑﺨﺸﯽ ﺍﺯ ﻣﺤﻞ ﻣﮑﺎﻧﻴﮑﯽ ﺷﻤﺎ ﺑﻪ ﺳﺮﺍﻍ‌ﺗﺎﻥ ﻣﯽ‌ﺁﻳﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻓﻀﺎﯼ ﻣﺤﻞ ﮐﺎﺭ ﺧﻮﺩ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ.

ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﺑﻪ ﭼﻨﺪ ﺭﻭﺵ ﮐﻪ ﺑﺎﻳﺪ ﺍﺯ ﺳﻮﯼ ﻣﻌﻠﻢ ﻫﺎ ﺑﻪ ﮐﺎﺭ ﮔﺮﻓﺘﻪ ﺷﻮﺩ ﺗﺎ ﺑﺎﻋﺚ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺷﻮﺩ ﻭ ﻣﻮﺟﺐ ﺗﻤﺮﮐﺰ ﺣﻮﺍﺱ ﺁﻧﻬﺎ ﺷﻮﺩ، ﺍﺷﺎﺭﻩ ﻣﯽ‌ﮐﻨﻴﻢ.

۱- ﺭﻭﺵ‌ﻫﺎﯼ ﺗﺪﺭﻳﺲ ﻣﺨﺘﻠﻒ: ﺩﺭﺑﺎﺭﻩ ﺭﻭﺵ ﺗﺪﺭﻳﺲ ﭼﻨﺪ ﻧﮑﺘﻪ ﻗﺎﺑﻞ ﺫﮐﺮ ﺍﺳﺖ ﮐﻪ ﻋﺒﺎﺭﺗﻨﺪ ﺍﺯ:

ﻫﺮ ﺭﻭﺵ ﺗﺪﺭﻳﺲ ﺑﺮﺍﯼ ﻧﻮﻉ ﻣﻌﻴﻨﯽ ﺍﺯ ﺩﺍﻧﺶ ﻣﻨﺎﺳﺐ ﺍﺳﺖ. ﭘﺲ ﻣﻌﻠﻢ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﻣﻮﺿﻮﻉ ﺩﺭﺳﯽ ﺑﺎﻳﺪ ﺭﻭﺵ ﺗﺪﺭﻳﺲ ﺧﻮﺩ ﺭﺍ ﺗﻌﻴﻴﻦ ﮐﻨﺪ.

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻳﮏ ﻧﻮﻉ ﺭﻭﺵ ﺗﺪﺭﻳﺲ ﺑﺮﺍﯼ ﻣﺪﺕ ﻃﻮﻻﻧﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺴﻴﺎﺭ ﺧﺴﺘﻪ‌ﮐﻨﻨﺪﻩ ﺑﺎﺷﺪ ﻭ ﺑﺎﻋﺚ ﺍﻓﺰﺍﻳﺶ ﺍﺣﺘﻤﺎﻝ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺷﺪﻩ ﻭ ﺗﻤﺮﮐﺰ ﺍﻭ ﺭﺍ ﺍﺯ ﺑﻴﻦ ﺑﺒﺮﺩ.

ﺭﻭﺵ ﺗﺪﺭﻳﺲ ﺑﺎﻳﺪ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺍﻧﺘﺨﺎﺏ ﺷﻮﺩ ﮐﻪ ﻣﻮﺟﺐ ﻧﺸﺎﻁ ﮐﻼﺱ ﺑﻮﺩﻩ ﻭ ﺑﺘﻮﺍﻧﺪ ﻣﻬﺎﺭﺕ‌ﻫﺎ، ﻣﻴﺰﺍﻥ ﻳﺎﺩﮔﻴﺮﯼ ﻭ ﺁﮔﺎﻫﯽ ﺭﺍ ﺩﺭ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺑﺎﻻ ﺑﺒﺮﺩ.

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﺗﺪﺭﻳﺲ ﺑﻪ ﺻﻮﺭﺕ ﺗﺮﮐﻴﺒﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﻳﮑﻨﻮﺍﺧﺘﯽ ﻭ ﮐﺴﻞ ﮐﻨﻨﺪﻩ ﺑﻮﺩﻥ ﮐﻼﺱ ﺑﮑﺎﻫﺪ، ﺗﻨﻮﻉ ﻭ ﺗﺎﺯﮔﯽ ﺩﺭ ﮐﻼﺱ ﺩﺭﺱ ﺍﻳﺠﺎﺩ ﮐﻨﺪ ﻭ ﺳﺒﺐ ﺟﻠﺐ ﺗﻮﺟﻪ ﻭ ﻫﻤﮑﺎﺭﯼ ﺑﻬﺘﺮ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺑﺸﻮﺩ.

٢- ﭘﻴﻮﻧﺪ ﻣﻄﺎﻟﺐ ﺟﺪﻳﺪ ﺑﺎﺗﺠﺮﺑﻪ‌ﻫﺎﯼ ﻭﺍﻗﻌﯽ: ﻳﮑﯽ ﺍﺯ ﺭﺍﻩ‌ﻫﺎﯼ ﺟﻠﺐ ﺗﻮﺟﻪ ﻭ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ  ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﻌﻠﻢ ﺩﺭﻭﺱ ﺭﺍ ﺑﺎ ﺗﺠﺮﺑﻪ‌ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﺯﻧﺪﮔﯽ ﻣﺮﺗﺒﻂ ﺳﺎﺯﺩ ﺗﺎ ﻣﻮﺿﻮﻉ ﺩﺭﺱ ﺑﺮﺍﯼ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺟﺎﻟﺐ ﺟﻠﻮﻩ ﮐﻨﺪ.

٣- ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﺤﺮﮎ ﻫﺎﯼ ﻣﺘﻨﻮﻉ: ﻣﻨﻈﻮﺭ ﺍﺯ ﻣﺤﺮﮎ ﺩﺭ ﺍﻳﻨﺠﺎ ﺣﺮﮐﺎﺕ ﺑﺪﻥ ﻣﻌﻠﻢ ﻭ ﺗﻦ ﺻﺪﺍﯼ ﺍﻭﺳﺖ. ﻣﻌﻠﻢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﻫﻨﮕﺎﻡ ﺗﺪﺭﻳﺲ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ ﺟﻨﺒﺎﻧﺪﻥ ﺳﺮ، ﺗﮑﺎﻥ ﺩﺍﺩﻥ ﺳﺮ، ﺣﺮﮐﺎﺕ ﺩﺳﺖ، ﺩﺳﺖ ﺑﻪ ﮐﻤﺮ ﮔﺬﺍﺷﺘﻦ ﻭ... ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﺪ ﺗﺎ ﺑﺘﻮﺍﻧﺪ ﻣﻮﺟﺐ ﺍﺳﺘﻤﺮﺍﺭ ﺗﻮﺟﻪ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺑﻪ ﮐﻼﺱ ﺷﻮﺩ.

ﺗﻐﻴﻴﺮ ﺗﻦ ﺻﺪﺍ ﻫﻢ ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﺍﺳﺖ ﮐﻪ ﻣﻮﺟﺐ ﻣﯽ‌ﺷﻮﺩ ﺗﻮﺟﻪ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺑﻪ ﮐﻼﺱ ﻭ ﻣﻄﺎﻟﺐ ﺩﺭﺳﯽ ﺟﻠﺐ ﺷﻮﺩ ﻭ ﻓﺮﺻﺖ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﻭ ﺍﻓﮑﺎﺭ ﺭﻭﻳﺎﻳﯽ ﺭﺍ ﺍﺯ ﺁﻧﻬﺎ ﺑﮕﻴﺮﺩ.

۴- ﺑﺨﺶ ﭘﺮﺳﺶ ﻭ ﭘﺎﺳﺦ: ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﻣﻌﻠﻢ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺑﺎ ﻃﺮﺡ ﺳﻮﺍﻝ ﻓﻌﺎﻟﻴﺖ ﺫﻫﻨﯽ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺭﺍ ﺑﻪ ﻳﮏ ﻣﺴﻴﺮ ﺩﺭﺳﺖ ﻭ ﻫﻢ ﺟﻬﺖ ﺑﺎ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﮐﻼﺳﯽ ﻫﺪﺍﻳﺖ ﮐﻨﺪ. ﻗﺎﺑﻞ ﺫﮐﺮ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺭﻭﺵ ﭘﺮﺳﺶ ﺧﻮﺏ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ ﺯﻳﺮﺍ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺭﺍ ﺑﻪ ﺗﻮﺟﻪ ﻭ ﻳﺎﺩﮔﻴﺮﯼ ﻣﻮﺛﺮ ﺭﺍﻫﻨﻤﺎﻳﯽ ﮐﻨﺪ. ﺩﻭ ﺭﻭﺵ ﺑﺮﺍﯼ ﻃﺮﺡ ﺳﻮﺍﻝ ﻭﺟﻮﺩ ﺩﺍﺭﺩ:

  • ﺳﻮﺍﻝ ﺍﺯ ﻳﮏ ﻧﻔﺮ: ﺍﮔﺮ ﻣﻌﻠﻢ ﺣﺲ ﮐﻨﺪ ﮐﻪ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﯼ ﺑﻪ ﻃﻮﺭ ﺧﺎﺹ ﺩﺭﮔﻴﺮ ﺧﻴﺎﻻﺕ ﺧﻮﺩﺵ ﺍﺳﺖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﻃﺮﺡ ﭘﺮﺳﺸﯽ ﺍﺯ ﺍﻭ، ﺑﺎﻋﺚ ﺑﺮﮔﺸﺘﻦ ﺫﻫﻦ ﺍﻭ ﺑﻪ ﮐﻼﺱ ﺷﻮﺩ.
  • ﺳﻮﺍﻝ ﺍﺯ ﮐﻼﺱ: ﺍﮔﺮ ﻣﻌﻠﻢ ﺣﺲ ﮐﻨﺪ ﻓﻀﺎﯼ ﮐﻠﯽ ﮐﻼﺱ ﻃﻮﺭﯼ ﺍﺳﺖ ﮐﻪ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺗﻤﺮﮐﺰ ﻻﺯﻡ ﺭﺍ ﻧﺪﺍﺭﻧﺪ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﻳﮏ ﭘﺮﺳﺶ ﻧﺎﮔﻬﺎﻧﯽ ﻭ ﻏﺎﻓﻠﮕﻴﺮ ﮐﻨﻨﺪﻩ، ﺣﻮﺍﺱ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺭﺍ ﺑﻪ ﻣﺒﺤﺚ ﻣﻮﺭﺩ ﺟﻤﻊ ﮐﻨﺪ. ﻣﺜﻼ ﻓﺮﺽ ﮐﻨﻴﺪ ﻣﻌﻠﻢ ﺗﺎﺭﻳﺦ، ﺩﺭ ﻣﻮﺭﺩ ﻳﮏ ﻣﺒﺤﺜﯽ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻫﺮﺍﻡ ﺛﻼﺛﻪ ﻭ ﺷﻴﻮﻩ ﺳﺎﺧﺖ ﺁﻥ ﺩﺭ ﻣﺼﺮ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﺪ ﺍﻣﺎ ﭼﻮﻥ ﻣﻮﺿﻮﻉ ﻳﮏ ﻣﻘﺪﺍﺭ ﺧﺸﮏ ﺍﺳﺖ، ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﺪ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺗﻤﺮﮐﺰ ﻻﺯﻡ ﺭﺍ ﻧﺪﺍﺭﻧﺪ. ﺑﻌﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯﺍﻥ ﺑﭙﺮﺳﺪ: ﺁﻳﺎ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﮐﻪ ﺍﻫﺮﺍﻡ ﺛﻼﺛﻪ ﻳﮑﯽ ﺍﺯ ﻋﺠﻴﺐ‌ﺗﺮﻳﻦ ﺳﺎﺧﺖ‌ﻫﺎﯼ ﺑﺸﺮ ﺩﻭﺭﺍﻥ ﺑﺎﺳﺘﺎﻥ ﺍﺳﺖ؟ ﻣﺜﻼ ﺍﻳﻨﮑﻪ ﻣﻮﻗﻌﻴﺖ ﺧﺎﺹ ﺳﺘﺎﺭﻩ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺣﺪﻭﺩ ۳۰۰۰ ﺳﺎﻝ ﭘﻴﺶ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ؟ ﻭ ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﮔﺮ ﮔﻮﺷﺖ ﺭﺍ ﺩﺭ ﺩﺍﺧﻞ ﻫﺮﻡ‌ﻫﺎﻳﯽ ﺑﺎ ﺍﺑﻌﺎﺩ ﺍﻫﺮﺍﻡ ﻣﺼﺮ ﻭ ﻳﺎ ﻣﺘﻨﺎﺳﺐ ﺑﺎ ﺁﻧﻬﺎ ﻗﺮﺍﺭ ﺩﻫﻴﻢ، ﺑﺎ ﻭﺟﻮﺩ ﺁﻧﮑﻪ ﺩﻭ ﺳﻮﻡ ﺍﺯ ﺁﺏ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻣﯽ ﺩﻫﺪ ﻫﺮﮔﺰ ﻓﺎﺳﺪ ﻧﻤﯽ ﺷﻮﺩ. ﺑﻪ ﻧﻈﺮ ﺷﻤﺎ ﺩﻟﻴﻞ ﻋﺠﺎﻳﺐ ﺍﻳﻦ ﺍﻫﺮﺍﻡ ﭼﻴﺴﺖ؟

٥- ﺗﺸﻮﻳﻖ: ﺗﺸﻮﻳﻖ ﻳﮑﯽ ﺍﺯ ﺭﺍﻩ‌ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﻣﻮﺛﺮ ﺑﺮﺍﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺧﻴﺎﻝ‌ﺑﺎﻓﯽ ﻭ ﺣﻮﺍﺱ‌ﭘﺮﺗﯽ ﺍﺳﺖ ﮐﻪ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺩﺍﻧﺶ‌ﺁﻣﻮﺯ ﺳﻌﯽ ﻭ ﺗﻼﺵ ﺑﻴﺸﺘﺮﯼ ﺩﺭ ﻳﺎﺩﮔﻴﺮﯼ ﺍﺯ ﺧﻮﺩ ﻧﺸﺎﻥ ﺩﻫﺪ ﻭ ﺩﺭ ﺑﺤﺚ‌ﻫﺎﯼ ﮐﻼﺳﯽ ﺑﻪ ﺷﮑﻞ ﻣﻮﺛﺮﺗﺮﯼ ﺷﺮﮐﺖ ﮐﻨﺪ.

ﻧﻮﻉ ﺗﺸﻮﻳﻖ ﺑﺮﺍﺳﺎﺱ ﺭﻭﻧﺪ ﮐﻼﺱ ﻭ ﺗﺠﺮﺑﻪ ﻣﻌﻠﻢ ﻣﺸﺨﺺ ﻣﯽ‌ﺷﻮﺩ. ﺗﺸﻮﻳﻖ ﺑﺎﻳﺪ ﻃﻮﺭﯼ ﺑﺎﺷﺪ ﮐﻪ ﺑﺎﻋﺚ ﺟﻠﺐ ﺗﻮﺟﻪ ﺷﺎﮔﺮﺩﺍﻥ، ﺗﺴﺮﻳﻊ ﺩﺭ ﺁﻣﻮﺯﺵ ﻭ ﻟﺬﺕ‌ﺑﺨﺶ‌ﺗﺮ ﺷﺪﻥ ﮐﻼﺱ ﺩﺭﺱ ﺑﺸﻮﺩ.

٦- ﺟﻤﻊ‌ﺑﻨﺪﯼ ﺩﺭﻭﺱ: ﺟﻤﻊ‌ﺑﻨﺪﯼ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭﻭﺱ ﺍﺭﺍﺋﻪ ﺷﺪﻩ ﺑﻪ ﻳﮏ ﻧﺘﻴﺠﻪ ﻣﻨﺎﺳﺐ ﻭ ﺩﻗﻴﻖ ﺑﺮﺳﺪ ﻭ ﺍﺯ ﭘﺮﺍﮐﻨﺪﮔﯽ ﻓﻌﺎﻟﻴﺖ ﻫﺎ ﺟﻠﻮﮔﻴﺮﯼ ﻣﯽ‌ﮐﻨﺪ.
