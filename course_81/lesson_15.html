ﺩﺭﺱ ﭼﻬﺎﺭﻡ- ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﻳﻮﺗﻴﻮﺏ

ﺷﻤﺎ ﺑﺮﺍﯼ ﻫﺮ ﮐﺎﺭﯼ ﻗﺒﻞ ﺍﺯ ﺁﻧﮑﻪ ﺑﺨﻮﺍﻫﻴﺪ ﺁﻥ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﻧﻴﺎﺯ ﺑﻪ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺩﺍﺭﻳﺪ. ﺣﺎﻻ ﺍﻳﻦ ﮐﺎﺭ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺭﺱ ﺧﻮﺍﻧﺪﻥ ﺑﺎﺷﺪ ﻳﺎ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﻳﻮﺗﻴﻮﺏ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺷﻤﺎ ﺑﺎ ﻓﺮﺁﻳﻨﺪﻩ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﺩﺭ ﻳﻮﺗﻴﻮﺏ ﺁﺷﻨﺎ ﻣﯽ ﺷﻮﻳﺪ ﻭ ﺧﻮﺍﻫﻴﺪ ﺁﻣﻮﺧﺖ ﭼﮕﻮﻧﻪ ﻳﮏ ﻭﻳﺪﻳﻮ ﺭﺍ ﺑﺎ ﺍﺳﺘﺎﻧﺪﺍﺭﻫﺎﯼ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﻳﻮﺗﻴﻮﺏ ﺿﺒﻂ ﮐﻨﻴﺪ.

 

ﻫﺪﻑ ﺧﻮﺩ ﺭﺍ ﺗﻌﺮﻳﻒ ﮐﻨﻴﺪ.

ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺩﺭ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﻳﻮﺗﻴﻮﺏ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻫﺪﻑ ﺧﻮﺩ ﺭﺍ ﺗﻌﺮﻳﻒ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻫﺪﻑ ﺧﻮﺩ ﺭﺍ ﺭﺍﺣﺖ ﺗﺮ ﺗﻌﺮﻳﻒ ﮐﻨﻴﺪ، ﺑﺎﻳﺪ ﺑﻪ ﺳﻮﺍﻻﺗﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﭘﺎﺳﺦ ﺩﻫﻴﺪ:

۱- ﭼﺮﺍ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ؟

ﭘﺎﺳﺦ ﺍﻳﻦ ﺳﻮﺍﻝ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﻩ ﻫﺎ ﻣﻮﺭﺩ ﺑﺎﺷﺪ. ﻣﺜﻼ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﺑﺨﻮﺍﻫﻴﺪ ﻭﻳﺪﻳﻮﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺩﻭﺳﺘﺎﻥ ﻭ ﺁﺷﻨﺎﻳﺎﻥ ﺧﻮﺩ ﺗﻮﻟﻴﺪ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺁﻧﮑﻪ ﺑﺮﺍﯼ ﻭﺏ ﺳﺎﻳﺖ ﻳﺎ ﻭﺑﻼﮔﯽ ﮐﻪ ﻣﺪﻳﺮﻳﺖ ﺁﻥ ﺭﺍ ﺑﺮ ﻋﻬﺪﻩ ﺩﺍﺭﻳﺪ ﺑﻪ ﺻﻮﺭﺕ ﻫﻔﺘﮕﯽ ﻭﻳﺪﻳﻮﻳﯽ ﺭﺍ ﻣﻨﺘﺸﺮ ﮐﻨﻴﺪ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﭘﺎﺳﺦ ﺍﻳﻦ ﺳﻮﺍﻝ ﺑﺴﺘﻪ ﺑﻪ ﻓﺎﮐﺘﻮﺭﻫﺎﯼ ﺩﻳﮕﺮﯼ ﻣﺎﻧﻨﺪ ﻧﻮﻉ ﻣﺨﺎﻃﺐ ﺗﺎﻥ ﺩﺍﺭﺩ ﮐﻪ ﺳﻮﺍﻝ ﺑﻌﺪﯼ ﺩﻗﻴﻘﺎ ﻫﻤﻴﻦ ﻣﻮﺭﺩ ﺍﺳﺖ.

۲- ﻣﺨﺎﻃﺐ ﺷﻤﺎ ﮐﻴﺴﺖ؟

ﻫﻤﺎﻧﮕﻮﻧﻪ ﮐﻪ ﮔﻔﺘﻪ ﺷﺪ، ﭘﺎﺳﺦ ﺳﻮﺍﻝ ﺍﻭﻝ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﮐﻪ ﻣﺨﺎﻃﺐ ﺷﻤﺎ ﮐﻴﺴﺖ، ﺑﺴﺘﮕﯽ ﺩﺍﺭﺩ. ﻣﻨﻈﻮﺭ ﺍﺯ ﺍﻳﻦ ﺳﻮﺍﻝ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﺎ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﭼﻪ ﮐﺴﺎﻧﯽ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺑﻴﻨﻨﺪﮔﺎﻥ ﻭﻳﺪﻳﻮﯼ ﺧﻮﺩ ﻫﺪﻑ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﻣﺜﻼ ﺍﮔﺮ ﻣﺨﺎﻃﺐ ﺷﻤﺎ ﺍﻓﺮﺍﺩ ﺟﻮﺍﻥ ﺑﺎﺷﻨﺪ، ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﻭﻳﺪﻳﻮﻫﺎﻳﯽ ﮐﻪ ﺗﻮﻟﻴﺪ ﻣﯽ ﮐﻨﻴﺪ ﮐﻮﺗﺎﻩ ﻭ ﺷﺎﻣﻞ ﻋﮑﺲ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﺎﺷﻨﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﮔﺮ ﺳﺮﻋﺖ ﺍﻳﻨﺘﺮﻧﺖ ﺁﻧﻬﺎ ﮐﻢ ﺍﺳﺖ، ﺑﺪ ﻧﻴﺴﺖ ﮐﻪ ﻭﻳﺪﻳﻮﻫﺎﯼ ﺗﻮﻟﻴﺪﯼ ﺗﺎﻥ ﺑﺴﻴﺎﺭ ﮐﻮﺗﺎﻩ ﺑﺎﺷﻨﺪ.

ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﻗﺒﻞ ﺍﺯ ﺁﻧﮑﻪ ﺷﺮﻭﻉ ﺑﻪ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﮐﻨﻴﺪ، ﮐﻤﯽ ﺩﺭ ﻣﻮﺭﺩ ﺑﻴﻨﻨﺪﮔﺎﻥ ﻭﻳﺪﻳﻮﯼ ﺧﻮﺩ ﻓﮑﺮ ﮐﻨﻴﺪ.

۳- ﭼﻪ ﭘﻴﺎﻣﯽ ﺭﺍ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﺑﺮﺳﺎﻧﻴﺪ؟

ﺍﻳﻦ ﺳﻮﺍﻝ ﺍﺯ ﺁﻥ ﺟﻬﺖ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺍﺳﺖ ﭼﻮﻥ ﺑﺎ ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﺁﻥ ﺑﻪ ﻧﻮﻋﯽ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﻣﻮﺿﻮﻉ ﻭﻳﺪﻳﻮﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﻣﺸﺨﺺ ﮐﻨﻴﺪ. ﻣﺜﻼ ﺍﮔﺮ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﭘﻴﺎﻡ ﺗﺎﻥ ﺩﺭ ﻣﻮﺭﺩ ﺁﻣﻮﺯﺵ ﻓﻼﻥ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺑﺎﺷﺪ، ﻣﺸﺨﺺ ﺍﺳﺖ ﮐﻪ ﻣﺤﺘﻮﺍﯼ ﻭﻳﺪﻳﻮﯼ ﺷﻤﺎ ﺁﻣﻮﺯﺷﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

۴- ﻭﻳﺪﻳﻮﻫﺎﯼ ﺗﻮﻟﻴﺪﯼ ﻣﺼﺮﻑ ﺷﺨﺼﯽ ﺩﺍﺭﻧﺪ ﻳﺎ ﺣﺮﻓﻪ ﺍﯼ؟

ﺑﺎ ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﺷﻤﺎ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺳﻄﺢ ﻭﻳﺪﻳﻮﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﻣﺸﺨﺺ ﮐﻨﻴﺪ. ﻣﺜﻼ ﺍﮔﺮ ﻭﻳﺪﻳﻮﯼ ﺷﻤﺎ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺩﺭ ﺯﻣﻴﻨﻪ ﺍﯼ ﺧﺎﺹ ﻓﻌﺎﻟﻴﺖ ﻣﯽ ﮐﻨﻨﺪ ﺗﻮﻟﻴﺪ ﺧﻮﺍﻫﺪ ﺷﺪ، ﺩﺭ ﻧﺘﻴﺠﻪ ﺳﻄﺢ ﺁﻥ ﻭﻳﺪﻳﻮ ﺣﺮﻓﻪ ﺍﯼ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺯﻣﺎﻧﯽ ﮐﻪ ﺑﻪ ۴ ﺳﻮﺍﻝ ﺑﺎﻻ ﺑﻪ ﺻﻮﺭﺕ ﻣﺸﺨﺺ ﭘﺎﺳﺦ ﺩﺍﺩﻳﺪ، ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻣﺮﺣﻠﻪ ﺑﻌﺪ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﺑﺮﻭﻳﺪ.

 

ﻓﺮﺁﻳﻨﺪ ﺗﻮﻟﻴﺪ ﻣﺤﺘﻮﺍ ﻭﻳﺪﻳﻮﻳﯽ

ﺣﺎﻻ ﮐﻪ ﻫﺪﻑ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﻳﻮﺗﻴﻮﺏ ﺗﻌﻴﻴﻦ ﮐﺮﺩﻩ ﺍﻳﺪ، ﺑﺎﻳﺪ ﺑﻪ ﺳﺮﺍﻍ ﻓﺮﺁﻳﻨﺪ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﺑﺮﻭﻳﺪ. ﺍﻳﻦ ﻓﺮﺁﻳﻨﺪ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ۵ ﻣﺮﺣﻠﻪ ﺗﻘﺴﻴﻢ ﮐﺮﺩ:

- ﭘﻴﺶ ﺍﺯ ﺗﻮﻟﻴﺪ: ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﻭ ﺗﺤﻘﻴﻘﺎﺕ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﻭﻳﺪﻳﻮ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻩ ﺑﺎﺷﻴﺪ. ﻣﺜﻼ ﺑﺎﻳﺪ ﺗﺼﺎﻭﻳﺮ، ﻣﺘﻦ ﻫﺎ ﻭ ﻣﮑﺎﻧﯽ ﮐﻪ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﻭﻳﺪﻳﻮ ﺭﺍ ﺩﺭ ﺁﻧﺠﺎ ﺿﺒﻂ ﮐﻨﻴﺪ ﺭﺍ ﻣﺸﺨﺺ ﮐﻨﻴﺪ. ﺑﻪ ﻃﻮﺭﯼ ﮐﻠﯽ ﻫﺮ ﺁﻧﭽﻪ ﺭﺍ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﻗﺒﻞ ﺍﺯ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﻗﺮﺍﺭ ﻣﯽ ﮔﻴﺮﺩ.

- ﺗﻮﻟﻴﺪ: ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺷﻤﺎ ﻭﻳﺪﻳﻮ ﻭ ﻓﺎﻳﻞ ﻫﺎﯼ ﻣﻮﺭﺩ ﻧﻴﺎﺯ ﺑﺮﺍﯼ ﻭﻳﺪﻳﻮﯼ ﺗﺎﻥ ﺭﺍ ﺿﺒﻂ ﻣﯽ ﮐﻨﻴﺪ ﻭ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺑﺨﻮﺍﻫﻴﺪ ﻋﮑﺲ ﻫﺎﻳﯽ ﺭﺍ ﺍﺯ ﻣﮑﺎﻥ ﺧﺎﺻﯽ ﺑﮕﻴﺮﻳﺪ، ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺁﻥ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺷﻤﺎ ﺑﺎﻳﺪ ﺗﻤﺎﻡ ﻣﺤﺘﻮﺍﯼ ﺗﺼﻮﻳﺮﯼ ﻭ ﺻﻮﺗﯽ ﺍﯼ ﮐﻪ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﻧﻬﺎﻳﯽ ﻭﻳﺪﻳﻮ ﻻﺯﻡ ﺩﺍﺭﻳﺪ ﺭﺍ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ.

- ﭘﺲ ﺍﺯ ﺗﻮﻟﻴﺪ: ﻧﺎﻡ ﺩﻳﮕﺮ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﻭﻳﺮﺍﻳﺶ (Editing) ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﺤﺘﻮﺍﯼ ﺗﺼﻮﻳﺮﯼ ﻭ ﺻﻮﺗﯽ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﺳﺎﺱ ﺑﺮﻧﺎﻣﻪ ﺍﯼ ﮐﻪ ﺩﺭ ﻣﺮﺣﻠﻪ ﭘﻴﺶ ﺍﺯ ﺗﻮﻟﻴﺪ ﺭﻳﺨﺘﻪ ﺍﻳﺪ، ﻭﻳﺮﺍﻳﺶ ﻭ ﻭﻳﺪﻳﻮ ﻧﻬﺎﻳﯽ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺗﻤﺎﺷﺎ ﺁﻣﺎﺩﻩ ﻣﯽ ﮐﻨﻴﺪ.

- ﻓﺸﺮﺩﻩ ﺳﺎﺯﯼ: ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺷﻤﺎ ﻭﻳﺪﻳﻮﯼ ﺗﺎﻥ ﺁﻣﺎﺩﻩ ﺑﺮﺍﯼ ﺗﻤﺎﺷﺎ ﺍﺳﺖ ﺍﻣﺎ ﻗﺒﻞ ﺍﺯ ﺁﻧﮑﻪ ﺁﻥ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﻳﻮﺗﻴﻮﺏ ﺑﺎﺭﮔﺬﺍﺭﯼ ﮐﻨﻴﺪ، ﻧﻴﺎﺯ ﺍﺳﺖ ﮐﻪ ﺁﻥ ﺭﺍ ﻓﺸﺮﺩﻩ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﻓﺸﺮﺩﻩ ﮐﺮﺩﻥ ﻭﻳﺪﻳﻮ ﺭﻭﺵ ﻫﺎ ﻭ ﺗﺮﻓﻨﺪﻫﺎﯼ ﺯﻳﺎﺩﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺍﺩﺍﻣﻪ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﻪ ﺁﻧﻬﺎ ﺍﺷﺎﺭﻩ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

- ﻧﻬﺎﻳﯽ ﮐﺮﺩﻥ ﻭﻳﺪﻳﻮ: ﺯﻣﺎﻧﯽ ﮐﻪ ﻭﻳﺪﻳﻮﯼ ﺷﻤﺎ ﺑﺮﺍﯼ ﺑﺎﺭﮔﺬﺍﺭﯼ ﺑﺮ ﺭﻭﯼ ﻳﻮﺗﻴﻮﺏ ﺁﻣﺎﺩﻩ ﺷﺪ، ﺑﺎﻳﺪ ﺁﺧﺮﻳﻦ ﻣﺮﺣﻠﻪ ﮐﻪ ﺷﺎﻣﻞ ﺍﻧﺘﺨﺎﺏ ﻋﻨﻮﺍﻥ، ﺗﻮﺿﻴﺤﺎﺕ ﻭﻳﺪﻳﻮ ﻭ ﻳﺎ ﻟﻴﻨﮏ ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﺎ ﻭﻳﺪﻳﻮ ﺍﺳﺖ ﺭﺍ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ ﺗﺎ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭ ﺯﻣﺎﻥ ﺑﺎﺭﮔﺬﺍﺭﯼ ﻭﻳﺪﻳﻮﯼ ﺗﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺷﺎﻳﺪ ﺑﺪ ﻧﺒﺎﺷﺪ ﻳﮏ ﻣﺜﺎﻝ ﺩﺭ ﻣﻮﺭﺩ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺑﺰﻧﻴﻢ ﺗﺎ ﺑﻪ ﺩﺭﮐﯽ ﺩﺭﺳﺖ ﺍﺯ ﻧﻬﺎﻳﯽ ﮐﺮﺩﻥ ﻭﻳﺪﻳﻮ ﺑﺮﺳﻴﺪ. ﺯﻣﺎﻧﯽ ﮐﻪ ﻳﮏ ﺩﯼ ﻭﯼ ﺩﯼ ﺗﻮﻟﻴﺪ ﻣﯽ ﺷﻮﺩ، ﺑﻪ ﻣﺮﺣﻠﻪ ﺍﯼ ﮐﻪ ﺩﺭ ﺁﻥ ﻣﻨﻮﻫﺎﯼ ﺩﯼ ﻭﯼ ﺩﯼ، ﺩﮐﻤﻪ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﻨﻮ ﻭ... ﻃﺮﺍﺣﯽ ﻣﯽ ﺷﻮﺩ، ﻣﺮﺣﻠﻪ ﻧﻬﺎﻳﯽ ﮐﺮﺩﻥ ﻭﻳﺪﻳﻮ ﮔﻔﺘﻪ ﻣﯽ ﺷﻮﺩ.

ﺣﺎﻻ ﮐﻪ ﺑﺎ ﻓﺮﺁﻳﻨﺪ ﮐﻠﯽ ﺗﻮﻟﻴﺪ ﻣﺤﺘﻮﺍﯼ ﻭﻳﺪﻳﻮﻳﯽ ﺁﺷﻨﺎ ﺷﺪﻳﺪ، ﺑﺪ ﻧﻴﺴﺖ ﮐﻪ ﻧﮕﺎﻫﯽ ﺑﻪ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﺻﻔﺤﻪ ﻫﺎﯼ ﮐﻮﭼﮏ ﻣﺎﻧﻨﺪ ﻳﻮﺗﻴﻮﺏ ﺑﻨﺪﺍﺯﻳﻢ.

 

ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﺻﻔﺤﻪ ﻫﺎﯼ ﮐﻮﭼﮏ

ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﻣﻮﺿﻮﻋﯽ ﺑﺴﻴﺎﺭ ﮔﺴﺘﺮﺩﻩ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻣﻮﺭﺩ ﺁﻧﻬﺎ ﻣﯽ ﺗﻮﺍﻥ ﭼﻨﺪﻳﻦ ﮐﺘﺎﺏ ﻧﻮﺷﺖ، ﺍﻣﺎ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻨﮑﻪ ﺍﻳﻦ ﺩﻭﺭﻩ ﺁﻣﻮﺯﺷﯽ ﺩﺭ ﻣﻮﺭﺩ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺩﺭ ﻳﻮﺗﻴﻮﺏ ﺍﺳﺖ، ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﺻﻔﺤﻪ ﻫﺎﯼ ﮐﻮﭼﮏ ﻣﺎﻧﻨﺪ ﻭﻳﺪﻳﻮﻫﺎﻳﯽ ﮐﻪ ﺑﺮ ﺭﻭﯼ ﻳﻮﺗﻴﻮﺏ ﺗﻤﺎﺷﺎ ﻣﯽ ﮐﻨﻴﺪ ﻣﯽ ﭘﺮﺩﺍﺯﻳﻢ.

ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺑﺎﺯ ﻣﯽ ﮔﺮﺩﺩ ﮐﻪ ﺷﻤﺎ ﺑﺮﺍﯼ ﭼﻪ ﻭﻳﺪﻳﻮ ﺗﻮﻟﻴﺪ ﻣﯽ ﮐﻨﻴﺪ؟ ﺁﻳﺎ ﻭﻳﺪﻳﻮﯼ ﺷﻤﺎ ﻗﺮﺍﺭ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﺣﺮﻓﻪ ﺍﯼ ﺩﺭ ﻣﻮﺭﺩ ﻣﻮﺿﻮﻋﯽ ﺻﺤﺒﺖ ﮐﻨﺪ ﻭ ﻳﺎ ﺁﻧﮑﻪ ﺑﺮﺍﯼ ﺗﻔﺮﻳﺢ ﻭ ﺗﻔﻨﻦ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﻭﻳﺪﻳﻮ ﺗﻮﻟﻴﺪ ﮐﻨﻴﺪ؟ ﭘﺎﺳﺦ ﺍﻳﻦ ﺳﻮﺍﻝ ﺑﺮ ﺭﻭﯼ ﻣﺪﺕ ﺯﻣﺎﻧﯽ ﮐﻪ ﺑﺮﺍﯼ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺍﺧﺘﺼﺎﺹ ﻣﯽ ﺩﻫﻴﺪ ﺗﺎﺛﻴﺮ ﻣﺴﺘﻘﻴﻤﯽ ﺩﺍﺭﺩ.

ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺑﺮﺍﯼ ﺁﻧﮑﻪ ﻳﮏ ﻭﻳﺪﻳﻮ ﺑﺎﮐﻴﻔﻴﺖ ﻭ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﻳﻮﺗﻴﻮﺏ ﺿﺒﻂ ﮐﻨﻴﺪ، ﺭﻋﺎﻳﺖ ﻧﮑﺎﺕ ﺯﻳﺮ ﺩﺭ ﻫﻨﮕﺎﻡ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺑﺴﻴﺎﺭ ﮐﺎﺭﺑﺮﺩﯼ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ:

- ﻫﻤﻮﺍﺭﻩ ﺍﺯ ﺳﻪ ﭘﺎﻳﻪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ: ﺣﺮﮐﺎﺕ ﻃﺒﻴﻌﯽ ﺩﻭﺭﺑﻴﻦ ﺩﺭ ﻫﻨﮕﺎﻡ ﺿﺒﻂ ﺁﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﻨﺎﺳﺐ ﻭ ﺗﺼﻮﻳﺮ ﺭﺍ ﻃﺒﻴﻌﯽ ﺗﺮ ﺟﻠﻮﻩ ﺩﻫﺪ، ﺍﻣﺎ ﺁﻧﭽﻪ ﮐﻪ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﺪ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺯﻣﺎﻥ ﻓﺸﺮﺩﻩ ﮐﺮﺩﻥ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﺑﺎﺭﮔﺬﺍﺭﯼ ﺑﺮ ﺭﻭﯼ ﻳﻮﺗﻴﻮﺏ، ﺍﻳﻦ ﺗﮑﺎﻥ ﻫﺎ ﺑﻴﺸﺘﺮ ﺑﻪ ﭼﺸﻢ ﺧﻮﺍﻫﻨﺪ ﺁﻣﺪ، ﻣﺨﺼﻮﺻﺎ ﺍﮔﺮ ﺁﻥ ﻭﻳﺪﻳﻮ ﺑﺮ ﺭﻭﯼ ﺻﻔﺤﻪ ﺍﯼ ﮐﻮﭼﮏ ﻣﺎﻧﻨﺪ ﺗﻠﻔﻦ ﻫﻤﺮﺍﻩ ﺗﻤﺎﺷﺎ ﺷﻮﺩ. ﺍﺯ ﻫﻤﻴﻦ ﺭﻭ ﻫﻤﻮﺍﺭﻩ ﺍﺯ ﺳﻪ ﭘﺎﻳﻪ ﺑﺮﺍﯼ ﻓﻴﻠﻢ ﺑﺮﺩﺍﺭﯼ ﻭﻳﺪﻳﻮ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ.

- ﺩﻭﺭﺑﻴﻦ ﻓﻴﻠﻢ ﺑﺮﺩﺍﺭﯼ ﺭﺍ ﺯﻳﺎﺩ ﺟﺎ ﺑﻪ ﺟﺎ ﻧﮑﻨﻴﺪ: ﺩﺭ ﺯﻣﺎﻧﯽ ﮐﻪ ﻓﻴﻠﻢ ﺑﺮﺩﺍﺭﯼ ﻣﯽ ﮐﻨﻴﺪ، ﺗﺎ ﺟﺎﻳﯽ ﮐﻪ ﺍﻣﮑﺎﻥ ﺩﺍﺭﺩ ﺍﺯ ﺟﺎ ﺑﻪ ﺟﺎ ﮐﺮﺩﻥ ﻭ ﻧﺰﺩﻳﮏ ﮐﺮﺩﻥ ﻳﺎ ﺩﻭﺭ ﮐﺮﺩﻥ ﺗﺼﻮﻳﺮ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺩﻗﺖ ﮐﻨﻴﺪ ﮐﻪ ﺩﺭ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺑﺮﺍﯼ ﺻﻔﺤﻪ ﻫﺎﯼ ﮐﻮﭼﮏ، ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﺗﺼﻮﻳﺮ ﺷﻤﺎ ﺛﺎﺑﺖ ﺑﺎﺷﺪ.

- ﻧﻮﺭﭘﺮﺩﺍﺯﯼ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ: ﻓﺎﺭﻍ ﺍﺯ ﺍﻳﻨﮑﻪ ﺍﺯ ﭼﻪ ﺗﺠﻬﻴﺰﺍﺗﯽ ﺑﺮﺍﯼ ﻧﻮﺭﭘﺮﺩﺍﺯﯼ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻴﺪ، ﭼﻪ ﻣﻴﺰﺍﻥ ﻧﻮﺭ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻳﺪ ﻭ...، ﻫﻤﻮﺍﺭﻩ ﺍﻳﻦ ﻧﮑﺘﻪ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻧﻮﺭ ﺩﺭ ﻣﺤﻴﻄﯽ ﮐﻪ ﻣﺸﻐﻮﻝ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﻫﺴﺘﻴﺪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺍﮔﺮ ﺷﻤﺎ ﺑﺎ ﺩﻭﺭﺑﻴﻦ ﭼﻨﺪ ﺩﻩ ﻣﻴﻠﻴﻮﻧﯽ ﻓﻴﻠﻢ ﺑﺮﺩﺍﺭﯼ ﺑﮑﻨﻴﺪ ﺍﻣﺎ ﻧﻮﺭﭘﺮﺩﺍﺯﯼ ﻣﻨﺎﺳﺒﯽ ﺩﺭ ﻣﺤﻴﻄﯽ ﮐﻪ ﺿﺒﻂ ﻣﯽ ﮐﻨﻴﺪ ﻭﺟﻮﺩ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﻧﺘﻴﺠﻪ ﮐﺎﺭ ﺑﺴﻴﺎﺭ ﻧﺎﺍﻣﻴﺪﮐﻨﻨﺪﻩ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

- ﮐﻴﻔﻴﺖ ﺻﺪﺍ ﻣﻬﻢ ﺍﺳﺖ: ﻫﻤﻮﺍﺭﻩ ﺑﻪ ﮐﻴﻔﻴﺖ ﺻﺪﺍﯼ ﻭﻳﺪﻳﻮﻳﯽ ﮐﻪ ﺿﺒﻂ ﻣﯽ ﮐﻨﻴﺪ ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﻭ ﺑﻪ ﻫﻴﭻ ﻋﻨﻮﺍﻥ ﻭ ﺗﺤﺖ ﻫﻴﭻ ﺷﺮﺍﻳﻄﯽ ﻭﻳﺪﻳﻮﻳﯽ ﺑﺎ ﮐﻴﻔﻴﺖ ﺻﺪﺍﯼ ﺑﺪ ﺗﻮﻟﻴﺪ ﻧﮑﻨﻴﺪ ﺯﻳﺮﺍ ﺑﺪﺗﺮﻳﻦ ﭼﻴﺰ ﺩﺭ ﺗﻮﻟﻴﺪ ﻳﮏ ﻭﻳﺪﻳﻮ، ﮐﻴﻔﻴﺖ ﺑﺪ ﺻﺪﺍﯼ ﺁﻥ ﺍﺳﺖ. ﺍﮔﺮ ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﻣﻴﮑﺮﻭﻓﻦ ﻭﺍﻗﻌﯽ ﻭ ﻳﺎ ﻫﺪﺳﺖ ﺑﺮﺍﯼ ﺿﺒﻂ ﻭﻳﺪﻳﻮ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ، ﺣﺘﻤﺎ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ.
