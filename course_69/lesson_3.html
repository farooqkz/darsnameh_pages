ﺑﻪ ﻧﻈﺮ ﻣﯽ ﺭﺳﺪ ﺩﺭ ﺍﺑﺘﺪﺍ ﺍﻳﻦ ﻣﻤﻨﻮﻋﻴﺖ ﺗﻨﻬﺎ ﺑﻪ ﺩﺍﺩﻥ ﻭﺍﻡ ﺑﺎ ﺑﻬﺮﻩ ﺑﻪ ﻓﻘﺮﺍ ﺍﺷﺎﺭﻩ ﺩﺍﺷﺘﻪ ﺍﺳﺖ٬ ﺍﻣﺎ ﺑﻌﺪﺍ ﺑﻪ ﻫﻤﻪ ﺗﻌﻤﻴﻢ ﺩﺍﺩﻩ ﺷﺪﻩ ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﻣﻤﻨﻮﻋﻴﺖ ﻓﻘﻂ ﺩﺭ ﻣﻴﺎﻥ ﺟﺎﻣﻌﻪ ﻣﻮﻣﻨﺎﻥ (ﻳﻬﻮﺩﻳﺎﻥ) ﺑﺮﻗﺮﺍﺭ ﺑﻮﺩﻩ ﺍﺳﺖ: «ﺑﻪ ﻳﮏ ﻏﺮﻳﺒﻪ ﻣﯽ ﺗﻮﺍﻧﯽ ﻭﺍﻡ ﺑﺎ ﺑﻬﺮﻩ ﺑﺪﻫﯽ٬ ﺍﻣﺎ ﺑﻪ ﺑﺮﺍﺩﺭﺕ ﻧﻤﯽ ﺑﺎﻳﺪ ﻭﺍﻡ ﺑﺎ ﺑﻬﺮﻩ ﺑﺪﻫﯽ.» (ﺳﻔﺮ ﺧﺮﻭﺝ)

 

ﺑﺎﺯﺍﺭ ﺳﻬﺎﻡ ﺩﺭ ﺭﻡ ﺑﺎﺳﺘﺎﻥ
ﺩﺭﺑﺎﺭﻩ ﻧﻘﻄﻪ ﭘﻴﺪﺍﻳﺶ ﺑﺎﺯﺍﺭ ﺳﻬﺎﻡ ﺍﺧﺘﻼﻑ ﺍﺳﺖ. ﻣﻄﺎﻟﻌﻪ ﺟﺎﻟﺒﯽ ﺩﺭ ﺍﻳﻦ ﺑﺎﺭﻩ ﻧﻘﻄﻪ ﭘﻴﺪﺍﻳﺶ ﺍﻳﻦ ﺑﺎﺯﺍﺭ ﺭﺍ ﺗﺎ ﺭﻡ ﺑﺎﺳﺘﺎﻥ ﺑﻪ ﻋﻘﺐ ﻣﯽ ﺑﺮﺩ. ﺍﻭﻟﺮﻳﮏ ﻣﺎﻟﻤﻨﺪﻳﺮ ﺩﺭ ﻣﻘﺎﻟﻪ ﺍﺵ ﺑﻪ ﺷﺮﮐﺖ ﻫﺎﯼ ﻋﻤﻮﻣﯽ ﻳﺎ  ﺳﻮﺳﻴﺘﺎﺱ ﭘﻮﺑﻠﻴﮑﺎﻧﻮﺭﻭﻡ (Societas Publicanorum) ﺍﺷﺎﺭﻩ ﻣﯽ ﮐﻨﺪ.

ﺍﻳﻦ ﺟﺎﻣﻌﻪ ﻣﺤﺼﻮﻝ ﺟﻤﻬﻮﺭﯼ ﺭﻡ (Roman Republic) ﺑﻮﺩ (۱). ﺩﺭ ﺁﻥ ﺯﻣﺎﻥ ﺩﺭ ﺟﻤﻬﻮﺭﯼ ﺭﻡ ﺧﺪﻣﺎﺕ ﻋﻤﻮﻣﯽ (Public Service) ﺑﻪ ﺻﻮﺭﺕ ﻗﺮﺍﺭﺩﺍﺩ ﺑﻪ ﺑﺨﺶ ﺧﺼﻮﺻﯽ ﺩﺍﺩﻩ ﻣﯽ ﺷﺪ. ﺑﻪ ﺍﻳﻦ ﭘﻴﻤﺎﻧﮑﺎﺭﺍﻥ ﺧﺼﻮﺻﯽ ﺳﻮﺳﻴﺘﻪ ﭘﺒﻠﻴﮑﺎﻧﻮﺭﻭﻡ ﻳﺎ ﺻﺎﺣﺒﺎﻥ ﺍﺟﺎﺭﻩ ﺑﻠﻨﺪ ﻣﺪﺕ ﺣﮑﻮﻣﺘﯽ (Government Leaseholder) ﮔﻔﺘﻪ ﻣﯽ ﺷﺪ.

ﭘﻮﺑﻠﻴﮑﺎﻥ ﻫﺎ ﺩﺭ ﺳﻪ ﻋﺮﺻﻪ ﻓﻌﺎﻟﻴﺖ ﻣﯽ ﮐﺮﺩﻧﺪ:

۱- ﻓﺮﺍﻫﻢ ﮐﺮﺩﻥ ﮐﺎﻻ ﻭ ﺧﺪﻣﺎﺕ ﺑﺮﺍﯼ ﻋﻤﻮﻡ

۲- ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﮑﺎﻥ ﻫﺎﯼ ﻋﻤﻮﻣﯽ

۳- ﺟﻤﻊ ﺁﻭﺭﯼ ﻣﺎﻟﻴﺎﺕ

ﺷﺮﺍﮐﺖ ﺭﻣﯽ ﻣﻌﻨﺎﻳﯽ ﻣﺘﻔﺎﻭﺕ ﺍﺯ ﺷﺮﺍﮐﺖ ﻣﺪﺭﻥ ﺩﺍﺷﺖ. ﺷﺮﮐﺎ ﻧﻤﯽ ﺗﻮﺍﻧﺴﺘﻨﺪ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺷﺮﮐﺖ ﻭ ﺷﺮﺍﮐﺖ ﻧﻤﯽ ﺗﻮﺍﻧﺴﺖ ﭘﺲ ﺍﺯ ﻣﺮﮒ ﻳﮑﯽ ﺍﺯ ﺷﺮﮐﺎ ﺍﺩﺍﻣﻪ ﭘﻴﺪﺍ ﮐﻨﺪ، ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻳﮑﯽ ﺍﺯ ﺷﺮﮐﺎ ﺧﻮﺩﺵ ﺭﺍ ﺍﺯ ﺷﺮﺍﮐﺖ ﺑﻴﺮﻭﻥ ﻣﯽ ﮐﺸﻴﺪ٬ ﺑﺎﻗﯽ ﺷﺮﮐﺎ ﻧﻤﯽ ﺗﻮﺍﻧﺴﺘﻨﺪ ﺍﺩﺍﻣﻪ ﺩﻫﻨﺪ. ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺩﻋﻮﺍﻳﯽ ﻗﺎﻧﻮﻧﯽ ﻣﻴﺎﻥ ﺷﺮﮐﺎ ﭘﻴﺶ ﻣﯽ ﺁﻣﺪ٬ ﻣﻮﺿﻮﻉ ﺷﺮﺍﮐﺖ ﻣﻠﻐﯽ ﻣﯽ ﺷﺪ.

ﺑﻪ ﻳﮏ ﻣﻌﻨﺎ ﺷﺮﮐﺖ ﺑﻪ ﺧﻮﺩﯼ ﺧﻮﺩ ﻋﺮﺻﻪ ﻋﻤﻞ ﻣﺤﺪﻭﺩﯼ ﺩﺍﺷﺖ. ﺍﻳﻦ ﺳﺎﺧﺘﺎﺭ ﻣﺸﮑﻠﯽ ﭘﻴﺶ ﻣﯽ ﺁﻭﺭﺩ. ﺍﺟﺎﺭﻩ ﺑﻠﻨﺪ ﻣﺪﺕ ﺣﮑﻮﻣﺘﯽ ﻧﻴﺎﺯﻣﻨﺪ ﺁﻥ ﺑﻮﺩ ﮐﻪ ﺷﺮﮐﺖ ﺑﺘﻮﺍﻧﺪ ﺍﻓﻖ ﺑﻠﻨﺪ ﻣﺪﺕ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺩﺭ ﺑﻠﻨﺪ ﻣﺪﺕ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﻳﻦ ﺟﺎ ﺑﻮﺩ ﮐﻪ ﺭﻣﯽ ﻫﺎ ﺑﺮﺍﯼ ﺁﺳﺎﻥ ﮐﺮﺩﻥ ﮐﺎﺭ ﺗﻐﻴﻴﺮﺍﺗﯽ ﺩﺭ ﻗﺎﻧﻮﻥ ﻣﺎﻟﮑﻴﺖ ﻭ ﺷﺮﺍﮐﺘﺸﺎﻥ ﭘﺪﻳﺪ ﺁﻭﺭﺩﻧﺪ.

ﺍﻳﻦ ﺳﺮﺁﻏﺎﺯ ﺗﻔﺎﻭﺕ ﻣﻌﻨﺎﺩﺍﺭ ﻣﻴﺎﻥ ﺷﺮﮐﺖ ﻋﻤﻮﻣﯽ ﻭ ﺷﺮﮐﺖ ﺳﺎﺩﻩ (Simple Societas) ﺑﻮﺩ. ﺍﻭﻟﺮﻳﮏ ﻣﺎﻟﻤﻨﺪﺭ ﺍﺯ ﭼﻬﺎﺭ ﺗﻔﺎﻭﺕ ﻋﻤﺪﻩ ﻣﻴﺎﻥ ﺍﻳﻦ ﺩﻭ ﻧﻮﻉ ﺷﺮﮐﺖ ﻧﺎﻡ ﻣﯽ ﺑﺮﺩ:

۱- ﻧﻤﺎﻳﻨﺪﮔﯽ: ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺟﺪﻳﺪ ﺍﺯ ﺷﺮﮐﺖ ﻳﮏ ﻧﻔﺮ ﻣﯽ ﺗﻮﺍﻧﺴﺖ ﺑﻪ ﻧﻤﺎﻳﻨﺪﮔﯽ ﺍﺯ ﺷﺮﮐﺖ ﻗﺮﺍﺭﺩﺍﺩ ﺑﺒﻨﺪﺩ. ﺍﻳﻦ ﺩﺭ ﻭﺍﻗﻊ ﺳﺮﺁﻏﺎﺯ ﭘﻴﺪﺍﻳﺶ ﺟﺎﻳﮕﺎﻩ ﻣﺪﻳﺮ ﻋﺎﻣﻞ ﺩﺭ ﺷﺮﮐﺖ ﻫﺎﯼ ﻋﻤﻮﻣﯽ ﺑﻮﺩ.

۲- ﺗﺪﺍﻭﻡ ﻭ ﺛﺒﺎﺕ: ﺩﻳﺪﻳﻢ ﮐﻪ ﺩﺭ ﻣﺪﻝ ﻗﺪﻳﻤﯽ ﺷﺮﮐﺖ٬ ﭘﺲ ﺍﺯ ﺧﺮﻭﺝ ﻳﮑﯽ ﺍﺯ ﺷﺮﮐﺎ٬ ﺍﻣﮑﺎﻥ ﺗﺪﺍﻭﻡ ﻧﺪﺍﺷﺖ. ﻧﻮﻉ ﺟﺪﻳﺪ، ﺍﻣﮑﺎﻥ ﺍﻳﻦ ﺗﺪﺍﻭﻡ ﻭ ﺛﺒﺎﺕ ﺭﺍ ﻓﺮﺍﻫﻢ ﻣﯽ ﺁﻭﺭﺩ. ﺩﺭ ﻭﺍﻗﻊ ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺟﺪﻳﺪ ﺷﺮﮐﺖ ﻣﻮﺟﻮﺩﻳﺘﯽ ﻣﺴﺘﻘﻞ ﺍﺯ ﺷﺮﮐﺎ ﺩﺍﺷﺖ.

۳- ﺳﺮﻣﺎﻳﻪ ﺧﺎﺭﺟﯽ: ﺩﺭ ﻧﻮﻉ ﺟﺪﻳﺪ ﺷﺮﮐﺖ، ﺍﻓﺮﺍﺩﯼ ﺑﻪ ﺟﺰ ﺷﺮﮐﺎ ﻣﯽ ﺗﻮﺍﻧﺴﺘﻨﺪ ﺳﻬﺎﻡ ﺷﺮﮐﺖ ﺭﺍ ﺑﺨﺮﻧﺪ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺑﻪ ﻳﮏ ﺷﺮﻳﮏ ﺗﺒﺪﻳﻞ ﺷﻮﻧﺪ ﻭ ﺩﺭ ﺑﺮﺍﺑﺮ ﺍﻋﻤﺎﻝ ﺷﺮﮐﺖ ﻣﺴﺌﻮﻟﻴﺘﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺍﻳﻦ ﮔﺎﻣﯽ ﻣﻬﻢ ﺑﻪ ﺳﻤﺖ ﭘﻴﺪﺍﻳﺶ ﺷﺮﮐﺖ ﻫﺎﯼ ﻋﻤﻮﻣﯽ ﮐﻨﻮﻧﯽ ﺑﻮﺩ ﮐﻪ ﺩﺭ ﺁﻥ ﺳﻬﺎﻣﺪﺍﺭﺍﻥ ﺑﺎ ﺧﺮﻳﺪ ﺳﻬﺎﻡ ﺩﺭ ﺑﺮﺍﺑﺮ ﺍﻋﻤﺎﻝ ﺷﺮﮐﺖ ﻣﺴﺌﻮﻟﻴﺖ ﻗﻀﺎﻳﯽ ﭘﻴﺪﺍ ﻧﻤﯽ ﮐﻨﻨﺪ.

ﺍﻳﻦ ﺳﻬﺎﻡ ﺩﺭ ﺁﻥ ﺯﻣﺎﻥ ﺩﺭ ﺑﺎﺯﺍﺭ ﺭﻡ ﺧﺮﻳﺪ ﻭ ﻓﺮﻭﺵ ﻣﯽ ﺷﺪ ﻭ ﻗﻴﻤﺘﺶ ﺑﺎﻻ ﻭ ﭘﺎﻳﻴﻦ ﻧﻴﺰ ﻣﯽ ﺷﺪ. ﺳﻴﺴﺮﻭ (۲) ﺩﺭ ﺁﺛﺎﺭﺵ ﺍﺯ ﺷﺮﮐﺖ ﻫﺎﻳﯽ ﻧﺎﻡ ﻣﯽ ﺑﺮﺩ ﮐﻪ ﺳﻬﺎﻣﺸﺎﻥ ﻗﻴﻤﺖ ﺧﻴﻠﯽ ﺑﺎﻻﻳﯽ ﺩﺍﺷﺖ.
