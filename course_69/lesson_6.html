ﺩﺭﺱ ﭼﻬﺎﺭﻡ- ﻣﺎﻟﮑﻴﺖ ﻭ ﺳﺎﺧﺘﺎﺭ ﻣﺎﻟﯽ ﺷﺮﮐﺖ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺸﻴﻦ ﺍﺯ ﺍﻭﺭﺍﻕ ﻗﺮﺿﻪ ﻭ ﺳﻬﺎﻡ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺍﻧﻮﺍﻉ ﻣﺎﻟﮑﻴﺖ ﻭ ﺳﺎﺧﺘﺎﺭ ﻣﺎﻟﯽ ﺷﺮﮐﺖ ﻣﯽ ﭘﺮﺩﺍﺯﻳﻢ.

ﻣﺎﻟﮑﻴﺖ ﻳﮏ ﺷﺮﮐﺖ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﺷﮑﺎﻝ ﻣﺨﺘﻠﻔﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺳﺎﺩﻩ ﺗﺮﻳﻦ ﻧﻮﻉ ﻣﺎﻟﮑﻴﺖ، ﻣﺎﻟﮑﻴﺖ ﻳﮏ ﻓﺮﺩ ﺍﺳﺖ.

 

ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ
ﻭﻗﺘﯽ ﻳﮏ ﻧﻔﺮ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻣﺎﻟﮏ ﻳﮏ ﺷﺮﮐﺖ ﺍﺳﺖ٬ ﻣﯽ ﮔﻮﻳﻴﻢ ﺍﻭ ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﺩﺍﺭﺩ. ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﻣﻌﻤﻮﻝ ﺗﺮﻳﻦ ﻭ ﻣﺘﺪﺍﻭﻝ ﺗﺮﻳﻦ ﻧﻮﻉ ﻣﺎﻟﮑﻴﺖ ﺩﺭ ﺟﻬﺎﻥ ﺍﺳﺖ. ﺗﻤﺎﻣﯽ ﺑﻴﺰﻧﺲ ﻫﺎﻳﯽ ﮐﻪ ﻣﺎﻟﮏ ﺷﺎﻥ ﻳﮏ ﻓﺮﺩ ﺍﺳﺖ ﺩﺭ ﺍﻳﻦ ﺩﺳﺘﻪ ﺟﺎ ﻣﯽ ﮔﻴﺮﻧﺪ. ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﮐﺸﻮﺭﻫﺎ ﺑﺮﺍﯼ ﺭﺍﻩ ﺍﻧﺪﺍﺧﺘﻦ ﭼﻨﻴﻦ ﺗﺠﺎﺭﺗﯽ ﺷﻤﺎ ﻧﻴﺎﺯﻣﻨﺪ ﺁﻥ ﻫﺴﺘﻴﺪ ﮐﻪ ﻣﺠﻮﺯ ﺑﮕﻴﺮﻳﺪ.

ﺑﺮﺧﯽ ﮐﺸﻮﺭﻫﺎ ﻫﻢ ﻫﺴﺘﻨﺪ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﭼﻨﻴﻦ ﻣﺠﻮﺯﯼ ﻧﺪﺍﺭﻧﺪ. ﻃﺒﻴﻌﺘﺎ ﺩﺭ ﺍﻳﻦ ﮐﺸﻮﺭﻫﺎ ﺭﺍﻩ ﺍﻧﺪﺍﺧﺘﻦ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺳﻬﻞ ﺗﺮ ﺍﺳﺖ. ﺳﻬﻮﻟﺖ ﺁﻏﺎﺯ ﻫﺮ ﮐﺴﺐ ﻭ ﮐﺎﺭﯼ ﻳﮑﯽ ﺍﺯ ﻓﺎﮐﺘﻮﺭﻫﺎﯼ ﻣﻬﻤﯽ ﺍﺳﺖ ﮐﻪ ﻧﻘﺸﯽ ﺍﺳﺎﺳﯽ ﺩﺭ ﺭﻭﻧﻖ ﻭ ﺭﺷﺪ ﺍﻗﺘﺼﺎﺩﯼ ﮐﺸﻮﺭﻫﺎ ﺩﺍﺭﺩ.

ﺷﺮﮐﺖ ﺑﺎ ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﺍﺭﺯﺍﻥ ﺗﺮﻳﻦ ﻧﻮﻉ ﺗﺸﮑﻴﻞ  ﺷﺮﮐﺖ ﺍﺳﺖ ﮐﻪ ﻋﻤﺪﺗﺎ ﺑﻪ ﻣﺠﻮﺯﻫﺎﯼ ﮐﻤﯽ ﻧﻴﺎﺯ ﺩﺍﺭﺩ. ﺷﺮﮐﺖ ﺑﺎ ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﻣﺎﻟﻴﺎﺕ ﻣﺮﺑﻮﻁ ﺑﻪ ﺷﺮﮐﺖ ﻫﺎ ﺭﺍ ﻧﻤﯽ ﭘﺮﺩﺍﺯﺩ. ﻣﺎﻟﻴﺎﺕ  ﺑﺮ ﺳﻮﺩ ﺷﺮﮐﺖ ﺗﻌﻠﻖ ﻣﯽ ﮔﻴﺮﺩ ﻭ ﺑﺎ ﺁﻥ ﻣﺸﺎﺑﻪ ﺑﺎ ﺩﺭﺁﻣﺪ ﻓﺮﺩﯼ ﺑﺮﺧﻮﺭﺩ ﻣﯽ ﺷﻮﺩ.

ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﻣﺴﺌﻮﻟﻴﺖ ﻧﺎﻣﺤﺪﻭﺩ ﺩﺍﺭﺩ. ﺻﺎﺣﺐ ﺷﺮﮐﺖ ﺩﺭ ﺑﺮﺍﺑﺮ ﺗﻤﺎﻣﯽ ﻗﺮﺽ ﻫﺎ ﻭ ﻣﺴﺌﻮﻟﻴﺖ ﻫﺎﯼ ﺷﺮﮐﺖ ﻣﺴﺌﻮﻟﻴﺖ ﺗﺎﻡ ﺩﺍﺭﺩ. ﻫﻴﭻ ﺗﻔﺎﻭﺗﯽ ﻣﻴﺎﻥ ﺩﺍﺭﺍﻳﯽ ﻫﺎﯼ ﺷﺮﮐﺖ ﻭ ﺩﺍﺭﺍﻳﯽ ﻫﺎﯼ ﺻﺎﺣﺐ ﺷﺮﮐﺖ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ.

ﺑﺪﻳﻦ ﻣﻌﻨﺎ ﺍﮔﺮ ﺷﺮﮐﺘﯽ ﺑﺎ ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﻗﺮﺿﯽ ﺩﺍﺭﺩ٬ ﺁﻥ ﻗﺮﺽ ﺑﺎﻳﺪ ﺍﺯ ﺍﻣﻮﺍﻝ ﺻﺎﺣﺐ ﺷﺮﮐﺖ ﭘﺮﺩﺍﺧﺖ ﺷﻮﺩ. ﻋﻤﺮ ﺷﺮﮐﺖ ﺑﺎ ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﻣﺤﺪﻭﺩ ﺍﺳﺖ ﺑﻪ ﻋﻤﺮ ﻣﺎﻟﮏ ﺷﺮﮐﺖ.

ﺗﻤﺎﻣﯽ ﺍﻣﻮﺍﻝ ﺷﺮﮐﺖ ﻣﺘﻌﻠﻖ ﺑﻪ ﻣﺎﻟﮏ ﺍﺳﺖ٬ ﺑﻨﺎﺑﺮﺍﻳﻦ ﺳﺮﻣﺎﻳﻪ ﮔﺬﺍﺭﯼ ﺩﺭ ﺷﺮﮐﺖ ﺗﻨﻬﺎ ﺍﺯ ﻃﺮﻳﻖ ﺍﻣﻮﺍﻝ ﺻﺎﺣﺐ ﺷﺮﮐﺖ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﺍﺳﺖ.

 

ﺷﺮﺍﮐﺖ (Partnership)
ﺩﻭ ﻧﻔﺮ ﻳﺎ ﺑﻴﺸﺘﺮ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﮔﺮﺩ ﻫﻢ ﺑﻴﺎﻳﻨﺪ ﻭ ﺑﺎ ﻫﻢ ﺷﺮﺍﮐﺖ ﮐﻨﻨﺪ. ﺷﺮﺍﮐﺖ ﺑﻪ ﺩﻭ ﺩﺳﺘﻪ ﮐﻠﯽ ﺗﻘﺴﻴﻢ ﻣﯽ ﺷﻮﺩ. ﺷﺮﺍﮐﺖ ﻧﺎﻣﺤﺪﻭﺩ (General Partnership) ﻭ ﺷﺮﺍﮐﺖ ﻣﺤﺪﻭﺩ (Limited Partnership).

ﺩﺭ ﺷﺮﺍﮐﺖ ﻧﺎﻣﺤﺪﻭﺩ ﻫﻤﻪ ﺷﺮﮐﺎ ﺗﻮﺍﻓﻖ ﻣﯽ ﮐﻨﻨﺪ ﮐﻪ ﺑﺨﺸﯽ ﺍﺯ ﮐﺎﺭ ﻭ ﭘﻮﻝ ﺭﺍ ﻓﺮﺍﻫﻢ ﮐﻨﻨﺪ ﻭ ﺩﺭ ﻣﻘﺎﺑﻞ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺭﺍ ﺑﺎ ﻫﻢ ﺗﻘﺴﻴﻢ ﮐﻨﻨﺪ. ﺗﻤﺎﻣﯽ ﺷﺮﮐﺎ ﻧﺴﺒﺖ ﺑﻪ ﺗﻤﺎﻣﯽ ﻗﺮﻭﺽ ﺷﺮﮐﺖ ﻣﺴﺌﻮﻟﻨﺪ. ﻳﮏ ﺗﻮﺍﻓﻖ ﻧﺎﻣﻪ ﺷﺮﺍﻳﻂ ﺷﺮﺍﮐﺖ ﺭﺍ ﻣﻌﻴﻦ ﻣﯽ ﮐﻨﺪ.

ﺷﺮﺍﮐﺖ ﻣﺤﺪﻭﺩ ﺍﺟﺎﺯﻩ ﻣﯽ ﺩﻫﺪ ﮐﻪ ﻣﺴﺌﻮﻟﻴﺖ ﺑﺮﺧﯽ ﺷﺮﮐﺎ ﻣﺤﺪﻭﺩ ﺑﻪ ﺳﺮﻣﺎﻳﻪ ﺍﯼ ﮐﻪ ﺁﻭﺭﺩﻩ ﺍﻧﺪ ﺑﺎﺷﺪ. ﺩﺭ ﺷﺮﺍﮐﺖ ﻣﺤﺪﻭﺩ ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﺣﺪﺍﻗﻞ ﻳﮑﯽ ﺍﺯ ﺷﺮﮐﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻧﺎﻣﺤﺪﻭﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺷﺮﮐﺎ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﺩﺭ ﻣﺪﻳﺮﻳﺖ ﺷﺮﮐﺖ ﺳﻬﻴﻢ ﻧﺒﺎﺷﻨﺪ.

ﺷﺮﺍﮐﺖ ﻣﻌﻤﻮﻻ ﺍﺭﺯﺍﻥ ﻭ ﺳﺎﺩﻩ ﺍﺳﺖ. ﺷﺮﮐﺎﯼ ﻧﺎﻣﺤﺪﻭﺩ ﻣﺴﺌﻮﻟﻴﺖ ﻧﺎﻣﺤﺪﻭﺩ ﺩﺍﺭﻧﺪ٬ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺷﺮﮐﺎ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﻧﺴﺒﺖ ﺑﻪ ﭘﻮﻟﯽ ﮐﻪ ﻣﻴﺎﻥ ﻧﻬﺎﺩﻩ ﺍﻧﺪ ﻣﺴﺌﻮﻟﻨﺪ. ﺍﮔﺮ ﻳﮑﯽ ﺍﺯ ﺷﺮﮐﺎﯼ ﻧﺎﻣﺤﺪﻭﺩ ﻧﺘﻮﺍﻧﺪ ﺳﻬﻤﺶ ﺭﺍ ﺑﭙﺮﺩﺍﺯﺩ، ﺷﺮﻳﮏ ﻧﺎﻣﺤﺪﻭﺩ ﺩﻳﮕﺮ ﺑﺎﻳﺪ ﻣﺴﺌﻮﻟﻴﺖ ﺑﺪﻫﯽ ﺍﻭ ﺭﺍ ﻗﺒﻮﻝ ﮐﻨﺪ.

ﺯﻣﺎﻧﯽ ﮐﻪ ﻳﮏ ﺷﺮﻳﮏ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻧﺎﻣﺤﺪﻭﺩ ﻓﻮﺕ ﮐﻨﺪ ﻳﺎ ﺷﺮﺍﮐﺖ ﺧﻮﺩ ﺭﺍ ﻟﻐﻮ ﮐﻨﺪ٬ ﺷﺮﺍﮐﺖ ﻧﺎﻣﺤﺪﻭﺩ ﺑﻪ ﻫﻢ ﻣﯽ ﺧﻮﺭﺩ. ﭼﻨﻴﻦ ﺍﻣﺮﯼ ﺩﺭ ﻣﻮﺭﺩ ﺷﺮﻳﮏ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﺻﺎﺩﻕ ﻧﻴﺴﺖ.

ﻳﮏ ﺷﺮﻳﮏ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻧﺎﻣﺤﺪﻭﺩ ﺗﻨﻬﺎ ﺩﺭ ﺻﻮﺭﺗﯽ ﻣﯽ ﺗﻮﺍﻧﺪ ﺳﻬﻤﺶ ﺭﺍ ﺑﻪ ﮐﺲ ﺩﻳﮕﺮﯼ ﻣﻨﺘﻘﻞ ﮐﻨﺪ ﮐﻪ ﻫﻤﻪ ﺷﺮﮐﺎﯼ ﺩﻳﮕﺮ ﻗﺒﻮﻝ ﮐﻨﻨﺪ. ﺷﺮﻳﮏ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﻣﯽ ﺗﻮﺍﻧﺪ ﺳﻬﻤﺶ ﺭﺍ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺑﻔﺮﻭﺷﺪ.

ﺷﺮﺍﮐﺖ ﮐﺎﺭ ﺭﺍ ﺑﺮﺍﯼ ﺗﺎﻣﻴﻦ ﺳﺮﻣﺎﻳﻪ ﺳﺨﺖ ﻣﯽ ﮐﻨﺪ. ﺑﺴﻴﺎﺭﯼ ﮐﺴﺐ ﻭ ﮐﺎﺭﻫﺎ ﺑﺎ ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﻳﺎ ﺷﺮﺍﮐﺖ ﺁﻏﺎﺯ ﻣﯽ ﺷﻮﻧﺪ ﻭ ﺑﻪ ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﺧﺘﻢ ﻣﯽ ﺷﻮﻧﺪ.

ﺑﺮ ﺩﺭﺁﻣﺪ ﺣﺎﺻﻞ ﺍﺯ ﺷﺮﺍﮐﺖ ﻫﻤﭽﻮﻥ ﺩﺭﺁﻣﺪ ﻓﺮﺩﯼ ﻣﺎﻟﻴﺎﺕ ﺗﻌﻠﻖ ﻣﯽ ﮔﻴﺮﺩ. ﮐﻨﺘﺮﻝ ﻣﺪﻳﺮﻳﺖ ﺩﺭ ﺩﺳﺖ ﺷﺮﮐﺎ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻧﺎﻣﺤﺪﻭﺩ ﺍﺳﺖ. ﺩﺭ ﻣﺴﺎﺋﻞ ﻣﻬﻢ ﺭﺍﯼ ﺍﮐﺜﺮﻳﺖ ﻧﻴﺎﺯ ﺍﺳﺖ.

ﻳﮏ ﺷﺮﮐﺖ ﺑﺰﺭﮒ ﺭﺍ ﺑﻪ ﺳﺨﺘﯽ ﻣﯽ ﺗﻮﺍﻥ ﺑﺎ ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﻳﺎ ﺷﺮﺍﮐﺖ ﺍﺩﺍﺭﻩ ﮐﺮﺩ. ﺍﻳﻦ ﺩﻭ ﺳﺎﺧﺘﺎﺭ ﺑﺮﺍﯼ ﺁﻏﺎﺯ ﺑﻪ ﮐﺎﺭ ﻳﮏ ﺷﺮﮐﺖ ﻣﻨﺎﺳﺒﻨﺪ ﺍﻣﺎ ﺑﺮﺍﯼ ﺍﺩﺍﻣﻪ ﮐﺎﺭ ﻣﻨﺎﺳﺐ ﻧﻴﺴﺘﻨﺪ. ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺑﻪ ﺧﺎﻃﺮ ﻣﺴﺌﻮﻟﻴﺖ ﻧﺎﻣﺤﺪﻭﺩ٬ ﻋﻤﺮ ﻣﺤﺪﻭﺩ ﻣﻮﺳﺴﻪ ﻭ ﺩﺷﻮﺍﺭﯼ ﺍﻧﺘﻘﺎﻝ ﺷﺮﮐﺖ ﺑﻪ ﻏﻴﺮ ﺍﺳﺖ.

 

ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ (The Corporation)
ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﻣﻬﻢ ﺗﺮﻳﻦ ﺷﮑﻞ ﺍﺯ ﺍﻧﻮﺍﻉ ﺳﺎﺧﺘﺎﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﻣﻮﺳﺴﺎﺕ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﺧﻮﺩ ﺑﮕﻴﺮﻧﺪ. ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﻳﮏ ﺷﺨﺼﻴﺖ ﺣﻘﻮﻗﯽ ﻣﺠﺰﺍ ﺍﺳﺖ. ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻣﮑﺎﻧﺎﺕ ﺣﻘﻮﻗﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻳﮏ ﺷﺨﺼﻴﺖ ﺣﻘﻴﻘﯽ ﺍﺳﺖ ﺭﺍ ﻣﺴﺘﻘﻼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ٬ ﺷﺮﮐﺖ ﻫﺎﯼ ﺳﻬﺎﻣﯽ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺍﻣﻮﺍﻝ ﻣﻨﻘﻮﻝ ﺭﺍ ﺑﺨﺮﻧﺪ ﻭ ﺑﻔﺮﻭﺷﻨﺪ. ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﻗﺮﺍﺭﺩﺍﺩ ﻣﻨﻌﻘﺪ ﮐﻨﻨﺪ، ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺷﮑﺎﻳﺖ ﮐﻨﻨﺪ ﻭ ﻣﻮﺭﺩ ﺷﮑﺎﻳﺖ ﻗﺮﺍﺭ ﺑﮕﻴﺮﻧﺪ.

ﺗﺎﺳﻴﺲ ﻳﮏ ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﺍﺯ ﺗﺎﺳﻴﺲ ﻳﮏ ﺷﺮﮐﺖ ﺑﺎ ﻣﺎﻟﮑﻴﺖ ﻳﮕﺎﻧﻪ ﻳﺎ ﺷﺮﺍﮐﺖ ﺩﺷﻮﺍﺭﺗﺮ ﺍﺳﺖ. ﺷﺮﮐﺖ ﻣﻌﻤﻮﻻ ﺩﺍﺭﺍﯼ ﻳﮏ ﺍﺳﺎﺳﻨﺎﻣﻪ ﺍﺳﺖ ﮐﻪ ﻧﮑﺎﺕ ﻣﻬﻢ ﺩﺭﺑﺎﺭﻩ ﺷﺮﮐﺖ ﺩﺭ ﺁﻥ ﺗﻌﻴﻴﻦ ﻣﯽ ﺷﻮﺩ. ﮐﻤﭙﺎﻧﯽ ﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﻣﻌﺎﻣﻠﻪ ﻳﺎ ﺗﺒﻠﻴﻎ ﺳﻬﺎﻡ ﺷﺎﻥ ﺑﻪ ﺻﻮﺭﺕ ﻋﻤﻮﻣﯽ ﻣﺠﺎﺯ ﻧﻴﺴﺖ ﺭﺍ ﺷﺮﮐﺖ ﻫﺎﯼ ﺳﻬﺎﻣﯽ ﺧﺎﺹ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﻣﯽ ﮔﻮﻳﻨﺪ.

ﺍﻳﻦ ﺷﺮﮐﺖ ﻫﺎ ﻣﻌﻤﻮﻻ ﺧﻴﻠﯽ ﺑﺰﺭﮒ ﻧﻴﺴﺘﻨﺪ. ﻣﺪﻳﺮ ﺷﺮﮐﺖ ﻣﻌﻤﻮﻻ ﮐﺴﯽ ﺍﺳﺖ ﮐﻪ ﺑﻴﺸﺘﺮﻳﻦ ﺳﻬﺎﻡ ﺭﺍ ﺩﺍﺭﺩ. ﺷﺮﮐﺖ ﻫﺎﯼ ﺳﻬﺎﻣﯽ ﺧﺎﺹ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﻋﻤﻮﻣﺎ ﺗﻮﺳﻂ ﺧﺎﻧﻮﺍﺩﻩ ﻫﺎ ﺍﺩﺍﺭﻩ ﻣﯽ ﺷﻮﻧﺪ. ﺗﺎﺳﻴﺲ ﻳﮏ ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﺧﺎﺹ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﻣﻌﻤﻮﻻ ﮔﺎﻡ ﻣﻬﻤﯽ ﺩﺭ ﺭﺍﻩ ﺗﺒﺪﻳﻞ ﺷﺪﻥ ﺑﻪ ﻳﮏ ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﻋﺎﻡ ﺍﺳﺖ.

ﺷﺮﮐﺖ ﻳﮏ ﻫﻴﺎﺕ ﻣﺪﻳﺮﻩ ﺩﺍﺭﺩ ﻭ ﻫﻴﺎﺕ ﻣﺪﻳﺮﻩ ﺭﺋﻴﺲ ﺩﺍﺭﺩ. ﻫﻴﺎﺕ ﻣﺪﻳﺮﻩ ﺍﯼ ﮐﻪ ﺗﻮﺳﻂ ﺳﻬﺎﻣﺪﺍﺭﺍﻥ ﺍﻧﺘﺨﺎﺏ ﻣﯽ ﺷﻮﺩ ﻣﺪﻳﺮ ﻋﺎﻣﻞ ﺷﺮﮐﺖ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻣﯽ ﮐﻨﺪ. ﺳﺎﺧﺘﺎﺭ ﻫﻴﺎﺕ ﻣﺪﻳﺮﻩ ﻣﯽ ﺗﻮﺍﻧﺪ ﮐﺸﻮﺭ ﺑﻪ ﮐﺸﻮﺭ ﻣﺘﻔﺎﻭﺕ ﺑﺎﺷﺪ.

ﻣﻬﻢ ﺗﺮﻳﻦ ﺗﻔﺎﻭﺕ ﻳﮏ ﻣﺮﺣﻠﻪ ﺍﯼ ﻳﺎ ﺩﻭ ﻣﺮﺣﻠﻪ ﺍﯼ ﺑﻮﺩﻥ ﺳﺎﺧﺘﺎﺭ ﻫﻴﺎﺕ ﻣﺪﻳﺮﻩ ﺍﺳﺖ. ﺩﺭ ﺳﺎﺧﺘﺎﺭ ﻳﮏ ﻣﺮﺣﻠﻪ ﺍﯼ ﮐﻪ ﺑﻴﺸﺘﺮ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﺷﻮﺩ٬ ﻫﻴﺎﺕ ﻣﺪﻳﺮﻩ ﺗﻮﺳﻂ ﺳﻬﺎﻣﺪﺍﺭﺍﻥ ﺍﻧﺘﺨﺎﺏ ﻣﯽ ﺷﻮﺩ.

ﺩﺭ ﺳﺎﺧﺘﺎﺭ ﺩﻭ ﻣﺮﺣﻠﻪ ﺍﯼ ﮐﻪ ﺩﺭ ﺑﺮﺧﯽ ﮐﺸﻮﺭﻫﺎ ﻣﺜﻞ ﺁﻟﻤﺎﻥ، ﻫﻠﻨﺪ، ﺩﺍﻧﻤﺎﺭﮎ ﻭ ﺍﺗﺮﻳﺶ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﺷﻮﺩ٬ ﻳﮏ ﻧﻬﺎﺩ ﻧﻈﺎﺭﺗﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ  ﮐﻪ ﻣﺪﻳﺮﻳﺖ ﺷﺮﮐﺖ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻣﯽ ﮐﻨﺪ ﻭ ﻣﺪﻳﺮﻳﺖ ﺑﻪ ﺁﻥ ﻧﻬﺎﺩ ﭘﺎﺳﺨﮕﻮ ﺍﺳﺖ. ﻫﻴﺌﺖ ﻧﻈﺎﺭﺕ ﻣﺘﺸﮑﻞ ﺍﺳﺖ ﺍﺯ ﻗﺮﺽ ﺩﻫﻨﺪﻩ ﻫﺎ، ﺳﻬﺎﻣﺪﺍﺭﺍﻥ ﻋﻤﺪﻩ، ﻧﻤﺎﻳﻨﺪﮔﺎﻥ ﺍﺗﺤﺎﺩﻳﻪ ﻫﺎ ﻭ... .

ﻣﺰﺍﻳﺎﯼ ﻳﮏ ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﺑﻪ ﺷﺮﺡ ﺯﻳﺮ ﺍﺳﺖ:

۱- ﺳﻬﺎﻡ ﺷﺮﮐﺖ ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﺳﺎﺩﮔﯽ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻣﻨﺘﻘﻞ ﺷﻮﺩ. ﺍﻳﻦ ﺑﺮﺧﻼﻑ ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻣﻮﺳﺴﻪ ﺑﺮﺍﺳﺎﺱ ﺷﺮﺍﮐﺖ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

۲- ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﻋﻤﺮ ﻧﺎﻣﺤﺪﻭﺩﯼ ﺩﺍﺭﺩ ﻭ ﺑﺎ ﻣﺮﮒ ﺷﺮﮐﺎ ﻳﺎ ﻣﺎﻟﮏ ﻳﮕﺎﻧﻪ ﺍﺯ ﺑﻴﻦ ﻧﻤﯽ ﺭﻭﺩ.

۳- ﻣﺴﺌﻮﻟﻴﺖ ﺳﻬﺎﻣﺪﺍﺭﺍﻥ ﻣﺤﺪﻭﺩ ﺍﺳﺖ.

ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﺑﺎ ﻭﺟﻮﺩ ﺍﻳﻦ ﺧﺼﺎﻳﺺ ﻣﺜﺒﺖ٬ ﻳﮏ ﻧﮑﺘﻪ ﻣﻨﻔﯽ ﻫﻢ ﺩﺍﺭﺩ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﮐﺸﻮﺭﻫﺎ ﻫﻢ ﻣﺎﻟﻴﺎﺕ ﺑﺮ ﺷﺮﮐﺖ ﻫﺎ ﺭﺍ ﺩﺍﺭﻧﺪ ﻭ ﻫﻢ ﻣﺎﻟﻴﺎﺕ ﺑﺮ ﺩﺭﺁﻣﺪ ﺍﻓﺮﺍﺩ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ٬ ﺳﻬﺎﻣﺪﺍﺭ ﻳﮏ ﺷﺮﮐﺖ ﺩﻭ ﺑﺎﺭ ﻣﺎﻟﻴﺎﺕ ﻣﯽ ﭘﺮﺩﺍﺯﺩ٬ ﻳﮏ ﺑﺎﺭ ﺍﺯ ﻃﺮﻳﻖ ﺷﺮﮐﺖ ﻭ ﺑﺎﺭ ﺩﻳﮕﺮ ﺑﻪ ﺧﺎﻃﺮ ﺩﺭﺁﻣﺪﺷﺎﻥ ﺍﺯ ﺁﻥ ﺷﺮﮐﺖ.

ﺧﺐ٬ ﺑﮕﺬﺍﺭﻳﺪ ﻣﺴﺌﻠﻪ ﺭﺍ ﻳﮏ ﺑﺎﺭ ﺩﻳﮕﺮ ﻣﺮﻭﺭ ﮐﻨﻴﻢ ﻭ ﺍﻳﻦ ﺑﺎﺭ ﺑﺎ ﻧﮕﺎﻫﯽ ﻭﻳﮋﻩ ﺑﻪ ﺍﻳﺮﺍﻥ.

ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﺧﺎﺹ ﻧﻮﻋﯽ ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﺍﺳﺖ ﮐﻪ ﺳﻬﺎﻡ ﺁﻥ ﻗﺎﺑﻞ ﺩﺍﺩ ﻭ ﺳﺘﺪ ﺩﺭ ﺑﻮﺭﺱ ﺍﻭﺭﺍﻕ ﺑﻬﺎﺩﺍﺭ ﻧﻴﺴﺖ ﻭﻟﯽ ﺑﻴﻦ ﻣﺮﺩﻡ ﺑﻪ ﺭﺍﺣﺘﯽ ﻗﺎﺑﻞ ﻣﻌﺎﻣﻠﻪ ﺍﺳﺖ. ﺗﻤﺎﻡ ﺳﺮﻣﺎﻳﻪ ﺍﻳﻦ ﺷﺮﮐﺖ ﺑﺎﻳﺪ ﻣﻮﻗﻊ ﺗﺄﺳﻴﺲ ﺗﻮﺳﻂ ﻣﻮﺳﺴﺎﻥ ﺗﺎﻣﻴﻦ ﻭ ﺑﺨﺸﯽ ﺍﺯ ﺳﺮﻣﺎﻳﻪ ﺩﺭ ﺑﺎﻧﮏ ﺳﭙﺮﺩﻩ‌ ﮔﺬﺍﺭﯼ ﺷﺪﻩ ﺑﺎﺷﺪ.

ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﻋﺎﻡ ﻧﻮﻋﯽ ﺷﺮﮐﺖ ﺳﻬﺎﻣﯽ ﺍﺳﺖ ﮐﻪ ﺳﻬﺎﻡ ﺁﻥ ﺩﺭ ﺑﻮﺭﺱ ﺍﻭﺭﺍﻕ ﺑﻬﺎﺩﺍﺭ ﺩﺍﺩ ﻭ ﺳﺘﺪ ﻣﯽ‌ﺷﻮﺩ. ﺳﺎﺯﻣﺎﻥ ﻫﺎﯼ ﺑﻮﺭﺱ ﺩﺭ ﺟﻬﺎﻥ ﻣﻌﻤﻮﻻ ﺷﺮﮐﺖ ﻫﺎﻳﯽ ﺭﺍ ﻣﯽ‌ﭘﺬﻳﺮﻧﺪ ﮐﻪ ﺳﺮﻣﺎﻳﻪ ﻭ ﺍﻋﺘﺒﺎﺭ ﺑﺴﻴﺎﺭ ﻗﻮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ.

ﻣﻮﺳﺴﺎﻥ ﺍﻳﻨﮕﻮﻧﻪ ﺷﺮﮐﺖ ﻫﺎ ﻗﺴﻤﺘﯽ ﺍﺯ ﺳﺮﻣﺎﻳﻪ ﺷﺮﮐﺖ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﻓﺮﻭﺵ ﺳﻬﺎﻡ ﺑﻪ ﻋﻤﻮﻡ ﻣﺮﺩﻡ ﺗﺎﻣﻴﻦ ﻣﯽ‌ﮐﻨﻨﺪ. ﺩﺭ ﺍﻳﻦ ﻧﻮﻉ ﺷﺮﮐﺖ ﻫﺎ، ﺻﺎﺣﺒﺎﻥ ﺳﻬﺎﻡ ﻣﯽ‌ ﺗﻮﺍﻧﻨﺪ ﺳﻬﺎﻡ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺑﻮﺭﺱ ﺑﻔﺮﻭﺷﻨﺪ ﻭ ﺩﺭ ﻭﺍﻗﻊ ﻣﺎﻟﮑﻴﺖ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻣﻨﺘﻘﻞ ﮐﻨﻨﺪ.

ﺷﺮﮐﺖ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﻧﻮﻋﯽ ﺷﺮﮐﺖ ﺍﺳﺖ ﮐﻪ ﺑﺮ ﺧﻼﻑ ﺷﺮﮐﺖ ﺗﻀﺎﻣﻨﯽ، ﻣﺴﺌﻮﻟﻴﺖ ﻫﺮ ﻳﮏ ﺍﺯ ﺳﻬﺎﻣﺪﺍﺭﺍﻥ ﺩﺭ ﻣﻮﺭﺩ ﺑﺪﻫﯽ ﻫﺎﯼ ﺷﺮﮐﺖ ﻣﺤﺪﻭﺩ ﺑﻪ ﻣﺒﻠﻎ ﺳﺮﻣﺎﻳﻪ‌ ﮔﺬﺍﺭﯼ ﺷﺪﻩ ﺩﺭ ﺷﺮﮐﺖ ﺍﺳﺖ ﻭ ﺩﺭ ﺻﻮﺭﺕ ﻋﺪﻡ ﮐﻔﺎﻳﺖ ﻣﺠﺒﻮﺭ ﻧﻴﺴﺘﻨﺪ ﺍﻣﻮﺍﻝ ﺷﺨﺼﯽ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ﻣﺼﺮﻑ ﮐﻨﻨﺪ.

ﺑﻪ ﺯﺑﺎﻥ ﺳﺎﺩﻩ‌ﺗﺮ٬ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺷﺮﮐﺖ ﻭﺭﺷﮑﺴﺘﻪ ﺷﻮﺩ ﻭ ﻧﺘﻮﺍﻧﺪ ﺟﻮﺍﺑﮕﻮﯼ ﻃﻠﺒﮑﺎﺭﺍﻥ ﺑﺎﺷﺪ٬ ﺗﻨﻬﺎ ﺳﺮﻣﺎﻳﻪ ﺳﻬﺎﻣﺪﺍﺭﺍﻥ ﺍﺯ ﺑﻴﻦ ﻣﯽ ﺭﻭﺩ ﻭ ﻃﻠﺒﮑﺎﺭﺍﻥ ﻧﻤﯽ ‌ﺗﻮﺍﻧﻨﺪ ﺑﺎﻗﻴﻤﺎﻧﺪﻩ ﻣﻄﺎﻟﺒﺎﺕ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺍﻣﻮﺍﻝ ﺷﺮﮐﺎﯼ ﺷﺮﮐﺖ ﻣﻄﺎﻟﺒﻪ ﮐﻨﻨﺪ.

ﺷﺮﮐﺖ ﺑﺎ ﻣﺴﺌﻮﻟﻴﺖ ﻣﺤﺪﻭﺩ ﺩﺭ ﺍﻳﺮﺍﻥ ﺗﻨﻬﺎ ﺑﺎ ﺩﻭ ﺷﺮﻳﮏ ﻗﺎﺑﻞ ﺛﺒﺖ ﺍﺳﺖ ﻭ ﻫﺮ ﮔﻮﻧﻪ ﻧﻘﻞ ﻭ ﺍﻧﺘﻘﺎﻝ ﺳﻬﺎﻡ ﺁﻥ ﺗﻨﻬﺎ ﺍﺯ ﻃﺮﻳﻖ ﺩﻓﺘﺮ ﺍﺳﻨﺎﺩ ﺭﺳﻤﯽ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﺍﺳﺖ.
