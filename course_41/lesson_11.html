ﺩﺭﺱ ﺍﻭﻝ- ﻣﻌﺎﻳﺐ ﻭ ﻣﺰﺍﻳﺎﯼ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ

ﻇﻬﻮﺭ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺑﺎ ﻣﺸﮑﻼﺕ ﻭ ﮐﺎﺳﺘﯽ ﻫﺎﯼ ﺑﺴﻴﺎﺭﯼ ﻫﻤﺮﺍﻩ ﺑﻮﺩ ﮐﻪ ﺑﻪ ﻣﺮﻭﺭ ﺯﻣﺎﻥ، ﺑﺎ ﭘﻴﺸﺮﻓﺖ ﻫﺎﯼ ﮔﺴﺘﺮﺩﻩ ﺩﺭ ﻋﺮﺻﻪ ﺗﮑﻨﻮﻟﻮﮊﯼ ﺩﻳﺠﻴﺘﺎﻝ ﻭ ﻣﻬﻨﺪﺳﯽ ﺍﭘﺘﻴﮏ، ﺗﺎ ﺣﺪ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﻣﺮﺗﻔﻊ ﺷﺪﻧﺪ. ﺑﻪ ﻃﻮﺭ ﻗﻄﻊ، ﺍﻣﺮﻭﺯﻩ ﻣﺰﺍﻳﺎ ﻭ ﺗﺴﻬﻴﻼﺗﯽ ﮐﻪ ﺗﻮﺳﻂ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻋﮑﺎﺳﺎﻥ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ، ﺍﻳﻦ ﻧﻮﻉ ﺍﺯ ﻋﮑﺎﺳﯽ ﺭﺍ ﺑﻪ ﻣﺮﺍﺗﺐ ﻣﺤﺒﻮﺏ ﺗﺮ ﺍﺯ ﻋﮑﺎﺳﯽ ﺁﻧﺎﻟﻮﮒ ﺳﺎﺧﺘﻪ، ﻳﻪ ﻃﻮﺭﯼ ﮐﻪ ﺭﻓﺘﻪ ﺭﻓﺘﻪ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺑﻪ ﻃﻮﺭ ﮐﺎﻣﻞ ﺟﺎﻳﮕﺰﻳﻦ ﻋﮑﺎﺳﯽ ﺁﻧﺎﻟﻮﮒ ﻣﯽ ﺷﻮﺩ.

ﺍﻟﺒﺘﻪ ﻫﻨﻮﺯ، ﻋﻠﻴﺮﻏﻢ ﻣﺤﺎﺳﻦ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩ ﺁﻥ ﻧﺴﺒﺖ ﺑﻪ ﻋﮑﺎﺳﯽ ﺁﻧﺎﻟﻮﮒ، ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺑﺎ ﻣﺸﮑﻼﺕ ﻭ ﺩﺷﻮﺍﺭﯼ ﻫﺎﻳﯽ ﻧﻴﺰ ﺩﺳﺖ ﺑﻪ ﮔﺮﻳﺒﺎﻥ ﺍﺳﺖ. ﺩﺭ ﺍﺩﺍﻣﻪ ﺑﻪ ﭘﺎﺭﻩ ﺍﯼ ﺍﺯ ﻣﺤﺎﺳﻦ ﻭ ﻣﻌﺎﻳﺐ ﻣﻬﻢ ﺩﺭ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ ﻣﯽ ﭘﺮﺩﺍﺯﻳﻢ.

 

ﻣﺤﺎﺳﻦ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ
- ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺍﻣﮑﺎﻥ ﺩﻳﺪﻥ ﻋﮑﺲ، ﺑﻼﻓﺎﺻﻠﻪ ﭘﺲ ﺍﺯ ﺛﺒﺖ ﺁﻥ ﺭﺍ ﻣﻤﮑﻦ ﻣﯽ ﮐﻨﺪ. ﺍﻳﻦ ﺍﻣﺮ ﻣﺸﮑﻞ ﻏﻴﺮ ﻗﺎﺑﻞ ﭘﻴﺶ ﺑﻴﻨﯽ ﺑﻮﺩﻥ ﻧﺘﺎﻳﺞ ﺭﺍ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺭﻓﻊ ﮐﺮﺩﻩ ﺍﺳﺖ. ﻫﻤﭽﻨﻴﻦ ﻋﮑﺎﺱ ﭘﺲ ﺍﺯ ﺩﻳﺪﻥ ﻋﮑﺲ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺭ ﻣﺤﻞ ﻋﮑﺎﺳﯽ ﻧﻘﺎﺋﺺ ﻣﺮﺑﻮﻁ ﺑﻪ ﺛﺒﺖ ﻋﮑﺲ ﺭﺍ ﺗﺸﺨﻴﺺ ﺩﺍﺩﻩ ﻭ ﺍﻗﺪﺍﻡ ﺑﻪ ﮔﺮﻓﺘﻦ ﻋﮑﺲ ﻫﺎﯼ ﺟﺪﻳﺪ ﺑﺎ ﺗﻨﻈﻴﻤﺎﺕ ﺟﺪﻳﺪ ﮐﻨﺪ ﺗﺎ ﻧﺘﻴﺠﻪ ﻣﻄﻠﻮﺏ ﺣﺎﺻﻞ ﺷﻮﺩ. ﺍﻳﻦ ﺩﺭ ﺣﺎﻟﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻋﮑﺎﺳﯽ ﺁﻧﺎﻟﻮﮒ، ﺍﻣﮑﺎﻥ ﺳﻌﯽ ﻭ ﺧﻄﺎ ﺩﺭ ﻣﺤﻞ ﻋﮑﺎﺳﯽ ﺑﺴﻴﺎﺭ ﭘﺎﻳﻴﻦ ﺍﺳﺖ.

- ﻋﮑﺲ ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﻧﻪ ﺗﻨﻬﺎ ﺍﺯ ﻧﻈﺮ ﮐﻴﻔﻴﺖ ﺑﺴﻴﺎﺭ ﻣﻄﻠﻮﺏ ﺑﻮﺩﻩ ﻭ ﺑﺎ ﻋﮑﺎﺳﯽ ﺁﻧﺎﻟﻮﮒ ﻗﺎﺑﻞ ﺭﻗﺎﺑﺖ ﻫﺴﺘﻨﺪ، ﺑﻠﮑﻪ ﺩﺭ ﺩﻭﺭﺑﻴﻦ ﻫﺎﯼ ﺣﺮﻓﻪ ﺍﯼ، ﮐﻴﻔﻴﺖ ﻋﮑﺲ ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺍﺯ ﺁﻧﺎﻟﻮﮒ ﭘﻴﺸﯽ ﮔﺮﻓﺘﻪ ﺍﺳﺖ.

- ﺩﺭ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ ﻧﻴﺎﺯ ﺑﻪ ﺗﻌﻮﻳﺾ ﺩﺍﺋﻤﯽ ﻓﻴﻠﻢ ﻫﺎ ﻣﺘﻨﺎﺳﺐ ﺑﺎ ﺷﺮﺍﻳﻂ ﻣﺤﻴﻂ ﻧﻴﺴﺖ. ﻫﻤﭽﻨﻴﻦ ﻧﻴﺎﺯ ﻧﻴﺴﺖ ﮐﻪ ﻋﮑﺎﺱ ﻣﻨﺘﻈﺮ ﺗﻤﺎﻡ ﺷﺪﻥ ﻓﻴﻠﻢ ﺑﺮﺍﯼ ﺗﻌﻮﻳﺾ ﺑﺎﺷﺪ ﻳﺎ ﻣﺠﺒﻮﺭ ﺷﻮﺩ ﻳﮏ ﺣﻠﻘﻪ ﺭﺍ ﻧﻴﻤﻪ ﮐﺎﺭﻩ ﺍﺯ ﺩﻭﺭﺑﻴﻦ ﺧﺎﺭﺝ ﮐﻨﺪ. ﺑﻪ ﺭﺍﺣﺘﯽ، ﺩﺭ ﻫﺮ ﺯﻣﺎﻥ ﻭ ﺑﺪﻭﻥ ﺗﻌﻮﻳﺾ ﺻﻔﺤﻪ ﺣﺴﺎﺱ، ﻣﯽ ﺗﻮﺍﻥ ﺗﻨﻈﻴﻤﺎﺕ ﺩﻭﺭﺑﻴﻦ ﺭﺍ ﻣﺘﻨﺎﺳﺐ ﺑﺎ ﺷﺮﺍﻳﻂ ﺗﻐﻴﻴﺮ ﺩﺍﺩ.

- ﺩﺭ ﮐﺎﺭﺕ ﻫﺎﯼ ﺣﺎﻓﻈﻪ ﮐﻮﭼﮏ ﺍﻣﺮﻭﺯﯼ، ﺑﺪﻭﻥ ﻧﻴﺎﺯ ﺑﻪ ﺗﻌﻮﻳﺾ، ﻣﯽ ﺗﻮﺍﻥ ﺗﻌﺪﺍﺩ ﺍﻧﺒﻮﻫﯽ ﺍﺯ ﻓﺎﻳﻞ ﻫﺎﯼ ﻋﮑﺲ ﺭﺍ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ ﮐﺮﺩ ﻭ ﺩﺭ ﻣﻘﺎﻳﺴﻪ ﺑﺎ ﻓﻴﻠﻢ ﺩﺭ ﻋﮑﺎﺳﯽ ﺁﻧﺎﻟﻮﮒ، ﻓﻀﺎﯼ ﺑﺴﻴﺎﺭ ﮐﻤﺘﺮﯼ ﺭﺍ ﺍﺷﻐﺎﻝ ﻣﯽ ﮐﻨﻨﺪ ﻭ ﺑﺴﻴﺎﺭ ﺳﺒﮏ ﺗﺮ ﻫﺴﺘﻨﺪ.

- ﮐﭙﯽ ﮐﺮﺩﻥ ﻭ ﺗﮑﺜﻴﺮ ﺗﺼﺎﻭﻳﺮ ﺩﻳﺠﻴﺘﺎﻝ ﺩﺭ ﺗﻌﺪﺍﺩ ﺑﺎﻻ ﻭ ﺑﺪﻭﻥ ﺍﻓﺖ ﮐﻴﻔﻴﺖ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ ﺍﺳﺖ.

- ﻃﺒﻘﻪ ﺑﻨﺪﯼ، ﺩﺳﺘﺮﺳﯽ ﻭ ﻧﮕﻬﺪﺍﺭﯼ ﺍﺯ ﻓﺎﻳﻞ ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺩﺭ ﻣﻘﺎﻳﺴﻪ ﺑﺎ ﻓﻴﻠﻢ ﻭ ﻋﮑﺲ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ ﺗﺮ ﺍﺳﺖ.

- ﺩﺭ ﻃﻮﻻﻧﯽ ﻣﺪﺕ ﺑﺎ ﺣﺬﻑ ﻫﺰﻳﻨﻪ ﺧﺮﻳﺪ ﻓﻴﻠﻢ، ﭼﺎﭖ ﻋﮑﺲ ﻭ ﺭﻓﺖ ﻭ ﺁﻣﺪ ﺑﺮﺍﯼ ﺁﻧﻬﺎ، ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺑﺴﻴﺎﺭ ﻣﻘﺮﻭﻥ ﺑﻪ ﺻﺮﻓﻪ ﺍﺳﺖ. ﻫﻤﭽﻨﻴﻦ ﺑﺎ ﺣﺬﻑ ﺯﻣﺎﻥ ﻣﻮﺭﺩ ﻧﻴﺎﺯ ﺑﺮﺍﯼ ﻣﺮﺍﺣﻞ ﻓﻮﻕ، ﺩﺭ ﺯﻣﺎﻥ ﻧﻴﺰ ﺻﺮﻓﻪ ﺟﻮﻳﯽ ﻣﯽ ﺷﻮﺩ.

- ﺍﺭﺳﺎﻝ ﻭ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺗﺼﺎﻭﻳﺮ ﺩﻳﺠﻴﺘﺎﻝ ﺗﻮﺳﻂ ﺷﺒﮑﻪ ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻟﯽ، ﺍﻳﻨﺘﺮﻧﺖ، ﺍﻳﻤﻴﻞ ﻭ ﻏﻴﺮﻩ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ ﻭ ﺳﺮﻳﻊ ﺍﺳﺖ. ﺍﻧﺘﻘﺎﻝ ﺍﻳﻦ ﺗﺼﺎﻭﻳﺮ ﻧﻴﺰ ﺑﻪ ﺭﺍﺣﺘﯽ ﻭ ﺑﺎ ﺭﻭﺵ ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﮔﻮﻧﺎﮔﻮﻥ ﺍﺯ ﻗﺒﻴﻞ ﺍﺭﺗﺒﺎﻁ ﺑﯽ ﺳﻴﻢ، ﮐﺎﺑﻞ، ﺍﻧﻮﺍﻉ ﮐﺎﺭﺕ ﻫﺎﯼ ﺣﺎﻓﻈﻪ ﻭ ﺍﻣﮑﺎﻧﺎﺕ ﺫﺧﻴﺮﻩ ﺳﺎﺯﯼ، ﺻﻮﺭﺕ ﻣﯽ ﭘﺬﻳﺮﺩ.

- ﺍﻋﻤﺎﻝ ﺑﺮﺧﯽ ﺍﺯ ﺗﻨﻈﻴﻤﺎﺕ، ﺗﻐﻴﻴﺮﺍﺕ ﻭ ﺟﻠﻮﻩ ﻫﺎﯼ ﺗﺼﻮﻳﺮﯼ ﺩﺭ ﺯﻣﺎﻥ ﺛﺒﺖ ﻋﮑﺲ ﻳﺎ ﺑﻼﻓﺎﺻﻠﻪ ﭘﺲ ﺍﺯ ﺁﻥ، ﺗﻮﺳﻂ ﺩﻭﺭﺑﻴﻦ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﺍﺳﺖ.

- ﺗﺼﺎﻭﻳﺮ ﺩﻳﺠﻴﺘﺎﻝ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﺭﺍﺣﺘﯽ ﻭ ﺩﺭ ﺍﺳﺮﻉ ﻭﻗﺖ ﻭ ﺑﻌﻀﺎ ﺑﺎ ﮐﻴﻔﻴﺖ ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﺑﺎﻻ ﺗﻮﺳﻂ ﭘﺮﻳﻨﺘﺮﻫﺎ ﭼﺎﭖ ﮐﺮﺩ. ﺍﻟﺒﺘﻪ ﻫﻨﻮﺯ ﺑﺮﺍﯼ ﭼﺎﭖ ﻫﺎﯼ ﺣﺮﻓﻪ ﺍﯼ ﻧﻴﺎﺯ ﺑﻪ ﻣﺎﺷﻴﻦ ﻫﺎ ﻭ ﺍﺑﺰﺍﺭ ﺩﻗﻴﻖ ﺍﺳﺖ.

- ﺩﺭ ﺍﻏﻠﺐ ﻣﺮﺍﺣﻞ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ، ﺩﻳﮕﺮ ﻧﻴﺎﺯﯼ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻮﺍﺩ ﺷﻴﻤﻴﺎﻳﯽ ﻣﻀﺮ ﺑﺮﺍﯼ ﺍﻧﺴﺎﻥ ﻭ ﻣﺤﻴﻂ ﺯﻳﺴﺖ ﻧﻴﺴﺖ.

- ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﻣﺤﺎﺳﻦ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ، ﮐﻨﺘﺮﻝ ﻭ ﺍﻣﮑﺎﻧﺎﺕ ﺑﺴﻴﺎﺭ ﮔﺴﺘﺮﺩﻩ ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺯﻣﻴﻨﻪ ﺍﺩﻳﺖ ﻭ ﺗﻨﻈﻴﻤﺎﺕ ﻋﮑﺲ ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻋﮑﺎﺱ ﺍﺳﺖ. ﺍﻳﻦ ﮐﻨﺘﺮﻝ ﻫﺎ ﻭ ﺍﻣﮑﺎﻧﺎﺕ ﺩﺭ ﻋﮑﺎﺳﯽ ﺁﻧﺎﻟﻮﮒ ﺑﺴﻴﺎﺭ ﺩﺷﻮﺍﺭ ﻳﺎ ﻏﻴﺮ ﻣﻤﮑﻦ ﻫﺴﺘﻨﺪ.

- ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺩﻭﺭﺑﻴﻦ ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ، ﻋﻼﻭﻩ ﺑﺮ ﺛﺒﺖ ﻋﮑﺲ، ﻗﺎﺩﺭ ﺑﻪ ﺛﺒﺖ ﻓﻴﻠﻢ ﻭ ﺻﺪﺍ ﻫﻢ ﻫﺴﺘﻨﺪ.

- ﻋﮑﺲ ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﺻﻮﺭﺕ Slide Show ﻭ ﻳﺎ ﺑﻪ ﺻﻮﺭﺕ Multimedia ﻫﻤﺮﺍﻩ ﺑﺎ ﺻﺪﺍ، ﻣﺘﻦ، ﻓﻴﻠﻢ ﻭ ﺳﺎﻳﺮ ﺟﻠﻮﻩ ﻫﺎﯼ ﻫﻨﺮﯼ ﺍﺭﺍﺋﻪ ﺩﺍﺩ.

 

ﻣﻌﺎﻳﺐ ﻭ ﺩﺷﻮﺍﺭﯼ ﻫﺎﯼ ﻋﮑﺎﺳﯽ ﺩﻳﺠﻴﺘﺎﻝ
