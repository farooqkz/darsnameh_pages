ﺍﮐﻨﻮﻥ ﮐﻪ ﻗﺎﻟﺐ ﻭ ﻣﺤﺘﻮﺍﯼ ﮐﻠﯽ ﭘﺎﺩﮐﺴﺖ ﺭﺍ ﻣﺸﺨﺺ ﮐﺮﺩﻳﻢ ﻭ ﻣﯽ‌ﺩﺍﻧﻴﻢ ﮐﻪ ﺩﺭ ﺁﻳﻨﺪﻩ ﻗﺮﺍﺭ ﺍﺳﺖ ﺩﺭﺑﺎﺭﻩ ﭼﻪ ﻣﻮﺿﻮﻋﺎﺕ ﺑﺎ ﺷﻨﻮﻧﺪﮔﺎﻥ ﺻﺤﺒﺖ ﮐﻨﻴﻢ، ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻤﯽ ﺟﺰﺋﯽ‌ﺗﺮ ﺑﻪ ﺩﺍﺳﺘﺎﻥ ﻧﮕﺎﻩ ﮐﻨﻴﻢ ﻭ ﺩﻗﻴﻖ‌ﺗﺮ ﻣﺎﺟﺮﺍ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﻨﻴﻢ. ﺁﻳﺎ ﻣﺸﺘﺎﻕ ﻫﺴﺘﻴﺪ ﺳﺮﻳﻊ‌ﺗﺮ ﺿﺒﻂ ﺍﻭﻟﻴﻦ ﻗﺴﻤﺖ ﭘﺎﺩﮐﺴﺖ ﺟﺬﺍﺑﺘﺎﻥ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﺪ؟ ﺍﻣﺎ ﻫﻨﻮﺯ ﺑﻪ ﮐﻤﯽ ﺁﻣﺎﺩﮔﯽ ﺑﻴﺸﺘﺮ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ، ﺗﺎ ﺑﺎ ﺍﻃﻤﻴﻨﺎﻥ ﺑﻴﺸﺘﺮﯼ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺭﺍ ﺑﺎﻻ ﺁﻭﺭﺩﻩ ﻭ ﺩﮐﻤﻪ ﺿﺒﻂ ﺭﺍ ﻓﺸﺎﺭ ﺩﻫﻴﺪ.

ﻓﻘﻂ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺗﻬﻴﻪ ﺍﻭﻟﻴﻦ ﻗﺴﻤﺖ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ، ﺍﺣﺘﻤﺎﻻ ﺑﺴﻴﺎﺭ ﻃﻮﻻﻧﯽ‌ﺗﺮ ﺍﺯ ﺁﻥ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﮐﻪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ. ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺗﻬﻴﻪ ﭘﺎﺩﮐﺴﺖ ﺑﻴﺶ ﺍﺯ ﺁﻧﭽﻪ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﺯﻣﺎﻥ ﻣﯽ‌ﺑﺮﺩ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﻻﺯﻡ ﺍﺳﺖ ﭘﻴﺶ ﺍﺯ ﺷﺮﻭﻉ ﮐﺎﺭ، ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﻫﺮ ﻗﺴﻤﺖ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﻣﺮﺣﻠﻪ ﺗﻮﻟﻴﺪ ﺳﺮﻳﻊ‌ﺗﺮ ﻭ ﺁﺳﺎﻥ‌ﺗﺮ ﭘﻴﺶ ﺑﺮﻭﺩ ﻭ ﺷﻤﺎ ﺑﺎ ﭘﻴﺶ ﺩﺳﺘﯽ، ﺑﺮﺍﯼ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺸﮑﻼﺕ ﺁﻣﺎﺩﮔﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

 

ﻃﺮﺡ ﮐﻠﯽ ﻭ ﺭﺋﻮﺱ ﺑﺮﻧﺎﻣﻪ ﺧﻮﺩ ﺭﺍ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ
ﻗﺒﻞ ﺍﺯ ﺁﻧﮑﻪ ﻫﺮ ﻗﺴﻤﺖ ﺭﺍ ﺿﺒﻂ ﮐﻨﻴﺪ، ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻃﺮﺡ ﮐﻠﯽ ﺑﺮﺍﯼ ﺁﻥ ﺑﻨﻮﻳﺴﻴﺪ. ﺍﻳﻦ ﻃﺮﺡ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻟﻴﺴﺘﯽ ﺍﺯ ﭼﻴﺰﻫﺎﻳﯽ ﺑﺎﺷﺪ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺩﺭ ﺍﻳﻦ ﻗﺴﻤﺖ ﺍﺯ ﺑﺮﻧﺎﻣﻪ ﺑﮕﻮﻳﻴﺪ. ﻳﮏ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺑﺮ ﺭﻭﯼ ﻣﺴﻴﺮ ﺍﺻﻠﯽ ﺗﻤﺮﮐﺰ ﮐﻨﻴﺪ ﻭ ﺩﺭ ﻣﻴﺎﻧﻪ ﺭﺍﻩ ﺿﺒﻂ ﺑﺮﻧﺎﻣﻪ، ﺳﺮﻧﺦ ﮐﺎﺭ ﺍﺯ ﺩﺳﺘﺘﺎﻥ ﺩﺭ ﻧﺮﻭﺩ.

ﺣﺘﯽ ﺍﮔﺮ ﭘﺎﺩﮐﺴﺖ ﺷﻤﺎ ﮐﺎﻣﻼ ﺳﺎﺩﻩ ﻭ ﺗﻨﻬﺎ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﺪﻭﻥ ﻫﻴﭻ ﮐﻠﻴﭗ ﻳﺎ ﻣﻮﺳﻴﻘﯽ ﺍﺿﺎﻓﻪ‌ﺍﯼ، ﺟﻠﻮﯼ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻧﺸﺴﺘﻪ ﻭ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﺪ؛ ﺑﺎﺯ ﻫﻢ ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻤﯽ ﺩﺭﺑﺎﺭﻩ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ‌ﺗﺎﻥ ﻓﮑﺮ ﮐﺮﺩﻩ ﻭ ﺑﺒﻴﻨﻴﺪ ﭼﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﮕﻮﻳﻴﺪ. ﺯﻳﺮﺍ ﻫﺪﻑ ﺷﻤﺎ ﺗﻮﻟﻴﺪ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺑﺎ ﮐﻴﻔﻴﺖ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺷﻨﻮﻧﺪﮔﺎﻥ ﺭﺍ ﺑﺎﺯ ﻫﻢ ﺑﺮﺍﯼ ﻗﺴﻤﺖ‌ﻫﺎﯼ ﺑﻌﺪﯼ ﺑﻪ ﭘﺎﺩﮐﺴﺖ ﺷﻤﺎ ﺑﺮﮔﺮﺩﺍﻧﺪ ﻭ ﻭﻓﺎﺩﺍﺭ ﻧﮕﻪ ﺩﺍﺭﺩ.

 ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻦ، ﺩﺍﺷﺘﻦ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ، ﻧﻮﺷﺘﻦ ﻣﺘﻦ ﻣﻌﺮﻓﯽ ﺑﺮﻧﺎﻣﻪ ﺭﺍ ﻫﻢ ﺑﺮﺍﻳﺘﺎﻥ ﺁﺳﺎﻥ‌ﺗﺮ ﻣﯽ‌ﮐﻨﺪ. ﺩﺭﺑﺎﺭﻩ ﭘﺴﺖ ﻣﻌﺮﻓﯽ ﻭ ﻧﮑﺎﺗﯽ ﮐﻪ ﺑﺎﻳﺪ ﺩﺭ ﺁﻥ ﺑﮕﻨﺠﺎﻧﻴﺪ ﺑﻌﺪﺍ ﺑﻴﺸﺘﺮ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ. ﺍﻣﺎ ﺍﮐﻨﻮﻥ ﺗﻨﻬﺎ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺍﺷﺎﺭﻩ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﭘﺴﺖ ﻣﻌﺮﻓﯽ ﻫﺮ ﻗﺴﻤﺖ ﺣﺎﻭﯼ ﺧﻼﺻﻪ‌ﺍﯼ ﺍﺯ ﻣﻮﺿﻮﻋﺎﺕ ﺁﻥ ﻗﺴﻤﺖ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻣﺨﺎﻃﺐ ﻣﯽ‌ﮔﻮﻳﺪ ﻗﺮﺍﺭ ﺍﺳﺖ ﺩﺭ ﻓﺎﻳﻞ ﺻﻮﺗﯽ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﺑﺸﻨﻮﺩ.

ﺗﻬﻴﻪ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ ﺗﻨﻬﺎ ﻳﮏ ﺭﺍﻩ ﻭ ﺭﻭﺵ ﻣﺸﺨﺺ ﻧﺪﺍﺭﺩ ﻭ ﺷﻤﺎ ﺁﻥ ﺭﺍ ﺑﻪ ﻫﺮ ﺷﮑﻠﯽ ﮐﻪ ﺑﺮﺍﻳﺘﺎﻥ ﮐﺎﺭﺍﻳﯽ ﺑﻬﺘﺮﯼ ﺩﺍﺭﺩ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﺗﻨﻬﺎ ﻻﺯﻡ ﺍﺳﺖ ﻃﯽ ﻣﺮﺍﺣﻞ ﺿﺒﻂ ﺑﺮﻧﺎﻣﻪ ﺁﻥ ﺭﺍ ﺑﻪ ﺭﺍﺣﺘﯽ ﺧﻮﺍﻧﺪﻩ ﻭ ﺩﺭﮎ ﮐﻨﻴﺪ. ﺑﻬﺘﺮ ﺍﺳﺖ ﺁﻥ ﺭﺍ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺳﺎﺯﻣﺎﻧﺪﻫﯽ ﮐﻨﻴﺪ ﮐﻪ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﻣﻨﻄﻘﯽ ﺍﺳﺖ ﻭ ﺑﻬﺘﺮ ﺩﺭﮐﺶ ﻣﯽ‌ﮐﻨﻴﺪ.

ﺑﺮﺧﯽ ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻫﻨﺪ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ ﺭﺍ ﺩﺭ ﻣﺎﻳﮑﺮﻭﺳﺎﻓﺖ ﻭﺭﺩ ﺗﺎﻳﭗ ﮐﺮﺩﻩ ﻭ ﺑﺮﺍﯼ ﻣﺮﺗﺐ ﮐﺮﺩﻥ ﺁﻥ ﺍﺯ ﺗﻨﻈﻴﻢ  outline view  ﺑﺮﻧﺎﻣﻪ ﻭﺭﺩ ﻣﺎﻳﮑﺮﻭﺳﺎﻓﺖ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪ. ﻋﮑﺲ ﺯﻳﺮ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﻧﻤﻮﻧﻪ‌ﺍﯼ ﻣﻨﺎﺳﺐ ﻭ ﺗﻘﺮﻳﺒﺎ ﺍﻳﺪﻩ ﺁﻝ ﺍﺯ ﻳﮏ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ ﺑﺮﺍﯼ ﭘﺎﺩﮐﺴﺖ ﺩﺍﻧﺴﺖ ﮐﻪ ﺑﻪ ﺧﻮﺑﯽ ﺟﺰﺋﻴﺎﺕ ﺭﺍ ﺷﺮﺡ ﺩﺍﺩﻩ ﻭ ﻣﺮﺗﺐ ﻫﻢ ﺷﺪﻩ ﺍﺳﺖ:

 

ﺍﻣﺎ ﺑﺎﺯ ﻫﻢ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﮐﻪ ﺑﻪ ﺩﻧﺒﺎﻝ ﭼﻴﺰﯼ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﺮﺍﻳﺘﺎﻥ ﮐﺎﺭﺍﻳﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﮔﺮ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺪﻭﻥ ﻳﺎﺩﺩﺍﺷﺖ ﺑﺮﺩﺍﺭﯼ ﻭ ﺩﺍﺷﺘﻦ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ، ﺗﻨﻬﺎ ﺑﺎ ﻳﺎﺭﯼ ﮔﺮﻓﺘﻦ ﺍﺯ ﺫﻫﻨﺘﺎﻥ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﭘﻴﺶ ﺑﺒﺮﻳﺪ، ﺑﺴﻴﺎﺭ ﻋﺎﻟﯽ ﺍﺳﺖ ﻭ ﺑﻬﺘﺮ ﺍﺳﺖ ﺩﺳﺖ ﺑﻪ ﮐﺎﺭ ﺷﻮﻳﺪ. ﺑﺮﺧﯽ ﭘﺎﺩﮐﺴﺖ ﺳﺎﺯ‌ﻫﺎ ﺗﻨﻬﺎ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺭﺍ ﺭﻭﺷﻦ ﮐﺮﺩﻩ ﻭ ﮐﺎﺭ ﺭﺍ ﺷﺮﻭﻉ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﻟﺒﺘﻪ ﺑﺮﺧﯽ ﺍﻭﻗﺎﺕ ﺍﻳﻦ ﺷﻴﻮﻩ ﺧﻮﺏ ﺍﺯ ﮐﺎﺭ ﺩﺭ ﻣﯽ‌ﺁﻳﺪ ﻭ ﮔﺎﻫﯽ ﺍﻭﻗﺎﺕ ﻫﻢ ﺣﺴﺎﺑﯽ ﺍﻭﺿﺎﻉ ﺑﻪ ﻫﻢ ﻣﯽ‌ﺭﻳﺰﺩ. ﻟﺬﺍ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺍﺯ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ، ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ ﮐﻪ ﻫﻢ ﮐﺎﺭ ﺿﺒﻂ ﺭﺍﺣﺖ‌ﺗﺮ ﭘﻴﺶ ﺧﻮﺍﻫﺪ ﺭﻓﺖ ﻭ ﻫﻢ ﺍﻳﻨﮑﻪ ﻧﺘﻴﺠﻪ ﮐﺎﺭ ﺑﻬﺘﺮ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﺩﺭ ﺗﺼﻮﻳﺮ ﺑﺎﻻ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﮐﻠﻴﭗ‌ﻫﺎﯼ ﺻﻮﺗﯽ ﻣﻴﺎﻥ ﺑﺮﻧﺎﻣﻪ ﺑﺎ ﺭﻧﮓ ﻗﺮﻣﺰ ﻣﺸﺨﺺ ﺷﺪﻩ‌ﺍﻧﺪ. ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﮐﻠﻴﭗ‌ﻫﺎﯼ ﻣﻴﺎﻥ ﺑﺮﻧﺎﻣﻪ ﺍﺯ ﺩﻳﮕﺮ ﻓﺎﻳﻞ‌ﻫﺎﯼ ﺻﻮﺗﯽ ﻭ ﻣﻮﺳﻴﻘﯽ ﮐﻪ ﻃﯽ ﺑﺮﻧﺎﻣﻪ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﻴﺪ، ﻣﺘﻤﺎﻳﺰ ﺷﺪﻩ ﻭ ﮐﺎﺭ ﺗﺸﺨﻴﺺ ﺁﻧﻬﺎ ﺭﺍﺣﺖ‌ﺗﺮ ﻣﯽ‌ﺷﻮﺩ. ﺷﺎﻳﺪ ﺷﻤﺎ ﺑﺨﻮﺍﻫﻴﺪ ﺩﺭ ﺑﺮﻧﺎﻣﻪ‌ﺗﺎﻥ ﺍﺯ ﻳﮏ ﮐﻠﻴﭗ ﻣﺼﺎﺣﺒﻪ ﺍﺯ ﭘﻴﺶ ﺿﺒﻂ ﺷﺪﻩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﻳﺎ ﺍﻳﻨﮑﻪ ﺑﺨﻮﺍﻫﻴﺪ ﺑﺨﺸﯽ ﺍﺯ ﻳﮏ ﻣﻮﺳﻴﻘﯽ ﺭﺍ ﭘﺨﺶ ﮐﺮﺩﻩ ﻭ ﺩﺭﺑﺎﺭﻩ ﺁﻥ ﺻﺤﺒﺖ ﮐﻨﻴﺪ. ﺍﻳﻦ ﮔﻮﻧﻪ ﺩﻳﮕﺮ، ﻣﻮﺳﻴﻘﯽ‌ﻫﺎﯼ ﻣﻴﺎﻥ ﺑﺮﻧﺎﻣﻪ ﺑﺎ ﺁﻥ ﻓﺎﻳﻞ‌ﻫﺎ ﺍﺷﺘﺒﺎﻩ ﮔﺮﻓﺘﻪ ﻧﻤﯽ‌ﺷﻮﻧﺪ.

ﻧﮑﺘﻪ ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺧﻼﺻﻪ ﺑﺮﻧﺎﻣﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﺴﺒﺖ ﺑﻪ ﻧﻤﻮﻧﻪ ﺫﮐﺮ ﺷﺪﻩ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ، ﺟﺰﺋﻴﺎﺕ ﮐﻤﺘﺮ ﻳﺎ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺴﺘﮕﯽ ﺑﻪ ﭘﻴﭽﻴﺪﮔﯽ ﺁﻥ ﻗﺴﻤﺖ ﺍﺯ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﻭ ﻫﻤﭽﻨﻴﻦ ﻓﺮﻡ ﺭﺳﻤﯽ ﻳﺎ ﻏﻴﺮ ﺭﺳﻤﯽ ﺩﺍﺭﺩ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻩ ﻭ ﭘﻴﺶ ﺑﺒﺮﻳﺪ.

 

ﻳﮏ ﻧﻤﺎﻳﺸﻨﺎﻣﻪ ﺑﻨﻮﻳﺴﻴﺪ
ﺷﺎﻳﺪ ﺩﻭﺳﺖ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﮐﻞ ﺩﺍﺳﺘﺎﻥ ﻫﺮ ﻗﺴﻤﺖ ﭘﺎﺩﮐﺴﺖ ﺭﺍ ﮐﻠﻤﻪ ﺑﻪ ﮐﻠﻤﻪ ﺭﻭﯼ ﮐﺎﻏﺬ ﺁﻭﺭﺩﻩ ﻭ ﺁﻥ ﺭﺍ ﺑﻨﻮﻳﺴﻴﺪ. ﺍﻳﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎﻋﺚ ﺷﻮﺩ ﮐﻪ ﻫﻤﻪ ﭼﻴﺰ ﺧﻴﻠﯽ ﻗﺮﺹ ﻭ ﻣﺤﮑﻢ ﻭ ﺍﺗﻮ ﮐﺸﻴﺪﻩ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﺪ. ﺑﺮﺧﯽ ﺍﻓﺮﺍﺩ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﻨﺪ ﺑﻪ ﻳﮏ ﻣﺘﻦ ﺩﻗﻴﻖ ﻧﻴﺎﺯ ﺩﺍﺭﻧﺪ. ﺑﺮﺧﯽ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺯ ﺭﻭﯼ ﻳﮏ ﻣﺘﻦ ﺑﺨﻮﺍﻧﻨﺪ، ﺍﻣﺎ ﮐﺎﻣﻼ ﻃﺒﻴﻌﯽ ﻭ ﺭﻭﺍﻥ ﻭ ﻓﯽ ﺍﻟﺒﺪﺍﻫﻪ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﻨﺪ.

ﺧﻼﺻﻪٔ ﺑﺮﻧﺎﻣﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺷﻤﺎ ﺭﺍ ﺩﺭ ﻣﺴﻴﺮ ﺻﺤﻴﺢ ﺑﺮﻧﺎﻣﻪ ﻧﮕﻪ ﺩﺍﺭﺩ. ﺍﻣﺎ ﺷﺎﻳﺪ ﺑﺨﻮﺍﻫﻴﺪ ﺑﺪﻭﻥ ﻣﺘﻦ ﮐﺎﻣﻞ، ﮔﻔﺘﻪ‌ﻫﺎ ﺟﻠﻮ ﺭﻓﺘﻪ ﻭ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﮐﺎﻣﻼ ﺩﺭ ﻟﺤﻈﻪ ﻭ ﺍﺯ ﺩﺭﻭﻥ ﺫﻫﻨﺘﺎﻥ ﺑﮕﻮﻳﻴﺪ. ﺑﺮﺧﯽ ﻫﻢ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﻳﮏ ﻣﺘﻦ ﺩﻗﻴﻖ ﻭ ﮐﺎﻣﻞ ﺭﺍ ﺩﻧﺒﺎﻝ ﮐﺮﺩﻩ ﻭ ﺑﺮﻧﺎﻣﻪ ﺭﺍ ﺑﺎ ﺭﻭﺧﻮﺍﻧﯽ ﺁﻥ ﺿﺒﻂ ﮐﻨﻨﺪ. ﺑﺎ ﺳﻌﯽ ﻭ ﺧﻄﺎ، ﺑﺒﻴﻨﻴﺪ ﮐﺪﺍﻡ ﻳﮏ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﻬﺘﺮ ﺍﺳﺖ ﻭ ﺑﺮﻧﺎﻣﻪ ﺑﯽ‌ﻧﻘﺺ ﺗﺮﯼ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ.

 

 ﺿﺒﻂ ﺑﺮﻧﺎﻣﻪ ﺑﻪ ﺻﻮﺭﺕ ﻳﮑﺠﺎ ﻳﺎ ﻭﻳﺮﺍﻳﺶ ﺁﻥ ﭘﺲ ﺍﺯ ﺿﺒﻂ ﻭ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ؟
ﺩﻭ ﺷﻴﻮﻩ ﺑﺮﺍﯼ ﺗﻬﻴﻪ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﻭﺟﻮﺩ ﺩﺍﺭﺩ: ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺻﻮﺭﺕ ﻳﮏ ﺟﺎ ﺗﻤﺎﻡ ﺑﺮﻧﺎﻣﻪ ﺭﺍ ﭘﺸﺖ ﺳﺮ ﻫﻢ ﺿﺒﻂ ﮐﺮﺩﻩ ﻭ ﮐﺎﺭ ﺭﺍ ﺗﻤﺎﻡ ﮐﻨﻴﺪ (on the fly) ﻳﺎ ﺍﻳﻨﮑﻪ ﺁﻥ ﺭﺍ ﺩﺭ ﺑﺨﺶ‌ﻫﺎﯼ ﺟﺪﺍﮔﺎﻧﻪ‌ﺍﯼ ﺿﺒﻂ ﮐﺮﺩﻩ ﻭ ﺳﭙﺲ ﺑﺎ ﻭﻳﺮﺍﻳﺶ ﻭ ﭘﺸﺖ ﺳﺮ ﻫﻢ ﮔﺬﺍﺭﺩﻥ ﺁﻧﻬﺎ (Post-Production) ﺑﺮﻧﺎﻣﻪ ﺭﺍ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ.

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﻪ ﺷﻴﻮﻩ On The Fly ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻴﺪ، ﺑﺎﻳﺪ ﺗﻤﺎﻡ ﻓﺎﻳﻞ‌ﻫﺎﯼ ﺻﻮﺗﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﺑﺮﻧﺎﻣﻪ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﻧﺪ، ﺍﺯ ﻗﺒﻞ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﭘﺨﺶ ﻳﮏ ﻣﻮﺳﻴﻘﯽ ﻣﻴﺎﻥ ﺑﺮﻧﺎﻣﻪ ﻫﻤﺰﻣﺎﻥ ﺑﺎ ﺿﺒﻂ، ﺩﮐﻤﻪ ﭘﺨﺶ ﺑﺮﻧﺎﻣﻪ ﻣﻮﺳﻴﻘﯽ ﺭﺍ ﺯﺩﻩ ﻭ ﺯﻣﺎﻥ ﻣﻨﺎﺳﺒﯽ ﺭﺍ ﺑﺮﺍﯼ ﺁﻥ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ. ﻭ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻫﻤﻪ ﭼﻴﺰ ﺑﻪ ﻃﻮﺭ ﺩﻗﻴﻖ ﻭ ﺑﺪﻭﻥ ﻧﻘﺺ ﻫﻤﺎﻧﻨﺪ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺭﺍﺩﻳﻮﻳﯽ ﺑﺎﻳﺪ ﭘﺸﺖ ﺳﺮ ﻫﻢ ﭘﻴﺶ ﺑﺮﻭﺩ.

ﺍﻳﻦ ﮐﺎﺭ ﻧﻴﺎﺯ ﺑﻪ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﻭ ﺩﻗﺖ ﺯﻳﺎﺩﯼ ﺩﺍﺭﺩ. ﺍﮔﺮ ﺷﻤﺎ ﻣﺮﺗﮑﺐ ﺍﺷﺘﺒﺎﻫﯽ ﺷﻮﻳﺪ، ﺑﺴﻴﺎﺭ ﻣﺸﮑﻞ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻋﻘﺐ ﺑﺮﮔﺸﺘﻪ ﻭ ﺁﻥ ﺭﺍ ﻭﻳﺮﺍﻳﺶ ﮐﻨﻴﺪ. ﺯﻳﺮﺍ ﺑﺎﻳﺪ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﺍﺯ ﺍﺑﺘﺪﺍ ﺿﺒﻂ ﮐﻨﻴﺪ. ﺷﺎﻳﺪ ﻫﻢ ﻣﺠﺒﻮﺭ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﺷﺘﺒﺎﻩ ﺭﺍ ﺑﻪ ﺣﺎﻝ ﺧﻮﺩ‌‌ ﺭﻫﺎ ﮐﺮﺩﻩ ﻭ ﺑﺮﻧﺎﻣﻪ ﺭﺍ ﺑﺎ‌‌ ﻫﻤﺎﻥ ﺍﻳﺮﺍﺩ ﻣﻨﺘﺸﺮ ﮐﻨﻴﺪ.

ﺍﻟﺒﺘﻪ ﺍﺯ ﺳﻮﯼ ﺩﻳﮕﺮ ﺍﻳﻦ ﺭﻭﺵ ﺩﺭ ﻭﻗﺖ ﺷﻤﺎ ﺻﺮﻓﻪ ﺟﻮﻳﯽ ﻓﺮﺍﻭﺍﻧﯽ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺷﻤﺎ ﺗﻨﻬﺎ ﻳﮏ ﺑﺎﺯﻩ ﺯﻣﺎﻧﯽ ﺭﺍ ﺑﺮﺍﯼ ﺿﺒﻂ ﺻﺮﻑ ﻣﯽ‌ﮐﻨﻴﺪ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﮐﺎﺭ ﻧﻴﺎﺯ ﺑﻪ ﺗﺠﺮﺑﻪ ﺩﺍﺭﺩ. ﻻﺯﻡ ﺍﺳﺖ ﺷﻤﺎ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺑﺎ ﻓﺮﺁﻳﻨﺪ ﺿﺒﻂ ﻭ ﮐﺎﺭﻫﺎﯼ ﺁﻥ ﺍﺣﺴﺎﺱ ﺭﺍﺣﺘﯽ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺁﻥ ﻣﺴﻠﻂ ﺷﺪﻩ ﺑﺎﺷﻴﺪ.

ﺑﺮﺍﯼ ﺷﺮﻭﻉ، ﭘﻴﺸﻨﻬﺎﺩ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺮﻧﺎﻣﻪ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺑﺨﺶ‌ﻫﺎﯼ ﺟﺪﺍﮔﺎﻧﻪ ﺿﺒﻂ ﮐﺮﺩﻩ ﻭ ﺁﻧﮕﺎﻩ ﻓﺎﻳﻞ‌ﻫﺎﯼ ﺻﻮﺗﯽ ﺭﺍ ﮐﻨﺎﺭ ﻫﻢ ﮔﺬﺍﺭﺩﻩ ﻭ ﻭﻳﺮﺍﻳﺶ ﮐﻨﻴﺪ. ﺩﺭ ‌‌ﻧﻬﺎﻳﺖ ﺷﻤﺎ ﻳﮏ ﻓﺎﻳﻞ ﺷﺴﺘﻪ ﺭﻓﺘﻪ ﻭ ﺧﻮﺏ ﺑﺮﺍﯼ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﺷﻴﻮﻩ Post-Production ﺷﺎﻣﻞ ﺗﻤﺎﻡ ﮐﺎﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﭘﺲ ﺍﺯ ﺍﺗﻤﺎﻡ ﺿﺒﻂ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ ﺗﺎ ﻓﺎﻳﻞ ﻧﻬﺎﻳﯽ ﺭﺍ ﺁﻣﺎﺩﻩ ﺍﻧﺘﺸﺎﺭ ﺩﺭ ﺍﻳﻨﺘﺮﻧﺖ ﮐﻨﻴﺪ.

ﭘﺲ ﺍﺯ ﺍﺗﻤﺎﻡ ﺿﺒﻂ، ﺑﻪ ﺯﻣﺎﻥ ﺑﺴﻴﺎﺭ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﻓﺎﻳﻞ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ. ﺑﺨﺶ ﺿﺒﻂ ﺟﺬﺍﺏ‌ﺗﺮﻳﻦ ﻟﺤﻈﺎﺕ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﭘﺎﺩﮐﺴﺖ ﻣﺤﺴﻮﺏ ﻣﯽ‌ﺷﻮﺩ. ﻭﻗﺖ‌ﻫﺎﻳﯽ ﮐﻪ ﭘﺲ ﺍﺯ ﺁﻥ ﺻﺮﻑ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﻓﺎﻳﻞ ﻣﯽ‌ﺷﻮﺩ، ﺣﺴﺎﺑﯽ ﺣﻮﺻﻠﻪ‌ﺗﺎﻥ ﺭﺍ ﺳﺮ ﺑﺮﺩﻩ ﻭ ﺧﺴﺘﻪ ﮐﻨﻨﺪﻩ ﻫﺴﺘﻨﺪ.

ﻗﺮﺍﺭ ﺩﺍﺩﻥ ﻗﻄﻌﺎﺕ ﺑﺮﻧﺎﻣﻪ ﺩﺭ ﮐﻨﺎﺭ ﻳﮑﺪﻳﮕﺮ ﺩﺭ ﺷﻴﻮﻩ Post-Production ﺑﻪ ﺯﻣﺎﻥ ﺑﻴﺸﺘﺮﯼ ﻧﻴﺎﺯ ﺩﺍﺭﺩ. ﺍﻣﺎ ﺑﻪ ﺷﻤﺎ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﺪ ﺗﺎ ﺩﺭ ﻫﺮ ﺯﻣﺎﻥ ﺗﻨﻬﺎ ﻳﮏ ﻗﺪﻡ ﺑﺮﺩﺍﺭﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺷﻴﻮﻩ ﻓﺸﺎﺭ ﺑﺴﻴﺎﺭ ﮐﻤﺘﺮﯼ ﺍﺯ ﻧﻈﺮ ﺍﻧﺠﺎﻡ ﺻﺤﻴﺢ ﮐﺎﺭ ﻣﺘﺤﻤﻞ ﺧﻮﺍﻫﻴﺪ ﺷﺪ. ﺯﻳﺮﺍ ﺍﺻﻼﺡ ﻭ ﻭﻳﺮﺍﻳﺶ ﺍﺷﺘﺒﺎﻫﺎﺕ ﻭ ﺣﺘﯽ ﺣﺬﻑ ﺁﻧﻬﺎ ﻭ ﺿﺒﻂ ﺩﻭﺑﺎﺭﻩ ﻗﺪﻡ ﺑﺴﻴﺎﺭ ﺁﺳﺎﻥ ﺗﺮﯼ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.
