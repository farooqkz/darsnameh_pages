ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ
ﻣﯽ‌ﺩﺍﻧﻢ ﮐﺎﻣﻼ ﻣﺸﺘﺎﻕ ﻫﺴﺘﻴﺪ ﺗﺎ ﺩﮐﻤﻪ ﺿﺒﻂ ﺑﺮﻧﺎﻣﻪ ﺭﺍ ﻓﺸﺮﺩﻩ ﻭ ﺻﺪﺍﻳﺘﺎﻥ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺍﻳﻨﺘﺮﻧﺖ ﺑﻪ ﮔﻮﺵ ﻫﻤﻪ ﺑﺮﺳﺎﻧﻴﺪ. ﺍﻣﺎ ﮐﻤﯽ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﺗﺎ ﺩﺭ ﺁﻳﻨﺪﻩ ﺑﺎ ﺗﻤﺮﮐﺰ ﺑﻴﺸﺘﺮﯼ ﺑﻪ ﮐﺎﺭﺗﺎﻥ ﺍﺩﺍﻣﻪ ﺩﻫﻴﺪ ﻭﺩﺭ ﻧﺘﻴﺠﻪ ﻣﺤﺼﻮﻝ ﺑﻬﺘﺮﯼ ﻣﻨﺘﺸﺮ ﻭ ﭘﺨﺶ ﻧﻤﺎﻳﻴﺪ ﺗﺎ ﮐﻪ ﺷﻨﻮﻧﺪﮔﺎﻥ ﻭ ﻃﺮﻓﺪﺍﺭﺍﻥ ﺑﻴﺸﺘﺮ ﻭ ﻭﻓﺎﺩﺍﺭﺗﺮﯼ ﺭﺍ ﺑﻪ ﺧﻮﺩ ﺟﺬﺏ ﮐﻨﻴﺪ. ﻫﻤﭽﻨﻴﻦ ﺑﺎ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﻣﻨﺎﺳﺐ، ﮐﺎﺭ ﺷﻤﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﺗﻬﻴﻪ ﮐﻨﻨﺪﻩ ﺑﺴﻴﺎﺭ ﺭﺍﺣﺖ‌ﺗﺮ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﺩﺭ ﺑﺨﺶ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ ﮐﻪ ﭼﮕﻮﻧﻪ ﺩﺭﺑﺎﺭﻩ ﺑﺮﺧﯽ ﻣﻮﺍﺭﺩ ﻣﻬﻢ ﺩﺭ ﺯﻣﻴﻨﻪ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ، ﻫﻤﭽﻮﻥ ﻣﻮﺿﻮﻉ ﻭ ﻓﺮﻣﺖ ﻳﺎ ﻗﺎﻟﺐ ﭘﺎﺩﮐﺴﺖ ﺗﺼﻤﻴﻢ ﮔﻴﺮﯼ ﮐﻨﻴﺪ. ﻫﻤﭽﻨﻴﻦ ﺑﻪ ﺷﻴﻮﻩ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﻫﺮ ﺍﭘﻴﺰﻭﺩ ﻳﺎ ﻗﺴﻤﺖ ﭘﺎﺩﮐﺴﺖ ﻗﺒﻞ ﺍﺯ ﺿﺒﻂ ﺁﻥ ﻫﻢ ﻧﮕﺎﻫﯽ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺷﺖ.

 

ﺗﻮﻟﻴﺪ
ﺍﻳﻦ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺭﺍ ﺭﻭﺷﻦ ﮐﺮﺩﻩ ﻭ ﺷﺮﻭﻉ ﺑﻪ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﺪ. ﺷﺎﻳﺪ ﻫﻢ ﻧﻘﺸﻪ ﺩﻳﮕﺮﯼ ﺑﻪ ﺟﺎﯼ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺍﻟﺒﺘﻪ ﻗﺒﻞ ﺍﺯ ﺍﻳﻦ، ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﺳﻮﻡ ﺗﺎ ﭘﻨﺠﻢ ﺩﺭ ﺧﺼﻮﺹ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﻣﮑﺎﻥ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﻭ ﺗﺠﻬﻴﺰﺍﺕ ﺳﺨﺖ‌ﺍﻓﺰﺍﺭﯼ ﻭ ﻧﺮﻡ‌ﺍﻓﺰﺍﺭﯼ ﺁﻥ ﺻﺤﺒﺖ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﻭ ﻣﻄﻤﺌﻨﺎ ﺷﻤﺎ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﺩﻡ ﺩﺳﺖ ﺣﺎﺿﺮ ﮐﺮﺩﻩ‌ﺍﻳﺪ.

 

ﺍﻧﺘﺸﺎﺭ
ﻫﻨﮕﺎﻣﯽ ﺍﻭﻟﻴﻦ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺭﺍ ﺗﻮﻟﻴﺪ ﮐﺮﺩﻳﺪ، ﻻﺯﻡ ﺍﺳﺖ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺍﻧﺘﺸﺎﺭ ﺁﻥ ﺩﺭ ﺍﻳﻨﺘﺮﻧﺖ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ. ﺍﻧﺘﺸﺎﺭ ﭘﺎﺩﮐﺴﺖ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﻣﺮﺍﺣﻠﯽ ﻫﻤﭽﻮﻥ ﺗﻮﻟﻴﺪ ﻓﺎﻳﻞ mp3، ﺩﺭﺝ ﺍﻃﻼﻋﺎﺕ ﺗﮓ‌ﻫﺎﯼ ID3، ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﻓﻀﺎﯼ ﻣﻴﺰﺑﺎﻧﯽ ﺭﺍﻳﮕﺎﻥ ﻳﺎ ﻏﻴﺮ ﺭﺍﻳﮕﺎﻥ، ﻧﺎﻣﮕﺬﺍﺭﯼ ﻓﺎﻳﻞ‌ﻫﺎﯼ ﭘﺎﺩﮐﺴﺖ، ﺁﭘﻠﻮﺩ ﮐﺮﺩﻥ ﻓﺎﻳﻞ‌ﻫﺎ، ﻧﻮﺷﺘﻦ ﻣﺘﻦ ﻣﻌﺮﻓﯽ ﻭ ﺍﻧﺘﺸﺎﺭ ﺁﻥ ﺩﺳﺘﻪ ﺑﻨﺪﯼ ﮐﺮﺩ.

 

ﺗﺮﻭﻳﺞ ﻭ ﺗﺒﻠﻴﻎ
ﺑﺪﻭﻥ ﺷﮏ ﺷﻤﺎ ﻫﻤﻴﺸﻪ ﺩﻧﺒﺎﻝ ﺷﻨﻮﻧﺪﮔﺎﻥ ﺑﻴﺸﺘﺮ ﻭ ﻭﻓﺎﺩﺍﺭﺗﺮﯼ ﺑﺮﺍﯼ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﻫﺴﺘﻴﺪ ﻭ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺩﺭ ﺍﻳﻨﺘﺮﻧﺖ ﺑﻪ ﻓﺮﺩﯼ ﺷﻨﺎﺧﺘﻪ ﺷﺪﻩ ﻭ ﭘﺮﻃﺮﻓﺪﺍﺭ ﺗﺒﺪﻳﻞ ﺷﻮﻳﺪ ﮐﻪ ﻫﻤﻪ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺷﻤﺎ ﺭﺍ ﺩﻧﺒﺎﻝ ﻣﯽ‌ﮐﻨﻨﺪ، ﺍﻳﻦ ﻃﻮﺭ ﻧﻴﺴﺖ؟ ﭘﺲ ﺩﺭ ﺑﺨﺸﯽ ﺍﺯ ﺩﻭﺭﻩ‌ﻣﺎﻥ ﻫﻢ ﺑﺎﻳﺪ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﭙﺮﺩﺍﺯﻳﻢ ﮐﻪ ﭼﮕﻮﻧﻪ ﺷﻨﻮﻧﺪﻩ‌ﻫﺎﯼ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﭘﺎﺩﮐﺴﺘﻤﺎﻥ ﺑﻴﺎﺑﻴﻢ.

 

ﺧﺐ ﺑﺮﺍﯼ ﺷﺮﻭﻉ ﺁﻣﺎﺩﻩ ﻫﺴﺘﻴﺪ؟ ﭘﺲ ﺑﻴﺎﻳﻴﺪ ﺩﺭ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺟﺎﻣﻊ ﺑﺮﺍﯼ ﭘﺎﺩﮐﺴﺖ ﻣﻮﺭﺩ ﻋﻼﻗﻪ‌ﻣﺎﻥ ﺑﺮﻳﺰﻳﻢ.

ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﺑﺎ ﮐﻤﯽ ﻭﻗﺖ ﮔﺬﺍﺭﺩﻥ ﻭ ﻓﮑﺮ ﮐﺮﺩﻥ ﺭﻭﯼ ﺍﻳﻦ ﺳﻮﺍﻻﺕ، ﭘﺎﺩﮐﺴﺖ ﺷﻤﺎ ﺩﺭ ﺁﻳﻨﺪﻩ ﺑﺴﻴﺎﺭ ﭘﺮﺑﺎﺭ‌ﺗﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺑﺎ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺻﺤﻴﺢ ﻭ ﻣﻨﺎﺳﺐ، ﺷﻨﻮﻧﺪﮔﺎﻥ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﻫﻢ ﺧﻮﺍﻫﺪ ﻳﺎﻓﺖ. ﺍﻳﻦ ﮐﺎﺭ ﻫﻤﭽﻨﻴﻦ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺷﻤﺎ ﺗﻤﺮﮐﺰﺗﺎﻥ ﺭﺍ ﺣﻔﻆ ﮐﻨﻴﺪ ﻭ ﮐﺎﺭﺗﺎﻥ ﺭﺍ ﺩﺭ ﻃﻮﻻﻧﯽ ﻣﺪﺕ ﺟﺪﯼ ﮔﺮﻓﺘﻪ ﻭ ﺁﻥ ﺭﺍ ﺁﺳﺎﻥ‌ﺗﺮ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ.

ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺩﺭﺑﺎﺭﻩ ﭼﻪ ﻣﻮﺿﻮﻋﯽ ﭘﺎﺩﮐﺴﺖ ﺑﺴﺎﺯﻳﺪ؟ ﺩﺭ ﺣﻘﻴﻘﺖ ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﺑﺘﺪﺍ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﭘﺎﺳﺦ ﺩﻫﻴﺪ ﮐﻪ: ﺍﺯ ﭼﻪ ﭼﻴﺰﯼ ﺧﻮﺷﺘﺎﻥ ﻣﯽ‌ﺁﻳﺪ؟ ﻳﺎ ﺍﻳﻨﮑﻪ، ﺑﻪ ﭼﻪ ﭼﻴﺰﯼ ﺷﺪﻳﺪﺍ ﻋﻼﻗﻪ ﺩﺍﺭﻳﺪ؟

ﺑﮕﺬﺍﺭﻳﺪ ﺭﻭ ﺭﺍﺳﺖ ﺑﺎﺷﻴﻢ. ﺳﺎﺧﺖ ﭘﺎﺩﮐﺴﺖ ﺑﺨﺸﯽ ﺍﺯ ﻭﻗﺖ ﺷﻤﺎ ﺭﺍ ﺑﺮﺍﯼ ﻫﻤﻴﺸﻪ ﺧﻮﺍﻫﺪ ﮔﺮﻓﺖ. ﺍﻣﺎ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﻫﻢ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﭘﺎﺩﮐﺴﺖ ﮐﺎﺭﯼ ﺳﺮﮔﺮﻡ ﮐﻨﻨﺪﻩ ﻭ ﻟﺬﺕ ﺑﺨﺶ ﺍﺳﺖ. ﻭ ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﺩﺭﺑﺎﺭﻩ ﻣﻮﺿﻮﻉ ﻣﻮﺭﺩ ﻋﻼﻗﻪ‌ﺗﺎﻥ ﭘﺎﺩﮐﺴﺖ ﺿﺒﻂ ﮐﻨﻴﺪ، ﺍﻳﻦ ﺧﺸﻨﻮﺩﯼ ﻭ ﻟﺬﺕ ﺩﻭ ﭼﻨﺪﺍﻥ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻳﮏ ﺷﺨﺺ ﻣﺸﻬﻮﺭ ﻳﺎ ﻳﮏ ﻣﺘﺨﺼﺺ ﺩﺭ ﺯﻣﻴﻨﻪ ﻣﻮﺭﺩ ﺍﻧﺘﺨﺎﺑﺘﺎﻥ ﺩﺭ ﺍﻳﻨﺘﺮﻧﺖ ﺗﺒﺪﻳﻞ ﺷﻮﻳﺪ. ﺷﺎﻳﺪ ﺍﻳﻦ ﺳﻮﺍﻻﺕ ﺑﻪ ﺷﻤﺎ ﺩﺭ ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ ﮐﻤﮏ ﮐﻨﻨﺪ:

ﺳﺮﮔﺮﻣﯽ‌ﻫﺎﯼ ﺷﻤﺎ ﭼﻪ ﻫﺴﺘﻨﺪ؟

  • ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺩﺭﺑﺎﺭﻩ ﭼﻪ ﭼﻴﺰﯼ ﺻﺤﺒﺖ ﮐﻨﻴﺪ؟
  • ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺣﺎﺿﺮﻳﺪ ﻫﻤﻴﺸﻪ ﺑﺮﺍﯼ ﺁﻧﻬﺎ ﭘﻮﻝ ﺧﺮﺝ ﮐﻨﻴﺪ ﻳﺎ ﺍﺻﻼ ﭼﻴﺰﯼ ﺑﺮﺍﻳﺸﺎﻥ ﻫﺰﻳﻨﻪ ﻧﻤﯽ‌ﮐﻨﻴﺪ؟
  • ﺩﺭ ﭼﻪ ﺯﻣﻴﻨﻪ‌ﻫﺎﻳﯽ ﺩﺍﺭﺍﯼ ﺗﺨﺼﺺ ﻭ ﻣﻬﺎﺭﺕ ﻫﺴﺘﻴﺪ؟
  • ﻣﻌﻤﻮﻻ ﺍﻓﺮﺍﺩ ﺑﺮﺍﯼ ﮐﻤﮏ ﮔﺮﻓﺘﻦ ﺩﺭ ﭼﻪ ﺯﻣﻴﻨﻪ‌ﻫﺎﻳﯽ ﺑﻪ ﺷﻤﺎ ﻣﺮﺍﺟﻌﻪ ﻣﯽ‌ﮐﻨﻨﺪ؟

ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻋﯽ ﮐﻪ ﺑﻴﺶ ﺍﺯ ﻫﻤﻪ ﺑﻪ ﺁﻥ ﻋﻼﻗﻪ ﺩﺍﺭﻳﺪ، ﮐﻴﻔﻴﺖ ﻭ ﻣﺤﺒﻮﺑﻴﺖ ﭘﺎﺩﮐﺴﺖ‌ﺗﺎﻥ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﻣﯽ‌ﺩﻫﺪ

ﺁﻳﺎ ﺑﻪ ﺍﺳﮑﯽ ﺭﻭﯼ ﺁﺏ ﻋﻼﻗﻪ ﺩﺍﺭﻳﺪ؟ ﺁﻳﺎ ﺩﺭ ﻣﻌﺮﻕ ﮐﺎﺭﯼ ﻳﺎ ﻣﻨﺒﺖ ﮐﺎﺭﯼ ﺗﺒﺤﺮ ﺩﺍﺭﻳﺪ؟ ﺁﻳﺎ ﻳﮏ ﺍﺳﺐ ﺳﻮﺍﺭ ﻳﺎ ﮐﺎﻳﺖ ﺳﻮﺍﺭ ﻗﺎﺑﻞ ﻫﺴﺘﻴﺪ؟ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﺍﻳﻦ ﺳﺮﮔﺮﻣﯽ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺗﺒﺪﻳﻞ ﺑﻪ ﻳﮏ ﻣﻮﺿﻮﻉ ﻋﺎﻟﯽ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﭘﺎﺩﮐﺴﺖ ﺷﻮﻧﺪ.

ﻋﻼﻗﻪ ﻭ ﺍﺷﺘﻴﺎﻕ ﺷﻤﺎ ﻫﺮ ﭼﻪ ﮐﻪ ﺑﺎﺷﺪ، ﻣﻄﻤﺌﻨﺎ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﺎ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﻋﻼﻗﻪ ﻭ ﺍﺷﺘﻴﺎﻕ ﺳﻬﻴﻢ ﻭ ﺷﺮﻳﮏ ﺑﺎﺷﻨﺪ. ﺍﻓﺮﺍﺩﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺩﺭﺑﺎﺭﻩ ﮐﺎﺭ ﺷﻤﺎ ﺑﺸﻮﻧﺪ ﻭ ﻳﺎﺩ ﺑﮕﻴﺮﻧﺪ، ﺯﻳﺮﺍ ﺁﻧﻬﺎ ﻫﻢ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺁﻥ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻨﺪ. ﺍﮔﺮ ﺗﺠﺮﺑﻴﺎﺕ، ﺩﺍﻧﺶ ﻭ ﺍﺷﺘﻴﺎﻕ‌ﺗﺎﻥ ﺭﺍ ﺑﺎ ﺁﻧﻬﺎ ﺩﺭ ﻣﻴﺎﻥ ﺑﮕﺬﺍﺭﻳﺪ، ﻣﻄﻤﺌﻨﺎ ﮔﻮﺵ ﺧﻮﺍﻫﻨﺪ ﺩﺍﺩ.

ﺍﮔﺮ ﺗﺨﺼﺺ ﻳﺎ ﺩﺍﻧﺶ ﺧﺎﺻﯽ ﺩﺍﺭﻳﺪ، ﺍﻓﺮﺍﺩﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺁﻥ ﺭﺍ ﻓﺮﺍ ﮔﻴﺮﻧﺪ. ﺁﻧﻬﺎ ﺑﻪ ﺯﻭﺩﯼ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﻨﺒﻊ ﺍﻃﻼﻋﺎﺗﯽ ﺧﻮﺍﻫﻨﺪ ﺷﻨﺎﺧﺖ ﻭ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺑﻪ ﻳﮑﯽ ﺍﺯ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺿﺮﻭﺭﻳﺸﺎﻥ ﺗﺒﺪﻳﻞ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﻫﻤﭽﻨﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺭﺑﺎﺭﻩ ﺷﻬﺮﯼ ﮐﻪ ﺩﺭ ﺁﻥ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﮐﻨﻴﺪ، ﺻﺤﺒﺖ ﮐﻨﻴﺪ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺗﺸﺮﻳﺢ ﮐﺎﺭ ﻭ ﮐﺴﺐ‌ﻫﺎﯼ ﻣﺤﻠﯽ ﻳﺎ ﺍﻃﻼﻋﺎﺕ ﮔﺮﺩﺷﮕﺮﯼ ﺑﭙﺮﺩﺍﺯﻳﺪ. ﻭ ﺍﻳﻦ ﻟﻴﺴﺖ ﻫﻤﭽﻨﺎﻥ ﺍﺩﺍﻣﻪ ﺩﺍﺭﺩ. ﻣﻬﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﺮﺧﯽ ﭼﻴﺰ‌ﻫﺎ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ. ﺩﻟﻴﻞ ﺍﻳﻨﮑﻪ ﺑﺮ ﺍﺷﺘﻴﺎﻗﺘﺎﻥ ﺗﺎﮐﻴﺪ ﺩﺍﺭﻳﻢ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﻣﻴﻞ ﻭ ﺭﻏﺒﺖ ﺷﻤﺎ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺗﺠﺮﺑﻪ ﺗﻮﻟﻴﺪ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺑﻪ ﮐﺎﺭﯼ ﺁﺳﺎﻥ‌ﺗﺮ ﻭ ﻟﺬﺕ ﺑﺨﺶ‌ﺗﺮ ﺗﺒﺪﻳﻞ ﺷﻮﺩ.

ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺩﺭ ﻃﻮﻻﻧﯽ ﻣﺪﺕ ﺍﺯ ﺳﺎﺧﺖ ﭘﺎﺩﮐﺴﺖ ﻟﺬﺕ ﺑﺒﺮﻳﺪ، ﭘﺲ ﻣﻮﺿﻮﻋﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﮐﻪ ﻋﺎﺷﻘﺶ ﻫﺴﺘﻴﺪ. ﺩﻭﺑﺎﺭﻩ ﺗﺎﮐﻴﺪ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﺍﻧﺘﺨﺎﺏ‌ﻫﺎ ﺑﯽ‌ﭘﺎﻳﺎﻥ ﻫﺴﺘﻨﺪ. ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﺟﺮﺍﯼ ﺑﺮﻧﺎﻣﻪ ﮐﻤﺪﯼ ﻭ ﻃﻨﺰ ﺑﭙﺮﺩﺍﺯﻳﺪ. ﻣﻮﺳﻴﻘﯽ ﺍﺟﺮﺍ ﮐﻨﻴﺪ ﻭ ﺣﺘﯽ ﺍﺯ ﺳﻴﺎﺳﺖ ﺑﮕﻮﻳﻴﺪ. ﻣﻬﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﻮﺿﻮﻋﯽ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ ﻭ ﮐﺎﺭ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﺪ.

ﺍﺧﺘﺼﺎﺻﯽ ﮐﺎﺭ ﮐﻨﻴﺪ
ﭘﺎﺩﮐﺴﺖ ﺍﺑﺰﺍﺭﯼ ﻗﺪﺭﺗﻤﻨﺪ ﺍﺳﺖ، ﺯﻳﺮﺍ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﺎﺭﯼ ﺍﻧﺠﺎﻡ ﺩﻫﺪ ﮐﻪ ﺍﺯ ﻋﻬﺪﻩ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﻋﻤﻮﻣﯽ ﻣﺎﻧﻨﺪ ﺭﺍﺩﻳﻮ ﻭ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺑﺮ ﻧﻤﯽ‌ﺁﻳﺪ. ﭘﺎﺩﮐﺴﺖ ﺍﺑﺰﺍﺭﯼ ﻋﺎﻟﯽ ﺑﺮﺍﯼ ﭘﺮﺩﺍﺧﺘﻦ ﺑﻪ ﻣﻮﺿﻮﻋﺎﺕ ﻭ ﻋﻼﻗﻪ ﻣﻨﺪﯼ‌ﻫﺎﯼ ﮐﻮﭼﮏ، ﻣﺘﻤﺮﮐﺰ ﻭ ﺑﻪ ﻇﺎﻫﺮ ﮐﻢ ﺍﻫﻤﻴﺘﯽ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺩﻳﮕﺮ ﺷﮑﻞ‌ﻫﺎ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺁﻧﻬﺎ ﺭﺍ ﭘﻮﺷﺶ ﺩﺍﺩ.

ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﻳﮏ ﻣﺤﺪﻭﺩﻩ ﻣﺸﺨﺺ ﺍﺯ ﻋﻼﻗﻪ‌ﻣﻨﺪﯼ‌ﻫﺎ ﻭ ﺗﻤﺮﮐﺰ ﺭﻭﯼ ﺁﻥ، ﭘﺎﺩﮐﺴﺖ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﺒﺪﻳﻞ ﺑﻪ ﻣﻨﺒﻌﯽ ﻗﺪﺭﺗﻤﻨﺪ ﺍﺯ ﺍﻃﻼﻋﺎﺕ ﻭ ﺳﺮﮔﺮﻣﯽ‌ﻫﺎ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩﯼ ﺷﻮﺩ ﮐﻪ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻋﻼﻗﻪ ﺩﺍﺭﻧﺪ ﻭ ﺁﻥ ﺭﺍ ﺩﻧﺒﺎﻝ ﻣﯽ‌ﮐﻨﻨﺪ.

ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﺷﻤﺎ ﺭﺍﺣﺖ‌ﺗﺮ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺭﺍ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﮐﺮﺩﻩ، ﺷﻨﻮﻧﺪﮔﺎﻧﺘﺎﻥ ﺭﺍ ﺣﻔﻆ ﻧﻤﻮﺩﻩ ﻭ ﺣﺘﯽ ﺍﺯ ﻃﺮﻳﻖ ﺍﻳﻦ ﺗﻤﺮﮐﺰ ﺭﻭﯼ ﻣﻮﺿﻮﻋﺎﺕ ﻣﺤﺪﻭﺩ، ﺭﺍﺣﺖ‌ﺗﺮ ﻭ ﺳﺮﻳﻊ‌ﺗﺮ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻫﺪﺍﻑ ﺗﺠﺎﺭﻳﺘﺎﻥ ﺑﺮﺳﻴﺪ.

ﺍﻣﺎ ﺍﮔﺮ ﺳﻌﯽ ﮐﻨﻴﺪ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍ ﺑﺮﺍﯼ ﻫﻤﻪ ﺍﻓﺮﺍﺩ ﺑﮕﻮﻳﻴﺪ، ﺩﺭ ﺁﺧﺮ ﭼﻴﺰﯼ ﺑﺮﺍﻳﺘﺎﻥ ﺑﺎﻗﯽ ﻧﺨﻮﺍﻫﺪ ﻣﺎﻧﺪ. ﺷﻤﺎ ﭼﻨﺎﻥ ﮔﺴﺘﺮﺵ ﺧﻮﺍﻫﻴﺪ ﻳﺎﻓﺖ ﮐﻪ ﺗﺒﺪﻳﻞ ﺑﻪ ﻭﺭﻗﻪ‌ﺍﯼ ﻧﺎﺯﮎ ﻭ ﺩﺭﻳﺎﻳﯽ ﺑﺎ ﻋﻤﻖ ﻳﮏ ﺑﻨﺪ ﺍﻧﮕﺸﺖ ﻣﯽ‌ﺷﻮﻳﺪ ﻭ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﻫﻢ ﭼﻨﺎﻥ ﺿﻌﻴﻒ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻧﮕﻬﺪﺍﺭﯼ ﺁﻥ ﮐﺎﺭﯼ ﺳﺨﺖ ﻭ ﺳﻨﮕﻴﻦ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﭼﮕﻮﻧﮕﯽ ﺗﺎﺛﻴﺮ ﺍﻫﺪﺍﻑ ﭘﺎﺩﮐﺴﺖ ﺑﺮ ﺭﻭﯼ ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ
ﺩﻟﻴﻞ ﺍﻳﻨﮑﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﺍﺯ ﺍﻫﺪﺍﻑ ﺷﻤﺎ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﭘﺎﺩﮐﺴﺖ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ، ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺍﻫﺪﺍﻑ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺭﻭﯼ ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ ﭘﺎﺩﮐﺴﺖ ﺗﺎﺛﻴﺮ ﮔﺬﺍﺭ ﺑﺎﺷﻨﺪ. ﺍﻳﻨﮑﻪ ﻧﮕﺎﻩ ﻭ ﺭﻭﻳﮑﺮﺩ ﺗﺠﺎﺭﯼ ﺑﻪ ﭘﺎﺩﮐﺴﺖ‌ﺗﺎﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﻳﺎ ﺑﺨﻮﺍﻫﻴﺪ ﺗﻌﺪﺍﺩ ﺯﻳﺎﺩﯼ ﺷﻨﻮﻧﺪﻩ ﮐﺴﺐ ﮐﻨﻴﺪ، ﺍﺯ ﺍﮐﻨﻮﻥ ﺑﺎﻳﺪ ﺑﻪ ﭘﺘﺎﻧﺴﻴﻞ ﻣﻮﺿﻮﻉ ﺍﺯ ﻧﻈﺮ ﺗﻌﺪﺍﺩ ﻣﺨﺎﻃﺐ ﺑﻴﺎﻧﺪﻳﺸﻴﺪ.

ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺩﺭﺑﺎﺭﻩ ﻧﮋﺍﺩﻫﺎﯼ ﮔﻮﻧﺎﮔﻮﻥ ﮐﺒﮏ ﻭ ﺷﮑﺎﺭ ﺁﻧﻬﺎ ﺻﺤﺒﺖ ﮐﻨﻴﺪ، ﻧﺒﺎﻳﺪ ﺗﻮﻗﻊ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﻭﺭﺯﺷﯽ ﻳﺎ ﺗﮑﻨﻮﻟﻮﮊﯼ ﺷﻨﻮﻧﺪﻩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺑﺎﺯﺍﺭ ﻣﻮﺿﻮﻉ ﺍﻧﺘﺨﺎﺑﻴﺘﺎﻥ ﭼﻪ ﺍﻧﺪﺍﺯﻩ ﺑﺰﺭﮒ ﺍﺳﺖ؟ ﺁﻳﺎ ﺗﻘﺎﺿﺎﯼ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﺁﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ؟ ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺍﺯ ﻃﺮﻳﻖ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺑﻪ ﮐﺴﺐ ﺩﺭﺁﻣﺪ ﺑﭙﺮﺩﺍﺯﻳﺪ، ﺑﻪ ﺗﺮﻭﻳﺞ ﻭ ﺗﺒﻠﻴﻎ ﮐﺎﺭ ﻭ ﮐﺴﺒﺘﺎﻥ ﺑﭙﺮﺩﺍﺯﻳﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﻣﺨﺎﻃﺒﺎﻥ ﻓﺮﺍﻭﺍﻧﯽ ﺑﺮﺍﯼ ﺁﻥ ﺩﺳﺖ ﻭ ﭘﺎ ﮐﻨﻴﺪ، ﭘﺲ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ ﮐﻪ ﺍﺯ ﺗﻘﺎﺿﺎﯼ ﺧﻮﺏ ﻭ ﮔﺴﺘﺮﺩﻩ ﺑﻪ ﻣﻮﺿﻮﻉ ﺍﻧﺘﺨﺎﺑﻴﺘﺎﻥ ﻣﻄﻤﺌﻦ ﺷﻮﻳﺪ.

ﺍﮔﺮ ﺑﻪ ﺩﻧﺒﺎﻝ ﺳﻮﺩ ﺩﻫﯽ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺩﺭ ﺁﻳﻨﺪﻩ ﻫﺴﺘﻴﺪ، ﺍﻃﻤﻴﻨﺎﻥ ﻳﺎﺑﻴﺪ ﮐﻪ ﻣﻮﺿﻮﻋﺘﺎﻥ ﺩﺭﺁﻣﺪ ﺯﺍ ﻭ ﺳﻮﺩﺁﻭﺭ ﺍﺳﺖ. ﺁﻳﺎ ﻣﺮﺩﻡ ﺣﺎﺿﺮﻧﺪ ﭘﻮﻟﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺻﺮﻑ ﮐﻨﻨﺪ؟ ﺁﻳﺎ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺍﻳﻨﻘﺪﺭ ﺑﺮﺍﻳﺸﺎﻥ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﮐﻴﻔﺸﺎﻥ ﺭﺍ ﺍﺯ ﺟﻴﺐ ﺑﻴﺮﻭﻥ ﺁﻭﺭﺩﻩ ﻭ ﮐﻤﯽ ﻫﺰﻳﻨﻪ ﮐﻨﻨﺪ؟

ﻭ ﺍﻟﺒﺘﻪ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ ﻣﻴﺰﺍﻥ ﺭﻗﺎﺑﺖ ﭼﻘﺪﺭ ﺍﺳﺖ؟ ﺧﺒﺮ ﺧﻮﺏ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﭘﺎﺩﮐﺴﺖ ﺁﻧﻘﺪﺭ ﺟﻮﺍﻥ ﺍﺳﺖ ﮐﻪ ﻫﻨﻮﺯ ﺯﻣﻴﻨﻪ‌ﻫﺎﯼ ﺑﮑﺮ ﻓﺮﺍﻭﺍﻧﯽ ﺩﺭ ﺁﻥ ﻳﺎﻓﺖ ﻣﯽ‌ﺷﻮﺩ. ﺑﺮﺍﯼ ﻫﻤﻴﻦ ﺷﺎﻧﺲ ﺯﻳﺎﺩﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺷﻤﺎ ﺑﺘﻮﺍﻧﻴﺪ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺍﻳﺠﺎﺩ ﮐﻨﻴﺪ ﮐﻪ ﺍﻭﻟﻴﻦ ﻧﻤﻮﻧﻪ ﻳﺎ ﺷﺎﻳﺪ ﻳﮑﯽ ﺍﺯ ﺍﻭﻟﻴﻦ‌ﻫﺎ ﺩﺭ ﻧﻮﻉ ﺧﻮﺩ ﺑﺎﺷﻴﺪ.

ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺟﻠﻮ ﻧﺮﻭﻳﺪ ﮐﻪ ﭘﺲ ﺍﺯ ﭼﻨﺪ ﻣﺎﻩ ﮐﺎﺭ، ﺗﺎﺯﻩ ﻣﺘﻮﺟﻪ ﺷﻮﻳﺪ ﺑﻪ ﺩﻟﻴﻞ ﻋﺪﻡ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺻﺤﻴﺢ ﺑﺮﺍﯼ ﻣﻮﺿﻮﻉ، ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻫﺪﺍﻓﺘﺎﻥ ﺩﺳﺖ ﻳﺎﺑﻴﺪ. ﭘﺲ ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻤﯽ ﺩﻳﺮ‌ﺗﺮ ﮐﺎﺭ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﺪ، ﺍﻣﺎ ﺑﺎ ﺗﺤﻘﻴﻖ ﻭ ﻓﮑﺮ ﺑﻴﺸﺘﺮ ﺑﻪ ﺍﻧﺠﺎﻡ ﺁﻥ ﺑﭙﺮﺩﺍﺯﻳﺪ. ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﻣﮑﺎﻥ ﭘﺸﻴﻤﺎﻧﯽ ﻭ ﺑﺮﻭﺯ ﺍﺷﺘﺒﺎﻩ ﺭﺍ ﺗﺎ ﺣﺪ ﻓﺮﺍﻭﺍﻧﯽ ﮐﺎﻫﺶ ﺩﺍﺩﻩ‌ﺍﻳﺪ.

 

ﺟﻤﻊ ﺑﻨﺪﯼ
ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺩﺭ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﭘﺎﺩﮐﺴﺖ، ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ ﺍﺳﺖ. ﻭ ﻫﻨﮕﺎﻡ ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ ﺗﻮﺟﻪ ﺑﻪ ﺩﻭ ﻧﮑﺘﻪ، ﺣﻴﺎﺗﯽ ﺍﺳﺖ:

  • ﭼﻴﺰﯼ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﻫﻴﺠﺎﻥ ﻣﯽ‌ﺁﻭﺭﺩ. ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺗﻮﻟﻴﺪ ﭘﺎﺩﮐﺴﺖ ﺍﮐﻨﻮﻥ ﻭ ﺩﺭ ﺩﺭﺍﺯ ﻣﺪﺕ ﺑﺮﺍﻳﺘﺎﻥ ﺗﺒﺪﻳﻞ ﺑﻪ ﻳﮏ ﺳﺮﮔﺮﻣﯽ ﺷﻮﺩ.
  • ﺍﻃﻤﻴﻨﺎﻥ ﻳﺎﺑﻴﺪ ﮐﻪ ﺗﻘﺎﺿﺎﯼ ﮐﺎﻓﯽ ﻭ ﮔﺮﺍﻳﺶ ﻻﺯﻡ ﺑﻪ ﻣﻮﺿﻮﻋﺘﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺑﻪ ﺧﺼﻮﺹ ﺍﮔﺮ ﺑﻪ ﺩﻧﺒﺎﻝ ﮐﺴﺐ ﺩﺭﺁﻣﺪ ﻳﺎ ﺗﺒﻠﻴﻎ ﮐﺴﺐ ﻭ ﮐﺎﺭﺗﺎﻥ ﺍﺯ ﻃﺮﻳﻖ ﭘﺎﺩﮐﺴﺖ ﻫﺴﺘﻴﺪ.

ﺍﮐﻨﻮﻥ ﭘﺲ ﺍﺯ ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ، ﻧﻮﺑﺖ ﺑﻪ ﺍﻧﺘﺨﺎﺏ ﻓﺮﻣﺖ ﻭ ﻗﺎﻟﺐ ﭘﺎﺩﮐﺴﺖ ﻣﯽ‌ﺭﺳﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺑﺮﺍﯼ ﺟﺬﺏ ﻣﺨﺎﻃﺐ ﻭ ﻫﻤﺮﺍﻫﯽ ﻭﯼ ﺗﺎ ﭘﺎﻳﺎﻥ ﻫﺮ ﺑﺨﺶ ﭘﺎﺩﮐﺴﺖ، ﺍﻣﺮﯼ ﺿﺮﻭﺭﯼ ﺍﺳﺖ. ﺍﮔﺮ ﺑﻪ ﺩﺭﺳﺘﯽ ﻣﺮﺍﻗﺐ ﻓﺮﻣﺖ ﻭ ﻗﺎﻟﺐ ﭘﺎﺩﮐﺴﺖ‌ﺗﺎﻥ ﻧﺒﺎﺷﻴﺪ ﻭ ﺁﻥ ﺭﺍ ﺳﺮ ﻭ ﺳﺎﻣﺎﻥ ﻧﺪﻫﻴﺪ، ﻧﻪ ﺗﻨﻬﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺸﺘﺮﮐﺎﻥ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﺑﺪﻫﻴﺪ، ﺑﻠﮑﻪ ﺍﺣﺘﻤﺎﻝ ﺍﻳﻨﮑﻪ ﺑﺎ ﻣﻴﺰﺑﺎﻥ ﺳﺎﻳﺖ ﻭ ﭘﺎﺩﮐﺴﺖ‌ﺗﺎﻥ ﻫﻢ ﺑﻪ ﻣﺸﮑﻞ ﺑﺨﻮﺭﻳﺪ ﺯﻳﺎﺩ ﺍﺳﺖ ﻭ ﺍﺯ ﺳﻮﻳﯽ ﺗﻮﻟﻴﺪ ﭘﺎﺩﮐﺴﺖ ﻫﻢ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﻪ ﮐﺎﺭﯼ ﺑﺴﻴﺎﺭ ﻭﻗﺖ ﮔﻴﺮ ﻭ ﺩﻳﻮﺍﻧﻪ ﮐﻨﻨﺪﻩ ﺗﺒﺪﻳﻞ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺩﺭ ﺩﺭﺱ ﺁﻳﻨﺪﻩ ﺑﻴﺸﺘﺮ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.
