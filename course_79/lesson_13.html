ﺩﺭﺱ ﺳﻮﻡ - ﺍﺗﺎﻕ ﺿﺒﻂ ﺻﺪﺍ ﻳﺎ ﺍﺳﺘﻮﺩﻳﻮ

ﺍﮐﻨﻮﻥ ﭘﺲ ﺍﺯ ﺁﺷﻨﺎﻳﯽ ﺑﺎ ﭘﺎﺩﮐﺴﺖ ﻭ ﺍﻧﻮﺍﻉ ﺁﻥ، ﻭ ﻣﺮﻭﺭﯼ ﺑﺮ ﺩﻻﻳﻞ ﺳﺎﺧﺖ ﭘﺎﺩﮐﺴﺖ، ﻣﺼﺼﻢ ﺷﺪﻩ‌ﺍﻳﻢ ﮐﻪ ﻣﺎ ﻫﻢ ﺑﺎﻳﺪ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﻭ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮐﺎﺭﻣﺎﻥ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﻢ. ﻣﺜﻼ ﻳﮏ ﻫﻔﺘﻪ ﻧﺎﻣﻪ ﺩﺭﺑﺎﺭﻩ ﺍﺩﺑﻴﺎﺕ ﻋﻠﻤﯽ-ﺗﺨﻴﻠﯽ ﻳﺎ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺭﻭﺯﺍﻧﻪ ﭼﻨﺪ ﺩﻗﻴﻘﻪ‌ﺍﯼ ﺩﺭﺑﺎﺭﻩ ﺍﺗﻔﺎﻗﺎﺕ ﺳﻴﺎﺳﯽ‪-ﺍﺟﺘﻤﺎﻋﯽ ﮐﺎﻣﻼ ﺑﺎ ﻋﻼﻗﻪ‌ﻣﺎﻥ ﻫﻢ ﺟﻮﺭ ﺩﺭ ﻣﯽ‌ﺁﻳﺪ. ﺍﺻﻼ ﺷﺎﻳﺪ ﮔﺰﻳﺪﻩ ﺭﻭﺯﻧﺎﻣﻪ‌ﻫﺎﯼ ﻭﺭﺯﺷﯽ ﻫﻢ ﺍﻳﺪﻩ ﺑﺪﯼ ﺑﺮﺍﯼ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﻧﺒﺎﺷﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺗﺼﻤﻴﻢ ﺩﺍﺭﻳﺪ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﻫﻔﺘﮕﯽ ﻳﺎ ﻫﻔﺘﻪ‌ﺍﯼ ﺩﻭ ﺑﺎﺭ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻫﺮﯼ ﭘﺎ‌ﺗﺮ ﺭﺍ ﺗﺒﺪﻳﻞ ﺑﻪ ﭘﺎﺩﮐﺴﺖ-ﮐﺘﺎﺏ ﺻﻮﺗﯽ ﮐﻨﻴﺪ.

ﺑﻪ ﻫﺮ ﺣﺎﻝ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ ﻧﻴﺎﺯﻣﻨﺪ ﺣﺪﺍﻗﻞ ﺗﺠﻬﻴﺰﺍﺕ ﻭ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺁﻧﻬﺎ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ. ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﺪ ﺑﻪ ﺻﻮﺭﺕ ﺣﺮﻓﻪ‌ﺍﯼ ﻭ ﻃﻮﻻﻧﯽ ﻣﺪﺕ ﺑﻪ ﺗﻮﻟﻴﺪ ﭘﺎﺩﮐﺴﺖ ﺑﭙﺮﺩﺍﺯﻳﺪ ﻭ ﻫﻤﺎﻧﻨﺪ ﺑﺮﺧﯽ ﻧﻤﻮﻧﻪ‌ﻫﺎﯼ ﻣﻮﻓﻖ ﺧﺎﺭﺟﯽ ﺁﻥ ﺭﺍ ﺑﻪ ﻳﮏ ﮐﺎﺭ ﻭ ﮐﺴﺐ ﻣﻮﻓﻖ ﺗﺒﺪﻳﻞ ﮐﻨﻴﺪ، ﺩﺍﻳﺮﻩ ﺍﻣﮑﺎﻧﺎﺕ ﻣﻮﺭﺩ ﻧﻴﺎﺯ ﺑﺴﻴﺎﺭ ﮔﺴﺘﺮﺩﻩ‌ﺗﺮ ﺷﺪﻩ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺷﺎﻣﻞ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﺍﺳﺘﻮﺩﻳﻮ ﺿﺒﻂ، ﺍﻧﺘﺨﺎﺏ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺣﺮﻓﻪ‌ﺍﯼ ﻭ ﻣﺎﻧﻨﺪ ﺁﻥ ﺷﻮﺩ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺳﻌﯽ ﺩﺍﺭﻳﻢ ﻧﮕﺎﻫﯽ ﮐﻮﺗﺎﻩ ﺑﻪ ﺗﻤﺎﻡ ﺍﻳﻦ ﻣﺒﺎﺣﺚ ﺩﺍﺷﺘﻪ ﻭ ﺑﻪ ﻃﻮﺭ ﮔﺬﺭﺍ ﺩﻳﺪﮔﺎﻩ ﻣﻨﺎﺳﺒﯽ ﻧﺴﺒﺖ ﺑﻪ ﺁﻧﻬﺎ ﺍﻳﺠﺎﺩ ﮐﻨﻴﻢ.

 

ﺳﺎﺧﺖ ﺍﺗﺎﻕ ﺿﺒﻂ ﺻﺪﺍ
ﺿﺒﻂ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺑﻪ ﺭﺍﺣﺘﯽ ﺩﺭ ﺍﺗﺎﻕ ﺷﺨﺼﯽ ﺷﻤﺎ ﻫﻢ ﻗﺎﺑﻞ ﺍﻧﺠﺎﻡ ﺍﺳﺖ ﻭ ﺗﻨﻬﺎ ﻧﻴﺎﺯ ﺑﻪ ﮐﻤﯽ ﺳﮑﻮﺕ ﻭ ﺁﺭﺍﻣﺶ ﺩﺍﺭﺩ. ﺳﺎﻋﺘﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﮐﻪ ﮐﺴﯽ ﺩﺭ ﻣﻨﺰﻝ ﻧﻴﺴﺖ، ﺗﺮﺟﻴﺤﺎ ﺳﺮ ﻭ ﺻﺪﺍﯼ ﺧﻴﺎﺑﺎﻥ ﮐﻤﺘﺮ ﺍﺳﺖ، ﻟﻮﺍﺯﻡ ﺑﺮﻗﯽ ﻣﻨﺰﻝ (ﻣﺎﻧﻨﺪ ﺟﺎﺭﻭﺑﺮﻗﯽ ﻭ ﻣﺎﺷﻴﻦ ﻟﺒﺎﺱ ﺷﻮﻳﯽ ﻭ...) ﺧﺎﻣﻮﺷﻨﺪ. ﺑﺮﺍﯼ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺗﻤﺎﺷﺎ ﮐﺮﺩﻥ ﻫﻢ ﻗﻠﻘﻠﮏ ﻧﻤﯽ‌ﺷﻮﻳﺪ! ﺑﻪ ﺍﺗﺎﻗﺘﺎﻥ ﺭﻓﺘﻪ، ﺩﺭ ﺭﺍ ﺑﺴﺘﻪ ﻭ ﻣﺸﻐﻮﻝ ﮐﺎﺭ ﺷﻮﻳﺪ.

ﺍﻣﺎ‌‌ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﮔﻔﺘﻴﻢ ﺍﻳﻦ ﺣﺪﺍﻗﻞ ﻧﻴﺎﺯ ﺑﺮﺍﯼ ﺿﺒﻂ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺍﺳﺖ. ﺯﻳﺮﺍ ﺩﻳﻮﺍﺭﻫﺎﯼ ﺍﺗﺎﻕ ﺷﻤﺎ ﺁﮐﻮﺳﺘﻴﮏ ﻧﺒﻮﺩﻩ ﻭ ﺑﺎﻋﺚ ﺍﻧﻌﮑﺎﺱ ﺻﺪﺍ ﻣﯽ‌ﺷﻮﻧﺪ. ﺍﺳﺒﺎﺏ ﻭ ﻟﻮﺍﺯﻡ ﻫﻢ ﮐﻪ ﺍﺯ ﺟﻨﺲ‌ﻫﺎﯼ ﻣﺘﻔﺎﻭﺗﯽ ﻫﺴﺘﻨﺪ، ﺗﺎﺛﻴﺮﺍﺕ ﮔﻮﻧﺎﮔﻮﻧﯽ ﺩﺍﺷﺘﻪ ﻭ ﺍﻧﻌﮑﺎﺱ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺍﺯ ﺻﺪﺍﻳﺘﺎﻥ ﺭﺍ ﺗﻮﻟﻴﺪ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﻟﺒﺘﻪ ﺍﺯ ﻃﺮﻑ ﺩﻳﮕﺮ، ﺳﺎﺧﺖ ﻳﺎ ﺍﺟﺎﺭﻩ ﻳﮏ ﺍﺳﺘﻮﺩﻳﻮ ﺣﺮﻓﻪ‌ﺍﯼ ﺿﺒﻂ ﺻﺪﺍ ﻭ ﻣﻮﺳﻴﻘﯽ (ﮐﻪ ﺍﻳﺪﻩ ﺁﻝ‌ﺗﺮﻳﻦ ﺣﺎﻟﺖ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﺍﺳﺖ ﻭ ﮐﻴﻔﻴﺘﯽ ﻫﻤﭽﻮﻥ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺭﺍﺩﻳﻮﻳﯽ ﺭﺍ ﺑﻪ ﺍﺭﻣﻐﺎﻥ ﻣﯽ‌ﺁﻭﺭﺩ)، ﺗﻘﺮﻳﺒﺎ ﺑﺮﺍﯼ ﻫﻴﭻ ﮐﺲ ﺍﺯ ﻧﻈﺮ ﻣﺎﻟﯽ ﻣﻘﺮﻭﻥ ﺑﻪ ﺻﺮﻓﻪ ﻭ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﻧﻴﺴﺖ. ﭘﺲ ﭼﻪ ﺑﺎﻳﺪ ﮐﺮﺩ؟

ﺭﺍﻩ ﺣﻞ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﮑﺎﻧﯽ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﻴﻢ ﻭ ﺗﺎ ﺣﺪ ﻣﻤﮑﻦ ﺍﻣﮑﺎﻥ ﺍﻧﻌﮑﺎﺱ ﺻﺪﺍ ﺭﺍ ﺩﺭ ﺁﻥ ﮐﺎﻫﺶ ﺩﻫﻴﻢ. ﻳﻌﻨﯽ ﺍﻳﻨﮑﻪ ﺍﻣﮑﺎﻧﺎﺕ ﻳﮏ ﺍﺳﺘﻮﺩﻳﻮ ﺿﺒﻂ ﺻﺪﺍ ﺭﺍ ﺑﺴﺘﻪ ﺑﻪ ﻫﺰﻳﻨﻪ‌ﺍﯼ ﮐﻪ ﺑﺮﺍﻳﻤﺎﻥ ﻣﻘﺪﻭﺭ ﺍﺳﺖ، ﺑﻪ ﺧﺎﻧﻪ ﺑﻴﺎﻭﺭﻳﻢ. ﺍﻣﺎ ﭼﮕﻮﻧﻪ ﻳﮏ ﺍﺗﺎﻕ ﺿﺒﻂ ﺻﺪﺍﯼ (ﺣﺎﻻ ﻧﻪ ﺍﻳﺪﻩ ﺁﻝ، ﺍﻣﺎ) ﺧﻮﺏ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ؟

ﻫﺮ ﭼﻪ ﺍﺗﺎﻗﺘﺎﻥ ﺩﺭ ﺑﺮﺍﺑﺮ ﻭﺭﻭﺩ ﺻﺪﺍ ﻣﻘﺎﻭﻣﺖ ﺑﻬﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻳﺎ ﺑﻪ ﺍﺻﻄﻼﺡ ﻋﺎﻳﻖ ﺻﺪﺍ ﺑﺎﺷﺪ، ﻧﺘﻴﺠﻪ ﺑﻬﺘﺮﯼ ﺧﻮﺍﻫﻴﺪ ﮔﺮﻓﺖ. ﺣﺘﻤﺎ ﺗﺠﺮﺑﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﮐﻪ ﭼﮕﻮﻧﻪ ﺩﺭ ﺁﭘﺎﺭﺗﻤﺎﻥ‌ﻫﺎ، ﭘﻨﺠﺮﻩ ﺩﻭ ﺟﺪﺍﺭﻩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺻﺪﺍﯼ ﻣﺰﺍﺣﻢ ﺧﻴﺎﺑﺎﻥ (ﻣﺎﺷﻴﻦ‌ﻫﺎ، ﺑﻮﻕ، ﻋﺎﺑﺮﺍﻥ ﻭ...) ﺭﺍ ﮐﺎﻫﺶ ﺩﺍﺩﻩ ﻭ ﺳﮑﻮﺕ ﻻﺯﻡ ﺭﺍ ﺩﺭ ﺧﺎﻧﻪ ﺑﻪ ﺍﺭﻣﻐﺎﻥ ﺁﻭﺭﺩ. ﺍﺯ ﺳﻮﻳﯽ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻣﺒﻠﻤﺎﻥ ﻭ ﻣﻴﺰِ ﺍﺗﺎﻕ ﺿﺒﻂ ﺻﺪﺍ ﻫﻢ ﺗﺎ ﺣﺪ ﻣﻤﮑﻦ ﺑﭽﻪ‌ﻫﺎﯼ ﺧﻮﺑﯽ ﺑﺎﺷﻨﺪ ﻭ ﺑﺎ ﻫﺮ ﺑﺎﺭ ﺟﺎﺑﺠﺎﻳﯽ ﻭ ﻧﺸﺴﺖ ﻭ ﺑﺮﺧﺎﺳﺖ، ﺩﺍﺩ ﻭ ﻗﺎﻝ ﺭﺍﻩ ﻧﻴﻨﺪﺍﺯﻧﺪ! ﻗﺮﺍﺭ ﻧﻴﺴﺖ ﺑﺎ ﻫﺮ ﺗﮑﺎﻧﯽ ﮐﻪ ﻣﯽ‌ﺧﻮﺭﻳﺪ ﻳﮏ ﺟﻴﻎ ﮔﻮﺵ ﺧﺮﺍﺵ ﺭﺍ ﻫﻢ ﺩﺭ ﭘﺲ ﺯﻣﻴﻨﻪ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺿﺒﻂ ﮐﻨﻴﺪ. ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﺯ ﺗﺮﺱ ﺻﺪﺍﯼ ﺻﻨﺪﻟﻴﺘﺎﻥ ﺗﺎ ﭘﺎﻳﺎﻥ ﺿﺒﻂ ﻫﻤﺎﻧﻨﺪ ﻣﺠﺴﻤﻪ ﺑﯽ‌ﺣﺮﮐﺖ ﺑﻨﺸﻴﻨﻴﺪ.

ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺑﺮﺍﯼ ﺁﮐﻮﺳﺘﻴﮏ ﻭ ﻋﺎﻳﻖ ﺳﺎﺯﯼ ﺩﻳﻮﺍﺭﻫﺎﯼ ﺍﺗﺎﻕ ﺿﺒﻂ ﺻﺪﺍ ﺍﻧﺘﺨﺎﺏ ﺍﺗﺎﻕ ﺍﺳﺖ. ﺁﻳﺎ ﺷﻤﺎ ﺗﺼﻤﻴﻢ ﺩﺍﺭﻳﺪ ﻳﮏ ﺍﺗﺎﻕ ﺭﺍ ﺑﻪ ﻃﻮﺭ ﺩﺍﺋﻤﯽ ﺑﺮﺍﯼ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ ﻭ ﺁﻥ ﺭﺍ ﺗﺒﺪﻳﻞ ﺑﻪ ﺍﺳﺘﻮﺩﻳﻮ ﮐﻨﻴﺪ، ﻳﺎ ﺍﻳﻨﮑﻪ‌‌ ﻫﻤﺎﻥ ﺍﺗﺎﻕ ﺧﻮﺍﺏ ﻭ ﻳﺎ ﺍﺗﺎﻕ ﮐﺎﺭ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﻣﻮﻗﺖ ﺗﺒﺪﻳﻞ ﺑﻪ ﺍﺳﺘﻮﺩﻳﻮ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﭘﺲ ﺍﺯ ﻫﺮ ﺑﺎﺭ ﺿﺒﻂ ﺁﻥ ﺭﺍ ﺩﻭﺑﺎﺭﻩ ﺑﻪ ﺣﺎﻟﺖ ﻗﺒﻞ ﺑﺮ ﻣﯽ‌ﮔﺮﺩﺍﻧﻴﺪ؟ ﻭ ﻣﻮﺭﺩ ﻣﻬﻢ‌ﺗﺮ ﺍﻳﻨﮑﻪ ﭼﻪ ﻣﻘﺪﺍﺭ ﻫﺰﻳﻨﻪ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﮐﻨﺎﺭ ﮔﺬﺍﺷﺘﻪ‌ﺍﻳﺪ؟ ﺁﻳﺎ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﺪﻭﻥ ﻫﻴﭻ ﻫﺰﻳﻨﻪ‌ﺍﯼ ﻭ ﺑﺎ ﻫﻤﻴﻦ ﺍﻣﮑﺎﻧﺎﺕ ﻣﻮﺟﻮﺩ ﺩﺭ ﻣﻨﺰﻝ، ﺗﺎ ﺣﺪ ﻣﻤﮑﻦ ﻋﺎﻳﻖ ﺳﺎﺯﯼ ﻭ ﺁﮐﻮﺳﺘﻴﮏ ﮐﺮﺩﻥ ﺍﺗﺎﻕ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ، ﻳﺎ ﺑﻮﺩﺟﻪ ﮐﺎﻓﯽ ﺑﺮﺍﯼ ﺧﺮﻳﺪ ﺗﺠﻬﻴﺰﺍﺕ ﻻﺯﻡ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺩﺍﺭﻳﺪ؟

ﺭﺍﻩ ﺍﻧﺪﺍﺯﯼ ﻳﮏ ﺍﺳﺘﻮﺩﻳﻮ ﺿﺒﻂ ﺣﺮﻓﻪ‌ﺍﯼ ﺻﺪﺍ، ﮐﺎﺭﯼ ﺑﺴﻴﺎﺭ ﭘﺮ ﻫﺰﻳﻨﻪ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺁﻥ ﺭﺍ ﺑﺮ ﻋﻬﺪﻩ ﺗﻴﻢ ﺣﺮﻓﻪ‌ﺍﯼ ﻭ ﻣﺘﺨﺼﺺ ﺑﮕﺬﺍﺭﻳﺪ. ﻟﺬﺍ ﺑﻪ ﺟﺎﯼ ﭘﺮﺩﺍﺧﺘﻦ ﺑﻪ ﺟﺰﺋﻴﺎﺕ ﺁﻥ، ﺗﻨﻬﺎ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﻴﻢ ﺣﺘﻤﺎ ﻭ ﺣﺘﻤﺎ ﺑﻪ ﺟﺎﯼ ﺍﻳﻨﮑﻪ ﺧﻮﺩﺗﺎﻥ ﺩﺳﺖ ﺑﻪ ﮐﺎﺭ ﭼﻨﻴﻦ ﺍﻣﺮﯼ ﺷﻮﻳﺪ، ﭘﺲ ﺍﺯ ﭘﺮﺱ ﻭ ﺟﻮ ﻭ ﻳﺎﻓﺘﻦ ﻳﮏ ﮔﺮﻭﻩ ﺣﺮﻓﻪ‌ﺍﯼ ﮐﻪ ﻗﻴﻤﺖ ﻣﻨﺎﺳﺒﯽ ﺭﺍ ﻫﻢ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﻨﺪ، ﺗﻤﺎﻡ ﮐﺎﺭ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺳﭙﺮﺩﻩ ﻭ ﺗﻨﻬﺎ ﻧﻈﺎﺭﺕ ﻭ ﺑﻬﺮﻩ ﺑﺮﺩﺍﺭﯼ ﺭﺍ ﺧﻮﺩﺗﺎﻥ ﺑﺮ ﻋﻬﺪﻩ ﺑﮕﻴﺮﻳﺪ. ﻭ ﺗﺒﺮﻳﮏ ﻣﯽ‌ﮔﻮﻳﻢ. ﭘﺲ ﺍﺯ ﻣﺪﺗﯽ ﺷﻤﺎ ﺻﺎﺣﺐ ﻳﮏ ﺍﺳﺘﻮﺩﻳﻮ ﻫﺴﺘﻴﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺷﺎﻧﻪ ﺑﻪ ﺷﺎﻧﻪ ﺑﻬﺘﺮﻳﻦ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺭﺍﺩﻳﻮﻳﯽ ﺑﻪ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﻭ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺻﻮﺗﯽ ﺑﭙﺮﺩﺍﺯﻳﺪ.

ﺍﻣﺎ ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﺎ ﮐﻤﺘﺮﻳﻦ ﻫﺰﻳﻨﻪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ:

ﺳﺎﺩﻩ‌ﺗﺮﻳﻦ ﮐﺎﺭ ﺑﺮﺍﯼ ﻋﺎﻳﻖ ﺳﺎﺯﯼ ﻣﻮﻗﺖ ﺍﺗﺎﻕ ﺟﻬﺖ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺍﻧﻌﮑﺎﺱ ﻭ ﺍﮐﻮ ﺷﺪﻥ ﺻﺪﺍ ﻭ ﻫﻤﭽﻨﻴﻦ ﻋﺪﻡ ﻧﻔﻮﺫ ﺻﺪﺍﻫﺎﯼ ﺑﻴﺮﻭﻧﯽ، ﭘﻮﺷﺎﻧﺪﻥ ﺗﻤﺎﻣﯽ ﺩﻳﻮﺍﺭ‌ﻫﺎ (ﻭ ﺑﻪ ﺧﺼﻮﺹ ﺍﻃﺮﺍﻑ ﮐﺎﻣﭙﻴﻮ‌ﺗﺮ ﻭ ﻣﺤﻞ ﻧﺸﺴﺘﻦ ﺧﻮﺩﺗﺎﻥ) ‌ ﺑﺎ ﭘﺮﺩﻩ‌ﻫﺎﯼ ﺿﺨﻴﻢ ﻭ ﭘﺘﻮ ﺍﺳﺖ. ﺍﻳﻦ ﮐﺎﺭ ﻧﻴﺎﺯﯼ ﺑﻪ ﻫﻴﭻ ﻫﺰﻳﻨﻪ ﺧﺎﺻﯽ ﻧﺪﺍﺭﺩ. ﺗﻨﻬﺎ ﮐﺎﻓﯽ ﺍﺳﺖ ﺗﻌﺪﺍﺩﯼ ﻣﻴﺦ ﺭﻭﯼ ﺩﻳﻮﺍﺭ‌ﻫﺎ ﺗﻌﺒﻴﻪ ﮐﻨﻴﺪ ﺗﺎ ﻫﻨﮕﺎﻡ ﻧﻴﺎﺯ ﭘﺘﻮ‌ﻫﺎ ﺭﺍ ﺁﻭﻳﺰﺍﻥ ﮐﺮﺩﻩ ﻭ ﭘﺲ ﺍﺯ ﺍﺗﻤﺎﻡ ﺿﺒﻂ ﺁﻧﻬﺎ ﺭﺍ ﺟﻤﻊ ﮐﻨﻴﺪ.

ﺍﮔﺮ ﮐﻒ ﺍﺗﺎﻗﺘﺎﻥ ﺳﺮﺍﻣﻴﮑﯽ ﺍﺳﺖ ﻭ ﺁﻥ ﺭﺍ ﻓﺮﺵ ﻧﮑﺮﺩﻩ‌ﺍﻳﺪ، ﺍﮐﻨﻮﻥ ﻭﻗﺖ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺍﺳﺖ. ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻳﮏ ﻗﺎﻟﯽ، ﻗﺎﻟﻴﭽﻪ ﻳﺎ ﻣﻮﮐﺖ ﺯﻳﺒﺎ ﺩﺭ ﮐﻒ ﺍﺗﺎﻕ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﻋﻼﻭﻩ ﺑﺮ ﺣﺬﻑ ﻧﻮﻳﺰ ﺻﺪﺍﯼ ﭘﺎ ﺩﺭ ﮐﻒ ﺍﺗﺎﻕ، ﺻﺪﺍﯼ ﭘﺎﻳﻪ‌ﻫﺎﯼ ﻣﻴﺰ ﻭ ﺻﻨﺪﻟﯽ ﻫﻢ ﮐﻤﺘﺮ ﺷﺪﻩ ﻭ ﺍﺯ ﺑﻴﻦ ﺑﺮﻭﺩ. ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻫﺮ ﭼﻪ ﻗﺎﻟﯽ ﺷﻤﺎ ﺑﺎﻓﺖ ﻧﺮﻡ‌ﺗﺮ، ﭖُﺭ‌ﺗﺮ ﻭ ﭘﺮﺯﻫﺎﯼ ﺑﻠﻨﺪﺗﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺩﺭ ﺟﺬﺏ ﺻﺪﺍ‌ﻫﺎ ﺑﻬﺘﺮ ﻋﻤﻞ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﻫﺮ ﭼﻪ ﺍﺷﻴﺎ ﻭ ﻭﺳﺎﻳﻞ ﺳﺨﺖ ﺗﺮﯼ ﺩﺭ ﺍﺗﺎﻕ ﺷﻤﺎ ﺑﺎﺷﻨﺪ، ﻣﻴﺰﺍﻥ ﺍﮐﻮ ﺷﺪﻥ ﺻﺪﺍ ﺑﻴﺸﺘﺮ ﻣﯽ‌ﺷﻮﺩ، ﮐﻒ ﭼﻮﺑﯽ ﻳﺎ ﺳﺮﺍﻣﻴﮑﯽ، ﺩﻳﻮﺍﺭﻫﺎﯼ ﺧﺎﻟﯽ، ﺍﺳﺒﺎﺏ ﻭ ﺍﺛﺎﺛﻴﻪ ﭼﻮﺑﯽ ﻭ ﺷﻴﺸﻪ‌ﺍﯼ، ﻣﺒﻠﻤﺎﻥ ﭼﻮﺑﯽ ﻭ... ﻫﻤﻪ ﺑﺎﻋﺚ ﺍﮐﻮ ﺷﺪﻥ ﺑﻴﺸﺘﺮ ﺻﺪﺍ ﺧﻮﺍﻫﻨﺪ ﺷﺪ. ﺍﻣﺎ ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻳﮏ ﮐﺎﻧﺎﭘﻪ ﻧﺮﻡ، ﻳﺎ ﭼﻨﺪ ﮔﻞ ﺁﭘﺎﺭﺗﻤﺎﻧﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﮔﺰﻳﻨﻪ‌ﻫﺎﯼ ﺧﻮﺑﯽ ﺑﺎﺷﻨﺪ. ﺯﻳﺮﺍ ﮔﻴﺎﻫﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﮔﺰﻳﻨﻪ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺟﺬﺏ ﻭ ﺑﯽ‌ﺍﺛﺮ ﮐﺮﺩﻥ ﺻﺪﺍ‌ﻫﺎ ﺑﺎﺷﻨﺪ.

ﺩﺭ ﻳﮏ ﮐﻼﻡ، ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﺑﺎﻳﺪ ﻳﮏ ﻣﺤﻴﻂ ﺍﻳﺰﻭﻟﻪ ﺭﺍ ﺑﺮﺍﯼ ﺍﺗﺎﻕ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﻓﺮﺍﻫﻢ ﮐﻨﻴﺪ. ﺑﺎ ﺑﺴﺘﻦ ﺩﺭﺏ ﻭ ﭘﻨﺠﺮﻩ، ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﭘﺮﺩﻩ‌ﻫﺎﯼ ﺿﺨﻴﻢ ﺩﺭ ﭘﺸﺖ ﺁﻧﻬﺎ ﻭ ﮐﺎﺭﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ ﺁﻥ، ﺟﻠﻮﯼ ﻭﺭﻭﺩ ﻫﺮﮔﻮﻧﻪ ﺻﺪﺍﯼ ﻣﺰﺍﺣﻢ ﺑﻪ ﺍﺗﺎﻕ ﻭ ﺍﻟﺒﺘﻪ ﺧﺮﻭﺝ ﺁﻥ ﺭﺍ ﺑﮕﻴﺮﻳﺪ. ﺍﻟﺒﺘﻪ ﺍﺯ ﺳﻘﻒ ﺍﺗﺎﻕ ﻫﻢ ﻏﺎﻓﻞ ﻧﺸﻮﻳﺪ. ﺍﮔﺮ ﺣﻮﺻﻠﻪ ﻭ ﺍﻣﮑﺎﻥ ﺁﻥ ﺭﺍ ﺩﺍﺭﻳﺪ، ﻳﮑﯽ ﺩﻭ ﭘﺘﻮﯼ ﺿﺨﻴﻢ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﭘﻮﺷﺸﯽ ﻋﺎﻟﯽ ﻭ ﺷﺎﻳﺪ ﺩﺍﺋﻤﯽ ﺑﺮﺍﯼ ﺳﻘﻒ ﺍﺗﺎﻗﺘﺎﻥ ﺑﺎﺷﻨﺪ.

ﺩﺭ ﺻﻮﺭﺗﯽ ﻫﻢ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﻋﺎﻳﻖ ﺳﺎﺯﯼ ﻭ ﺁﮐﻮﺳﺘﻴﮏ ﮐﺮﺩﻥ ﺍﺗﺎﻕ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺩﺍﺋﻤﯽ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ، ﺑﺮﺧﯽ ﻣﻮﺍﺩ ﺟﺎﻟﺐ، ﺍﺭﺯﺍﻥ ﺍﻣﺎ ﮐﺎﺭﺁﻣﺪ ﺭﺍ ﺩﺭ ﺑﺎﺯﺍﺭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻴﺎﺑﻴﺪ. ﻳﮑﯽ ﺍﺯ ﺍﻳﻦ ﮔﺰﻳﻨﻪ‌ﻫﺎ ﮐﻪ ﺷﺎﻳﺪ ﺑﺎ ﺷﻮﺧﯽ ﺍﺯ ﮐﻨﺎﺭ ﺁﻥ ﺑﮕﺬﺭﻳﺪ، ﺷﺎﻧﻪ ﺗﺨﻢ ﻣﺮﻍ ﺍﺳﺖ. ﺑﻠﻪ! ﺷﺎﻧﻪ ﺗﺨﻢ ﻣﺮﻍ ﺑﻪ ﺩﻟﻴﻞ ﺑﺎﻓﺖ ﻣﻘﻮﺍﻳﯽ ﻭ ﻫﻤﭽﻨﻴﻦ ﺷﮑﻞ ﺧﺎﺹ ﺧﻮﺩ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻓﻀﺎﯼ ﺍﺗﺎﻕ ﺭﺍ ﺁﮐﻮﺳﺘﻴﮏ ﮐﻨﺪ. ﺑﺎ ﭘﻮﺷﺎﻧﺪﻥ ﺩﻳﻮﺍﺭ‌ﻫﺎ ﻭ ﺳﻘﻒ ﺍﺗﺎﻕ ﺑﺎ ﻳﮏ ﻳﺎ ﺩﻭ ﻻﻳﻪ ﺍﺯ ﺍﻳﻦ ﭘﻮﺷﺶ ﺍﺭﺯﺍﻥ ﻭ ﺩﺭ ﺩﺳﺘﺮﺱ، ﻳﮏ ﻣﺤﻴﻂ ﭼﺸﻢ ﻧﻮﺍﺯ ﻭ ﺗﺎ ﺣﺪ ﻓﺮﺍﻭﺍﻧﯽ ﺁﮐﻮﺳﺘﻴﮏ ﺭﺍ ﻓﺮﺍﻫﻢ ﮐﺮﺩﻩ‌ﺍﻳﺪ.

ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﻣﻮﺍﺩﯼ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺻﻮﺭﺕ ﻣﺠﺰﺍ ﻭ ﻳﺎ ﺍﻳﻨﮑﻪ ﺑﻪ ﺻﻮﺭﺕ ﺗﺮﮐﻴﺒﯽ ﺑﺎ ﺷﺎﻧﻪ‌ﻫﺎﯼ ﺗﺨﻢ ﻣﺮﻍ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﮔﻴﺮﺩ، ﭘﺸﻢ ﺷﻴﺸﻪ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻋﺎﻳﻖ ﻗﺪﺭﺗﻤﻨﺪ ﺻﺪﺍ ﻭ ﺣﺮﺍﺭﺕ ﻣﺤﺴﻮﺏ ﻣﯽ‌ﺷﻮﺩ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﻋﺎﻳﻖ ﺗﺎ ﺣﺪﯼ ﻫﺰﻳﻨﻪ ﮐﺎﺭ ﺭﺍ ﺑﺎﻻ ﺑﺮﺩﻩ ﻭ ﻧﻴﺎﺯﻣﻨﺪ ﺩﻗﺖ ﻭ ﮐﻤﯽ ﺗﺨﺼﺺ ﺩﺭ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺳﺖ. ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻦ، ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﻪ ﻓﮑﺮ ﭘﻮﺷﺶ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺭﻭﯼ ﺁﻥ ﺑﺎﺷﻴﺪ ﺗﺎ ﺧﺎﺻﻴﺖ ﺁﮐﻮﺳﺘﻴﮏ ﺭﺍ ﺣﻔﻆ ﮐﻨﺪ.

ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﺍﻫﻞ ﮐﻤﯽ ﺧﺮﺝ ﮐﺮﺩﻥ ﺑﺎﺷﻴﺪ، ﺷﺎﻳﺪ ﺑﺘﻮﺍﻧﻴﺪ ﺑﺎ ﻗﻴﻤﺖ ﻣﻨﺎﺳﺒﯽ، ﭼﻨﺪ ﮐﺎﺷﯽ ﺁﮐﻮﺳﺘﻴﮏ (Acoustic Tile) ﻫﻢ ﺩﺭ ﺑﺎﺯﺍﺭ ﻳﺎﻓﺘﻪ ﻭ ﺩﻳﻮﺍﺭﻩ‌ﻫﺎﯼ ﺍﺗﺎﻕ ﮐﻮﭼﮏ‌ﺗﺎﻥ ﺭﺍ ﺑﺎ ﺁﻧﻬﺎ ﺑﭙﻮﺷﺎﻧﻴﺪ. ﻣﻄﻤﺌﻨﺎ ﺍﻳﻦ ﮔﺰﻳﻨﻪ، ﻳﮏ ﮔﺰﻳﻨﻪ ﺍﻳﺪﻩ ﺁﻝ ﺑﺮﺍﯼ ﻳﮏ ﺍﺳﺘﻮﺩﻳﻮ ﺧﺎﻧﮕﯽ ﮐﻢ ﻫﺰﻳﻨﻪ ﻭ ﺳﺎﺩﻩ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻫﻢ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﺁﻳﺎ ﻫﺰﻳﻨﻪ‌ﺍﯼ ﭼﻨﺪ ﺻﺪ ﻫﺰﺍﺭ ﺗﻮﻣﺎﻧﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺷﯽ‌ﻫﺎ ﺑﺮﺍﻳﺘﺎﻥ ﺑﻪ ﺻﺮﻓﻪ ﺍﺳﺖ؟ ﺍﻣﺎ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﺮﺍﯼ ﮐﻴﻔﻴﺖ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺣﺘﻤﺎ ﺑﺎ ﺍﺭﺯﺵ ﺍﺳﺖ.
