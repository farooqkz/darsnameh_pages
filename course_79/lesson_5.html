ﺩﺭﺱ ﭼﻬﺎﺭﻡ - ﺍﺑﺰﺍﺭﻫﺎﯼ ﻻﺯﻡ ﺑﺮﺍﯼ ﺿﺒﻂ ﺻﺪﺍ

ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﺑﻪ ﺷﻴﻮﻩ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﻳﮏ ﺍﺳﺘﻮﺩﻳﻮﯼ ﺧﺎﻧﮕﯽ ﺑﺮﺍﯼ ﺿﺒﻂ ﺻﺪﺍ ﭘﺮﺩﺍﺧﺘﻴﻢ. ﺍﻳﻨﮑﻪ ﭼﮕﻮﻧﻪ ﭘﺎﺩﮐﺴﺘﺘﺎﻥ ﺭﺍ ﺑﺪﻭﻥ ﺍﻧﮑﻌﺎﺱ ﺻﺪﺍ ﻭ ﺍﮐﻮ ﺿﺒﻂ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺍﻳﻨﮑﻪ ﺻﺪﺍﻫﺎﯼ ﻣﺰﺍﺣﻢ ﺑﻴﺮﻭﻥ ﺭﺍ ﻫﻨﮕﺎﻡ ﺿﺒﻂ ﺑﻪ ﺣﺪﺍﻗﻞ ﺑﺮﺳﺎﻧﻴﺪ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ ﻭ ﺩﺭ ﺍﺩﺍﻣﻪ ﻗﺼﺪ ﺩﺍﺭﻳﻢ ﺑﻪ ﻟﻴﺴﺘﯽ ﺍﺯ ﺗﺠﻬﻴﺰﺍﺕ ﻣﻮﺭﺩ ﻧﻴﺎﺯ ﺑﺮﺍﯼ ﺿﺒﻂ ﺻﺪﺍ ﺑﭙﺮﺩﺍﺯﻳﻢ. ﺍﻟﺒﺘﻪ‌‌ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﭘﻴﺶ ﺍﺯ ﺍﻳﻦ ﮔﻔﺘﻴﻢ ﺍﻳﻦ ﮐﺎﺭ ﺗﻨﻬﺎ ﺑﺎ ﻳﮏ ﻟﭗ ﺗﺎﭖ ﻭ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻣﻌﻤﻮﻟﯽ ﻫﻢ ﻗﺎﺑﻞ ﺍﻧﺠﺎﻡ ﺍﺳﺖ ﻭ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﺑﺨﺶ ﻳﮏ ﭘﺎﺩﮐﺴﺖ، ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ «ﭼﻪ ﭼﻴﺰﯼ ﺑﺮﺍﯼ ﮔﻔﺘﻦ ﻭ ﺍﺭﺍﺋﻪ ﺩﺍﺭﻳﻢ» ﻭ ﺷﻴﻮﻩ ﺍﺭﺍﺋﻪ ﺍﺯ ﻧﻈﺮ ﺍﻫﻤﻴﺖ ﺩﺭ ﺭﺗﺒﻪ ﺑﻌﺪﯼ ﻗﺮﺍﺭ ﺩﺍﺭﺩ.

ﺍﻣﺎ ﺍﺯ ﺳﻮﯼ ﺩﻳﮕﺮ ﺑﺎﻳﺪ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﻧﻴﺰ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﺭﺍﺋﻪ ﺯﻳﺒﺎ، ﺻﺤﻴﺢ ﻭ ﺑﺎ ﮐﻴﻔﻴﺖ ﻣﺤﺘﻮﺍ ﺗﺎﺛﻴﺮ ﺯﻳﺎﺩﯼ ﺑﺮ ﺟﺬﺍﺑﻴﺖ ﺁﻥ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﻫﺮ ﭼﻪ ﮐﻴﻔﻴﺖ ﺻﺪﺍﯼ ﻣﺎ ﺑﻬﺘﺮ ﺑﺎﺷﺪ ﻭ ﺷﻨﻮﻧﺪﻩ ﺁﻥ ﺭﺍ ﺷﻔﺎﻑ‌ﺗﺮ ﻭ ﺑﻬﺘﺮ ﺑﺸﻨﻮﺩ، ﺑﻴﺸﺘﺮ ﺟﺬﺏ ﺁﻥ ﺷﺪﻩ ﻭ ﺩﺭﮎ ﺩﺭﺳﺖ ﺗﺮﯼ ﺍﺯ ﻣﺤﺘﻮﺍ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﻣﻌﺮﻓﯽ ﺑﺮﺧﯽ ﺍﺑﺰﺍﺭ‌ﻫﺎ ﺑﭙﺮﺩﺍﺯﻳﻢ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺩﺭ ﺿﺒﻂ ﺑﻬﺘﺮ ﻭ ﺑﺎ ﮐﻴﻔﻴﺖ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺗﺎﺛﻴﺮ ﻓﺮﺍﻭﺍﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ.

 

ﻣﻴﮑﺮﻭﻓﻮﻥ
ﺍﻭﻟﻴﻦ ﻭ ﺿﺮﻭﺭﯼ‌ﺗﺮﻳﻦ ﺍﺑﺰﺍﺭ ﺳﺎﺧﺖ ﭘﺎﺩﮐﺴﺖ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺍﺳﺖ. ﺷﺎﻳﺪ ﺑﺘﻮﺍﻥ ﺁﻥ ﺭﺍ ﻫﻢ ﺭﺩﻩ ﺑﺎ ﻗﻠﻢ ﻣﻮﯼ ﻳﮏ ﻧﻘﺎﺵ ﻳﺎ ﺩﻭﺭﺑﻴﻦ ﻳﮏ ﻋﮑﺎﺱ ﺩﺍﻧﺴﺖ ﻭ ﺑﻪ‌‌ ﻫﻤﺎﻥ ﻧﺴﺒﺖ ﺷﻤﺎ ﺍﻧﺘﺨﺎﺏ‌ﻫﺎﯼ ﻓﺮﺍﻭﺍﻧﯽ ﺭﺍ ﻧﻴﺰ ﭘﻴﺶ ﺭﻭ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﻳﮏ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺳﺎﺩﻩ ﭼﻨﺪ ﻫﺰﺍﺭ ﺗﻮﻣﺎﻧﯽ ﻣﻌﻤﻮﻟﯽ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ، ﻳﺎ ﺣﺘﯽ ﺍﺯ‌‌ ﻫﻤﺎﻥ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺗﻌﺒﻴﻪ ﺷﺪﻩ ﺭﻭﯼ ﻫﺪﻓﻮﻧﺘﺎﻥ ﺑﻬﺮﻩ ﺑﺒﺮﻳﺪ. ﻳﺎ ﺍﻳﻨﮑﻪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺧﺮﻳﺪ ﻳﮏ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺣﺮﻓﻪ‌ﺍﯼ ﺧﺒﺮﻧﮕﺎﺭﯼ ﺑﺎﺷﻴﺪ. ﻭ ﺍﻟﺒﺘﻪ ﻣﻴﮑﺮﻭﻓﻮﻥ‌ﻫﺎﯼ ﺍﺳﺘﻮﺩﻳﻮﻳﯽ ﺟﺬﺍﺏ ﭼﻨﺪ ﺻﺪ ﻫﺰﺍﺭ ﺗﻮﻣﺎﻧﯽ ﻫﻢ ﺩﺭ ﺩﺍﻳﺮﻩ ﺍﻧﺘﺨﺎﺑﺘﺎﻥ ﻗﺮﺍﺭ ﺩﺍﺭﻧﺪ.

ﺷﺎﻳﺪ ﻫﻢ ﺑﺨﻮﺍﻫﻴﺪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺳﻴﺴﺘﻢ‌ﻫﺎﯼ ﮐﺎﻣﻞ ﺍﺳﺘﻮﺩﻳﻮ ﭘﺎﺩﮐﺴﺖ ﺭﻓﺘﻪ ﻭ ﺧﻴﺎﻝ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺑﺎﺑﺖ ﻫﻤﻪ ﭼﻴﺰ ﺭﺍﺣﺖ ﮐﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻳﮏ ﮐﺎﺭﺕ ﺻﺪﺍﯼ ﺍﮐﺴﺘﺮﻧﺎﻝ، ﻫﺪﻓﻮﻥ، ﻣﻴﮑﺮﻭﻓﻮﻥ ﻭ ﻣﻴﮑﺴﺮ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﻭ ﮐﻴﻔﻴﺖ ﺻﺪﺍﯼ ﭘﺎﺩﮐﺴﺖ ﺷﻤﺎ ﺩﺳﺖ ﮐﻤﯽ ﺍﺯ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺣﺮﻓﻪ‌ﺍﯼ ﺭﺍﺩﻳﻮ ﻭ ﺗﻠﻮﻳﺰﻳﻮﻧﯽ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺷﺖ. ﺍﮔﺮﭼﻪ ﻣﻄﻤﺌﻨﺎ ﺑﺮﺍﯼ ﭼﻨﻴﻦ ﮐﻴﻔﻴﺘﯽ ﺑﺎﻳﺪ ﭼﻨﺪ ﻣﻴﻠﻴﻮﻥ ﺗﻮﻣﺎﻧﯽ ﻫﺰﻳﻨﻪ ﮐﻨﻴﺪ.

 

ﻫﺮﭼﻨﺪ ﺩﺭ ﺍﻳﻦ ﻣﻴﺎﻥ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻧﮑﺘﻪ‌ﺍﯼ ﮐﻪ ﺑﺎﻳﺪ ﺑﺨﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﺑﺰﺍﺭﯼ ﺑﺎ ﮐﻤﺘﺮﻳﻦ ﻧﻮﻳﺰ ﻭ ﻭﺍﺿﺢ‌ﺗﺮﻳﻦ ﺻﺪﺍﯼ ﻣﻤﮑﻦ ﺍﺳﺖ. ﮔﺎﻫﯽ ﺷﻤﺎ ﻻﺯﻡ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺍﺯ ﻳﮏ ﻣﻴﮑﺮﻭﻓﻮﻥ ﮔﺮﺍﻥ ﻗﻴﻤﺖ ﺍﺳﺘﻮﺩﻳﻮﻳﯽ ﺑﻬﺮﻩ ﺑﺒﺮﻳﺪ ﻭ ﮔﺎﻫﯽ ﻫﻢ ﻓﺮﺩﯼ ﺑﻪ ﻳﮏ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻣﻴﺎﻥ ﻗﻴﻤﺖ ﺍﮐﺘﻔﺎ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺳﺮﺍﻍ ﮐﻠﮏ‌ﻫﺎﻳﯽ ﭼﻮﻥ ﭘﺎﭖ ﻓﻴﻠﺘﺮ ﻣﯽ‌ﺭﻭﺩ ﺗﺎ ﺗﻴﺰﯼ ﺻﺪﺍ ﮔﺮﻓﺘﻪ ﺷﻮﺩ. ﺑﻪ ﻫﺮ ﺣﺎﻝ ﺑﻬﺘﺮ ﺍﺳﺖ ﺣﺘﻤﺎ ﺍﺯ ﻳﮏ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻣﻨﺎﺳﺐ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﮐﻪ ﻇﺎﻫﺮ ﺧﻮﺑﯽ ﻫﻢ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺑﻪ ﺧﺼﻮﺹ ﺍﮔﺮ ﺑﻪ ﺩﻧﺒﺎﻝ ﺗﻬﻴﻪ ﻭﻳﺪﻳﻮ ﮐﺴﺖ ﻫﻢ ﻫﺴﺘﻴﺪ.

ﻭ ﺁﺧﺮﻳﻦ ﻧﮑﺘﻪ ﺩﺭ ﺧﺼﻮﺹ ﻣﻴﮑﺮﻭﻓﻮﻥ‌ﻫﺎ ﺍﻳﻨﮑﻪ ﭘﺎﻳﻪ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ. ﺯﻳﺮﺍ ﻫﻨﮕﺎﻡ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺩﺳﺘﺎﻧﺘﺎﻥ ﺁﺯﺍﺩ ﺑﺎﺷﻨﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺑﺎ ﮐﺎﻣﭙﻴﻮﺗﺮﺗﺎﻥ ﮐﺎﺭ ﮐﻨﻴﺪ، ﺑﺮﮔﻪ‌ﻫﺎﯼ ﻳﺎﺩﺩﺍﺷﺘﺘﺎﻥ ﺭﺍ ﺳﺮ ﻭ ﺳﺎﻣﺎﻥ ﺩﻫﻴﺪ، ﻳﺎ ﺍﮔﺮ ﺩﺭ ﺣﺎﻝ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﻭﻳﺪﻳﻮﻳﯽ ﻫﺴﺘﻴﺪ، ﺍﺯ ﺣﺮﮐﺎﺕ ﺩﺳﺖ ﺑﺮﺍﯼ ﺭﺳﺎﻧﺪﻥ ﻣﻔﻬﻮﻡ ﺑﻬﺮﻩ ﺑﺒﺮﻳﺪ. ﭘﺲ ﭼﻨﺪﺍﻥ ﺧﻮﺷﺎﻳﻨﺪ ﻧﻴﺴﺖ ﻫﻤﻴﺸﻪ ﻳﮏ ﺩﺳﺘﺘﺎﻥ ﺩﺭﮔﻴﺮ ﻧﮕﻬﺪﺍﺭﯼ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺑﺎﺷﺪ ﻭ ﻫﻤﭽﻨﻴﻦ ﺣﺮﮐﺖ‌ﻫﺎﯼ ﻧﺎﺧﻮﺩﺍﮔﺎﻩ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺍﻳﺠﺎﺩ ﻧﻮﻳﺰ ﻫﻢ ﺑﮑﻨﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﻫﻢ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺿﺒﻂ ﺻﺪﺍ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺍﻳﺴﺘﺎﺩﻩ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﻳﮏ ﭘﺎﻳﻪ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺭﻭﻣﻴﺰﯼ ﮐﺎﺭﺗﺎﻥ ﺭﺍ ﺭﺍﻩ ﻧﻤﯽ‌ﺍﻧﺪﺍﺯﺩ!

 

ﻫﺪﻓﻮﻥ
ﻫﻨﮕﺎﻡ ﺿﺒﻂ ﺻﺪﺍ، ﻻﺯﻡ ﺍﺳﺖ ﺷﻤﺎ ﺻﺪﺍﯼ ﺩﺭﻳﺎﻓﺘﯽ ﺗﻮﺳﻂ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺭﺍ ﺑﺸﻨﻮﻳﺪ. ﺯﻳﺮﺍ ﮐﻨﺘﺮﻝ‌ ﮔﺎﻩ ﻭ ﺑﻴﮕﺎﻩ ﺁﻥ ﺩﺭ ﻃﻮﻝ ﺑﺮﻧﺎﻣﻪ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺍﺯ ﻣﺸﮑﻼﺕ ﺍﺣﺘﻤﺎﻟﯽ ﻣﻄﻠﻊ ﺷﺪﻩ ﻭ‌‌ ﻫﻤﺎﻥ ﻫﻨﮕﺎﻡ ﺁﻧﻬﺎ ﺭﺍ ﺑﺮﻃﺮﻑ ﮐﻨﻴﺪ. ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺠﺒﻮﺭ ﺷﻮﻳﺪ ﻫﻨﮕﺎﻡ ﻭﻳﺮﺍﻳﺶ ﺑﺮﻧﺎﻣﻪ ﺩﻭﺑﺎﺭﻩ ﺳﺮﺍﻍ ﺿﺒﻂ ﺑﻴﺎﻳﺪ. ﺍﻣﺎ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﺳﭙﻴﮑﺮﻫﺎﯼ ﮐﺎﻣﭙﻴﻮ‌ﺗﺮ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﻧﻴﺴﺖ. ﭼﻮﻥ ﺑﺎﺯ ﭘﺨﺶ ﺻﺪﺍﯼ ﺍﺳﭙﻴﮑﺮ ﺩﺭ ﻓﻀﺎﯼ ﺍﺗﺎﻕ ﺑﺎﻋﺚ ﺿﺒﻂ ﺩﻭﺑﺎﺭﻩ ﺁﻥ ﺍﺯ ﻃﺮﻳﻖ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻭ ﺍﻳﺠﺎﺩ ﺻﺪﺍﯼ ﻣﺰﺍﺣﻢ ﻭ ﺣﺘﯽ ﺍﮐﻮ ﺷﺪﻥ ﺻﺪﺍ ﻣﯽ‌ﺷﻮﺩ. ﭘﺲ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻳﮏ ﻫﺪﻓﻮﻥ ﺧﻮﺏ ﻭ ﺭﺍﺣﺖ ﻫﻨﮕﺎﻡ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﺍﻣﺮﯼ ﺿﺮﻭﺭﯼ ﺍﺳﺖ. ﺍﮔﺮ ﺑﻪ ﺩﻧﺒﺎﻝ ﮐﻴﻔﻴﺖ ﺧﻮﺏ ﭘﺎﺩﮐﺴﺖ ﻫﺴﺘﻴﺪ، ﺣﺘﻤﺎ ﺍﺯ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻭ ﻫﺪﻓﻮﻥ ﺟﺪﺍﮔﺎﻧﻪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ، ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﻣﻴﮑﺮﻭﻓﻮﻧﯽ ﺑﺎ ﮐﻴﻔﻴﺖ ﻣﻨﺎﺳﺐ ﺗﻬﻴﻪ ﮐﻨﻴﺪ.

ﺍﻟﺒﺘﻪ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺑﻪ ﺩﻧﺒﺎﻝ ﮐﻢ ﮐﺮﺩﻥ ﻫﺰﻳﻨﻪ‌ﻫﺎ ﺑﺎﺷﻴﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺑﻴﻦ ﻫﺪﻓﻮﻥ ﻣﻴﮑﺮﻭﻓﻮﻥ‌ﻫﺎﯼ ﻣﻮﺟﻮﺩ، ﮔﺰﻳﻨﻪ ﻣﻨﺎﺳﺒﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﺟﺎﯼ ﮐﻤﺘﺮﯼ ﻫﻢ ﮔﺮﻓﺘﻪ ﺷﺪﻩ ﻭ ﮐﺎﺭ ﺿﺒﻂ ﺻﺪﺍ ﻫﻢ ﺑﺎ ﺩﺭﺩﺳﺮ ﮐﻤﺘﺮﯼ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﮔﻴﺮﺩ. ﺍﻟﺒﺘﻪ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﺣﺪﺍﻗﻞ ﺍﺯ ﻫﺪﺳﺘﯽ (ﺗﺮﮐﻴﺐ ﻫﺪﻓﻮﻥ ﻣﻴﮑﺮﻭﻓﻮﻥ) ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﮐﻪ ﭘﺎﻳﻪ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺁﻥ ﺍﻧﻌﻄﺎﻑ ﭘﺬﻳﺮ ﺑﺎﺷﺪ ﻭ ﺑﺘﻮﺍﻧﻴﺪ ﻓﺎﺻﻠﻪ ﺁﻥ ﺭﺍ ﺑﺎ ﺩﻫﺎﻥ ﺧﻮﺩ ﺗﻨﻈﻴﻢ ﮐﻨﻴﺪ. ﻧﮑﺘﻪ ﻣﻬﻢ ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺮ ﺭﻭﯼ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺁﻥ ﭘﻮﺷﺶ ﺍﺑﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺯﻳﺮﺍ ﺍﻳﻦ ﭘﻮﺷﺶ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﺎﺭ ﭘﺎﭖ ﻓﻴﻠﺘﺮ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﺪ.

 

ﮐﺎﺑﻞ
ﻋﻼﻭﻩ ﺑﺮ ﻣﻴﮑﺮﻭﻓﻮﻥ، ﺷﻤﺎ ﺑﻪ ﻣﻘﺪﺍﺭﯼ ﮐﺎﺑﻞ ﻫﻢ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ. ﺩﺭﺳﺖ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﮐﻨﺎﺭ ﻟﭗ ﺗﺎﭖ ﻳﺎ ﮐﺎﻣﭙﻴﻮﺗﺮﺗﺎﻥ ﻭ ﺩﺭ ﺧﺎﻧﻪ ﻳﺎ ﻣﺤﻞ ﮐﺎﺭ ﻣﺸﻐﻮﻝ ﺿﺒﻂ ﭘﺎﺩﮐﺴﺖ ﻫﺴﺘﻴﺪ. ﺍﻣﺎ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻫﻢ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﺷﺎﻳﺪ ﺭﻭﺯﯼ ﺑﺨﻮﺍﻫﻴﺪ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺯﻧﺪﻩ ﺭﺍ ﺿﺒﻂ ﮐﻨﻴﺪ. ﻳﺎ ﺍﻳﻨﮑﻪ ﻻﺯﻡ ﺑﺎﺷﺪ ﺍﺯ ﻳﮏ ﻣﺮﺍﺳﻢ ﻳﺎ ﻧﻤﺎﻳﺸﮕﺎﻩ ﺧﺎﺹ ﮔﺰﺍﺭﺵ ﺗﻬﻴﻪ ﮐﻨﻴﺪ. ﭘﺲ ﻫﻤﻴﺸﻪ ﺑﻬﺘﺮ ﺍﺳﺖ ﻣﻘﺪﺍﺭﯼ ﺳﻴﻢ ﺍﺿﺎﻓﯽ ﺩﺭ ﺩﺳﺖ ﻭ ﭘﺎﻳﺘﺎﻥ ﭘﻴﭽﻴﺪﻩ ﺑﺎﺷﺪ ﺗﺎ ﺍﻳﻨﮑﻪ ﮔﺎﻫﯽ ﺍﻭﻗﺎﺕ ﺳﻴﻢ ﮐﻢ ﺑﻴﺎﻭﺭﻳﺪ.

 

ﭘﺎﭖ ﻓﻴﻠﺘﺮ
ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﺍﺑﺰﺍﺭﻫﺎﯼ ﺿﺮﻭﺭﯼ ﺑﺮﺍﯼ ﺿﺒﻂ ﺻﺪﺍ، ﻓﻴﻠﺘﺮ pop ﻳﺎ ﭘﺎﭖ ﻓﻴﻠﺘﺮ ﺍﺳﺖ. ﺑﺎ ﺍﻳﻦ ﻓﻴﻠﺘﺮ ﺻﺪﺍﯼ ﻫﻮﺍﯼ ﺗﻨﺪ ﺑﺮﺧﯽ ﺣﺮﻭﻑ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﮔﺮﻓﺘﻪ ﺷﺪﻩ ﻭ ﺿﺒﻂ ﻳﮑﻨﻮﺍﺧﺖ ﻭ ﻳﮑﺪﺳﺖ ﺗﺮﯼ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﺍﮔﺮ ﺩﻗﺖ ﮐﻨﻴﺪ ﻫﻨﮕﺎﻡ ﺍﺩﺍﯼ ﺑﺮﺧﯽ ﺣﺮﻭﻑ، ﺑﻪ ﺧﺼﻮﺹ ﺣﺮﻑ «ﭖ» ﻭ «ﺕ» ﺣﺠﻢ ﻫﻮﺍﯼ ﺯﻳﺎﺩﯼ ﺑﻪ ﺳﻤﺖ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺷﻠﻴﮏ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺻﻮﺭﺕ ﮐﻨﺘﺮﻝ ﻧﺸﺪﻥ ﺗﻮﺳﻂ ﻳﮏ ﻓﻴﻠﺘﺮ، ﺗﺎﺛﻴﺮ ﺑﺪﯼ ﺭﻭﯼ ﺻﺪﺍﯼ ﺿﺒﻂ ﺷﺪﻩ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﺑﺎ ﺧﺮﻳﺪ ﻳﮏ ﻓﻴﻠﺘﺮ ﭘﺎﭖ (ﺩﺭ ﻓﺎﺭﺳﯽ ﺑﻪ ﺑﺎﺩﮔﻴﺮ ﻣﺸﻬﻮﺭ ﺍﺳﺖ) ﻭ ﻳﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻳﮏ ﻓﻴﻠﺘﺮ ﺩﺳﺖ ﺳﺎﺯ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺍﻳﻦ ﻧﻮﻳﺰ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﺪ. ﺍﻟﺒﺘﻪ ﺑﺨﺸﯽ ﺍﺯ ﮐﻨﺘﺮﻝ ﺍﻳﻦ ﻧﻮﻳﺰ ﻫﻢ ﺑﻪ ﺗﻤﺮﻳﻨﺎﺕ ﺗﻠﻔﻆ ﺻﺤﻴﺢ ﺣﺮﻭﻑ ﮐﻠﻤﺎﺕ ﻭ ﺻﺪﺍ‌ﻫﺎ ﺗﻮﺳﻂ ﺷﻤﺎ ﺑﺴﺘﮕﯽ ﺩﺍﺭﺩ. ﺍﻳﻦ ﺗﻤﺮﻳﻨﺎﺕ ﺑﻴﺸﺘﺮ ﺩﺭ ﮐﻼﺱ‌ﻫﺎﯼ ﮔﻮﻳﻨﺪﮔﯽ ﻭ ﺑﺎﺯﻳﮕﺮﯼ ﻣﺘﺪﺍﻭﻝ ﻫﺴﺘﻨﺪ.

 

ﺑﺮﺍﯼ ﺳﺎﺧﺖ ﻳﮏ ﭘﺎﭖ ﻓﻴﻠﺘﺮ ﺧﺎﻧﮕﯽ ﮐﺎﻓﯽ ﺍﺳﺖ ﻳﮏ ﺟﺎ ﺭﺧﺘﯽ ﺳﻴﻤﯽ ﻧﺎﺯﮎ ﺭﺍ ﺑﻪ ﺷﮑﻞ ﺩﺍﻳﺮﻩ ﺩﺭﺁﻭﺭﺩﻩ ﻭ ﻳﮏ ﺟﻮﺭﺍﺏ ﻧﺴﺒﺘﺎ ﻧﺎﺯﮎ ﺯﻧﺎﻧﻪ ﺭﺍ ﺭﻭﯼ ﺁﻥ ﺑﮑﺸﻴﺪ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺨﺸﯽ ﺍﺯ ﻣﻴﻠﻪ ﺑﺎﺯ ﺷﺪﻩ ﺟﺎ ﺭﺧﺘﯽ ﺭﺍ ﻫﻢ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺎﻳﻪ ﭘﺎﭖ ﻓﻴﻠﺘﺮ ﺑﺮﺍﯼ ﺍﺗﺼﺎﻝ ﺁﻥ ﺑﻪ ﻣﻴﮑﺮﻭﻓﻮﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﺑﻪ ﺩﻧﺒﺎﻝ ﻳﮏ ﭘﺎﭖ ﻓﻴﻠﺘﺮ ﺩﺳﺖ ﺳﺎﺯ ﺣﺮﻓﻪ‌ﺍﯼ‌ﺗﺮ ﻭ ﺯﻳﺒﺎ‌ﺗﺮ ﺑﺎﺷﻴﺪ، ﺍﻳﻦ ﺭﺍﻫﻨﻤﺎﯼ ﻗﺪﻡ ﺑﻪ ﻗﺪﻡ ﺑﺎ ﺗﻮﺿﻴﺤﺎﺕ ﻭ ﻋﮑﺲ ﮐﺎﻣﻞ، ﮔﺰﻳﻨﻪ ﻣﻨﺎﺳﺒﯽ ﺍﺳﺖ.

 

ﺍﮐﻨﻮﻥ ﺷﻤﺎ ﺗﻘﺮﻳﺒﺎ ﺗﻤﺎﻣﯽ ﺍﺑﺰﺍﺭﻫﺎﯼ ﻣﻮﺭﺩ ﻧﻴﺎﺯ ﺑﺮﺍﯼ ﺿﺒﻂ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻳﺪ. ﭘﺲ ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﺎﻣﭙﻴﻮ‌ﺗﺮ ﺭﺍ ﺭﻭﺷﻦ ﮐﺮﺩﻩ ﻭ ﺿﺒﻂ ﺍﻭﻟﻴﻦ ﻗﺴﻤﺖ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﺪ. ﺍﻣﺎ ﺻﺒﺮ ﮐﻨﻴﺪ، ﻫﻨﻮﺯ ﺑﺮﺧﯽ ﭼﻴﺰ‌ﻫﺎ ﮐﻢ ﺍﺳﺖ. ﺩﺍﺳﺘﺎﻥ ﻧﺮﻡ ﺍﻓﺰﺍﺭ‌ﻫﺎ ﺑﺎﻗﯽ ﻣﺎﻧﺪﻩ...
