ﺩﺭﺱ ﺩﻭﻡ - ﭼﺮﺍ ﺑﺎﻳﺪ ﭘﺎﺩﮐﺴﺖ ﺑﺴﺎﺯﻳﻢ؟

ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﺑﻪ ﭼﻴﺴﺘﯽ ﭘﺎﺩﮐﺴﺖ، ﺗﺎﺭﻳﺨﭽﻪ ﻭ ﺍﻧﻮﺍﻉ ﺁﻥ ﭘﺮﺩﺍﺧﺘﻴﻢ. ﺍﻣﺎ ﺑﮕﺬﺍﺭﻳﺪ ﻗﺒﻞ ﺍﺯ ﺷﺮﻭﻉ ﺑﺤﺚ‌ﻫﺎﯼ ﺗﺨﺼﺼﯽ، ﺍﻭﻝ ﺍﺯ ﻫﻤﻪ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺑﭙﺮﺩﺍﺯﻳﻢ ﮐﻪ ﺁﻳﺎ ﺍﺻﻼ ﻣﺎ ﻧﻴﺎﺯﯼ ﺑﻪ ﺳﺎﺧﺖ ﭘﺎﺩﮐﺴﺖ ﺩﺍﺭﻳﻢ؟ ﭼﺮﺍ ﺑﺎﻳﺪ ﭘﺎﺩﮐﺴﺖ ﺑﺴﺎﺯﻳﻢ؟ ﻣﺰﺍﻳﺎﯼ ﭘﺎﺩﮐﺴﺖ ﭼﻪ ﻫﺴﺘﻨﺪ؟ ﺁﻳﺎ ﭘﺎﺩﮐﺴﺖ ﺑﻪ ﺩﺭﺩ ﻣﺎ ﻣﯽ‌ﺧﻮﺭﺩ؟ ﻭ ﺳﻮﺍﻝ‌ﻫﺎﻳﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ.

 

ﻣﺰﺍﻳﺎﯼ ﭘﺎﺩﮐﺴﺖ
۱- ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﺣﻮﺯﻩ ﺟﺪﻳﺪﯼ ﺍﺯ ﮐﺎﺭﺑﺮﺍﻥ
ﻣﺮﺩﻡ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺍﺯ ﺭﺳﺎﻧﻪ ﺑﻪ ﺷﮑﻞ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪ. ﻟﺬﺍ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﻳﮏ ﺍﻧﺘﺨﺎﺏ ﺭﺳﺎﻧﻪ‌ﺍﯼ ﺗﺎﺯﻩ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﻤﮏ ﺧﻮﺑﯽ ﺟﻬﺖ ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﮔﺮﻭﻩ ﺟﺪﻳﺪﯼ ﺍﺯ ﮐﺎﺭﺑﺮﺍﻥ ﺑﺎﺷﺪ. ﭘﺎﺩﮐﺴﺖ‌ﻫﺎﯼ ﺻﻮﺗﯽ (ﺑﺮ ﺧﻼﻑ ﭘﺎﺩﮐﺴﺖ ﻭﻳﺪﻳﻮﻳﯽ) ﺑﻪ ﮐﺎﺭﺑﺮ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻨﺪ ﭼﻨﺪ ﻭﻇﻴﻔﻪ‌ﺍﯼ ﻭ ﻣﺎﻟﺘﯽ ﺗﺴﮏ ﺑﺎﺷﺪ. ﻳﻌﻨﯽ ﺁﻧﻬﺎ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻫﻤﺰﻣﺎﻥ ﺑﺎ ﮔﻮﺵ ﺩﺍﺩﻥ ﺑﻪ ﭘﺎﺩﮐﺴﺖ ﺷﻤﺎ، ﺑﻪ ﮐﺎﺭﻫﺎﯼ ﺩﻳﮕﺮﺷﺎﻥ ﻫﻢ ﺑﺮﺳﻨﺪ: ﺑﻪ ﺟﺴﺘﺠﻮﯼ ﻭﺏ ﺑﭙﺮﺩﺍﺯﻧﺪ ﻳﺎ ﺣﺘﯽ ﻣﺸﻐﻮﻝ ﺭﺍﻧﻨﺪﮔﯽ، ﻭﺭﺯﺵ ﻳﺎ ﮐﺎﺭﻫﺎﯼ ﻣﻨﺰﻝ ﺷﻮﻧﺪ.

۲- ﺍﻳﺠﺎﺩ ﺭﺍﺑﻄﻪ ﻧﺰﺩﻳﮏ‌ﺗﺮ ﺑﺎ ﺷﻨﻮﻧﺪﮔﺎﻥ
ﭘﺎﺩﮐﺴﺖ ﺑﺎ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﺻﺪﺍﯼ ﺷﻤﺎ ﺑﻪ ﻭﺏ‌ﺳﺎﻳﺖ ﺷﺨﺼﯽ ﻳﺎ ﺷﺮﮐﺘﯽ ﺩﺭ ﻣﺸﺨﺺ ﮐﺮﺩﻥ ﻳﮏ ﻧﺸﺎﻥ ﻭ ﻋﻼﻣﺖ ﻣﺸﺨﺼﻪ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﺷﻨﺎﺧﺘﻪ ﺷﺪﻥ ﺷﻤﺎ ﻳﺎ ﮐﺎﺭ ﻭ ﮐﺴﺐ‌ﺗﺎﻥ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ. ﺯﻳﺮﺍ ﭘﺎﺩﮐﺴﺖ‌ﻫﺎ ﮐﺎﻣﻼ ﺍﻧﺘﺨﺎﺑﯽ ﻫﺴﺘﻨﺪ ﻭ ﺗﻨﻬﺎ ﮐﺴﺎﻧﯽ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺻﺪﺍﯼ ﺷﻤﺎ ﺭﺍ ﺑﺸﻨﻮﻧﺪ ﻭ ﻳﺎ ﺟﺬﺏ ﺁﻧﭽﻪ ﻣﯽ‌ﮔﻮﻳﻴﺪ ﺷﺪﻩ‌ﺍﻧﺪ، ﺁﻥ ﺭﺍ ﺩﺍﻧﻠﻮﺩ ﻣﯽ‌ﮐﻨﻨﺪ.

۳- ﺑﺮﺟﺴﺘﻪ ﮐﺮﺩﻥ ﺗﻮﺍﻧﺎﻳﯽ ﺷﻤﺎ
ﺑﺎ ﺍﺭﺍﺋﻪ ﺑﺮﺧﯽ ﺍﻃﻼﻋﺎﺕ ﻣﻔﻴﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﻮﺍﻧﺎﻳﯽ‌ﻫﺎ ﻭ ﻣﻬﺎﺭﺕ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺯﻣﻴﻨﻪ ﮐﺎﺭﯼ‌ﺗﺎﻥ ﺑﺮﺟﺴﺘﻪ ﮐﻨﻴﺪ ﻭ ﺍﻳﻦ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺁﻳﻨﺪﻩ ﺑﺮﺍﯼ ﻣﺮﺩﻣﯽ ﮐﻪ ﻧﻴﺎﺯ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﭼﻨﻴﻦ ﻣﻬﺎﺭﺕ‌ﻫﺎﻳﯽ ﺩﺍﺭﻧﺪ، ﺍﻋﺘﻤﺎﺩ ﺁﻓﺮﻳﻦ ﺑﺎﺷﺪ. ﻫﻤﭽﻨﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺳﻤﻴﻨﺎﺭ‌ﻫﺎ ﻭ ﮐﻨﻔﺮﺍﻧﺲ‌‌ﻫﺎﻳﺘﺎﻥ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﭘﺎﺩﮐﺴﺖ ﺿﺒﻂ ﮐﺮﺩﻩ ﻭ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﻓﺮﺍﺩﯼ ﻗﺮﺍﺭ ﺩﻫﻴﺪ ﮐﻪ ﺍﻣﮑﺎﻥ ﺷﺮﮐﺖ ﺩﺭ ﺁﻧﻬﺎ ﺭﺍ ﻧﺪﺍﺭﻧﺪ. ﺑﺮﺟﺴﺘﻪ ﺷﺪﻥ ﺗﻮﺍﻧﺎﻳﯽ‌ﻫﺎﻳﺘﺎﻥ ﺣﺘﯽ ﺍﻣﮑﺎﻥ ﺑﻪ ﺍﺭﻣﻐﺎﻥ ﺁﻭﺭﺩﻥ ﺑﺎﺯﺍﺭﻫﺎﯼ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺟﺪﻳﺪ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﺎﺷﺪ.

۴- ﺍﻳﺠﺎﺩ ﭘﺘﺎﻧﺴﻴﻞ ﺗﺒﻠﻴﻐﺎﺗﯽ ﻭ ﺩﺭﺁﻣﺪﺯﺍﻳﯽ
ﺍﮔﺮ ﭘﺎﺩﮐﺴﺖ ﺷﻤﺎ ﻣﺤﺒﻮﺑﻴﺖ ﻻﺯﻡ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﺪ، ﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺘﻮﺍﻧﻴﺪ ﺑﺨﺶ‌ﻫﺎﯼ ﺗﺒﻠﻴﻐﺎﺗﯽ ﺩﺭ ﻣﻴﺎﻥ ﺁﻥ ﺑﺮﺍﯼ ﻓﺮﻭﺵ ﻋﺮﺿﻪ ﮐﻨﻴﺪ.

۵- ﭘﺎﺳﺦ ﺑﻪ ﺑﺎﺯﺧﻮﺭﺩﻫﺎﯼ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﮔﺎﻥ ﻭ ﻣﺸﺘﺮﻳﺎﻥ
ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﭘﺎﺩﮐﺴﺖ‌ﺗﺎﻥ ﺑﻪ ﻋﻨﻮﺍﻥ ﺷﺎﻧﺴﯽ ﺟﻬﺖ ﭘﺎﺳﺦ ﺑﻪ ﻧﮕﺮﺍﻧﯽ‌ﻫﺎ ﻭ ﻭﺍﮐﻨﺶ‌ﻫﺎﯼ ﮐﺎﺭﺑﺮﺍﻥ ﺳﺎﻳﺖ ﻭ ﻳﺎ ﻣﺸﺘﺮﻳﺎﻥ ﺷﺮﮐﺖ‌ﺗﺎﻥ ﺑﻬﺮﻩ ﺑﺒﺮﻳﺪ.

۶- ﺍﺭﺗﺒﺎﻁ ﺻﻤﻴﻤﺎﻧﻪ ﺑﺎ ﺷﻨﻮﻧﺪﮔﺎﻥ
ﺩﺭ ﭘﺎﺩﮐﺴﺖ، ﻣﺤﺘﻮﺍ ﻣﺴﺘﻘﻴﻤﺎ ﺑﻪ ﺻﻮﺭﺕ ﮐﻼﻣﯽ ﺑﺎ ﺷﻨﻮﻧﺪﻩ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌ﮐﻨﺪ. ﺑﺮﺍﯼ ﺭﺳﺎﻧﺪﻥ ﻣﺤﺘﻮﺍ ﺑﻪ ﮐﺎﺭﺑﺮ، ﺍﻳﻦ ﺷﻴﻮﻩ ﺑﺴﻴﺎﺭ ﺻﻤﻴﻤﺎﻧﻪ‌ﺗﺮ ﺍﺯ ﻧﻮﺷﺘﻦ ﺍﻳﻤﻴﻞ ﻳﺎ ﺩﺭﺝ ﻳﮏ ﭘﺴﺖ ﺩﺭ ﻭﺑﻼﮒ ﻭ ﻭﺏ ﺳﺎﻳﺖ ﺍﺳﺖ.

۷- ﺭﺍﻩ ﺣﻞ ﻣﻨﺎﺳﺐ ﻭ ﺳﺎﺩﻩ ﺍﻧﺘﻘﺎﻝ ﻣﺤﺘﻮﺍ
ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﮐﺎﺭﺑﺮ ﻳﮏ ﺑﺎﺭ ﻣﺸﺘﺮﮎ ﻓﻴﺪ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﺷﻮﺩ، ﻧﺴﺨﻪ‌ﻫﺎﯼ ﺟﺪﻳﺪ ﺑﻪ ﺻﻮﺭﺕ ﺧﻮﺩﮐﺎﺭ ﺩﺭ ﮐﺎﻣﭙﻴﻮﺗﺮﺵ ﺩﺍﻧﻠﻮﺩ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻫﺮ ﺯﻣﺎﻥ ﮐﻪ ﺧﻮﺍﺳﺖ ﺑﻪ ﺁﻧﻬﺎ ﮔﻮﺵ ﮐﻨﺪ.

۸- ﮐﺎﻫﺶ ﻫﺰﻳﻨﻪ‌ﻫﺎ
ﺑﻪ ﺩﻟﻴﻞ ﺍﻳﻨﮑﻪ ﭘﺎﺩﮐﺴﺖ‌ﻫﺎ ﺑﻪ ﺻﻮﺭﺕ ﺩﻳﺠﻴﺘﺎﻝ ﺍﺭﺳﺎﻝ ﻣﯽ‌ﺷﻮﻧﺪ، ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻫﺰﻳﻨﻪ‌ﻫﺎ ﻫﻤﭽﻮﻥ ﭘﺴﺖ، ﭼﺎﭖ، ﺿﺒﻂ ﻳﺎ ﺳﯽ ﺩﯼ ﺣﺬﻑ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺎ ﺍﻳﻦ ﺷﻴﻮﻩ، ﻫﻤﭽﻨﻴﻦ ﻫﺰﻳﻨﻪ‌ﻫﺎﻳﯽ ﻫﻤﭽﻮﻥ ﻣﻼﻗﺎﺕ ﺣﻀﻮﺭﯼ ﻫﻢ ﺩﻳﮕﺮ ﺩﺭ ﻣﻴﺎﻥ ﻧﺨﻮﺍﻫﻨﺪ ﺑﻮﺩ. ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻦ ﺑﻪ ﺭﻭﺯﺭﺳﺎﻧﯽ ﻭ ﺁﺭﺷﻴﻮ ﺁﻧﻬﺎ ﻫﻢ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

۹- ﺗﻬﻴﻪ ﭘﺎﺩﮐﺴﺖ ﻳﮏ ﺷﻴﻮﻩ ﺍﺭﺗﺒﺎﻃﯽ ﺑﺎ ﺻﺮﻓﻪ‌ﺟﻮﻳﯽ ﻣﻨﺎﺳﺐ ﺩﺭ ﺯﻣﺎﻥ ﺍﺳﺖ
ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻫﻤﺰﻣﺎﻥ ﺑﺎ ﺍﻧﺠﺎﻡ ﺩﻳﮕﺮ ﮐﺎﺭ‌ﻫﺎﻳﺘﺎﻥ ﺩﺭ ﺍﺩﺍﺭﻩ، ﺧﺎﻧﻪ ﻳﺎ ﺣﺘﯽ ﺩﺭ ﻣﺴﺎﻓﺮﺕ ﺑﻪ ﭘﺎﺩﮐﺴﺖ ﮔﻮﺵ ﺩﻫﻴﺪ. ﺑﻪ ﻟﻄﻒ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﭘﺎﺩﮐﺴﺖ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺮﺧﯽ ﺍﺯ ﺟﻠﺴﺎﺕ ﻭ ﻣﻼﻗﺎﺕ‌ﻫﺎ ﺭﺍ ﺣﺬﻑ ﮐﺮﺩ ﻭ ﺻﺮﻓﻪ‌ﺟﻮﻳﯽ ﺧﻮﺑﯽ ﺩﺭ ﺯﻣﺎﻥ ﻭ ﺑﻬﺒﻮﺩ ﮐﺎﺭﺍﻳﯽ ﻭ ﺳﻮﺩﻣﻨﺪﯼ ﺩﺍﺷﺖ.

 

ﺁﻳﺎ ﻣﻦ ﺑﺎﻳﺪ ﭘﺎﺩﮐﺴﺖ ﺑﺴﺎﺯﻡ
ﻣﺮﺩﻡ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﻣﺤﺘﻮﺍ ﻭ ﺭﺳﺎﻧﻪ ﺭﺍ ﺑﻪ ﺷﮑﻞ‌ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﻣﺘﻨﻮﻋﯽ ﻣﺼﺮﻑ ﮐﻨﻨﺪ ﻭ‌‌ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﮔﻔﺘﻴﻢ، ﭘﺎﺩﮐﺴﺖ ﺷﻨﻮﻧﺪﮔﺎﻥ ﻭ ﮐﺎﺭﺑﺮﺍﻥ ﺟﺪﻳﺪﯼ ﺭﺍ ﺑﺮﺍﻳﺘﺎﻥ ﺑﻪ ﺍﺭﻣﻐﺎﻥ ﺧﻮﺍﻫﺪ ﺁﻭﺭﺩ. ﻫﻤﭽﻨﻴﻦ ﺗﻬﻴﻪ ﭘﺎﺩﮐﺴﺖ‌ﻫﺎ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺁﺳﺎﻥ ﺍﺳﺖ. ﻳﮏ ﻣﻴﮑﺮﻭﻓﻮﻥ ﻧﺴﺒﺘﺎ ﺧﻮﺏ ﻭ ﻳﮏ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻭﻳﺮﺍﻳﺶ ﺻﺪﺍ ﻫﻤﻪ ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﺮﺍﯼ ﺷﺮﻭﻉ ﭘﺎﺩﮐﺴﺖ‌ﺗﺎﻥ ﺑﻪ ﺁﻥ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﻳﺪ.
ﺍﻣﺎ ﺑﺎ ﻭﺟﻮﺩ ﺳﻬﻮﻟﺖ ﺧﻠﻖ ﻳﮏ ﭘﺎﺩﮐﺴﺖ، ﻗﺒﻞ ﺍﺯ ﻫﺮ ﮐﺎﺭﯼ ﺑﺎﻳﺪ ﺑﺮﺧﯽ ﺳﻮﺍﻝ‌ﻫﺎ ﺭﺍ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺑﭙﺮﺳﻴﺪ ﺗﺎ ﺑﺎ ﺍﻃﻤﻴﻨﺎﻥ ﮐﺎﻣﻞ ﮐﺎﺭﺗﺎﻥ ﺭﺍ ﺁﻏﺎﺯ ﮐﻨﻴﺪ.

۱- ﺻﺪﺍﻳﺘﺎﻥ ﺑﻪ ﭼﻪ ﮐﺴﺎﻧﯽ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﺮﺳﺪ؟
ﺍﻳﻦ ﺳﻮﺍﻟﯽ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﻫﻤﻴﺸﻪ ﭼﺸﻢ‌ﺍﻧﺪﺍﺯ ﺷﻤﺎ ﺑﺎﺷﺪ. ﭼﻪ ﮔﺮﻭﻩ ﻫﺪﻓﺘﯽ ﺷﻨﻮﻧﺪﻩ ﺷﻤﺎ ﺍﺳﺖ؟ ﺍﮔﺮ ﭼﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﻓﮑﺮ ﮐﻨﻴﺪ ﺗﻨﻬﺎ ﺟﻮﺍﻥ‌ﻫﺎﯼ ﻋﺸﻖ ﺗﮑﻨﻮﻟﻮﮊﯼ ﭘﺎﺩﮐﺴﺖ‌ﻫﺎ ﺭﺍ ﺩﺍﻧﻠﻮﺩ ﻣﯽ‌ﮐﻨﻨﺪ، ﺍﻣﺎ ﻳﮏ ﺗﺤﻘﻴﻖ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﭘﺎﺩﮐﺴﺖ‌ﻫﺎ ﺩﺭ ﻣﻴﺎﻥ ﺩﻳﮕﺮ ﺍﻗﺸﺎﺭ ﻣﺮﺩﻡ ﻫﻢ ﻣﺤﺒﻮﺏ ﻫﺴﺘﻨﺪ.
ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﮔﻔﺖ ﺩﺍﻧﻠﻮﺩ ﮐﻨﻨﺪﻩ‌ﻫﺎﯼ ﭘﺎﺩﮐﺴﺖ ﺍﻓﺮﺍﺩ ﺟﻮﺍﻥ، ﻣﺮﺩ ﻭ ﺗﺤﺼﻴﻞ ﮐﺮﺩﻩ ﻫﺴﺘﻨﺪ. ﺍﻣﺎ ﺑﻪ ﺷﮑﻞ ﻗﺎﺑﻞ ﺗﻮﺟﻬﯽ ﺍﻓﺮﺍﺩ ﺑﻴﻦ ۳۵ ﺗﺎ ۵۴ ﺳﺎﻟﻪ ﻧﻴﺰ ﺗﻘﺮﻳﺒﺎ ﻧﻴﻤﯽ ﺍﺯ ﺷﻨﻮﻧﺪﮔﺎﻥ ﭘﺎﺩﮐﺴﺖ ﻣﻮﺭﺩ ﺗﺤﻘﻴﻖ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﻨﺪ.
ﻗﺒﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﭘﺎﺩﮐﺴﺖ‌ﺗﺎﻥ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﺪ، ﺍﺑﺘﺪﺍ ﺩﺭﺑﺎﺭﻩ ﻗﺸﺮ ﻫﺪﻑ‌ﺗﺎﻥ ﺑﻴﻨﺪﻳﺸﻴﺪ. ﺍﮔﺮ ﺟﺎﻣﻌﻪ ﻫﺪﻑ‌ﺗﺎﻥ ﻋﻼﻗﻪ‌ﺍﯼ ﺑﻪ ﭘﺎﺩﮐﺴﺖ ﻧﺪﺍﺭﻧﺪ، ﭘﺲ ﺷﺎﻳﺪ ﻭﻗﺖ ﮔﺬﺍﺷﺘﻦ ﺑﺮﺍﯼ ﺗﻬﻴﻪ ﺁﻥ ﺍﺭﺯﺵ ﭼﻨﺪﺍﻧﯽ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ.

۲- ﭼﺮﺍ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺻﺪﺍﻳﺘﺎﻥ ﺑﻪ ﺁﻧﻬﺎ ﺑﺮﺳﺪ
ﺍﮔﺮ ﺗﻨﻬﺎ ﺑﻪ ﺩﻧﺒﺎﻝ ﺍﻳﻦ ﻫﺴﺘﻴﺪ ﮐﻪ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﭘﺎﺩﮐﺴﺖ ﺑﻪ ﺗﺒﻠﻴﻎ ﺍﺟﻨﺎﺱ‌ﺗﺎﻥ ﺑﭙﺮﺩﺍﺯﻳﺪ ﻭ ﻫﻤﺎﻧﻨﺪ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﻣﺎﻫﻮﺍﺭﻩ‌ﺍﯼ ﻓﺮﻭﺵ ﮐﺎﻻ، ﻳﮏ ﻓﺮﻭﺷﮕﺎﻩ ﺻﻮﺗﯽ ﺑﺎﺷﻴﺪ، ﭘﺲ ﺍﺣﺘﻤﺎﻻ ﻣﻮﻓﻘﻴﺖ ﭼﻨﺪﺍﻧﯽ ﺩﺭ ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﻣﺸﺘﺮﮎ ﺛﺎﺑﺖ ﻧﺨﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﺍﻣﺎ ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﻪ ﻣﺮﺩﻡ ﻣﻬﺎﺭﺕ‌ﻫﺎﯼ ﺟﺪﻳﺪﯼ ﺑﻴﺎﻣﻮﺯﻳﺪ ﻭ ﻳﺎ ﺩﺍﻧﺶ ﻭ ﺁﮔﺎﻫﯽ ﺟﺪﻳﺪﯼ ﺩﺭﺑﺎﺭﻩ ﺻﻨﻌﺖ ﻭ ﺷﻐﻞ‌ﺗﺎﻥ ﺭﺍ ﺑﺮﺍﻳﺸﺎﻥ ﻓﺮﺍﻫﻢ ﺁﻭﺭﻳﺪ، ﭘﺲ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﺍﻳﻦ ﭘﺎﺩﮐﺴﺖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﺎﺛﻴﺮ ﺧﻮﺑﯽ ﺭﻭﯼ ﮐﺎﺭ ﻭ ﮐﺴﺐ‌ﺗﺎﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.
ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﻟﺤﻈﻪ‌ﺍﯼ ﺑﻴﻨﺪﻳﺸﻴﺪ... ﺍﮔﺮ ﻣﺤﺘﻮﺍﯼ ﺍﺭﺯﺷﻤﻨﺪﯼ ﺭﺍ ﺑﺮﺍﯼ ﺷﻨﻮﻧﺪﻩ ﺗﺎﻣﻴﻦ ﻧﻤﯽ‌ﮐﻨﻴﺪ (ﺣﺎﻝ ﺩﺭ ﻫﺮ ﺯﻣﻴﻨﻪ‌ﺍﯼ)، ﺍﺻﻼ ﭼﺮﺍ ﭘﺎﺩﮐﺴﺖ ﺿﺒﻂ ﻣﯽ‌ﮐﻨﻴﺪ؟

۳- ﺁﻳﺎ ﺑﻪ ﻃﻮﺭ ﻣﻨﻈﻢ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭘﺎﺩﮐﺴﺖ ﺗﻬﻴﻪ ﮐﻨﻴﺪ؟
ﺍﮔﺮ ﻧﺘﻮﺍﻧﻴﺪ ﮐﺎﺭﺗﺎﻥ ﺭﺍ ﺑﺮﺍﺳﺎﺱ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺛﺎﺑﺖ ﻭ ﻣﻨﻈﻢ ﺟﻠﻮ ﺑﺒﺮﻳﺪ، ﻋﺮﺿﻪ ﭘﺎﺩﮐﺴﺖ، ﻋﻤﻠﯽ ﺑﻴﻬﻮﺩﻩ ﻭ ﺗﻠﻒ ﮐﺮﺩﻥ ﻭﻗﺖ ﺍﺳﺖ. ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﻣﺸﺘﺮﮐﺎﻥ ﻭ ﺷﻨﻮﻧﺪﮔﺎﻥ‌ﺗﺎﻥ ﺭﺍ ﺭﺍﺿﯽ ﻭ ﺧﻮﺷﻨﻮﺩ ﻧﮕﻪ ﺩﺍﺭﻳﺪ، ﺑﻪ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﺯﻣﺎﻥ‌ﺑﻨﺪﯼ ﻣﻨﻈﻢ ﺑﭽﺴﺒﻴﺪ ﻭ ﻫﻴﭻ‌ﮔﺎﻩ ﺁﻥ ﺭﺍ‌‌ ﺭﻫﺎ ﻧﮑﻨﻴﺪ.

 

ﺁﻳﺎ ﻫﻤﭽﻨﺎﻥ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﭼﻴﺰﯼ ﺩﺭ ﭼﻨﺘﻪ ﺩﺍﺭﻳﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺁﻥ ﺭﺍ ﺩﺭ ﻗﺎﻟﺐ ﭘﺎﺩﮐﺴﺖ ﺍﺭﺍﺋﻪ ﮐﻨﻴﺪ ﻭ ﺟﻤﻌﻴﺖ ﻫﺪﻑ ﺷﻤﺎ ﻫﻢ ﺩﺭ ﻫﻤﻴﻦ ﻗﺎﻟﺐ ﺧﻮﺍﻫﺎﻥ ﺁﻥ ﻫﺴﺘﻨﺪ؟ ﺗﻮﺍﻧﺎﻳﯽ ﻋﺮﺿﻪ ﻣﻨﻈﻢ ﺁﻥ ﺭﺍ ﻫﻢ ﺩﺍﺭﻳﺪ؟ ﭘﺲ ﺩﺭ ﺍﺩﺍﻣﻪ ﺩﻭﺭﻩ ﺑﺎ ﻣﺎ ﺑﺎﺷﻴﺪ ﺗﺎ ﺑﺒﻴﻨﻴﻢ ﭼﮕﻮﻧﻪ ﺑﺎﻳﺪ ﻳﮏ ﭘﺎﺩﮐﺴﺖ ﻫﻮﺍ ﮐﻨﻴﻢ.
