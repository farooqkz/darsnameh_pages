ﺩﺭﺱ ۲- ﻃﺮﺍﺣﯽ ﺍﺷﺨﺎﺹ ﺳﺨﻨﺮﺍﻧﯽ

ﺑﻪ ﺩﻭﻣﻴﻦ ﺩﺭﺱ ﻣﻴﻨﯽ ﺩﻭﺭﻩ ﺳﺨﻨﺮﺍﻧﯽ ﻭ ﻓﻦ ﺑﻴﺎﻥ ﺧﻮﺵ ﺁﻣﺪﻳﺪ!

ﺩﻭﻣﻴﻦ ﻃﺮﺍﺣﯽ ﮐﻪ ﺑﺎﻳﺪ ﺩﺭ ﻳﮏ ﺳﺨﻨﺮﺍﻧﯽ ﺍﻧﺠﺎﻡ ﺑﺪﻫﻴﻢ، ﻃﺮﺍﺣﯽ ﺍﺷﺨﺎﺹ ﺳﺨﻨﺮﺍﻧﯽ ﺍﺳﺖ. ﻭﺍﻗﻌﻴﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻫﻤﻮﺍﺭﻩ ﻳﮏ ﺳﺨﻨﺮﺍﻧﯽ ﺭﺍ ﺑﺮﺍﯼ ﻋﺪﻩ ﺧﺎﺻﯽ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ ﻭ ﺍﮔﺮ ﺳﺨﻨﺮﺍﻧﯽ ﺷﻤﺎ ﻣﻄﺎﺑﻖ ﺑﺎ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﯼ ﺁﻥ ﺍﻓﺮﺍﺩ ﻧﺒﺎﺷﺪ، ﺳﺨﻨﺮﺍﻧﯽ ﺷﻤﺎ ﺷﮑﺴﺖ ﺧﻮﺭﺩﻩ ﺗﻠﻘﯽ ﻣﯽ‌ﺷﻮﺩ.

ﺍﺯ ﻫﻤﻴﻦ ﺭﻭ ﺑﺎﻳﺪ ﺷﻨﺎﺧﺖ ﺑﺴﻴﺎﺭ ﺩﻗﻴﻘﯽ ﻧﺴﺒﺖ ﺑﻪ ﻣﺨﺎﻃﺒﺎﻥ ﺳﺨﻨﺮﺍﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﻭ ﺑﺪﺍﻧﻴﻢ ﮐﻪ ﭼﻪ ﭼﻴﺰﯼ ﺭﺍ ﺑﺮﺍﯼ ﭼﻪ ﮐﺴﯽ ﺁﻣﺎﺩﻩ ﻣﯽ‌ﮐﻨﻴﻢ؟

ﺍﺟﺎﺯﻩ ﺑﺪﻫﻴﺪ ﺍﻫﻤﻴﺖ ﻣﺨﺎﻃﺐ ﺷﻨﺎﺳﯽ ﺭﺍ ﺑﺎ ﻳﮏ ﻣﺜﺎﻝ ﺗﻮﺿﻴﺢ ﺑﺪﻫﻢ، ﻓﺮﺽ ﮐﻨﻴﺪ ﻗﺮﺍﺭ ﺍﺳﺖ ﻳﮏ ﻣﻮﺿﻮﻉ ﺗﺨﺼﺼﯽ ﻣﺜﻞ «ﺗﻌﺎﺩﻝ ﻫﻴﺪﺭﻭﺍﺳﺘﺎﺗﻴﮏ ﺩﺭ ﻣﺮﮐﺰ ﺳﺘﺎﺭﮔﺎﻥ» ﺭﺍ ﺩﺭ ﻳﮏ ﺳﺨﻨﺮﺍﻧﯽ ﺍﺭﺍﺋﻪ ﮐﻨﻴﻢ. ﺁﻳﺎ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﻨﺎﺳﺐ ﺍﺳﺖ؟ 

ﺳﻮﺍﻝ ﻣﻬﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﺑﺮﺍﯼ ﭼﻪ ﮐﺴﺎﻧﯽ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺗﻮﺿﻴﺢ ﺑﺪﻫﻴﺪ؟ ﺍﮔﺮ ﻣﺨﺎﻃﺒﺎﻥ ﺷﻤﺎ ﮐﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﺍﺭﺍﯼ ﻣﺪﺭﮎ ﺩﮐﺘﺮﯼ ﺍﺧﺘﺮﻓﻴﺰﻳﮏ ﻫﺴﺘﻨﺪ، ﺑﺎﻳﺪ ﺑﮕﻮﻳﻢ ﮐﻪ ﺧﻴﺮ! ﺯﻳﺮﺍ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ ﻭ ﭘﻴﺶ ﭘﺎ ﺍﻓﺘﺎﺩﻩ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻭ ﺍﮔﺮ ﻣﺨﺎﻃﺒﺎﻥ ﺷﻤﺎ ﮐﻮﺩﮐﺎﻥ ﻳﮏ ﻣﻬﺪ ﮐﻮﺩﮎ ﺑﺎﺷﻨﺪ ﻧﻴﺰ ﺑﻪ ﻫﻤﻴﻦ ﺻﻮﺭﺕ، ﻣﺤﺘﻮﺍ ﺑﺴﻴﺎﺭ ﭘﻴﭽﻴﺪﻩ ﻭ ﻏﻴﺮ ﻗﺎﺑﻞ ﺩﺭﮎ ﺍﺳﺖ! ﺍﻣﺎ ﻫﻤﻴﻦ ﺻﺤﺒﺖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﺩﺍﻧﺸﺠﻮﻳﺎﻥ ﮐﺎﺭﺷﻨﺎﺳﯽ ﻓﻴﺰﻳﮏ ﺑﺴﻴﺎﺭ ﺟﺎﻟﺐ ﻭ ﻣﻨﺎﺳﺐ ﺑﺎﺷﺪ.

ﭘﺲ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﻫﻴﭽﮕﺎﻩ ﻳﮏ ﺳﺨﻨﺮﺍﻧﯽ ﺧﻮﺏ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ، ﻳﮏ ﺳﺨﻨﺮﺍﻧﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺧﻮﺏ ﻳﺎ ﺑﺪ ﺑﺎﺷﺪ، ﺑﺴﺘﻪ ﺑﻪ ﺍﻳﻦ ﮐﻪ ﺑﺮﺍﯼ ﭼﻪ ﮐﺴﺎﻧﯽ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﺷﻮﺩ.

ﻫﻤﻮﺍﺭﻩ ﺑﺮﺍﯼ ﺷﻨﺎﺧﺖ ﻣﺨﺎﻃﺒﺎﻥ ﺩﺭ ﺳﺨﻨﺮﺍﻧﯽ ﺑﺎﻳﺪ ﺑﻪ ﻧﮑﺎﺕ ﺯﻳﺮ ﺗﻮﺟﻪ ﮐﺮﺩ:

  • ﺳﻦ ﻣﺨﺎﻃﺒﺎﻥ ﭼﻘﺪﺭ ﺍﺳﺖ؟
  • ﺟﻨﺴﻴﺖ ﻣﺨﺎﻃﺒﺎﻥ ﭼﻴﺴﺖ؟
  • ﺳﻄﺢ ﻓﺮﻫﻨﮕﯽ ﺁﻧﻬﺎ ﺩﺭ ﭼﻪ ﺣﺪﯼ ﺍﺳﺖ؟
  • ﺩﺭ ﭼﻪ ﺳﻄﺢ ﺩﺭﺁﻣﺪﯼ ﻗﺮﺍﺭ ﺩﺍﺭﻧﺪ؟
  • ﭼﻘﺪﺭ ﺩﺍﻧﺶ ﺩﺍﺭﻧﺪ؟
  • ﺻﺤﺒﺖ ﺷﻤﺎ ﻧﻴﺎﺯ ﺑﻪ ﭼﻪ ﭘﻴﺶ‌ﻧﻴﺎﺯﻫﺎﻳﯽ ﺩﺍﺭﺩ؟
  • ﻋﻼﻳﻖ ﺁﻧﻬﺎ ﭼﻴﺴﺖ؟
  • ﺑﻪ ﭼﻪ ﺩﻟﻴﻞ ﺩﺭ ﺳﺨﻨﺮﺍﻧﯽ ﺷﻤﺎ ﺣﻀﻮﺭ ﺩﺍﺭﻧﺪ؟

ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺍﮔﺮ ﺷﻤﺎ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﺮﺍﯼ ﺩﺍﻧﺶ ﺁﻣﻮﺯﺍﻥ ﺩﻭﺭﻩ ﺩﺑﻴﺮﺳﺘﺎﻥ ﺳﺨﻨﺮﺍﻧﯽ ﮐﻨﺒﺪ، ﺍﻧﺘﺨﺎﺏ ﺭﺷﺘﻪ ﻳﮑﯽ ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﻣﻮﺿﻮﻋﺎﺕ ﻭ ﺩﻏﺪﻏﻪ‌ﻫﺎﯼ ﺫﻫﻨﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻭ ﺑﺮﺍﯼ ﺩﺍﻧﺸﺠﻮﻳﺎﻥ، ﺍﻧﺘﺨﺎﺏ ﺷﻐﻞ ﻭ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩ ﮐﻬﻨﺴﺎﻝ، ﻣﺒﺎﺣﺚ ﻣﺮﺑﻮﻁ ﺑﻪ ﺑﺎﺯﻧﺸﺴﺘﮕﯽ!

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺣﺘﻤﺎ ﻗﺒﻞ ﺍﺯ ﺍﺟﺮﺍﯼ ﻫﺮ ﺳﺨﻨﺮﺍﻧﯽ ﺑﻪ ﺳﻮﺍﻻﺕ ﺯﻳﺮ ﭘﺎﺳﺦ ﺩﻫﻴﺪ:

  • ﻣﺨﺎﻃﺒﺎﻥ ﻣﻦ ﺩﻗﻴﻘﺎ ﭼﻪ ﮐﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ؟
  • ﻣﻴﺎﻧﮕﻴﻦ ﺳﻨﯽ ﺁﻧﻬﺎ ﭼﻘﺪﺭ ﺧﻮﺍﻫﺪ ﺑﻮﺩ؟
  • ﺳﻄﺢ ﺳﻮﺍﺩ ﺁﻧﻬﺎ ﭼﻘﺪﺭ ﺍﺳﺖ؟
  • ﺁﻳﺎ ﺑﺮﺍﯼ ﺷﻨﻴﺪﻥ ﺻﺤﺒﺖ‌ﻫﺎﯼ ﻣﻦ ﻧﻴﺎﺯ ﺑﻪ ﭘﻴﺶ‌ﻧﻴﺎﺯﯼ ﺩﺍﺭﻧﺪ؟
  • ﺍﺻﻠﯽ‌ﺗﺮﻳﻦ ﻋﻼﻗﻪ ﺁﻧﻬﺎ ﭼﻴﺴﺖ ﻭ ﭼﻄﻮﺭ ﻣﯽ‌ﺗﻮﺍﻧﻢ ﺑﻪ ﻋﻼﻗﻪ ﺁﻧﻬﺎ ﺩﺭ ﺻﺤﺒﺘﻢ ﺍﺷﺎﺭﻩ ﮐﻨﻢ؟

ﺣﺎﻝ ﺑﺎ ﺍﻳﻦ ﺗﻮﺿﻴﺤﺎﺕ ﺩﻭﺑﺎﺭﻩ ﺑﺮﮔﺮﺩﻳﻢ ﺑﻪ ﻓﺎﻳﻞ ﺻﻮﺗﯽ ﮐﻪ ﺟﻠﺴﻪ ﻗﺒﻞ ﺿﺒﻂ ﮐﺮﺩﻩ‌ﺍﻳﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﻣﻌﺮﻓﯽ ﺧﻮﺩ ﻭﻳﮋﮔﯽ‌ﺍﯼ ﺭﺍ ﻣﻄﺮﺡ ﮐﺮﺩﻩ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﺴﻴﺎﺭ ﺟﺬﺍﺏ ﺑﺎﺷﺪ ﺍﻣﺎ ﺑﺮﺍﯼ ﻣﺨﺎﻃﺐ ﺷﻤﺎ ﺍﺻﻼ ﺟﺬﺍﺑﻴﺘﯽ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺍﮔﺮ ﺍﻳﻦ ﻣﻌﺮﻓﯽ ﺭﺍ ﺑﺮﺍﯼ ﻣﺪﻳﺮ ﻳﮏ ﺷﺮﮐﺖ ﺗﻮﻟﻴﺪﯼ ﺿﺒﻂ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺗﺎ ﻧﻈﺮ ﺍﻭ ﺭﺍ ﺟﻠﺐ ﮐﻨﻴﺪ ﻭ ﺍﻭ ﺷﻤﺎ ﺭﺍ ﺍﺳﺘﺨﺪﺍﻡ ﮐﻨﺪ، ﺑﺪﺗﺮﻳﻦ ﮐﺎﺭ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﻋﻼﻳﻖ ﺧﻮﺩ ﺑﺎ ﺍﻭ ﺻﺤﺒﺖ ﮐﻨﻴﺪ. ﺍﻳﻦ ﮐﻪ ﺷﻤﺎ ﭼﻘﺪﺭ ﺑﻪ ﺷﻨﺎ ﻋﻼﻗﻪ ﺩﺍﺭﻳﺪ، ﺍﺻﻼ ﺑﺮﺍﯼ ﻳﮏ ﻣﺪﻳﺮ ﻳﮏ ﺷﺮﮐﺖ ﺗﻮﻟﻴﺪﯼ ﺍﻫﻤﻴﺖ ﻧﺪﺍﺭﺩ ﻭ ﻣﻬﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﭼﻘﺪﺭ ﺗﺴﻠﻂ ﺑﻪ ﮐﺎﺭ ﺑﺎ ﺩﺳﺘﮕﺎﻩ‌ﻫﺎﯼ ﺁﻥ ﺷﺮﮐﺖ ﺩﺍﺭﻳﺪ.

ﻫﻤﭽﻨﻴﻦ ﺩﺭ ﻳﮏ ﺟﻤﻊ ﺩﻭﺳﺘﺎﻧﻪ، ﺯﻣﺎﻧﯽ ﮐﻪ ﺍﺯ ﺷﻤﺎ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺧﻮﺩ ﺭﺍ ﻣﻌﺮﻓﯽ ﮐﻨﻴﺪ، ﺍﺻﻼ ﻣﻬﻢ ﻧﻴﺴﺖ ﮐﻪ ﺷﻤﺎ ﭼﻪ ﺗﺴﻠﻄﯽ ﺩﺭ ﮐﺎﺭ ﺑﺎ ﺩﺳﺘﮕﺎﻩ ﺟﻮﺵ ﺻﻨﻌﺘﯽ ﺩﺍﺭﻳﺪ! ﻣﻬﻢ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﭼﻪ ﺗﻔﺮﻳﺤﺎﺕ ﻭ ﻋﻼﻗﻪ‌ﻣﻨﺪﯼ‌ﻫﺎﻳﯽ ﺩﺍﺭﻳﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﺑﺎﺭ ﺩﻳﮕﺮ ﺑﺎ ﻓﺮﺽ ﺍﻳﻦ ﮐﻪ ﭼﻪ ﮐﺴﯽ ﻣﺨﺎﻃﺐ ﻓﺎﻳﻞ ﺻﻮﺗﯽ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﺑﻮﺩ، ﻓﺎﻳﻞ ﺻﻮﺗﯽ ﺧﻮﺩ ﺭﺍ ﻣﺘﻨﺎﺳﺐ ﺑﺎ ﻣﺨﺎﻃﺒﺎﻥ ﺧﻮﺩ ﻃﺮﺍﺣﯽ ﻭ ﺍﺟﺮﺍ ﮐﻨﻴﺪ.
