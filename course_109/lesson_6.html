ﺩﺭﺱ ﺷﺸﻢ- ﺍﺟﺘﻨﺎﺏ ﺍﺯ ﺑﺮﺧﯽ ﺭﻭﺵ ﻫﺎ- ﺍﺩﺍﻣﻪ

ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﺑﻪ ﺑﻪ ﺗﻌﺪﺍﺩﯼ ﺍﺯ ﺭﻭﺵ­‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺁﻣﻮﺯﺵ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺑﺎﻳﺪ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺍﺟﺘﻨﺎﺏ ﮐﺮﺩ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ. ﺍﻳﻦ ﺍﺻﻮﻝ ﺑﻪ ﺍﺧﺘﺼﺎﺭ ﻋﺒﺎﺭﺕ ﺑﻮﺩﻧﺪ ﺍﺯ ﻋﺪﻡ ﺗﻬﺪﻳﺪ ﮐﻮﺩﮎ، ﻋﺪﻡ ﺗﺤﻘﻴﺮ ﻭ ﺗﻄﻤﻴﻊ ﺍﻭ، ﺗﺤﻤﻴﻞ ﻧﮑﺮﺩﻥ ﻧﻴﺖ ﺧﻮﺏ ﺑﻪ ﮐﻮﺩﮎ، ﺣﻤﺎﻳﺖ ﻧﮑﺮﺩﻥ ﺯﻳﺎﺩ ﺍﺯ ﺣﺪ ﺍﺯ ﺍﻭ ﻭ ﺯﻳﺎﺩ ﺻﺤﺒﺖ ﻧﮑﺮﺩﻥ ﻭ ﻋﺪﻡ ﺩﺭﺧﻮﺍﺳﺖ ﺍﻃﺎﻋﺖ ﮐﻮﺭﮐﻮﺭﺍﻧﻪ ﻭ ﺑﻼﻓﺎﺻﻠﻪ ﺍﺯ ﮐﻮﺩﮎ . ﺍﻳﻨﮏ ﺭﻭﺵ­‌ﻫﺎﯼ ﺩﻳﮕﺮﯼ ﮐﻪ ﺩﺭ ﺁﻣﻮﺯﺵ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺑﺎﻳﺪ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺍﺟﺘﻨﺎﺏ ﮐﺮﺩ ﺑﺮﺭﺳﯽ ﻣﯽ‌­ﮐﻨﻴﻢ.
                  
۸- ﺯﻳﺎﺩﯼ ﮔﺬﺷﺖ ﻧﮑﻨﻴﺪ ﻭ ﺍﻭ ﺭﺍ ﻧﺎﺯﭘﺮﻭﺭﺩﻩ ﺑﺎﺭ ﻧﻴﺎﻭﺭﻳﺪ.
ﻗﺒﻮﻝ ﺗﻤﺎﻡ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﯼ ﻛﻮﺩﮎ ﻫﻤﺎﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﻛﻪ ﺍﻛﺜﺮ ﻭﺍﻟﺪﻳﻦ «ﻟﻮﺱ ﻛﺮﺩﻥ» ﻣﯽ‌ﻧﺎﻣﻨﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺯﻣﺎﻧﯽ ﻛﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﺧﻴﻠﯽ ﺯﻭﺩ ﺩﺭ ﻣﻘﺎﺑﻞ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﯼ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺗﺴﻠﻴﻢ ﻣﯽ‌ﺷﻮﻧﺪ، ﻣﺜﻞ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺍﺯ ﻧﻪ ﮔﻔﺘﻦ ﺑﻪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺗﺮﺱ ﺩﺍﺭﻧﺪ ﻭ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﺁﻥ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﺑﺮﺍﯼ ﺍﻭ ﺗﻌﻴﻴﻦ ﻛﺮﺩﻩ‌ﺍﻧﺪ ﺭﻋﺎﻳﺖ ﻛﻨﻨﺪ. ﭼﻨﻴﻦ ﺭﻓﺘﺎﺭﯼ ﺑﻪ ﮐﻮﺩﮎ ﺍﻳﻦ ﮔﻮﻧﻪ ﺍﻟﻘﺎ ﻣﯽ‌­ﮐﻨﺪ ﻛﻪ ﺗﻤﺎﻡ ﻗﻮﺍﻧﻴﻦ ﻗﺎﺑﻞ‌ ﺍﻧﻌﻄﺎﻑ ﻫﺴﺘﻨﺪ ﻭ ﺑﺎ ﻛﻤﯽ ﭘﺎﻓﺸﺎﺭﯼ ﻣﯽ‌ﺗﻮﺍﻥ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺯﻳﺮ ﭘﺎ ﮔﺬﺍﺷﺖ. ﺍﻳﻦ ﻣﻤﻜﻦ ﺍﺳﺖ ﺩﺭ ﻭﺍﺣﺪ ﺧﺎﻧﻮﺍﺩﻩ ﺍﻣﻜﺎﻥ‌ﭘﺬﻳﺮ ﺑﺎﺷﺪ، ﺍﻣﺎ ﺩﺭ ﺗﻤﺎﺱ ﺑﺎ ﺩﻧﻴﺎﯼ ﺧﺎﺭﺝ ﺍﺯ ﺧﺎﻧﻮﺍﺩﻩ ﻛﻮﺩﮎ ﺭﺍ ﺩﭼﺎﺭ ﻣﺴﺎﺋﻞ ﺩﺭﺩﻧﺎﻛﯽ ﺧﻮﺍﻫﺪ ﻛﺮﺩ. ﻗﺒﻮﻝ ﻛﺮﺩﻥ ﺯﻳﺎﺩﻩ ﺍﺯ ﺣﺪ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﯼ ﻛﻮﺩﮎ ﻣﻌﺎﺩﻝ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩﻥ ﺷﺎﻧﺲ ﺍﻭ ﺑﺮﺍﯼ ﺗﺒﺪﻳﻞ ﺷﺪﻥ ﺑﻪ ﺍﻧﺴﺎﻥ ﻣﺴﺘﻘﻞ ﺍﺳﺖ ﻛﻪ ﻗﺎﺩﺭ ﺑﻪ ﺍﺩﺍﺭﻩ ﺧﻮﺩ ﺑﺎﺷﺪ.
 
۹- ﺩﺭ ﻣﻘﺎﺑﻞ ﻗﻮﺍﻧﻴﻦ ﻭ ﻣﻘﺮﺭﺍﺕ ﻫﻤﻴﺸﻪ ﻳﻜﺴﺎﻥ ﺭﻓﺘﺎﺭ ﻛﻨﻴﺪ.
ﺍﮔﺮ ﻣﺎﺩﺭﯼ ﺷﻨﺒﻪ ﺑﺎ ﺍﻋﺼﺎﺏ ﺭﺍﺣﺖ ﺑﮕﺬﺍﺭﺩ ﻓﺮﺯﻧﺪﺵ ﻫﺮ ﻛﺎﺭﯼ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻜﻨﺪ ﻭﻟﯽ ﻳﻜﺸﻨﺒﻪ ﻫﻤﺎﻥ ﻛﺎﺭﻫﺎﯼ ﻏﻠﻂ ﺭﺍ ﺷﺪﻳﺪﺍ ﻣﻤﻨﻮﻉ ﻛﻨﺪ، ﺍﺛﺮﯼ ﭼﻨﻴﻦ ﺭﻓﺘﺎﺭﯼ ﺑﺮ ﺭﻭﯼ ﮐﻮﺩﮎ ﺗﻘﺮﻳﺒﺎ ﻣﺜﻞ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﭼﺮﺍﻍ ﻗﺮﻣﺰ ﺑﺮﺍﯼ ﺭﺍﻧﻨﺪﮔﺎﻥ ﻭﺳﺎﻳﻞ ﻧﻘﻠﻴﻪ ﻳﮏ ﺭﻭﺯ ﺑﻪ ﻣﻌﻨﺎﯼ ﻋﺒﻮﺭ ﻭ ﺭﻭﺯﯼ ﺩﻳﮕﺮ ﺑﻪ ﻣﻌﻨﺎﯼ ﺗﻮﻗﻒ ﻛﺮﺩﻥ ﺑﺎﺷﺪ. ﻛﻮﺩﮎ ﻣﺤﺘﺎﺝ ﺑﻪ ﻧﻮﻋﯽ ﻳﻜﻨﻮﺍﺧﺘﯽ ﻭ ﺛﺒﺎﺕ ﺩﺭ ﻣﻮﺭﺩ ﺍﻧﺘﻈﺎﺭﺍﺗﯽ ﺍﺳﺖ ﻛﻪ ﺍﺯ ﺍﻭ ﺩﺍﺭﻧﺪ.
 
۱۰- ﻣﻘﺮﺭﺍﺗﯽ ﻛﻪ ﺑﺎ ﺳﻦ ﻛﻮﺩﮎ ﺗﻄﺎﺑﻖ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭﺿﻊ ﻧﮑﻨﻴﺪ.
ﺍﮔﺮ ﺍﺯ ﻛﻮﺩﻛﯽ ﺩﻭ ﺳﺎﻟﻪ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻛﻪ ﻣﺎﻧﻨﺪ ﻃﻔﻠﯽ ﭘﻨﺞ ﺳﺎﻟﻪ ﺍﺯ ﺷﻤﺎ ﺍﻃﺎﻋﺖ ﻛﻨﺪ، ﺗﻨﻬﺎ ﻧﺘﻴﺠﻪ‌ﺍﺵ ﻧﺎﺭﺍﺣﺖ ﻛﺮﺩﻥ ﺍﻭ ﻭ ﺑﺮﺍﻧﮕﻴﺨﺘﻦ ﻧﻔﺮﺗﺶ ﻋﻠﻴﻪ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺍﻳﻦ ﺍﻧﺘﻈﺎﺭ ﺷﻤﺎ ﺑﻪ ﻣﻌﻨﯽ ﺧﻮﺍﺳﺘﻦ ﺍﻧﺠﺎﻡ ﻛﺎﺭﯼ ﺍﺳﺖ ﻛﻪ ﺍﻭ ﻫﻨﻮﺯ ﭘﺨﺘﮕﯽ ﻻﺯﻡ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻣﺶ ﺭﺍ ﻧﺪﺍﺭﺩ ﻭ ﺍﺛﺮﯼ ﺑﺴﻴﺎﺭ ﺑﺪﯼ ﺑﺮ ﺭﺷﺪ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﺍﻭ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ.
 
۱۱- ﺍﺯ ﻣﻮﻋﻈﻪ ﻛﺮﺩﻥ ﻭ ﺑﻪ ﻛﺎﺭ ﺑﺮﺩﻥ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﻛﻪ ﻣﻮﺟﺐ ﻋﺬﺍﺏ ﻭﺟﺪﺍﻥ ﻣﯽ‌ﺷﻮﻧﺪ ﭘﺮﻫﻴﺰ ﮐﻨﻴﺪ.
ﺑﻪ ﻛﺎﺭ ﮔﺮﻓﺘﻦ ﭼﻨﻴﻦ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺑﺎﻋﺚ ﺷﻜﻞ‌ﮔﻴﺮﯼ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﺍﺯ ﻧﻮﻉ ﺑﺪﯼ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺭﻭﺵ‌ﻫﺎﯼ ﺗﺮﺑﻴﺘﯽ‌ﺍﯼ ﻛﻪ ﺍﺣﺴﺎﺱ ﮔﻨﺎﻩ ﺭﺍ ﺍﻟﻘﺎ ﻛﻨﻨﺪ، ﺁﻥ‌ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﻛﻪ ﭘﻴﺎﻡ‌ﻫﺎﻳﯽ ﺍﻳﻦ ﭼﻨﻴﻦ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﻨﺘﻘﻞ ﻣﯽ‌ﻛﻨﻨﺪ: «ﻛﺎﺭﯼ ﻛﻪ ﺗﻮ ﻛﺮﺩﯼ ﺧﻴﻠﯽ ﺑﺪ ﺑﻮﺩ. ﺗﻮ ﺑﭽﻪ ﺑﺪﯼ ﻫﺴﺘﯽ، ﭼﻄﻮﺭ ﻣﯽ‌ﺗﻮﻧﯽ ﭼﻨﻴﻦ ﻛﺎﺭﯼ ﺭﻭ ﺩﺭ ﺣﻖ ﻣﺎﻣﺎﻥ ﻛﻪ ﺍﻳﻦ ﻫﻤﻪ ﺑﺮﺍﯼ ﺗﻮ ﺯﺣﻤﺖ ﻣﯽ‌ﻛﺸﻪ ﺑﻜﻨﯽ؟»
ﺩﻓﻌﻪ ﺑﻌﺪ ﻛﻪ ﺑﻪ ﻓﻜﺮﺗﺎﻥ ﺭﺳﻴﺪ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪ ﻛﻢ‌ﺗﺮ ﺍﺯ ﺷﺶ ﺳﺎﻟﻪ‌ﺗﺎﻥ ﻣﻮﻋﻈﻪ ﻛﻨﻴﺪ، ﭼﺮﺍ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺑﻪ ﺯﺑﺎﻥ ﺍﺳﭙﺎﻧﻴﻮﻟﯽ، ﺁﻟﻤﺎﻧﯽ ﻭ ﻳﺎ ﺍﺻﻼ ﺑﻪ ﻛﺎﺭ ﺑﺮﺩﻥ ﻛﻠﻤﺎﺕ ﺩﺭ ﻫﻢ ﻭ ﺑﺮﻫﻢ ﻭ ﻏﻴﺮ ﻗﺎﺑﻞ‌ ﺩﺭﮎ ﺍﻧﺠﺎﻡ ﻧﺪﻫﻴﺪ؟ ﺍﻳﻦ ﻛﺎﺭ ﻫﻤﺎﻥ ﺍﺛﺮﯼ ﺭﺍ ﺩﺍﺭﺩ ﻛﻪ ﻣﻮﻋﻈﻪ‌ﺍﯼ ﺑﻪ ﺯﺑﺎﻥ ﻓﺎﺭﺳﯽ ﺑﺮﺍﻳﺶ ﺩﺍﺭﺩ ﺑﺎ ﺍﻳﻦ ﺗﻔﺎﻭﺕ ﮐﻪ ﮐﻮﺩﮎ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﺍﺣﺴﺎﺱ ﻧﻤﯽ‌ﻛﻨﺪ ﻛﺎﺭ ﺑﺪﯼ ﻛﺮﺩﻩ ﺍﺳﺖ.
ﻫﺮ ﺭﻭﺯﻩ ﻭﺍﻟﺪﻳﻦ ﻓﺮﺯﻧﺪﺍﻥ ﺑﻴﭽﺎﺭﻩ‌ﺷﺎﻥ ﺭﺍ ﺩﻩ‌ﻫﺎ ﻫﺰﺍﺭ ﺑﺎﺭ ﺳﺮﺯﻧﺶ ﻣﯽ‌ﻛﻨﻨﺪ. ﺍﮔﺮ ﺿﺒﻂ ﺻﻮﺗﯽ ﺩﺭ ﮔﻮﺷﻪ ﻭ ﻛﻨﺎﺭ ﭘﻨﻬﺎﻥ ﻣﯽ‌ﺷﺪ ﻭ ﺩﻗﻴﻘﺎ ﺍﻳﻦ ﺳﺮﺯﻧﺶ‌ﻫﺎﯼ ﺑﯽ‌ﭘﺎﻳﺎﻥ ﺭﺍ ﺿﺒﻂ ﻣﯽ‌ﻛﺮﺩ، ﻭ ﺳﭙﺲ ﻣﺎﺩﺭ‌ﻫﺎ ﺁﻥ‌ﻫﺎ ﺭﺍ ﮔﻮﺵ ﻣﯽ‌ﻛﺮﺩﻧﺪ، ﺍﺯ ﺷﻨﻴﺪﻥ ﭼﻴﺰﻫﺎﻳﯽ ﻛﻪ ﻫﺮ ﺭﻭﺯ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻥ‌ﺷﺎﻥ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﺗﻌﺠﺐ ﻣﯽ‌ﻛﺮﺩﻧﺪ. ﺍﮔﺮ ﭼﻨﻴﻦ ﻧﻮﺍﺭﻫﺎﻳﯽ ﺿﺒﻂ ﻣﯽ‌ﺷﺪﻧﺪ ﺷﺎﻣﻞ ﻣﻮﻋﻈﻪ‌ﻫﺎﯼ ﻛﻮﭼﮏ ﺍﺧﻼﻗﯽ، ﻓﺮﻳﺎﺩﻫﺎ، ﻏﺮﻫﺎ ﻭ ﻓﺤﺶ‌ﻫﺎ ﻭ ﺗﺬﻛﺮﺍﺕ ﺗﺤﻘﻴﺮ ﻛﻨﻨﺪﻩ ﺑﻮﺩﻧﺪ.
ﺩﺭ ﻛﻮﺩﻛﯽ ﻛﻪ ﺗﺤﺖ ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ ﻗﺮﺍﺭ ﺩﺍﺭﺩ ﭘﺪﻳﺪﻩ ﺟﺎﻟﺒﯽ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻳﺪ. ﮔﻮﺵ‌ﻫﺎﻳﺶ ﺑﺴﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ. ﺍﻭ ﺑﻪ ﺗﻨﻬﺎ ﺳﻼﺣﯽ ﻛﻪ ﻋﻠﻴﻪ ﺍﻳﻦ ﺳﻴﻞ ﺻﺤﺒﺖ‌ ﻛﺮﺩﻥ‌ﻫﺎ ﺩﺍﺭﺩ ﻣﺘﻮﺳﻞ ﻣﯽ‌ﺷﻮﺩ. ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﻛﻪ ﺍﻭ ﻛﺎﻣﻼ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ «ﺻﺪﺍ ﺭﺍ ﻗﻄﻊ ﻛﻨﺪ»، ﺁﻥ ﺭﺍ ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ، ﻭ ﺁﻥ ﻫﻢ ﺑﻪ ﺻﻮﺭﺕ ﺑﺴﻴﺎﺭ ﺧﻮﺑﯽ، ﺿﺒﻂ ﻣﯽ‌ﻛﻨﺪ ﻭ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺑﻪ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌ﺍﺵ ﻟﻄﻤﻪ ﺯﻳﺎﺩﯼ ﻭﺍﺭﺩ ﻣﯽ‌ﺷﻮﺩ.
ﺳﺮﺯﻧﺶ‌ﻫﺎﯼ ﺷﻔﺎﻫﯽ ﺍﺛﺮ ﺑﺴﻴﺎﺭ ﮐﻤﯽ ﺭﻭﯼ ﺭﻓﺘﺎﺭ ﻛﻮﺩﮎ ﺩﺍﺭﺩ. ﺗﻨﻬﺎ ﺗﺎﺛﻴﺮﺵ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺗﻴﺸﻪ ﺑﻪ ﺭﻳﺸﻪ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻛﻮﺩﮎ ﻣﯽ‌ﺯﻧﺪ.
 
۱۲- ﺍﺯ ﺩﺍﺩﻥ ﺩﺳﺘﻮﺭﻫﺎﻳﯽ ﻛﻪ ﻗﺼﺪ ﺍﺟﺮﺍﻳﺶ ﺭﺍ ﻧﺪﺍﺭﻳﺪ ﭘﺮﻫﻴﺰ ﮐﻨﻴﺪ.
ﺍﻳﻦ ﺻﺤﻨﻪ‌­ﺍﯼ ﺑﺴﻴﺎﺭ ﺭﺍﻳﺞ ﺍﺳﺖ: ﻣﺎﺩﺭﯼ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﻣﯽ‌ﮔﻮﻳﺪ: «ﺑﺎﻻﯼ ﺻﻨﺪﻟﯽ ﻧﺮﻭ.» ﻭ ﻛﻮﺩﮎ ﺑﻪ ﺑﺎﻻﺭﻓﺘﻦ ﺍﺯ ﺻﻨﺪﻟﯽ ﺍﺩﺍﻣﻪ ﻣﯽ‌ﺩﻫﺪ. «ﺭﻳﭽﺎﺭﺩ، ﺑﻪ ﺗﻮ ﮔﻔﺘﻢ ﺍﺯ ﺻﻨﺪﻟﯽ ﺑﺎﻻ ﻧﺮﻭ.» ﻭ ﻛﻮﺩﻙ، ﺑﺪﻭﻥ ﺗﻮﺟﻪ ﺑﻪ ﻣﺎﺩﺭﺵ ﺑﻪ ﺑﺎﻻ ﺭﻓﺘﻨﺶ ﺍﺩﺍﻣﻪ ﻣﯽ‌ﺩﻫﺪ. «ﺭﻳﭽﺎﺭﺩ، ﻣﮕﻪ ﻧﻤﯽ‌ﺷﻨﻮﯼ؟ ﻫﻤﻴﻦ ﺍﻻﻥ ﺑﻪ ﺗﻮ ﮔﻔﺘﻢ  ﺑﺎﻻﯼ ﺻﻨﺪﻟﯽ ﻧﺮﯼ!» ﻛﻮﺩﮎ ﻫﻤﭽﻨﺎﻥ ﺑﯽ‌ﺗﻮﺟﻪ ﺑﻪ ﻣﺎﺩﺭﺵ، ﻛﻪ ﺍﻛﻨﻮﻥ ﺩﻳﮕﺮ ﺳﻌﯽ ﺩﺭ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺑﺎﻻ ﺭﻓﺘﻦ ﺍﻭ ﺍﺯ ﺻﻨﺪﻟﯽ ﻧﻤﯽ‌ﻛﻨﺪ ﺑﻪ ﻛﺎﺭﺵ ﺍﺩﺍﻣﻪ ﻣﯽ‌ﺩﻫﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻣﺎﺩﺭ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺁﻣﻮﺯﺩ ﻛﻪ ﺑﻪ ﺩﺳﺘﻮﺭﻫﺎ ﻭ ﻓﺮﻣﺎﻥ‌ﻫﺎﻳﺶ ﺑﯽ‌ﺗﻮﺟﻬﯽ ﻧﺸﺎﻥ ﺩﻫﺪ. ﺑﺮﺍﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﭼﻨﻴﻦ ﺁﻣﻮﺯﺵ ﻣﻨﻔﯽ‌ﺍﯼ، ﺩﺳﺘﻮﺭﯼ ﻛﻪ ﻗﺼﺪ ﺍﺟﺮﺍﻳﺶ ﺭﺍ ﻧﺪﺍﺭﻳﺪ ﻧﺪﻫﻴﺪ.
 
 
ﺗﺎ ﺍﻳﻦ‌ﺟﺎ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ، ﺩﺭ ﻣﻮﺭﺩ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ. ﺑﻴﺎﻳﺪ ﭘﻴﺶ ﺍﺯ ﺑﻪ ﭘﺎﻳﺎﻥ ﺭﺳﺎﻧﺪﻥ ﺍﻳﻦ ﺑﺤﺚ ﻳﮏ ﺑﺮﺭﺳﯽ ﺍﺟﻤﺎﻟﯽ ﺍﺯ ﺁﻥ‌ﭼﻪ ﺗﺎ ﺣﺎﻻ ﮔﻔﺘﻴﻢ ﺑﮑﻨﻴﻢ. 
 
ﺷﻤﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺷﺨﺺ ﺑﺎﻟﻐﯽ ﺷﻮﺩ ﻛﻪ ﺑﺘﻮﺍﻧﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻛﺎﺭﻫﺎﻳﺶ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﺪ ﻭ ﺍﺯ ﻋﻬﺪﻩ ﺍﺩﺍﺭﻩ ﻛﺮﺩﻥ ﺧﻮﺩ ﺑﺮﺁﻳﺪ، ﺑﻪ ﻋﺒﺎﺭﺗﯽ ﺩﻳﮕﺮ ﺗﺒﺪﻳﻞ ﺑﻪ ﺷﺨﺼﯽ ﻣﺴﺌﻮﻝ ﮔﺮﺩﺩ. ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ، ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﭘﻴﺶ ﺍﺯ ﻣﺪﺭﺳﻪ ﺭﻓﺘﻦ ﺗﺎ ﺳﻨﻴﻦ ﭘﻴﺸﺮﻓﺘﻪ ﺑﻠﻮﻍ، ﺑﺎﻳﺪ ﻗﻮﺍﻧﻴﻦ ﻭ ﻣﻘﺮﺭﺍﺕ ﺍﻧﻀﺒﺎﻃﯽ ﻛﻪ ﺍﺯ ﺟﺎﻧﺐ ﻭﺍﻟﺪﻳﻦ ﺑﺮﺍﯼ ﺭﺍﻫﻨﻤﺎﻳﯽ ﺍﻭ ﻭﺿﻊ ﻣﯽ‌ﮔﺮﺩﺩ ﺭﻋﺎﻳﺖ ﻛﻨﺪ، ﺗﺎ ﺯﻣﺎﻧﯽ ﻛﻪ ﺧﻮﺩ ﻗﺎﺩﺭ ﺑﻪ ﺍﺩﺍﺭﻩ ﺧﻮﺩ ﮔﺮﺩﺩ.
 
ﺩﻳﺴﻴﭙﻠﻴﻦ ﺭﺍ ﻋﻤﺪﺗﺎ ﺑﻪ ﺻﻮﺭﺕ ﺗﻌﻠﻴﻤﺎﺗﯽ ﻛﻪ ﺍﺯ ﻃﺮﻑ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻥ‌ ﺩﺭ ﺣﺎﻝ ﻓﺮﺍﮔﻴﺮﻳﺸﺎﻥ ﻋﺮﺿﻪ ﻣﯽ­‌ﺷﻮﺩ ﺗﻌﺮﻳﻒ ﻛﺮﺩﻳﻢ. ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻳﻦ ‌ﻛﻪ ﺩﻳﺴﻴﭙﻠﻴﻦ ﻳﮏ ﺭﻭﻧﺪ ﺗﻌﻠﻴﻤﺎﺗﯽ ﺍﺳﺖ، ﺗﺎﺑﻊ ﺍﺻﻮﻝ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ‌ﺍﯼ ﺍﺳﺖ ﻛﻪ ﺑﺮ ﻳﺎﺩﮔﻴﺮﯼ، ﭼﻪ ﻧﺰﺩ ﺣﻴﻮﺍﻧﺎﺕ ﻭ ﭼﻪ ﻧﺰﺩ ﺍﻧﺴﺎﻥ‌ﻫﺎ، ﺣﺎﻛﻢ ﺍﺳﺖ. ﺑﻌﻀﯽ ﺍﺯ ﺍﻳﻦ ﺍﺻﻮﻝ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻫﻢ ﺑﺮﺍﯼ ﺍﻧﺴﺎﻥ‌ﻫﺎ ﻭ ﻫﻢ ﺑﺮﺍﯼ ﺣﻴﻮﺍﻧﺎﺕ ﻗﺎﺑﻞ ﭘﻴﺎﺩﻩ ﻛﺮﺩﻥ ﻫﺴﺘﻨﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﺸﻨﺎﺳﻴﻢ ﻭ ﺑﻪ ﺻﻮﺭﺗﯽ ﻫﻮﺷﻴﺎﺭﺍﻧﻪ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻴﻢ. ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺷﻜﺴﺖ ﻣﺎ ﺩﺭ ﺩﺭﺍﺯ ﻣﺪﺕ ﺑﺮﺍﯼ ﺗﺮﺑﻴﺖ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺧﻴﻠﯽ ﺑﻴﺸﺘﺮ ﺍﺯ ﺷﻜﺴﺘﯽ ﺍﺳﺖ ﻛﻪ ﺗﺮﺑﻴﺖ‌ﻛﻨﻨﺪﮔﺎﻥ ﺩﻟﻔﻴﻦ‌ﻫﺎ ﺑﺮﺍﯼ ﻳﺎﺩ ﺩﺍﺩﻥ ﭼﻴﺰﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﻪ ﺁﻥ‌ﻫﺎ ﻣﺘﺤﻤﻞ ﻣﯽ‌ﺷﻮﻧﺪ.
 
ﻭﻟﯽ ﺍﺻﻮﻝ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺑﻴﺘﯽ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻛﻪ ﻣﺨﺘﺺ ﺍﻧﺴﺎﻥ‌ﻫﺎﺳﺖ ﻭ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﺑﺎﻳﺪ ﺣﺘﻤﺎ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﺸﻨﺎﺳﻨﺪ. ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺧﻮﺍﻫﻨﺪ ﻛﺮﺩ ﻛﻪ ﻣﺤﻜﻮﻡ ﺑﻪ ﺷﻜﺴﺖ ﺑﻮﺩﻩ ﻭ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺟﺰ ﺩﻟﺴﺮﺩﯼ ﺑﺮﺍﯼ ﺧﻮﺩ ﻭ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺩﺭ ﺑﺮﻧﺪﺍﺭﺩ.
 
ﺍﮔﺮ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺑﻴﺘﯽ ﻛﻪ ﺗﺠﺎﺭﺏ ﻋﻠﻤﯽ ﺗﺎﺛﻴﺮ ﺧﻮﺑﺸﺎﻥ ﺭﺍ ﻧﺸﺎﻥ ﺩﺍﺩﻩ‌­ﺍﻧﺪ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻴﻢ، ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻥ‌ﻣﺎﻥ ﺍﺯ ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ ﺗﺎ ﺯﻣﺎﻥ ﺑﻠﻮﻍ ﺑﻴﺎﻣﻮﺯﻳﻢ ﭼﮕﻮﻧﻪ ﺧﻮﺩ ﺭﺍ ﺍﺩﺍﺭﻩ ﻛﻨﻨﺪ ﻭ ﺩﻳﮕﺮ ﻣﺠﺒﻮﺭ ﻧﺨﻮﺍﻫﻴﻢ ﺑﻮﺩ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺭﺍ ﺍﺯ ﺧﺎﺭﺝ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺗﺤﻤﻴﻞ ﻛﻨﻴﻢ.
 
ﺑﻪ ﻛﺎﺭ ﺑﺮﺩﻥ ﭼﻨﻴﻦ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﻮﺛﺮ ﻭ ﺑﺎﺍﺭﺯﺷﯽ ﻭ ﻋﺪﻡ ﺑﻪ‌‌ﻛﺎﺭﮔﻴﺮﯼ ﺭﻭﺵ‌ﻫﺎﯼ ﺑﺪ ﭼﻪ ﺗﺎﺛﻴﺮﯼ ﺩﺍﺭﻧﺪ؟ ﺁﻳﺎ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ «ﻛﻮﺩﮎ ﻧﻤﻮﻧﻪ»‌ﺍﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ؟ ﺍﻣﻴﺪﻭﺍﺭﻡ ﻛﻪ ﺍﻳﻦ‌ﻃﻮﺭ ﻧﺒﺎﺷﺪ. ﻣﻦ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﺭﻭﺍﻧﺸﻨﺎﺱ ﺯﻳﺎﺩ ﺑﻪ «ﻛﻮﺩﮎ ﻧﻤﻮﻧﻪ» ﻋﻼﻗﻪ‌ﺍﯼ ﻧﺪﺍﺭﻡ. «ﻛﻮﺩﮎ ﻧﻤﻮﻧﻪ» ﻧﻪ ﺧﻮﺷﺒﺨﺖ ﺍﺳﺖ ﻭ ﻧﻪ ﻗﺎﺩﺭ ﺑﻪ ﺍﺩﺍﺭﻩ ﺧﻮﺩ. ﺍﻭ ﻛﻮﺩﻛﯽ ﺍﺳﺖ ﻛﻪ ﻓﻘﻂ ﺭﻭﯼ ﺳﻜﻪ ﺭﺍ ﺑﻪ ﻣﺎ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺍﻭ ﺭﺍ ﺑﺎ ﺗﻬﺪﻳﺪ ﻭ ﺍﺭﻋﺎﺏ ﻣﺠﺒﻮﺭ ﻛﺮﺩﻩ‌ﺍﻧﺪ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻗﻮﺍﻧﻴﻦ ﺑﻴﺮﻭﻧﯽ ﺗﻄﺎﺑﻖ ﻛﺎﻣﻞ ﺑﺪﻫﺪ، ﻟﻴﻜﻦ ﺩﺭ ﺩﺭﻭﻥ ﺍﻭ ﻧﺎﺭﺍﺣﺘﯽ ﻋﺎﻃﻔﯽ ﻗﺎﺑﻞ ﻣﻼﺣﻈﻪ‌ﺍﯼ ﺑﺎﻗﯽ ﻣﺎﻧﺪﻩ ﺍﺳﺖ. ﺍﮔﺮ ﻛﻮﺩﻛﯽ ﻛﻪ ﻫﻨﻮﺯ ﺑﻪ ﻣﺪﺭﺳﻪ ﻧﺮﻓﺘﻪ (ﺷﺶ ﺳﺎﻟﻪ) ﻫﻤﻴﺸﻪ ﺁﺭﺍﻡ ﺑﺎﺷﺪ ﻭ ﺩﺭ ﺗﻤﺎﻣﯽ ﻟﺤﻈﺎﺕ ﺑﺎ ﺍﺩﺏ، ﻫﺮﮔﺰ ﻋﺼﻴﺎﻥ ﻧﻜﻨﺪ، ﻫﻤﻴﺸﻪ ﺗﺤﺖ ﻛﻨﺘﺮﻝ ﻣﺎ ﺑﺎﺷﺪ، ﺍﺯ ﺍﻧﺠﺎﻡ ﻛﺎﺭﻫﺎﻳﯽ ﻛﻪ ﺑﺰﺭﮒ‌ﺗﺮﻫﺎ ﺍﺯ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺧﺮﺳﻨﺪ ﺑﺎﺷﺪ، ﻛﻪ ﻫﻴﭻ‌ﻛﺲ ﻭ ﻫﻴﭻ‌ﭼﻴﺰﯼ ﺩﺭ ﺍﻭ ﻫﻴﭻ ﺍﺣﺴﺎﺱ ﻣﻨﻔﯽ‌ﺍﯼ ﺑﻪ‌ﻭﺟﻮﺩ ﻧﻴﺎﻭﺭﺩ، ﻫﻴﭻ‌ﮔﺎﻩ ﺩﺭﻭﻍ ﻧﮕﻮﻳﺪ، ﻫﺮﮔﺰ ﺧﻮﺍﻫﺮ ﻭ ﺑﺮﺍﺩﺭﻫﺎ ﻳﺎ ﺩﻭﺳﺘﺎﻧﺶ ﺭﺍ ﻛﺘﮏ ﻧﺰﻧﺪ، ﻛﻪ ﺷﺠﺎﻉ ﻭ ﻧﻮﻉ‌ﺩﻭﺳﺖ ﺑﺎﺷﺪ، ﺑﻪ ﺍﺻﻮﻝ ﺍﺧﻼﻗﯽ ﭘﺎﻳﺒﻨﺪ ﺑﺎﺷﺪ، ﻛﻪ ﻫﻴﭻ ﻋﻼﻗﻪ‌ﺍﯼ ﺑﻪ ﻣﺴﺎﺋﻠﯽ ﻛﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻣﺴﺎﺋﻞ ﺟﻨﺴﯽ ﺍﺳﺖ ﻧﺸﺎﻥ ﻧﺪﻫﺪ، ﻛﻪ ﺑﺎ ﻭﺟﺪﺍﻥ ﺑﺎﺷﺪ، ﺗﻤﻴﺰ ﺑﺎﺷﺪ ﻭ ﺑﻪ ﺍﻣﻮﺍﻝ ﺩﻳﮕﺮﺍﻥ ﺍﺣﺘﺮﺍﻡ ﺑﮕﺬﺍﺭﺩ، ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﺎ ﻳﮏ ﻛﻮﺩﮎ ﺳﺮ ﻭ ﻛﺎﺭ ﻧﺪﺍﺭﻳﻢ. ﻛﺴﯽ ﻛﻪ ﺩﺭ ﻣﻘﺎﺑﻞ ﻣﺎﺳﺖ، ﻣﻮﺟﻮﺩﯼ ﺍﺳﺖ ﻛﻪ ﺑﺎ ﺍﺭﻋﺎﺏ ﺍﻭ ﺭﺍ ﺗﺒﺪﻳﻞ ﺑﻪ ﺷﺨﺼﯽ ﺑﺎﻟﻎ ﻛﺮﺩﻩ‌ﺍﻧﺪ ﻭﻟﯽ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻇﺎﻫﺮ ﻳﮏ ﻛﻮﺩﮎ ﻧﻤﺎﻳﺶ ﻣﯽ‌ﺩﻫﺪ.
