ﺩﺭﺱ ﺩﻭﺍﺯﺩﻫﻢ- ﭼﮕﻮﻧﻪ ﺭﺷﺪ ﺯﺑﺎﻥ ﺭﺍ ﺩﺭ ﻛﻮﺩﮎ ﺑﺮﺍﻧﮕﻴﺰﻳﻢ؟

 ﺩﺭ ﭼﻨﺪ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﺩﺭ ﻣﻮﺭﺩ ﺁﻣﻮﺯﺵ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﮐﻮﺩﮎ ﻭ ﺍﻫﻤﻴﺖ ﺁﻥ ﮔﻔﺘﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺩﺭ ﻣﻮﺭﺩ ﺭﺷﺪ ﺯﺑﺎﻥ ﮐﻮﺩﮎ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ. ﺯﺑﺎﻥ ﻛﻮﺩﮎ، ﺑﺎﻳﺪ ﺩﺭ ﺩﻭ ﺯﻣﻴﻨﻪ ﺑﺮﺭﺳﯽ ﺷﻮﺩ: ﺯﺑﺎﻥ ﺷﻔﺎﻫﯽ ﻭ ﺯﺑﺎﻥ ﻛﺘﺒﯽ. ﺭﺍﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺑﺮﺍﯼ ﺑﺮﺍﻧﮕﻴﺨﺘﻦ ﺍﻳﻦ ﺩﻭ ﺯﺑﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

ﮐﺘﺎﺏ
ﺍﻭﻟﻴﻦ ﺭﺍﻩ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺑﺮﺍﯼ ﺍﻭ ﺩﺍﺳﺘﺎﻥ ﻭ ﻛﺘﺎﺏ ﺑﺨﻮﺍﻧﻴﻢ. ﺍﺯ ﺳﻦ ﺳﻪ ﺳﺎﻟﮕﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻭ ﺭﺍ ﺑﻪ ﻛﺘﺎﺑﺨﺎﻧﻪ ﺑﺒﺮﻳﺪ ﻭ ﻛﻤﻜﺶ ﻛﻨﻴﺪ ﺗﺎ ﻛﺘﺎﺏ ﺍﻧﺘﺨﺎﺏ ﻛﻨﺪ. ﺍﻭ ﺭﺍ ﺑﻪ ﻣﺴﻮﻭﻝ ﻛﺘﺎﺑﺨﺎﻧﻪ ﻣﻌﺮﻓﯽ ﻛﻨﻴﺪ ﻭ ﺳﻌﯽ ﻛﻨﻴﺪ ﻛﻪ ﻛﺘﺎﺑﺪﺍﺭ ﺑﺎ ﺍﻭ ﺩﻭﺳﺖ ﺷﻮﺩ.

ﻭﻟﯽ ﺧﻮﺩ ﺭﺍ ﻣﺤﺪﻭﺩ ﺑﻪ ﻛﺘﺎﺑﺨﺎﻧﻪ ﻧﻜﻨﻴﺪ. ﻛﺘﺎﺏ‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺧﻮﺩﺵ ﺑﺨﺮﻳﺪ. ﺑﻬﺘﺮﻳﻦ ﻛﺎﺭ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﻫﺮ ﭼﻨﺪ ﻭﻗﺖ ﻳﮏ‌ﺑﺎﺭ ﺑﻪ ﻛﺘﺎﻓﺮﻭﺷﯽ‌ﻫﺎ ﺑﺮﻭﻳﺪ ﻭ ﻳﻜﯽ ﺩﻭ ﺳﺎﻋﺘﯽ ﻭﻗﺖ ﺑﮕﺬﺍﺭﻳﺪ ﺗﺎ ﺑﺒﻴﻨﻴﺪ ﭼﻪ ﻛﺘﺎﺏ‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﻛﻮﺩﻛﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺍﺯ ﺍﻫﻤﻴﺖ ﻛﺘﺎﺑﯽ ﻛﻪ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻣﯽ‌ﺧﺮﻧﺪ ﺑﯽ‌ﺧﺒﺮﻧﺪ ﻭ ﺑﺮﺣﺴﺐ ﺍﺗﻔﺎﻕ ﻛﺘﺎﺑﯽ ﺭﺍ ﻛﻪ ﺩﺭ ﻳﮏ ﺳﻮﭘﺮﻣﺎﺭﻛﺖ ﻣﯽ‌ﺑﻴﻨﺪ ﻣﯽ‌ﺧﺮﻧﺪ. ﺍﻳﻦ ﻛﺎﺭ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ. ﺍﻧﺘﺨﺎﺏ ﻛﺘﺎﺏ ﺍﻫﻤﻴﺖ ﻓﻮﻕ‌ﺍﻟﻌﺎﺩﻩ‌ﺍﯼ ﺩﺍﺭﺩ.

ﻛﺘﺎﺏ ﺧﻮﺍﻧﺪﻥ ﺑﺮﺍﯼ ﻛﻮﺩﮎ ﻓﻦ ﺧﺎﺻﯽ ﺩﺍﺭﺩ. ﺍﻭﻻ، ﻳﺎ ﺍﻭ ﺭﺍ ﺭﻭﯼ ﺯﺍﻧﻮﻳﺘﺎﻥ ﺑﻨﺸﺎﻧﻴﺪ ﻭ ﻳﺎ ﺍﻳﻨﻜﻪ ﺧﻴﻠﯽ ﻧﺰﺩﻳﻜﺶ ﺑﻨﺸﻴﻨﻴﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻳﻦ ﻛﺘﺎﺏ ﺧﻮﺍﻧﺪﻥ ﻣﻨﺤﺼﺮ ﺑﻪ ﻳﮏ ﺭﺍﺑﻄﻪ ﻓﻜﺮﯼ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ ﺑﻠﻜﻪ ﺗﺒﺪﻳﻞ ﺑﻪ ﺭﺍﺑﻄﻪ‌ﺍﯼ ﮔﺮﻡ ﻭ ﺷﻴﺮﻳﻦ ﻣﻴﺎﻥ ﺷﻤﺎ ﻭ ﻛﻮﺩﮎ ﻣﯽ‌ﺷﻮﺩ. ﻗﺒﻞ ﺍﺯ ﺧﻮﺍﻧﺪﻥ ﻛﺘﺎﺏ ﻧﺎﻡ ﻧﻮﻳﺴﻨﺪﻩ، ﻧﺎﻡ ﻛﺘﺎﺏ، ﻧﺎﺷﺮ ﻭ ﺳﺎﻝ ﺍﻧﺘﺸﺎﺭ ﺁﻥ ﺭﺍ ﺑﮕﻮﻳﻴﺪ. ﺍﮔﺮ ﻛﺘﺎﺏ ﻣﺼﻮﺭ ﺑﺎﺷﺪ ﻭ ﻧﺎﻡ ﻧﻘﺎﺵ ﺗﺼﺎﻭﻳﺮ ﺩﺭ ﺁﻥ ﺩﺭﺝ ﺷﺪﻩ، ﻧﺎﻡ ﺍﻭ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ.
ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺁﻣﻮﺯﻳﺪ ﻛﻪ ﺍﻧﺴﺎﻥ ﺑﺎﻳﺪ ﺑﻪ ﻛﺴﯽ ﻛﻪ ﻛﺘﺎﺏ ﺭﺍ ﻧﻮﺷﺘﻪ ﻭ ﺗﺼﺎﻭﻳﺮ ﺁﻥ ﺭﺍ ﻛﺸﻴﺪﻩ، ﻛﺴﯽ ﻛﻪ ﺁﻥ ﺭﺍ ﭼﺎﭖ ﻛﺮﺩﻩ ﻭ ﺯﻣﺎﻥ ﭼﺎﭖ ﻛﺘﺎﺏ ﺗﻮﺟﻪ ﻛﻨﺪ. ﺍﻳﻦ ﻋﺎﺩﺕ ﺩﺭ ﺁﻳﻨﺪﻩ، ﺯﻣﺎﻧﯽ ﻛﻪ ﺩﺭ ﻣﺪﺭﺳﻪ ﺍﻳﻦ ﺟﺰﺋﻴﺎﺕ ﺍﺯ ﻧﻈﺮ ﺍﺩﺑﯽ ﻭ ﺗﺎﺭﻳﺨﯽ ﺍﻫﻤﻴﺖ ﭘﻴﺪﺍ ﻣﯽ‌ﻛﻨﻨﺪ، ﺑﺴﻴﺎﺭ ﻣﻔﻴﺪ ﺍﺳﺖ.

ﻃﻮﺭﯼ ﺑﻨﺸﻴﻨﻴﺪ ﻛﻪ ﻭﻗﺘﯽ ﻛﺘﺎﺏ ﺭﺍ ﺑﺮﺍﻳﺶ ﻣﯽ‌ﺧﻮﺍﻧﻴﺪ، ﺑﺘﻮﺍﻧﺪ ﺗﺼﺎﻭﻳﺮ ﺭﺍ ﺑﺒﻴﻨﺪ. ﺭﻳﺰﻩ‌ﻛﺎﺭﯼ‌ﻫﺎﯼ ﻣﻬﻢ ﺍﻳﻦ ﺗﺼﺎﻭﻳﺮ ﺭﺍ ﻧﺸﺎﻧﺶ ﺩﻫﻴﺪ: «ﻣﻮﺵ ﺭﻭ ﺗﻮ ﺍﻳﻦ ﮔﻮﺷﻪ ﻣﯽ‌ﺑﻴﻨﯽ؟ ﺑﻪ‌ﻧﻈﺮ ﺗﻮ ﭼﯽ ﻛﺎﺭ ﻣﯽ‌ﻛﻨﻪ؟» ﺗﺸﻮﻳﻘﺶ ﻛﻨﻴﺪ ﻛﻪ ﺑﻪ ﺍﺟﺰﺍﺀ ﺗﺼﺎﻭﻳﺮ ﺗﻮﺟﻪ ﻛﻨﺪ. ﺑﺎ ﺍﻳﻦ ﻛﺎﺭ ﻧﻪ ﺗﻨﻬﺎ ﻛﺘﺎﺏ ﺧﻮﺍﻧﺪﻥ ﺭﺍ ﺑﺮﺍﯼ ﺍﻭ ﺟﺎﻟﺐ‌ﺗﺮ ﻣﯽ‌ﻛﻨﻴﺪ، ﺑﻠﻜﻪ ﺩﻗﺖ ﺍﻭ ﺭﺍ ﻧﻴﺰ ﺑﻴﺸﺘﺮ ﻣﯽ‌ﻛﻨﻴﺪ.

ﮐﺘﺎﺏ ﺭﺍ ﺑﺎ ﻟﺤﻦ ﻳﮑﻨﻮﺍﺧﺖ ﻧﺨﻮﺍﻧﻴﺪ. ﺍﺩﺍ ﺩﺭ ﺁﻭﺭﻳﺪ. ﺻﺪﺍﻳﺘﺎﻥ ﺭﺍ ﺑﺎﻻ ﻭ ﭘﺎﻳﻴﻦ ﺑﻴﺎﻭﺭﻳﺪ ﮐﻪ ﺩﺍﺳﺘﺎﻥ ﺟﺎﻟﺐ‌­ﺗﺮ ﺷﻮﺩ.
ﻫﺮ ﺑﺎﺭ ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﻳﮏ ﺑﭽﻪ ﮐﺘﺎﺏ ﺑﺨﻮﺍﻧﻴﺪ. ﮐﺘﺎﺏ ﺧﻮﺍﻧﺪﻥ ﺑﺮﺍﯼ ﭼﻨﺪ ﺑﭽﻪ ﺩﺭ ﺳﻨﻴﻦ ﻣﺘﻔﺎﻭﺕ ﮐﺎﺭ ﺑﺴﻴﺎﺭ ﻣﺸﮑﻠﯽ ﺍﺳﺖ. ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺗﻮﺻﻴﻪ ﻧﻤﯽ‌ﺷﻮﺩ.

ﺁﻳﺎ ﺑﺎﻳﺪ ﺯﻳﺎﺩ ﮐﺘﺎﺏ ﺧﻮﺍﻧﺪ؟ ﻫﺮ ﭼﻘﺪﺭ ﮐﻪ ﺩﻟﺘﺎﻥ‌ ﻣﯽ‌­ﺧﻮﺍﻫﺪ. ﺍﻣﺎ ﺑﻪ ﻫﺮ ﺣﺎﻝ ﺳﻌﯽ ﮐﻨﻴﺪ ﻫﺮ ﺷﺐ ﻳﮏ ﺩﺍﺳﺘﺎﻥ ﮐﺎﻣﻞ ﻳﺎ ﮐﺘﺎﺏ ﮐﻮﭼﮑﯽ ﺭﺍ ﺑﺮﺍﻳﺶ ﺑﺨﻮﺍﻧﻴﺪ.
 

ﺁﻣﻮﺯﺵ ﻧﻮﺷﺘﻦ
ﺩﻭﻣﻴﻦ ﮐﺎﺭﯼ ﮐﻪ ﺑﺮﺍﯼ ﺭﺷﺪ ﺯﺑﺎﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﺁﻣﻮﺯﺵ ﻧﻮﺷﺘﻦ ﺑﻪ ﺍﻭﺳﺖ.
ﻣﺜﻼ ﺣﺪﻭﺩ ﺳﻪ­‌ ﺳﺎﻝ ﻭ ﻧﻴﻤﮕﯽ ﺷﺮﻭﻉ ﻛﻨﻴﺪ. ﺣﺮﻭﻑ ﺍﻟﻔﺒﺎ ﻭ ﻳﮏ ﻣﺎﮊﻳﮏ ﺗﻬﻴﻪ ﻛﻨﻴﺪ (ﺑﺎ ﻣﺎﮊﻳﮏ ﺭﺍﺣﺖ‌ﺗﺮ ﺍﺯ ﻣﺪﺍﺩ ﻭ ﻳﺎ ﺧﻮﺩﻛﺎﺭ ﻣﯽ‌ﺗﻮﺍﻥ ﻧﻮﺷﺖ.)
ﺑﻌﻀﯽ ﺍﺯ ﺍﻓﺮﺍﺩ (ﻣﻌﻠﻤﺎﻥ ﻛﻮﺩﻛﺴﺘﺎﻥ‌ﻫﺎ ﻭ ﺁﻣﺎﺩﮔﯽ‌ﻫﺎ) ﻣﻤﻜﻦ ﺍﺳﺖ ﺑﻪ ﺷﻤﺎ ﺑﮕﻮﻳﻨﺪ ﻛﻪ ﺭﻭﺵ ﺧﺎﺻﯽ ﺑﺮﺍﯼ ﻳﺎﺩ ﺩﺍﺩﻥ ﻧﻮﺷﺘﻦ ﺑﻪ ﻛﻮﺩﻛﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺑﻪ ﺁﻥ ﺗﻮﺟﻬﯽ ﻧﻜﻨﻴﺪ. ﻫﻴﭻ ﺭﻭﺵ ﺧﺎﺻﯽ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﻫﺮ ﻃﻮﺭﯼ ﻛﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺘﻮﺍﻧﺪ ﺣﺮﻭﻑ ﺭﺍ ﺑﻨﻮﻳﺴﺪ ﺑﺮﺍﻳﺶ ﻣﻨﺎﺳﺐ‌ﺗﺮ ﺍﺳﺖ. ﺑﻪ‌ﻫﺮ‌ﺻﻮﺭﺕ ﺍﻭ ﺑﻌﺪﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺭﻭﺵ ﻧﻮﺷﺘﻨﺶ ﺭﺍ ﻋﻮﺽ ﻛﻨﺪ.

ﻫﻴﭻ ﭼﻴﺰﯼ ﺷﻤﺎ ﺭﺍ ﻣﺠﺒﻮﺭ ﻧﻤﯽ‌ﻛﻨﺪ ﻛﻪ ﺍﺯ ﺍﻟﻒ ﺷﺮﻭﻉ ﻛﻨﻴﺪ ﻭ ﺑﻪ ﯼ ﺑﺮﺳﻴﺪ. ﺍﻭﻝ ﺑﺎ ﺁﻥ‌ﻫﺎﻳﯽ ﺷﺮﻭﻉ ﻛﻨﻴﺪ ﻛﻪ ﺍﺯ ﻫﻤﻪ ﺁﺳﺎﻥ‌ﺗﺮ ﻫﺴﺘﻨﺪ. ﻣﺜﻼ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺻﻮﺭﺕ ﺯﻳﺮ ﺷﺮﻭﻉ ﻛﻨﻴﺪ. ﺍ-ﺏ-ﭖ-ﺕ-ﻥ-ﺩ-ﺫ-ﺭ-ﺯ-ﻭ-ﻑ-ﺱ-ﺵ-ﺙ-ﻭ-ﻡ-ﻕ-ﻝ-ﮎ-ﮒ-ﺝ-ﭺ-ﺥ-ﻉ-ﻍ-ﯼ-ﻡ-ﺹ-ﺽ-ﻁ-ﻅ-ﻩـ . ﺍﻟﺒﺘﻪ ﺩﺭ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻫﻴﭻ ﺍﺟﺒﺎﺭﯼ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﻓﻘﻂ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻛﻪ ﺍﺯ ﺁﺳﺎﻥ‌ﺗﺮﻫﺎ ﺷﺮﻭﻉ ﻛﻨﻴﺪ ﻭ ﺑﻪ ﺳﺨﺖ‌ﺗﺮﻳﻦ‌ﻫﺎ ﺑﺮﺳﻴﺪ. ﺑﺪﺍﻧﻴﺪ ﻛﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺑﺮﺍﯼ «ﺑﭽﻪ‌ﻫﺎ ﺑﻪ ﻃﻮﺭ ﻛﻠﯽ ﺍﺳﺖ» ﻭ ﻓﺮﺯﻧﺪ ﺷﻤﺎ «ﺑﭽﻪ‌ﻫﺎ ﺑﻪ ﻃﻮﺭ ﻛﻠﯽ» ﻧﻴﺴﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﯽ‌ﺷﮏ ﺑﺮﺍﯼ ﺍﻭ ﺍﺯ ﺁﺳﺎﻥ‌ﺗﺮﻳﻦ ﺑﻪ ﺳﺨﺖ‌ﺗﺮﻳﻦ‌ﻫﺎ ﺗﺮﺗﻴﺐ ﺩﻳﮕﺮﯼ ﺩﺍﺭﺩ ﻛﻪ ﻓﻘﻂ ﺧﺎﺹ ﺧﻮﺩ ﺍﻭﺳﺖ.

ﻣﺪﺗﯽ ﻃﻮﻝ ﻣﯽ‌­ﻛﺸﺪ ﺗﺎ ﻛﻮﺩﮎ ﻧﻮﺷﺘﻦ ﺍﻳﻦ ﺣﺮﻭﻑ ﺭﺍ ﻳﺎﺩ ﺑﮕﻴﺮﺩ ، ﻭ ﻫﻤﺎﻧﻄﻮﺭ ﻛﻪ ﻗﺒﻼ ﮔﻔﺘﻴﻢ ﺑﻌﻀﯽ‌ﻫﺎﻳﺸﺎﻥ ﺑﺮﺍﻳﺶ ﻣﺸﻜﻞ‌ﺗﺮ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ. ﻭﻟﯽ ﺻﺒﻮﺭ ﺑﺎﺷﻴﺪ. ﻧﺸﺎﻧﺶ ﺩﻫﻴﺪ ﻛﻪ ﭼﮕﻮﻧﻪ ﻳﮏ ﺣﺮﻑ ﻧﻮﺷﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻫﺮ ﭼﻘﺪﺭ ﻛﻪ ﺯﻣﺎﻥ ﻻﺯﻡ ﺩﺍﺭﺩ ﺑﻪ ﺍﻭ ﺑﺪﻫﻴﺪ ﺗﺎ ﻧﻮﺷﺘﻦ ﺁﻥ ﺭﺍ ﻳﺎﺩ ﺑﮕﻴﺮﺩ. ﺷﺎﻳﺪ ﺍﮔﺮ ﺩﺭ ﺳﻪ ﺳﺎﻝ ﻭ ﻧﻴﻤﮕﯽ ﺷﺮﻭﻉ ﺑﻜﻨﻴﺪ ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ ﻓﻘﻂ ﻳﮏ ﺳﺎﻝ ﻭﻗﺖ ﻻﺯﻡ ﺑﺎﺷﺪ. ﻫﻤﻪ ﭼﻴﺰ ﺑﺴﺘﮕﯽ ﺑﻪ ﺍﻭ ﺩﺍﺭﺩ، ﻭﻟﯽ ﻫﻴﭻ ‌ﻋﺠﻠﻪ‌ﺍﯼ ﺩﺭ ﺍﻳﻦ ﻛﺎﺭ ﻧﻴﺴﺖ. ﻓﺮﺯﻧﺪﺗﺎﻥ ﺗﻤﺎﻡ ﻋﻤﺮﺵ ﺑﺮﺍﯼ ﻳﺎﺩ ﮔﺮﻓﺘﻦ ﻓﺮﺻﺖ ﺩﺍﺭﺩ ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﻭ ﺭﺍ ﺗﺤﺖ ﻓﺸﺎﺭ ﻧﮕﺬﺍﺭﻳﺪ.

ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺣﺮﻭﻑ ﺍﺳﻤﺶ ﺭﺍ ﻳﺎﺩ ﮔﺮﻓﺖ، ﺑﻪ ﺍﻭ ﻧﺸﺎﻥ ﺩﻫﻴﺪ ﻛﻪ ﭼﮕﻮﻧﻪ ﺁﻥ ﺭﺍ ﻣﯽ‌ﻧﻮﻳﺴﻨﺪ. ﺑﺮﺍﯼ ﻳﮏ ﻛﻮﺩﮎ ﭘﻴﺶ ﺍﺯ ﺳﻨﻴﻦ ﻣﺪﺭﺳﻪ ﺍﻳﻦ ﻟﺤﻈﻪ ﻛﻪ ﭼﮕﻮﻧﻪ ﺍﺳﻤﺶ ﻧﻮﺷﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﺧﻴﻠﯽ ﻣﻬﻢ ﺍﺳﺖ.
ﺍﻭ ﺣﺘﻤﺎ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻛﻪ ﺍﺳﻢ ﻳﮏ ﻧﻔﺮ ﺩﻳﮕﺮ ﺭﺍ ﻫﻢ ﺑﻨﻮﻳﺴﺪ، ﻣﺜﻼ ﺍﺳﻢ ﻣﺎﻣﺎﻥ ﻳﺎ ﺑﺎﺑﺎ ﻭ ﻳﺎ ﻳﮏ ﺩﻭﺳﺖ ﺧﻮﺏ. ﻣﻤﻜﻦ ﺍﺳﺖ ﻗﺒﻞ ﺍﺯ ﻫﺮﭼﻴﺰ ﺑﺨﻮﺍﻫﺪ ﺍﻭﻝ ﻧﻮﺷﺘﻦ ﺍﺳﻤﺶ ﺭﺍ ﻳﺎﺩ ﺑﮕﻴﺮﺩ ﻭ ﺩﺭ ﺍﻳﻦ ﻛﺎﺭ ﻣﺼﺮ ﺑﺎﺷﺪ. ﺍﮔﺮ ﺍﻳﻦ ﺑﺮﺍﻳﺶ ﺟﺎﻟﺐ ﺑﺎﺷﺪ، ﭼﺮﺍ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﻧﻜﻨﻴﻢ؟ ﺍﻳﻦ ﻳﮏ ﻗﺎﻧﻮﻥ ﺍﺳﺎﺳﯽ ﺩﺭ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺍﺳﺖ، ﻫﺮ ﻭﻗﺖ ﻛﻮﺩﻛﯽ ﻋﻼﻗﻪ‌ﺍﺵ ﺑﻪ ﻳﺎﺩ ﮔﺮﻓﺘﻦ ﭼﻴﺰﯼ ﺭﺍ ﻧﺸﺎﻥ ﺩﺍﺩ، ﺑﺮﻧﺎﻣﻪ‌ﺍﯼ ﺭﺍ ﻛﻪ ﺍﺯ ﻗﺒﻞ ﭘﻴﺶ‌ﺑﻴﻨﯽ ﻛﺮﺩﻩ ﺑﻮﺩﻳﺪ ﺑﻪ ﻛﻨﺎﺭﯼ ﺑﮕﺬﺍﺭﻳﺪ. ﺑﻼﻓﺎﺻﻠﻪ ﭼﻴﺰﯼ ﺭﺍ ﻛﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﺪ.

ﺭﺍﻩ ﺳﻮﻣﯽ ﻛﻪ ﺑﻪ ﺭﺷﺪ ﺯﺑﺎﻥ ﻛﻮﺩﮎ ﻛﻤﮏ ﻣﯽ‌ﻛﻨﺪ ﻋﺒﺎﺭﺕ ﺍﺯ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﺪ ﺧﻮﺩﺵ ﻛﺘﺎﺏ ﺩﺭﺳﺖ ﻛﻨﺪ. ﺩﺭﺳﺖ ﻛﺮﺩﻥ ﻛﺘﺎﺏ ﻳﻜﯽ ﺍﺯ ﺟﺎﻟﺐ‌ﺗﺮﻳﻦ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﻳﯽ‌ ­ﺍﺳﺖ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﭘﻴﺸﻨﻬﺎﺩ ﻛﻨﻴﺪ. ﺯﻣﺎﻧﯽ ﺭﺍ ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ ﺍﻧﺘﺨﺎﺏ ﻛﻨﻴﺪ ﻛﻪ ﻫﺮ ﺩﻭﯼ ﺷﻤﺎ ﺳﺮﺣﺎﻝ ﺑﺎﺷﻴﺪ ﻭ ﺑﻪ ﺍﻭ ﭘﻴﺸﻨﻬﺎﺩ ﻛﻨﻴﺪ: «ﺣﺎﻻ ﻣﺎ ﻳﮏ ﻛﺘﺎﺏ ﻣﯽ‌ﻧﻮﻳﺴﻴﻢ. ﻣﺎﻣﺎﻥ ﺑﻬﺖ ﻧﺸﻮﻥ ﻣﯽ‌ﺩﻩ ﻛﻪ ﭼﻄﻮﺭ ﺍﻳﻦ ﻛﺎﺭﻭ ﻣﯽ‌ﻛﻨﻦ. ﺍﻳﻦ ﻛﺘﺎﺏ ﺍﺛﺮ ﺗﻮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ!» ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ ﻣﻘﺪﺍﺭﯼ ﻭﺳﺎﻳﻞ ﻻﺯﻡ ﺍﺳﺖ: ﻣﺠﻠﻪ‌ﻗﺪﻳﻤﯽ- ﺧﺼﻮﺻﺎ ﺁﻥ‌ﻫﺎﻳﯽ ﻛﻪ ﺩﺭ ﺁﻥ‌ﻫﺎ ﺗﺼﺎﻭﻳﺮ ﺯﻳﺎﺩﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﻗﻴﭽﯽ، ﻛﺎﻏﺬ،‌ ﻣﺎﺷﻴﻦ ﺩﻭﺧﺖ ﻭ ﻳﺎ ﻧﺦ (ﺑﺮﺍﯼ ﻭﺻﻞ ﻛﺮﺩﻥ ﺻﻔﺤﺎﺕ ﺑﻪ ﻫﻢ)، ﻣﻘﻮﺍ ﺑﺮﺍﯼ ﺟﻠﺪ ﻛﺘﺎﺏ، ﻣﺎﮊﻳﮏ ﻭ ﻣﺪﺍﺩ ﺭﻧﮕﯽ ﻭ ﭘﺎﺳﺘﻞ.
ﺩﻭ ﻧﻮﻉ ﻛﺘﺎﺏ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭﺳﺖ ﻛﺮﺩ: ﺍﻭﻝ ﺁﻥ ﻧﻮﻋﯽ ﻛﻪ ﻛﻮﺩﮎ ﺑﻪ ﺷﻤﺎ ﺩﻳﻜﺘﻪ ﻣﯽ‌ﻛﻨﺪ ﻭ ﺷﻤﺎ ﺁﻥ ﺭﺍ ﻣﯽ‌ﻧﻮﻳﺴﻴﺪ ﻭ ﺩﻳﮕﺮﯼ ﻛﺘﺎﺑﯽ ﺍﺳﺖ ﻛﻪ ﺧﻮﺩ ﺍﻭ، ﺯﻣﺎﻧﯽ ﻛﻪ ﺩﺭ ﻧﻮﺷﺘﻦ ﺗﺒﺤﺮ ﭘﻴﺪﺍ ﻛﺮﺩ ﺧﻮﺍﻫﺪ ﻧﻮﺷﺖ. ﻃﺒﻴﻌﺘﺎ ﻛﺘﺎﺑﯽ ﻛﻪ ﺍﻭﻝ ﺑﺎﺭ ﻧﻮﺷﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﻛﺘﺎﺑﯽ ﺍﺳﺖ ﻛﻪ ﺍﻭ ﺑﻪ ﺷﻤﺎ ﺩﻳﻜﺘﻪ ﺧﻮﺍﻫﺪ ﻛﺮﺩ. ﺍﻳﻦ ﺑﺎﺯﯼ ﻧﻴﺰ ﻣﺜﻞ ﻫﺮ ﺑﺎﺯﯼ ﺩﻳﮕﺮ ﺗﺮﺑﻴﺘﯽ ﺑﺎﻳﺪ ﺣﺪﻭﺩ ﺳﻦ ﺳﻪ ﺳﺎﻟﮕﯽ ﺷﺮﻭﻉ ﺷﻮﺩ، ﻣﻌﻤﻮﻻ ﺍﻳﻦ ﺳﻦ، ﺳﻦ ﺗﻌﺎﺩﻝ ﺍﺳﺖ ﻭ ﻛﻮﺩﮎ ﺑﻴﺸﺘﺮ ﺑﺎ ﻭﺍﻟﺪﻳﻨﺶ ﻫﻤﻜﺎﺭﯼ ﻣﯽ‌ﻛﻨﺪ (ﻧﺴﺒﺖ ﺑﻪ ﺳﻦ 2 ﺗﺎ 3 ﺳﺎﻟﮕﯽ).

ﻣﻮﺿﻮﻉ ﻛﺘﺎﺏ ﭼﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎﺷﺪ؟ ﺍﺟﺎﺯﻩ ﺑﺪﻫﻴﺪ ﻛﻪ ﻃﻔﻞ ﺧﻮﺩ ﻣﻮﺿﻮﻉ ﻛﺘﺎﺏ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻛﻨﺪ. ﺍﻳﻦ ﺑﺴﺘﮕﯽ ﺑﻪ ﭼﻴﺰﻫﺎﯼ ﻣﻮﺭﺩ ﻋﻼﻗﻪ ﺍﻭ ﺩﺍﺭﺩ. ﻣﻤﻜﻦ ﺍﺳﺖ ﺑﺨﻮﺍﻫﺪ ﺩﺭ ﻣﻮﺭﺩ ﺳﮓ، ﮔﺮﺑﻪ، ﻛﺎﻣﻴﻮﻥ، ﻋﺮﻭﺳﮏ ﻳﺎ ﺩﺍﻳﻨﺎﺳﻮﺭﻫﺎ، ﻣﻬﻤﺎﻧﯽ ﺧﺎﻧﻪ ﻣﺎﺩﺭﺑﺰﺭﮒ ﻭ ﻳﺎ ﻻﮎ‌ﭘﺸﺖ ﺑﻨﻮﻳﺴﺪ. ﻣﻮﺿﻮﻉ ﺍﻫﻤﻴﺘﯽ ﻧﺪﺍﺭﺩ. ﺑﻪ ﻫﻤﻴﻦ ﻋﻠﺖ ﻣﺴﺎﻟﻪ ﺟﺎﻟﺐ ﺍﺳﺖ. ﻓﻜﺮ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ ﺧﺴﺘﻪ ﻧﻜﻨﻴﺪ ﭼﺮﺍ ﻛﻪ ﺍﻳﻦ ﻛﺘﺎﺏ ﻣﺎﻝ ﺍﻭ، ﻭ ﺧﻮﺩ ﺍﻭ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻓﻘﻂ ﺑﮕﻮﻳﻴﺪ: «ﺑﺮﺍﯼ ﻛﺘﺎﺏ ﻧﻮﺷﺘﻦ ﺁﺩﻡ ﺑﺎﻳﺪ ﭼﻴﺰﯼ ﺭﻭ ﻛﻪ ﻣﯽ‌ﺧﻮﺍﺩ ﺑﻨﻮﻳﺴﻪ. ﺗﻮ ﺑﮕﻮ، ﻣﻦ ﻣﯽ‌ﻧﻮﻳﺴﻢ. ﺣﺎﻻ ﺷﺮﻭﻉ ﻛﻨﻴﻢ.»

ﻣﺜﺎﻝ ﺯﻳﺮ ﻳﮏ ﻣﺜﺎﻝ ﻭﺍﻗﻌﯽ ﺍﺳﺖ ﻛﻪ ﻳﮏ ﭘﺴﺮ ﺳﻪ‌ﺳﺎﻟﻪ ﺑﻪ ﻣﺎﺩﺭﺵ ﻛﻪ ﺍﺯ ﺍﻭ ﭘﺮﺳﻴﺪﻩ ﺑﻮﺩ ﺩﺭ ﻣﻮﺭﺩ ﭼﻪ ﭼﻴﺰﯼ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻛﺘﺎﺏ ﺑﻨﻮﻳﺴﺪ ﮔﻔﺘﻪ ﺑﻮﺩ ﺩﺭﺑﺎﺭﻩ «ﺑﺖ‌ﻣﻦ».
- ﺧﻴﻠﯽ ﺧﻮﺏ. ﺩﺭ ﻣﻮﺭﺩ ﺑﺖ‌ﻣﻦ ﭼﯽ ﻣﯽ‌ﺧﻮﺍﯼ ﺑﻨﻮﻳﺴﯽ؟
- ﻛﻪ ﺍﻭﻥ ﺧﻴﻠﯽ ﻣﻬﺮﺑﻮﻧﻪ.
- ﺧﻴﻠﯽ ﺧﻮﺏ. ﺍﻳﻦ ﻣﯽ‌ﺷﻪ ﺍﻭﻟﻴﻦ ﺻﻔﺤﻪ ﻛﺘﺎﺏ.
«ﺑﺖ ﻣﻦ ﺧﻴﻠﯽ ﻣﻬﺮﺑﺎﻥ ﺍﺳﺖ.» ﺍﻭ ﻋﻜﺲ ﺑﺖ‌ﻣﻦ ﺭﺍ ﻧﺪﺍﺷﺖ ﺑﻨﺎﺑﺮﺍﻳﻦ ﻳﮏ ﺟﺎﯼ ﺧﺎﻟﯽ ﮔﺬﺍﺷﺖ ﻛﻪ ﺑﻌﺪﺍ ﻓﺮﺯﻧﺪﺵ ﻋﻜﺲ ﺍﻭ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﺑﻜﺸﺪ ﻭ ﻳﺎ ﺗﺼﻮﻳﺮﺵ ﺭﺍ ﺩﺭ ﺁﻥ ﺑﭽﺴﺒﺎﻧﺪ. ﺳﭙﺲ ﻣﺎﺩﺭ ﭘﺮﺳﻴﺪ: «ﺧﻮﺏ، ﺣﺎﻻ ﺩﺭ ﻣﻮﺭﺩ ﺑﺖ‌ﻣﻦ ﭼﯽ ﻣﯽ‌ﺧﻮﺍﯼ ﺑﻨﻮﻳﺴﯽ؟
- ﺍﻭﻥ ﺑﻪ ﭘﻠﻴﺲ ﻛﻤﮏ ﻣﯽ‌ﻛﻨﻪ ﻛﻪ ﺩﺯﺩﻫﺎ ﺭﻭ ﺩﺳﺘﮕﻴﺮ ﻛﻨﻪ.
ﺍﻭ ﺍﻳﻦ ﺟﻤﻠﻪ ﺭﺍ ﺩﺭ ﺻﻔﺤﻪ ﺑﻌﺪ ﻧﻮﺷﺖ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺗﺮﺗﻴﺐ ﺑﺎ ﺻﻔﺤﻪ‌ﺍﯼ ﻳﮏ ﺟﻤﻠﻪ ﻛﺎﺭ ﺭﺍ ﺍﺩﺍﻣﻪ ﺩﺍﺩ. ﻣﺎﺩﺭ ﻫﺮ ﭼﻪ ﺭﺍ ﻛﻪ ﻓﺮﺯﻧﺪﺵ ﻣﯽ‌ﮔﻔﺖ ﻣﯽ‌ﻧﻮﺷﺖ ﺑﺪﻭﻥ ﺍﻳﻨﻜﻪ ﻓﻜﺮ ﻛﻨﺪ ﻛﻪ ﻛﻮﺩﮎ ﻣﯽ‌ﺗﻮﺍﻧﺴﺖ ﻣﻮﺿﻮﻉ ﺟﺎﻟﺐ‌ﺗﺮﯼ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻛﻨﺪ ﻭ ﻳﺎ ﺳﺒﮏ ﺑﻬﺘﺮﯼ ﺑﺮﺍﯼ ﺟﻤﻼﺕ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻛﻨﺪ. ﺯﻳﺮﺍ ﺍﻳﻦ ﻛﺘﺎﺏ ﻧﻪ ﻣﺎﻝ ﺍﻭ، ﺑﻠﻜﻪ ﺍﺯ ﺁﻥ ﻓﺮﺯﻧﺪﺵ ﺑﻮﺩ.

ﺗﺎ ﺯﻣﺎﻧﯽ ﻛﻪ ﻛﻮﺩﮎ ﺑﻪ ﻣﻮﺿﻮﻉ ﻋﻼﻗﻪ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺑﻪ ﺷﻤﺎ ﺩﻳﻜﺘﻪ ﻣﯽ‌ﻛﻨﺪ، ﺷﻤﺎ ﺑﻨﻮﻳﺴﻴﺪ. ﺯﻣﺎﻧﯽ ﻛﻪ ﻛﻢ‌ﻛﻢ ﻋﻼﻗﻪ‌ﺍﺵ ﺑﻪ ﻣﻮﺿﻮﻉ ﻛﻢ ﻣﯽ‌ﺷﻮﺩ، ﻳﻌﻨﯽ ﻛﻪ ﺑﺎﻳﺪ ﻛﺘﺎﺏ ﺭﺍ ﺗﻤﺎﻡ ﻛﺮﺩ. ﻳﮏ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ ﺳﺮﻳﻊ ﺑﻜﻨﻴﺪ ﻭ ﮐﺘﺎﺏ ﺭﺍ ﺗﻤﺎﻡ ﮐﻨﻴﺪ. ﺣﺘﯽ ﺍﮔﺮ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻫﻨﻮﺯ ﺧﻮﺍﻧﺪﻥ ﻭ ﻧﻮﺷﺘﻦ ﺭﺍ ﻳﺎﺩ ﻧﮕﺮﻓﺘﻪ ﺍﺳﺖ، ﻛﺘﺎﺏ ﺭﺍ ﺑﻪ ﻃﺮﺯ ﺻﺤﻴﺤﯽ ﺑﻨﻮﻳﺴﻴﺪ.

ﻫﻤﻴﻦ‌ﻛﻪ ﻧﻮﺷﺘﻦ ﻛﺘﺎﺏ ﺗﻤﺎﻡ ﺷﺪ، ﺁﻥ ﺭﺍ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﺑﺮﺍﻳﺶ ﺑﺨﻮﺍﻧﻴﺪ. ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﻛﻪ ﺍﻭ ﺧﻮﺩﺵ ﺍﻳﻦ ﻛﺘﺎﺏ ﺭﺍ ﻧﻮﺷﺘﻪ ﺍﺳﺖ، ﻣﺴﻠﻤﺎ ﺍﺯ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﺧﻮﺍﺳﺖ ﻛﻪ ﺩﻭ ﺳﻪ ﺑﺎﺭ ﺁﻥ‌ ﺭﺍ ﺑﺮﺍﻳﺶ ﺑﺨﻮﺍﻧﻴﺪ. ﻭ ﺍﻳﻦ ﻛﺘﺎﺏ ﻭﺍﻗﻌﺎ ﻛﺎﺭ ﺧﻮﺩ ﺍﻭﺳﺖ.
ﺍﺳﻢ ﺍﻭﻟﻴﻦ ﻛﺘﺎﺏ ﺑﺎﻳﺪ ﺷﺎﻣﻞ ﺍﺳﻢ ﺧﻮﺩ ﻛﻮﺩﮎ ﻧﻴﺰ ﺑﺎﺷﺪ. ﺭﻭﯼ ﺟﻠﺪ ﻣﻘﻮﺍﻳﯽ ﻛﺘﺎﺏ ﺑﻨﻮﻳﺴﻴﺪ «ﻛﺘﺎﺏ ﻣﺎﻳﻜﻞ.» ﺑﻌﺪ ﻳﮏ ﻋﻜﺲ ﺍﺯ ﻛﻮﺩﮎ ﺭﺍ ﺭﻭﯼ ﺟﻠﺪ ﻛﺘﺎﺏ ﺑﭽﺴﺒﺎﻧﻴﺪ.
