ﺩﺭﺱ ﭼﻬﺎﺭﺩﻫﻢ- ﺟﻤﻊ ﺑﻨﺪﯼ ﻭ ﻣﺮﻭﺭﯼ ﺑﺮ ﺩﻭﺭﻩ

ﺍﺯ ﺁﻏﺎﺯ ﮔﻔﺘﮕﻮﻳﻤﺎﻥ ﺩﺭﺑﺎﺭﻩ ﮐﻮﺩﮐﺎﻥ ﺧﺮﺩﺳﺎﻝ، ﺩﺭﺑﺎﺭﻩ ﻣﺴﺎﻳﻞ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ. ﺗﻌﺪﺍﺩ ﺍﻳﻦ ﻣﺴﺎﻳﻞ ﺣﻴﺮﺕ‌­ﺍﻧﮕﻴﺰ ﺍﺳﺖ. ﺩﺭﺑﺎﺭﻩ ﻫﻤﻪ ﭼﻴﺰ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ ﺍﺯ ﺧﺸﻢ ﮔﺮﻓﺘﻪ ﺗﺎ ﺁﻣﻮﺯﺵ ﺧﻮﺍﻧﺪﻥ، ﺣﺘﯽ ﻃﺮﺯ ﺗﻬﻴﻪ ﺧﻤﻴﺮ ﺑﺮﺍﯼ ﺧﻤﻴﺮﺑﺎﺯﯼ ﺭﺍ ﻫﻢ ﮔﻒ
ﺗﻴﻢ.
ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻣﺴﺎﺋﻞ ﺭﻭﺷﻦ‌­ﺗﺮ ﺷﻮﺩ، ﺍﻳﺪﻩ­‌ﻫﺎﯼ ﺍﺻﻠﯽ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭﺳﻨﺎﻣﻪ ﺭﺍ ﺑﺮﺍﻳﺘﺎﻥ ﺑﻪ ﺻﻮﺭﺕ ﺳﻴﺰﺩﻩ ﺗﻮﺻﻴﻪ ﺧﻼﺻﻪ ﻣﯽ‌­ﮐﻨﻴﻢ ﮐﻪ ﺑﺮﺍﯼ ﺑﺰﺭﮒ ﮐﺮﺩﻥ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺑﺎﻳﺪ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. 
 
۱- ﺑﺰﺭﮒ ﮐﺮﺩﻥ ﻓﺮﺯﻧﺪ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩﻥ ﺭﺍﺑﻄﻪ‌ﺍﯼ ﺍﻧﺴﺎﻧﯽ ﺍﺳﺖ ﻭ ﺭﻭﺍﺑﻂ ﺍﻧﺴﺎﻧﯽ ﻧﻤﯽ‌­ﺗﻮﺍﻧﻨﺪ ﺩﺭ ﻗﻮﺍﻧﻴﻦ ﺧﻼﺻﻪ ﺷﻮﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻫﻴﭻ ﻗﺎﻧﻮﻧﯽ ﺭﺍ ﺣﺘﯽ ﺁﻥ‌ﻫﺎﻳﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺧﻮﺍﻧﺪﻳﺪ ﺑﻪ ﺻﻮﺭﺗﯽ ﺧﺸﮏ ﺍﺟﺮﺍ ﻧﮑﻨﻴﺪ. ﻗﻮﺍﻧﻴﻦ ﻓﻘﻂ ﺟﻬﺖ ﺭﺍﻫﻨﻤﺎﻳﯽ ﺍﺳﺖ. ﺷﻤﺎ ﻭ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻫﺮ ﺩﻭ ﻣﻮﺟﻮﺩﺍﺗﯽ ﻣﻨﺤﺼﺮ ﺑﻪ ﻓﺮﺩ ﻫﺴﺘﻴﺪ. ﺭﺍﺑﻄﻪ ﺑﻴﻦ ﺷﻤﺎ ﺩﻭ ﻧﻔﺮ ﻧﻮﻉ ﺧﺎﺻﯽ ﺍﺳﺖ ﮐﻪ ﻣﺘﻔﺎﻭﺕ ﺑﺎ ﻫﺮﮔﻮﻧﻪ ﺭﺍﺑﻄﻪ‌­ﺍﯼ ﺍﺳﺖ ﮐﻪ ﺩﻭ ﻧﻔﺮ ﺩﻳﮕﺮ ﺩﺭ ﺟﻬﺎﻥ ﻣﯽ‌­ﺗﻮﺍﻧﻨﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻣﺮﺗﮑﺐ ﺍﻳﻦ ﺍﺷﺘﺒﺎﻩ ﻧﺸﻮﻳﺪ ﮐﻪ ﺑﺨﻮﺍﻫﻴﺪ ﺍﻳﻦ ﺭﺍﺑﻄﻪ ﺧﺎﺹ ﺭﺍ ﺷﺒﻴﻪ ﺑﻪ ﺍﻟﮕﻮﯼ ﮐﻠﯽ‌­ﺍﯼ ﮐﻨﻴﺪ.
 
۲- ﺷﺶ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﺳﺎﻝ‌ﻫﺎ ﺑﺮﺍﯼ ﺷﮑﻞ ﮔﻴﺮﯼ ﻋﺎﺩﺍﺕ ﻭ ﺭﻭﺵ‌ﻫﺎﯼ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻣﺴﺎﺋﻞ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺗﻤﺎﻡ ﻋﻤﺮ ﺍﺩﺍﻣﻪ ﺧﻮﺍﻫﻨﺪ ﺩﺍﺷﺖ. ﺭﻭﺍﺑﻄﯽ ﮐﻪ ﺑﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺭﻓﺘﻦ ﺍﻭ ﺑﻪ ﺩﺑﺴﺘﺎﻥ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌ﮐﻨﻴﺪ ﺗﻌﻴﻴﻦ ﮐﻨﻨﺪﻩ ﺭﻭﺍﺑﻂ ﺍﻭ ﺑﺎ ﺷﻤﺎ ﺩﺭ ﺗﻤﺎﻡ ﻃﻮﻝ ﻋﻤﺮﺵ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻣﺜﻼ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﺳﻨﻴﻦ ﺑﻠﻮﻍ ﺑﻪ ﺧﺎﻃﺮ ﺳﺘﻴﺰﻩ‌­ﺟﻮﺋﯽ، ﺗﺒﺪﻳﻞ ﺑﻪ  ﺑﺰﻩ­ﮐﺎﺭ ﺟﻮﺍﻧﯽ ﻧﺸﻮﺩ، ﻭﻗﺖ ﮐﺎﻓﯽ ﺑﮕﺬﺍﺭﻳﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺑﺎ ﺍﻭ ﻗﺒﻞ ﺍﺯ ﻭﺭﻭﺩﺵ ﺑﻪ ﺩﺑﺴﺘﺎﻥ، ﺭﺍﺑﻄﻪ ﻋﺎﻃﻔﯽ ﻣﺤﺘﺮﻣﺎﻧﮥ ﺩﻭﻃﺮﻓﻪ ﻭ ﻣﺴﺘﺤﮑﻤﯽ ﺑﺮﻗﺮﺍﺭ ﮐﻨﻴﺪ.
 
۳- ﺑﻪ ﺭﻏﻢ ﺗﻤﺎﻡ ﻧﮑﺎﺗﯽ ﮐﻪ ﻋﻠﻢ ﺩﺭﺑﺎﺭﻩ ﮐﻮﺩﮐﺎﻥ ﺑﻪ ﺷﻤﺎ ﺁﻣﻮﺧﺘﻪ ﺍﺳﺖ، ﺍﮔﺮ ﻧﺘﻮﺍﻧﻴﺪ ﻣﻌﻨﺎﯼ ﮐﻮﺩﮐﯽ ﺭﺍ ﺩﺭﮎ ﮐﻨﻴﺪ، ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﺪ ﺭﺍﻫﻨﻤﺎﯼ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺎﺷﻴﺪ. ﻣﺜﻞ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺍﺯ ﻳﮏ ﺗﺼﻨﻴﻒ ﺷﻌﺮﺵ ﺭﺍ ﺑﺪﺍﻧﻴﺪ ﻭﻟﯽ ﺁﻫﻨﮓ ﺁﻥ ﺭﺍ ﻧﺸﻨﺎﺳﻴﺪ. ﺑﺮﺍﯼ ﺩﺭﮎ ﮐﻮﺩﮐﯽ ﺑﺎﻳﺪ ﺑﺎ ﮐﻮﺩﮐﯽ ﮐﻪ ﺩﺭ ﮐﻨﻪ ﻭﺟﻮﺩﺗﺎﻥ ﺍﺳﺖ ﺩﻭﺑﺎﺭﻩ ﭘﻴﻮﻧﺪ ﺑﺮﻗﺮﺍﺭ ﮐﻨﻴﺪ. ﺍﻳﻦ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻫﻨﻤﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌­ﺗﻮﺍﻧﻴﺪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.
 
۴- ﻫﺮﻳﮏ ﺍﺯ ﻓﺮﺯﻧﺪﺍﻥ ﺷﻤﺎ ﺗﺮﮐﻴﺐ ﮊﻧﺘﻴﮑﯽ ﺧﺎﺻﯽ ﺩﺍﺭﺩ ﮐﻪ ﻧﻪ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﻭ ﺟﻮﺩ ﺩﺍﺷﺘﻪ ﻭ ﻧﻪ ﭘﺲ ﺍﺯ ﺍﻳﻦ ﻭﺟﻮﺩ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ. ﺑﻪ ﻋﻼﻭﻩ، ﺑﺮﺣﺴﺐ ﺍﻳﻦ‌ﮐﻪ ﺍﻭ ﻓﺮﺯﻧﺪ ﭼﻨﺪﻡ ﺧﺎﻧﻮﺍﺩﻩ ﺑﺎﺷﺪ، ﺍﺯ ﻧﻈﺮ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺩﺭ ﻣﺤﻴﻂ ﻣﺘﻔﺎﻭﺗﯽ ﺍﺯ ﺧﻮﺍﻫﺮ ﻭ ﺑﺮﺍﺩﺭﻫﺎﻳﺶ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻦ ﺗﺮﮐﻴﺐ ﮊﻧﺘﻴﮑﯽ ﺧﺎﺹ ﻭ ﺍﻳﻦ ﻣﺤﻴﻂ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺧﺎﺹ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﻪ ﻣﻌﻨﺎﯼ ﻭﺍﻗﻌﯽ ﮐﻠﻤﻪ ﺩﺭ ﻧﻮﻉ ﺧﻮﺩ ﺑﯽ‌­ﻧﻈﻴﺮ ﺍﺳﺖ، ﺩﺭﺳﺖ ﻣﺎﻧﻨﺪ ﺍﺛﺮ ﺍﻧﮕﺸﺘﺎﻥ ﺩﺳﺘﺶ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺭﻓﺘﺎﺭﺗﺎﻥ ﺑﺎ ﺍﻭ ﺑﺎﻳﺪ ﺑﺎ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﺍﻳﻦ ﺍﺻﻞ ﺑﺎﺷﺪ. ﭘﺲ ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﺍﻭ ﺭﺍ ﺑﺎ ﻳﮏ ﺍﻟﮕﻮﯼ ﺍﺯ ﭘﻴﺶ ﺳﺎﺧﺘﻪ ﻣﻘﺎﻳﺴﻪ ﮐﻨﻴﺪ ﻭ ﻓﮑﺮ ﻧﮑﻨﻴﺪ ﮐﻪ ﺍﻭ ﺑﺎﻳﺪ ﭼﻨﻴﻦ ﻭ ﭼﻨﺎﻥ ﺑﺎﺷﺪ: ﻣﻬﻢ‌ﺗﺮﻳﻦ ﮐﺎﺭﯼ ﻛﻪ ﻣﯽ‌­ﺗﻮﺍﻧﻴﺪ ﺑﮑﻨﻴﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﺷﺨﺼﻴﺖ ﺍﻭ ﮐﻪ ﺁﻣﺎﺩﻩ ﺗﺜﺒﻴﺖ ﺷﺪﻥ ﺍﺳﺖ ﻭ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺷﮑﻞ ﮔﻴﺮﯼ ﺁﻥ ﺭﺍ ﺩﺭ ﻭﺟﻮﺩﺵ ﺣﺲ ﻣﯽ‌ﮐﻨﺪ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﺩﻫﺪ
 
۵- ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺷﮑﻮﻓﺎﻳﻴﯽ ﮐﺎﻣﻞ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﺩ ﺩﺭ ﺑﻄﻦ ﺧﺎﻧﻮﺍﺩ‌‌ﻩ‌­ﺍﯼ ﻣﺴﺘﺤﮑﻢ ﻭ ﺍﺳﺘﻮﺍﺭ ﺑﺰﺭﮒ ﺷﻮﺩ ﮐﻪ ﺗﻮﺳﻂ ﺑﺰﺭﮔﺴﺎﻻﻥ ﺍﺩﺍﺭﻩ ﻣﯽ­‌ﺷﻮﺩ. ﺍﻭ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﻭﺍﻟﺪﻳﻨﯽ ﻗﻮﯼ ﻭ ﻣﻤﻠﻮ ﺍﺯ ﻋﺸﻖ ﺑﺮﺍﯼ ﺭﺍﻫﻨﻤﺎﺋﻴﺶ ﺩﺍﺭﺩ، ﻭ ﻧﻪ ﻭﺍﻟﺪﻳﻨﯽ ﺑﺮﺍﯼ ﻣﺠﺒﻮﺭ ﮐﺮﺩﻥ ﻭ ﻣﺴﺘﺒﺪﺍﻧﻪ ﺭﻓﺘﺎﺭ ﮐﺮﺩﻥ.
 
۶- ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﺭﻓﺘﺎﺭﺵ ﺭﺍﻫﻨﻤﺎﻳﯽ ﻣﯽ‌ﮐﻨﻴﺪ، ﺑﺎﻳﺪ ﺑﻴﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻭ ﻭ ﺍﻋﻤﺎﻟﺶ ﺗﻔﺎﻭﺕ ﻗﺎﻳﻞ ﺷﻮﻳﺪ. ﺑﻪ ﻃﻮﺭ ﻃﺒﻴﻌﯽ ﺑﺎﻳﺪ ﺍﺯ ﮐﻮﺩﮎ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺖ ﮐﻪ ﮐﻨﺘﺮﻝ ﺍﻋﻤﺎﻝ ﺧﻮﺩ ﺭﺍ ﺑﺮﺣﺴﺐ ﺳﻦ ﻭ ﻣﺮﺣﻠﻪ ﺭﺷﺪﺵ ﻳﺎﺩ ﺑﮕﻴﺮﺩ. ﺍﻣﺎ ﻧﺒﺎﻳﺪ ﺍﺯ ﮐﻮﺩﮎ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺖ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﺪ، ﺯﻳﺮﺍ ﺍﺣﺴﺎﺳﺎﺕ ﻣﺎﻧﻨﺪ ﺍﻓﮑﺎﺭ ﻓﯽ­‌ﺍ‌‌‌ﻟﺒﺪﺍﻫﻪ ﺩﺭ ﺍﻧﺴﺎﻥ ﻇﺎﻫﺮ ﻣﯽ‌­ﺷﻮﻧﺪ.
 
۷- ﻣﺤﺪﻭﺩﻳﺖ‌­ﻫﺎ ﻭ ﻗﻮﺍﻧﻴﻨﯽ ﮐﻪ ﺑﺮﺍﯼ ﺍﻋﻤﺎﻝ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺗﻌﻴﻴﻦ ﻣﯽ‌­ﮐﻨﻴﺪ، ﺑﺎﻳﺪ ﺑﻪ ﮔﻮﻧﻪ ﻗﺎﺑﻞ ﺍﺟﺮﺍﺋﯽ ﺑﺎﺷﻨﺪ، ﺳﭙﺲ ﺍﻳﻦ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎ ﻭ ﻗﻮﺍﻧﻴﻦ ﺭﺍ ﺑﻪ ﺻﻮﺭﺗﯽ ﻣﻨﻄﻘﯽ ﻭ ﻣﻨﺴﺠﻢ ﺍﺟﺮﺍ ﮐﻨﻴﺪ. ﻫﻴﭻ ﻧﻈﺎﻡ ﺟﺎﺩﻭﺋﯽ ﻗﻮﺍﻧﻴﻨﯽ ﮐﻪ ﺑﺮﺍﯼ ﺗﻤﺎﻣﯽ ﺧﺎﻧﻮﺍﺩﻩ‌ﻫﺎ ﺧﻮﺏ ﺑﺎﺷﺪ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﻫﻤﻪ ﭼﻴﺰ ﺑﻪ ﺍﻳﻦ ﺍﺻﻞ ﺑﺴﺘﮕﯽ ﺩﺍﺭﺩ ﮐﻪ: ﭼﻪ ﻗﻮﺍﻧﻴﻨﯽ ﺍﺯ ﻧﻈﺮ ﺷﻤﺎ ﻣﻬﻢ ﺍﺳﺖ ﻭ ﺍﻳﻨﮑﻪ ﭼﻪ ﻗﻮﺍﻧﻴﻨﯽ ﺑﻬﺘﺮ ﺍﺳﺖ ﺭﻋﺎﻳﺖ ﺷﻮﺩ؟
 
 ۸- ﺷﻤﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ، ﻣﺴﺌﻮﻟﻴﺖ ﺁﻣﻮﺯﺵ ﮐﻨﺘﺮﻝ ﺍﻋﻤﺎﻝ ﮐﻮﺩﮎ ﺭﺍ ﺑﻪ ﻋﻬﺪﻩ ﺩﺍﺭﻳﺪ ﻭﻟﯽ ﺍﺟﺎﺯﻩ ﺑﻴﺎﻥ ﺁﺯﺍﺩﺍﻧﻪ ﺍﺣﺴﺎﺳﺎﺗﺶ ﻧﻴﺰ ﺍﻫﻤﻴﺖ ﺑﻪ ﺳﺰﺍﻳﯽ ﺩﺍﺭﺩ. ﺑﺎ ﺍﻳﻦ ﮐﺎﺭﺗﺎﻥ ﺍﺻﻼ ﺍﺯ ﺍﺣﺘﺮﺍﻡ ﺍﻭ ﺑﻪ ﺷﻤﺎ ﮐﺎﺳﺘﻪ ﻧﺨﻮﺍﻫﺪ ﺷﺪ. ﺑﻠﮑﻪ ﺑﺮﻋﮑﺲ ﺁﻥ ﺭﺍ ﺑﻴﺸﺘﺮ ﺧﻮﺍﻫﺪ ﮐﺮﺩ. ﺯﻳﺮﺍ ﺍﻭ ﺩﺭﮎ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺪﺭ ﻳﺎ ﻣﺎﺩﺭ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺑﻪ ﺧﻮﺩ ﺍﻃﻤﻴﻨﺎﻥ ﺩﺍﺭﻳﺪ ﻭ ﺑﻪ ﺍﻭ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻋﻀﻮ ﺟﻮﺍﻥ ﺧﺎﻧﻮﺍﺩﻩ ﺑﻪ ﺻﻮﺭﺗﯽ ﺩﻣﮑﺮﺍﺗﻴﮏ، ﺍﺟﺎﺯﻩ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺗﺶ ﺭﺍ ﻣﯽ‌ﺩﻫﻴﺪ. ﺍﮔﺮ ﺍﻧﺴﺎﻥ ﺁﺯﺍﺩﻩ‌­ﺍﯼ ﻫﺴﺘﻴﺪ، ﺩﺭ ﺧﺎﻧﻮﺍﺩﻩ ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﺟﺎﻣﻌﻪ، ﻗﻮﺍﻧﻴﻦ ﺩﻣﮑﺮﺍﺗﻴﮏ ﺑﺎﻳﺪ ﺭﻋﺎﻳﺖ ﺷﻮﺩ.
 
۹- ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ ﺑﺮﺍﯼ ﺑﺮﻗﺮﺍﺭﯼ ﻭ ﺣﻔﻆ ﺭﻭﺍﺑﻄﯽ ﻣﺴﺘﺤﮑﻢ ﻭ ﺳﻮﺩﻣﻨﺪ ﺑﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺍﻭ ﺑﻔﻬﻤﺎﻧﻴﺪ ﺍﻭ ﺭﺍ ﺩﺭﮎ ﻣﯽ‌­ﮐﻨﻴﺪ. ﻧﻪ ﺍﻳﻨﮑﻪ ﺑﻪ ﺍﻭ ﺑﮕﻮﺋﻴﺪ: «ﻣﯽ ﻓﻬﻤﻢ ﮐﻪ ﭼﻪ ﺍﺣﺴﺎﺳﯽ ﺩﺍﺭﯼ»، ﻧﻪ، ﺑﻠﮑﻪ ﺑﻪ ﺻﻮﺭﺗﯽ ﻣﺘﻘﺎﻋﺪ ﮐﻨﻨﺪﻩ ﻣﯽ‌­ﺗﻮﺍﻧﻴﺪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺑﮑﻨﻴﺪ ﻭ ﺍﻳﻦ ﺍﻣﺮ ﺍﺯ ﻃﺮﻳﻖ ﺑﺎﺯﺗﺎﺏ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻭ ﺑﻪ ﺧﻮﺩﺵ ﺍﻧﺠﺎﻡ ﭘﺬﻳﺮﺍﺳﺖ. ﺍﻳﻦ ﺭﺍﻩ ﺑﺴﻴﺎﺭ ﺧﻮﺑﯽ ﺍﺳﺖ ﮐﻪ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺑﻪ ﺟﺎﯼ ﺍﻭ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺩﻧﻴﺎ ﺭﺍ ﺍﺯ ﭼﺸﻢ ﺍﻭ ﺑﻨﮕﺮﻳﺪ.
 
۱۰- ﺑﺎ ﻣﻨﻀﺒﻂ ﺑﺎﺭ ﺁﻭﺭﺩﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ، ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﮐﻨﻴﺪ ﺗﺎ ﺟﻮﺷﺶ­‌ﻫﺎﯼ ﺩﺭﻭﻧﯽ ﻣﺜﺒﺘﺶ ﺑﻪ ﺳﻮﯼ ﺍﻫﺪﺍﻑ ﺑﺎ ﺍﺭﺯﺷﯽ ﻫﺪﺍﻳﺖ ﺷﻮﻧﺪ. ﻫﺮﺭﻓﺘﺎﺭﯼ ﮐﻪ ﺗﺸﻮﻳﻖ ﺷﻮﺩ ﮔﺮﺍﻳﺶ ﺑﻪ ﺗﮑﺮﺍﺭ ﺩﺍﺭﺩ. ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﺯ ﺧﻮﺩ ﺍﺳﺘﻘﻼﻝ ﻧﺸﺎﻥ ﻣﯽ­‌ﺩﻫﺪ ﻭ ﻳﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺭﻭﺣﻴﻪ ﺧﻼﻕ ﻭ ﻣﻬﺮﺑﺎﻧﯽ ﺍﺯ ﺧﻮﺩ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﮐﻨﻴﺪ. ﻭﻗﺘﯽ ﮔﺮﻳﻪ ﻣﯽ­‌ﮐﻨﺪ، ﺧﺠﺎﻟﺘﯽ ﺍﺳﺖ ﻭ ﻳﺎ ﺧﺮﺍﺑﮑﺎﺭﯼ ﻣﯽ‌­ﮐﻨﺪ ﻭ ﺧﺸﻮﻧﺖ ﺑﻪ ﺧﺮﺝ ﻣﯽ ﺩﻫﺪ ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﻧﮑﻨﻴﺪ. ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﻣﯽ­‌ﺗﻮﺍﻧﻴﺪ ﮐﻤﮑﺶ ﮐﻨﻴﺪ ﺗﺎ ﺍﻧﺴﺎﻧﯽ ﻣﻨﻄﻘﯽ ﻭ ﻣﻨﻀﺒﻂ ﺷﻮﺩ.
 
۱۱- ﻣﺪﺭﺳﻪ ﺍﺯ ﺧﺎﻧﻪ ﺁﻏﺎﺯ ﻣﯽ‌ﺷﻮﺩ. «ﺑﺮﻧﺎﻣﻪ» ﻓﻌﺎﻟﻴﺖ­‌ﻫﺎﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﺧﺎﻧﻪ ﻫﻤﺎﻥ ﺍﻫﻤﻴﺘﯽ ﺭﺍ ﺩﺍﺭﺩ ﮐﻪ ﺑﺮﻧﺎﻣﻪ ﺩﺭﺳﯽ ﺩﺭ ﻣﺪﺭﺳﻪ. ﺍﻭ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ­‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﮐﺘﺎﺏ ﻭ ﺳﯽ‌­ﺩﯼ ﻫﺎﻳﯽ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎ ﺩﻗﺖ ﻫﺮ ﭼﻪ ﺑﻴﺸﺘﺮ ﺍﻧﺘﺨﺎﺏ ﺷﻮﻧﺪ، ﺯﻳﺮﺍ ﺍﻳﻦ‌ﻫﺎ ﻭﺳﺎﻳﻞ ﺁﻣﻮﺯﺷﯽ ﺍﻭ ﺩﺭ ﺧﺎﻧﻪ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ­‌ﺩﻫﻨﺪ.
 
۱۲- ﺑﺎﻳﺪ ﺭﺷﺪ ﺫﻫﻨﯽ ﮐﻮﺩﮎ ﺭﺍ ﺑﺮﺍﻧﮕﻴﺨﺖ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﻪ ﺣﺪﺍﮐﺜﺮ ﺭﺷﺪ ﻫﻮﺷﯽ ﺧﻮﺩ ﺑﺮﺳﺪ، ﺍﻧﮕﻴﺰﺵ­‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﺩﺭﻳﺎﻓﺖ ﻣﯽ‌­ﮐﻨﺪ ﺍﻫﻤﻴﺖ ﻓﻮﻕ ﺍﻟﻌﺎﺩﻩ‌­ﺍﯼ ﺩﺍﺭﺩ. ﺭﺷﺪ ﺯﺑﺎﻥ ﺭﺍ ﺩﺭ ﮐﻮﺩﮎ ﺗﻘﻮﻳﺖ ﮐﻨﻴﺪ، ﺗﻮﺟﻬﺶ ﺭﺍ ﺑﻪ ﮐﻠﻤﺎﺕ ﺑﺮﺍﻧﮕﻴﺰﻳﺪ، ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﺪ ﮐﻪ ﮐﺘﺎﺏ ﺭﺍ ﺩﻭﺳﺖ ﺑﺪﺍﺭﺩ، ﻧﻮﺷﺘﻦ ﺭﺍ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﺪ. ﺑﺎ ﺍﻭ ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﺑﮑﻨﻴﺪ ﮐﻪ ﺫﻫﻦ ﻣﻨﻄﻘﯽ ﻭ ﺭﻳﺎﺿﯽ ﺭﺍ ﺩﺭ ﺍﻭ ﺭﺷﺪ ﺩﻫﺪ. ﺍﻳﻦ‌ﻫﺎ ﻣﻴﺮﺍﺙ ﺫﻫﻨﯽ­‌ﺍ‌ﯼ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺮﺍﯼ ﺍﻭ ﺗﺎ ﺷﺶ ﺳﺎﻟﮕﯽ ﺑﮕﺬﺍﺭﻳﺪ.
