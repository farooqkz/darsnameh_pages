ﻫﻴﭻ ﻛﺪﺍﻡ ﺍﺯ ﻣﺎ ﺩﻟﻤﺎﻥ ﻧﻤﯽ‌ﺧﻮﺍﻫﺪ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺩﺭ ﺁﻳﻨﺪﻩ ﺗﺒﺪﻳﻞ ﺑﻪ ﺍﻓﺮﺍﺩﯼ ﺧﺸﻦ ﺷﻮﻧﺪ. ﭼﮕﻮﻧﻪ ﺟﻠﻮﯼ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺑﮕﻴﺮﻳﻢ؟
 
ﺍﻣﺮﻭﺯﻩ، ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺮﺩﻡ ﺩﺭﺑﺎﺭﻩ ﺧﺸﻮﻧﺖ ﺑﻪ‌ ﺻﻮﺭﺗﯽ ﻣﺒﻬﻢ ﻭ ﺑﺴﻴﺎﺭ ﻋﺎﻡ ﺻﺤﺒﺖ ﻣﯽ‌ﻛﻨﻨﺪ. ﺑﻪ ﻧﻈﺮ ﻣﻦ ﺧﺸﻮﻧﺖ، ﺧﺸﻢ ﻭ ﺍﺣﺴﺎﺱ ﺿﺪﻳﺘﯽ ﺍﺳﺖ ﻛﻪ ﺑﻪ ﺷﺪﻳﺪ‌ﺗﺮﻳﻦ ﺻﻮﺭﺕ ﻭ ﺗﺨﺮﻳﺐ‌ﻛﻨﻨﺪﻩ‌ﺗﺮﻳﻦ ﻭﺟﻪ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺁﻥ ﺭﺍ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺗﻌﺮﻳﻒ ﻣﯽ‌ﻛﻨﻢ: «ﺧﺸﻮﻧﺖ ﺧﺸﻤﯽ ﺍﺳﺖ ﻛﻪ ﺑﻪ‌ ﺻﻮﺭﺕ ﺍﻋﻤﺎﻝ ﺷﺪﻳﺪﯼ ﻛﻪ ﻫﺪﻓﺶ ﺯﺧﻤﯽ ﻛﺮﺩﻥ ﻭ ﻳﺎ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩﻥ ﺟﺴﻤﺎﻧﯽ ﻛﺴﯽ ﺑﺎﺷﺪ ﺑﻴﺎﻥ ﺷﻮﺩ.»
 
ﺑﺎﻳﺪ ﻣﻴﺎﻥ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﻛﻪ ﺩﻭ ﭼﻴﺰ ﻛﺎﻣﻼ ﻣﺘﻔﺎﻭﺗﻨﺪ ﺗﻤﺎﻳﺰ ﻗﺎﺋﻞ ﺷﺪ. ﻃﺒﻖ ﺗﻌﺮﻳﻔﯽ ﻛﻪ ﺍﺯ ﺧﺸﻮﻧﺖ ﺍﺭﺍﺋﻪ ﺷﺪ، ﺧﺸﻮﻧﺖ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻋﻤﺎﻝ ﺧﺼﻤﺎﻧﻪ ﺍﺳﺖ. ﻣﺎ ﺁﺭﺯﻭ ﺩﺍﺭﻳﻢ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺩﺭ ﺩﻭﺭﺍﻥ ﻛﻮﺩﻛﯽ ﻭ ﺯﻣﺎﻧﯽ ﻛﻪ ﺑﺰﺭﮒ ﺷﺪﻧﺪ ﻗﺎﺩﺭ ﺑﻪ ﻛﻨﺘﺮﻝ ﭼﻨﻴﻦ ﺍﻋﻤﺎﻟﯽ ﺑﺎﺷﻨﺪ.
 
ﺍﺣﺴﺎﺳﺎﺕ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ، ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ ﻭ ﻣﺎ ﻗﺎﺩﺭ ﺑﻪ ﻛﻨﺘﺮﻝ ﺁﻥ‌ﻫﺎ ﻧﻴﺴﺘﻴﻢ. ﺩﺭ ﺳﺮﺗﺎﺳﺮ ﺍﻳﻦ ﻧﻮﺷﺘﻪ ﺳﻌﯽ ﻛﺮﺩﻩ‌ﺍﻡ ﻧﺸﺎﻥ ﺩﻫﻢ ﻛﻪ ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﺑﻪ ﺧﻮﺩ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﻧﺪ: ﺧﻮﺷﺤﺎﻟﯽ ﻭ ﺧﺮﺳﻨﺪﯼ، ﻧﺎﺭﺍﺣﺘﯽ ﻭ ﺍﻓﺴﺮﺩﮔﯽ، ﻋﺸﻖ، ﺧﺸﻢ ﻭ ﺿﺪﻳﺖ. ﺍﺣﺴﺎﺳﺎﺕ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﻧﻴﺰ ﺑﺎ ﺩﻳﮕﺮ ﺍﺣﺴﺎﺳﺎﺕ ﺗﻔﺎﻭﺗﯽ ﻧﺪﺍﺭﻧﺪ.
 
ﺍﮔﺮ ﺑﺎ ﺧﻮﺩﻣﺎﻥ ﺭﻭ ﺭﺍﺳﺖ ﺑﺎﺷﻴﻢ ﻣﯽ‌ﺑﻴﻨﻴﻢ ﻛﻪ ﮔﻪ‌ﮔﺎﻩ ﭼﻨﻴﻦ ﺍﺣﺴﺎﺱ ﭘﺮﺧﺎﺷﮕﺮﯼ‌ﺍﯼ ﺩﺭ ﻣﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻫﺮ ﭼﻨﺪ ﻭﻗﺖ ﻳﮏ‌ﺑﺎﺭ ﺯﻥ ﻭ ﺷﻮﻫﺮ ﻧﺴﺒﺖ ﺑﻪ ﻫﻢ ﻋﺼﺒﺎﻧﻴﺖ ﺷﺪﻳﺪﯼ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﻛﻨﻨﺪ. ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﻛﻮﺩﻛﯽ ﻛﺎﺭﯼ ﻣﯽ‌ﻛﻨﺪ ﻛﻪ ﻣﺎ ﺭﺍ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﻛﻨﺪ، ﺧﺸﻤﻨﺎﮎ ﻣﯽ‌ﺷﻮﻳﻢ. ﻫﺮ ﻛﺴﯽ ﻛﻪ ﺑﮕﻮﻳﺪ ﻫﻴﭻ‌ﻭﻗﺖ ﭼﻨﻴﻦ ﺍﺣﺴﺎﺳﯽ ﺭﺍ ﻧﺴﺒﺖ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﻧﺪﺍﺷﺘﻪ ﺩﺭ ﺗﻮﻫﻢ ﺑﻪ ﺳﺮ ﻣﯽ‌ﺑﺮﺩ. ﺍﻳﻦ ﻋﻜﺲ‌ﺍﻟﻌﻤﻠﯽ ﻃﺒﻴﻌﯽ ﺍﺳﺖ. (ﺍﻟﺒﺘﻪ ﺍﮔﺮ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﻛﻪ ﺍﻛﺜﺮ ﻣﻮﺍﻗﻊ ﭼﻨﻴﻦ ﺍﺣﺴﺎﺳﯽ ﻧﺴﺒﺖ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺩﺍﺭﻳﺪ، ﺑﺎﻳﺪ ﺑﻪ ﺭﻭﺍﻧﺸﻨﺎﺱ ﻣﺮﺍﺟﻌﻪ ﻛﻨﻴﺪ ﺗﺎ ﺑﺮﺍﯼ ﺭﻓﻊ ﻣﺴﺎﺋﻞ ﻋﺎﻃﻔﯽ ﻭ ﺧﺎﻧﻮﺍﺩﮔﯽ‌ﺗﺎﻥ ﺑﻪ ﺷﻤﺎ ﻛﻤﮏ ﻛﻨﺪ.)
 
ﺍﻃﻔﺎﻝ ﻧﻴﺰ ﻋﻜﺲ‌ﺍﻟﻌﻤﻞ‌ﺷﺎﻥ ﻣﺸﺎﺑﻪ ﻣﺎﺳﺖ. ﻭ ﺍﻳﻦ ﻃﺒﻴﻌﯽ ﺍﺳﺖ. ﻛﻮﺩﻛﯽ ﻛﻪ ﻋﺼﺒﺎﻧﯽ ﺍﺳﺖ ﻭ ﺍﺯ ﻋﺼﺒﺎﻧﻴﺖ ﻓﺮﻳﺎﺩ ﻣﯽ‌ﻛﺸﺪ، ﻣﺴﻠﻤﺎ ﺧﺸﻮﻧﺖ ﺩﺭ ﺍﻭ ﺑﻴﺪﺍﺭ ﺷﺪﻩ ﺍﺳﺖ.
 
ﺗﻔﺎﻭﺕ ﻗﺎﺋﻞ ﺷﺪﻥ ﺑﻴﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﻭ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﺩﺍﺭﺩ. ﻣﺎ ﻧﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﻭ ﻧﻪ ﺍﺣﺴﺎﺳﺎﺕ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺭﺍ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﻛﻨﺘﺮﻝ ﻛﻨﻴﻢ، ﺩﺭ ﺻﻮﺭﺗﯽ‌ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺑﻴﺎﻣﻮﺯﻳﻢ ﻛﻪ ﺍﻋﻤﺎﻝ‌ﻣﺎﻥ ﺭﺍ ﻛﻨﺘﺮﻝ ﻛﻨﻴﻢ ﻭ ﻣﺎﻧﻊ ﺍﺯ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻥ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﺷﻮﻳﻢ.
 
ﻣﺎ ﻧﻤﯽ‌ﺧﻮﺍﻫﻴﻢ ﻓﻘﻂ ﺑﺮ ﺟﻨﺒﻪ‌ﻫﺎﯼ ﻣﻨﻔﯽ ﺍﻳﻦ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﻏﻠﺒﻪ ﻛﻨﻴﻢ. ﺑﻠﻜﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺟﻨﺒﻪ‌ﻫﺎﯼ ﻣﺜﺒﺖ ﻋﺸﻖ، ﻫﻤﻜﺎﺭﯼ ﻭ ﻋﻼﻗﻪ‌ﻣﻨﺪﯼ ﻭﺍﻗﻌﯽ ﺑﻪ ﺧﻮﺷﺒﺨﺘﯽ ﺩﻳﮕﺮﺍﻥ ﻭ ﺍﺣﺴﺎﺱ ﻫﻤﺪﺭﺩﯼ ﺑﺎ ﻛﺴﺎﻧﯽ ﻛﻪ ﺍﺯ ﻣﺴﺌﻠﻪ‌ﺍﯼ ﺭﻧﺞ ﻣﯽ‌ﺑﺮﻧﺪ ﺭﺍ ﺑﻴﺎﻣﻮﺯﻳﻢ. ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﻳﺎﺩ ﺑﮕﻴﺮﻧﺪ ﭼﮕﻮﻧﻪ ﺍﺯ ﻋﻘﻞ ﺳﻠﻴﻢ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻨﺪ ﻭ ﺑﺮﺍﯼ ﺣﻞ ﻣﺴﺎﻳﻞ ﺑﻴﻦ ﺍﻧﺴﺎﻥ‌ﻫﺎ ﺍﺯ ﮔﻔﺘﮕﻮ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻨﺪ.
 
ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ ﭼﮕﻮﻧﻪ ﺑﺎﻳﺪ ﻋﻤﻞ ﻛﻨﻴﻢ؟ ﭼﮕﻮﻧﻪ ﻣﺎﻧﻊ ﺑﺮﻭﺯ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺑﺮﺍﯼ ﺣﻞ ﻣﺴﺎﻳﻠﯽ ﺷﻮﻳﻢ ﮐﻪ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺑﺮﺍﻳﺸﺎﻥ ﭘﻴﺶ ﻣﯽ‌­ﺁﻳﺪ؟ ﻫﻴﭻ ﻣﻄﺎﻟﻌﻪ ﻋﻠﻤﯽ‌ﺍﯼ‌ ﻧﺸﺎﻥ ﻧﺪﺍﺩﻩ ﺍﺳﺖ ﻛﻪ ﺍﻧﺴﺎﻥ‌ﻫﺎ ﺑﻪ‌ﻃﻮﺭ ﻣﺎﺩﺭﺯﺍﺩﯼ ﺧﺸﻦ ﻫﺴﺘﻨﺪ. ﻣﺮﺩﻡ ﺧﺸﻮﻧﺖ ﺭﺍ ﻓﺮﺍ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﻫﻴﭻ‌ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﺑﻪ‌ ﻃﻮﺭ ﺁﮔﺎﻫﺎﻧﻪ ﺍﻳﻦ ﺧﺸﻮﻧﺖ ﺭﺍ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻧﻤﯽ‌ﺁﻣﻮﺯﺩ. ﻭﻟﯽ ﺑﻪ‌ ﻫﺮ‌ﺻﻮﺭﺕ ﻧﺘﻴﺠﻪ ﻳﻜﯽ ﺍﺳﺖ. ﻛﻮﺩﮎ ﺍﺯ ﻃﺮﻳﻖ ﺗﻘﻠﻴﺪ ﻭ ﻫﻤﺎﻧﻨﺪ‌ﺳﺎﺯﯼ ﻣﯽ‌ﺁﻣﻮﺯﺩ. ﻃﻔﻞ ﺁﻥ ﻛﺎﺭﯼ ﺭﺍ ﻣﯽ‌ﻛﻨﺪ ﻛﻪ ﻣﯽ‌ﺑﻴﻨﺪ!
 
ﺣﺎﻻ ﺩﺭ ﺍﻳﻦ ﻓﻜﺮﻡ ﻛﻪ ﻣﺎﺩﺭﯼ ﻛﻪ ﻛﻤﯽ ﭘﻴﺶ ﺧﺸﻤﻨﺎﮎ ﺷﺪﻩ ﻭ ﺍﻛﻨﻮﻥ ﺍﻳﻦ ﺳﻄﻮﺭ ﺭﺍ ﻣﯽ‌ﺧﻮﺍﻧﺪ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﺪ: «ﻭﺍﯼ، ﺧﺪﺍﯼ ﻣﻦ، ﺷﺎﻳﺪ ﻣﻦ ﺁﺩﻡ ﺧﺸﻨﯽ ﻫﺴﺘﻢ! ﻭ ﺷﺎﻳﺪ ﺍﺛﺮ ﻭﺣﺸﺘﻨﺎﻛﯽ ﺭﻭﯼ ﻓﺮﺯﻧﺪﻡ ﮔﺬﺍﺷﺘﻪ‌ﺍﻡ!» ﺍﮔﺮ ﭼﻨﻴﻦ ﺩﻏﺪﻏﻪ‌ ﺧﺎﻃﺮﯼ ﺩﺍﺭﻳﺪ، ﻧﮕﺮﺍﻥ ﻧﺒﺎﺷﻴﺪ. ﺍﻳﻦ‌ﻛﻪ ﮔﻪ‌ﮔﺎﻩ ﺧﺸﻤﻨﺎﮎ ﺷﻮﻳﺪ ﻭ ﻳﺎ ﭼﻴﺰﯼ ﺭﺍ ﭘﺮﺗﺎﺏ ﻛﻨﻴﺪ، ﺩﻟﻴﻞ ﺑﺮ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﺷﺨﺺ ﺧﻄﺮﻧﺎﮐﯽ ﻫﺴﺘﻴﺪ.

ﻣﻦ ﺣﺘﯽ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎﻳﯽ ﺭﺍ ﺩﻳﺪﻩ‌ﺍﻡ ﻛﻪ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻨﺪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻛﺘﮏ ﺑﺰﻧﻨﺪ. ﺍﮔﺮ ﻛﻮﺩﮎ ﺭﺍ ﺍﺯ ﭼﻴﺰﯼ ﻣﻨﻊ ﻣﯽ‌ﻛﺮﺩﻧﺪ، ﺍﻭ ﺧﺸﻤﻨﺎﮎ ﺷﺪﻩ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻣﯽ‌ﺯﺩ ﻭ ﻳﺎ ﺑﻪ ﺁﻥ‌ﻫﺎ ﻟﮕﺪ ﻣﯽ‌ﺯﺩ. ﺁﻥ‌ﻭﻗﺖ ﻣﺎﺩﺭ ﻣﯽ‌ﮔﻔﺖ: ﺧﻴﻠﯽ ﺑﭽﻪ ﺑﺪﯼ ﻫﺴﺘﯽ، ﻭﻟﯽ ﺟﻠﻮﯼ ﻛﺘﮏ ﺯﺩﻥ ﺍﻭ ﺭﺍ ﻧﻤﯽ‌ﮔﺮﻓﺖ. ﺩﺭ ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ ﻏﺎﻟﺒﺎ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﺑﻪ «ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻛﻮﺩﻙ» ﻓﺮﺿﯽ‌ﺍﯼ ﺗﻤﺴﮏ ﻣﯽ‌ﺟﻮﻳﻨﺪ ﺗﺎ ﺗﺮﺱ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﻧﻪ ﮔﻔﺘﻦ ﺑﻪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻭ ﺗﻌﻴﻴﻦ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﯼ ﻣﺸﺨﺼﯽ ﺑﺮﺍﯼ ﺍﻭ ﭘﻨﻬﺎﻥ ﻛﻨﻨﺪ. ﺁﻥ‌ﻫﺎ ﻏﺎﻟﺒﺎ ﺍﺩﻋﺎ ﻣﯽ‌ﻛﻨﻨﺪ ﻛﻪ ﻧﻤﯽ‌ﺧﻮﺍﻫﻨﺪ ﺍﻭ ﺭﺍ ﺍﺯ ﭼﻴﺰﯼ «ﻣﻨﻊ» ﻛﻨﻨﺪ.
 
ﺑﺎ ﺍﻳﻦ ﻫﻤﻪ ﺑﺎﻳﺪ ﺑﻌﻀﯽ ﺍﺯ ﺍﻋﻤﺎﻝ ﻛﻮﺩﮎ ﺭﺍ «ﻣﻨﻊ» ﻛﺮﺩ! ﻣﺜﻼ ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﺩﻭﺳﺘﺎﻥ ﻭ ﻳﺎ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﺭﺍ ﻛﺘﮏ ﻣﯽ‌ﺯﻧﺪ. ﻧﺒﺎﻳﺪ ﮔﺬﺍﺷﺖ ﻛﻮﺩﻛﺎﻥ ﺑﺎ ﺣﻴﻮﺍﻧﺎﺕ ﺳﺘﻤﮕﺮﺍﻧﻪ ﺭﻓﺘﺎﺭ ﻛﻨﻨﺪ ﻳﺎ ﺧﺮﺍﺑﻜﺎﺭﯼ ﻛﻨﻨﺪ. ﻧﺒﺎﻳﺪ ﮔﺬﺍﺷﺖ ﻛﻪ ﺍﺯ ﻛﻴﻒ ﻣﺎﺩﺭﺷﺎﻥ ﭘﻮﻝ ﺑﺮﺩﺍﺭﻧﺪ ﻭ ﺍﺯ ﻓﺮﻭﺷﮕﺎﻩ‌ﻫﺎ ﭼﻴﺰﯼ ﻙِﺵ ﺑﺮﻭﻧﺪ. ﺍﮔﺮ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﺩﺭ ﻣﻘﺎﺑﻞ ﭼﻨﻴﻦ ﺍﻋﻤﺎﻝ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻫﻴﭻ‌ﮔﻮﻧﻪ ﻣﻤﺎﻧﻌﺘﯽ ﺑﻪ ﻋﻤﻞ ﻧﻴﺎﻭﺭﻧﺪ، ﺍﺷﺘﺒﺎﻩ ﺑﺴﻴﺎﺭ ﺑﺰﺭﮔﯽ ﻣﺮﺗﻜﺐ ﻣﯽ‌ﺷﻮﻧﺪ. ﻭﻟﯽ ﺑﺎﻳﺪ ﺍﺯ ﺻﻤﻴﻢ ﻗﻠﺐ ﺁﺭﻭﺯ ﻛﻨﻴﻢ ﻛﻪ ﺩﺭ ﻣﻮﺭﺩ ﺍﺣﺴﺎﺳﺎﺕ ﺁﻥ‌ﻫﺎ ﻫﻴﭻ‌ﮔﻮﻧﻪ ﻣﻤﺎﻧﻌﺘﯽ ﺑﻪ ﻋﻤﻞ ﻧﻴﺎﻭﺭﻳﻢ. ﻭ ﺍﻳﻦ ﭼﻴﺰﯼ ﺍﺳﺖ ﻛﻪ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﻣﯽ‌ﭘﺴﻨﺪﺩ ﻭ ﻋﻘﻞ ﺳﻠﻴﻢ ﺣﻜﻢ ﻣﯽ‌ﻛﻨﺪ.
 
ﺍﮔﺮ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﻣﻘﺎﺑﻞ ﺗﻈﺎﻫﺮﺍﺕ ﺟﺴﻤﺎﻧﯽ ﺿﺪﻳﺖ ﻭ ﻣﺨﺎﻟﻔﺖ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﯼ ﻣﺸﺨﺼﯽ ﺗﻌﻴﻴﻦ ﻧﻜﻨﻨﺪ، ﻛﻮﺩﮎ ﻗﺎﺩﺭ ﺑﻪ ﻛﻨﺘﺮﻝ ﺍﻋﻤﺎﻝ ﻧﺎﺷﯽ ﺍﺯ ﻫﻴﺠﺎﻧﺎﺗﺶ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ. ﺩﺭ ﺍﻳﻦ ﺑﺎﺭﻩ ﺍﺷﺘﺒﺎﻩ ﻧﻜﻨﻴﺪ.
 
ﺑﺮﺍﯼ ﮐﻮﺩﮎ ﺣﻴﺎﺗﯽ ﺍﺳﺖ  ﮐﻪ ﺍﻳﻦ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﺍﻋﻤﺎﻝ ﺧﺸﻦ ﺟﺴﻤﺎﻧﯽ ﺩﺭﮎ ﻛﻨﺪ. ﺍﻭ ﺑﻴﻢ ﺩﺍﺭﺩ ﻛﻪ ﻛﻨﺘﺮﻝ ﺍﻋﻤﺎﻝ ﻧﺎﺷﯽ ﺍﺯ ﻫﻴﺠﺎﻧﺎﺗﺶ ﺍﺯ ﺩﺳﺘﺶ ﺩﺭﺭﻓﺘﻪ ﻭ ﺩﺭ ﺁﻥ ﻏﺮﻕ ﮔﺮﺩﺩ. ﻭﻗﺘﯽ ﺍﺯ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﯼ ﻣﺸﺨﺺ ﻭ ﺍﻧﻌﻄﺎﻑ‌ﻧﺎﭘﺬﻳﺮ ﺻﺤﺒﺖ ﻣﯽ‌ﻛﻨﻴﻢ، ﻣﻨﻈﻮﺭ ﺍﻳﻦ ﻧﻴﺴﺖ ﻛﻪ ﺣﺘﻤﺎ ﻛﻮﺩﮎ ﺭﺍ ﺑﺎﻳﺪ ﺗﻨﺒﻴﻪ ﻛﺮﺩ. ﻛﺘﮏ ﺯﺩﻥ ﻛﻮﺩﻛﯽ ﻛﻪ ﻋﻤﻠﯽ ﺧﺼﻤﺎﻧﻪ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﺪ، ﺑﻪ ﻣﻌﻨﺎﯼ ﺟﻮﺍﺏ ﺩﺍﺩﻥ ﺧﺸﻮﻧﺖ ﺑﺎ ﺧﺸﻮﻧﺖ ﺍﺳﺖ (ﺁﻥ ﺗﺼﻮﻳﺮ ﻓﻜﺎﻫﯽ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﺁﻭﺭﻳﺪ ﻛﻪ ﭘﺪﺭﯼ ﺩﺭ ﺣﺎﻟﯽ ﻛﻪ ﺩﺭ ﺣﺎﻝ ﺗﻨﺒﻴﻪ ﭘﺴﺮﺵ ﺑﻮﺩ ﺑﻪ ﺍﻭ ﻣﯽ‌ﮔﻔﺖ: «ﺍﻳﻦ‌ﻃﻮﺭﯼ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﯼ ﻛﻪ ﻣﺮﺩﻡ‌ ﺭﻭ ﭼﻄﻮﺭ ﺑﺰﻧﯽ!») ﺑﺎ ﺍﻳﻦ‌ ﻫﻤﻪ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﻪ ﻛﻮﺩﻛﯽ ﻛﻪ ﻳﮏ ﺧﺮﺍﺑﻜﺎﺭﯼ ﻛﺮﺩﻩ ﻭ ﻳﺎ ﻛﺴﯽ ﺭﺍ ﻛﺘﮏ ﺯﺩﻩ ﺍﺳﺖ ﻳﮏ ﺩﺭﻟﻨﺒﺮﯼ ﺑﺰﻧﻴﻢ ﺗﺎ ﺍﻳﻨﻜﻪ ﺳﺮﻣﺎﻥ ﺭﺍ ﺑﺮﮔﺮﺩﺍﻧﻴﻢ ﻭ ﺑﻪ ﺭﻭﯼ ﺧﻮﺩﻣﺎﻥ ﻧﻴﺎﻭﺭﻳﻢ ﻛﻪ ﺍﻭ ﻋﻤﻠﯽ ﺿﺪ ﺍﺟﺘﻤﺎﻋﯽ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻩ ﺍﺳﺖ. ﻭﻟﯽ ﺭﺍﻩ‌ﻫﺎﯼ ﺑﻬﺘﺮﯼ ﻧﻴﺰ ﺑﺮﺍﯼ ﺗﺴﻠﻂ ﺑﺮ ﺍﻭﺿﺎﻉ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.
