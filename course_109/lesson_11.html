ﺩﺭﺱ ﺩﻭﻡ- ﺧﻮﺩﺍﻧﮕﺎﺭﻩ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﺩﻳﺪﻳﻢ ﮐﻪ ﺑﻌﻀﯽ ﺍﺯ ﺍﺻﻮﻝ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺑﻴﺖ ﻫﻢ ﺑﺮﺍﯼ ﺍﻧﺴﺎﻥ‌ﻫﺎ ﻭ ﻫﻢ ﺑﺮﺍﯼ ﺣﻴﻮﺍﻧﺎﺕ ﻗﺎﺑﻞ ﭘﻴﺎﺩﻩ ﻛﺮﺩﻥ ﺍﺳﺖ. ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﺸﻨﺎﺳﻴﻢ ﻭ ﺑﻪ ﺻﻮﺭﺗﯽ ﻫﻮﺷﻴﺎﺭﺍﻧﻪ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻴﻢ. ﻭﻟﯽ ﺍﺻﻮﻝ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺑﻴﺘﯽ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻛﻪ ﻣﺨﺘﺺ ﺍﻧﺴﺎﻥ‌ﻫﺎﺳﺖ ﻭ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﺑﺎﻳﺪ ﺣﺘﻤﺎ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﺸﻨﺎﺳﻨﺪ. ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺧﻮﺍﻫﻨﺪ ﻛﺮﺩ ﻛﻪ ﻣﺤﻜﻮﻡ ﺑﻪ ﺷﻜﺴﺖ ﺑﻮﺩﻩ ﻭ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺟﺰ ﺩﻟﺴﺮﺩﯼ ﭼﻪ ﺑﺮﺍﯼ ﺧﻮﺩﺷﺎﻥ ﻭ ﭼﻪ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺩﺭ ﺑﺮﻧﺪﺍﺭﺩ.                                                                               
ﺍﮔﺮ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺑﻴﺘﯽ ﻛﻪ ﺗﺠﺎﺭﺏ ﻋﻠﻤﯽ ﺗﺎﺛﻴﺮ ﺧﻮﺑﺸﺎﻥ ﺭﺍ ﻧﺸﺎﻥ ﺩﺍﺩﻩ­‌ﺍﻧﺪ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻴﻢ، ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻥ‌ﻣﺎﻥ ﺍﺯ ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ ﺗﺎ ﺯﻣﺎﻥ ﺑﻠﻮﻍ ﺑﻴﺎﻣﻮﺯﻳﻢ ﭼﮕﻮﻧﻪ ﺧﻮﺩ ﺭﺍ ﺍﺩﺍﺭﻩ ﻛﻨﻨﺪ ﻭ ﺩﻳﮕﺮ ﻣﺠﺒﻮﺭ ﻧﺨﻮﺍﻫﻴﻢ ﺑﻮﺩ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺭﺍ ﺍﺯ ﺧﺎﺭﺝ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺗﺤﻤﻴﻞ ﻛﻨﻴﻢ. ﺭﻭﺵ­‌ﻫﺎﯼ ﺁﻣﻮﺯﺵ ﺩﻳﺴﻴﭙﻠﻴﻨﯽ ﺭﺍ ﮐﻪ ﻫﻢ ﺑﺮﺍﯼ ﺣﻴﻮﺍﻧﺎﺕ ﻭ ﻫﻢ ﺑﺮﺍﯼ ﺍﻧﺴﺎﻥ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ‌­ﮔﻴﺮﻧﺪ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ. ﺍﮐﻨﻮﻥ ﺭﻭﺵ‌ﻫﺎﯼ ﺁﻣﻮﺯﺵ ﺩﻳﺴﻴﭙﻠﻴﻨﯽ ﺭﺍ ﺑﺮﺭﺳﯽ ﻣﯽ‌ﻛﻨﻴﻢ ﻛﻪ ﺧﺎﺹ ﺍﻃﻔﺎﻝ ﺑﻮﺩﻩ ﻭ ﺩﺭ ﻣﻮﺭﺩ ﺣﻴﻮﺍﻧﺎﺗﯽ ﭼﻮﻥ ﺳﮓ، ﺩﻟﻔﻴﻦ ﻳﺎ ﻃﻮﻃﯽ ﻗﺎﺑﻞ ‌ﺍﺟﺮﺍ ﻧﻴﺴﺖ. ﻧﻘﻄﻪ ﻣﺸﺘﺮﮎ ﺭﻭﺵ‌ﻫﺎﯼ ﺍﺧﻴﺮ ﺗﻘﻮﻳﺖ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻛﻮﺩﮎ ﺍﺳﺖ. ﺳﮓ ﻳﺎ ﺩﻟﻔﻴﻦ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻧﺪﺍﺭﻧﺪ ﺩﺭ ﺻﻮﺭﺗﯽ ﻛﻪ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺩﺍﺭﺩ.
 
ﺧﻮﺩﺍﻧﮕﺎﺭﻩ، ﺗﺼﻮﻳﺮﯼ ﺍﺳﺖ ﻛﻪ ﮐﻮﺩﮎ ﺍﺯ ﺧﻮﺩ ﻭ ﺗﻮﺍﻧﺎﻳﯽ‌ﻫﺎﻳﺶ ﺩﺍﺭﺩ. ﺍﮔﺮ ﺍﻭ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌ﺍﯼ ﻗﻮﯼ ﻭ ﻣﺜﺒﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻧﻪ ﺗﻨﻬﺎ ﺩﺭ ﻣﺪﺭﺳﻪ ﺑﻠﻜﻪ ﺩﺭ ﺳﺮﺗﺎﺳﺮ ﺯﻧﺪﮔﯽ‌ﺍﺵ ﻣﻮﻓﻖ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﻛﻪ ﻫﺪﻑ ﻣﺎ ﺍﺯ ﺁﻣﻮﺧﺘﻦ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺑﻪ ﻛﻮﺩﮎ ﺗﺒﺪﻳﻞ ﺍﻭ ﺑﻪ ﻓﺮﺩﯼ ﺍﺳﺖ ﻛﻪ ﻗﺎﺩﺭ ﺑﺮ ﺗﺴﻠﻂ ﺑﺮ ﺧﻮﺩ ﺑﺎﺷﺪ. ﻫﺮ ﭼﻘﺪﺭ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﺍﻭ ﻗﻮﯼ‌ﺗﺮ ﺑﺎﺷﺪ، ﻫﻤﺎﻥ‌ﻗﺪﺭ ﺗﺴﻠﻂ ﺑﺮ ﺧﻮﺩﺵ ﺑﻴﺸﺘﺮ ﺧﻮﺍﻫﺪ ﺷﺪ.
 
ﻫﺮ ﻗﺪﺭ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﻛﻮﺩﮎ ﺩﺭ ﻣﻴﺎﻥ ﺁﻥ‌ﻫﺎ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﻛﻨﺪ ﺳﻨﺠﻴﺪﻩ‌ﺗﺮ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ، ﻣﺴﺎﺋﻠﯽ ﻛﻪ ﺑﺮﺍﯼ ﺁﻣﻮﺯﺵ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺑﺎ ﺁﻥ­‌ﻫﺎ ﺭﻭﺑﻪ‌ﺭﻭ ﻣﯽ‌ﺷﻮﻳﺪ ﻛﻤﺘﺮ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ.
ﻛﻨﺘﺮﻝ ﻭ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﻣﺤﻴﻂ ﺯﻧﺪﮔﯽ ﻛﻮﺩﮎ ﻛﻤﺘﺮ ﺍﺯ ﻫﺮ ﭼﻴﺰﯼ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻭﺍﻟﺪﻳﻦ ﺍﺳﺖ. ﻧﮕﺎﻫﯽ ﺑﻪ ﺩﻭﺭ ﻭ ﺑﺮﺗﺎﻥ ﺩﺭ ﺧﺎﻧﻪ ﺑﻴﻨﺪﺍﺯﻳﺪ. ﺁﻳﺎ ﺧﺎﻧﻪ ﺷﻤﺎ ﺍﺯ ﺁﻥ ﺧﺎﻧﻪ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﻛﻪ ﭼﻴﺰ ﺟﺎﻟﺒﯽ ﺑﺮﺍﯼ ﺗﻔﺮﻳﺢ ﻛﻮﺩﻛﺎﻥ ﺩﺭ ﺁﻥ ﭘﻴﺪﺍ ﻧﻤﯽ‌ﺷﻮﺩ ﻭ ﻣﻤﻠﻮ ﺍﺯ ﺍﺷﻴﺎﯼ ﻣﺨﺼﻮﺹ «ﺑﺰﺭﮒ‌ﺗﺮﻫﺎﺳﺖ» ﮐﻪ ﻧﺒﺎﻳﺪ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺩﺳﺖ ﺯﺩ؟ ﺍﮔﺮ ﺍﻳﻦ ﻃﻮﺭ ﺑﺎﺷﺪ، ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﺯﻳﺎﺩ ﻣﺴﺎﺋﻞ ﺯﻳﺎﺩ ﻭ ﺑﻴﻬﻮﺩﻩ‌ﺍﯼ ﺟﻬﺖ ﺁﻣﻮﺯﺵ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﻭﻟﯽ ﺍﮔﺮ ﺑﺮﻋﻜﺲ ﺧﺎﻧﻪ‌ﺗﺎﻥ ﻭ ﺍﮔﺮ ﺣﻴﺎﻃﯽ ﺩﺍﺭﻳﺪ ﺣﻴﺎﻁ ﺧﺎﻧﻪ‌ﺗﺎﻥ ﺍﻣﻜﺎﻧﺎﺕ ﺟﺎﻟﺐ ﻭ ﺍﻧﮕﻴﺰﺵ‌ﻫﺎ ﻻﺯﻡ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ، ﻣﺴﺌﻠﻪ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﺷﻤﺎ ﭘﻴﺶ ﻧﺨﻮﺍﻫﺪ ﺁﻣﺪ.
ﻣﺜﻼ ﺑﻪ ﻣﺴﻴﺮﻫﺎﯼ ﻃﻮﻻﻧﯽ ﻛﻪ ﺑﺎ ﻣﺎﺷﻴﻦ ﻃﯽ ﻣﯽ‌ﻛﻨﻴﺪ ﻓﻜﺮ ﻛﻨﻴﺪ. ﭘﺪﺭ  ﻭ ﻣﺎﺩﺭﻫﺎﻳﯽ ﺭﺍ ﻣﯽ‌ﺷﻨﺎﺳﻢ ﻛﻪ ﭼﻨﻴﻦ ﻣﺴﺎﻓﺮﺕ‌ﻫﺎﻳﯽ ﺑﺮﺍﻳﺸﺎﻥ ﻳﮏ ﻛﺎﺑﻮﺱ ﺍﺳﺖ، ﺯﻳﺮﺍ ﻧﻤﯽ‌ﺩﺍﻧﻨﺪ ﭼﮕﻮﻧﻪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺭﺍ ﺩﺭ ﺍﻳﻦ ﻣﺴﺎﻓﺮﺕ‌­ﻫﺎ‌ ﺳﺮﮔﺮﻡ ﻛﻨﻨﺪ. ﺁﻥ‌ﻫﺎ ﺑﻪ ﻓﻜﺮﺷﺎﻥ ﻧﻤﯽ‌ﺭﺳﺪ ﻛﻪ ﺑﻪ ﻫﻤﺮﺍﻩ ﺧﻮﺩ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺑﺮﺩﺍﺭﻧﺪ ﻳﺎ ﻫﺪﻳﻪ‌ﻫﺎﯼ ﻛﻮﭼﻜﯽ ﻫﻤﺮﺍﻩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ ﻛﻪ ﻭﻗﺘﯽ ﺟﻨﮓ ﻭ ﺩﻋﻮﺍﯼ ﺧﻮﺍﻫﺮ ﻭ ﺑﺮﺍﺩﺭﻫﺎ ﺑﺎﻻ ﮔﺮﻓﺖ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺑﺪﻫﻨﺪ. ﺩﺭ ﺑﻴﻦ ﺭﺍﻩ ﺗﻮﻗﻒ‌ﻫﺎﯼ ﻛﻮﺗﺎﻩ‌ ﻧﻤﯽ‌ﻛﻨﻨﺪ ﺗﺎ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺑﺘﻮﺍﻧﻨﺪ ﭼﻨﺪ ﺩﻗﻴﻘﻪ‌ﺍﯼ ﺍﺯ ﻣﺎﺷﻴﻦ ﭘﻴﺎﺩﻩ ﺷﻮﻧﺪ ﻭ ﺑﺪﻭﻧﺪ. ﺁﻥ ﻭﻗﺖ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﻨﺪ ﭼﺮﺍ ﺑﭽﻪ‌ﻫﺎ ﺑﺎ ﻫﻢ ﺩﻋﻮﺍ ﻣﯽ‌­ﮐﻨﻨﺪ، ﮔﺮﻳﻪ ﻣﯽ‌ﻛﻨﻨﺪ ﻭ ﺳﻔﺮ ﺭﺍ ﻏﻴﺮﻗﺎﺑﻞ‌ﺗﺤﻤﻞ ﻣﯽ‌ﻛﻨﻨﺪ. ﺍﻧﺪﻛﯽ ﻛﻨﺘﺮﻝ ﻭ ﻛﻤﯽ ﺗﺎﻣﻞ ﺩﺭ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﺳﻔﺮﺗﺎﻥ ﻣﻮﺟﺐ ﻣﯽ‌ﺷﻮﺩ ﺩﻳﮕﺮ ﺗﻤﺎﻡ ﺍﻧﺮﮊﯼ‌ﺗﺎﻥ ﺭﺍ ﺩﺭ ﻃﻮﻝ ﺳﻔﺮ ﺻﺮﻑ ﺁﺭﺍﻡ ﻛﺮﺩﻥ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﻧﻜﻨﻴﺪ.
 
ﺭﺍﺑﻄﻪ‌ ﺷﺨﺼﯽ‌ﺍﯼ ﻛﻪ ﺑﺎ ﻫﺮ ﻳﮏ ﺍﺯ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌ﻛﻨﻴﺪ، ﺭﺷﺪ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌ﺍﯼ ﻗﻮﯼ ﺭﺍ ﺩﺭ ﺍﻭ ﺗﻘﻮﻳﺖ ﻣﯽ‌ﻛﻨﺪ.
ﺑﻪ ﻃﻮﺭ ﻧﻈﺮﯼ ﻭﺍﻟﺪﻳﻦ ﻣﯽ‌ﺩﺍﻧﻨﺪ ﻛﻪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻣﻨﺤﺼﺮﺑﻪ‌ﻓﺮﺩ ﻫﺴﺘﻨﺪ، ﻭﻟﯽ ﺩﺭ ﻋﻤﻞ ﺳﻌﯽ ﻣﯽ‌ﻛﻨﻨﺪ ﻛﻪ ﺭﻭﺵ ﺩﻳﺴﻴﭙﻠﻴﻦ ﻣﺸﺘﺮﮐﯽ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﺗﻤﺎﻡ ﺁﻥ‌ﻫﺎ ﺑﻪ ﻛﺎﺭ ﮔﻴﺮﻧﺪ، ﮔﻮﻳﯽ ﻫﻤﻪ ﺁﻥ‌ﻫﺎ ﻳﻜﺴﺎﻥ ﻫﺴﺘﻨﺪ. ﺩﺭ ﺣﺎﻟﯽ ﻛﻪ ﺑﺪﻳﻬﯽ ﺍﺳﺖ ﻛﻪ ﺁﻥ‌ﻫﺎ ﻳﻜﺴﺎﻥ ﻧﻴﺴﺘﻨﺪ. ﺍﺯ ﻧﻈﺮ ﺑﻴﻮﻟﻮﮊﻳﻜﯽ، ﻣﻤﻜﻦ ﺍﺳﺖ ﻛﻪ ﻳﻜﯽ ﺍﺯ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺍﺯ ﺑﻘﻴﻪ ﺳﺮﺳﺨﺖ‌ﺗﺮ ﻳﺎ ﺁﺭﺍﻡ‌ﺗﺮ ﺑﺎﺷﺪ.
ﺣﺘﯽ ﺩﺭ ﻳﮏ ﺧﺎﻧﻮﺍﺩﻩ، ﻧﺴﺒﺖ ﺑﻪ ﺍﻳﻦ ﻛﻪ ﻛﻮﺩﮎ ﻓﺮﺯﻧﺪ ﺍﻭﻝ، ﺩﻭﻡ ﻳﺎ ﺳﻮﻡ ﻳﺎ ... ﺑﺎﺷﺪ ﻣﺤﻴﻄﯽ ﻛﻪ ﺩﺭ ﺁﻥ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﺩ ﻧﺴﺒﺖ ﺑﻪ ﺧﻮﺍﻫﺮ ﺑﺮﺍﺩﺭﻫﺎﻳﺶ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﻓﺮﺯﻧﺪ ﺍﻭﻝ ﺩﺭ ﻣﺤﻴﻄﯽ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﺩ ﻛﻪ ﺗﺎ ﺯﻣﺎﻥ ﺗﻮﻟﺪ ﻓﺮﺯﻧﺪ ﺩﻭﻡ ﻓﻘﻂ ﺍﻓﺮﺍﺩ ﺑﺰﺭﮒ ﺩﻭﺭ ﻭ ﺑﺮﺵ ﺭﺍ ﮔﺮﻓﺘﻪ‌ﺍﻧﺪ؛ ﺍﻭ ﺑﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ ﭼﮕﻮﻧﮕﯽ ﺑﺰﺭﮒ ﻛﺮﺩﻥ ﻛﻮﺩﮎ ﺭﺍ ﻣﯽ‌ﺁﻣﻮﺯﺩ (ﻋﻠﺖ ﺍﻳﻨﻜﻪ ﺍﻛﺜﺮ ﺑﻴﻤﺎﺭﺍﻥِ ﺭﻭﺍﻧﺸﻨﺎﺳﺎﻥ ﺩﺭ ﺭﻭﺍﻥ‌­ﺩﺭﻣﺎﻧﯽ ﻓﺮﺯﻧﺪﻫﺎﯼ ﺍﻭﻝ ﺧﺎﻧﻮﺍﺩﻩ ﻫﺴﺘﻨﺪ ﻧﻴﺰ ﻫﻤﻴﻦ ﺍﺳﺖ.) ﺑﭽﻪ ﺩﻭﻡ ﻫﻤﻴﺸﻪ ﻳﻜﯽ ﻗﺒﻞ ﺍﺯ ﺧﻮﺩﺵ ﺭﺍ ﭘﻴﺶ ﺭﻭ ﺩﺍﺭﺩ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺍﻭ ﻧﮕﺎﻩ ﻛﻨﺪ، ﺍﺯ ﺍﻭ ﻗﻮﯼ‌ﺗﺮ ﺍﺳﺖ ﻭ ﺍﺯ ﺁﻥ ﺟﺎﻳﯽ ﻛﻪ ﺍﺯ ﻭﯼ ﺑﺰﺭﮒ‌ﺗﺮ ﺍﺳﺖ ﺍﺯ ﺍﻭ ﺑﻴﺸﺘﺮ ﻣﯽ‌ﺩﺍﻧﺪ.
ﺑﺎ ﺗﻮﻟﺪ ﺳﻮﻣﻴﻦ ﻓﺮﺯﻧﺪ، ﻓﺮﺯﻧﺪ ﺩﻭﻡ، ﺑﭽﻪ ﻭﺳﻄﯽ ﻣﯽ‌ﺷﻮﺩ. ﺍﻭ ﻧﻪ ﺍﻣﺘﻴﺎﺯ ﺑﭽﻪ ﺍﻭﻝ ﺭﺍ ﺩﺍﺭﺩ ﻭ ﻧﻪ ﺍﻣﺘﻴﺎﺯﺍﺕ ﺗﻪ‌ﺗﻐﺎﺭﯼ ﺑﻮﺩﻥ ﺭﺍ. ﺍﻭ ﻫﻤﻴﺸﻪ ﻛﻤﯽ ﻓﺮﺍﻣﻮﺵ ﺷﺪﻩ ﺍﺳﺖ. ﺣﺎﻻ ﺳﻮﻣﯽ ﺍﺳﺖ ﻛﻪ ﻧﻮﺯﺍﺩ ﺍﺳﺖ ﻭ ﻣﻼﺣﻈﺎﺗﯽ ﻛﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﻛﻮﭼﻚ‌ﺗﺮﻳﻦ ﺍﺳﺖ ﺷﺎﻣﻞ ﺍﻭ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻦ ﻛﻪ ﭼﻨﺪ ﻛﻮﺩﮎ ﺩﺭ ﻳﮏ ﺧﺎﻧﻮﺍﺩﻩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺍﻫﻤﻴﺘﯽ ﻧﺪﺍﺭﺩ: ﻫﻤﻪ ﺁﻥ‌ﻫﺎ ﺩﺭ ﻣﺤﻴﻄﯽ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﻧﺪ ﻛﻪ ﻣﺨﺘﺺ ﺧﻮﺩ ﺁﻥ‌ﻫﺎﺳﺖ.
ﺗﺮﻛﻴﺐ ﮊﻧﺘﻴﮏ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻭ ﺍﻳﻨﻜﻪ ﻓﺮﺯﻧﺪ ﭼﻨﺪﻡ ﺧﺎﻧﻮﺍﺩﻩ ﺑﺎﺷﺪ، ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﺍﺳﺖ ﻛﻪ ﺍﻭ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺗﻔﺎﻭﺕ ﺩﺍﺭﺩ. ﺑﻪ ﻫﻤﻴﻦ ﻋﻠﺖ ﺑﺎﻳﺪ ﺭﻭﺵ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺑﺮﺍﯼ ﺗﺮﺑﻴﺖ ﺁﻥ‌ﻫﺎ ﺑﻪ ﻛﺎﺭ ﺑﺮﻳﺪ. ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻛﻮﺩﻛﯽ ﻣﻀﻄﺮﺏ ﻛﻪ ﺩﺭ ﺭﻭ­ﺩﺭ­ﺭﻭﻳﯽ ﺑﺎ ﻣﺴﺎﺋﻞ ﺩﭼﺎﺭ ﺍﺷﻜﺎﻻﺗﯽ ﻫﺴﺖ، ﺑﺎ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻛﻮﺩﻛﯽ ﺑﺸﺎﺵ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﺍﻳﻦ ﻣﺴﺎﺋﻞ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﻧﻤﯽ‌­ﮔﻴﺮﻧﺪ. ﺁﻥ‌ﻫﺎ ﺩﺭ ﭘﯽِ ﺭﻭﺵ‌ﻫﺎﯼ ﺩﻳﺴﻴﭙﻠﻴﻨﯽ ﻋﻤﻮﻣﯽ ﻭ ﻣﻄﻠﻖ ﻫﺴﺘﻨﺪ ﻛﻪ ﺑﺘﻮﺍﻥ ﺩﺭ ﻣﻮﺭﺩ ﺗﻤﺎﻡ ﻛﻮﺩﻛﺎﻥ ﺑﻪ ﻛﺎﺭ ﺑﺮﺩ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﺑﺎﻳﺪ ﮔﻔﺖ ﻛﻪ ﺗﻨﻬﺎ ﺭﻭﺵ‌ﻫﺎﯼ ﺩﻳﺴﻴﭙﻠﻴﻨﯽ ﻛﻪ ﻋﻤﻮﻣﯽ ﺍﺳﺖ ﻭ ﺭﻭﯼ ﺗﻤﺎﻡ ﺑﭽﻪ‌ﻫﺎ ﺍﺛﺮ ﻣﯽ‌ﮔﺬﺍﺭﺩ، ﺭﻭﺵ‌ﻫﺎﯼ ﻣﻨﻔﯽ‌ﺍﯼ ﻫﺴﺘﻨﺪ ﻛﻪ ﻧﺎﺭﺳﺎﻳﯽ‌ﻫﺎﻳﯽ ﺩﺭ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻛﻮﺩﻛﺎﻥ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻭﺭﻧﺪ، ﻓﺮﻗﯽ ﻫﻢ ﻧﻤﯽ‌ﮐﻨﺪ ﮐﻮﺩﮎ ﭼﮕﻮﻧﻪ ﺷﺨﺼﻴﺘﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺯﻣﺎﻧﯽ ﻛﻪ ﺑﺎ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺳﺮﻭﻛﺎﺭ ﺩﺍﺭﻳﻢ ﻛﻪ ﺑﺎﻋﺚ ﺗﻘﻮﻳﺖ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻛﻮﺩﮎ ﻣﯽ‌ﺷﻮﺩ، ﺍﻳﻦ ﺭﻭﺵ‌ﻫﺎ ﺑﺎﻳﺪ ﺑﺎ ﺧﺼﻮﺻﻴﺎﺕ ﻫﺮ ﻛﻮﺩﮎ ﺗﻄﺒﻴﻖ ﺩﺍﺩﻩ ﺷﻮﻧﺪ. ﺍﻳﻦ ﻳﻌﻨﯽ ﻣﺎ ﺑﺎﻳﺪ ﺑﻪ ﺩﻗﺖ ﺧﺼﻮﺻﻴﺎﺕ ﺷﺨﺼﯽ ﻭ ﮔﺮﺍﻳﺶ‌ﻫﺎﯼ ﻫﺮ ﻳﮏ ﺍﺯ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺭﺍ ﺷﻨﺎﺳﺎﻳﯽ ﻛﻨﻴﻢ ﻭ ﺑﺮﺍﺳﺎﺱ ﺁﻥ ﺑﺎ ﺍﻭ ﺭﻓﺘﺎﺭ ﻛﻨﻴﻢ.
 
ﺍﮔﺮ ﺩﺭ ﻫﺮ ﻣﺮﺣﻠﻪ ﺍﺯ ﺭﺷﺪ ﻛﻮﺩﮎ ﺑﻪ ﺍﻭ ﺍﻣﻜﺎﻥ ﺗﻔﺤﺺ ﺩﺭ ﻣﺤﻴﻄﺶ ﺭﺍ ﺑﺪﻫﻴﺪ ﻭ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﺑﻪ ﻣﺤﺾ ﺁﻥ ﻛﻪ ﺍﻣﻜﺎﻧﺶ ﻭﺟﻮﺩ ﺩﺍﺭﺩ  ﺧﻮﺩ ﺭﺍ ﺍﺩﺍﺭﻩ ﻛﻨﺪ،  ﺑﻪ ﺍﻭ ﺍﻣﻜﺎﻥ ﻣﯽ‌ﺩﻫﻴﺪ ﻛﻪ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻣﺜﺒﺘﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.
ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺮﺍﯼ ﺍﻭﻟﻴﻦ ﺑﺎﺭ ﻗﺎﺷﻖ ﺑﻪ ﺩﺳﺖ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﺑﻪ ﺷﻤﺎ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﻛﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻏﺬﺍ ﺑﺨﻮﺭﺩ، ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﺍﻣﺘﺤﺎﻥ ﻛﻨﺪ، ﻛﻤﯽ ﺑﻴﺸﺘﺮ ﻳﺎ ﻛﻤﺘﺮ «ﻛﺜﻴﻔﯽ» ﺩﺭ ﺍﻳﻦ ﺳﻦ ﺍﻫﻤﻴﺖ ﭼﻨﺪﺍﻧﯽ ﻧﺪﺍﺭﺩ، ﭼﺮﺍﻛﻪ ﺍﻭ ﺩﺭ ﺣﺎﻝ ﻳﺎﺩﮔﻴﺮﯼ ﻛﻨﺘﺮﻝ ﻭ ﺍﺩﺍﺭﻩ ﻛﺮﺩﻥ ﺧﻮﺩ ﺍﺳﺖ. ﺍﮔﺮ ﺍﺟﺎﺭﻩ ﺩﻫﻴﻢ ﻛﻮﺩﮎ ﺧﻮﺩﺵ ﻛﺎﺭﻫﺎﻳﺶ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﺪ، ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﺍﻳﺴﺖ ﻛﻪ ﺗﺼﻤﻴﻢ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﮔﺮﻓﺘﻪ ﺑﺎﺷﻴﻢ ﻛﻪ ﻛﻮﺩﮎ ﺑﺰﺭﮒ ﺷﻮﺩ.
ﻏﺎﻟﺒﺎ، ﺑﻪ ﻋﻠﺖ ﺍﻳﻨﻜﻪ ﺩﺭ ﻋﻤﻖ ﻭﺟﻮﺩﻣﺎﻥ ﺩﻟﻤﺎﻥ ﻧﻤﯽ‌ﺧﻮﺍﻫﺪ ﻓﺮﺯﻧﺪﻣﺎﻥ ﺑﺰﺭﮒ ﺷﻮﺩ، ﻣﺮﺩﺩ ﻫﺴﺘﻴﻢ ﺑﻪ ﺍﻭ ﺍﺟﺎﺯﻩ ﺩﻫﻴﻢ ﻛﺎﺭﻫﺎﻳﺶ ﺭﺍ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺍﻧﺠﺎﻡ ﺩﻫﺪ ﻭ ﺑﺎ ﺍﻳﻦ ﻫﻤﻪ، ﺑﺎ ﺍﺣﺘﺮﺍﻡ ﮔﺬﺍﺷﺘﻦ ﺑﻪ ﻣﻴﻞ ﻏﺮﻳﺰﯼ ﺁﻥ‌ﻫﺎ ﺑﻪ ﺁﺯﺍﺩﯼ ﻭ ﺍﻣﻜﺎﻥ ﺍﻳﻦ ﻛﻪ ﺧﻮﺩ ﺭﺍ ﺭﺷﺪ ﺩﻫﻨﺪ، ﻛﻤﮏ ﺷﺎﻳﺎﻧﯽ ﺑﻪ ﺁﻥ‌ﻫﺎ ﻣﯽ‌ﻛﻨﻴﻢ. ﻫﻤﺎﻥ ﻃﻮﺭ ﻛﻪ ﭘﻴﺶ ﺍﺯ ﺍﻳﻦ ﮔﻔﺘﻢ ﺑﺮﺧﻮﺭﺩﯼ ﻣﺘﻔﺎﻭﺕ ﺩﺭﺑﺎﺭﻩ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺍﻋﻤﺎﻝ ﻛﻮﺩﻙ، ﺭﺷﺪ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌ﺍﯼ ﻣﺜﺒﺖ ﺭﺍ ﺩﺭ ﻛﻮﺩﮎ ﺗﻘﻮﻳﺖ ﻣﯽ‌ﻛﻨﺪ.
 
ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻗﺪﺭﺕ ﺑﺴﻴﺎﺭ ﻧﻴﺮﻭﻣﻨﺪ ﺗﻘﻠﻴﺪ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ، ﻋﺎﻣﻞ ﺑﺴﻴﺎﺭ ﻣﻬﻤﯽ ﺩﺭ ﺷﻜﻞ‌ﮔﻴﺮﯼ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌­ﺍﯼ ﻣﺜﺒﺖ ﺩﺭ ﻛﻮﺩﮎ ﺍﺳﺖ.
ﻛﻮﺩﻛﺎﻥ ﻣﻘﻠﺪﺍﻥ ﻓﻮﻕ‌ﺍﻟﻌﺎﺩﻩ‌ﺍﯼ ﻫﺴﺘﻨﺪ ﻭ ﺩﺭ ﺳﺎﻳﻪ ﭼﻨﻴﻦ ﻋﻄﻴﻪ‌ﺍﯼ، ﻣﺎ ﻭﺳﻴﻠﻪ ﺁﻣﻮﺯﺷﯽ ﺑﺴﻴﺎﺭ ﺧﻮﺑﯽ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻳﻢ. ﺍﮔﺮ ﺑﺘﻮﺍﻧﻴﻢ ﻣﺪﻝ ﺯﻧﺪﻩ‌ﺍﯼ ﺍﺯ ﺷﺨﺼﻴﺘﯽ ﺑﺎ ﺧﺼﻠﺖ‌ﻫﺎﯼ ﻣﺜﺒﺖ ﺍﺭﺍﺋﻪ ﺩﻫﻴﻢ، ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺁﻥ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﺗﻘﻠﻴﺪ ﻓﺮﺍ ﺧﻮﺍﻫﻨﺪ ﮔﺮﻓﺖ.
ﺍﮔﺮ ﺍﻳﻦ ﻣﻔﻬﻮﻡ ﺭﺍ ﺩﺭﮎ ﻛﻨﻴﻢ، ﺍﺯ ﻣﺒﺎﺭﺯﺍﺕ ﺑﻴﻬﻮﺩﻩ ﺑﺎ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺑﺮ ﺣﺬﺭ ﺧﻮﺍﻫﻴﻢ ﺑﻮﺩ. ﺍﮔﺮ ﺑﺎ ﺩﺭﺳﺖ ﻏﺬﺍ ﺧﻮﺭﺩﻥ ﻣﺜﺎﻝ ﺧﻮﺑﯽ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺍﺭﺍﺋﻪ ﺩﻫﻴﻢ، ﺯﻣﺎﻧﯽ ﻛﻪ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﻛﺎﻓﯽ ﺑﺰﺭﮒ ﺷﻮﻧﺪ، ﺍﺯ ﻣﺎ ﺗﻘﻠﻴﺪ ﺧﻮﺍﻫﻨﺪ ﻛﺮﺩ. ﻧﻪ ﺩﺭ ﺩﻭ ﺳﺎﻟﮕﯽ ﻭ ﻧﻪ ﺩﺭ ﭼﻬﺎﺭ ﺳﺎﻟﮕﯽ ﺑﻠﻜﻪ ﺩﻳﺮﺗﺮ. ﺍﮔﺮ ﺑﺎ ﺍﻭﻟﻴﻦ ﻣﺸﻜﻼﺕ ﺟﺎ ﻧﺰﻧﻴﻢ ﻭ ﺩﺭ ﻛﺎﺭﻫﺎﻳﻤﺎﻥ ﭘﺸﺖ ﻛﺎﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ، ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺍﺯ ﺍﻳﻦ ﻣﺒﺎﺭﺯﻩ ﻛﺮﺩﻥ ﻣﺎ ﺗﻘﻠﻴﺪ ﺧﻮﺍﻫﻨﺪ ﻛﺮﺩ. ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻪ ﺣﻘﻮﻕ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﺩﻳﮕﺮﺍﻥ ﺍﺣﺘﺮﺍﻡ ﺑﮕﺬﺍﺭﻧﺪ، ﺑﺎﻳﺪ ﺑﺎ ﺍﺣﺘﺮﺍﻡ ﮔﺬﺍﺷﺘﻦ ﺑﻪ ﺣﻘﻮﻕ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﺁﻥ‌ﻫﺎ ﺷﺮﻭﻉ ﻛﻨﻴﻢ. ﺑﻬﺘﺮﻳﻦ ﺭﻭﺵ ﺑﺮﺍﯼ ﺁﻣﻮﺯﺵ، ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﻧﻤﻮﻧﻪ‌­ﺍﯼ ﺧﻮﺏ ﺍﺳﺖ. ﺯﻣﺎﻧﯽ ﻛﻪ ﻣﺎ ﻛﻮﺩﻛﯽ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺍﺣﺘﺮﺍﻡ ﻧﮕﺬﺍﺷﺘﻦ ﺑﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺩﻳﮕﺮﯼ ﺳﺮﺯﻧﺶ ﻣﯽ‌ﻛﻨﻴﻢ، ﺑﺎ ﺍﻳﻦ ﻛﺎﺭﻣﺎﻥ ﺑﻪ ﺍﻭ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻴﻢ ﻛﻪ ﺧﻮﺩ ﻣﺎ ﻧﻴﺰ ﺑﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻭ ﺍﺣﺘﺮﺍﻡ ﻧﻤﯽ‌ﮔﺬﺍﺭﻳﻢ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺮﺩ ﺗﺮﺑﻴﺘﯽ ﺍﻋﻤﺎﻝ ﻣﺎ ﺧﻴﻠﯽ ﺑﻴﺸﺘﺮ ﺍﺯ ﺑﺮﺩ ﺗﺮﺑﻴﺘﯽ ﮔﻔﺘﺎﺭﻣﺎﻥ ﺍﺳﺖ.
 
 
ﺣﻤﺎﻳﺖ ﻋﺎﻃﻔﯽ ﻭﺍﻟﺪﻳﻦ ﺍﺯ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺑﻪ ﻛﻮﺩﮎ ﻛﻤﮏ ﻣﯽ‌ﻛﻨﺪ ﺗﺎ ﺑﺮ ﺍﺣﺴﺎﺳﺎﺕ ﻋﺪﻡ‌ ﺗﻄﺎﺑﻖ ﺧﻮﺩ ﻓﺎﺋﻖ ﺁﻣﺪﻩ ﻭ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﻣﺴﺘﺤﻜﻤﯽ ﻛﺴﺐ ﻛﻨﻨﺪ.
 ﺗﻤﺎﻡ ﻛﻮﺩﻛﺎﻥ ﺑﻪ ﻋﻠﺖ ﻗﺪﺷﺎﻥ ﻭ ﻧﺪﺍﺷﺘﻦ ﺗﺠﺮﺑﻪ ﻛﺎﻓﯽ، ﺩﺭ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﺩﻧﻴﺎ ﺩﭼﺎﺭ ﺍﺣﺴﺎﺱ ﻋﺪﻡ‌ ﺗﻄﺎﺑﻖ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺑﻪ ﺍﻳﻦ ﺩﻭ ﻋﺎﻣﻞ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﻧﺎﺷﯽ ﺍﺯ ﺁﻥ‌ﻫﺎ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭ ﻛﻤﯽ ﻣﯽ‌ﺩﻫﻨﺪ.
ﻛﻮﺩﻛﺎﻥ ﺧﺮﺩﺳﺎﻝ، ﻧﺎﺗﻮﺍﻥ ﻭ ﻧﺴﺒﺖ ﺑﻪ ﻭﺍﻟﺪﻳﻦ ﻭ ﺩﻳﮕﺮ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺩﻭﺭ ﻭ ﺑﺮﺷﺎﻥ ﺩﺭ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻣﺤﻴﻂ ﺯﻧﺪﮔﻴﺸﺎﻥ ﺩﺭ ﺳﻄﺢ ﭘﺎﻳﻴﻦ‌ﺗﺮﯼ ﻗﺮﺍﺭ ﺩﺍﺭﻧﺪ. ﺍﮔﺮ ﺍﻳﻦ ﺭﺍ ﺑﺎﻭﺭ ﻧﺪﺍﺭﻳﺪ، ﺗﺠﺮﺑﻪ ﺯﻳﺮ ﺭﺍ ﺑﻜﻨﻴﺪ. ﻣﺪﺗﯽ ﺭﻭﯼ ﺯﺍﻧﻮﻫﺎﻳﺘﺎﻥ ﺭﺍﻩ ﺑﺮﻭﻳﺪ. ﺑﺒﻴﻨﻴﺪ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﻳﻦ ﻏﻮﻝ‌ﻫﺎﯼ ﺩﻧﻴﺎﯼ ﺑﺰﺭﮔﺴﺎﻻﻥ ﭼﻪ ﺍﺣﺴﺎﺳﯽ ﻣﯽ‌ﻛﻨﻴﺪ. ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﻧﺎﺗﻮﺍﻧﯽ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﻛﻨﻴﺪ ﻫﻤﺎﻧﯽ ﺍﺳﺖ ﻛﻪ ﻛﻮﺩﻛﺎﻥ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﻛﻨﻨﺪ. ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﺁﻥ‌ﻫﺎ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﺣﻤﺎﻳﺖ ﻋﺎﻃﻔﯽ ﻭ ﺗﺸﻮﻳﻖ ﻭﺍﻟﺪﻳﻨﺸﺎﻥ ﺩﺍﺭﻧﺪ ﺗﺎ ﺍﺯ ﺍﻳﻦ ﻃﺮﻳﻖ ﺑﺮ ﺍﻳﻦ ﺍﺣﺴﺎﺱ ﻋﺪﻡ‌ ﺗﻄﺎﺑﻖ ﺧﻮﺩ ﻓﺎﺋﻖ ﺁﻳﻨﺪ.
ﺩﺭ ﺩﻭﺭﻩ ﺍﻭﻝ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﮐﻮﺩﮎ ﺍﺯ ﻓﻦ ﺍﻧﻌﮑﺎﺱ ﺍﺣﺴﺎﺳﺎﺕ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ. ﺑﺎ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻥ ﺍﻳﻦ ﻓﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻪ ﮐﻮﺩﮎ ﻧﺸﺎﻥ ﺩﻫﻴﻢ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻭ ﺭﺍ ﺩﺭﮎ ﻣﯽ‌ﮐﻨﻴﻢ. ﺍﻳﻦ ﺗﻨﻬﺎ ﻛﺎﺭﯼ ﺍﺳﺖ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﺮﺍﯼ ﺍﻃﻤﻴﻨﺎﻥ ﺧﺎﻃﺮ ﺩﺍﺩﻥ ﺑﻪ ﺍﻭ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ. ﺍﻳﻦ ﺣﻘﻴﻘﺖ ﻛﻪ ﻳﮏ ﺍﻧﺴﺎﻥ ﺑﺎﻟﻎ ﺑﻪ ﺍﻭ ﺑﻔﻬﻤﺎﻧﺪ ﻧﮕﺮﺍﻧﯽ ﺍﻭ ﺭﺍ ﺩﺭﮎ ﻣﯽ‌ﻛﻨﺪ، ﺑﻪ ﻃﺮﺯ ﺣﻴﺮﺕ‌ﺍﻧﮕﻴﺰﯼ ﺑﻪ ﺍﻭ ﻗﻮﺕ ﻗﻠﺐ ﻣﯽ‌ﺩﻫﺪ.
