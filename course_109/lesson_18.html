ﺩﺭﺱ ﻫﺸﺘﻢ- ﺧﺸﻮﻧﺖ-ﺍﺩﺍﻣﻪ

ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﺍﺯ ﺍﻫﻤﻴﺖ ﺗﻔﺎﻭﺕ ﻗﺎﺋﻞ ﺷﺪﻥ ﺑﻴﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﻭ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ. ﮔﻔﺘﻴﻢ ﮐﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﺑﻪ ﺧﻮﺩ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﻧﺒﺎﻳﺪ ﺟﻠﻮﯼ ﺍﺑﺮﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ ﺭﺍ ﮔﺮﻓﺖ. ﺑﺮﺍﯼ ﺟﻠﻮﮔﻴﺮﯼ ﺍﺯ ﺗﺒﺪﻳﻞ ﺷﺪﻥ ﻓﺮﺯﻧﺪﻣﺎﻥ ﺑﻪ ﻓﺮﺩﯼ ﺧﺸﻦ، ﺑﺎﻳﺪ ﮐﻨﺘﺮﻝ ﺍﻋﻤﺎﻝ ﺭﺍ ﺑﻪ ﮐﻮﺩﮎ ﺑﻴﺎﻣﻮﺯﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺩﻭ ﺑﺤﺚ ﺑﺴﻴﺎﺭ ﺩﻳﮕﺮ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺧﺸﻮﻧﺖ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ. ﺍﺳﺤﻠﻪ‌ﻫﺎﯼ ﺧﻄﺮﻧﺎﮎ ﻭ ﺧﺸﻮﻧﺖ ﻏﻴﺮﻭﺍﻗﻌﯽ.

ﺍﺳﻠﺤﻪ‌ﻫﺎﯼ ﺧﻄﺮﻧﺎﮎ
ﺍﻭﻝ ﺍﺳﻠﺤﻪ‌ﻫﺎﯼ ﺧﻄﺮﻧﺎﮎ ﺭﺍ ﺑﺮﺭﺳﯽ ﻛﻨﻴﻢ. ﺧﻴﻠﯽ ﺻﺮﻳﺢ ﺑﮕﻮﻳﻴﻢ ﻛﻪ ﺩﺍﺷﺘﻦ ﺍﺳﻠﺤﻪ‌ﻫﺎﯼ ﺧﻄﺮﻧﺎﮎ ﺩﺭ ﺧﺎﻧﻪ ﭼﻴﺰ ﺧﻮﺑﯽ ﻧﻴﺴﺖ. ﺍﺳﻠﺤﻪ‌ﻫﺎﯼ ﺧﻄﺮﻧﺎﮎ ﻓﻘﻂ ﻣﻮﺟﺐ ﺍﻳﺠﺎﺩ ﺣﻮﺍﺩﺙ ﺩﻟﺨﺮﺍﺵ ﺷﺪﻩ ﻭ ﻭﺟﻮﺩﺷﺎﻥ ﺗﺤﺮﻳﻜﯽ ﺩﺍﺋﻤﯽ ﺑﺮﺍﯼ ﺧﺸﻮﻧﺖ ﺍﺳﺖ.
ﺑﯽ‌ﺷﮏ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺎﺩﺭﺍﻥ ﺍﺯ ﺍﻧﻔﺠﺎﺭ ﺧﺸﻮﻧﺖ ﺩﺭ ﺟﺎﻣﻌﻪ ﻧﮕﺮﺍﻧﻨﺪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺳﻌﯽ ﻣﯽ‌ﻛﻨﻨﺪ ﻛﻪ ﻫﺮ ﮔﻮﻧﻪ ﺗﻈﺎﻫﺮ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰﯼ ﺩﺭ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺭﺍ ﺳﺮﻛﻮﺏ ﻛﻨﻨﺪ، ﺁﻥ‌ﻫﺎ  ﻣﺎﻧﻊ ﺍﺯ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺑﺎ ﺍﺳﻠﺤﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻳﺎ ﺳﺮﺑﺎﺯﺍﻥ ﭘﻼﺳﺘﻴﻜﯽ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﻧﻤﯽ‌ﮔﺬﺍﺭﻧﺪ ﻓﻴﻠﻢ‌ﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﺩﺭ ﺁﻥ‌ﻫﺎ ﺻﺤﻨﻪ‌ﻫﺎﯼ ﺧﺸﻮﻧﺖ‌ﺑﺎﺭ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻧﮕﺎﻩ ﻛﻨﻨﺪ. ﺑﻌﻀﯽ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺣﺘﯽ ﻧﻤﯽ‌ﮔﺬﺍﺭﻧﺪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻛﺎﺭﺗﻮﻥ‌ﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﺩﺭ ﺁﻥ‌ﻫﺎ «ﺧﺸﻮﻧﺖ» ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺑﺒﻴﻨﻨﺪ: ﻣﺜﻼ ﺣﻴﻮﺍﻧﺎﺗﯽ ﻛﻪ ﺑﻪ ﺳﺮ ﻭ ﻛﻠﻪ ﻫﻢ ﻣﯽ‌ﺯﻧﻨﺪ ﻭ ﻳﺎ ﺑﻪ ﻫﻢ ﺗﻴﺮﺍﻧﺪﺍﺯﯼ ﻣﯽ‌ﻛﻨﻨﺪ.
ﺍﻳﻦ ﺗﺮﺱ ﺷﺪﻳﺪ ﺍﺯ ﺧﺸﻮﻧﺖ ﺩﺭ ﺑﺎﺯﯼ‌ﻫﺎ ﻭ ﻧﻤﺎﻳﺸﺎﺕ ﻏﻴﺮﻃﺒﻴﻌﯽ ﺍﺳﺖ. ﻋﻤﻞ ﺍﻳﻦ ﻭﺍﻟﺪﻳﻦ ﻭﺍﻗﻌﺎ ﺩﺭ ﺟﻬﺖ ﺣﻔﺎﻇﺖ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻋﻠﻴﻪ ﺧﺸﻮﻧﺖ ﻧﻴﺴﺖ. ﺭﻓﺘﺎﺭ ﺍﻳﻦ ﻭﺍﻟﺪﻳﻦ ﻓﻘﻂ ﻣﻮﺟﺐ ﻣﻨﺰﻭﯼ ﻛﺮﺩﻥ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻭ ﺣﻤﺎﻳﺖ ﺯﻳﺎﺩ ﺍﺯ ﺣﺪ ﺍﺯ ﺁﻥ‌ﻫﺎ  ﻣﯽ‌ﺷﻮﺩ.
ﺍﻳﻦ ﻋﻘﻴﺪﻩ ﺯﻳﺎﺩ ﺍﺯ ﺣﺪ ﺣﻤﺎﻳﺖ ﻛﺮﺩﻥ ﺍﺯ ﻓﺮﺯﻧﺪﺍﻥ ﺍﺯ ﻛﺠﺎ ﻧﺎﺷﯽ ﻣﯽ‌ﺷﻮﺩ؟ ﻣﺎﺩﺭ ﺍﺯ ﺯﻭﺩ ﺧﺸﻤﯽ ﺧﻮﺩ ﻫﺮﺍﺳﻨﺎﮎ ﺍﺳﺖ. ﺍﻳﻦ ﻣﺎﺩﺭﺍﻥ ﻧﮕﺮﺍﻧﻨﺪ ﻛﻪ ﻧﺘﻮﺍﻧﻨﺪ ﺧﻮﺩ ﺭﺍ ﻛﻨﺘﺮﻝ ﻛﻨﻨﺪ ﻭ ﻣﺮﺗﻜﺐ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰ ﺷﻮﻧﺪ. ﺍﻣﺎ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﺑﻪ‌ ﻃﻮﺭ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺍﺳﺖ. ﻛﻮﭼﮏ‌ﺗﺮﻳﻦ ﺻﺤﻨﻪ ﺧﺸﻮﻧﺖ ﺩﺭ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻭ ﻳﺎ ﺳﻴﻨﻤﺎ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﻛﻪ ﺍﻭ ﻧﮕﺮﺍﻥ ﻇﺎﻫﺮ ﺷﺪﻥ ﻭ ﺍﻧﻔﺠﺎﺭ ﺧﺸﻮﻧﺖ ﺩﺭ ﺧﻮﺩﺵ ﮔﺮﺩﺩ ﻭ ﻣﻮﺟﺐ ﺷﻮﺩ ﻛﻪ ﺷﻮﻫﺮ ﻭ ﻓﺮﺯﻧﺪﺍﻧﺶ ﺭﺍ ﻛﺘﮏ ﺑﺰﻧﺪ. ﺍﻳﻦ ﺗﺮﺱ ﺁﻧﭽﻨﺎﻥ ﻗﻮﯼ ﺍﺳﺖ ﻛﻪ ﺍﻭ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﺪ ﻓﺮﺯﻧﺪﺵ ﻧﻴﺰ ﻣﺎﻧﻨﺪ ﺧﻮﺩ ﺍﻭﺳﺖ ﻭ ﻫﻤﺎﻥ ﺍﺣﺴﺎﺱ ﺭﺍ ﺩﺍﺭﺩ. ﺑﻪ ﻫﻤﻴﻦ ﻋﻠﺖ ﺑﻌﻀﯽ ﺍﺯ ﺍﺷﺨﺎﺹ ﻧﮕﺮﺍﻧﻨﺪ ﻛﻪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺩﺭ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻭ ﻳﺎ ﺳﻴﻨﻤﺎ ﺻﺤﻨﻪ‌ﻫﺎﯼ ﺧﺸﻮﻧﺖ‌ﺑﺎﺭ ﺑﺒﻴﻨﻨﺪ ﻭ ﻳﺎ ﺑﺎ ﺳﺮﺑﺎﺯﻫﺎﯼ ﭘﻼﺳﺘﻴﻜﯽ ﺑﺎﺯﯼ ﻛﻨﻨﺪ.
ﺍﻳﻦ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ. ﻛﻮﺩﮎ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﺩ ﺧﻮﺩ ﺭﺍ ﺗﺨﻠﻴﻪ ﻛﻨﺪ. ﺍﮔﺮ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺍﺟﺎﺯﻩ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺑﻪ ﺍﻭ ﻧﺪﻫﻨﺪ ﺍﻭ ﺧﻮﺩﺵ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺧﻮﺍﻫﺪ ﻛﺮﺩ. ﺍﮔﺮ ﻣﺎﺩﺭ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺵ ﺍﺳﻠﺤﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻧﺨﺮﺩ، ﺍﻭ ﺧﻮﺩﺵ ﺁﻥ ﺭﺍ ﺑﺎ ﻳﮏ ﺗﻜﻪ ﭼﻮﺏ، ﻳﮏ ﻟﻮﻟﻪ ﻭ ﻳﺎ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﺑﺎ ﺍﻧﮕﺸﺘﺎﻧﺶ ﺩﺭﺳﺖ ﻣﯽ‌ﻛﻨﺪ: «ﺩﻧﮓ! ﻣﺮﺩﯼ!» ﻗﺮﻥ‌ﻫﺎﺳﺖ ﻛﻪ ﻛﻮﺩﻛﺎﻥ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺑﺎ ﺳﺮﺑﺎﺯﻫﺎﯼ ﻛﻮﭼﮏ ﺑﺎﺯﯼ ﻛﻨﻨﺪ. ﺍﮔﺮ ﻭﺍﻟﺪﻳﻨﯽ ﻛﻪ ﺯﻳﺎﺩ ﺍﺯ ﺣﺪ ﺍﺯ ﺁﻥ‌ﻫﺎ  ﺣﻤﺎﻳﺖ ﻣﯽ‌ﻛﻨﻨﺪ ﺑﺮﺍﻳﺸﺎﻥ ﺍﺯ ﺍﻳﻦ ﺳﺮﺑﺎﺯﺍﻥ ﻧﺨﺮﻧﺪ، ﺧﻮﺩﺷﺎﻥ ﺁﻥ‌ﻫﺎ  ﺭﺍ ﺩﺭﺳﺖ ﻣﯽ‌ﻛﻨﻨﺪ.

ﺑﺎﻳﺪ ﺑﻴﻦ ﻛﻮﺩﮎ ﭘﺮﺧﺎﺷﮕﺮ ﻭ ﻛﻮﺩﮎ ﺧﺸﻦ ﺗﻔﺎﻭﺕ ﻗﺎﺋﻞ ﺷﺪ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﺍﻛﺜﺮ ﻭﺍﻟﺪﻳﻦ ﺍﻳﻦ ﺩﻭ ﺭﺍ ﺑﺎ ﻫﻢ ﻳﻜﯽ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﻣﻌﻨﯽ ﻣﺨﺎﻟﻒ ﺧﺸﻦ ﺁﺭﺍﻡ ﺍﺳﺖ. ﻣﻌﻨﯽ ﻣﺨﺎﻟﻒ ﭘﺮﺧﺎﺷﮕﺮ ﻣﻨﻔﻌﻞ ﺍﺳﺖ.
ﻓﺮﺯﻧﺪﺍﻥ ﻣﺎ ﻧﺒﺎﻳﺪ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺧﺸﻨﯽ ﺑﺸﻮﻧﺪ. ﻭﻟﯽ ﺑﺎﻳﺪ ﺑﻪ ﭘﺴﺮﻫﺎ ﻧﻮﻋﯽ ﭘﺮﺧﺎﺷﮕﺮﯼ ﺭﺍ ﺁﻣﻮﺧﺖ، ﭼﺮﺍ ﻛﻪ ﻣﺮﺩ ﺑﺎﻟﻎ ﻣﻨﻔﻌﻞ ﭼﻨﺪﺍﻥ ﺟﺎﻟﺐ ﻧﻴﺴﺖ. ﻣﻨﻔﻌﻞ ﺑﻮﺩﻥ ﺩﺭ ﺭﻝ ﻣﺮﺩﺍﻧﮕﯽ ﺍﻭ ﻳﮏ ﻧﻘﻄﻪ ﺿﻌﻒ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻣﺎﺩﺭﯼ ﻛﻪ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﺍﺯ ﻫﺮ ﻧﻮﻉ ﻧﻤﻮﺩ ﺧﺸﻮﻧﺖ ﺑﺮﺣﺬﺭ ﻣﯽ‌ﺩﺍﺭﺩ ﻭ ﺑﻴﻤﺎﺭﮔﻮﻧﻪ ﺍﺯ ﺍﻭ ﺣﻤﺎﻳﺖ ﻣﯽ‌ﻛﻨﺪ، ﺩﺭ ﺟﻬﺖ ﺻﻼﺡ ﻓﺮﺯﻧﺪﺵ ﻋﻤﻞ ﻧﻤﯽ‌ﻛﻨﺪ. ﻣﺎﺩﺭﯼ ﻛﻪ ﺑﺎﺯﯼ ﺑﺎ ﺳﺮﺑﺎﺯﻫﺎﯼ ﭘﻼﺳﺘﻴﻜﯽ ﻭ ﻫﻔﺖ‌ﺗﻴﺮ، ﺩﻳﺪﻥ ﻓﻴﻠﻢ‌ﻫﺎﯼ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻭ ﺳﻴﻨﻤﺎ ﻭ ﻛﺎﺭﺗﻮﻥ‌ﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﺩﺭ ﺁﻥ‌ﻫﺎ  «ﺧﺸﻮﻧﺖ» ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﺩ ﺑﺮ ﻓﺮﺯﻧﺪ ﺧﺮﺩﺳﺎﻟﺶ ﻣﻤﻨﻮﻉ ﻣﯽ‌ﻛﻨﺪ، ﺍﻭ ﺭﺍ ﺩﺭ ﺟﻬﺖ ﺗﺒﺪﻳﻞ ﺑﻪ «ﺩﺧﺘﺮ ﻛﻮﭼﻮﻟﻮﻳﯽ» ﻣﻨﻔﻌﻞ ﻫﺪﺍﻳﺖ ﻣﯽ‌ﻛﻨﺪ. ﺍﻳﻦ ﮔﻮﻧﻪ ﻣﺎﺩﺭﺍﻥ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻛﻤﻜﯽ ﺩﺭ ﺟﻬﺖ ﻛﺴﺐ ﻧﻴﺮﻭ ﻭ ﭘﺮﺧﺎﺷﮕﺮﯼ ﻃﺒﻴﻌﯽ‌ﺍﯼ ﻛﻪ ﺁﻥ‌ﻫﺎ  ﺑﺮﺍﯼ ﺗﻤﺎﺱ ﺑﺎ ﭘﺴﺮﺍﻥ ﻫﻢ ﺳﻦ ﻭ ﺳﺎﻟﺸﺎﻥ ﻧﻴﺎﺯ ﺩﺍﺭﻧﺪ ﻧﻤﯽ‌ﻛﻨﻨﺪ.
ﺍﻟﺒﺘﻪ ﻧﺒﺎﻳﺪ ﺑﺮﺩﺍﺷﺖ ﻏﻠﻄﯽ ﺍﺯ ﺍﻳﻦ ﺻﺤﺒﺖ‌ﻫﺎ ﺑﻜﻨﻴﺪ. ﺍﺯ ﺁﻥ ﻃﺮﻑ ﻫﻢ ﻧﺒﺎﻳﺪ ﺍﻓﺘﺎﺩ. ﺍﺟﺎﺯﻩ ﻧﺪﻫﻴﺪ ﻛﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻫﺮ ﮔﻮﻧﻪ ﺍﺳﻠﺤﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺭﺍ ﺑﺨﺮﺩ. ﺍﮔﺮ ﻳﮏ ﻭﻗﺖ ﺩﻳﺪﻳﺪ ﻛﻪ ﺩﺍﺭﻧﺪ ﺗﺒﻠﻴﻎ ﻣﯽ‌ﻛﻨﻨﺪ ﻛﻪ «ﺗﻔﻨﮓ ﺑﺮﺍﯼ ﻛﻮﻣﺎﻧﺪﻭﯼ ﻣﺮﮒ ﺍﺯ ﺟﻨﺲ ﭘﻼﺳﺘﻴﮏ ﺑﺎ ﺭﻭﻛﺸﯽ ﺍﺯ ﺟﻨﺲ ﻛﺮﻡ» ﻭ ﻳﺎ «ﻣﺴﻠﺴﻞ ﺑﺮﺍﯼ ﭼﺮﻳﮏ‌ﻫﺎ ﻛﻪ ﻭﻗﺘﯽ ﺷﻠﻴﮏ ﻣﯽ‌ﻛﻨﻨﺪ ﺧﻮﻥ ﻣﺼﻨﻮﻋﯽ ﺍﺯ ﺁﻥ ﻓﻮﺭﺍﻥ ﻣﯽ‌ﻛﻨﺪ!»، ﻓﻜﺮ ﻧﻜﻨﻴﺪ ﻛﻪ ﻣﺠﺒﻮﺭﻳﺪ ﭼﻨﻴﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﻭﺣﺸﺘﻨﺎﻛﯽ ﺭﺍ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺨﺮﻳﺪ.

ﺍﮔﺮ ﻣﺎﺩﺭﯼ ﺍﺳﺘﻘﻼﻝ، ﻗﻮﻩ ﺗﺤﺮﮎ ﻭ ﻗﺎﺑﻠﻴﺖ ﺩﻓﺎﻉ ﺩﺭ ﻣﻘﺎﺑﻞ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺩﺭ ﻓﺮﺯﻧﺪﺵ ﺗﻘﻮﻳﺖ ﻛﻨﺪ، ﭘﺴﺮﺵ ﺗﺒﺪﻳﻞ ﺑﻪ ﻣﺮﺩﯼ ﺧﻮﺍﻫﺪ ﺷﺪ ﻛﻪ ﺑﻪ ﻃﺮﺯﯼ ﺳﺎﻟﻢ ﺧﺼﻠﺖ ﭘﺮﺧﺎﺷﮕﺮﯼ ﻻﺯﻡ ﺭﺍ ﺑﺪﻭﻥ ﺍﻳﻨﻜﻪ ﺍﻧﺴﺎﻥ ﺧﺸﻨﯽ ﺑﺎﺷﺪ ﺩﺍﺭﺩ.

ﻧﺒﺎﻳﺪ ﺿﺪﻳﺖ ﻭ ﺧﺸﻮﻧﺖ ﺭﺍ ﺑﺎ ﭘﺮﺧﺎﺷﮕﺮﯼ ﻳﻜﯽ ﮔﺮﻓﺖ. ﻭﻗﺘﯽ ﭘﺴﺮﺍﻥ ﺷﻤﺎ ﺑﺰﺭﮒ ﻣﯽ‌ﺷﻮﻧﺪ ﺑﺎﻳﺪ ﺑﻪ ﻣﺮﺩﺍﻥ ﭘﻮﻳﺎ ﻭ ﭘﺮﺧﺎﺷﮕﺮﯼ ﺗﺒﺪﻳﻞ ﺷﻮﻧﺪ. ﻣﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺁﻥ‌ﻫﺎ  ﺑﻪ ﺷﻬﺮﻭﻧﺪﺍﻥ ﺁﺭﺍﻣﯽ ﺗﺒﺪﻳﻞ ﺷﻮﻧﺪ ﻛﻪ ﻣﺴﺎﺋﻠﺸﺎﻥ ﺭﺍ ﻧﻪ ﺍﺯ ﺭﺍﻩ ﺧﺸﻮﻧﺖ، ﺑﻠﻜﻪ ﺍﺯ ﻃﺮﻳﻖ ﻣﻨﻄﻖ ﻭ ﺭﺍﻩ ﺣﻞ‌ﻫﺎﯼ ﻣﺴﺎﻟﻤﺖ‌ﺁﻣﻴﺰ ﺣﻞ ﻛﻨﻨﺪ.

ﺍﻛﻨﻮﻥ ﺑﻪ ﺑﺮﺭﺳﯽ ﻣﺴﺌﻠﻪ ﺧﺸﻮﻧﺖ ﺩﺭ ﻭﺳﺎﻳﻞ ﺍﺭﺗﺒﺎﻁ ﺟﻤﻌﯽ ﺑﭙﺮﺩﺍﺯﻳﻢ. ﻓﺮﺽ ﻛﻨﻴﻢ ﻛﻪ ﺷﻤﺎ ﺑﻪ ﻃﻮﺭ ﻛﻠﯽ، ﺗﻮﺻﻴﻪ‌ﻫﺎﯼ ﺗﺮﺑﻴﺘﯽ ﺭﺍ ﻛﻪ ﺗﺎ ﺍﻳﻨﺠﺎ ﮔﻔﺘﻪ‌ﺍﻳﻢ، ﺩﻧﺒﺎﻝ ﻣﯽ‌ﻛﻨﻴﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺭﻭﺣﯽ ﺍﺳﺎﺳﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺍﺭﺿﺎ ﻣﯽ‌ﻛﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻋﻠﺘﯽ ﻧﺪﺍﺭﺩ ﻛﻪ ﺍﻭ ﺍﺣﺴﺎﺱ ﻣﺤﺮﻭﻣﻴﺖ ﻛﻨﺪ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﺿﺪﻳﺖ ﻭ ﺧﺸﻮﻧﺖ‌ﺑﺎﺭ ﺩﺭ ﻋﻤﻖ ﻭﺟﻮﺩﺵ ﺗﻐﺬﻳﻪ ﮔﺮﺩﻧﺪ. ﺷﻤﺎ ﺑﺮﺍﯼ ﺍﻭ ﻣﺤﻴﻄﯽ ﻓﺮﺍﻫﻢ ﻛﺮﺩﻩ‌ﺍﻳﺪ ﻛﻪ ﺍﻧﮕﻴﺰﺵ‌ﻫﺎﯼ ﻻﺯﻡ ﺑﺮﺍﯼ ﺍﻭ ﺩﺭ ﺁﻥ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻪ ﺷﺪﻩ ﻭ ﺍﻭ ﻫﻤﻴﺸﻪ ﺍﻣﻜﺎﻥ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﺩﺭ ﺩﺍﺧﻞ ﻭ ﺧﺎﺭﺝ ﺍﺯ ﺧﺎﻧﻪ ﺭﺍ ﺩﺍﺭﺩ. ﻫﺮ ﻭﻗﺖ ﻛﻪ ﺩﻟﺶ ﺧﻮﺍﺳﺖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻄﺎﻟﻌﻪ ﻛﻨﺪ ﻭ ﻳﺎ ﺑﻪ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﻫﻨﺮﯼ ﺑﭙﺮﺩﺍﺯﺩ ﻭ ﺍﻣﻜﺎﻥ ﺍﻧﺠﺎﻡ ﻛﺎﺭﻫﺎﯼ ﺟﺎﻟﺐ ﻭ ﺁﻣﻮﺯﺷﯽ ﺑﺴﻴﺎﺭﯼ ﺭﺍ ﻋﻼﻭﻩ ﺑﺮ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻧﮕﺎﻩ ﻛﺮﺩﻥ ﺩﺍﺭﺩ. ﺍﻭ ﺭﻭﺯﯼ ﭼﻬﺎﺭ ﭘﻨﺞ ﺳﺎﻋﺖ ﺟﻠﻮﯼ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻧﻤﯽ‌ﻧﺸﻴﻨﺪ ﻭ ﻣﺎﺕ ﻭ ﻣﺒﻬﻮﺕ ﺑﻪ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺁﻥ ﻧﮕﺎﻩ ﻧﻤﯽ‌ﻛﻨﺪ. ﺷﻤﺎ ﺑﻪ ﻃﻮﺭ ﻣﻨﻄﻘﯽ ﻣﺪﺕ ﺯﻣﺎﻧﯽ ﻛﻪ ﺍﻭ ﺍﺟﺎﺯﻩ ﺩﺍﺭﺩ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻧﮕﺎﻩ ﻛﻨﺪ ﻣﺸﺨﺺ ﻣﯽ‌ﻛﻨﻴﺪ. ﺍﻭ ﺭﺍ ﺗﺮﻏﻴﺐ ﻣﯽ‌ﻛﻨﻴﺪ ﻛﻪ ﻓﻘﻂ ﺑﻪ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺧﻮﺏ ﻛﻮﺩﻛﺎﻥ ﻧﮕﺎﻩ ﻛﻨﺪ ﻭ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﻛﻨﻴﺪ ﺩﺭ ﭼﻨﻴﻦ ﺣﺎﻟﺘﯽ، ﺍﮔﺮ ﺍﻭ ﺑﺮﻧﺎﻣﻪ‌ﺍﯼ ﺭﺍ ﺩﻳﺪ ﻛﻪ ﺩﺭ ﺁﻥ ﺧﺸﻮﻧﺖ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﺩ، ﺍﺻﻼ ﻧﮕﺮﺍﻥ ﻧﺒﺎﺷﻴﺪ.
ﻛﻮﺩﻛﺎﻥ ﺧﺸﻮﻧﺖ ﺭﺍ ﺑﺎ ﺩﻳﺪﻥ ﺻﺤﻨﻪ‌ﻫﺎﯼ ﺧﺸﻮﻧﺖ‌ﺑﺎﺭ ﺩﺭ ﺗﻠﻮﻳﺰﻳﻮﻥ ﻳﺎﺩ ﻧﻤﯽ‌ﮔﻴﺮﻧﺪ. ﺍﻳﻦ ﻋﻘﻴﺪﻩ ﻛﺎﻣﻼ ﻏﻠﻂ ﺍﺳﺖ. ﻫﻴﭻ ﭘﺎﻳﻪ ﻋﻠﻤﯽ‌ﺍﯼ ﺑﺮﺍﯼ ﺍﺛﺒﺎﺕ ﺍﻳﻦ ﻧﻈﺮ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ، ﻛﻮﺩﻛﺎﻥ ﺩﺭ ﺻﻮﺭﺗﯽ ﺧﺸﻦ ﻣﯽ‌ﺷﻮﻧﺪ ﻛﻪ ﻭﺍﻟﺪﻳﻨﺸﺎﻥ ﺧﺸﻮﻧﺖ ﺭﺍ ﺑﻪ ﺁﻥ‌ﻫﺎ  ﺑﻴﺎﻣﻮﺯﻧﺪ، ﻳﻌﻨﯽ:

  • ﻭﻗﺘﯽ ﻛﻪ ﻭﺍﻟﺪﻳﻦ ﻧﺘﻮﺍﻧﻨﺪ ﺍﺣﺘﻴﺎﺟﺎﺕ ﺭﻭﺣﯽ ﺭﻭﺍﻧﯽ ﺁﻧﺎﻥ ﺭﺍ ﺑﺮﻃﺮﻑ ﻛﻨﻨﺪ ﻭ ﺧﺸﻢ ﻭ ﺧﺸﻮﻧﺖ ﺭﺍ ﺑﻪ ﺁﻥ‌ﻫﺎ  ﺍﻟﻘﺎ ﻛﻨﻨﺪ.
  • ﻭﻗﺘﯽ ﻛﻪ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺧﺸﻦ ﺧﻮﺩ ﺗﻘﻠﻴﺪ ﻛﻨﻨﺪ.
  • ﻭﻗﺘﯽ ﻛﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺍﻋﻤﺎﻝ ﺧﺸﻮﻧﺖ‌ﺑﺎﺭ ﺁﻥ‌ﻫﺎ  ﺭﺍ ﺗﺸﻮﻳﻖ ﻛﻨﻨﺪ ﻭ ﺩﺭ ﻣﻘﺎﺑﻞ ﻛﺘﮏ ﺯﺩﻥ ﻛﻮﺩﻛﺎﻥ ﺩﻳﮕﺮ ﻭ ﻳﺎ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩﻥ ﺍﺷﻴﺎﺀ ﺁﻥ ﻃﻮﺭ ﻛﻪ ﺑﺎﻳﺪ ﻋﻜﺲ‌ﺍﻟﻌﻤﻞ ﻧﺸﺎﻥ ﻧﺪﻫﻨﺪ ﻭ ﺟﻠﻮﯼ ﺁﻥ‌ﻫﺎ  ﺭﺍ ﻧﮕﻴﺮﻧﺪ.
  • ﻭﻗﺘﯽ ﻛﻪ ﺑﻪ ﺁﻥ‌ﻫﺎ  ﺍﺟﺎﺯﻩ ﺗﺨﻠﻴﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺧﺸﻮﻧﺖ‌ﺁﻣﻴﺰﺷﺎﻥ ﺩﺍﺩﻩ ﻧﺸﻮﺩ.
     

ﺍﮔﺮ ﺷﻤﺎ ﺍﺯ ﻧﻜﺎﺕ ﺑﺎﻻ ﺍﺟﺘﻨﺎﺏ ﻛﻨﻴﺪ، ﻧﺒﺎﻳﺪ ﻫﻴﭻ ﺗﺮﺳﯽ ﺍﺯ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

ﺗﺎ ﺍﻳﻨﺠﺎ ﻫﻴﭻ ﺻﺤﺒﺘﯽ ﺍﺯ ﻋﺎﻣﻞ ﺍﺻﻠﯽ ﺿﺪ ﺧﺸﻮﻧﺖ ﻧﻜﺮﺩﻳﻢ. ﺍﻳﻦ ﻋﺎﻣﻞ ﻋﺸﻖ ﻧﺎﻡ ﺩﺍﺭﺩ.
ﻓﺮﺍﻣﻮﺵ ﻧﻜﻨﻴﻢ ﻛﻪ ﺑﺰﺭﮔﺴﺎﻻﻥ ﺧﺸﻦ ﺍﺷﺨﺎﺻﯽ ﻫﺴﺘﻨﺪ ﻛﻪ ﺩﺭ ﺩﻭﺭﺍﻥ ﻛﻮﺩﻛﯽ ﺍﺯ ﻋﺸﻖ ﻭ ﻣﺤﺒﺖ ﻭﺍﻗﻌﯽ ﻣﺤﺮﻭﻡ ﺑﻮﺩﻩ‌ﺍﻧﺪ. ﺑﺰﺭﮔﺴﺎﻻﻧﯽ ﻛﻪ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﻣﯽ‌ﻛﺸﻨﺪ ﻭ ﻳﺎ ﻣﺠﺮﻭﺡ ﻣﯽ‌ﻛﻨﻨﺪ ﺍﺷﺨﺎﺻﯽ ﻋﻘﺪﻩ‌ﺍﯼ ﻭ ﺑﻴﻤﺎﺭ ﻫﺴﺘﻨﺪ ﻛﻪ ﺳﺮﺗﺎﺳﺮ ﻭﺟﻮﺩﺷﺎﻥ ﭘﺮ ﺍﺯ ﻧﻔﺮﺕ ﺍﺳﺖ ﭼﺮﺍ ﻛﻪ ﺧﻮﺩ ﺍﺯ ﻋﺸﻖ ﻣﺤﺮﻭﻡ ﺑﻮﺩﻩ‌ﺍﻧﺪ.
ﺑﻬﺘﺮﻳﻦ ﻭﺍﻛﺴﻦ ﺗﺨﻠﻴﻪ ﻋﻠﻴﻪ ﻭﻳﺮﻭﺱ ﺧﺸﻮﻧﺖ ﺩﺭ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﻋﺸﻖ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ‌ﻣﺎﻥ ﺍﺳﺖ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻪ ﺁﻥ‌ﻫﺎ  ﻋﺮﺿﻪ ﻛﻨﻴﻢ. ﺑﻪ ﻋﻮﺽ ﺍﻳﻦ ﻛﻪ ﺑﻴﺸﺘﺮ ﻭﻗﺖ ﺧﻮﺩ ﺭﺍ ﺻﺮﻑ ﺍﻳﻦ ﻛﻨﻴﻢ ﻛﻪ ﻭﻳﺮﻭﺳﯽ ﻛﻪ ﻣﺎ ﺭﺍ ﺍﺣﺎﻃﻪ ﻛﺮﺩﻩ ﺑﻪ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺳﺮﺍﻳﺖ ﻧﻜﻨﺪ، ﺳﻌﯽ ﻛﻨﻴﻢ ﻛﻪ ﺭﻭﺍﺑﻂ ﺳﺎﻟﻢ ﻭ ﺩﻭﺳﺘﺎﻧﻪ‌ﺍﯼ ﺑﺎ ﺁﻥ‌ﻫﺎ  ﺑﺮﻗﺮﺍﺭ ﻛﻨﻴﻢ. ﻓﻘﻂ ﻛﺎﻓﯽ ﻧﻴﺴﺖ ﻛﻪ ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﻢ ﺳﺎﺩﻳﺴﻢ ﺭﺍ ﻳﺎﺩ ﻧﮕﻴﺮﻧﺪ: ﺑﺎﻳﺪ ﺑﻪ ﺁﻥ‌ﻫﺎ  ﺑﻴﺎﻣﻮﺯﻳﻢ ﻛﻪ ﺍﺭﺯﺵ‌ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﻭ ﻣﺜﺒﺖ ﻣﺤﺒﺖ ﭼﻴﺴﺖ، ﺑﻪ ﺁﻥ‌ﻫﺎ  ﺑﻴﺎﻣﻮﺯﻳﻢ ﻛﻪ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺩﻭﺳﺖ ﺑﺪﺍﺭﻧﺪ ﻭ ﺍﺯ ﺧﻮﺷﺤﺎﻟﯽ ﺩﻳﮕﺮﺍﻥ ﺧﻮﺷﺤﺎﻝ ﺑﺎﺷﻨﺪ. ﺑﻪ ﺁﻥ‌ﻫﺎ  ﺑﻴﺎﻣﻮﺯﻳﻢ ﻛﻪ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﻣﺜﻞ ﺧﻮﺩﺷﺎﻥ ﺩﻭﺳﺖ ﺑﺪﺍﺭﻧﺪ. ﺑﮕﻮﻳﻴﻢ ﻛﻪ ﺧﺮﺍﺏ ﻛﺮﺩﻥ ﻣﺤﻴﻂ ﺯﻳﺴﺖ، ﻧﮋﺍﺩ‌ﭘﺮﺳﺘﯽ ﻭ ﻓﻘﺮ ﭼﻴﺰﻫﺎﯼ ﺑﺪﯼ ﻫﺴﺘﻨﺪ.

ﺩﺭ ﺻﻮﺭﺗﯽ ﺣﺲ ﺍﻧﺴﺎﻥ‌ﺩﻭﺳﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﻓﺮﺯﻧﺪﺍﻥ ﻣﺎ ﺭﺷﺪ ﻛﻨﺪ ﻛﻪ ﺁﻥ‌ﻫﺎ  ﻋﺸﻖ ﻣﺎﺩﺭﯼ ﻭ ﭘﺪﺭﯼ ﻣﺎ ﺭﺍ ﺩﺭﻳﺎﻓﺖ ﻛﺮﺩﻩ ﺑﺎﺷﻨﺪ. ﺍﮔﺮ ﻣﺎ ﺣﺮﻓﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻋﺸﻖ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﻢ ﻛﻪ ﻓﺮﺯﻧﺪﺍﻧﯽ ﺻﻠﺢ‌ﻃﻠﺐ ﻭ ﻣﺘﻜﯽ ﺑﻪ ﻧﻔﺲ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺷﺖ ﻛﻪ ﻫﻴﭻ‌ﮔﺎﻩ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﺧﺸﻨﯽ ﻧﺨﻮﺍﻫﻨﺪ ﺷﺪ.
