ﺩﺭﺱ ﺳﻴﺰﺩﻫﻢ- ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ ﻫﺎﯼ ﺧﻮﺏ

ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﺎﺭﻫﺎ ﺩﺭ ﻣﻮﺭﺩ ﺍﺳﺒﺎﺏ‌‌ﺑﺎﺯﯼ ﻭ ﺍﻫﻤﻴﺖ ﺁﻥ‌ﻫﺎ ﺣﺮﻑ ﺯﺩﻳﻢ. ﭘﻴﺶ ﺍﺯ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ، ﺑﻴﺎﻳﻴﺪ ﻣﺮﻭﺭﯼ ﮐﻨﻴﻢ ﺑﺮ ﺍﻳﻦ ﮐﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺧﻮﺏ ﮐﺪﺍﻡ‌ﻫﺎ ﻫﺴﺘﻨﺪ.

ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺧﻮﺏ ﻗﺒﻞ ﺍﺯ ﻫﺮ ﭼﻴﺰ ﻧﺒﺎﻳﺪ ﺧﻄﺮﻧﺎﮎ ﺑﺎﺷﺪ؛ ﻟﺒﻪ‌ﻫﺎﯼ ﺗﻴﺰ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺭﻧﮓ ﺁﻥ ﺳﻤﯽ ﻧﺒﺎﺷﺪ. ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺧﻮﺏ ﻣﺤﻜﻢ ﺍﺳﺖ ﻭ ﺩﺭ ﺻﻮﺭﺕ ﺷﻜﺴﺘﻦ ﻗﻄﻌﺎﺗﺶ ﺧﻄﺮﻧﺎﮎ ﻧﻴﺴﺖ ﻭ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎﻋﺚ ﺑﺮﻳﺪﮔﯽ ﺷﻮﺩ. ﻳﮏ ﭼﻨﻴﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ ﺷﺎﻣﻞ ﻗﻄﻌﺎﺕ ﺑﺴﻴﺎﺭ ﺭﻳﺰ ﻧﻴﺴﺖ، ﭼﺮﺍ ﻛﻪ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻛﻮﺩﮎ ﺑﺎ ﺑﻪ ﺩﻫﺎﻥ ﮔﺬﺍﺷﺘﻦ ﺁﻥ ﻗﻄﻌﺎﺕ ﻣﻤﻜﻦ ﺍﺳﺖ ﺩﭼﺎﺭ ﺧﻔﮕﯽ ﺷﻮﺩ.

ﺧﺼﻠﺖ ﺩﻳﮕﺮ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺧﻮﺏ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺯﻭﺩ ﺧﺮﺍﺏ ﻧﻤﯽ‌ﺷﻮﺩ. ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ، ﺑﺴﺘﻪ ﺑﻪ ﺳﻦ ﻭ ﺷﺨﺼﻴﺖ ﻛﻮﺩﻛﯽ ﻛﻪ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﻛﻨﺪ ﻣﺪﺕ ﻧﺴﺒﺘﺎ ﻃﻮﻻﻧﯽ‌ﺍﯼ ﻋﻤﺮ ﻣﯽ‌ﻛﻨﺪ. ﻭﻟﯽ ﻣﻌﻤﻮﻻ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﭘﻼﺳﺘﻴﻜﯽ ﻭ ﺁﻫﻦ ﺳﻔﻴﺪ ﺍﺯ ﺑﻘﻴﻪ ﻋﻤﺮ ﻛﻤﺘﺮﯼ ﺩﺍﺭﻧﺪ: ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﭼﻮﺑﯽ ﻭ ﻳﺎ ﺁﻥ‌ﻫﺎﻳﯽ ﻛﻪ ﺍﺯ ﻓﻠﺰﺍﺕ ﻣﺤﻜﻢ ﺩﺭﺳﺖ ﺷﺪﻩ‌ﺍﻧﺪ ﻋﻤﺮ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﻧﺪ. ﺑﺎ ﺗﻤﺎﻡ ﺍﻳﻦ‌ﻫﺎ ﺑﺎﻳﺪ ﺩﺍﻧﺴﺖ ﻛﻪ ﭼﻪ ﻧﻮﻉ ﭼﻮﺑﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻛﻨﻴﻢ. ﻣﻨﻈﻮﺭ ﺁﻥ ﺍﺳﺒﺎ‌ﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺍﺳﺖ ﻛﻪ ﺑﺴﻴﺎﺭ ﮔﺮﺍﻥ ﻗﻴﻤﺖ ﻫﺴﺘﻨﺪ ﻭ ﺑﻌﺪ ﺍﺯ ﺩﻭ ﺳﻪ ﺑﺎﺭ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﭘﻮﺳﺘﻪ ﭘﻮﺳﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﻣﯽ‌ﺷﻜﻨﻨﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺎﻳﺪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﻛﻪ ﺑﺎ ﭼﻮﺏ‌ﻫﺎﯼ ﻣﺤﻜﻢ ﺳﺎﺧﺘﻪ ﺷﺪﻩ‌ﺍﻧﺪ ﺗﻬﻴﻪ ﻛﻨﻴﺪ. ﺍﺯ ﻳﮏ ﻣﻬﺪ‌ﻛﻮﺩﮎ ﺩﻳﺪﻥ ﻛﻨﻴﺪ ﻭ ﺩﺭ ﺁﻧﺠﺎ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﺭﺍ ﻛﻪ ﻋﻤﺮﯼ ﻃﻮﻻﻧﯽ ﺩﺍﺭﻧﺪ ﺧﻮﺍﻫﻴﺪ ﺷﻨﺎﺧﺖ. ﺍﺯ ﺍﻳﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎ ﺩﺭ ﺍﻳﻦ ﻣﻬﺪﻛﻮﺩﮎ‌ﻫﺎ ﺑﻴﺸﺘﺮﻳﻦ ﺍﺳﺘﻔﺎﺩﻩ‌ﻫﺎ ﺭﺍ ﻣﯽ‌ﻛﻨﻨﺪ ﻭ ﻭﺍﻗﻌﺎ ﺑﻼﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺑﻪ ﺳﺮﺷﺎﻥ ﻣﯽ‌ﺁﻭﺭﻧﺪ ﻭ ﺑﺎ ﺍﻳﻦ ﻫﻤﻪ ﭼﻨﺪﻳﻦ ﺳﺎﻝ ﻋﻤﺮ ﻣﯽ‌ﻛﻨﻨﺪ.

ﺁﻥ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ ﺧﻮﺏ ﺍﺳﺖ ﻛﻪ ۹۰ ﺩﺭﺻﺪ ﺑﺎﺯﯼ ﺍﺯ ﺑﭽﻪ ﺑﺎﺷﺪ ﻭ ۱۰ ﺩﺭﺻﺪﺵ ﺍﺯ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ. ﺍﮔﺮ ۹۰ ﺩﺭﺻﺪ ﺑﺎﺯﯼ ﺍﺯ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺑﺎﺷﺪ ﻭ ۱۰ ﺩﺭﺻﺪ ﺍﺯ ﺑﭽﻪ، ﺁﻥ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ، ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺧﻮﺑﯽ ﻧﻴﺴﺖ.
ﺑﻴﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ ﻛﻪ ﺑﺎ ﻓﺸﺎﺭ ﻳﮏ ﺩﮐﻤﻪ ﻛﻨﺘﺮﻝ ﺍﺯ ﺭﺍﻩ ﺩﻭﺭ ﻛﺎﺭ ﻣﯽ‌ﻛﻨﺪ ﻭ ﻣﻜﻌﺐ‌ﻫﺎﯼ ﭼﻮﺑﯽ ﺗﻔﺎﻭﺕ ﻗﺎﺋﻞ ﺷﻮﻳﺪ. ﻣﺜﻼ ﻳﮏ ﺳﮕﯽ ﻛﻪ ﺑﺎ ﺑﺎﻃﺮﯼ ﻛﺎﺭ ﻣﯽ‌ﻛﻨﺪ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ. ﻭﻗﺘﯽ ﻳﮏ ﺩﮐﻤﻪ  ﺭﺍ ﻓﺸﺎﺭ ﻣﯽ‌ﺩﻫﻴﺪ ﺳﮓ ﺭﻭﯼ ﺯﻣﻴﻦ ﺭﺍﻩ ﻣﯽ‌ﺭﻭﺩ، ﻭﺍﻕ‌ﻭﺍﻕ ﻣﯽ‌ﻛﻨﺪ ﻭ ﺩﻣﺶ ﺭﺍ ﺗﻜﺎﻥ ﻣﯽ‌ﺩﻫﺪ: ۹۰ ﺩﺭﺻﺪ ﺑﺎﺯﯼ ﺩﺭ ﺍﻳﻨﺠﺎ ﺍﺯ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺍﺳﺖ. ﻛﻮﺩﮎ ﺗﻨﻬﺎ ﻛﺎﺭﯼ ﻛﻪ ﻣﯽ‌ﻛﻨﺪ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺩﮐﻤﻪ ﺭﺍ ﻓﺸﺎﺭ ﻣﯽ‌ﺩﻫﺪ.
ﺑﺎ ﻳﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻛﻪ ﺷﺎﻣﻞ ﻣﻜﻌﺐ‌ﻫﺎﯼ ﭼﻮﺑﯽ ﺧﺎﻧﻪ‌ﺳﺎﺯﯼ ﺍﺳﺖ، ﻭﺿﻌﻴﺖ ﻛﺎﻣﻼ ﻣﺘﻔﺎﻭﺕ ﺍﺳﺖ: ﺩﺭ ﺍﻳﻦ ﺑﺎﺯﯼ ۹۰ ﺩﺭﺻﺪ ﺑﺎﺯﯼ ﺍﺯ ﻃﻔﻞ ﺍﺳﺖ. ﺍﻳﻦ ﺍﺳﺒﺎﺏ‌‌ﺑﺎﺯﯼ ﻓﻘﻂ ﺑﺮﺍﯼ ﻳﮏ ﻛﺎﺭ ﻧﻴﺴﺖ. ﺍﻣﻜﺎﻧﺎﺕ ﺑﺎﺯﯼ ﺑﺎ ﻳﮏ ﭼﻨﻴﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ ﺗﻘﺮﻳﺒﺎ ﻧﺎﻣﺤﺪﻭﺩ ﺍﺳﺖ.

ﻭﻱ‍‍ﮊﮔﯽ ﺩﻳﮕﺮ ﻳﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺧﻮﺏ ﻛﻪ ﻭﺍﻟﺪﻳﻦ ﺑﺎﻳﺪ ﺑﻪ ﺁﻥ ﺗﻮﺟﻪ ﻛﻨﻨﺪ، ﺧﺼﻠﺘﯽ ﻛﺎﻣﻼ ﻣﺘﻔﺎﻭﺕ ﺑﺎ ﺩﻭ ﺗﺎﯼ ﺩﻳﮕﺮ ﺍﺳﺖ. ﻫﺮ ﭼﻘﺪﺭ ﻛﻮﺩﮎ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺑﻴﺸﺘﺮ ﻓﻌﺎﻟﻴﺖ ﻛﻨﺪ ﻭ ﻫﺮ ﭼﻘﺪﺭ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻛﻤﺘﺮ ﺑﺮﺍﯼ ﺍﻭ ﻛﺎﺭ ﻛﻨﺪ، ﻫﻤﺎﻧﻘﺪﺭ ﺧﻮﺩ ﺍﻧﮕﺎﺭﻩ ﻃﻔﻞ ﻭ ﻗﻮﻩ ﺍﺑﺘﻜﺎﺭﺵ ﺑﻴﺸﺘﺮ ﺭﺷﺪ ﻣﯽ‌ﻛﻨﺪ ﻭ ﭼﻴﺰﻫﺎﯼ ﺑﻴﺸﺘﺮﯼ ﺍﺯ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ. ﻫﺮ ﭼﻘﺪﺭ ﻛﻮﺩﮎ ﻛﻤﺘﺮ ﻓﻌﺎﻟﻴﺖ ﻛﻨﺪ ﻭ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺑﻪ ﺟﺎﯼ ﺍﻭ ﻛﺎﺭ ﻛﻨﺪ، ﻫﻤﺎﻥ‌ﻗﺪﺭ ﻛﻤﺘﺮ ﺍﺯ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﭼﻴﺰ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﺧﺼﻠﺖ‌ﻫﺎﯼ ﺧﻮﺏ ﺩﺭ ﺍﻭ ﻛﻤﺘﺮ ﺭﺷﺪ ﻣﯽ‌ﻛﻨﺪ.

ﺍﮔﺮ ﺍﻳﻦ ﺍﺻﻞ ﺭﺍ ﺩﺭﮎ ﻛﺮﺩﻳﺪ، ﺁﻥ ﮔﺎﻩ ﺧﻮﺍﻫﻴﺪ ﻓﻬﻤﻴﺪ ﻛﻪ ﭼﺮﺍ ﻳﮏ ﺟﻌﺒﻪ ﺑﺰﺭﮒ ﺧﺎﻟﯽ ﻛﻪ ﻛﻮﺩﮎ ﺑﺘﻮﺍﻧﺪ ﺍﺯ ﺁﻥ ﺑﺎﻻ ﺑﺮﻭﺩ ﻳﺎ ﺗﻮﯼ ﺁﻥ ﺑﺮﻭﺩ ﻳﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻋﺎﻟﯽ ﺑﻪ‌ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ. ﺟﻌﺒﻪ ﺑﻴﺸﺘﺮ ﺍﺯ ﻳﮏ ﭼﻴﺰ ﻣﺠﺮﺩ ﻭ ﺧﺎﺹ ﺍﺳﺖ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺍﺯ ﭼﻴﺰﻫﺎ ﺑﺎﺷﺪ: ﻛﺸﺘﯽ، ﺧﺎﻧﻪ، ﺯﻳﺮﺩﺭﻳﺎﻳﯽ، ﻛﻠﺒﻪ، ﻫﻮﺍﭘﻴﻤﺎ، ﺁﺩﻡ‌ﺁﻫﻨﯽ ﻭ ﺧﻴﻠﯽ ﭼﻴﺰﻫﺎﯼ ﺩﻳﮕﺮ ﻛﻪ ﻗﺪﺭﺕ ﺍﺑﺘﻜﺎﺭ ﻛﻮﺩﮎ ﺭﺍ ﺑﺮﻣﯽ‌ﺍﻧﮕﻴﺰﺩ. ﻛﻮﺩﮎ ﺑﺮﺣﺴﺐ ﺗﺼﻮﺭﺍﺕ ﻭ ﺗﺨﻴﻼﺗﺶ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﺪ. ﻣﺜﻼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺑﻌﻀﯽ ﺟﺎﻫﺎﻳﺶ ﻛﻪ ﺩﻟﺶ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺳﻮﺭﺍﺥ‌ﻫﺎﻳﯽ ﺑﺒﺮﺩ. ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺭﻭﯼ ﺁﻥ ﻧﻘﺎﺷﯽ ﻛﻨﺪ. ﻭﺍﻗﻌﺎ ﻫﻴﭻ‌ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺩﻳﮕﺮﯼ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﻳﮏ ﺟﻌﺒﻪ ﺑﺰﺭﮒ ﺧﺎﻟﯽ ﺟﺎﻟﺐ ﻧﻴﺴﺖ، ﻭ ﺑﺎ ﺗﻤﺎﻡ ﺍﻳﻦ‌ﻫﺎ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﻴﺪ ﻛﻪ ﭼﻘﺪﺭ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺑﻪ ﺍﻳﻦ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﻨﺪ ﻛﻪ ﺑﻪ ﺩﺭ ﻣﻐﺎﺯﻩ‌ﺍﯼ ﺑﺮﻭﻧﺪ ﻭ ﻳﮏ ﺟﻌﺒﻪ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺍﺯ ﺁﻧﺠﺎ ﺑﮕﻴﺮﻧﺪ، ﺩﺭ ﺻﻮﺭﺗﯽ‌ﻛﻪ ﺍﻳﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻛﻮﭼﮏ‌ﺗﺮﻳﻦ ﺧﺮﺟﯽ ﻫﻢ ﺑﺮﺍﯼ ﺁﻥ‌ﻫﺎ ﻧﺪﺍﺭﺩ؟


ﭼﻬﺎﺭﻣﺎ ﻳﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺧﻮﺏ ﺑﺎﻳﺪ ﺳﺮﮔﺮﻡ‌ﻛﻨﻨﺪﻩ ﻭ ﺟﺎﻟﺐ ﺑﺎﺷﺪ. ﻣﻤﻜﻦ ﺍﺳﺖ ﻳﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺁﻣﻮﺯﻧﺪﻩ ﺑﺎﺷﺪ ﻭﻟﯽ ﺍﮔﺮ ﺟﺎﻟﺐ ﻭﺳﺮﮔﺮﻡ‌ﻛﻨﻨﺪﻩ ﻧﺒﺎﺷﺪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺧﻮﺑﯽ ﻧﻴﺴﺖ. ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺑﻴﻦ ﺳﺮﮔﺮﻡ‌ﻛﻨﻨﺪﻩ ﺩﺭ ﻛﻮﺗﺎﻩ ﻣﺪﺕ ﻭ ﺩﺭﺍﺯ ﻣﺪﺕ ﻓﺮﻕ ﻗﺎﺋﻞ ﺷﻮﻳﻢ.

ﻣﺴﻠﻢ ﺍﺳﺖ ﻛﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ ﻛﻪ ﺑﺎ ﻛﻨﺘﺮﻝ ﺍﺯ ﺭﺍﻩ ﺩﻭﺭ ﻛﺎﺭ ﻣﯽ‌ﻛﻨﺪ ﻭ ﺑﺎ ﻓﺸﺎﺭ ﻳﮏ ﺩﮐﻤﻪ  ﺑﻪ ﺭﺍﻩ ﻣﯽ‌ﺍﻓﺘﺪ، ﺩﺭ ﻛﻮﺩﮎ ﺧﺮﺩﺳﺎﻝ ﺧﻮﺷﺤﺎﻟﯽ ﺑﻼﻓﺎﺻﻠﻪ ﺯﻳﺎﺩﯼ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﻛﻨﺪ. ﺍﻭ ﺑﺎ ﻓﺸﺎﺭ ﺩﺍﺩﻥ ﺩﮐﻤﻪ ﺣﻴﻮﺍﻥ ﻭ ﻳﺎ ﻣﺎﺷﻴﻨﯽ ﺭﺍ ﻛﻪ ﺩﻭﺭ ﺍﺗﺎﻕ ﺑﻪ ﺣﺮﻛﺖ ﺩﺭﻣﯽ‌ﺁﻳﺪ ﺑﺎ ﺷﮕﻔﺘﯽ ﻧﮕﺎﻩ ﻣﯽ‌ﻛﻨﺪ ﻭ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﺩ.

ﺑﺎ ﺍﻳﻦ‌ﻫﻤﻪ ﺑﻌﺪ ﺍﺯ ﭼﻨﺪ ﺳﺎﻋﺖ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﺑﺎ ﻳﮏ ﭼﻨﻴﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ، ﺍﻳﻦ ﺗﻮﺟﻪ ﻛﻢ‌ﻛﻢ ﺍﺯ ﺑﻴﻦ ﻣﯽ‌ﺭﻭﺩ ﻭ ﻛﺸﺶ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﻛﻮﺩﮎ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﺪ. ﺩﺭ ﻋﻮﺽ ﻣﻜﻌﺐ‌ﻫﺎﯼ ﭼﻮﺑﯽ ﺧﺎﻧﻪ‌ﺳﺎﺯﯼ، ﺭﻭﺯﻫﺎﯼ ﻣﺘﻤﺎﺩﯼ ﺍﻭ ﺭﺍ ﺳﺮﮔﺮﻡ ﻣﯽ‌ﻛﻨﺪ؛ ﻋﻤﻞ ﻳﮏ ﭼﻨﻴﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ ﺩﺭ ﻃﻮﻻﻧﯽ ﻣﺪﺕ ﺍﺳﺖ.

ﭘﻨﺠﻢ ﺍﻳﻨﻜﻪ، ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺑﺎﻳﺪ ﻣﻨﺎﺳﺐ ﺳﻦ ﻭ ﺭﺷﺪ ﻛﻮﺩﮎ ﺑﺎﺷﺪ. ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻨﺪ ﻛﻪ ﺩﺭ ﻫﺮ ﻣﺮﺣﻠﻪ ﺍﺯ ﺭﺷﺪ ﻓﺮﺯﻧﺪﺷﺎﻥ ﭼﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ ﺑﺮﺍﯼ ﺍﻭ ﻣﻨﺎﺳﺐ ﺍﺳﺖ. ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺁﺧﺮﻳﻦ ﺩﺭﺱ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭﺳﻨﺎﻣﻪ ﻣﺮﺍﺟﻌﻪ ﻛﻨﻴﺪ، ﻭﻟﯽ ﻳﮏ ﻣﺎﺩﺭ ﺑﻪ ﭼﻴﺰﯼ ﺑﻴﺶ ﺍﺯ ﻳﮏ ﺩﺭﺱ ﻳﺎ ﻛﺘﺎﺏ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﺩ. ﺍﻭ ﺑﺎﻳﺪ ﻓﺮﺯﻧﺪﺵ ﺭﺍ ﺑﺸﻨﺎﺳﺪ، ﺑﺪﺍﻧﺪ ﻛﻪ ﭼﻪ ﭼﻴﺰﯼ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﻭ ﭼﻪ ﭼﻴﺰﯼ ﺭﺍ ﺩﻭﺳﺖ ﻧﺪﺍﺭﺩ. ﻫﺮ ﻣﺎﺩﺭﯼ ﺑﺎﻳﺪ ﻓﻜﺮ ﻛﻨﺪ: «ﺁﻳﺎ ﺍﻳﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻣﻨﺎﺳﺐ ﺳﻦ ﻭ ﺭﺷﺪ ﺍﻭ ﻫﺴﺖ؟ ﺁﻳﺎ ﺍﻳﻦ ﺍﺯ ﺁﻥ ﻧﻮﻉ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﻳﯽ ﻫﺴﺖ ﻛﻪ ﺍﻭ ﺩﻭﺳﺖ ﺩﺍﺭﺩ؟ ﺁﻳﺎ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻣﻨﺎﺳﺐ ﻗﺪ ﺍﻭ ﻫﺴﺖ؟ ﺁﻳﺎ ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺁﻥ ﺭﺍ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺑﻪ ﺩﺳﺖ ﺑﮕﻴﺮﺩ ﻳﺎ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ ﺑﻪ ﺍﻭ ﻛﻤﮏ ﻛﺮﺩ؟»

ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻣﯽ‌ﺧﺮﻳﺪ ﺑﻪ‌ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻛﻪ ﻫﺮ ﭼﻘﺪﺭ ﻛﻮﺩﮎ ﻛﻮﭼﻚ‌ﺗﺮ ﺑﺎﺷﺪ، ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﺍﯼ ﻛﻪ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﻛﻨﻴﺪ ﺑﺎﻳﺪ ﺑﺰﺭﮒ‌ﺗﺮ ﺑﺎﺷﺪ. ﻣﺜﻼ ﻛﻮﺩﻛﺎﻥ ﺧﺮﺩﺳﺎﻝ ﺩﺭ ﻣﻘﺎﻳﺴﻪ ﺑﺎ ﻛﻮﺩﻛﺎﻥ ﺑﺰﺭﮒ‌ﺗﺮ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﻣﻜﻌﺐ‌ﻫﺎﯼ ﺧﺎﻧﻪ‌ﺳﺎﺯﯼ ﺑﺰﺭﮒ‌ﺗﺮﯼ ﺩﺍﺭﻧﺪ، ﻣﺪﺍﺩﻫﺎﻳﺸﺎﻥ ﺑﺎﻳﺪ ﺑﺰﺭﮒ‌ﺗﺮ ﺑﺎﺷﺪ، ﻗﻠﻢ‌‌ﻣﻮﯼ ﻧﻘﺎﺷﯽ‌ﺷﺎﻥ ﺑﺎﻳﺪ ﺑﺰﺭﮔﺘﺮ ﺑﺎﺷﺪ.

ﮔﺎﻫﯽ ﺍﻭﻗﺎﺕ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺧﻮﺑﯽ ﻛﻪ ﻣﯽ‌ﺑﻴﻨﻨﺪ ﺑﻪ ﻋﻠﺖ ﮔﺮﺍﻧﻴﺸﺎﻥ ﻋﻘﺐ‌ﻧﺸﻴﻨﯽ ﻣﯽ‌ﻛﻨﻨﺪ. ﺑﺎﻳﺪ ﻣﻨﻈﻮﺭ ﺍﺯ ﺧﻴﻠﯽ ﮔﺮﺍﻥ ﺭﺍ ﺗﻮﺿﻴﺢ ﺩﺍﺩ. ﻣﻤﻜﻦ ﺍﺳﺖ ﻛﻪ ﻭﻗﺘﯽ ﻳﮏ ﻛﺎﻣﻴﻮﻥ ﺑﺰﺭﮒ ﭘﻼﺳﺘﻴﻜﯽ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﯽ‌ﺧﺮﻳﺪ ﻭ ﻫﺰﺍﺭ ﺗﻮﻣﺎﻥ ﺑﺮﺍﻳﺶ ﭘﻮﻝ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﺪ، ﺩﺭ ﻧﻈﺮ ﺍﻭﻝ ﺑﺮﺍﻳﺘﺎﻥ ﺍﺭﺯﺍﻥ ﺑﺎﺷﺪ. ﻭﻟﯽ ﺍﮔﺮ ﻛﻮﺩﮎ ﻣﻮﻓﻖ ﺷﺪ ﻛﻪ ﺑﻌﺪ ﺍﺯ ﭘﻨﺞ ﺩﻗﻴﻘﻪ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﺁﻥ ﺭﺍ ﺑﺸﻜﻨﺪ، ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺍﺭﺯﺍﻥ، ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻧﻴﺴﺖ.

ﺍﮔﺮ ﻣﺎﺩﺭﺍﻥ ﻗﻴﻤﺖ ﺗﻤﺎﻡ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﭘﻼﺳﺘﻴﻜﯽ ﺍﺭﺯﺍﻥ ﻗﻴﻤﺘﯽ ﺭﺍ ﻛﻪ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﻣﯽ‌ﺧﺮﻧﺪ ﺟﻤﻊ ﻣﯽ‌ﻛﺮﺩﻧﺪ، ﺣﺘﻤﺎ ﺍﺯ ﺟﻤﻊ ﻣﺒﻠﻐﯽ ﻛﻪ ﭘﺮﺩﺍﺧﺘﻪ‌ﺍﻧﺪ ﺗﻌﺠﺐ ﻣﯽ‌ﻛﺮﺩﻧﺪ. ﺑﺎ ﻫﻤﻴﻦ ﻣﺒﻠﻎ ﺁﻥ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﺴﺘﻨﺪ ﺑﻌﻀﯽ ﺍﺯ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ «ﮔﺮﺍﻥ» ﻭﻟﯽ ﻣﺤﻜﻤﯽ ﺭﺍ ﺑﺨﺮﻧﺪ ﻛﻪ ﻫﻢ ﺑﺮﺍﯼ ﺭﺷﺪ ﻃﻔﻞ ﻣﻨﺎﺳﺐ ﻫﺴﺘﻨﺪ ﻭ ﻫﻢ ﻗﻮﻩ ﺧﻼﻗﻪ ﺭﺍ ﺩﺭ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺭﺷﺪ ﻣﯽ‌ﺩﻫﺪ.ﻗﻴﻤﺖ ﭘﻨﺞ ﺍﺳﺒﺎ‌ﺏ‌ﺑﺎﺯﯼ ﭘﻼﺳﺘﻴﻜﯽ ﺍﺭﺯﺍﻥ ﻣﻌﺎﺩﻝ ﻗﻴﻤﺖ ﻳﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻣﺤﻜﻢ ﭼﻮﺑﯽ ﻳﺎ ﻓﻠﺰﯼ ﻣﺤﻜﻢ ﺍﺳﺖ ﻛﻪ ﻣﺪﺕ ﺯﻳﺎﺩﯼ ﻋﻤﺮ ﻣﯽ‌ﻛﻨﺪ.

ﻓﺮﺍﻣﻮﺵ ﻧﻜﻨﻴﺪ ﻛﻪ ﻣﻬﻢ‌ﺗﺮﻳﻦ «ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌»ﺍﯼ ﻛﻪ ﻛﻮﺩﮎ ﺩﺭ ﺗﻤﺎﻡ ﻋﻤﺮﺵ ﺩﺭﻳﺎﻓﺖ ﻣﯽ‌ﻛﻨﺪ، «ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ»ﺍﯼ ﻛﻪ ﺑﻴﺶ ﺍﺯ ﻫﺮ ﭼﻴﺰ ﺩﻳﮕﺮﯼ ﺩﻭﺳﺘﺶ ﺩﺍﺭﺩ، «ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ»ﺍﯼ ﻛﻪ ﺑﻴﺶ ﺍﺯ ﻫﺮ ﭼﻴﺰﯼ ﺑﺮ ﺍﻭ ﺍﺛﺮ ﺧﻮﺍﻫﺪ ﻛﺮﺩ، ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ ﻳﻜﺘﺎﻳﯽ ﺍﺳﺖ ﻛﻪ ﺍﺯ ﭘﻮﺳﺖ ﻭ ﮔﻮﺷﺖ ﻭ ﺍﺳﺘﺨﻮﺍﻥ ﺩﺭﺳﺖ ﺷﺪﻩ ﺍﺳﺖ، ﻳﻌﻨﯽ ﺷﻤﺎ! ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﺑﺎ ﺷﻤﺎ ﺑﻴﺶ ﺍﺯ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﺑﺎ ﻫﺮ ﭼﻴﺰ ﺩﻳﮕﺮﯼ ﺑﺮﺍﯼ ﺍﻭ ﺍﺭﺯﺵ ﺩﺍﺭﺩ.

ﺻﺪﻫﺎ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺧﺎﺭﻕ‌ﺍﻟﻌﺎﺩﻩ ﻭ ﭘﺮﺯﺭﻕ‌ﻭ‌ﺑﺮﻕ ﺩﺭ ﻣﻘﺎﻳﺴﻪ ﺑﺎ ﻟﺤﻈﺎﺗﯽ ﻛﻪ ﺑﺮﺍﯼ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﺑﺎ ﺍﻭ ﻣﯽ‌ﮔﺬﺭﺍﻧﻴﺪ ﻭ ﺍﻭ ﺭﺍ ﺩﺭ ﺯﻣﻴﻨﻪ ﻋﺎﻃﻔﯽ ﺍﺭﺿﺎ ﻣﯽ‌ﻛﻨﻴﺪ ﻫﻴﭻ ﺍﺳﺖ.

ﺗﻠﻮﻳﺰﻳﻮﻥ ﺑﻪ ﺭﻏﻢ ﺗﻤﺎﻣﯽ ﺑﺪﯼ‌ﻫﺎﻳﺶ، ﻳﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺁﻣﻮﺯﻧﺪﻩ ﺑﺴﻴﺎﺭ ﻋﺎﻟﯽ ﺍﺳﺖ ﻭ ﻧﺒﺎﻳﺪ ﺣﺎﻟﺖ ﻣﻨﻔﻌﻠﯽ ﻧﺴﺒﺖ ﺑﻪ ﺁﻥ ﮔﺮﻓﺖ، ﻛﺎﺭﯼ ﻛﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﻣﯽ‌ﻛﻨﻨﺪ ﻭ ﺑﻪ ﻧﻈﺮ ﻣﻦ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ. ﻛﻤﯽ ﻭﻗﺖ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺑﻌﻀﯽ ﺍﺯ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎ ﺭﺍ ﺑﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻧﮕﺎﻩ ﻛﻨﻴﺪ، ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺧﻮﺏ ﺭﺍ ﺧﻮﺍﻫﻴﺪ ﺷﻨﺎﺧﺖ.
ﻫﻤﺎﻧﻄﻮﺭ ﻛﻪ ﻗﺒﻼ ﻫﻢ ﮔﻔﺘﻢ، ﻣﻦ ﺑﻪ ﺍﻳﻦ ﻣﺴﺌﻠﻪ ﻭﺍﻗﻔﻢ ﻛﻪ ﺍﻏﻠﺐ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺍﺯ ﻛﻴﻔﻴﺖ ﭘﺎﻳﻴﻨﯽ ﺑﺮﺧﻮﺭﺩﺍﺭﻧﺪ ﻭ ﺁﺭﺯﻭ ﺩﺍﺭﻡ ﻛﻪ ﺗﺎ ﺁﻧﺠﺎ ﻛﻪ ﻣﻤﻜﻦ ﺍﺳﺖ ﺍﺯ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺧﺸﻮﻧﺖ ﺩﺭ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺟﻠﻮﮔﻴﺮﯼ ﺷﻮﺩ. ﺑﺎ ﺍﻳﻦ ﻫﻤﻪ ﺑﻪ ﺭﻏﻢ ﺗﻤﺎﻡ ﺍﺷﻜﺎﻻﺗﯽ ﻛﻪ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺩﺍﺭﺩ، ﻧﺒﺎﻳﺪ ﺍﺭﺯﺵ‌ﻫﺎﯼ ﻣﺜﺒﺘﯽ ﻛﻪ ﻧﮕﺎﻩ ﻛﺮﺩﻥ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺑﺮﺍﯼ ﻛﻮﺩﮎ ﺩﺭﺑﺮﺩﺍﺭﺩ ﻧﺎﺩﻳﺪﻩ ﮔﺮﻓﺖ. ﻣﺜﻼ ﺩﺭ ﺯﻣﻴﻨﻪ ﺭﺷﺪ ﺯﺑﺎﻥ ﻛﻮﺩﮎ ﻭ ﺭﺷﺪ ﮔﻨﺠﻴﻨﻪ ﻟﻐﺎﺕ ﺍﻭ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺩﺍﺭﺍﯼ ﺍﺛﺮﺍﺕ ﺑﺴﻴﺎﺭ ﻣﺜﺒﺘﯽ ﺍﺳﺖ. ﺗﺤﻘﻴﻘﺎﺕ ﻧﺸﺎﻥ ﺩﺍﺩﻩ‌ﺍﻧﺪ ﻛﻪ ﺍﺯ ﺯﻣﺎﻥ ﻫﻤﻪ‌ﮔﻴﺮ ﺷﺪﻥ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺩﺭ ﺧﺎﻧﻮﺍﺩﻩ‌ﻫﺎﯼ ﺁﻣﺮﻳﻜﺎﻳﻲ، ﮔﻨﺠﻴﻨﻪ ﻟﻐﺎﺕ ﻛﻮﺩﻛﺎﻧﯽ ﻛﻪ ﺑﻪ ﻛﻮﺩﻛﺴﺘﺎﻥ ﻣﯽ‌ﺭﻭﻧﺪ ﺑﺴﻴﺎﺭ ﻏﻨﯽ‌ﺗﺮ ﺷﺪﻩ ﺍﺳﺖ. ﻧﺒﺎﻳﺪ ﺟﻨﺒﻪ‌ﻫﺎﯼ ﻣﺜﺒﺖ ﺗﻠﻮﻳﺰﻳﻮﻥ ﺭﺍ ﻛﻤﺘﺮ ﺍﺯ ﺁﻧﭽﻪ ﻫﺴﺖ ﺍﺭﺯﻳﺎﺑﯽ ﻛﺮﺩ. ﻭ ﭼﻨﺎﻧﭽﻪ ﻛﻮﺩﮎ ﺧﺮﺩﺳﺎﻝ ﺑﺎ ﺍﻳﻦ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﻣﺘﻮﺳﻂ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮔﻨﺠﻴﻨﻪ ﻟﻐﺎﺗﺶ ﺭﺍ ﻏﻨﯽ‌ﺗﺮ ﻛﻨﺪ، ﺍﮔﺮ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎ ﺧﻮﺏ ﺑﻮﺩﻧﺪ، ﺁﻥ‌ﻭﻗﺖ ﭼﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺩﺍﺷﺖ!
ﺗﻠﻮﻳﺰﻳﻮﻥ ﺑﻪ ﺭﻏﻢ ﺗﻤﺎﻣﯽ ﺑﺪﯼ‌ﻫﺎﻳﺶ، ﻳﮏ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﺁﻣﻮﺯﻧﺪﻩ ﺑﺴﻴﺎﺭ ﻋﺎﻟﯽ ﺍﺳﺖ ﻭ ﻧﺒﺎﻳﺪ ﺣﺎﻟﺖ ﻣﻨﻔﻌﻠﯽ ﻧﺴﺒﺖ ﺑﻪ ﺁﻥ ﮔﺮﻓﺖ، ﻛﺎﺭﯼ ﻛﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﻣﯽ‌ﻛﻨﻨﺪ ﻭ ﺑﻪ ﻧﻈﺮ ﻣﻦ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ. ﻛﻤﯽ ﻭﻗﺖ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺑﻌﻀﯽ ﺍﺯ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎ ﺭﺍ ﺑﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻧﮕﺎﻩ ﻛﻨﻴﺪ، ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺧﻮﺏ ﺭﺍ ﺧﻮﺍﻫﻴﺪ ﺷﻨﺎﺧﺖ.
ﺣﻀﻮﺭ ﺷﻤﺎ، ﺗﺸﻮﻳﻖ ﻭ ﻳﺎ ﭘﺎﺩﺍﺵ ﺑﻪ‌ﺣﺴﺎﺏ ﻣﯽ‌ﺁﻳﺪ. ﻛﺎﺭﯼ ﻛﻪ ﺷﻤﺎ ﻫﻮﺷﻴﺎﺭﺍﻧﻪ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ ﺑﻪ ﺍﻭ ﻣﯽ‌ﺁﻣﻮﺯﺩ ﻛﻪ ﭼﮕﻮﻧﻪ ﺍﻧﺘﺨﺎﺏ ﻛﻨﺪ.

ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﻄﺎﻟﻌﻪ ﻛﻨﺪ ﻭ ﺍﻳﻦ ﻋﺎﺩﺕ ﺭﺍ ﺣﻔﻆ ﻛﻨﺪ، ﺷﻤﺎ، ﻳﻌﻨﯽ ﻭﺍﻟﺪﻳﻦ ﺑﺎﻳﺪ ﺍﻳﻦ ﻋﺎﺩﺕ ﺭﺍ ﺩﺭ ﺍﻭ ﺑﻪ ﻭﺟﻮﺩ ﺑﻴﺎﻭﺭﻳﺪ ﻭ ﻣﻮﺍﻇﺐ ﺑﺎﺷﻴﺪ ﻛﻪ ﺍﻳﻦ ﻋﺎﺩﺕ ﺗﺮﮎ ﻧﺸﻮﺩ ﻭ ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﺳﺎﻝ‌ﻫﺎ ﻫﺴﺘﻨﺪ.

ﺍﺯ ﺗﻤﺎﻡ ﭼﻴﺰﻫﺎﻳﯽ ﻛﻪ ﮔﻔﺘﻴﻢ ﻳﮏ ﺍﺻﻞ ﺍﺳﺎﺳﯽ ﺭﺍ ﻧﺘﻴﺠﻪ‌ﮔﻴﺮﯼ ﻣﯽ‌ﻛﻨﻴﻢ: ﺯﻳﺎﺩ ﺭﺍﺟﻊ‌ﺑﻪ ﭼﻴﺰﯼ ﻛﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﯽ‌ﺧﻮﺍﻧﺪ ﺳﺨﺖ‌ﮔﻴﺮﯼ ﻧﻜﻨﻴﺪ. ﻫﺮ ﻣﺘﻦ ﭼﺎﭘﯽ‌ﺍﯼ ﻛﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﻭﺳﺖ ﺩﺍﺭﺩ ﺑﺨﻮﺍﻧﺪ، ﺣﺎﻻ ﭼﻪ ﻛﺘﺎﺏ ﺑﺎﺷﺪ ﭼﻪ ﻣﺠﻠﻪ ﭼﻪ ﺭﻭﺯﻧﺎﻣﻪ ﻭ ﭼﻪ ﻛﺘﺎﺏ‌ﻫﺎﯼ ﻣﺼﻮﺭ، ﻫﻤﻪ‌ﺷﺎﻥ ﺑﺮﺍﻳﺶ ﻣﻔﻴﺪ ﺍﺳﺖ. ﺧﺼﻮﺻﺎ ﺩﺭ ﻣﻮﺭﺩ ﻛﻮﺩﻛﺎﻥ ﺑﺎﻻﯼ ﺷﺶ ﺳﺎﻝ. ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﻭﺍﻟﺪﻳﻦ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﻨﺪ: «ﺍﯼ ﺑﺎﺑﺎ، ﺍﻭﻥ ﻓﻘﻂ ﺧﺰﻋﺒﻼﺕ ﻣﯽ‌ﺧﻮﻧﻪ، ﻛﺘﺎﺏ‌ﻫﺎﯼ ﻣﺼﻮﺭ ﻭ ﺍﺣﻤﻘﺎﻧﻪ، ﺩﻟﻢ ﻣﯽ‌ﺧﻮﺍﺩ ﺍﻭ ﺍﺩﺑﻴﺎﺕ ﻭﺍﻗﻌﯽ ﻣﻄﺎﻟﻌﻪ ﻛﻨﻪ.» ﻧﮕﺮﺍﻥ ﻧﺒﺎﺷﻴﺪ، ﺍﻳﻦ ﺧﺰﻋﺒﻼﺗﯽ ﻛﻪ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻧﺪ، ﺣﺘﻤﺎ ﺑﺎﻳﺪ ﻳﮏ ﺍﺣﺘﻴﺎﺝ ﺭﻭﺍﻧﯽ ﺁﻧﯽ ﺍﻭ ﺭﺍ ﺍﺭﺿﺎ ﻛﻨﺪ ﻭ ﻫﻤﻴﻦ‌ﻃﻮﺭﯼ ﺧﻴﻠﯽ ﺧﻮﺏ ﺍﺳﺖ. ﻫﻤﻴﻦ‌ﻛﻪ ﺍﻭ ﻣﻄﺎﻟﻌﻪ ﻛﻨﺪ ﻛﺎﻓﯽ ﺍﺳﺖ، ﺳﻠﻴﻘﻪ‌ﺍﺵ ﻧﻴﺰ ﺑﻪ‌ﺗﺪﺭﻳﺞ ﺗﺼﻔﻴﻪ ﻭ ﺗﻠﻄﻴﻒ ﺧﻮﺍﻫﺪ ﺷﺪ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺑﻪ ﺍﺻﻄﻼﺡ ﻣﺴﺌﻠﻪ ﺍﺩﺑﻴﺎﺕ ﺑﯽ‌ﺍﺭﺯﺵ، ﺩﺭ ﺳﻨﻴﻦ ﭘﻴﺶ ﺍﺭ ﺩﺑﺴﺘﺎﻥ ﻇﺎﻫﺮ ﻧﻤﯽ‌ﺷﻮﺩ ﭼﺮﺍ ﻛﻪ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﺁﻥ ﭼﻴﺰﯼ ﺭﺍ «ﻣﯽ‌ﺧﻮﺍﻧﺪ» ﻛﻪ ﺷﻤﺎ ﺑﺮﺍﯼ ﺍﻭ ﺍﻧﺘﺨﺎﺏ ﻛﺮﺩﻩ‌ﺍﻳﺪ ﻭ ﺧﺮﻳﺪﻩ‌ﺍﻳﺪ (ﻭ ﻳﺎ ﺍﮔﺮ ﺍﺯ ﺷﺶ ﺳﺎﻟﮕﯽ ﺧﻮﺍﻧﺪﻥ ﺭﺍ ﺁﻣﻮﺧﺘﻪ ﺑﺎﺷﺪ، ﭼﻴﺰﯼ ﺭﺍ ﻛﻪ ﺷﻤﺎ ﺑﺮﺍﻳﺶ ﺍﻧﺘﺨﺎﺏ ﻛﺮﺩﻩ‌ﺍﻳﺪ) ﻭ ﺍﻧﺘﺨﺎﺏ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﺳﺎﻟﻬﺎ‌، ﺍﻫﻤﻴﺖ ﻓﻮﻕ‌ﺍﻟﻌﺎﺩﻩ‌ﺍﯼ ﺩﺭ ﺑﻪ‌ﻭﺟﻮﺩ ﺁﻣﺪﻥ ﻋﺸﻖ ﺑﻪ ﻣﻄﺎﻟﻌﻪ ﺩﺭ‌ ﺁﻳﻨﺪﻩ ﺍﻭ ﺩﺍﺭﺩ.
ﭼﮕﻮﻧﻪ ﺍﻧﺘﺨﺎﺏ ﻛﻨﻴﻢ؟ ﺗﻌﺎﺩﻟﯽ ﺩﺭ ﺁﺛﺎﺭ ﺗﺨﻴﻠﯽ ﻭ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﺭﻋﺎﻳﺖ ﻛﻨﻴﺪ. ﻛﻮﺩﮎ ﻫﻤﺎﻥ‌ﻃﻮﺭ ﻛﻪ ﺩﺭ ﺗﻐﺬﻳﻪ ﻣﺤﺘﺎﺝ ﺑﻪ ﺭﮊﻳﻢ‌ﻏﺬﺍﻳﯽ ﻣﺘﻌﺎﺩﻟﯽ ﺍﺳﺖ ﺩﺭ ﻣﻄﺎﻟﻌﻪ ﻧﻴﺰ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﺗﻌﺎﺩﻝ ﺩﺍﺭﺩ. ﺍﮔﺮ ﺍﻭ ﻓﻘﻂ ﺑﻪ ﻣﻄﺎﻟﻌﻪ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﺑﭙﺮﺩﺍﺯﺩ، ﻗﻮﻩ ﺗﺨﻴﻞ ﻭ ﺧﻼﻕ ﺍﻭ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﻛﺎﻓﯽ ﺭﺷﺪ ﻧﺨﻮﺍﻫﺪ ﻛﺮﺩ. ﻭﻟﯽ ﺍﮔﺮ ﺑﻪ ﻋﻜﺲ، ﻓﻘﻂ ﺑﻪ ﻣﻄﺎﻟﻌﻪ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﯼ ﺗﺨﻴﻠﯽ ﻭ ﺗﻔﻨﻨﯽ ﺑﭙﺮﺩﺍﺯﺩ، ﻣﻄﺎﻟﻌﻪ ﺑﺮﺍﻳﺶ ﺑﻪ‌ﻋﻨﻮﺍﻥ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺩﺭﮎ ﻭﺍﻗﻌﻴﺎﺗﯽ ﻛﻪ ﺩﺭ ﺁﻥ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﻛﻨﺪ ﺟﻠﻮﻩ‌ﮔﺮ ﻧﺨﻮﺍﻫﺪ ﺷﺪ.
ﻛﻮﺩﮎ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﺪ ﺗﻌﺎﺩﻟﯽ ﺑﻴﻦ ﻛﺘﺎﺏ‌ﻫﺎﻳﯽ ﻛﻪ ﺍﺯ ﻛﺘﺎﺑﺨﺎﻧﻪ ﻗﺮﺽ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﻳﺎ ﺍﺯ ﻛﺘﺎﺏ‌ﻓﺮﻭﺷﯽ ﻣﯽ‌ﺧﺮﺩ ﻭ ﺩﺭ ﻛﺘﺎﺑﺨﺎﻧﻪ ﺷﺨﺼﯽ‌ﺍﺵ ﻛﻪ ﺗﻌﺪﺍﺩﺵ ﺑﺎﻳﺪ ﺭﻭ ﺑﻪ ﺍﻓﺰﺍﻳﺶ ﺑﺎﺷﺪ، ﺍﻳﺠﺎﺩ ﻛﻨﺪ.
