ﺩﺭﺱ ﺳﻮﻡ- ﺗﻘﻮﻳﺖ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﮐﻮﺩﮎ

ﺩﺭ ﺩﺭﺱ ﭘﻴﺶ ﻣﻮﺍﺭﺩﯼ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ ﮐﻪ ﻣﻮﺟﺐ ﺗﻘﻮﻳﺖ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﮐﻮﺩﮎ ﻣﯽ‌­ﺷﻮﺩ. ﺍﺯ ﺟﻤﻠﻪ ﺑﻪ ﺍﻧﺘﺨﺎﺏ ﺳﻨﺠﻴﺪﻩ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﮐﻮﺩﮎ ﺩﺭﻣﻴﺎﻥ ﺁﻧﻬﺎ ﺯﻧﺪﮔﯽ ﻣﯽ­‌ﮐﻨﺪ، ﺭﺍﺑﻄﻪ ﺷﺨﺼﯽ‌ﺍﯼ ﮐﻪ ﺑﺎ ﺗﮏ‌ﺗﮏ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌­ﮐﻨﻴﻢ، ﺍﺟﺎﺯﻩ ﺩﺍﺩﻥ ﺑﻪ ﮐﻮﺩﮎ ﺑﺮﺍﯼ ﺍﺩﺍﺭﻩ ﺧﻮﺩ ﺑﻪ ﻣﺤﺾ ﺁﻧﮑﻪ ﻗﺎﺩﺭ ﺑﻪ ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﺷﺪ، ﺩﺍﺷﺘﻦ ﺑﺮﺧﻮﺭﺩﯼ ﻣﺘﻔﺎﻭﺕ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺍﻋﻤﺎﻝ ﮐﻮﺩﮎ ﻭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻗﺪﺭﺕ ﺑﺴﻴﺎﺭ ﻧﻴﺮﻭﻣﻨﺪ ﺗﻘﻠﻴﺪ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ. ﺍﮐﻨﻮﻥ ﻭﺟﻮﻩ ﺩﻳﮕﺮ ﺁﻣﻮﺯﺵ ﺩﻳﺴﻴﭙﻠﻴﻦ ﺭﺍ ﺑﺮﺭﺳﯽ ﻣﯽ‌­ﮐﻨﻴﻢ.
 
ﺍﮔﺮ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﺯ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﺍﻋﻤﺎﻟﺶ ﻋﺒﺮﺕ ﺑﮕﻴﺮﺩ، ﺑﻪ ﺍﻭ ﺩﺭ ﻛﺴﺐ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ‌ﺍﯼ ﻗﻮﯼ ﻛﻤﮏ ﻛﺮﺩﻩ‌ﺍﻳﺪ.
 ﺍﻳﻦ ﻳﻜﯽ ﺍﺯ ﺑﺎﺍﺭﺯﺵ‌ﺗﺮﻳﻦ ﻭﺳﺎﻳﻠﯽ ﺍﺳﺖ ﻛﻪ ﻣﺎ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻳﻢ. ﻣﺘﺎﺳﻔﺎﻧﻪ ﺗﻌﺪﺍﺩ ﺑﺴﻴﺎﺭ ﻛﻤﯽ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﻛﻨﻨﺪ. ﭼﮕﻮﻧﮕﯽ ﺭﻭﻧﺪ ﺍﻳﻦ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﺭﺍ ﺑﺮﺭﺳﯽ ﻛﻨﻴﻢ.
ﻛﻮﺩﻛﯽ ﺻﺒﺤﺎﻧﻪ‌ﺍﺵ ﺭﺍ ﻧﻤﯽ‌ﺧﻮﺭﺩ. ﻓﺲ‌ﻓﺲ ﻣﯽ‌ﻛﻨﺪ، ﺑﺎﺯﯼ ﻣﯽ‌ﻛﻨﺪ ﺧﻼﺻﻪ ﻫﻤﻪ ﻛﺎﺭ ﻣﯽ‌ﻛﻨﺪ ﺟﺰ ﺧﻮﺭﺩﻥ. ﻣﺎﺩﺭ ﻋﺼﺒﺎﻧﯽ ﻧﻤﯽ‌ﺷﻮﺩ، ﺗﻬﺪﻳﺪﯼ ﺑﻪ ﺗﻨﺒﻴﻪ ﻛﻮﺩﮎ ﻧﻤﯽ‌ﻛﻨﺪ. ﻓﻘﻂ ﺑﻌﺪ ﺍﺯ ﭘﺎﻳﺎﻥ ﺻﺒﺤﺎﻧﻪ، ﻣﻴﺰ ﺭﺍ ﺟﻤﻊ ﻣﯽ‌ﻛﻨﺪ ﻭ ﻣﯽ‌‌ﮔﺬﺍﺭﺩ ﻛﻪ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﻛﺎﺭ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺑﻜﻨﻨﺪ. ﺍﺣﺘﻤﺎﻻ ﻛﻤﯽ ﺑﻌﺪ ﻛﻮﺩﮎ «ﭼﻴﺰﯼ» ﺧﻮﺍﻫﺪ ﺧﻮﺍﺳﺖ. ﻣﺎﺩﺭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺟﻮﺍﺏ ﺩﻫﺪ: «ﺧﻴﻠﯽ ﻣﺘﺎﺳﻔﻢ ﻛﻪ ﮔﺮﺳﻨﻪ‌ﺍﯼ، ﻇﻬﺮ ﻧﺎﻫﺎﺭ ﻣﯽ‌ﺧﻮﺭﻳﻢ. ﺧﻴﻠﯽ ﻧﺎﺭﺍﺣﺖ‌ﻛﻨﻨﺪﻩ ﺍﺳﺖ ﻛﻪ ﺑﺎﻳﺪ ﺍﻳﻦ ﻫﻤﻪ ﺻﺒﺮ ﻛﻨﯽ.» ﮔﺮﺳﻨﮕﯽ‌ﺍﯼ ﻛﻪ ﻛﻮﺩﮎ ﺭﺍ ﺁﺯﺍﺭ ﻣﯽ‌ﺩﻫﺪ ﻧﺘﻴﺠﻪ ﻃﺒﻴﻌﯽ ﻧﺨﻮﺭﺩﻥ ﺻﺒﺤﺎﻧﻪ‌ﺍﺵ ﺍﺳﺖ. ﺍﻳﻦ ﮔﺮﺳﻨﮕﯽ ﺑﻬﺘﺮ ﻭ ﺳﺮﻳﻊ‌ﺗﺮ ﺍﺯ ﻫﺮ ﻏﺮ ﻭ ﺗﻨﺒﻴﻬﯽ ﺭﻭﯼ ﺍﻭ ﺍﺛﺮ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﺗﺎ ﺩﺭ ﺭﻓﺘﺎﺭﺵ ﺗﺠﺪﻳﺪﻧﻈﺮ ﻛﻨﺪ.
ﺍﮔﺮ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﺍﻋﻤﺎﻝ ﻛﻮﺩﮎ ﻣﻄﺒﻮﻉ ﺑﺎﺷﺪ، ﺍﻭ ﻣﻌﻤﻮﻻ ﺩﺭ ﺭﻭﺵ ﻋﻤﻠﺶ ﺗﻐﻴﻴﺮﯼ ﻧﻤﯽ‌ﺩﻫﺪ. ﺍﮔﺮ ﺍﻳﻦ ﻧﺘﺎﻳﺞ ﻧﺎﻣﻄﺒﻮﻉ ﺑﺎﺷﻨﺪ، ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﻛﻮﺩﮎ ﻟﺰﻭﻡ ﺗﻐﻴﻴﺮ ﺭﻓﺘﺎﺭﺵ ﺭﺍ ﺍﺣﺴﺎﺱ ﺧﻮﺍﻫﺪ ﻛﺮﺩ، ﻣﮕﺮ ﺁﻧﻜﻪ ﻣﺎ ﻭﺍﻟﺪﻳﻦ، ﺳﻌﯽ ﻛﻨﻴﻢ ﺍﺯ ﺍﻭ ﺩﺭ ﻣﻘﺎﺑﻞ ﺁﻥ ﻧﺘﺎﻳﺠﯽ ﻛﻪ ﻃﺒﻴﻌﺘﺎ ﺑﺎﻳﺪ ﭘﻴﺶ ﺑﻴﺎﻳﻨﺪ، ﺣﻔﺎﻇﺖ ﮐﻨﻴﻢ. ﻣﺘﺎﺳﻔﺎﻧﻪ، ﺍﻏﻠﺐ ﺍﻭﻗﺎﺕ ﻭﺿﻊ ﺑﺪﻳﻦ ﻣﻨﻮﺍﻝ ﺍﺳﺖ.
ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺮﺍﯼ ﻣﺤﺎﻓﻈﺖ ﻓﺮﺯﻧﺪﺷﺎﻥ ﺍﺯ ﻧﺘﺎﻳﺞ ﻧﺎﻣﻄﺒﻮﻉ ﻭ ﻃﺒﻴﻌﯽ ﺍﻋﻤﺎﻟﺶ ﺩﺧﺎﻟﺖ ﻣﯽ‌ﻛﻨﻨﺪ، ﺍﻭ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﺍﺭﺯﺵ ﺗﺮﺑﻴﺘﯽ ﺗﺠﺮﺑﻪ‌ﺍﺵ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﺪ. ﺍﻳﻦ ﺑﺮﺍﯼ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﺍﻭ ﻣﻀﺮ ﺍﺳﺖ ﻭ ﻣﺎﻧﻊ ﻣﯽ‌ﺷﻮﺩ ﺑﺘﻮﺍﻧﺪ ﺭﻭﯼ ﺧﻮﺩﺵ ﺣﺴﺎﺏ ﻛﻨﺪ. ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﻋﺎﻗﻼﻧﻪ ﺭﻓﺘﺎﺭ ﻛﺮﺩ ﻭ ﺍﺯ ﺍﻳﻦ ﻣﻔﻬﻮﻡ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﺑﻪ ﺻﻮﺭﺗﯽ ﻋﺎﻗﻼﻧﻪ ﺍﺳﺘﻔﺎﺩﻩ ﻛﺮﺩ. ﺍﮔﺮ ﺷﻤﺎ ﺑﻪ ﻓﺮﺯﻧﺪ ﺑﺴﻴﺎﺭ ﺧﺮﺩﺳﺎﻟﺘﺎﻥ ﻛﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺍﺯ ﺧﻴﺎﺑﺎﻧﯽ ﻛﻪ ﻣﺎﺷﻴﻦ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺍﺯ ﺁﻥ ﻋﺒﻮﺭ ﻣﯽ‌ﻛﻨﻨﺪ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﺭﺩ ﺷﻮﺩ ﺗﺎ ﺍﺯ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﺍﻋﻤﺎﻟﺶ ﺁﮔﺎﻩ ﺷﻮﺩ، ﺍﻳﻦ ﻛﺎﺭ ﻣﻤﻜﻦ ﺍﺳﺖ ﺑﻪ ﺟﺎﻫﺎﯼ ﺧﻴﻠﯽ ﺑﺎﺭﻳﮏ ﺑﻜﺸﺪ. ﺑﻨﺎﺑﺮ ﺍﻳﻦ ﻭﻗﺘﯽ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺍﺯ ﺧﻴﺎﺑﺎﻥ ﺭﺩ ﺷﻮﺩ، ﺑﺎﻳﺪ ﺩﺧﺎﻟﺖ ﻛﺮﺩ ﻭ ﺍﻭ ﺭﺍ ﺍﺯ ﺍﻳﻦ ﻛﺎﺭ ﻣﻨﻊ ﻛﺮﺩ. ﻟﻴﻜﻦ ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﺍﻋﻤﺎﻝ ﻛﻮﺩﮎ ﻓﻘﻂ ﻳﮏ ﻧﺎﺧﺸﻨﻮﺩﯼ ﻳﺎ ﻧﺎﺭﺍﺣﺘﯽ ﺍﺳﺖ، ﺩﺧﺎﻟﺘﯽ ﻧﻜﻨﻴﺪ ﻭ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﻛﻪ ﻭﻗﺎﻳﻊ ﻣﺴﻴﺮ ﻣﻨﻄﻘﯽ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﻃﯽ ﻛﻨﻨﺪ.
ﺍﮔﺮ ﻣﯽ‌­ﺗﻮﺍﻧﺴﺘﻴﻢ ﻛﺎﻣﻼ ﺭﻭﯼ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﺍﻋﻤﺎﻝ ﻛﻮﺩﮎ ﺟﻬﺖ ﺗﺮﺑﻴﺖ ﺍﻭ ﺗﻜﻴﻪ ﻛﻨﻴﻢ ﺧﻴﻠﯽ ﻋﺎﻟﯽ ﺑﻮﺩ. ﻭﻟﯽ ﻣﺘﺎﺳﻔﺎﻧﻪ ﺍﻳﻦ ﻛﺎﺭ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻛﺎﻓﯽ ﻧﻴﺴﺖ.  ﮔﺎﻫﯽ ﻣﺠﺒﻮﺭﻳﻢ ﻧﺘﺎﻳﺠﯽ ﻣﺼﻨﻮﻋﯽ ﺑﺮﺍﯼ ﺍﻋﻤﺎﻝ ﺍﻭ ﺍﻳﺠﺎﺩ ﻛﻨﻴﻢ ﺗﺎ ﻋﻤﻠﯽ ﺭﺍ ﻣﺠﺎﺯﺍﺕ ﻛﻨﻴﻢ. ﺑﺎﻳﺪ ﻧﺘﺎﻳﺞ ﻣﺼﻨﻮﻋﯽ ﻭ ﺍﺭﺍﺩﯼ‌ﺍﯼ ﺍﺯ ﺟﺎﻧﺐ ﺷﻤﺎ ﺑﺮﺍﯼ ﻛﻮﺩﮎ ﻭﺿﻊ ﺷﻮﺩ ﻛﻪ ﻣﺤﺪﻭﺩﻩ ﻛﺎﻣﻼ ﻣﺸﺨﺼﯽ ﺭﺍ ﺑﺮﺍﯼ ﺍﻭ ﺗﻌﻴﻴﻦ ﻛﻨﺪ،‌ ﺑﻪ ﺻﻮﺭﺗﯽ ﻛﻪ ﺍﻳﻦ ﻧﺘﺎﻳﺞ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﺑﺎﺷﺪ ﻛﻪ «ﺩﻳﮕﺮ ﻫﺮﮔﺰ ﺍﻳﻦ ﻛﺎﺭ ﺗﻜﺮﺍﺭ ﻧﺸﻮﺩ!» ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ، ﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﺯ ﺳﻪ ﺭﻭﺵ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻴﻢ:
 
۱- ﻣﯽ‌ﺗﻮﺍﻥ ﻛﻮﺩﮎ ﺭﺍ ﺍﺯ ﭼﻴﺰﯼ ﻛﻪ ﺑﻪ ﺁﻥ ﻋﻼﻗﻪ‌ﻣﻨﺪ ﺍﺳﺖ ﻣﺤﺮﻭﻡ ﮐﺮﺩ.
ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻭ ﺭﺍ ﺍﺯ ﭼﻴﺰ ﺑﻪ‌ ﺧﺼﻮﺻﯽ ﻣﺤﺮﻭﻡ ﻛﻨﻴﺪ، ﻣﺜﻼ ﺑﮕﻮﻳﻴﺪ: «ﺑﺒﻴﻦ ﺗﻮ ﺍﻭﻥ ﻗﺪﺭ ﺑﺰﺭﮒ ﺷﺪﯼ ﻛﻪ ﺑﺪﻭﻧﯽ ﺭﻭﯼ ﺩﻳﻮﺍﺭ ﻧﻤﯽ‌ﻧﻮﻳﺴﻦ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺗﺎ ﺳﻪ ﺭﻭﺯ ﺍﺟﺎﺯﻩ ﻧﺪﺍﺭﯼ ﺍﺯ ﻣﺪﺍﺩ ﺭﻧﮕﯽ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﯽ، ﺍﻭﻧﻮﻗﺖ ﻣﯽ‌ﻓﻬﻤﯽ ﻛﻪ ﺁﺩﻡ ﺍﺯ ﻣﺪﺍﺩ ﺑﺮﺍﯼ ﻧﻮﺷﺘﻦ ﺭﻭﯼ ﻛﺎﻏﺬ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﻛﻨﻪ ﻧﻪ ﺑﺮﺍﯼ ﺧﻂ ﻛﺸﻴﺪﻥ ﺭﻭﯼ ﺩﻳﻮﺍﺭ.»
 
۲- ﻣﯽ‌ﺗﻮﺍﻥ ﻛﻮﺩﮎ ﺭﺍ ﺍﺯ ﺟﻤﻊ ﻛﻨﺎﺭ ﮔﺬﺍﺷﺖ ﻭ ﻳﺎ ﺍﻭ ﺭﺍ ﺑﻪ ﺍﺗﺎﻗﺶ ﻓﺮﺳﺘﺎﺩ.
ﻓﺮﺽ ﻛﻨﻴﺪ ﻛﻪ ﻓﺮﺯﻧﺪ ﭼﻬﺎﺭ ﺳﺎﻟﻪ‌ﺗﺎﻥ ﻣﺎﻧﻊ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﮔﺮﻭﻫﯽ ﻛﻮﺩﮎ ﺩﺭ ﺣﻴﺎﻁ ﻳﺎ ﺩﺍﺧﻞ ﺧﺎﻧﻪ‌ ﻣﯽ‌ﺷﻮﺩ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ: «ﻣﯽ‌ﺑﻴﻨﻢ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﻧﯽ ﺑﺎ ﺑﭽﻪ‌ﻫﺎ ﺑﺎﺯﯼ ﮐﻨﯽ، ﻫﻤﻪ‌ﺍﺵ ﺩﺍﺭﯼ ﺍﻭﻧﺎﺭﻭ ﻣﯽ‌ﺯﻧﯽ ﻭ ﺷﻠﻮﻍ ﻣﯽ‌ﮐﻨﯽ. ﺑﺮﻭ ﺗﻮ ﺍﺗﺎﻗﺖ ﻭ ﻫﺮ ﻭﻗﺖ ﻛﻪ ﺗﻮﻧﺴﺘﯽ ﻣﻬﺮﺑﻮﻥ‌ﺗﺮ ﺑﺎﺷﯽ ﻭ ﺧﻮﺩﺕ ﺭﻭ ﻛﻨﺘﺮﻝ ﻛﻨﯽ ﺑﻴﺎ  ﺑﻪ ﻣﻦ ﺑﮕﻮ.»
ﻫﺮ ﺑﺎﺭ ﻛﻪ ﺍﺯ ﺟﺪﺍ ﻛﺮﺩﻥ ﻃﻔﻞ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻭﺳﻴﻠﻪ ﺗﻨﺒﻴﻬﯽ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﻛﻨﻴﺪ، ﻣﻬﻢ ﺍﺳﺖ ﻛﻪ ﺑﻪ ﺍﺻﻄﻼﺡ ﺩﺭﻫﺎ ﺭﺍ ﺑﺎﺯ ﺑﮕﺬﺍﺭﻳﺪ. ﺍﻭ ﺭﺍ ﺁﻥ ﻃﻮﺭ ﺑﻪ ﺍﺗﺎﻗﺶ ﻧﻔﺮﺳﺘﻴﺪ ﻛﻪ ﺍﻧﮕﺎﺭ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﻫﻤﻴﺸﻪ ﺩﺭ ﺁﻥ ﺟﺎ ﺑﻤﺎﻧﺪ. ﻫﺪﻑ ﺷﻤﺎ ﺣﺒﺲ ﻛﺮﺩﻥ ﺍﻭ ﻧﻴﺴﺖ، ﺑﻠﻜﻪ ﻗﺼﺪ ﺩﺍﺭﻳﺪ ﻛﻪ ﺩﺭ ﺭﻓﺘﺎﺭﺵ ﺗﻐﻴﻴﺮﯼ ﺑﺪﻫﺪ. ﺍﻭ ﺑﺎﻳﺪ ﺑﺪﺍﻧﺪ ﻛﻪ ﻫﺮ ﮔﺎﻩ ﺑﺘﻮﺍﻧﺪ ﺑﺎ ﻣﻬﺮﺑﺎﻧﯽ ﺑﺎ ﺭﻓﻘﺎﻳﺶ ﺑﺎﺯﯼ ﻛﻨﺪ ﻭ ﺭﻓﺘﺎﺭﺵ ﺭﺍ ﺗﻐﻴﻴﺮ ﺑﺪﻫﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﮔﺮﺩﺩ.
 
۳- ﻣﯽ‌ﺗﻮﺍﻥ ﻳﮏ ﺩﺭﻟﻨﺒﺮﯼ ﺑﻪ ﻛﻮﺩﮎ ﺯﺩ.
ﺍﮔﺮ ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﺧﺮﺍﺑﻜﺎﺭﯼ ﺍﻭ ﺭﺍ ﻛﺸﻒ ﻣﯽ‌ﮐﻨﻴﺪ ﺧﻴﻠﯽ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻴﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻼﻓﺎﺻﻠﻪ ﻳﮏ ﺩﺭﻟﻨﺒﺮﯼ ﺑﻪ ﺍﻭ ﺑﺰﻧﻴﺪ. ﺍﻳﻦ ﻳﮏ ﻧﻮﻉ ﻧﺘﻴﺠﻪ ﻧﺎﻣﻄﻠﻮﺏ ﻣﺼﻨﻮﻋﯽ ﺑﺮﺍﯼ ﻛﻮﺩﮎ ﺍﺳﺖ. ﺍﻣﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﭼﻴﺰﯼ ﺭﺍ ﻣﺸﺨﺺ ﻛﻨﻢ ﻭ ﺁﻥ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺩﺭﻟﻨﺒﺮﯼ «ﺧﻮﺏ» ﺩﺍﺭﻳﻢ ﻭ ﺩﺭﻟﻨﺒﺮﯼ «ﺑﺪ». ﻣﻨﻈﻮﺭﻡ ﺍﺯ ﺩﺭﻟﻨﺒﺮﯼ «ﺑﺪ»، ﻛﺘﮏ ﺯﺩﻥ ﻛﻮﺩﮎ ﺑﻪ ﺻﻮﺭﺗﯽ ﺳﺎﺩﻳﺴﺘﯽ ﻭ ﺳﺘﻤﮕﺮﺍﻧﻪ ﺍﺳﺖ ﻛﻪ ﺗﻨﻬﺎ ﺍﺣﺴﺎﺳﯽ ﻛﻪ ﺩﺭ ﻛﻮﺩﮎ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﮐﻨﺪ ﺍﺣﺴﺎﺱ ﺗﻨﻔﺮ ﻭ ﺍﻧﺘﻘﺎﻡ‌ﺟﻮﻳﯽ ﺍﺳﺖ. ﺍﻳﻦ ﺗﻨﺒﻴﻬﯽ ﺍﺳﺖ ﻛﻪ ﺑﺎ ﺷﻼﻕ ﻳﺎ ﺗﺮﻛﻪ ﻭ ﻳﺎ ﻫﺮ ﻧﻮﻉ «ﺍﺳﻠﺤﻪ» ﺩﻳﮕﺮﯼ ﻛﻪ ﺑﺰﺭﮔﺴﺎﻻﻥ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻧﺪ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ. ﺳﻴﻠﯽ‌ﺍﯼ ﻛﻪ ﻛﻮﺩﮎ ﺭﺍ ﺗﺤﻘﻴﺮ ﻣﯽ‌ﮐﻨﺪ ﻧﻴﺰ ﺩﺭ ﺭﺩﻩ ﻫﻤﻴﻦ ﻧﻮﻉ ﺗﻨﺒﻴﻬﺎﺕ ﺍﺳﺖ.
ﻃﺮﻳﻖ «ﺧﻮﺏ» ﺗﻨﺒﻴﻪ ﻛﺮﺩﻥ ﻳﮏ ﻛﻮﺩﮎ ﺍﺣﺘﻴﺎﺝ ﺑﻪ ﻭﺳﻴﻠﻪ ﻧﺪﺍﺭﺩ. ﺩﺳﺖ ﻣﺎﺩﺭﯼ ﻳﺎ ﭘﺪﺭﯼ، ﺩﻭ ﻳﺎ ﺳﻪ ﺑﺎﺭ ﭘﺸﺖ ﻛﻮﺩﮎ ﺭﺍ ﻣﯽ‌ﻧﻮﺍﺯﺩ ﻭ ﺩﻳﮕﺮ ﺗﻤﺎﻡ ﻣﯽ‌ﺷﻮﺩ. ﺍﮔﺮ ﺩﺭﻟﻨﺒﺮﯼ ﺭﺍ ﺑﻪ ﻣﻮﻗﻊ ﺍﺳﺘﻔﺎﺩﻩ ﻛﻨﻴﺪ، ﭼﻴﺰ ﺧﻮﺑﯽ ﺍﺳﺖ. ﺩﺭﻟﻨﺒﺮﯼ ﺟﻮ ﺭﺍ ﺳﺎﻟﻢ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺑﻪ ﻣﻮﻋﻈﻪ‌ﻫﺎﯼ ﺍﺧﻼﻗﯽ ﻛﻪ ﻛﻮﺩﮎ ﺭﺍ ﺩﭼﺎﺭ ﺍﺣﺴﺎﺱ ﮔﻨﺎﻩ ﻣﯽ‌ﮐﻨﺪ ﺗﺮﺟﻴﺢ ﺩﺍﺭﺩ.
ﻓﻘﻂ ﺯﻣﺎﻧﯽ ﻛﻮﺩﮎ ﺭﺍ ﻛﺘﮏ ﺑﺰﻧﻴﺪ ﻛﻪ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻴﺪ. ﻛﻮﺩﮎ ﺧﻴﻠﯽ ﺧﻮﺏ ﺩﺭﮎ ﻣﯽ‌ﮐﻨﺪ ﻛﻪ ﻭﻗﺘﯽ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻴﺪ ﭼﺮﺍ ﺍﻭ ﺭﺍ ﻣﯽ‌ﺯﻧﻴﺪ. ﺍﻭ ﻣﯽ‌ﺩﺍﻧﺪ ﺍﺯ ﺩﺳﺘﺶ ﺧﻴﻠﯽ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻴﺪ ﻭ ﻋﻠﺘﺶ ﺭﺍ ﻫﻢ ﻣﯽ‌ﺩﺍﻧﺪ. ﭼﻴﺰﯼ ﻛﻪ ﻃﻔﻞ ﻧﻤﯽ‌ﻓﻬﻤﺪ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﻣﺎﺩﺭ ﺑﻪ ﻓﺮﺯﻧﺪﺵ ﺩﺭ ﺳﺎﻋﺖ ﺩﻩ ﺻﺒﺢ ﺑﮕﻮﻳﺪ : «ﺧﻴﻠﯽ ﺧﻮﺏ ، ﺍﻣﺸﺐ ﻭﻗﺘﯽ ﺑﺎﺑﺎﺕ ﺍﻭﻣﺪ ﺧﻮﻧﻪ ﺣﺴﺎﺑﺘﻮ ﻣﻲ­ﺫﺍﺭﻩ ﻛﻒ ﺩﺳﺘﺖ!» ﻭ ﻭﻗﺘﯽ ﭘﺪﺭ ﺑﻪ ﺧﺎﻧﻪ ﺑﺮﮔﺸﺖ ﺍﺯ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻳﮏ ﺩﺭﻟﻨﺒﺮﯼ ﺑﻪ ﺍﻭ ﺑﺰﻧﺪ ﻭ ﻓﻜﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ «ﺩﺭﺱ ﺧﻮﺑﯽ» ﺑﻪ ﻛﻮﺩﮎ ﻣﯽ‌ﺩﻫﻨﺪ. ﺩﺭ ﺻﻮﺭﺗﯽ ﻛﻪ ﺍﻳﻦ ﻧﻮﻉ ﺗﻨﺒﻴﻪ ﺭﺍ ﻛﻪ ﺩﺭ ﻛﻤﺎﻝ ﺧﻮﻧﺴﺮﺩﯼ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ ﻛﻮﺩﮎ ﻧﻪ ﻣﯽ ﺗﻮﺍﻧﺪ ﺩﺭﮎ ﻛﻨﺪ ﻭ ﻧﻪ ﺑﺒﺨﺸﺪ.
ﺁﻥ ﻧﻮﻉ ﺩﺭﻟﻨﺒﺮﯼ ﻛﻪ ﻣﻦ ﺳﻔﺎﺭﺵ‌ﺍﺵ ﺭﺍ ﻣﯽ‌ﮐﻨﻢ ﺍﺯ ﻧﻮﻉ «ﺩﺭﻕ، ﺁﯼ!» ﺍﺳﺖ. «ﺩﺭﻕ»، ﺷﻤﺎ ﺑﻼﻓﺎﺻﻠﻪ «ﺁﯼ» ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺑﻪ ﺩﻧﺒﺎﻝ ﺩﺍﺭﺩ. ﻓﻘﻂ ﻭﻗﺘﯽ ﻛﻪ ﺧﻴﻠﯽ ﺍﺯ ﺩﺳﺘﺶ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻴﺪ ﻭ ﻣﻲ‌ﺧﻮﺍﻫﻴﺪ ﻫﻤﺎﻥ ﻟﺤﻈﻪ ﻣﺴﺌﻠﻪ ﺭﺍ ﺗﻤﺎﻡ ﻛﻨﻴﺪ ﺑﻪ ﺍﻭ ﺩﺭﻟﻨﺒﺮﯼ ﺑﺰﻧﻴﺪ. ﺑﺎﻳﺪ ﻳﮏ ﺑﺎﺭ ﻳﺎ ﺩﻭ ﺑﺎﺭ ﺑﻪ ﻛﻮﺩﮎ ﺩﺭ ﻣﻮﺭﺩ ﻛﺎﺭﯼ ﻛﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻜﻨﺪ ﻭ ﻳﺎ ﻧﻜﻨﺪ ﺗﺬﻛﺮ ﺑﺪﻫﻴﺪ. ﺁﻥ ﻭﻗﺖ ﺍﮔﺮ ﺍﻭ ﺍﻃﺎﻋﺖ ﻧﻜﺮﺩ ﻭ ﺍﻳﻦ ﻛﺎﺭﺵ ﺷﻤﺎ ﺭﺍ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﮐﻨﺪ ﺑﻼﻓﺎﺻﻠﻪ ﺗﻨﺒﻴﻬﺶ ﻛﻨﻴﺪ. ﺑﻌﺪ ﺍﺯ ﺩﺭﻟﻨﺒﺮﯼ ﻣﻤﻜﻦ ﺍﺳﺖ ﺍﺣﺴﺎﺱ ﻧﺎﺭﺍﺣﺘﯽ ﻭ ﮔﻨﺎﻩ ﻛﻨﻴﺪ ﻛﻪ ﭼﺮﺍ ﺧﻮﻧﺴﺮﺩﻳﺘﺎﻥ ﺭﺍ ﺣﻔﻆ ﻧﻜﺮﺩﻳﺪ. ﺧﺎﻧﻢ ﺟﺎﻥ، ﺁﻗﺎ ﺟﺎﻥ، ﺷﻬﺎﻣﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﺭﺍﻩ ﺑﺎﺯﮔﺸﺖ ﻭﺟﻮﺩ ﺩﺍﺭﺩ!
ﻫﻤﻴﺸﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﻛﻠﻤﺎﺕ ﺩﻝ ﺍﻭ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﻳﺪ،‌ ﻣﺜﻼ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﮕﻮﻳﻴﺪ: «ﺑﺒﻴﻦ ﻣﺎﻣﺎﻥ ﻳﮏ ﻛﺎﺭ ﺍﺣﻤﻘﺎﻧﻪ ﻛﺮﺩ. ﻋﺼﺒﺎﻧﯽ ﺷﺪﻡ ﻭ ﺍﺯ ﺍﻳﻦ ﻛﺎﺭﻡ ﺧﻴﻠﯽ ﻧﺎﺭﺍﺣﺘﻢ. ﺍﺯﺕ ﻣﻌﺬﺭﺕ ﻣﯽ‌ﺧﻮﺍﻡ.» ﻭ ﺑﺪﻭﻥ ﻧﺎﺭﺍﺣﺘﯽ ﻭ ﺍﺣﺴﺎﺱ ﮔﻨﺎﻩ ﺑﻪ ﻛﺎﺭﺗﺎﻥ ﺍﺩﺍﻣﻪ ﺩﻫﻴﺪ.
 
ﺻﺒﺮ ﻛﻨﻴﺪ ﻛﻪ ﺣﺎﺩﺛﻪ‌ﺍﯼ ﻛﻪ ﺍﺗﻔﺎﻕ ﺍﻓﺘﺎﺩﻩ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻛﻨﻴﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﻛﺎﺭ ﻣﻤﻜﻦ ﺍﺳﺖ ﭘﻨﺞ ﺩﻗﻴﻘﻪ ﻻﺯﻡ ﺑﺎﺷﺪ ﻳﺎ ﭘﻨﺞ ﺳﺎﻋﺖ. ﻭﻟﯽ ﺍﮔﺮ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﺍﺷﺘﺒﺎﻩ ﻛﺮﺩﻩ‌ﺍﻳﺪ، ﻣﻬﻢ ﺍﺳﺖ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﻳﻦ ﺭﺍ ﺑﺪﺍﻧﺪ. ﺧﺼﻮﺻﺎ ﻃﻮﺭﯼ ﻣﺴﺌﻠﻪ ﺭﺍ ﻣﻄﺮﺡ ﻧﻜﻨﻴﺪ ﻛﻪ ﺍﻭ ﻓﻜﺮ ﻛﻨﺪ ﺍﻭ ﺭﺍ ﻓﻘﻂ ﺑﺮﺍﯼ ﺧﻮﺑﯽ ﺧﻮﺩﺵ ﺗﻨﺒﻴﻪ ﻛﺮﺩﻩ‌ﺍﻳﺪ. ﺯﻳﺮﺍ ﺍﻳﻦ ﺁﻧﻘﺪﺭ ﻏﻠﻂ ﺍﺳﺖ ﻛﻪ ﻛﻮﺩﮎ ﺁﻥ ﺭﺍ ﺩﺭﮎ ﺧﻮﺍﻫﺪ ﻛﺮﺩ.
ﻫﺪﻑ ﺍﺻﻠﯽ ﺩﺭﻟﻨﺒﺮﯼ، ﻫﺮ ﭼﻨﺪ ﻛﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﻋﻘﻴﺪﻩ ﺩﻳﮕﺮﯼ ﺩﺍﺭﻧﺪ، ﺳﺒﮏ ﻛﺮﺩﻥ ﻭﺍﻟﺪﻳﻦ ﺍﺳﺖ ﻭ ﻫﻤﻪ ﻣﺎ ﮔﻬﮕﺎﻩ ﺑﻪ ﺍﻳﻦ ﺳﺒﮏ ﻛﺮﺩﻥ، ﺯﻣﺎﻧﯽ ﻛﻪ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ «ﺍﻋﺼﺎﺑﻤﺎﻥ ﺭﺍ ﺧﻮﺭﺩ ﻣﯽ‌ﮐﻨﻨﺪ» ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﻳﻢ.
 
ﺑﻌﻀﯽ ﺍﺯ ﺧﻮﺍﻧﻨﺪﻩ‌ﻫﺎﯼ ﻣﺎ ﺍﻳﻦ ﻛﻪ ﻣﯽ‌ﺑﻴﻨﻨﺪ ﺩﺭﻟﻨﺒﺮﯼ ﻋﻤﺪﺗﺎ ﺑﺮﺍﯼ ﺳﺒﮏ ﮐﺮﺩﻥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺍﺳﺖ ﻧﺎﺭﺍﺣﺖ ﻣﯽ‌ﺷﻮﻧﺪ. ﺷﺎﻳﺪ ﻫﻨﻮﺯ ﺑﺮﺍﯼ ﺷﻤﺎ ﺍﻳﻦ ﺗﻮﻫﻢ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭﻟﻨﺒﺮﯼ ﺑﺮﺍﯼ ﺧﻮﺑﯽ ﻛﻮﺩﮎ ﺍﺳﺖ. ﺍﮔﺮ ﺷﻤﺎ ﺍﺯ ﺍﻳﻦ ﺩﺳﺘﻪ ﻫﺴﺘﻴﺪ ﺷﻤﺎ ﺭﺍ ﺑﻪ ﺁﻥ ﮐﺎﺭﻳﮑﺎﺗﻮﺭﯼ ﺭﺟﻌﺖ ﻣﯽ‌ﺩﻫﻢ ﻛﻪ ﭘﺪﺭﯼ ﺩﺭ ﺣﺎﻝ ﺯﺩﻥ ﻓﺮﺯﻧﺪﺵ ﺍﺳﺖ ﻭ ﻣﯽ‌ﮔﻮﻳﺪ: «ﺍﻳﻦ ﻃﻮﺭﯼ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﯼ ﻛﻪ ﭼﻪ‌ﻃﻮﺭﯼ ﻣﺮﺩﻡ ﺭﺍ ﻛﺘﮏ ﺑﺰﻧﯽ» (ﻣﺴﻠﻤﺎ).
 
ﺑﺎ ﺍﻳﻦ ﻫﻤﻪ، ﻣﺎ ﻭﺍﻟﺪﻳﻦ ﻧﻴﺰ ﺍﻧﺴﺎﻥ ﻫﺴﺘﻴﻢ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﻋﻠﺖ ﻣﯽ‌ﮔﻮﻳﻢ ﺩﺭ ﺻﻮﺭﺕ ﻟﺰﻭﻡ ﺑﺎ ﺯﺩﻥ ﻳﮏ ﺩﺭﻟﻨﺒﺮﯼ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺳﺒﮏ ﻛﻨﻴﺪ. ﻭﻟﯽ ﺍﻣﻴﺪﻭﺍﺭﻡ ﻛﻪ ﺑﺎ ﺑﻪ‌ﻛﺎﺭﮔﻴﺮﯼ ﺗﻮﺻﻴﻪ‌ﻫﺎﻳﯽ ﻛﻪ ﮐﺮﺩﻩ‌ﺍﻳﻢ، ﺧﻴﻠﯽ ﻛﻤﺘﺮ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺗﺎ ﺍﻳﻦ ﺳﻄﺢ ﭘﺎﻳﻴﻦ ﺑﻴﺎﻭﺭﻳﺪ ﻭ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺑﻜﻨﻴﺪ.
 
ﺍﮔﺮ ﺑﺎ ﺧﻮﺩﺗﺎﻥ ﺭﻭ ﺭﺍﺳﺖ ﺑﺎﺷﻴﺪ، ﺍﺫﻋﺎﻥ ﺧﻮﺍﻫﻴﺪ ﻛﺮﺩ ﻛﻪ ﮔﺎﻫﯽ ﺧﻴﻠﯽ ﺯﻭﺩ ﺍﺯ ﻛﻮﺭﻩ ﺩﺭ ﻣﯽ‌ﺭﻭﻳﺪ ﻭ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﻳﺪ ﻛﻪ ﻧﺒﺎﻳﺪ ﺁﻧﻘﺪﺭ ﺧﺸﻦ ﺭﻓﺘﺎﺭ ﻣﯽ‌ﮐﺮﺩﻳﺪ. ﻣﻤﻜﻦ ﺍﺳﺖ ﻋﻠﺖ ﺍﻳﻦ ﺍﻣﺮ ﻋﺼﺒﺎﻧﻴﺖ ﺷﻤﺎ ﻋﻠﻴﻪ ﻫﻤﺴﺎﻳﻪ ﻳﺎ ﺷﻮﻫﺮﺗﺎﻥ ﺑﻮﺩﻩ ﻭ ﻳﺎ ﺍﺻﻼ ﺁﻥ ﺭﻭﺯ ﺍﺯ ﻧﻈﺮ ﻋﺼﺒﯽ ﻇﺮﻓﻴﺖ ﺯﻳﺎﺩﯼ ﻧﺪﺍﺷﺘﻴﺪ ﻭ ﺗﻼﻓﯽ‌ﺍﺵ ﺭﺍ ﺳﺮ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭﺁﻭﺭﻳﺪ.
ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﭼﻪ ﻛﺎﺭ ﻛﻨﻴﻢ؟ ﻣﺴﻠﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻫﻤﻴﺸﻪ ﻧﻘﺶ ﺁﺩﻡ ﻛﺎﻣﻞ ﻭ ﺧﺪﺷﻪ‌ﻧﺎﭘﺬﻳﺮ ﺭﺍ ﺑﺎﺯﯼ ﻛﻨﻴﺪ ﻭ ﺑﮕﻮﻳﻴﺪ ﻛﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﺴﺘﺤﻖ ﺩﺭﻟﻨﺒﺮﯼ ﺑﻮﺩ ﻭ ﻳﺎ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺷﺠﺎﻉ‌ﺗﺮ ﺑﺎﺷﻴﺪ ﻭ ﻣﺜﻼ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ: «ﻣﯽ‌ﺩﻭﻧﯽ، ﻣﺎﻣﺎﻥ ﻋﺼﺒﺎﻧﯽ ﺷﺪ ﻭ ﺗﻮ ﺭﻭ ﺧﻴﻠﯽ ﺩﻋﻮﺍ ﻛﺮﺩ. ﻭﻟﯽ ﺍﻻﻥ ﻛﻪ ﻓﻜﺮ ﻣﯽ‌ﮐﻨﻢ ﻣﯽ‌ﺑﻴﻨﻢ ﻛﺎﺭﯼ ﻛﻪ ﻛﺮﺩﯼ ﺧﻴﻠﯽ ﻫﻢ ﺑﺪ ﻧﺒﻮﺩ. ﻣﻦ ﺍﻣﺮﻭﺯ ﺧﻮﺩﻡ ﺍﺯ ﻗﺒﻞ ﻋﺼﺒﯽ ﺑﻮﺩﻡ ﻭﻟﯽ ﺗﻘﺼﻴﺮ ﺗﻮ ﻧﺒﻮﺩ. ﺍﻳﻨﻪ ﻛﻪ ﺍﺯ ﺗﻮ ﻣﻌﺬﺭﺕ ﻣﯽ‌ﺧﻮﺍﻡ.»
ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺎ ﻣﺸﺎﻫﺪﻩ ﺍﻳﻨﻜﻪ ﺷﻤﺎ ﻫﻢ ﺍﻧﺴﺎﻥ ﻫﺴﺘﻴﺪ ﻭ ﺷﻤﺎ ﻫﻢ ﺍﺷﺘﺒﺎﻩ ﻣﯽ‌ﮐﻨﻴﺪ، ‌ﺍﺣﺴﺎﺱ ﻫﻤﺪﺭﺩﯼ ﮔﺮﻣﯽ ﻧﺴﺒﺖ ﺑﻪ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﻛﺮﺩ ﻭ ﻧﺘﻴﺠﻪ ﺑﺮﺍﯼ ﺷﻤﺎ ﻭ ﺧﻮﺩﺍﻧﮕﺎﺭﻩ ﺍﻭ ﺑﺴﻴﺎﺭ ﺧﻮﺏ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.
 
ﻣﺮﻭﺭﯼ ﺑﻜﻨﻴﻢ. ﮔﻔﺘﻴﻢ ﻛﻪ ﻧﺘﺎﻳﺞ ﻃﺒﻴﻌﯽ ﺭﻓﺘﺎﺭ ﻛﻮﺩﮎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﻧﺘﺎﻳﺞ ﻣﺼﻨﻮﻋﯽ ﻛﻪ ﺷﻤﺎ ﺑﺮﺍﻳﺶ ﺗﻌﻴﻴﻦ ﻣﯽ‌ﮐﻨﻴﺪ ﺗﻘﻮﻳﺖ ﮔﺮﺩﺩ. ﻭﻟﯽ ﺣﺘﯽ ﺍﮔﺮ ﻧﺘﺎﻳﺞ ﺍﻋﻤﺎﻝ ﻛﻮﺩﮎ ﺑﻴﺸﺘﺮ ﻣﺼﻨﻮﻋﯽ ﺑﺎﺷﺪ ﺗﺎ ﻃﺒﻴﻌﯽ، ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺁﻧﻬﺎ ﺑﺎﻳﺪ ﺗﺎﺑﻊ ﻳﮏ ﺳﺮﯼ ﺍﺻﻮﻝ ﭘﺎﻳﻪ‌ﺍﯼ ﺑﺎﺷﺪ.
