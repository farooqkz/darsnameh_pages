ﺩﺭﺱ ﻳﺎﺯﺩﻫﻢ- ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺳﺖ ﭼﭗ

ﺩﺭ ﺩﻭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﺩﺭ ﻣﻮﺭﺩ ﺁﻣﻮﺯﺵ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﮐﻮﺩﮎ ﺻﺤﺒﺖ ﮐﺮﺩﻳﻢ ﻭ ﺩﻭ ﻧﻮﻉ ﻓﮑﺮ ﮐﺮﺩﻥ ﺭﺍ ﺗﻌﺮﻳﻒ ﮐﺮﺩﻳﻢ. ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺳﺖ ﺭﺍﺳﺖ ﻭ ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺳﺖ ﭼﭗ. ﻣﻮﺭﺩ ﺍﻭﻝ ﺭﺍ ﺗﺎ ﺍﻳﻨﺠﺎ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ ﻭ ﺣﺎﻝ ﺑﻪ ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺳﺖ ﭼﭗ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ.

ﺣﺎﻻ «ﻓﻜﺮ ﻛﺮﺩﻥ ﺩﺳﺖ ﭼﭗ» ﺭﺍ ﺑﺮﺭﺳﯽ ﻛﻨﻴﻢ. ﺍﻳﻦ ﺑﺎﺭ ﺑﺎﻳﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﻛﻤﮏ ﻛﻨﻴﺪ ﺗﺎ ﺫﻫﻨﺶ ﺭﺍ ﻧﻪ ﺗﻨﻬﺎ ﺩﺭ ﺟﻬﺖ ﻏﺮﻳﻀﯽ ﻭ ﺗﺨﻴﻠﯽ ﺑﻠﻜﻪ ﺩﺭ ﺟﻬﺖ ﻣﻨﻄﻘﯽ ﻭ ﺁﮔﺎﻫﺎﻧﻪ ﺭﺷﺪ ﺩﻫﺪ. ﭼﮕﻮﻧﻪ؟


ﻗﺒﻞ ﺍﺯ ﻫﺮ ﭼﻴﺰ ﺫﻫﻦ ﺧﻼﻕ ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﻛﻨﻴﺪ.
ﻣﺘﺎﺳﻔﺎﻧﻪ ﺩﺭ ﺯﻣﺎﻥ ﻣﺎ ﺍﻳﻦ ﺍﺻﻄﻼﺡ ﺑﻪ ﻏﻠﻂ ﻭ ﻧﺎﻏﻠﻂ ﺑﻪ ﻛﺎﺭ ﮔﺮﻓﺘﻪ ﻣﯽ‌ﺷﻮﺩ. ﺗﻮﻟﻴﺪ‌ﻛﻨﻨﺪﮔﺎﻥ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺩﺭ ﻣﻮﺭﺩ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺧﻮﺩ ﺩﺭﺳﺖ ﻳﺎ ﻏﻠﻂ ﺗﺒﻠﻴﻎ ﻣﯽ‌ﻛﻨﻨﺪ ﻛﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﻳﺸﺎﻥ ﻣﻮﺟﺐ ﺑﺎﻻ ﺭﻓﺘﻦ ﺧﻼﻗﻴﺖ ﺫﻫﻨﯽ ﻛﻮﺩﻛﺎﻥ ﻣﯽ‌ﺷﻮﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺎﻳﺪ ﺑﻪ ﺍﺻﻞ ﻣﻌﻨﯽ ﺍﻳﻦ ﻛﻠﻤﻪ ﺑﺎﺯ ﮔﺮﺩﻳﻢ.

ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﭼﻴﺰﯼ ﺭﺍ ﺧﻠﻖ ﻣﯽ‌ﻛﻨﺪ، ﻳﻌﻨﯽ ﺍﻳﻨﻜﻪ ﺍﻭ ﺍﺯ ﻣﻮﺍﺩ ﺍﻭﻟﻴﻪ‌ﺍﯼ ﻛﻪ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﻣﯽ‌ﮔﺬﺍﺭﻳﺪ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﻣﯽ‌ﺳﺎﺯﺩ ﻛﻪ ﺳﺎﺧﺘﺎﺭ ﺁﻥ ﺭﺍ ﺩﺭ ﺫﻫﻨﺶ ﺭﻳﺨﺘﻪ ﺍﺳﺖ. ﻣﺜﻼ ﻧﻘﺎﺷﯽ ﻛﺮﺩﻥ ﺭﻭﯼ ﻛﺎﻏﺬ ﺳﻔﻴﺪ ﺭﺍ ﺑﺎ ﺭﻧﮓ ﻛﺮﺩﻥ ﻳﮏ ﺁﻟﺒﻮﻡ ﻧﻘﺎﺷﯽ ﺁﻣﺎﺩﻩ ﻣﻘﺎﻳﺴﻪ ﻛﻨﻴﺪ. ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻳﮏ ﺁﻟﺒﻮﻡ ﻧﻘﺎﺷﯽ ﺭﺍ ﺭﻧﮓ‌ﺁﻣﻴﺰﯼ ﻣﯽ‌ﻛﻨﺪ، ﺳﺎﺧﺘﺎﺭ ﻛﺘﺎﺏ ﺍﺯ ﻗﺒﻞ ﺭﻳﺨﺘﻪ ﺷﺪﻩ ﺍﺳﺖ. ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﭼﻴﺰﯼ ﺧﻠﻖ ﻧﻤﯽ‌ﻛﻨﺪ. ﺗﻨﻬﺎ ﭼﻴﺰﯼ ﻛﻪ ﻳﺎﺩ ﻣﯽ‌ﮔﻴﺮﺩ ﺑﻴﺮﻭﻥ ﻧﺰﺩﻥ ﺍﺯ ﻣﺤﺪﻭﺩﻩ ﻧﻘﺎﺷﯽ ﻫﻨﮕﺎﻡ ﺭﻧﮓ ﻛﺮﺩﻥ ﺍﺳﺖ. ﺍﻳﻨﺠﺎ ﺧﻼﺋﯽ ﺑﺮﺍﯼ «ﺧﻼﻗﻴﺖ» ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﻭﻟﯽ ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﺍﻭ ﺭﻭﯼ ﻛﺎﻏﺬ ﺳﻔﻴﺪ ﻧﻘﺎﺷﯽ ﻣﯽ‌ﻛﻨﺪ، ﭼﻴﺰﯼ ﺧﻠﻖ ﻣﯽ‌ﻛﻨﺪ ﻛﻪ ﺳﺎﺧﺘﺎﺭﺵ ﺩﺭ ﺫﻫﻦ ﺧﻮﺩ ﺍﻭ ﺭﻳﺨﺘﻪ ﺷﺪﻩ ﺍﺳﺖ، ﻣﻮﺍﺩ ﺍﻭﻟﻴﻪ ﺍﻭ ﻧﻴﺰ ﻣﺪﺍﺩ ﻭ ﻛﺎﻏﺬ ﺑﻮﺩﻩ ﺍﺳﺖ.


ﺍﺯ ﺍﻳﻦ ﺟﻬﺖ ﺍﺳﺖ ﻛﻪ ﺗﺎﻛﻴﺪ ﺩﺍﺭﻳﻢ ﻛﻪ ﻛﻮﺩﮎ ﺑﺎﻳﺪ ﻣﻮﺍﺩ ﺍﻭﻟﻴﻪ ﺧﺎﻡ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻛﻪ ﺍﺯ ﻗﺒﻞ ﺩﺭﺳﺖ ﻧﺸﺪﻩ ﺑﺎﺷﻨﺪ: ﻣﺜﻼ ﻣﺪﺍﺩ، ﻛﺎﻏﺬ، ﻣﺎﮊﻳﮏ، ﺁﺑﺮﻧﮓ، ﺧﻤﻴﺮ، ﻣﻜﻌﺐ‌ﻫﺎﯼ ﺧﺎﻧﻪ‌ﺳﺎﺯﯼ، ﻟﮕﻮ، ﺧﺎﻙ، ﺷﻦ ﻭ ﻏﻴﺮﻩ. ﺫﻫﻦ ﻓﺮﺯﻧﺪ ﺷﻤﺎﺳﺖ ﻛﻪ ﺑﻪ ﺍﻳﻦ ﻣﻮﺍﺩ ﺷﻜﻞ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺳﺎﺧﺘﺎﺭﯼ ﺑﺮﺍﯼ ﺁﻥ‌ﻫﺎ ﻣﯽ‌ﺳﺎﺯﺩ.

ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﺎﻳﺪ ﭼﻴﺰﯼ ﺭﺍ ﺑﻪ ﻭﺍﻟﺪﻳﻦ ﺧﺎﻃﺮ ﻧﺸﺎﻥ ﻛﻨﻴﻢ. ﺁﻥ‌ﻫﺎ ﮔﺎﻫﯽ ﺑﺮﺩﺍﺷﺖ ﺑﺪﯼ ﺍﺯ ﺍﺻﺮﺍﺭﯼ ﻛﻪ ﺑﺮﺍﯼ ﺁﻣﺎﺩﻩ ﻛﺮﺩﻥ ﺷﺮﺍﻳﻂ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ «ﺧﻼﻗﻴﺖ» ﺩﺭ ﻛﻮﺩﮎ ﻣﯽ‌ﺷﻮﺩ ﺩﺍﺭﻧﺪ ﻭ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﻨﺪ ﻛﻪ: «ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﺍﻫﻤﻴﺖ ﺯﻳﺎﺩﯼ ﻧﺪﺍﺭﻩ، ﺑﭽﻪ ﻣﻦ ﻛﻪ ﻧﻤﯽ‌ﺧﻮﺍﺩ ﻫﻨﺮﻣﻨﺪ ﺑﺸﻪ.» ﻭﻟﯽ ﺍﺻﺮﺍﺭ ﻣﺎ ﺑﺮﺍﯼ ﺑﻪ‌ ﻭﺟﻮﺩ ﺁﻭﺭﺩﻥ ﺷﺮﺍﻳﻂ ﺧﻼﻗﻴﺖ ﺑﺮﺍﯼ ﻛﻮﺩﮎ ﺍﺻﻼ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﻛﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺍﻭ ﻫﻨﺮﻣﻨﺪ ﺑﺸﻮﺩ. ﺷﻤﺎ ﺑﻪ ﺍﻭ ﻛﻤﮏ ﻧﻤﯽ‌­ﮐﻨﻴﺪ ﮐﻪ ﻫﻨﺮﻣﻨﺪ ﺷﻮﺩ، ﺑﻠﻜﻪ ﺑﻪ ﺍﻭ ﻛﻤﮏ ﻣﯽ‌ﻛﻨﻴﺪ ﻛﻪ ﺭﻭﺣﻴﻪ‌ﺍﯼ ﺧﻼﻕ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺑﻪ ﺍﻭ ﺍﻣﻜﺎﻥ ﻣﯽ‌ﺩﻫﻴﺪ ﻛﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺴﺶ ﺭﺍ  ﺭﺷﺪ ﺩﻫﺪ ﺯﻳﺮﺍ ﻣﯽ‌ﺑﻴﻨﺪ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻣﻮﺍﺩ ﺧﺎﻡ ﺷﻜﻞ ﻭ ﻧﻈﻢ ﺑﺪﻫﺪ. ﺷﻤﺎ ﺑﺎ ﺍﻳﻦ ﻛﺎﺭﺗﺎﻥ ﻗﺪﺭﺕ ﺩﺭﮎ ﺍﻭ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﺍﺣﺴﺎﺳﺎﺕ، ﺁﮔﺎﻫﯽ ﺍﺯ ﻭﺍﻗﻌﻴﺎﺕ، ﺑﻜﺮ ﺑﻮﺩﻥ ﻭ ﺍﻣﻜﺎﻥ ﺗﻄﺎﺑﻖ ﺩﺍﺩﻧﺶ ﺭﺷﺪ ﻣﯽ‌ﺩﻫﻴﺪ. ﺍﻳﻦ ﺧﺼﻠﺖ «ﺧﻼﻕ» ﺩﺭ ﻫﺮ ﻛﺎﺭﯼ ﻛﻪ ﺍﻭ ﺩﺭ ﺁﻳﻨﺪﻩ ﺍﻧﺠﺎﻡ ﺩﻫﺪ ﺑﺮﺍﻳﺶ ﻻﺯﻡ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﻗﺒﻼ ﺭﻭﯼ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﻛﻪ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺕ ﻛﻮﺩﮎ ﻣﻮﺟﺐ ﺳﻼﻣﺘﯽ ﺭﻭﺣﯽ ﺍﻭ ﻣﯽ‌ﺷﻮﺩ ﺗﺎﻛﻴﺪ ﻛﺮﺩﻩ‌ﺍﻳﻢ. ﺍﻳﻦ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺕ، ﺑﺮﺍﯼ ﺍﻳﻨﻜﻪ ﺍﻭ ﺑﻪ ﺣﺪﺍﻛﺜﺮ ﺭﺷﺪ ﺫﻫﻨﯽ ﺑﺮﺳﺪ ﻧﻴﺰ ﻻﺯﻡ ﺍﺳﺖ. ﺷﻤﺎ ﺑﺎ ﺗﺸﻮﻳﻖ ﺍﻭ ﺑﻪ ﺑﻴﺎﻥ ﺁﻥ­ﭼﻪ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﻛﻨﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻛﻤﻜﺶ ﻛﻨﻴﺪ ﺭﺍﺣﺖ‌ﺗﺮ ﺑﻪ ﺿﻤﻴﺮ ﻧﺎﺧﻮﺩﺁﮔﺎﻫﺶ ﺩﺳﺖ ﻳﺎﺑﺪ. ﺑﺎﻳﺪ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﺪ ﻛﻪ ﺩﺭ ﻣﻮﺭﺩ ﺍﺣﺴﺎﺳﺎﺕ، «ﻭﺍﻗﻌﯽ»، «ﺍﺷﺘﺒﺎﻩ»، «ﺧﻮﺏ» ﻭ ﻳﺎ «ﺑﺪ» ﺍﺻﻼ ﻣﻌﻨﯽ ﻧﺪﺍﺭﺩ. ﻭﺍﻗﻌﯽ، ﺍﺷﺘﺒﺎﻩ، ﺑﺪ ﻭ ﺧﻮﺏ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻋﻤﺎﻝ ﻫﺴﺘﻨﺪ ﻭ ﺩﺭ ﻣﻮﺭﺩ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺍﻓﻜﺎﺭ ﺩﺭﻭﻧﯽ ﺻﺪﻕ ﻧﻤﯽ‌ﻛﻨﻨﺪ. ﻳﻜﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ‌ﻫﺎ ﺑﺮﺍﯼ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩﻥ ﺧﻼﻗﻴﺖ ﺩﺭ ﻳﮏ ﻓﺮﺩ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﺩﺭ ﻛﻮﺩﻛﯽ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻧﺪ ﻛﻪ ﭼﻨﻴﻦ ﻭ ﭼﻨﺎﻥ ﺍﺣﺴﺎﺳﯽ ﺧﻮﺏ ﻳﺎ ﺑﺪ ﺍﺳﺖ. ﺩﺭ ﺣﺎﻟﯽ‌ﻛﻪ ﻣﯽ‌ﺩﺍﻧﻴﻢ ﻫﺮ ﭼﻘﺪﺭ ﻳﮏ ﺷﺨﺺ ﺑﺎﻟﻎ ﺍﻧﺪﻳﺸﻪ­‌ﻫﺎﯼ ﺍﻧﻘﻼﺑﯽ، ﺑﺪﻳﻊ ﻭ ﺗﺎﺯﻩ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻫﻤﺎﻧﻘﺪﺭ ﺍﻣﻜﺎﻥ ﻣﻮﻓﻘﻴﺖ ﺣﺮﻓﻪ‌ﺍﯼ ﺍﻭ ﻧﻴﺰ ﺑﻴﺸﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﺑﻪ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺗﺶ ﻛﻨﻴﺪ. ﺑﺎ ﺑﻴﺎﻥ ﺍﻳﻦ ﺍﺣﺴﺎﺳﺎﺕ ﺍﺳﺖ ﻛﻪ ﻛﻮﺩﮎ ﻭ ﻳﺎ ﻓﺮﺩﯼ ﺑﺎﻟﻎ ﺑﻪ ﺿﻤﻴﺮ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺧﻮﺩ ﺩﺳﺖ ﻣﯽ‌ﻳﺎﺑﺪ ﻭ ﺍﺯ ﺍﻳﻦ ﺿﻤﻴﺮ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺍﺳﺖ ﻛﻪ ﺍﻳﺪﻩ‌ﻫﺎﯼ ﺑﺰﺭﮒ ﻭ ﺧﻼﻗﻴﺖ‌ﻫﺎ ﻣﯽ‌ﺟﻮﺷﻨﺪ. ﺍﺷﺨﺎﺹ ﺑﺎﻟﻎ ﺧﺸﮏ، ﻓﻀﻞ‌ﻓﺮﻭﺵ ﻭ ﻛﺴﻞ‌ﻛﻨﻨﺪﻩ ﻛﺴﺎﻧﯽ ﻫﺴﺘﻨﺪ ﻛﻪ ﺩﺭﻫﺎﯼ ﺯﻧﺪﮔﯽ ﺍﺣﺴﺎﺳﺎﺗﺸﺎﻥ ﺭﺍ ﺑﺴﺘﻪ‌ﺍﻧﺪ ﻭ ﻫﻴﭻ‌ﮔﻮﻧﻪ ﺭﺍﺑﻄﻪ‌ﺍﯼ ﺑﺎ ﺿﻤﻴﺮ ﻧﺎﺧﻮﺩﺁﮔﺎﻫﺸﺎﻥ ﻧﺪﺍﺭﻧﺪ.

ﻣﺎ ﺑﺎﻳﺪ ﺩﺭﻫﺎﯼ ﺯﻧﺪﮔﯽ ﻋﺎﻃﻔﯽ ﺭﺍ ﺗﺎ ﺯﻣﺎﻧﯽ ﻛﻪ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺧﺮﺩﺳﺎﻟﻨﺪ ﺑﺎﺯ ﻛﻨﻴﻢ ﻭ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﺎﺯ ﻧﮕﺎﻩ ﺩﺍﺭﻳﻢ.
ﺑﺮﺍﯼ ﺍﻓﺰﺍﻳﺶ ﻗﺪﺭﺕ ﺗﺨﻴﻞ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﺎﻳﺪ ﻣﻨﺎﺑﻊ ﺯﻳﺎﺩﯼ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﺑﮕﺬﺍﺭﻳﺪ. ﺩﺭ ﺩﻭﺭﻩ ﻣﺎ ﻣﺮﺩﻡ ﺑﻴﺸﺘﺮ ﺑﻪ ﻭﺍﻗﻌﻴﺎﺕ ﺗﻮﺟﻪ ﻣﯽ‌ﻛﻨﻨﺪ ﻭ ﺗﻔﻨﻦ ﻭ ﺗﺨﻴﻼﺕ ﺭﺍ ﺑﻪ ﻛﻨﺎﺭﯼ ﮔﺬﺍﺷﺘﻪ‌ﺍﻧﺪ. ﻛﺘﺎﺏ‌ﻫﺎﯼ ﺧﻮﺑﯽ ﻛﻪ ﺩﺭﺑﺎﺭﻩ ﻣﻮﺿﻮﻋﺎﺕ ﻭﺍﻗﻌﯽ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺑﺎﺷﻨﺪ ﺑﻪ ﻭﻓﻮﺭ ﻳﺎﻓﺖ ﻣﯽ‌ﺷﻮﻧﺪ، ﺣﺎﻝ ﺁﻥ­ﮐﻪ ﻛﺘﺎﺏ‌ﻫﺎﯼ ﺧﻮﺏ ﺗﺨﻴﻠﯽ ﻧﺎﺩﺭﻧﺪ. ﺩﺭﺣﺎﻟﯽ ﻛﻪ ﺗﺨﻴﻞ ﺭﺍ ﺑﺎﻳﺪ ﺭﺷﺪ ﺩﺍﺩ ﻧﻪ ﺍﻳﻦ ﻛﻪ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩ. ﻛﻮﺩﻛﯽ ﻛﻪ ﻛﺘﺎﺏ‌ﻫﺎﻳﯽ ﻧﻈﻴﺮ ﺁﻟﻴﺲ ﺩﺭ ﺳﺮﺯﻣﻴﻦ ﻋﺠﺎﻳﺐ ﻭ ﻳﺎ ﺷﺎﺯﺩﻩ ﻛﻮﭼﻮﻟﻮ ﺭﺍ ﺑﺮﺍﻳﺶ ﻧﺨﻮﺍﻧﺪﻩ ﺑﺎﺷﻨﺪ ﺍﺯ ﻣﻨﺒﻊ ﺑﺰﺭﮔﯽ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﺴﺖ ﻗﻮﻩ ﺗﺼﻮﺭ ﻭ ﺗﺨﻴﻞ ﻛﻮﺩﻛﺎﻧﻪ‌­ﺍﺵ ﺭﺍ ﺭﺷﺪ ﺩﻫﺪ ﻣﺤﺮﻭﻡ ﻣﺎﻧﺪﻩ ﺍﺳﺖ.


ﺑﺮﺍﯼ ﺭﺷﺪ ﻗﻮﻩ ﺗﺼﻮﺭ ﻭ ﺗﺨﻴﻞ ﺩﺭ ﻛﻮﺩﮎ ﺍﺯ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﻳﯽ ﭼﻮﻥ ﺭﻗﺺ، ﻣﻮﺳﻴﻘﯽ، ﻫﻨﺮﻫﺎﯼ ﺗﺠﺴﻤﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﻬﺮﻩ ﺟﻮﻳﻴﻢ. ﺍﻳﻦ ﮔﻮﻧﻪ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎ ﺑﻪ ﻛﻮﺩﮎ ﺍﻣﻜﺎﻥ ﺑﻴﺎﻥ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺭﺷﺪ ﻗﻮﻩ ﺗﺨﻴﻞ ﻭ ﺗﺼﻮﺭﺵ ﺭﺍ ﻣﯽ‌ﺩﻫﻨﺪ.
ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﺮﺍﯼ ﺍﻭ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﻳﯽ ﺗﻌﺮﻳﻒ ﻛﻨﻴﻢ ﻭ ﺍﻭ ﺭﺍ ﺗﺸﻮﻳﻖ ﺑﻪ ﺗﻌﺮﻳﻒ ﺩﺍﺳﺘﺎﻥ‌ﻫﺎﯼ ﻣﺸﺎﺑﻪ ﻛﻨﻴﻢ. ﺗﻔﺎﻭﺕ ﺑﺴﻴﺎﺭﯼ ﺍﺳﺖ ﺑﻴﻦ ﺩﺍﺳﺘﺎﻧﯽ ﻛﻪ ﺑﺮﺍﯼ ﺍﻭ ﻣﯽ‌ﺧﻮﺍﻧﻴﺪ ﻭ ﺩﺍﺳﺘﺎﻧﯽ ﻛﻪ ﺑﺮﺍﻳﺶ ﺗﻌﺮﻳﻒ ﻣﯽ‌ﻛﻨﻴﺪ. ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﺷﻤﺎ ﺧﻮﺩﺗﺎﻥ ﺩﺍﺳﺘﺎﻥ ﻭ ﻗﺼﻪ‌ﺍﯼ ﺭﺍ ﺑﺮﺍﯼ ﺍﻭ ﺗﻌﺮﻳﻒ ﻣﯽ‌ﻛﻨﻴﺪ، ﻛﻤﻜﺶ ﻣﯽ‌ﻛﻨﻴﺪ ﺑﺎﻭﺭ ﻛﻨﺪ ﻛﻪ ﺍﻭ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺍﺳﺘﺎﻧﯽ ﺍﺯ ﺧﻮﺩ ﺩﺭ ﺑﻴﺎﻭﺭﺩ.

ﻛﻮﺩﻛﺎﻥ ﺷﻴﻔﺘﻪ‌ ﺷﻨﻴﺪﻥ ﺗﺠﺎﺭﺏ ﺧﻮﺩ ﺷﻤﺎ ﻫﻨﮕﺎﻣﯽ ﮐﻪ «ﻫﻢ‌ﺳﻦ ﻭ ﺳﺎﻝ ﺍﻭ ﺑﻮﺩﻳﺪ» ﻫﺴﺘﻨﺪ. ﺍﺯ ﺍﻳﻨﻜﻪ ﻛﻤﯽ ﺑﻪ ﻣﺴﺎﺋﻞ ﭘﺮﻭﺑﺎﻝ ﺑﺪﻫﻴﺪ ﺗﺎ ﺟﺮﻳﺎﻥ ﺟﺎﻟﺐ‌ﺗﺮ ﺑﺎﺷﺪ ﺍﺑﺎﻳﯽ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.
ﻭﻗﺘﯽ ﻗﺼﻪ‌ﺍﯼ ﺗﻌﺮﻳﻒ ﻣﯽ‌ﻛﻨﻴﺪ ﺻﺪﺍﻳﺘﺎﻥ ﺭﺍ ﺗﻐﻴﻴﺮ ﺑﺪﻫﻴﺪ. ﮔﺎﻫﯽ ﺳﺮﻳﻊ ﻭ ﮔﺎﻫﯽ ﺁﺭﺍﻡ ﻭ ﺑﺎ ﻟﺤﻨﯽ ﻛﻪ ﺍﻧﮕﺎﺭ ﻧﻤﺎﻳﺸﻨﺎﻣﻪ ﺑﺎﺯﯼ ﻣﯽ‌ﻛﻨﻴﺪ. ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﺑﻠﻨﺪ ﻭ ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﺁﻫﺴﺘﻪ ﺻﺤﺒﺖ ﻛﻨﻴﺪ. ﻫﺮ ﺟﺎ ﻛﻪ ﺍﻣﻜﺎﻥ ﺩﺍﺭﺩ ﺻﺪﺍﻫﺎﻳﯽ ﺷﺒﻴﻪ «ﻭﻳﺰﺯﺯ» (ﺻﺪﺍﯼ ﺯﻧﺒﻮﺭ) ﻭ ﻳﺎ ﻫﻮﻫﻮ ﭼﯽ‌ﭼﯽ (ﺻﺪﺍﯼ ﻗﻄﺎﺭ) ﺩﺭ ﺑﻴﺎﻭﺭﻳﺪ، ﻛﻮﺩﻛﺎﻥ ﻗﺒﻞ ﺍﺯ ﺷﺶ ﺳﺎﻝ ﺍﻳﻦ ﻧﻮﻉ ﺻﺪﺍﻫﺎ ﺭﺍ ﺧﻴﻠﯽ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ.

ﻳﮏ ﺭﺍﻩ ﺧﻮﺏ ﺑﺮﺍﯼ ﺗﺸﻮﻳﻖ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺟﻬﺖ ﺩﺍﺳﺘﺎﻥ ﺩﺭﺳﺖ ﻛﺮﺩﻥ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻳﮏ ﻋﻜﺲ ﻭ ﻳﺎ ﺗﺼﻮﻳﺮ ﺑﺮﺍﯼ ﺷﺮﻭﻉ ﻛﺮﺩﻥ ﺩﺍﺳﺘﺎﻧﺘﺎﻥ ﺍﺳﺖ. ﻳﮏ ﺗﺼﻮﻳﺮ ﺟﺎﻟﺐ ﺩﺭ ﻳﮏ ﻣﺠﻠﻪ ﭘﻴﺪﺍ ﻛﻨﻴﺪ ﻭ ﻧﺸﺎﻧﺶ ﺩﻫﻴﺪ ﻭ ﺑﻪ ﺍﻭ ﻧﺸﺎﻥ ﺩﻫﻴﺪ ﻛﻪ ﭼﮕﻮﻧﻪ ﺍﺯ ﺭﻭﯼ ﺁﻥ ﺗﺼﻮﻳﺮ ﻣﯽ‌ﺗﻮﺍﻥ ﻳﮏ ﺩﺍﺳﺘﺎﻥ ﻛﻮﺗﺎﻩ ﻧﻘﻞ ﻛﺮﺩ. ﺑﻌﺪ ﻳﮏ ﺗﺼﻮﻳﺮ ﺩﻳﮕﺮ ﻧﺸﺎﻧﺶ ﺩﻫﻴﺪ ﻭ ﺑﮕﻮﻳﻴﺪ ﺣﺎﻻ ﻧﻮﺑﺖ ﺗﻮﺳﺖ.

ﺭﻭﺵ ﺩﻳﮕﺮ: ﺍﻭﻝ ﺩﺍﺳﺘﺎﻥ ﺭﺍ ﺗﻌﺮﻳﻒ ﻛﻨﻴﺪ ﻭ ﺑﮕﺬﺍﺭﻳﺪ ﺍﻭ ﺁﻥ ﺭﺍ ﺗﻤﺎﻡ ﻛﻨﺪ. ﺍﮔﺮ ﺗﻌﺪﺍﺩ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺑﻴﺶ ﺍﺯ ﻳﻜﯽ ﺑﺎﺷﺪ، ﻫﺮ ﻛﺪﺍﻡ ﺑﻪ ﻧﻮﺑﺖ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺟﻤﻠﻪ‌ﺍﯼ ﺑﮕﻮﻳﻨﺪ.
ﻣﺴﻠﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﮕﻮﻳﻴﺪ: «ﻧﻪ، ﻣﻦ ﻧﻤﯽ‌ﺗﻮﻧﻢ. ﻣﻦ ﺑﻠﺪ ﻧﻴﺴﺘﻢ ﺩﺍﺳﺘﺎﻥ ﺗﻌﺮﻳﻒ ﻛﻨﻢ!» ﺍﮔﺮ ﺍﻳﻦ ﻃﻮﺭ ﻓﻜﺮ ﻣﯽ‌ﻛﻨﻴﺪ ﺑﮕﺬﺍﺭﻳﺪ ﺧﻴﺎﻟﺘﺎﻥ ﺭﺍ ﺭﺍﺣﺖ ﻛﻨﻢ: ﺣﺘﯽ ﺍﮔﺮ ﺑﺪﺗﺮﻳﻦ، ﻧﺎﺷﯽ‌ﺗﺮﻳﻦ ﻭ ﺑﯽ‌ﺍﺳﺘﻌﺪﺍﺗﺮﻳﻦ ﻧﻘﺎﻝ‌ﻫﺎﯼ ﺩﻧﻴﺎ ﻫﻢ ﺑﺎﺷﻴﺪ، ﻛﻮﭼﻮﻟﻮﯼ ﺷﻤﺎ، ﺷﻤﺎ ﺭﺍ ﺑﻬﺘﺮﻳﻦ ﻧﻘﺎﻝ ﺩﻧﻴﺎ ﺗﺼﻮﺭ ﺧﻮﺍﻫﺪ ﻛﺮﺩ! ﻫﺮﮔﺰ ﺷﻨﻮﻧﺪﻩ‌ﺍﯼ ﺑﻬﺘﺮ ﺍﺯ ﺍﻭ ﻧﺨﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺗﺮﺱ ﺭﺍﻩ ﻧﺪﻫﻴﺪ  ﻭ ﺍﻣﺘﺤﺎﻥ ﻛﻨﻴﺪ.


ﻭ ﺑﺎﻻﺧﺮﻩ، ﺑﺮﺍﯼ ﺍﻳﻨﻜﻪ ﻛﻮﺩﮎ ﻗﻮﻩ ﺗﺨﻠﻴﺶ ﺭﺷﺪ ﺷﺪﻳﺪﯼ ﻳﺎﺑﺪ، ﺑﺮﺩﺍﺷﺖ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺍﺯ ﺯﻧﺪﮔﯽ ﺍﻫﻤﻴﺖ ﺯﻳﺎﺩﯼ ﺩﺍﺭﺩ. ﺑﻌﻀﯽ‌ﻫﺎ ﺑﺎﻭﻗﺎﺭﺗﺮ ﻭ ﭘﺮﻃﻤﻄﺮﺍﻕ‌ﺗﺮﻧﺪ، ﺑﻌﻀﯽ‌ﻫﺎ ﺭﻳﺴﮏ ﺑﻴﺸﺘﺮﯼ ﻣﯽ‌ﻛﻨﻨﺪ. ﻫﺮ ﭼﻪ ﻫﺴﺘﻴﺪ ﻫﻤﺎﻥ ﺑﻤﺎﻧﻴﺪ. ﺳﻌﯽ ﻧﻜﻨﻴﺪ ﭼﻴﺰ ﺩﻳﮕﺮﯼ ﺑﺎﺷﻴﺪ ﻭﻟﯽ ﻫﺮ ﭼﻘﺪﺭ ﺑﻴﺸﺘﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺩﻟﻘﮏ ﺑﺎﺯﯼ ﻛﻨﻴﺪ ﻫﻤﺎﻧﻘﺪﺭ ﻗﺪﺭﺕ ﺗﺨﻴﻞ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﻴﺸﺘﺮ ﺭﺷﺪ ﻣﯽ‌ﻳﺎﺑﺪ.

ﺑﺎﻳﺪ ﺑﻪ ﻛﻮﺩﮎ ﻛﻤﮏ ﻛﺮﺩ ﺗﺎ ﺗﺼﺎﻭﻳﺮ ﺫﻫﻨﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻣﻨﻈﻮﺭ، ﺗﺼﺎﻭﻳﺮ ﺫﻫﻨﯽ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺍﺳﺖ. ﭼﻨﺪﯼ ﭘﻴﺶ ﺍﺯ ﺗﻌﺪﺍﺩﯼ ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﺗﺴﺖ ﻫﻮﺵ ﮔﺮﻓﺘﻨﺪ. ﺗﺴﺖ‌ﻫﺎﯼ ﻣﺮﺑﻮﻁ ﺑﻪ ﺣﺴﺎﺏ ﻫﻤﻴﺸﻪ ﺑﺎ ﺟﻤﻠﻪ ﺯﻳﺮ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﺪ: «ﺣﺎﻻ ﺑﺒﻴﻨﻢ ﻭﺿﻊ ﺣﺴﺎﺏ ﺷﻤﺎ ﭼﻄﻮﺭ ﺍﺳﺖ.» ﻭ ﻫﺮ ﺩﻓﻌﻪ ﺣﺪﺍﻗﻞ ﻳﮏ ﻧﻔﺮ ﻣﯽ‌ﮔﻔﺖ: «ﻣﻄﻤﺌﻨﻢ ﻛﻪ ﺭﺩ ﻣﯽ‌ﺷﻢ ﻣﻦ ﻫﻤﻴﺸﻪ ﺗﻮ ﺭﻳﺎﺿﻴﺎﺕ ﺿﻌﻴﻒ ﺑﻮﺩﻡ.» ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺷﺨﺎﺹ ﺗﺼﻮﻳﺮ ﺫﻫﻨﯽ ﻧﺎﺧﻮﺩﺁﮔﺎﻫﯽ ﻛﻪ ﺍﺯ ﺭﻳﺎﺿﯽ ﺧﻮﺩ ﺩﺍﺷﺘﻨﺪ ﻣﻨﻔﯽ ﺑﻮﺩ، ﺯﻳﺮﺍ ﺍﻭﻟﻴﻦ ﺑﺮﺧﻮﺭﺩﻫﺎﯼ ﺁﻧﺎﻥ ﺑﺎ ﺭﻳﺎﺿﻴﺎﺕ ﺩﺭ ﺩﻭﺭﺍﻥ ﻛﻮﺩﻛﯽ ﻣﻨﻔﯽ ﺑﻮﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺎﻳﺪ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻛﻤﮏ ﻛﻨﻴﺪ ﻛﻪ ﺗﺼﺎﻭﻳﺮ ﻣﺜﺒﺘﯽ ﺍﺯ ﻣﻮﺍﺩ ﻣﺨﺘﻠﻒ ﺩﺭﺳﯽ ﻛﻪ ﺩﺭ ﻣﺪﺭﺳﻪ ﺧﻮﺍﻫﺪ ﺁﻣﻮﺧﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.


ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﭼﻨﻴﻦ ﻫﺪﻓﯽ ﭼﻪ ﺑﺎﻳﺪ ﻛﺮﺩ؟ ﺳﻌﯽ ﻛﻨﻴﺪ ﻛﻪ ﺍﻭﻟﻴﻦ ﺑﺮﺧﻮﺭﺩﺵ ﺑﺎ ﻳﮏ ﻣﻮﺿﻮﻉ ﺩﺭ ﻣﺤﻴﻂ ﺟﺎﻟﺐ ﻭ ﺧﻮﺑﯽ ﺑﺎﺷﺪ.
ﻣﺜﻼ ﺷﻴﻤﯽ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ ﻛﻪ ﻳﻜﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺍﺳﺖ ﻛﻪ ﺧﻴﻠﯽ ﺧﻮﺏ ﺑﺎ «ﺗﺠﺎﺭﺏ» ﭘﻴﺶ ﺍﺯ ﺩﺑﺴﺘﺎﻥ ﺳﺎﺯﮔﺎﺭﯼ ﺩﺍﺭﺩ. ﺣﻞ ﻛﺮﺩﻥ ﻧﻤﮏ ﻳﺎ ﺷﻜﺮ ﺩﺭ ﺁﺏ ﻣﯽ‌ﺗﻮﺍﻧﺪ «ﻳﮏ ﺁﺯﻣﺎﻳﺶ ﺷﻴﻤﻴﺎﻳﯽ» ﺑﺸﻮﺩ. ﻭ ﻳﺎ ﺧﻴﻠﯽ ﺳﺎﺩﻩ‌ﺗﺮ، ﺷﻴﺮﻳﻨﯽ ﺩﺭﺳﺖ ﻛﺮﺩﻥ،  ﺩﺭ ﻣﻮﺭﺩ ﺑﻪ ﻛﺎﺭ ﺑﺮﺩﻥ ﻛﻠﻤﻪ ﺷﻴﻤﯽ ﺗﺮﺩﻳﺪ ﺑﻪ ﺧﻮﺩ ﺭﺍﻩ ﻧﺪﻫﻴﺪ ﻭ ﺑﻪ ﺍﻭ ﺗﻮﺻﻴﻪ ﻛﻨﻴﺪ ﻛﻪ ﺗﺠﺎﺭﺑﯽ ﺍﻧﺠﺎﻡ ﺩﻫﺪ. «ﺁﺯﻣﺎﻳﺸﮕﺎﻩ» ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻇﺮﻓﺸﻮﻳﯽ ﺁﺷﭙﺰﺧﺎﻧﻪ ﺑﺎﺷﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﻄﺎﻟﻌﻪ ﺩﺭﺱ ﺷﻴﻤﯽ ﺭﺍ ﺩﺭ ﺩﺑﻴﺮﺳﺘﺎﻥ ﺁﻏﺎﺯ ﻛﻨﺪ، ﺍﺯ ﺧﻴﻠﯽ ﭘﻴﺸﺘﺮ، ﺗﺼﻮﻳﺮﯼ ﻣﺜﺒﺖ ﺍﺯ ﺷﻴﻤﯽ ﺩﺍﺭﺩ. ﻫﻤﻴﻦ ﻛﺎﺭ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﻣﻮﺭﺩ ﻓﻴﺰﻳﮏ، ﺯﺑﺎﻥ‌ﻫﺎﯼ ﺧﺎﺭﺟﯽ، ﺭﻳﺎﺿﻴﺎﺕ ﻭ ﺧﻴﻠﯽ ﺍﺯ ﻣﻮﺍﺩ ﺩﺭﺳﯽ ﺩﻳﮕﺮ ﺍﻧﺠﺎﻡ ﺩﺍﺩ ﻭ ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺍﻭ ﺭﺍ ﻛﻤﮏ ﻛﺮﺩ ﺗﺎ ﺗﺼﻮﻳﺮﯼ ﻣﺜﺒﺖ ﺍﺯ ﺍﻳﻦ ﻣﻮﺍﺩ ﺩﺭﺳﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.
