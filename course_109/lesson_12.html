ﺩﺭﺱ ﺩﻫﻢ- ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺳﺖ ﺭﺍﺳﺖ

ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﮔﻔﺘﻴﻢ ﭘﻨﺞ ﺳﺎﻝ ﺍﻭﻝ ﺯﻧﺪﮔﯽ ﮐﻮﺩﮎ ﺩﺭ ﺭﺷﺪ ﺫﻫﻨﯽ ﺍﻭ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺩﺍﺭﺩ، ﺑﻪ ﺍﻳﻦ ﻋﻠﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺳﺎﻝ‌ﻫﺎ ﺍﺳﺖ ﮐﻪ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎﯼ ﺍﺳﺎﺳﯽ ﻳﺎﺩﮔﻴﺮﯼ ﺭﺍ ﮐﺴﺐ ﻣﯽ‌ﮐﻨﺪ. ﺍﻳﻦ ﻗﺎﺑﻠﻴﺖ‌ﻫﺎ ﮐﻤﮏ ﺷﺎﻳﺎﻧﯽ ﺑﻪ ﺑﺎﻻ ﺑﺮﺩﻥ ﻫﻮﺵ ﻭ ﺳﺮﺯﻧﺪﮔﯽ ﮐﻮﺩﮎ ﻣﯽ‌ﮐﻨﺪ.

ﺩﺭ ﺍﻭﻟﻴﻦ ﻗﺪﻡ، ﺍﺯ ﺍﻫﻤﻴﺖ ﺁﻣﻮﺯﺵ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﮐﻮﺩﮎ ﮔﻔﺘﻴﻢ ﻭ ﺩﻭ ﻧﻮﻉ ﻓﮑﺮ ﮐﺮﺩﻥ ﺭﺍ ﺗﻌﺮﻳﻒ ﮐﺮﺩﻳﻢ: «ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺳﺖ ﺭﺍﺳﺖ» ﻭ «ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺳﺖ ﭼﭗ». ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﻧﻮﻉ ﺍﻭﻝ ﺭﺍ ﺑﺮﺭﺳﯽ ﮐﺮﺩﻳﻢ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻧﻴﺰ ﺁﻥ ﺭﺍ ﺍﺩﺍﻣﻪ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺩ. ﺩﺭ ﺩﺭﺱ ﺑﻪ ﺑﻌﺪ «ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺳﺖ ﭼﭗ» ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.

ﻳﻜﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ‌ﻫﺎﯼ ﻭﺳﻌﺖ ﺑﺨﺸﻴﺪﻥ ﺑﻪ ﺍﻣﻜﺎﻥ ﺗﺠﺮﺑﻪ ﻛﺮﺩﻥ ﻓﺮﺯﻧﺪﺗﺎﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﻛﻪ ﺩﺭ ﺑﻌﻀﯽ ﻣﺪﺍﺭﺱ ﺁﻥ ﺭﺍ «ﺑﺮﺭﺳﯽ ﻣﺤﻴﻂ» ﻣﯽ‌ﻧﺎﻣﻨﺪ. ﺩﺭ ﻫﺮ ﺷﻬﺮ ﻛﻮﭼﮏ ﻭ ﺑﺰﺭﮔﯽ ﻣﻜﺎﻥ‌ﻫﺎﯼ ﺟﺎﻟﺐ ﺑﺴﻴﺎﺭﯼ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﻳﮏ ﻛﻮﺩﮎ ﻗﺒﻞ ﺍﺯ ﺳﻨﻴﻦ ﻣﺪﺭﺳﻪ ﺯﻣﻴﻨﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺗﺠﺮﺑﻪ ﺑﺎﺷﺪ. ﻓﺮﺯﻧﺪﺍﻥ ﺷﻤﺎ ﺑﺎ ﻫﻴﺠﺎﻥ ﺯﻳﺎﺩﯼ ﺍﺯ ﻣﺤﻞ ﺁﺗﺶ‌ﻧﺸﺎﻧﯽ، ﺍﺩﺍﺭﻩ ﭘﻠﻴﺲ، ﻛﺎﺭﺧﺎﻧﻪ ﺷﻴﺮ ﭘﺎﺳﺘﻮﺭﻳﺰﻩ، ﭼﺎﭘﺨﺎﻧﻪ، ﻛﺘﺎﺑﺨﺎﻧﻪ، ﻛﺎﺭﺧﺎﻧﻪ ﺭﻳﺨﺘﻪ‌ﮔﺮﯼ، ﻛﺎﺭﺧﺎﻧﻪ ﺟﻮﺵ‌ﻛﺎﺭﯼ، ﺑﺎﻧﮏ ﻭ ﻛﺎﺭﺧﺎﻧﻪ ﻣﺎﻛﺎﺭﻭﻧﯽ‌ﺳﺎﺯﯼ ﺑﺎﺯﺩﻳﺪ ﺧﻮﺍﻫﻨﺪ ﻛﺮﺩ.

ﻛﺎﺭﺧﺎﻧﻪ ﺭﻳﺨﺘﻪ‌ﮔﺮﯼ ﺑﺮﺍﯼ ﺁﻥ‌ﻫﺎ ﺍﺯ ﻫﻤﻪ ﺟﺎﻟﺐ‌ﺗﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻭﻗﺘﯽ ﺑﺒﻴﻨﻨﺪ ﺗﻮﺩﻩ ﺁﻫﻦ ﻗﺮﻣﺰ ﺷﺪﻩ ﺭﺍ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺗﻐﻴﻴﺮ ﺷﻜﻞ ﺩﺍﺩ، ﺧﻴﻠﯽ ﺧﻮﺏ ﻣﯽ‌ﻓﻬﻤﻨﺪ ﻛﻪ ﺍﺟﺴﺎﻡ ﺁﻫﻨﯽ ﺭﺍ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺳﺎﺯﻧﺪ ﻭ ﺩﻟﺸﺎﻥ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﺪﺍﻧﻨﺪ ﻛﻪ ﺍﺟﺴﺎﻡ ﺩﻳﮕﺮ ﺭﺍ ﭼﮕﻮﻧﻪ ﺩﺭﺳﺖ ﻣﯽ‌ﻛﻨﻨﺪ.

ﺍﮔﺮ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺑﻪ ﭼﻨﻴﻦ ﻣﺤﻞ‌ﻫﺎﻳﯽ ﺑﺒﺮﻳﺪ ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ ﻛﻪ ﺑﺎﺯﺩﻳﺪ ﺍﺯ ﺍﻳﻦ ﻣﻜﺎﻥ‌ﻫﺎ ﺑﺎ ﻛﻮﺩﻛﺎﻥ ﭼﻘﺪﺭ ﺟﺎﻟﺐ ﺍﺳﺖ. ﺩﺭ ﻣﻮﻗﻌﻴﺖ‌ ﻋﺎﺩﯼ، ﺍﻓﺮﺍﺩ ﺑﺎﻟﻎ ﻭﻗﺖ ﻭ ﺍﻣﻜﺎﻥ ﭼﻨﻴﻦ ﻛﺎﺭﻫﺎﻳﯽ ﺭﺍ ﻧﺪﺍﺭﻧﺪ. ﻣﺎ ﻛﻤﺘﺮ ﺟﺮﺍﺕ ﻣﯽ‌ﻛﻨﻴﻢ ﺑﻪ ﻣﺎﻣﻮﺭ ﺁﺗﺶ‌ﻧﺸﺎﻧﯽ ﺑﮕﻮﻳﻴﻢ: «ﻣﺎﺷﻴﻦ ﺁﺗﺶ‌ﻧﺸﺎﻧﻴﺘﻮﻥ ﺭﻭ ﺑﻪ ﻣﺎ ﻧﺸﺎﻥ ﺑﺪﻫﻴﺪ.» ﻭﻟﯽ ﺍﮔﺮ ﺑﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﻪ ﺍﻳﻦ ﮔﻮﻧﻪ ﻣﺤﻞ‌ﻫﺎ ﺑﺮﻭﻳﺪ، ﺍﻭ ﺩﺭ ﻭﺍﻗﻊ ﺑﻠﻴﻂ ﻭﺭﻭﺩﯼ ﺷﻤﺎ ﺑﻪ ﺁﻧﺠﺎﺳﺖ. ﻣﻦ ﺩﻳﺪﻩ‌ﺍﻡ ﻛﻪ ﺣﺘﯽ ﺗﻠﺦ‌ﺗﺮﻳﻦ ﺁﺩﻡ‌ﻫﺎ ﻧﻴﺰ ﺍﺯ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﻛﺎﺭﺧﺎﻧﻪ ﻭ ﻳﺎ ﮐﺎﺭﮔﺎﻫﺸﺎﻥ ﺑﻪ ﻳﮏ ﻛﻮﺩﮎ ﺧﻮﺷﺸﺎﻥ ﻣﯽ‌ﺁﻳﺪ. ﺑﻌﺪ ﺍﺯ ﭼﻨﻴﻦ ﺑﺎﺯﺩﻳﺪﻫﺎﻳﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺗﺮﻏﻴﺐ ﺑﻪ ﻛﺸﻴﺪﻥ ﻧﻘﺎﺷﯽ‌ﻫﺎﻳﯽ ﺩﺭ ﻣﻮﺭﺩ ﺁﻥ ﭼﻪ ﺩﻳﺪﻩ ﻭ ﺗﺎﺛﻴﺮﺍﺗﯽ ﻛﻪ ﺑﺮ ﺭﻭﯼ ﺍﻭ ﻛﺮﺩﻩ ﺑﻜﻨﻴﺪ. ﭼﺮﺍ ﺑﺎ ﺍﻳﻦ ﻧﻘﺎﺷﯽ‌ﻫﺎ «ﻳﮏ ﻛﺘﺎﺏ ﻛﻮﭼﻮﻟﻮ» ﺩﺭﺳﺖ ﻧﻜﻨﻴﺪ؟

ﻭﻗﺘﯽ ﺩﺭ ﻓﻜﺮ ﺑﺎﻻ ﺑﺮﺩﻥ ﻣﻌﻠﻮﻣﺎﺕ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﻫﺴﺘﻴﺪ ﻻﺯﻡ ﻧﻴﺴﺖ ﻛﻪ ﺭﺍﻩ ﺯﻳﺎﺩ ﺩﻭﺭﯼ ﺑﺮﻭﻳﺪ؛  ﺍﺯ ﭼﻴﺰﻫﺎﯼ ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﻏﺎﻓﻞ ﻧﺸﻮﻳﺪ. ﻣﺜﻼ ﻣﻮﺯﻩ‌ﻫﺎ ﻣﻨﺒﻊ ﻋﻈﻴﻤﯽ ﺍﺯ ﺍﻃﻼﻋﺎﺕ ﻭ ﺗﺠﺮﺑﻴﺎﺕ ﻫﺴﺘﻨﺪ. ﻭﻟﯽ ﺑﻪ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻛﻪ ﻛﻮﺩﻛﯽ ﺩﺭ ﺳﻨﻴﻦ ﻗﺒﻞ ﺍﺯ ﻣﺪﺭﺳﻪ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺗﻮﺟﻬﺶ ﺭﺍ ﺑﺮﺍﯼ ﻣﺪﺗﯽ ﻃﻮﻻﻧﯽ ﻣﺘﻮﺟﻪ ﻳﮏ ﭼﻴﺰ ﻛﻨﺪ، ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺎﺯﺩﻳﺪ ﺍﺯ ﺍﻳﻦ ﮔﻮﻧﻪ ﺍﻣﺎﻛﻦ ﺑﺎﻳﺪ ﺗﺎ ﺣﺪ ﻣﻤﻜﻦ ﻛﻮﺗﺎﻩ ﺑﺎﺷﺪ. ﻣﺴﺎﻓﺮﺕ ﻧﻴﺰ ﻣﻨﺒﻊ ﺑﺴﻴﺎﺭ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﻭﺳﻌﺖ ﺑﺨﺸﻴﺪﻥ ﺗﺠﺮﺑﻴﺎﺕ ﻛﻮﺩﮎ ﺍﺳﺖ، ﺧﺼﻮﺻﺎ ﺍﮔﺮ ﺍﻣﻜﺎﻥ ﻣﺴﺎﻓﺮﺕ ﺑﻪ ﺧﺎﺭﺝ ﺑﺮﺍﻳﺘﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻛﺘﺎﺏ، ﺳﯽ‌ﺩﯼ، ﻓﻴﻠﻢ‌ﻫﺎ ﻭ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺗﻠﻮﻳﺰﻳﻮﻧﯽ ﻧﻴﺰ ﻛﻤﮏ ﺷﺎﻳﺎﻧﯽ ﺑﻪ ﮔﺴﺘﺮﺵ ﺍﻓﻖ ﺩﻳﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﯽ‌ﻛﻨﺪ. ﻫﻤﻴﺸﻪ ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻫﺮ ﭼﻘﺪﺭ ﻓﺮﺯﻧﺪﻣﺎﻥ ﺑﻴﺸﺘﺮ ﺑﺒﻴﻨﺪ ﻭ ﺑﺸﻨﻮﺩ، ﻫﻤﺎﻥ‌ﻗﺪﺭ ﺑﻴﺸﺘﺮ ﻣﯽ‌ﺑﻴﻨﺪ ﻭ ﻣﯽ‌ﺷﻨﻮﺩ.


ﺍﻳﻨﻜﻪ ﺑﮕﺬﺍﺭﻳﻢ ﺍﺯ ﺭﻭﯼ ﺗﺠﺮﺑﻴﺎﺕ ﻋﻴﻨﯽ، ﻓﻜﺮ ﻣﺠﺮﺩ ﻧﺰﺩ ﻛﻮﺩﮎ ﺭﺷﺪ ﻳﺎﺑﺪ، ﺍﻫﻤﻴﺖ ﺑﻪ‌ﺳﺰﺍﻳﯽ ﺩﺍﺭﺩ. ﻛﻮﺩﻛﺎﻥ ﺧﺮﺩﺳﺎﻝ ﺑﻪ ﻫﻴﭻ‌ﻭﺟﻪ ﺧﺎﺭﺝ ﺍﺯ ﺩﺍﺩﻩ‌ﻫﺎﯼ ﻋﻴﻨﯽ ﻭ ﺑﻪ‌ ﺻﻮﺭﺕ ﻣﺠﺮﺩ ﻗﺎﺩﺭ ﺑﻪ ﻓﻜﺮ ﻛﺮﺩﻥ ﻧﻴﺴﺘﻨﺪ. ﻭﻟﯽ ﺍﮔﺮ ﻭﺳﺎﻳﻠﯽ ﻛﻪ ﺑﺘﻮﺍﻧﻨﺪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﺒﻴﻨﻨﺪ ﻭ ﻟﻤﺲ ﻛﻨﻨﺪ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺷﺎﻥ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ ﺑﻪ‌ ﺧﻮﺑﯽ ﻗﺎﺩﺭ ﺑﻪ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﻛﺎﺭ (ﻓﻜﺮ ﻛﺮﺩﻥ ﺑﻪ ﺻﻮﺭﺕ ﻣﺠﺮﺩ) ﻣﯽ‌­ﺷﻮﻧﺪ. ﺁﮔﺎﻫﯽ ﺑﻪ ﺍﻫﻤﻴﺖ «ﺁﻣﻮﺯﺵ ﺍﺯ ﻃﺮﻳﻖ ﺣﺲ ﻻﻣﺴﻪ» ﺍﻳﺪﻩ ﺑﺴﻴﺎﺭ ﻋﺎﻟﯽ ﻣﺎﺭﻳﺎ ﻣﻮﻧﺖِﺳﻮﺭﯼ ﺍﺳﺖ ﻛﻪ ﺩﺭ ﻭﺍﻗﻊ ﭘﺎﻳﻪ «ﺟﻨﺒﺶ ﻣﻮﻧﺖِﺳﻮﺭﯼ» ﺍﺳﺖ ﻛﻪ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ.

ﻣﻮﻧﺖِﺳﻮﺭﯼ ﻣﯽ‌ﮔﻮﻳﺪ ﭼﻴﺰﻫﺎﻳﯽ ﻋﻴﻨﯽ ﻛﻪ ﺑﺮﺍﯼ ﺍﺷﺘﻐﺎﻝ ﺣﻮﺍﺱ ﺍﺳﺖ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻛﻮﺩﻛﺎﻥ ﺑﮕﺬﺍﺭﻳﺪ. ﺍﻭ ﺍﻳﺪﻩ‌ﻫﺎﯼ ﻣﺠﺮﺩﯼ ﺭﺍ ﻛﻪ ﺑﻪ‌ ﺻﻮﺭﺕ ﻋﻴﻨﯽ ﺩﺭﺁﻣﺪﻩ‌ﺍﻧﺪ ﻭ ﻛﻮﺩﻛﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺣﻴﻦ ﻛﺎﺭﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻟﻤﺲ ﻛﺮﺩﻩ ﻭ ﺩﺳﺘﻜﺎﺭﯼ ﻛﻨﻨﺪ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺷﺎﻥ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﮐﻪ ﻳﮏ ﺳﺮﯼ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺁﻣﻮﺯﻧﺪﻩ ﻃﺮﺍﺣﯽ ﻛﺮﺩﻩ ﻛﻪ ﺑﻪ ﻛﻮﺩﻛﺎﻥ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﺪ ﻓﻘﻂ ﺑﺎ ﺩﺳﺘﻜﺎﺭﯼ ﻛﺮﺩﻥ ﺁﻥ‌ﻫﺎ ﻓﻴﺰﻳﻚ، ﺭﻳﺎﺿﯽ، ﺧﻮﺍﻧﺪﻥ ﻭ ﻧﻮﺷﺘﻦ ﺭﺍ ﻳﺎﺩ ﺑﮕﻴﺮﻧﺪ.


ﺭﺍﻩ ﻣﻮﻧﺖِﺳﻮﺭﯼ ﺭﺍ ﻣﺘﺨﺼﺼﻴﻦ ﺩﻳﮕﺮ ﺗﻌﻠﻴﻢ ﻭ ﺗﺮﺑﻴﺖ ﻛﻮﺩﻛﺎﻥ ﺩﻧﺒﺎﻝ ﻛﺮﺩﻩ‌ﺍﻧﺪ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﺩﺭ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﻓﺮﻭﺷﯽ‌ﻫﺎ، ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺁﻣﻮﺯﻧﺪﻩ‌ﺍﯼ ﺭﺍ ﻛﻪ ﺑﺴﻴﺎﺭ ﺧﻮﺏ ﺩﺭﺳﺖ ﺷﺪﻩ‌ﺍﻧﺪ ﻭ ﺑﺎﺯﯼ ﻛﺮﺩﻥ ﺑﺎ ﺁﻥ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻓﻜﺮ ﻣﺠﺮﺩ ﺭﺍ ﻧﺰﺩ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺷﺪ ﺩﻫﺪ، ﺑﺨﺮﻳﺪ.

ﺍﺳﺎﺱ ﺍﻳﻦ‌ﮔﻮﻧﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎ ﺑﺮ ﺍﻳﻦ ﺍﺳﺖ ﻛﻪ ﻛﻮﺩﮎ ﺭﺍ ﺁﺯﺍﺩ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﻛﻪ ﺧﻮﺩ ﻛﺸﻒ ﻛﻨﺪ ﻛﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ ﭼﻴﺴﺖ. ﭼﻴﺰﯼ ﻛﻪ ﺍﻭ ﺧﻮﺩ ﻛﺸﻒ ﻣﯽ‌ﻛﻨﺪ ﺧﻴﻠﯽ ﺑﻬﺘﺮ ﺍﺯ ﻫﺮ ﭼﻴﺰﯼ ﻛﻪ ﺑﻪ ﺍﻭ ﺗﻮﺿﻴﺢ ﻣﯽ‌ﺩﻫﻨﺪ ﺩﺭ ﻣﻐﺰﺵ ﺟﺎﯼ ﻣﯽ‌ﮔﻴﺮﺩ.

ﻣﺜﺎﻟﯽ ﻣﯽ‌ﺯﻧﻴﻢ: ﺷﻤﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺁﺯﺍﺩ ﻣﯽ‌ﮔﺬﺍﺭﻳﺪ ﺗﺎ ﺑﺎ ﺁﺑﺮﻧﮓ ﻧﻘﺎﺷﯽ ﻛﻨﺪ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻭ ﺗﻮﺿﻴﺢ ﺩﻫﻴﺪ ﻛﻪ ﺑﺎ ﻣﺨﻠﻮﻁ ﻛﺮﺩﻥ ﺩﻭ ﺭﻧﮓ ﺯﺭﺩ ﻭ ﺁﺑﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺭﻧﮓ ﺳﺒﺰ ﺭﺍ ﺑﻪ‌ﺩﺳﺖ ﺁﻭﺭﺩ. ﻭﻟﯽ ﺑﺎ ﺍﻳﻦ ﻛﺎﺭ ﺭﺍ ﺍﻭ ﺭﺍ ﺍﺯ ﻟﺬﺕ ﻛﺸﻒ ﺍﻳﻦ ﻣﺴﺎﻟﻪ ﻣﺤﺮﻭﻡ ﻛﺮﺩﻩ‌ﺍﻳﺪ. ﭘﺲ ﭼﺮﺍ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺩﻳﮕﺮ ﻋﻤﻞ ﻧﻜﻨﻴﻢ؟ ﺭﻧﮓ ﺯﺭﺩ ﻭ ﺁﺑﯽ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺵ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﻣﺜﻼ ﺑﮕﻮﻳﻴﺪ: «ﭼﻄﻮﺭﻩ ﺭﻧﮓ ﺯﺭﺩ ﻭ ﺁﺑﯽ ﺭﻭ ﺑﺎ ﻫﻢ ﻣﺨﻠﻮﻁ ﻛﻨﯽ ﺑﺒﻴﻨﯽ ﭼﻪ ﭘﻴﺶ ﻣﯽ‌ﺁﺩ.» ﺍﻭ ﺍﻣﺘﺤﺎﻥ ﺧﻮﺍﻫﺪ ﻛﺮﺩ ﻭ ﺑﺎ ﺧﻮﺷﺤﺎﻟﯽ ﺑﻪ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﮔﻔﺖ: «ﻧﮕﺎﻩ ﻛﻦ، ﺳﺒﺰ ﺷﺪ!»

ﺑﺮﺍﯼ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎﻳﯽ ﻛﻪ ﻋﺎﺩﺕ ﻛﺮﺩﻩ‌ﺍﻧﺪ ﻫﻤﻴﺸﻪ ﺑﮕﻮﻳﻨﺪ ﭼﻪ ﻛﺎﺭ ﻛﻦ ﻭ ﭼﻪ ﻛﺎﺭ ﻧﻜﻦ، ﺑﻪ‌ﻛﺎﺭ ﮔﺮﻓﺘﻦ ﺭﻭﺵ ﺍﻛﺘﺸﺎﻑ ﺷﺨﺼﯽ ﺑﻪ ﻋﻨﻮﺍﻥ ﺭﻭﺵ ﺗﺮﺑﻴﺘﯽ ﻛﻤﯽ ﻣﺸﻜﻞ ﺍﺳﺖ. ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻛﻤﮏ ﻛﻨﻴﺪ ﺗﺎ ﻣﻔﺎﻫﻴﻢ ﻛﻠﻴﺪﯼ‌ﺍﯼ ﺭﺍ ﻛﻪ ﺑﺮﺍﯼ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﻭ ﺩﺭﮎ ﺩﻧﻴﺎﻳﺶ ﻧﻴﺎﺯ ﺩﺍﺭﺩ ﻳﺎﺩ ﺑﮕﻴﺮﺩ. ﻛﻮﺩﻛﺎﻥ ﺩﺍﺋﻤﺎ ﺗﺤﺖ ﺗﻬﺎﺟﻢ ﺍﻃﻼﻋﺎﺗﯽ ﻫﺴﺘﻨﺪ ﻛﻪ ﺍﻃﺮﺍﻓﻴﺎﻥ ﺑﻪ ﺁﻥ‌ﻫﺎ ﻣﯽ‌ﺩﻫﻨﺪ. ﺁﻥ‌ﻫﺎ ﺗﻤﺎﻡ ﻣﺪﺕ ﺩﺭ ﭘﯽ ﺭﺑﻂ ﺩﺍﺩﻥ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﭘﺮﺍﮐﻨﺪﻩ ﻫﺴﺘﻨﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻨﺪ ﻣﻌﻨﺎﯼ ﺟﻬﺎﻥ ﺩﻭﺭ ﻭﺑﺮﺷﺎﻥ ﺩﺭﮎ ﮐﻨﻨﺪ. ﻛﻮﺩﻛﺎﻥ ﺳﻌﯽ ﻣﯽ‌ﻛﻨﻨﺪ ﺭﻭﺍﺑﻂ ﻋﻠﺖ ﻭ ﻣﻌﻠﻮﻟﯽ ﺭﺍ ﻛﺸﻒ ﻛﺮﺩﻩ ﻭ ﺍﻃﻼﻋﺎﺗﯽ ﺭﺍ ﻛﻪ ﺣﻮﺍﺳﺸﺎﻥ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺷﺎﻥ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﻃﺒﻘﻪ‌ﺑﻨﺪﯼ ﻛﻨﻨﺪ.

ﻣﻨﻈﻮﺭ ﺍﺯ «ﻣﻔﺎﻫﻴﻢ ﻛﻠﻴﺪﯼ ۱۲۳۴۵»، ﻣﻔﺎﻫﻴﻤﯽ ﺍﺳﺖ ﻛﻪ ﻛﻮﺩﮎ ﺑﺘﻮﺍﻧﺪ ﺑﺎ ﺁﻥ‌ﻫﺎ ﭘﺪﻳﺪﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺭﺍ ﺗﺤﻠﻴﻞ ﻛﺮﺩﻩ ﻭ ﺑﺮﺍﯼ ﻣﺤﻴﻂ ﺧﻮﺩ، ﺑﺮﺍﺳﺎﺱ ﺁﻥ‌ﻫﺎ، ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﻛﻨﺪ. ﺍﺯ ﺟﻤﻠﻪ ﻣﻔﺎﻫﻴﻢ ﻛﻠﻴﺪﯼ ﺍﯼ ﻛﻪ ﻳﮏ ﻛﻮﺩﮎ ﺩﺭ ﺳﻨﻴﻦ ﻗﺒﻞ ﺍﺯ ﻣﺪﺭﺳﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻓﺮﺍﮔﻴﺮﺩ ﻣﻮﺍﺭﺩ ﺯﻳﺮ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻧﺎﻡ ﺑﺒﺮﻳﻢ:

  • ﻧﻴﺮﻭﯼ ﺟﺎﺫﺑﻪ ﻧﻴﺮﻭﻳﯽ ﺍﺳﺖ ﻏﻴﺮ ﻗﺎﺑﻞ ﺭﻭﻳﺖ ﻛﻪ ﺍﺟﺴﺎﻡ ﺭﺍ ﺑﻪ ﻃﺮﻑ ﺯﻣﻴﻦ ﻣﯽ‌ﻛﺸﺪ.
  • ﺗﻤﺎﻡ ﺣﻴﻮﺍﻧﺎﺕ ﻳﮏ ﻭﺳﻴﻠﻪ ﺩﻓﺎﻋﯽ ﻋﻠﻴﻪ ﺩﺷﻤﻨﺎﻧﺸﺎﻥ ﺩﺍﺭﻧﺪ.
  • ﻫﻨﮕﺎﻣﯽ ﻛﻪ ﺩﻭ ﺟﺴﻢ ﺭﺍ ﺑﻪ ﻫﻢ ﺑﻤﺎﻟﻴﻢ، ﺍﻳﻦ ﻣﺎﻟﺶ ﺑﺎﻋﺚ ﮔﺮﻡ ﺷﺪﻥ ﺁﻥ‌ﻫﺎ ﻣﯽ‌ﺷﻮﺩ.


ﺍﮔﺮ ﻛﻮﺩﻛﯽ ﻳﻜﯽ ﺍﺯ ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﻛﻠﻴﺪﯼ ﺭﺍ ﻳﺎﺩ ﺑﮕﻴﺮﺩ، ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺍﺯ ﭘﺪﻳﺪﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺭﺍ ﻛﻪ ﺩﻭﺭ ﻭ ﺑﺮﺵ ﺍﺗﻔﺎﻕ ﻣﯽ‌ﺍﻓﺘﺪ ﺩﺭﮎ ﻣﯽ‌ﻛﻨﺪ. ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺧﻮﺍﻧﺪﻥ ﻛﺘﺎﺏ‌ﻫﺎﻳﯽ ﻛﻪ ﺑﺮﺍﯼ ﺍﻭ ﻗﺎﺑﻞ ﺩﺭﮎ ﺑﺎﺷﻨﺪ ﺑﻪ ﺍﻭ ﺑﻴﺎﻣﻮﺯﻳﺪ.

ﺑﺎ ﺗﻮﺿﻴﺢ ﻭﻗﺎﻳﻊ ﺭﻭﺯﻣﺮﻩ ﺯﻧﺪﮔﯽ ﻧﻴﺰ ﻫﻤﻴﻦ ﻛﺎﺭ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻜﻨﻴﺪ. ﻣﺜﻼ ﻣﺴﺎﻟﻪ ﻧﻴﺮﻭﯼ ﺟﺎﺫﺑﻪ ﺯﻣﻴﻦ ﺭﺍ ﺧﻴﻠﯽ ﺭﺍﺣﺖ ﺑﺎ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺍﻳﻨﻜﻪ ﻫﺮﺟﺴﻤﯽ ﺭﺍ ﮐﻪ ﺭﻫﺎ ﻛﻨﻴﺪ ﺑﻪ ﺯﻣﻴﻦ ﻣﯽ‌ﺍﻓﺘﺪ ﻭ ﺑﻪ ﻫﻮﺍ ﻧﻤﯽ‌ﺭﻭﺩ ﻣﯽ‌ﺗﻮﺍﻥ ﺗﻮﺿﻴﺢ ﺩﺍﺩ. ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺗﻮﺿﻴﺢ ﺑﺪﻫﻴﻢ ﻛﻪ ﺩﺭ ﻓﻀﺎ ﻧﻴﺮﻭﯼ ﺟﺎﺫﺑﻪ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ ﻭ ﺍﻧﺴﺎﻥ ﺩﺭ ﺁﻥ ﻏﻮﻃﻪ‌ﻭﺭ ﻣﯽ‌ﺷﻮﺩ ﻣﮕﺮ ﺍﻳﻨﻜﻪ ﺑﻪ ﻳﮏ ﺳﻴﺎﺭﻩ‌ﺍﯼ ﺑﺮﺳﺪ ﻛﻪ ﺩﺭ ﺁﻥ ﻧﻴﺮﻭﯼ ﺟﺎﺫﺑﻪ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﻳﺎﺩ ﺑﮕﻴﺮﻳﺪ ﻛﻪ ﺩﺭ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺁﻣﻮﺯﻧﺪﻩ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺷﺮﻛﺖ ﻛﻨﻴﺪ. ﺗﻤﺎﻡ ﻛﻮﺩﻛﺎﻥ ﺩﻭﺳﺖ ﺩﺍﺭﻧﺪ ﺑﺎﺯﯼ ﻛﻨﻨﺪ. ﺍﺳﺒﺎﺏ ﺑﺎﺯﯼ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﻫﺴﺖ ﻛﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺧﺮﻳﺪ ﻭﻟﯽ ﺑﺮﺍﯼ ﻛﻮﺩﻛﺎﻥ ﺩﺭ ﺳﻨﻴﻦ ﭘﻴﺶ ﺩﺑﺴﺘﺎﻧﯽ ﻻﺯﻡ ﻧﻴﺴﺖ ﺭﺍﻩ ﺩﻭﺭﯼ ﺑﺮﻭﻳﺪ، ﺧﻮﺩﺗﺎﻥ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻳﻦ‌ﮔﻮﻧﻪ ﺍﺳﺒﺎﺏ‌ﺑﺎﺯﯼ‌ﻫﺎ ﺭﺍ ﺩﺭﺳﺖ ﻛﻨﻴﺪ.
