ﻣﻌﺮﻓﯽ ﺩﻭﺭﻩ ﻭ ﺍﻫﺪﺍﻑ ﺁﻥ
ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻣﺘﻠﺐ (MATLAB) ﻳﮑﯽ ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﺍﺑﺰﺍﺭﻫﺎﯼ ﻣﺤﺎﺳﺒﺎﺗﯽ ﻣﻬﻨﺪﺳﺎﻥ ﻭ ﭘﮋﻭﻫﺸﮕﺮﺍﻥ ﺭﺷﺘﻪ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺳﺖ ﻭ ﺍﻣﺮﻭﺯ ﺩﺭ ﻫﻤﻪ ﺩﺍﻧﺸﮕﺎﻩ‌ﻫﺎ ﻭ ﺑﻪ ﺗﻘﺮﻳﺐ ﺩﺭ ﻫﻤﻪ ﺭﺷﺘﻪ‌ﻫﺎ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﺩ. ﺑﺮﺧﻼﻑ ﭼﻴﺰﯼ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺍﺑﺘﺪﺍ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﺪ، ﻳﺎﺩﮔﻴﺮﯼ ﻭ ﺑﮑﺎﺭ ﺑﺴﺘﻦ ﻣﺘﻠﺐ ﮐﺎﺭ ﺳﺨﺘﯽ ﻧﻴﺴﺖ. ﺩﻭﺭﻩ ﻣﻘﺪﻣﺎﺗﯽ ﻣﺘﻠﺐ (MATLAB) ﺷﻤﺎ ﺭﺍ ﺑﺎ ﻣﻔﺎﻫﻴﻢ ﺍﻭﻟﻴﻪ ﺍﻳﻦ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺁﺷﻨﺎ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺩﺭ ﭘﺎﻳﺎﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﺷﻤﺎ ﻗﺎﺩﺭ ﺧﻮﺍﻫﻴﺪ ﺑﻮﺩ:

  • ﺍﺯ ﺍﻳﻦ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺑﺮﺍﯼ ﺣﻞ ﻣﺴﺎﻳﻞ ﺭﻳﺎﺿﯽ ﻳﺎ ﺭﺳﻢ ﻧﻤﻮﺩﺍﺭ ﺗﻮﺍﺑﻊ ﻭ ﺟﺪﺍﻭﻝ ﺩﺍﺩﻩ‌ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ.
  • ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺳﺎﺩﻩ‌ﺍﯼ ﺑﺮﺍﯼ ﺗﺤﻠﻴﻞ ﺩﺍﺩﻩ‌ﻫﺎﯼ ﺑﺪﺳﺖ ﺁﻣﺪﻩ ﺍﺯ ﻳﮏ ﺁﻣﺎﺭ ﮔﻴﺮﯼ ﻳﺎ ﺁﺯﻣﺎﻳﺶ ﺑﻨﻮﻳﺴﻴﺪ.
  • ﻭ ﺑﻪ ﺭﺍﺣﺘﯽ ﺭﺍﻩ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﺣﻞ ﻣﺴﺎﻳﻞ ﺟﺪﻳﺪ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻨﺎﺑﻊ ﮐﻤﮑﯽ ﺍﻳﻦ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﭘﻴﺪﺍ ﮐﻨﻴﺪ.

ﺳﻄﺢ ﺍﻃﻼﻋﺎﺕ ﺍﺭﺍﺋﻪ ﺷﺪﻩ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺭ ﺣﺪﯼ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﻫﻤﮕﺎﻥ ﻗﺎﺑﻞ ﺍﺳﺘﻔﺎﺩﻩ ﺑﺎﺷﺪ ﻭﻟﯽ ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺍﮔﺮ ﺩﺍﻧﺶ ﻭ ﺗﺠﺮﺑﻪ‌ﺍﯼ ﻫﺮ ﭼﻨﺪ ﺍﻧﺪﮎ، ﻭ ﺣﺘﯽ ﺩﺭ ﺣﺪ ﺁﺷﻨﺎﻳﯽ، ﺑﺎ ﺯﺑﺎﻥ‌ﻫﺎﯼ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻣﻔﺎﻫﻴﻢ ﺑﺨﺶ‌ﻫﺎﻳﯽ ﺍﺯ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﺮﺍﯼ ﺷﻤﺎ ﻣﻠﻤﻮﺱ‌ﺗﺮ ﻭ ﺩﺭﮎ ﺁﻧﻬﺎ ﺳﺎﺩﻩ ﺗﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

 

ﻣﻘﺪﻣﻪ
ﺍﻳﻦ ﻧﮑﺘﻪ ﺭﺍ ﻧﺒﺎﻳﺪ ﻓﺮﺍﻣﻮﺵ ﮐﺮﺩ ﮐﻪ ﺍﺳﺎﺱ ﺗﻔﮑﺮﯼ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﺳﺎﺧﺘﻪ ﺷﺪﻥ ﻭ ﭘﻴﺸﺮﻓﺖ ﺭﺍﻳﺎﻧﻪ‌ﻫﺎ ﺷﺪ، ﻧﻴﺎﺯ ﺑﺸﺮ ﺑﻪ ﺣﻞ ﻣﺴﺎﻳﻞ ﺭﻳﺎﺿﯽ ﭘﻴﭽﻴﺪﻩ ﻭ ﺗﺤﻠﻴﻞ ﺭﻳﺎﺿﯽ ﺩﺍﺩﻩ‌ﻫﺎﯼ ﺁﻣﺎﺭﯼ ﻭ ﻋﻠﻤﯽ ﺑﻮﺩ. ﻫﺮﭼﻪ ﺩﺍﻧﺶ ﺑﺸﺮﯼ ﺯﻳﺎﺩﺗﺮ ﺷﻮﺩ ﻧﻴﺎﺯ ﺍﻭ ﺑﻪ ﺣﻞ ﻣﺴﺎﻳﻞ ﭘﻴﭽﻴﺪﻩ ﺭﻳﺎﺿﯽ ﻧﻴﺰ ﺍﻓﺰﺍﻳﺶ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ. ﺍﻣﺮﻭﺯ ﺣﻞ ﻣﺴﺎﻳﻞ ﭘﻴﭽﻴﺪﻩ‌ﺍﯼ ﻣﺎﻧﻨﺪ ﭘﻴﺶ ﺑﻴﻨﯽ ﻭﺿﻌﻴﺖ ﺁﺏ ﻭ ﻫﻮﺍ ﺑﺎ ﻣﺘﻐﻴﺮﻫﺎ ﻭ ﭘﺎﺭﺍﻣﺘﺮﻫﺎﯼ ﮔﻮﻧﺎﮔﻮﻥ، ﺑﺪﻭﻥ ﻳﺎﺭﯼ ﺭﺍﻳﺎﻧﻪ‌ﻫﺎﯼ ﭘﻴﺸﺮﻓﺘﻪ ﻭ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﻳﯽ ﮐﻪ ﻗﺎﺩﺭ ﺑﻪ ﺣﻞ ﻣﺴﺎﻳﻞ ﭘﻴﭽﻴﺪﻩ ﺭﻳﺎﺿﯽ ﺑﺎﺷﻨﺪ، ﻏﻴﺮ ﻣﻤﮑﻦ ﺍﺳﺖ.

ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﺗﻌﺪﺍﺩ ﺯﻳﺎﺩﯼ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺑﺮﺍﯼ ﺣﻞ ﻣﺴﺎﻳﻞ ﺭﻳﺎﺿﯽ ﻳﺎ ﺷﺒﻴﻪ ﺳﺎﺯﯼ‌ﻫﺎﯼ ﭘﺪﻳﺪﻩ‌ﻫﺎﯼ ﻓﻴﺰﻳﮑﯽ ﺑﻮﺳﻴﻠﻪ ﻣﺪﻝ ﺳﺎﺯﯼ ﺭﻳﺎﺿﯽ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﮐﻪ ﻣﺘﺨﺼﺼﺎﻥ ﺑﺴﺘﻪ ﺑﻪ ﻧﻴﺎﺯ ﻭ ﺷﺮﺍﻳﻂ ﮐﺎﺭ ﺍﺯ ﺑﺮﺧﯽ ﺍﺯ ﺁﻧﻬﺎ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﻳﺎ ﺑﻄﻮﺭ ﺗﺮﮐﻴﺒﯽ ﺑﺎ ﺩﻳﮕﺮ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎ ﺑﻬﺮﻩ ﻣﯽ‌ﺑﺮﻧﺪ. ﺍﺯ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﯼ ﻣﻨﺒﻊ ﺑﺎﺯ ﺭﺍﻳﮕﺎﻥ ﺗﺎ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﺧﺎﺹ ﭼﻨﺪ ﺩﻩ ﻫﺰﺍﺭ ﺩﻻﺭﯼ.

 

ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﯼ ﻣﺤﺎﺳﺒﺎﺕ ﺭﻳﺎﺿﯽ

ﻣﺘﻠﺐ ﭼﻴﺴﺖ؟
ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻣﺘﻠﺐ، ﻳﮏ ﻣﺤﻴﻂ ﻧﺮﻡ ﺍﻓﺰﺍﺭﯼ ﻭ ﻳﮏ ﺯﺑﺎﻥ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﺩﺭ ﻳﮏ ﻣﺠﻤﻮﻋﻪ ﺑﻪ ﻫﻢ ﭘﻴﻮﺳﺘﻪ ﻭ ﻣﻨﺴﺠﻢ ﺑﺮﺍﯼ ﻣﺤﺎﺳﺒﺎﺕ ﺭﻳﺎﺿﯽ ﺍﺳﺖ. ﺍﻳﻦ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺿﻤﻦ ﺍﺭﺍﺋﻪ ﺍﻣﮑﺎﻧﺎﺕ ﻣﺨﺘﻠﻒ ﺑﺮﺍﯼ ﺣﻞ ﻣﺴﺎﻳﻞ ﺭﻳﺎﺿﯽ ﻭ ﺗﺤﻠﻴﻞ ﺩﺍﺩﻩ‌ﻫﺎ ﺍﺑﺰﺍﺭﻫﺎ ﻭ ﺭﺍﻫﮑﺎﺭﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺑﺮﺍﯼ ﻧﻤﺎﻳﺶ ﻭ ﺑﻪ ﺗﺼﻮﻳﺮ ﺩﺭﺁﻭﺭﺩﻥ ﺍﻃﻼﻋﺎﺕ ﻋﺪﺩﯼ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﻫﻤﻪ ﺍﻳﻦ ﺍﻣﮑﺎﻧﺎﺕ ﺑﻪ ﺳﺎﺩﻩ ﺗﺮﻳﻦ ﺭﺍﻩ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪﻩ ﺍﺯ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ ﺑﻄﻮﺭﻳﮑﻪ ﻳﺎﺩﮔﻴﺮﯼ ﺍﻳﻦ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻭ ﮐﺎﺭ ﮐﺮﺩﻥ ﺑﺎ ﺁﻥ ﺑﺴﻴﺎﺭ ﺳﺎﺩﻩ‌ﺗﺮ ﺍﺯ ﺁﻧﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻧﮕﺎﻩ ﺍﻭﻝ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ.

ﺯﺑﺎﻥ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﻣﺘﻠﺐ ﻧﻴﺰ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﺑﺮﺍﯼ ﺣﻞ ﻣﺴﺎﻳﻞ ﺭﻳﺎﺿﯽ ﺭﺍ ﺑﻄﻮﺭ ﺑﺎﻭﺭ ﻧﮑﺮﺩﻧﯽ ﺳﺎﺩﻩ ﮐﺮﺩﻩ ﺍﺳﺖ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺟﻤﻊ ﺩﻭ ﻣﺎﺗﺮﻳﺲ 10 ﺩﺭ 10 ﺩﺭ ﺯﺑﺎﻥ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﻓﺮﺗﺮﻥ ﺣﺪﺍﻗﻞ ﺑﻪ ﺷﺶ ﺳﻄﺮ ﺩﺳﺘﻮﺭ ﻧﻴﺎﺯ ﺩﺍﺭﺩ:

ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻣﺘﻠﺐ ﺑﺮﺍﯼ ﻫﻤﻪ ﺳﻴﺴﺘﻢ‌ﻫﺎﯼ ﻋﺎﻣﻞ (ﺑﻪ ﺗﻘﺮﻳﺐ) ﺩﺭ ﺩﺳﺘﺮﺱ ﺍﺳﺖ ﻭ ﺍﻣﺮﻭﺯﻩ ﺣﺘﯽ ﻧﻤﻮﻧﻪ ﺳﺎﺩﻩ ﺷﺪﻩ‌ﺍﯼ ﺍﺯ ﺁﻥ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺮ ﺭﻭﯼ ﮔﻮﺷﯽ‌ﻫﺎﯼ ﻣﻮﺑﺎﻳﻞ ﻫﻮﺷﻤﻨﺪ ﺑﺎ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺁﻧﺪﺭﻭﺋﻴﺪ ﻳﺎ ﺁﯼ ﺍﻭ ﺍﺱ ﺍﭘﻞ ﻧﺼﺐ ﮐﺮﺩ.

 

ﺑﺎ ﻣﺘﻠﺐ ﭼﻪ ﮐﺎﺭﻫﺎﻳﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﮐﺮﺩ ﻭ ﭼﻪ ﮐﺎﺭﻫﺎﻳﯽ ﺍﺯ ﻣﺘﻠﺐ ﺑﺮ ﻧﻤﯽ‌ﺁﻳﺪ؟
ﻋﻼﻭﻩ ﺑﺮ ﻧﮑﺎﺗﯽ ﮐﻪ ﺩﺭ ﺯﻣﺎﻥ ﻣﻌﺮﻓﯽ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻣﻄﺮﺡ ﺷﺪ، ﻣﺘﻠﺐ ﻧﺮﻡ ﺍﻓﺰﺍﺭﯼ ﺧﻮﺏ ﺑﺮﺍﯼ ﻃﺮﺍﺣﯽ ﻭ ﻧﻤﻮﻧﻪ ﺳﺎﺯﯼ ﺳﻴﺴﺘﻢ‌ﻫﺎﯼ ﻣﺤﺎﺳﺒﺎﺗﯽ ﻭ ﮐﻨﺘﺮﻝ ﺍﺳﺖ. ﭼﻮﻥ:

ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻧﻤﻮﻧﻪ‌ﻫﺎﯼ ﺍﺟﺮﺍ ﺷﺪﻩ ﺑﺎ ﺁﻥ ﺑﺮﺍﯼ ﭘﺮﻭﮊﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﻪ ﺩﻟﻴﻞ ﮔﺴﺘﺮﺩﮔﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺩﺭ ﺩﺍﻧﺸﮕﺎﻩ‌ﻫﺎ ﻭ ﻣﺮﺍﮐﺰ ﺗﺤﻘﻴﻘﺎﺗﯽ ﻭ ﺻﻨﻌﺖ ﺑﻪ ﺳﺎﺩﮔﯽ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﺍﺳﺖ.

ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺑﺰﺭﮒ ﺍﺯ ﮐﺘﺎﺑﺨﺎﻧﻪ‌ﻫﺎ ﻭ ﺍﺑﺰﺍﺭﻫﺎ ﺑﺮﺍﯼ ﺍﻫﺪﺍﻑ ﻣﺨﺘﻠﻔﯽ ﻣﺎﻧﻨﺪ ﺗﺤﻠﻴﻞ ﺗﺼﻮﻳﺮ ﻳﺎ ﺳﻴﮕﻨﺎﻝ ﺭﺍ ﺑﻪ ﻫﻤﺮﺍﻩ ﺩﺍﺭﺩ.

ﺍﻣﮑﺎﻥ ﻧﻮﺷﺘﻦ ﺍﺑﺰﺍﺭﻫﺎ ﻭ ﮐﺘﺎﺑﺨﺎﻧﻪ‌ﻫﺎﻳﯽ ﺑﻪ ﺯﺑﺎﻥ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﻣﺎﻧﻨﺪ C ﻭ FORTRAN ﻭ ﻧﺼﺐ ﻭ ﻓﺮﺍﺧﻮﺍﻧﯽ ﺁﻧﻬﺎ ﺩﺭ ﺑﺮﻧﺎﻣﻪ ﺍﺻﻠﯽ ﺑﻪ ﺳﺎﺩﮔﯽ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﺍﺳﺖ.

ﻗﺎﺑﻠﻴﺖ ﺧﻮﺑﯽ ﺩﺭ ﺑﻪ ﻧﻤﺎﻳﺶ ﺩﺭﺁﻭﺭﺩﻥ ﺩﺍﺩﻩ‌ﻫﺎ ﺑﻪ ﺻﻮﺭﺕ ﻧﻤﻮﺩﺍﺭﻫﺎﯼ ﺩﻭ ﺑﻌﺪﯼ ﻳﺎ ﺳﻪ ﺑﻌﺪﯼ ﺛﺎﺑﺖ ﻳﺎ ﻣﺘﺤﺮﮎ (ﺍﻧﻴﻤﻴﺸﻦ) ﺩﺍﺭﺩ.

ﺑﺮﺧﻼﻑ ﺯﺑﺎﻥ‌ﻫﺎﯼ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﻣﺘﺪﺍﻭﻟﯽ ﻣﺎﻧﻨﺪ ﺟﺎﻭﺍ ﻳﺎ ﺳﯽ، ﺯﺑﺎﻥ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺴﯽ ﻣﺘﻠﺐ ﺯﺑﺎﻧﯽ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺳﺎﺩﮔﯽ ﻭ ﻣﺴﺘﻘﻴﻢ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺁﻥ ﺭﺍ ﺑﻪ ﺯﺑﺎﻥ ﻣﺎﺷﻴﻦ ﺗﺮﺟﻤﻪ ﮐﺮﺩ. ﺑﻪ ﺑﻴﺎﻥ ﺩﻳﮕﺮ ﺯﺑﺎﻧﯽ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﺍﺟﺮﺍ ﺑﻪ ﻣﻔﺴﺮ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﺩ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺑﺪﻭﻥ ﻫﺴﺘﻪ ﺍﺻﻠﯽ ﻣﺘﻠﺐ ﻏﻴﺮ ﻗﺎﺑﻞ ﺍﺟﺮﺍ ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﺭﺍﻩ‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺍﺟﺮﺍﻳﯽ ﮐﺮﺩﻥ ﻭ ﻣﺴﺘﻘﻞ ﮐﺮﺩﻥ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎ ﺍﺯ ﻫﺴﺘﻪ ﻣﺤﺎﺳﺒﺎﺗﯽ ﻣﺘﻠﺐ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻭﻟﯽ ﺩﺭ ﺣﺎﻟﺖ ﮐﻠﯽ ﺑﺎﻳﺪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺭﺍ ﻫﻤﻴﺸﻪ ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺖ. ﻧﻴﺎﺯ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺩﺭ ﻣﺘﻠﺐ ﺑﻪ ﻫﺴﺘﻪ ﺍﺻﻠﯽ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﻣﻮﺟﺐ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻧﺘﻮﺍﻥ ﺍﺯ ﺍﻳﻦ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎ ﺩﺭ ﻣﺤﻴﻂ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺁﻧﻬﺎ ﺭﺍ ﺗﺤﺖ ﻭﺏ ﺑﮑﺎﺭ ﺑﺮﺩ.

ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻦ ﺑﺎﻳﺪ ﺩﺍﻧﺴﺖ ﮐﻪ ﻣﺘﻠﺐ ﺑﺮﺍﯼ ﮐﺎﺭ ﺑﺎ ﻣﺠﻤﻮﻋﻪ ﺩﺍﺩﻩ‌ﻫﺎﯼ ﻣﺘﻌﺎﺭﻑ ﻃﺮﺍﺣﯽ ﺷﺪﻩ ﺍﺳﺖ ﻧﻪ ﻣﺠﻤﻮﻋﻪ‌ﻫﺎﯼ ﺑﺴﻴﺎﺭ ﺑﺰﺭﮔﯽ ﺍﺯ ﺍﻃﻼﻋﺎﺕ. ﺍﮔﺮﭼﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﭼﻨﻴﻦ ﮐﺎﺭﻫﺎﻳﯽ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﺍﻳﻦ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺳﭙﺮﺩ ﻭﻟﯽ ﺑﺎﻳﺪ ﺍﺯ ﺍﺑﺘﺪﺍ ﺩﺍﻧﺴﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﻧﺒﺎﻳﺪ ﺍﻧﺘﻈﺎﺭ ﭘﺎﺳﺨﯽ ﺳﺮﻳﻊ ﺍﺯ ﺳﻴﺴﺘﻢ ﺑﺎﺷﻴﻢ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺍﮔﺮ ﻗﺮﺍﺭ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺗﺤﻠﻴﻞ ﺗﺼﺎﻭﻳﺮ ﺣﺎﺻﻞ ﺍﺯ ﻳﮏ ﺩﻭﺭﺑﻴﻦ ﮐﻨﺘﺮﻝ ﻫﻤﺰﻣﺎﻥ ﺳﻴﺴﺘﻤﯽ ﺭﺍ ﺑﺪﺳﺖ ﺑﮕﻴﺮﻳﻢ، ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﺑﺠﺎﯼ ﻣﺘﻠﺐ ﺍﺯ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﯼ ﺧﺎﺹ ﺍﻳﻦ ﮐﺎﺭ ﺑﻬﺮﻩ ﺑﺒﺮﻳﻢ.

 

ﻣﻨﺎﺑﻊ ﮐﻤﮑﯽ ﻭ ﺩﺳﺘﻮﺭ HELP
ﻳﮑﯽ ﺍﺯ ﺷﺎﺧﺼﻪ‌ﻫﺎﯼ ﻣﺘﻤﺎﻳﺰ ﮐﻨﻨﺪﻩ ﻣﺘﻠﺐ ﺍﺯ ﺩﻳﮕﺮ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﯼ ﻣﺸﺎﺑﻪ، ﮔﺴﺘﺮﺩﮔﯽ ﻭ ﻓﺮﺍﻭﺍﻧﯽ ﻣﻨﺎﺑﻊ ﮐﻤﮑﯽ ﻭ ﻣﺴﺘﻨﺪﺍﺕ ﻭ ﻣﺜﺎﻝ‌ﻫﺎﯼ ﻓﺮﺍﻭﺍﻧﯽ ﺍﺳﺖ ﮐﻪ ﻫﻢ ﺑﻮﺳﻴﻠﻪ ﺷﺮﮐﺖ ﺗﻮﺳﻌﻪ ﺩﻫﻨﺪﻩ ﻭ ﻫﻢ ﺑﻮﺳﻴﻠﻪ ﮐﺎﺭﺑﺮﺍﻥ ﺩﺭ ﺩﺳﺘﺮﺱ ﻋﻼﻗﻤﻨﺪﺍﻥ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺍﺳﺖ. ﺳﺎﺩﻩ‌ﺗﺮﻳﻦ ﻭ ﺩﺭ ﺩﺳﺘﺮﺱ‌ﺗﺮﻳﻦ ﺍﻳﻦ ﻣﻨﺎﺑﻊ ﺩﺳﺘﻮﺭ "ﮐﻤﮏ" ﻳﺎ HELP ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺭﻭﺵ‌ﻫﺎﯼ ﮔﻮﻧﺎﮔﻮﻥ ﻗﺎﺑﻞ ﺩﺳﺘﺮﺳﯽ ﺍﺳﺖ ﻭ ﺑﻪ ﺗﻘﺮﻳﺐ ﻫﻤﻪ ﻧﻴﺎﺯﻫﺎﯼ ﺍﻃﻼﻋﺎﺗﯽ ﺍﻭﻟﻴﻪ ﮐﺎﺭﺑﺮ ﺭﺍ ﺑﺮﺁﻭﺭﺩﻩ ﻣﯽ‌ﮐﻨﺪ.

ﺩﺳﺘﻮﺭ HELP ﺑﻪ ﺩﻭ ﺭﻭﺵ ﺍﺻﻠﯽ ﺩﺭ ﺩﺳﺘﺮﺱ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ؛ ﺭﺟﻮﻉ ﺑﻪ ﻣﻨﻮﯼ ﺑﺮﻧﺎﻣﻪ ﻳﺎ ﻧﻮﺷﺘﻦ ﺁﻥ ﺩﺭ ﺧﻂ ﻓﺮﻣﺎﻥ (Command Prompt) ﺑﺮﻧﺎﻣﻪ. ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻨﻮﯼ HELP ﻭ ﺟﺴﺘﺠﻮﯼ ﺩﺳﺘﻮﺭ ﻳﺎ ﺗﺎﺑﻌﯽ ﮐﻪ ﺩﺭﺑﺎﺭﻩ ﺁﻥ ﺳﻮﺍﻝ ﺩﺍﺭﻳﺪ ﺩﺭ ﺑﺨﺶ ﺟﺴﺘﺠﻮ، ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﻓﻬﺮﺳﺘﯽ ﺩﺳﺖ ﻳﺎﻓﺖ ﮐﻪ ﺷﺎﻣﻞ ﻧﺘﺎﻳﺞ ﺟﺴﺘﺠﻮ ﺩﺭ ﻫﻤﻪ ﺑﺨﺶ‌ﻫﺎ ﻭ ﺗﻮﺍﺑﻊ ﻭ ﺍﺑﺰﺍﺭﻫﺎﯼ ﻣﺘﻠﺐ ﺍﺳﺖ. ﻫﺮﮐﺪﺍﻡ ﺍﺯ ﻧﺘﺎﻳﺞ ﺷﺎﻣﻞ ﺷﺮﺡ ﺩﺳﺘﻮﺭ، ﭼﮕﻮﻧﮕﯽ ﮐﺎﺭﮐﺮﺩ، ﺣﺎﻟﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺳﺘﻔﺎﺩﻩ، ﻭ ﭼﻨﺪ ﻣﺜﺎﻝ ﺍﺳﺖ ﻭ ﮔﺎﻫﯽ ﺑﺮﺍﯼ ﺑﺮﺧﯽ ﺍﺯ ﻓﺮﺍﻣﻴﻦ، ﻗﻄﻌﺎﺕ ﭼﻨﺪ ﺭﺳﺎﻧﻪ‌ﺍﯼ ﻭ ﻣﺤﺎﻭﺭﻩ‌ﺍﯼ ﻧﻴﺰ ﺑﺮﺍﯼ ﺗﻔﻬﻴﻢ ﺑﻬﺘﺮ ﻣﻮﺿﻮﻉ ﺍﺿﺎﻓﻪ ﺷﺪﻩ ﺍﺳﺖ.
