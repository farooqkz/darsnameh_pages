ﺷﻴﻮﻩ ﻧﻤﺎﻳﺶ ﻫﻢ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﺳﺖ ﮐﻪ ﭘﺲ ﺍﺯ ﻧﻤﺎﻳﺶ ﮐﻮﺗﺎﻩ ﻣﺪﺕ ﻣﺘﻦ ﻧﻮﺗﻴﻔﻴﮑﻴﺸﻦ ﻫﺎ ﺑﺮ ﺭﻭﯼ ﺻﻔﺤﻪ ﻣﻮﺑﺎﻳﻞ، ﻫﻤﻪ ﺁﻧﻬﺎ ﺑﻪ ﺣﺎﻟﺖ ﮐﺸﻮﻳﯽ ﺩﺭ ﺑﺨﺶ ﺑﺎﻻﻳﯽ ﺻﻔﺤﻪ ﺟﻤﻊ ﻭ ﻣﺨﻔﯽ ﻣﯽ ﺷﻮﻧﺪ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻫﻤﻪ ﺁﻧﻬﺎ ﺭﺍ ﻳﮏ ﺟﺎ ﺑﺒﻴﻨﻴﺪ، ﺗﻨﻬﺎ ﮐﺎﻓﯽ ﺍﺳﺖ ﺍﻧﮕﺸﺖ ﺗﺎﻥ ﺭﺍ ﺩﺭ ﺑﺎﻻﯼ ﺻﻔﺤﻪ ﻣﻮﺑﺎﻳﻞ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﻭ ﺑﻪ ﺳﻤﺖ ﭘﺎﻳﻴﻦ ﺑﮑﺸﻴﺪ ﺗﺎ ﻧﻮﺍﺭ ﺁﺑﺸﺎﺭﯼ ﺁﮔﺎﻩ ﺳﺎﺯﯼ ﻫﺎ ﺑﺎﺯ ﺷﻮﺩ.

ﻳﮑﯽ ﺍﺯ ﺑﺰﺭﮔﺘﺮﻳﻦ ﻣﺰﻳﺖ ﻫﺎﯼ ﺍﻧﺪﺭﻭﻳﺪ ﻧﺴﺒﺖ ﺑﻪ ﺩﻳﮕﺮ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺎﯼ ﻣﻮﺑﺎﻳﻞ ﻫﻢ ﭼﻮﻥ iOS ﺍﻣﮑﺎﻥ ﺷﺨﺼﯽ ﺳﺎﺯﯼ ﺻﻔﺤﻪ ﺧﺎﻧﻪ (Home) ﺍﺳﺖ. ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺗﻨﻬﺎ ﺑﻪ ﻳﮏ ﺻﻔﺤﻪ ﺧﺎﻧﮕﯽ (ﺍﺻﻠﯽ) ﺍﮐﺘﻔﺎ ﻧﮑﺮﺩﻩ ﻭ ﺩﺭ ﺍﮐﺜﺮ ﻣﻮﺑﺎﻳﻞ ﻫﺎ ۵ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺗﺎﻥ ﻣﯽ ﮔﺬﺍﺭﺩ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﻭﻳﺠﺖ ﻫﺎ ﻭ ﺁﻳﮑﻮﻥ ﻫﺎﯼ ﻣﻴﺎﻧﺒﺮ ﺑﺮﻧﺎﻣﻪ ﻫﺎ ﺭﺍ ﺑﻪ ﺳﻠﻴﻘﻪ ﺧﻮﺩ ﺩﺭ ﻫﺮ ﻳﮏ ﭼﻴﺪﻣﺎﻥ ﮐﻨﻴﺪ.

ﺍﺻﻼﺡ ﻭ ﺗﻐﻴﻴﺮ ﺻﻔﺤﻪ ﺧﺎﻧﮕﯽ ﮐﺎﻣﻼ ﺳﺎﺩﻩ ﺍﺳﺖ. ﺗﻨﻬﺎ ﮐﺎﻓﯽ ﺍﺳﺖ ﺑﺮ ﺭﻭﯼ ﺑﺨﺶ ﺧﺎﻟﯽ ﺍﺯ ﺻﻔﺤﻪ ﺧﺎﻧﮕﯽ ﻳﮏ ﺗﭗ ﻃﻮﻻﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﻳﻌﻨﯽ ﺍﻳﻨﮑﻪ ﺍﻧﮕﺸﺖ ﺗﺎﻥ ﺭﺍ ﺑﺮﺍﯼ ﻣﺪﺕ ﺯﻣﺎﻧﯽ ﺣﺪﻭﺩ ۱۰ ﺛﺎﻧﻴﻪ ﺑﺮ ﺭﻭﯼ ﺻﻔﺤﻪ ﮔﺬﺍﺷﺘﻪ ﻭ ﺳﭙﺲ ﺁﻥ ﺭﺍ ﺑﺮﺩﺍﺭﻳﺪ ﺗﺎ ﻣﻨﻮﯼ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﻣﻴﺎﻧﺒﺮﻫﺎ ﻭ ﻭﻳﺠﺖ ﻫﺎ ﺑﺎﺯ ﺷﻮﺩ.

ﺍﻳﻦ ﻣﻴﺎﻧﺒﺮﻫﺎ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺷﺎﻣﻞ ﻫﺮ ﭼﻴﺰﯼ ﺑﺎﺷﻨﺪ. ﺍﺯ ﺁﻳﮑﻮﻥ ﺍﭘﻠﻴﮑﻴﺸﻦ ﻫﺎ ﻭ ﺑﺎﺯﯼ ﻫﺎ ﮔﺮﻓﺘﻪ ﺗﺎ ﺍﻃﻼﻋﺎﺕ ﺗﻤﺎﺱ ﻳﮏ ﺷﺨﺺ ﺧﺎﺹ ﻳﺎ ﺑﻮﮎ ﻣﺎﺭﮎ ﺑﺮﺧﯽ ﺳﺎﻳﺖ ﻫﺎ ﺑﺮﺍﯼ ﺩﺳﺘﺮﺳﯽ ﺳﺮﻳﻊ ﺑﻪ ﺁﻧﻬﺎ. ﺷﻤﺎ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺩﺳﺘﻪ ﺑﻨﺪﯼ ﺻﻔﺤﺎﺕ ﺧﺎﻧﮕﯽ ﺍﻧﺪﺭﻭﻳﺪ ﺧﻮﺩ، ﻫﺮ ﺻﻔﺤﻪ ﺭﺍ ﺑﻪ ﮔﺮﻭﻩ ﺧﺎﺻﯽ ﺍﺯ ﻣﻴﺎﻧﺒﺮﻫﺎ ﻳﺎ ﻭﻳﺠﺖ ﻫﺎ ﺍﺧﺘﺼﺎﺹ ﺩﻫﻴﺪ.

ﺗﻔﺎﻭﺕ ﻭﻳﺠﺖ ﻫﺎ ﺑﺎ ﻣﻴﺎﻧﺒﺮ ﺩﺭ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻣﻴﺎﻧﺒﺮ، ﺗﻨﻬﺎ ﻟﻴﻨﮑﯽ ﺑﻪ ﺑﺮﻧﺎﻣﻪ ﺍﺻﻠﯽ ﺍﺳﺖ ﻭ ﻫﻴﭻ ﻓﻌﺎﻟﻴﺖ ﻭ ﺗﻌﺎﻣﻞ ﺩﻳﮕﺮﯼ ﻧﺪﺍﺭﺩ. ﺍﻣﺎ ﻭﻳﺠﺖ ﻫﺎ ﺩﺭ ﻭﺍﻗﻊ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﮐﻮﭼﮑﯽ ﺑﺎ ﻗﺎﺑﻠﻴﺖ ﺗﻌﺎﻣﻞ، ﻭ ﺩﺭﻳﺎﻓﺖ ﻭ ﻧﻤﺎﻳﺶ ﺍﻃﻼﻋﺎﺕ ﻫﺴﺘﻨﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻳﮏ ﻭﻳﺠﺖ ﺁﺏ ﻭ ﻫﻮﺍ ﻣﺮﺗﺒﺎ ﺍﻃﻼﻋﺎﺕ ﻫﻮﺍﺷﻨﺎﺳﯽ ﺷﻬﺮ ﻣﻨﺘﺨﺐ ﺷﻤﺎ ﺭﺍ ﺍﺯ ﺍﻳﻨﺘﺮﻧﺖ ﺩﺭﻳﺎﻓﺖ ﮐﺮﺩﻩ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺻﻔﺤﻪ ﺍﺻﻠﯽ ﺑﻪ ﻧﻤﺎﻳﺶ ﻣﯽ ﮔﺬﺍﺭﺩ. ﻳﺎ ﻭﻳﺠﺖ ﺳﺎﻋﺖ، ﺯﻣﺎﻥ ﻭ ﺗﺎﺭﻳﺦ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ ﺩﻫﺪ. ﻭﻳﺠﺘﯽ ﻣﺎﻧﻨﺪ ﻓﻴﺲ ﺑﻮﮎ ﻫﻢ ﻣﯽ ﺗﻮﺍﻧﺪ ﺁﺧﺮﻳﻦ ﺁﭘﺪﻳﺖ ﻫﺎﯼ ﺍﮐﺎﻧﺖ ﻓﻴﺲ ﺑﻮﮎ ﺗﺎﻥ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺻﻔﺤﻪ ﺑﻪ ﻧﻤﺎﻳﺶ ﺑﮕﺬﺍﺭﺩ ﻭ ﺑﺮﺍﯼ ﭼﮏ ﮐﺮﺩﻥ ﺣﺴﺎﺏ ﺗﺎﻥ ﻧﻴﺎﺯﯼ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﺮﻭﺭﮔﺮ ﻳﺎ ﺍﭘﻠﻴﮑﻴﺸﻦ ﻓﻴﺲ ﺑﻮﮎ ﻧﺪﺍﺭﻳﺪ.

ﮐﭙﯽ ﻭ ﭘﻴﺴﺖ
ﮐﭙﯽ ﻭ ﭘﻴﺴﺖ ﺩﺭ ﺍﻧﺪﺭﻭﻳﺪ ﺑﺴﻴﺎﺭ ﺷﺒﻴﻪ ﺑﻪ ﺩﻳﮕﺮ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺎﯼ ﻣﻮﺑﺎﻳﻞ ﺍﺳﺖ ﻭ ﭼﻨﺪﺍﻥ ﮐﺎﺭﮐﺮﺩ ﭘﻴﭽﻴﺪﻩ ﺍﯼ ﻧﺪﺍﺭﺩ. ﺷﺎﻳﺪ ﺗﻨﻬﺎ ﺗﻔﺎﻭﺕ ﺁﻥ ﺭﺍ ﺑﺘﻮﺍﻥ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺁﻳﮑﻮﻥ ﺑﻪ ﺟﺎﯼ ﮐﻠﻤﺎﺕ ﻫﻨﮕﺎﻡ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ، ﺩﺍﻧﺴﺖ. ﺗﻨﻬﺎ ﮐﺎﻓﯽ ﺍﺳﺖ ﺍﻧﮕﺸﺖ ﺗﺎﻥ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﻣﺘﻨﯽ ﮐﻪ ﻣﯽ ﺧﻮﺍﻫﻴﺪ ﮐﭙﯽ ﺷﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ ﻭ ﺁﻥ ﺭﺍ ﺗﺎ ﺯﻣﺎﻥ ﺁﺷﮑﺎﺭ ﺷﺪﻥ ﺫﺭﻩ ﺑﻴﻦ ﻧﮕﻪ ﺩﺍﺭﻳﺪ. ﺍﮐﻨﻮﻥ ﺩﻭ ﺩﺳﺘﮕﻴﺮﻩ ﺳﺒﺰ ﺭﻧﮓ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﻳﺪ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﮐﻠﻤﺎﺕ ﻳﺎ ﺟﻤﻼﺕ ﻣﻮﺭﺩ ﻧﻈﺮ ﺑﺮﺍﯼ ﮐﭙﯽ ﺭﺍ ﻫﺎﻳﻼﻳﺖ (ﺭﻧﮕﯽ) ﮐﻨﻴﺪ. ﺣﺎﻝ ﺍﺯ ﻣﻨﻮﯼ ﮐﻮﭼﮑﯽ ﮐﻪ ﺩﺭ ﺑﺎﻻﯼ ﻣﺘﻦ ﻗﺮﺍﺭ ﺩﺍﺭﺩ، ﻋﻤﻠﻴﺎﺕ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ.

ﺁﻳﮑﻮﻧﯽ ﮐﻪ ﺷﺒﻴﻪ ﻭﺭﻗﻪ ﻫﺎﯼ ﮐﺎﻏﺬ ﺍﺳﺖ، ﮐﻞ ﺑﺨﺶ ﻫﺎﻳﻼﻳﺖ ﺷﺪﻩ ﺭﺍ ﺩﺭﻭﻥ ﮐﻠﻴﭗ ﺑﺮﺩ ﮐﭙﯽ ﻣﯽ ﮐﻨﺪ. ﺁﻳﮑﻮﻥ ﺫﺭﻩ ﺑﻴﻦ ﺑﻪ ﺟﺴﺘﺠﻮﯼ ﺑﺨﺶ ﺍﻧﺘﺨﺎﺏ ﺷﺪﻩ ﺩﺭ ﻭﻳﮑﯽ ﭘﺪﻳﺎ ﻣﯽ ﭘﺮﺩﺍﺯﺩ ﻭ ﺁﻳﮑﻮﻥ ﺁﺧﺮ ﻫﻢ ﻭﻇﻴﻔﻪ ﺗﻮﻟﻴﺪ Short URL ﻳﺎ ﺁﺩﺭﺱ ﮐﻮﺗﺎﻩ (ﺩﺭ ﺻﻮﺭﺕ ﻫﺎﻳﻼﻳﺖ ﮐﺮﺩﻥ ﻳﮏ ﺁﺩﺭﺱ ﺍﻳﻨﺘﺮﻧﺘﯽ) ﺭﺍ ﺩﺍﺭﺩ.

 

ﻳﮑﭙﺎﺭﭼﮕﯽ ﺑﺎ ﮔﻮﮔﻞ
ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺍﻧﺘﻈﺎﺭ ﻣﯽ ﺭﻭﺩ، ﮔﻮﮔﻞ ﺧﺪﻣﺎﺕ ﻭ ﻣﺤﺼﻮﻻﺕ ﺁﻧﻼﻳﻦ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺍﺩﻏﺎﻡ ﺷﺪﻩ ﻭ ﻳﮑﭙﺎﺭﭼﻪ ﺑﺎ ﺍﻧﺪﺭﻭﻳﺪ ﻋﺮﺿﻪ ﻣﯽ ﮐﻨﺪ ﻭ ﺷﻤﺎ ﺑﺮﺍﯼ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺎﻣﻞ ﺍﺯ ﺍﻣﮑﺎﻧﺎﺕ ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻗﺒﻞ ﺍﺯ ﻫﺮ ﭼﻴﺰ ﻧﻴﺎﺯ ﺑﻪ ﻳﮏ ﺍﮐﺎﻧﺖ ﮔﻮﮔﻞ (ﺍﻳﻤﻴﻞ ﺟﻴﻤﻴﻞ) ﺩﺍﺭﻳﺪ.

ﺍﻧﺪﺭﻭﻳﺪ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﻳﻤﻴﻞ ﻫﺎ، ﺍﻃﻼﻋﺎﺕ ﺗﻤﺎﺱ ﺩﻭﺳﺘﺎﻥ (ﮐﺎﻧﺘﮑﺖ) ﻭ ﻫﻤﻪ ﺯﻧﺪﮔﯽ ﮔﻮﮔﻠﯽ ﺗﺎﻥ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﻫﻤﺴﺎﻥ ﺑﺎ ﻭﺏ (Sync)، ﺑﺮ ﺭﻭﯼ ﺗﻠﻔﻦ ﻋﺮﺿﻪ ﮐﻨﺪ. ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﺍﺯ ﺍﻳﻦ ﭘﺲ ﺗﻤﺎﻡ ﺍﻃﻼﻋﺎﺕ ﺗﻤﺎﺱ، ﺍﻳﻤﻴﻞ ﻫﺎ، ﻳﺎﺩﺩﺍﺷﺖ ﻫﺎ ﻭ ﻳﺎﺩﺁﻭﺭﻫﺎﯼ ﺗﻘﻮﻳﻢ ﻭ ﺩﻳﮕﺮ ﻣﺤﺘﻮﺍﻫﺎﯼ ﺗﻠﻔﻦ ﺍﻧﺪﺭﻭﻳﺪﯼ ﺗﺎﻥ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺍﮐﺎﻧﺖ ﮔﻮﮔﻞ ﻫﻢ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﻭ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺁﻧﻬﺎ ﺑﺮ ﺭﻭﯼ ﺩﻳﮕﺮ ﺍﺑﺰﺍﺭﻫﺎﯼ ﺍﻧﺪﺭﻭﻳﺪﯼ ﺗﺎﻥ ﻧﻴﺰ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ.

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﺮﺍﯼ ﺍﻭﻟﻴﻦ ﺑﺎﺭ ﺍﺯ ﻣﻮﺑﺎﻳﻞ ﺍﻧﺪﺭﻭﻳﺪﯼ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻴﺪ، ﺍﺯ ﺷﻤﺎ ﺧﻮﺍﺳﺘﻪ ﻣﯽ ﺷﻮﺩ ﺍﻃﻼﻋﺎﺕ ﺍﮐﺎﻧﺖ ﮔﻮﮔﻞ ﺗﺎﻥ ﺭﺍ ﺩﺭ ﺁﻥ ﺛﺒﺖ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻗﺒﻼ ﺍﻧﺠﺎﻡ ﻧﺪﺍﺩﻳﺪ، ﺍﮐﻨﻮﻥ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺭﻓﺘﻦ ﺑﻪ ﺑﺨﺶ Settings ﻭ ﺍﻧﺘﺨﺎﺏ ﮔﺰﻳﻨﻪ Accounts & Sync ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﭘﺲ ﺍﺯ ﺍﻳﻦ ﮐﺎﺭ ﮔﻮﮔﻞ ﺑﻪ ﺻﻮﺭﺕ ﺧﻮﺩﮐﺎﺭ ﻭ ﺩﻭ ﻃﺮﻓﻪ ﺍﻃﻼﻋﺎﺕ ﺣﺴﺎﺏ ﺗﺎﻥ ﺍﺯ ﺟﻤﻠﻪ ﺷﻤﺎﺭﻩ ﻫﺎﯼ ﺗﻤﺎﺱ ﻭ ﺍﻳﻤﻴﻞ ﻫﺎ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﺍﺑﺰﺍﺭ ﺍﻧﺪﺭﻭﻳﺪﯼ ﻭ ﺳﺮﻭﺭ ﺧﻮﺩ ﺳﻴﻨﮏ ﻭ ﻫﻤﺴﺎﻥ ﺳﺎﺯﯼ ﻣﯽ ﮐﻨﺪ.

 

ﻳﮑﭙﺎﺭﭼﻪ ﺳﺎﺯﯼ ﺑﺎ ﺩﻳﮕﺮ ﺳﺮﻭﻳﺲ ﻫﺎ  ﻭ ﺑﺮﻧﺎﻣﻪ ﻫﺎ
ﻧﻪ ﺗﻨﻬﺎ ﺳﺮﻭﻳﺲ ﻫﺎﯼ ﮔﻮﮔﻞ ﺑﻪ ﺻﻮﺭﺕ ﻳﮑﭙﺎﺭﭼﻪ ﻭ ﺍﺩﻏﺎﻡ ﺷﺪﻩ ﺑﺎ ﺍﻧﺪﺭﻭﻳﺪ ﻗﺎﺑﻞ ﺍﺳﺘﻔﺎﺩﻩ ﺍﻧﺪ، ﺑﻠﮑﻪ ﺧﺪﻣﺎﺕ ﺳﺎﻳﺖ ﻫﺎﯼ ﺩﻳﮕﺮﯼ ﺭﺍ ﻫﻢ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﭘﻠﻴﮑﻴﺸﻦ ﻫﺎﯼ ﻭﻳﮋﻩ ﺁﻧﻬﺎ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﺻﻮﺭﺕ ﻳﮑﭙﺎﺭﭼﻪ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﺩﺍﺩ. ﺍﺯ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﻓﻴﺴﺒﻮﮎ، ﺗﻮﻳﻴﺘﺮ، ﺍﺳﮑﺎﻳﭗ ﻭ ﺑﺮﺧﯽ ﺳﺮﻭﻳﺲ ﻫﺎﯼ ﺍﻳﻤﻴﻞ ﺍﺷﺎﺭﻩ ﮐﺮﺩ. ﺑﺮﺍﯼ ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﺍﻳﻦ ﺍﮐﺎﻧﺖ ﻫﺎ ﻧﻴﺰ ﻫﻤﺎﻧﻨﺪ ﺑﺨﺶ ﻗﺒﻠﯽ ﻣﯽ ﺗﻮﺍﻥ ﺍﺯ ﻣﻨﻮﯼ Accounts & Sync ﺩﺭ ﻗﺴﻤﺖ Settings ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩ.

ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﺗﻨﻬﺎ ﮔﻮﺷﻪ ﺍﯼ ﺍﺯ ﺍﻣﮑﺎﻧﺎﺕ ﮔﺴﺘﺮﺩﻩ ﺍﻧﺪﺭﻭﻳﺪ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﻪ ﺁﻧﻬﺎ ﺍﺷﺎﺭﻩ ﮐﺮﺩﻳﻢ. ﺍﻟﺒﺘﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﮐﺎﺭﮐﺮﺩﻫﺎﯼ ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﻢ ﮐﺎﻣﻼ ﻫﻤﺎﻧﻨﺪ ﺑﻘﻴﻪ ﻣﻮﺑﺎﻳﻞ ﻫﺎ ﺍﺳﺖ ﻭ ﺑﺎ ﮐﻤﯽ ﺟﺴﺘﺠﻮ، ﮐﺎﺭ ﺑﺎ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺭﺍﺣﺘﯽ ﻣﯽ ﺁﻣﻮﺯﻳﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﮐﺎﺭﻫﺎﻳﯽ ﻫﻢ ﭼﻮﻥ ﺷﻤﺎﺭﻩ ﮔﻴﺮﯼ، ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﺷﻤﺎﺭﻩ ﺑﻪ ﻟﻴﺴﺖ ﺩﻓﺘﺮ ﺗﻠﻔﻦ، ﺟﺴﺘﺠﻮ ﺩﺭ ﻣﻴﺎﻥ ﺷﻤﺎﺭﻩ ﻫﺎ، ﺍﺭﺳﺎﻝ ﺍﺱ ﺍﻡ ﺍﺱ ﻭ... ﺗﻔﺎﻭﺕ ﭼﻨﺪﺍﻧﯽ ﺩﺭ ﻣﻮﺑﺎﻳﻞ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻧﺪﺍﺭﻧﺪ ﻭ ﺑﻪ ﺭﺍﺣﺘﯽ ﻗﺎﺑﻞ ﺍﺳﺘﻔﺎﺩﻩ ﺍﻧﺪ.

ﺍﻟﺒﺘﻪ ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﺑﺨﺶ ﻫﺎﯼ ﮐﻠﻴﺪﯼ ﺍﻧﺪﺭﻭﻳﺪ ﺍﭘﻠﻴﮑﻴﺸﻦ ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﯽ ﺗﻮﺍﻥ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﻧﺼﺐ ﮐﺮﺩﻩ ﻭ ﮐﺎﺭﻫﺎﯼ ﺟﺬﺍﺏ ﻭ ﻭﻳﮋﻩ ﺍﯼ ﺑﺎ ﺁﻧﻬﺎ ﺍﻧﺠﺎﻡ ﺩﺍﺩ. ﺩﺭ ﺩﺭﺱ «ﺷﻴﻮﻩ ﻫﺎﯼ ﻧﺼﺐ ﺑﺮﻧﺎﻣﻪ ﺩﺭ ﺍﻧﺪﺭﻭﻳﺪ»، ﺑﻴﺸﺘﺮ ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.
