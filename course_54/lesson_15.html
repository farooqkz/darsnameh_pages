ﺗﻘﻮﻳﺖ ﺣﺲ ﻫﻤﺪﻟﯽ

ﻫﻤﺪﻟﯽ ﻳﻌﻨﯽ ﺍﻳﻨﮑﻪ ﺑﺎ ﭼﺸﻢ ﺩﻳﮕﺮﺍﻥ ﺑﺒﻴﻨﻴﻢ، ﺑﺎ ﮔﻮﺵ ﺩﻳﮕﺮﺍﻥ ﺑﺸﻨﻮﻳﻢ ﻭ ﺑﺎ ﻗﻠﺐ ﺩﻳﮕﺮﺍﻥ ﺣﺲ ﮐﻨﻴﻢ. ﻧﺪﺍﺷﺘﻦ ﺣﺲ ﻫﻤﺪﻟﯽ ﻧﺘﻴﺠﻪ‌ﺍﺵ ﺍﺭﺗﺒﺎﻃﺎﺕ ﺿﻌﻴﻒ ﻭ ﺩﺭﮎ ﻧﮑﺮﺩﻥ ﺩﻳﮕﺮﺍﻥ ﺍﺳﺖ. ﺩﺭ ﻣﻬﺎﺭ ﻭ ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ، ﺟﺎﻟﺐ ﺍﺳﺖ ﺑﺪﺍﻧﻴﺪ ﮐﻪ ﺩﻟﻴﻞ ﺧﺸﻢ ﻣﺎ ﺗﺮﮐﻴﺒﯽ ﺍﺯ ﺭﻓﺘﺎﺭ ﺩﻳﮕﺮﺍﻥ ﻭ ﻧﺪﺍﺷﺘﻦ ﺣﺲ ﻫﻤﺪﻟﯽ ﺍﺳﺖ.

ﻫﻤﺪﻟﯽ، ﻗﺎﺑﻠﻴﺖ ﻭ ﺗﻮﺍﻧﺎﻳﯽ ﺩﻳﺪﻥ ﻭ ﺗﺠﺮﺑﻪ ﺩﻧﻴﺎ ﺍﺯ ﺩﻳﺪ ﻳﮏ ﻧﻔﺮ ﺩﻳﮕﺮ‌ ﺍﺳﺖ، ﺣﺘﯽ ﺍﮔﺮ ﺑﺎ ﺍﻭ ﻣﻮﺍﻓﻖ ﻭ ﻫﻢ ﻧﻈﺮ ﻧﺒﺎﺷﻴﻢ.

ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺍﺣﺴﺎﺳﺎﺕ‌ﻣﺎﻥ ﺑﻪ ﺍﻃﺮﺍﻓﻴﺎﻥ ﻳﮑﯽ ﺍﺯ ﺳﺨﺖ‌ﺗﺮﻳﻦ ﮐﺎﺭﻫﺎ ﺍﺳﺖ. ﺗﻮﺍﻧﺎﻳﯽ ﺍﻳﻨﮑﻪ ﺑﺪﺍﻧﻴﻢ ﭼﻪ ﺣﺴﯽ ﺩﺍﺭﻳﻢ ﻭ ﺍﻳﻨﮑﻪ ﺍﺣﺴﺎﺱ ﺁﻧﻬﺎﻳﯽ ﺭﺍ ﮐﻪ ﺍﻃﺮﺍﻑ ﻣﺎ ﻫﺴﺘﻨﺪ ﺩﺭﮎ ﮐﻨﻴﻢ، ﺑﻪ ﻣﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺭﺍﺑﻄﻪ ﻣﺜﺒﺘﯽ ﺑﺎ ﺑﻘﻴﻪ ﺩﺭﺳﺖ ﮐﻨﻴﻢ. ﺑﻪ ﺍﻳﻦ ﻗﺎﺑﻠﻴﺖ ﻭ ﺧﺼﻴﺼﻪ ﻫﻤﺪﻟﯽ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

ﺗﺎ ﺑﻪ ﺣﺎﻝ ﺁﻳﺎ ﺑﻪ  ﺍﻳﻦ ﺗﻮﺍﻧﺎﻳﯽ ﻓﮑﺮ ﮐﺮﺩﻩ ﺑﻮﺩﻳﺪ؟ ﺗﺎ ﺣﺎﻻ ﺑﻪ ﺍﻳﻦ ﻣﺘﻬﻢ ﺷﺪﻩ ﺑﻮﺩﻳﺪ ﮐﻪ ﻓﻘﻂ ﺑﻪ ﻓﮑﺮ ﺧﻮﺩﺗﺎﻥ ﻫﺴﺘﻴﺪ؟ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﮐﺴﯽ ﺑﻪ ﺷﻤﺎ ﮔﻔﺘﻪ ﮐﻪ ﺍﻧﮕﺎﺭ ﺑﺮﺍﻳﺘﺎﻥ ﻓﺮﻗﯽ ﻧﻤﯽ‌ﮐﻨﺪ ﺑﺮﺍﯼ ﺍﻃﺮﺍﻓﻴﺎﻧﺘﺎﻥ ﭼﻪ ﺍﺗﻔﺎﻗﯽ ﺑﻴﻔﺘﺪ؟

ﺍﮔﺮ ﺟﻮﺍﺏ ﺷﻤﺎ ﻣﺜﺒﺖ ﺍﺳﺖ، ﺷﺎﻳﺪ ﺑﻬﺘﺮ ﺑﺎﺷﺪ ﮐﻤﯽ ﺑﻴﺸﺘﺮ ﺭﺍﺟﻊ ﺑﻪ ﻫﻤﺪﻟﯽ ﺑﺪﺍﻧﻴﺪ.

 

ﻫﻤﺪﻟﯽ ﭼﺮﺍ ﻣﻬﻢ ﺍﺳﺖ؟
ﻧﺪﺍﺷﺘﻦ ﻫﻤﺪﻟﯽ ﺑﺎﻋﺚ ﺿﻌﻒ ﺩﺭ ﺍﺭﺗﺒﺎﻃﺎﺕ ﻭ ﺩﺭﮎ ﻧﮑﺮﺩﻥ ﻭ ﻧﻔﻬﻤﻴﺪﻥ ﺩﻳﮕﺮﺍﻥ ﻣﯽ‌ﺷﻮﺩ. ﻫﻤﺪﻟﯽ ﺩﺭ ﺳﻄﺢ ﺷﺨﺼﯽ ﻫﻢ ﻣﻬﻢ ﺍﺳﺖ ﭼﺮﺍ ﮐﻪ ﺑﻪ ﻣﺎ ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺑﺪﻩ ﺑﺴﺘﺎﻥ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﺩﺭﮎ ﮐﻨﻴﻢ ﻭ ﺭﻓﺘﺎﺭ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﭘﻴﺶ‌ﺑﻴﻨﯽ ﮐﻨﻴﻢ.

ﻫﻤﺪﻟﯽ، ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻫﻢ ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﻣﺎ ﺭﺍ ﺑﻬﺘﺮ ﺩﺭﮎ ﮐﻨﻨﺪ. ﻭﻗﺘﯽ ﺁﺩﻡ‌ﻫﺎﯼ ﻫﻤﺪﻝ ﺩﻭﺭ ﻭ ﺑﺮﻣﺎﻥ ﺑﺎﺷﻨﺪ ﺁﻧﻬﺎ ﻫﻢ ﺩﻧﻴﺎ ﺭﺍ ﺍﺯ ﺩﻳﺪ ﻣﺎ ﻣﯽ‌ﺑﻴﻨﻨﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺟﺎﯼ ﻣﺎ ﺑﮕﺬﺍﺭﻧﺪ.

ﭼﻘﺪﺭ ﺑﺮﺍﻳﺘﺎﻥ ﭘﻴﺶ ﺁﻣﺪﻩ ﮐﻪ ﺁﺭﺯﻭ ﮐﺮﺩﻳﺪ ﺍﯼ ﮐﺎﺵ، ﻫﻤﺴﺮ ﻭ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ، ﻫﻤﮑﺎﺭﺍﻥ ﻭ ﺩﻭﺳﺘﺎﻥ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺟﺎﯼ ﺷﻤﺎ ﻣﯽ‌ﮔﺬﺍﺷﺘﻨﺪ. ﺍﮔﺮ ﭼﻨﻴﻦ ﻟﺤﻈﺎﺗﯽ ﺩﺍﺷﺘﻪ‌ﺍﻳﺪ، ﺷﻤﺎ ﻫﻢ ﻧﻴﺎﺯ ﺑﻪ ﺗﻘﻮﻳﺖ ﻫﻤﺪﻟﯽ ﺭﺍ ﻣﯽ‌ﺩﺍﻧﻴﺪ.

ﻫﺮ ﭼﻪ ﺣﺲ ﻫﻤﺪﻟﯽ ﺑﺎ ﺁﺩﻡ‌ﻫﺎ ﺑﻴﺸﺘﺮ ﺑﺎﺷﺪ، ﮐﻤﺘﺮ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ ﺍﺯ ﺩﺳﺖ ﺁﻧﻬﺎ ﻋﺼﺒﺎﻧﯽ ﺷﻮﻳﻢ.

ﺭﺍﻩ‌ﻫﺎﻳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﻪ ﻣﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﻓﻘﻂ ﺑﻪ ﺧﻮﺩﻣﺎﻥ ﻓﮑﺮ ﻧﮑﻨﻴﻢ. ﻣﺜﻼ: 

ﺩﺭﺳﺖ ﮔﻮﺵ ﮐﺮﺩﻥ ﺑﻪ ﺩﻳﮕﺮﺍﻥ:
ﻳﮑﯽ ﺍﺯ ﺭﺍﻩ‌ﻫﺎﻳﺶ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺪﻭﻥ ﻓﻴﻠﺘﺮﻫﺎﯼ‌ﻣﺎﻥ ﺑﻪ ﺣﺮﻑ ﺑﻘﻴﻪ ﮔﻮﺵ ﮐﻨﻴﻢ، ﻳﻌﻨﯽ ﺍﻳﻨﮑﻪ ﺣﺮﻑ‌ﻫﺎﯼ ﺁﻧﻬﺎ ﺭﺍ ﻧﻪ ﺁﻥ ﺟﻮﺭﯼ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ، ﺑﻠﮑﻪ ﺟﻮﺭﯼ ﮐﻪ ﻭﺍﻗﻌﺎ ﻣﻨﻈﻮﺭ ﺁﻧﻬﺎﺳﺖ ﮔﻮﺵ ﮐﻨﻴﻢ. ﺑﻌﻀﯽ ﺍﺯ ﺍﻳﻦ ﻓﻴﻠﺘﺮﻫﺎﯼ ﻣﻌﻤﻮﻝ، ﮐﻪ ﮔﻮﺵ ﺩﺍﺩﻥ ﻣﺎ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﻨﺪ ﻋﺒﺎﺭﺗﻨﺪ ﺍﺯ:

ﺩﺭﺳﺖ/ﻏﻠﻂ: ﺍﮔﺮ ﺑﻪ ﺣﺮﻑ ﮐﺴﯽ ﮔﻮﺵ ﻣﯽ‌ﺩﻫﻴﻢ ﻭ ﻣﻴﺎﻥ ﺣﺮﻑ‌ﻫﺎﻳﺶ ﺩﻧﺒﺎﻝ ﺩﻟﻴﻞ ﺩﺭﺳﺖ ﺑﻮﺩﻥ ﻧﻈﺮ ﺧﻮﺩﻣﺎﻥ ﻭ ﻏﻠﻂ ﺑﻮﺩﻥ ﻧﻈﺮ ﺍﻭ ﻫﺴﺘﻴﻢ، ﺍﻳﻦ ﺍﺳﻤﺶ ﻫﻤﺪﻟﯽ ﻧﻴﺴﺖ!

ﺍﻧﺘﻘﺎﺩ/ﺿﺎﻳﻊ ﮐﺮﺩﻥ: ﺍﮔﺮ ﻗﺮﺍﺭ ﺑﺎﺷﺪ ﻫﺮ ﺍﻇﻬﺎﺭ ﻧﻈﺮﯼ ﮐﻪ ﺑﻘﻴﻪ ﻣﯽ‌ﮐﻨﻨﺪ ﺭﺍ ﺑﺎ ﺍﻳﻦ ﺩﻳﺪ ﺑﺸﻨﻮﻳﻢ ﮐﻪ ﺩﻟﻴﻠﯽ ﺍﺳﺖ ﺑﺮﺍﯼ ﺿﺎﻳﻊ ﮐﺮﺩﻥ ﻣﺎ، ﺍﻳﻨﺠﺎﺳﺖ ﮐﻪ ﻣﻨﻈﻮﺭ ﻓﺮﺩ ﻣﻘﺎﺑﻞ ﺭﺍ  ﻧﮕﺮﻓﺘﻪ‌ﺍﻳﻢ ﻭ ﺍﺣﺴﺎﺳﺶ ﺭﺍ ﺍﺷﺘﺒﺎﻩ ﻓﻬﻤﻴﺪﻩ‌ﺍﻳﻢ.

ﻣﻦ ﺑﻌﺪ ﭼﻪ ﻣﯽ‌ﺧﻮﺍﻫﻢ ﺑﮕﻮﻳﻢ؟: ﻭﻗﺘﯽ ﮐﺴﯽ ﺩﺍﺭﺩ ﺣﺮﻑ ﻣﯽ‌ﺯﻧﺪ ﻭ ﻣﺎ ﺑﻪ ﻓﮑﺮ ﺟﻮﺍﺏ ﺧﻮﺩﻣﺎﻥ ﻫﺴﺘﻴﻢ، ﻣﻌﻨﯽ‌ﺍﺵ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻫﺪﻑ‌ﻣﺎﻥ ﺍﺯ ﮔﻮﺵ ﮐﺮﺩﻥ، ﻫﻤﺪﻟﯽ ﻧﻴﺴﺖ. ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺣﺮﻑ‌ﻫﺎﯼ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﺑﺰﻧﻴﻢ ﻭ ﺍﻳﻦ ﺩﻳﮕﺮ ﮔﻮﺵ ﮐﺮﺩﻥ ﺑﻪ ﺩﻳﮕﺮﯼ ﻧﻴﺴﺖ!

ﺑﺮﻳﻢ ﺳﺮ ﺍﺻﻞ ﻣﻄﻠﺐ: ﺍﮔﺮ ﺑﻪ ﺩﻭﺳﺖ‌ﺗﺎﻥ ﮔﻮﺵ ﻣﯽ‌ﺩﻫﻴﺪ ﻭ ﺍﻭ ﺩﺍﺭﺩ ﺑﺎ ﺟﺰﺋﻴﺎﺕ ﺣﺮﻑ  ﻣﯽ‌ﺯﻧﺪ ﻭ ﺷﻤﺎ ﻣﺮﺗﺐ ﺩﻧﺒﺎﻝ ﺁﺧﺮ، «ﺍﺻﻞ ﻣﻄﻠﺐ» ﻣﯽ‌ﮔﺮﺩﻳﺪ، ﺷﻤﺎ ﺑﺎ ﺍﻭ ﻫﻤﺪﻝ ﻧﺸﺪﻳﺪ. ﺷﻤﺎ ﺗﺠﺮﺑﻪ ﺍﻭ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﻴﺪ، ﺑﺎ ﺍﻭ ﺷﺮﻳﮏ ﺗﺠﺮﺑﻪ‌ﺍﺵ ﻧﻤﯽ‌ﺷﻮﻳﺪ. ﻓﻘﻂ ﺩﻧﺒﺎﻝ ﺑﺨﺶ ﻣﻬﻢ ﺩﺍﺳﺘﺎﻥ ﺁﻥ ﻫﻢ ﻣﻬﻢ ﺍﺯ ﻧﻈﺮ ﺧﻮﺩﺗﺎﻥ ﻫﺴﺘﻴﺪ!

ﻫﻤﻪ ﻣﺎ ﺩﺭ ﻃﻮﻝ ﺭﻭﺯ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ، ﺩﺭ ﺧﻴﺎﺑﺎﻥ، ﺩﺭ ﺧﺎﻧﻪ، ﻣﻮﻗﻊ ﺭﺍﻧﻨﺪﮔﯽ ﻭ ﺧﺮﻳﺪ، ﺣﺘﯽ ﺧﻴﻠﯽ ﻭﻗﺖ‌ﻫﺎ ﻫﻤﺰﻣﺎﻥ ﺑﺎ ﮐﺎﺭﻫﺎﯼ ﺩﻳﮕﺮﯼ ﮐﻪ ﺩﺍﺭﻳﻢ ﺣﺮﻑ‌ﻫﺎﻳﯽ ﻫﻢ ﺑﺎ ﺧﻮﺩﻣﺎﻥ ﻣﯽ‌ﺯﻧﻴﻢ. ﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﮔﺎﻫﯽ ﺁﻧﻘﺪﺭ ﺣﻮﺍﺱ‌ﻣﺎﻥ ﮔﺮﻡ ﮐﺎﺭﻫﺎﯼ ﺩﻳﮕﺮ ﺑﺎﺷﺪ ﮐﻪ ﺍﺻﻼ ﻣﺘﻮﺟﻪ‌ﺍﺵ ﻧﺒﺎﺷﻴﻢ.

ﺍﻳﻦ ﻣﮑﺎﻟﻤﻪ ﺩﺭﻭﻧﯽ ﻭ ﺣﺮﻑ‌ﻫﺎﻳﯽ ﮐﻪ ﺧﻮﺩﻣﺎﻥ ﺑﻪ ﺧﻮﺩﻣﺎﻥ ﻣﯽ‌ﺯﻧﻴﻢ ﺭﻭﯼ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻥ ﻭ ﺷﺪﺕ ﮔﺮﻓﺘﻦ ﺍﺣﺴﺎﺳﺎﺕ‌ﻣﺎﻥ ﺍﺯ ﺟﻤﻠﻪ ﻋﺼﺒﺎﻧﻴﺖ، ﺗﺎﺛﻴﺮ ﺩﺍﺭﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﭘﺮﻭﺳﻪ ﮐﻨﺘﺮﻝ ﺧﺸﻢ، ﺷﻨﺎﺧﺘﻦ ﺍﻳﻦ ﻣﮑﺎﻟﻤﻪ ﺩﺭﻭﻧﯽ ﺧﻴﻠﯽ ﻣﻬﻢ ﺍﺳﺖ.

ﺍﻳﻦ ﻣﮑﺎﻟﻤﻪ ﺑﺎ ﺧﻮﺩ ﻣﻤﮑﻦ ﺍﺳﺖ ﺧﻴﻠﯽ ﻭﻗﺖ‌ﻫﺎ ﻣﻨﻔﯽ ﻭ ﺩﻭﺭ ﺍﺯ ﻭﺍﻗﻌﻴﺖ ﺑﺎﺷﺪ ﮐﻪ ﻣﺎ ﺑﺮﺍﯼ ﮐﻨﺘﺮﻝ ﻭ ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ‌ﻣﺎﻥ، ﻫﻢ ﺑﺎﻳﺪ ﺁﻧﻬﺎ ﺭﺍ ﺑﺸﻨﺎﺳﻴﻢ ﻭ ﺍﮔﺮ ﻭﺍﻗﻌﯽ ﻧﻴﺴﺘﻨﺪ ﺗﻐﻴﻴﺮﺷﺎﻥ ﺩﻫﻴﻢ. ﺍﻟﺒﺘﻪ ﮐﺎﺭ ﺁﺳﺎﻧﯽ ﻧﻴﺴﺖ؛ ﺑﺮﺍﯼ ﺁﻧﮑﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﺣﺘﯽ ﻣﺘﻮﺟﻪ ﺁﻥ ﻧﺒﺎﺷﻴﺪ. ﻭﻟﯽ ﺗﮑﺮﺍﺭ ﻣﮑﺎﻟﻤﻪ‌ﻫﺎﯼ ﻏﻴﺮ ﻭﺍﻗﻌﯽ ﺑﺎﻋﺚ ﺑﻮﺟﻮﺩ ﺁﻣﺪﻥ ﻳﮑﺴﺮﯼ ﺍﺣﺴﺎﺳﺎﺕ ﻣﯽ‌ﺷﻮﻧﺪ ﻣﺜﻞ ﻋﺼﺒﺎﻧﻴﺖ.

ﺗﻐﻴﻴﺮ ﮔﻔﺘﮕﻮﯼ ﺩﺭﻭﻧﯽ ﺑﻪ ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ، ﻧﻮﻉ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﺁﻥ ﻭ ﻣﺪﺕ ﺯﻣﺎﻧﯽ ﮐﻪ ﺧﺸﻢ‌ﺗﺎﻥ ﺍﺩﺍﻣﻪ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺭﻓﺘﺎﺭﯼ ﮐﻪ ﺩﺭ ﺯﻣﺎﻥ ﻋﺼﺒﺎﻧﻴﺖ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻴﺪ، ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ.

ﮐﺎﺭﻫﺎﻳﯽ ﻫﺴﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺮﺍﯼ ﺗﻐﻴﻴﺮ ﻣﮑﺎﻟﻤﻪ ﺩﺭﻭﻧﯽ ﺗﻤﺮﻳﻦ ﮐﺮﺩ.

ﻣﺴﺌﻮﻟﻴﺖ ﮐﺎﺭﻣﺎﻥ ﺭﺍ ﺧﻮﺩﻣﺎﻥ ﺑﻌﻬﺪﻩ ﺑﮕﻴﺮﻳﻢ:
ﺣﺮﻑ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﻌﻤﻮﻻ ﺑﻪ ﺧﻮﺩﻣﻮﻥ ﻣﯽ‌ﮔﻮﻳﻴﻢ: «ﺍﻳﻦ ﺑﭽﻪ ﺍﻋﺼﺎﺏ ﻣﻨﻮ ﺧﻮﺭﺩ ﻣﯽ‌ﮐﻨﻪ.». ﻭﻗﺘﯽ ﻣﺴﺌﻮﻟﻴﺖ ﮐﺎﺭ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﻗﺒﻮﻝ ﮐﺮﺩﻩ ﺑﺎﺷﻴﻢ، ﺑﻪ ﺧﻮﻣﺎﻥ ﻣﯽ‌ﮔﻮﻳﻴﻢ: «ﻣﻦ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻡ ﻭ ﺟﻮﺵ ﺁﻭﺭﺩﻡ.»

ﺣﻮﺍﺱ‌ﻣﺎﻥ ﺑﻪ ﻓﮑﺮﻫﺎﯼ ﻧﺎﺧﻮﺁﮔﺎﻩ ﺑﺎﺷﺪ:
ﺧﻴﻠﯽ ﻭﻗﺖ‌ﻫﺎ ﻣﺎ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺑﺪﺍﻧﻴﻢ ﻭ ﻣﺘﻮﺟﻪ ﺑﺎﺷﻴﻢ، ﺩﺭ ﺷﺮﺍﻳﻄﯽ ﺑﺮﺍﺳﺎﺱ ﺩﻳﺪ ﮐﻠﯽ‌ﻣﺎﻥ ﺷﺮﻭﻉ ﺑﻪ ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺭ ﻣﻮﺭﺩ ﻣﺴﺎﺋﻞ ﻣﺨﺘﻠﻒ ﻣﯽ‌ﮐﻨﻴﻢ. ﺍﻳﻦ ﺑﻪ ﺻﻮﺭﺕ ﺧﻮﺩﮐﺎﺭ ﺍﺗﻔﺎﻕ ﻣﯽ‌ﺍﻓﺘﺪ. ﻣﺜﻞ ﻳﮏ ﺑﺎﺯﯼ ﮐﻪ ﭘﻨﺞ ﻣﺮﺣﻠﻪ ﺩﺍﺭﺩ ﻭ ﺷﻤﺎ ﺑﻌﺪ ﺍﺯ ﻫﺮ ﻣﺮﺣﻠﻪ ﺑﻪ ﺷﮑﻞ ﺧﻮﺩﮐﺎﺭ ﺑﻪ ﻣﺮﺣﻠﻪ ﺑﻌﺪﯼ ﻣﯽ‌ﺭﻭﻳﺪ.

ﻣﺮﺣﻠﻪ ﺍﻭﻝ- ﻳﮏ ﺍﺗﻔﺎﻕ ﺳﺎﺩﻩ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺍﺗﻔﺎﻕ ﻣﯽ‌ﺍﻓﺘﺪ: ﺩﺧﺘﺮﺗﺎﻥ ﺑﺪﻭﻥ ﺍﺟﺎﺯﻩ ﺷﻤﺎ ﺳﺮ ﮐﺎﻣﭙﻴﻮﺗﺮ ﺷﻤﺎ ﺭﻓﺘﻪ ﺍﺳﺖ.

ﻣﺮﺣﻠﻪ ﺩﻭﻡ- ﻓﮑﺮﻫﺎﻳﯽ ﮐﻪ ﻧﺎﺧﻮﺩﺍﮔﺎﻩ ﺑﻪ ﺫﻫﻦ ﺷﻤﺎ ﻣﯽ‌ﺁﻳﺪ: ﺍﻳﻦ ﺑﭽﻪ ﺧﻴﻠﯽ ﺑﯽ‌ﺟﺎ ﮐﺮﺩﻩ ﺑﯽ‌ﺍﺟﺎﺯﻩ ﺑﻪ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻣﻦ ﺩﺳﺖ ﺯﺩﻩ. ﻣﻦ ﻫﺰﺍﺭ ﺑﺎﺭ ﺑﻬﺶ ﮔﻔﺘﻢ ﻣﻦ ﻧﻴﺴﺘﻢ ﺣﻖ ﻧﺪﺍﺭﯼ ﺑﻪ ﻭﺳﺎﻳﻞ ﻣﻦ ﺩﺳﺖ ﺑﺰﻧﯽ. ﭼﺮﺍ ﻫﻤﻴﺸﻪ ﻳﮏ ﭼﻴﺰﯼ ﺑﺎﻳﺪ ﺑﺎﺷﻪ ﺍﻋﺼﺎﺏ ﺁﺩﻣﻮ ﺧﻮﺭﺩ ﮐﻨﻪ؟

ﻣﺮﺣﻠﻪ ﺳﻮﻡ- ﺍﺣﺴﺎﺳﯽ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﺪ: ﺣﺮﺹ ﻣﯽ‌ﺧﻮﺭﻳﺪ، ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﻮﻳﺪ.

ﻣﺮﺣﻠﻪ ﭼﻬﺎﺭﻡ- ﺩﺭ ﻣﮑﺎﻟﻤﻪ ﺩﺭﻭﻧﯽ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ‌ﺗﺎﻥ ﺗﺠﺪﻳﺪﻧﻈﺮ ﮐﻨﻴﺪ: ﺍﻳﻦ ﺑﭽﻪ ﺍﺻﻼ ﻧﻤﯽ‌ﺩﻭﻧﻪ ﺑﺎ ﺍﻳﻦ‌ ﮐﺎﺭﺵ ﻣﻨﻮ ﺍﻳﻨﻘﺪﺭ ﻧﺎﺭﺍﺣﺖ ﮐﺮﺩﻩ. ﺷﺎﻳﺪ ﻭﺍﻗﻌﺎ ﭼﻨﺪ ﺩﻗﻴﻘﻪ ﻳﻪ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻻﺯﻡ ﺩﺍﺷﺘﻪ. ﺷﺎﻳﺪ ﮐﺎﻣﭙﻴﻮﺗﺮ ﺧﻮﺩﺵ ﺧﺮﺍﺏ ﺑﻮﺩﻩ ﻳﺎ ﺭﻭﺷﻦ ﻧﺸﺪﻩ. ﺥُﺏ ﻣﻤﮑﻨﻪ ﻣﻨﻢ ﻳﻪ ﺭﻭﺯ ﺑﻪ ﮐﺎﻣﭙﻴﻮﺗﺮ ﺍﻭﻥ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻢ ﻭ ﻧﺘﻮﻧﻢ ﺍﺯﺵ ﺍﺟﺎﺯﻩ ﺑﮕﻴﺮﻡ.

ﻣﺮﺣﻠﻪ ﭘﻨﺠﻢ: ﻧﺘﻴﺠﻪ ﺗﻐﻴﻴﺮ ﻣﮑﺎﻟﻤﻪ ﺩﺭﻭﻧﯽ‌ﺗﺎﻥ:

  • ﺑﺎﻻ ﺭﻓﺘﻦ ﺗﺤﻤﻞ ﻭ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﺩﻻﻳﻞ ﻣﺨﺘﻠﻒ ﮐﺎﺭ ﻓﺮﺯﻧﺪﺗﺎﻥ
  • ﻗﻀﻴﻪ ﺑﺮﺍﯼ ﺷﻤﺎ ﺷﺨﺼﯽ ﻧﺸﺪﻩ، ﺩﺧﺘﺮﺗﻮﻥ ﺍﻳﻨﮑﺎﺭ ﺭﺍ ﻧﮑﺮﺩﻩ ﺑﻮﺩ ﺗﺎ ﺷﻤﺎ ﺭﺍ ﺁﺯﺍﺭ ﺩﻫﺪ.
  • ﺩﺭﮎ ﺑﻴﺸﺘﺮ ﻣﺸﮑﻼﺕ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﺍﻓﺮﺍﺩ ﺧﺎﻧﻮﺍﺩﻩ

​ﺷﻤﺎ ﺩﺭ ﭘﻨﺞ ﻣﺮﺣﻠﻪ ﺑﺎﻻ ﺩﺭ ﺣﻘﻴﻘﺖ ﺑﻌﺪ ﺍﺯ ﻣﺮﺣﻠﻪ ﺳﻮﻡ، ﻣﺮﺣﻠﻪ ﭼﻬﺎﺭﻡ ﻭ ﭘﻨﺠﻢ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻳﺪ. ﺍﻧﺘﺨﺎﺑﯽ ﺁﮔﺎﻫﺎﻧﻪ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺗﻐﻴﻴﺮ ﺩﺭ ﺭﻓﺘﺎﺭ.

 

ﭘﺎﺳﺦ ﺑﻪ ﺟﺎﯼ ﻋﮑﺲ ﺍﻟﻌﻤﻞ
ﻧﻈﺮﺗﺎﻥ ﺩﺭ ﻣﻮﺭﺩ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺍﺣﺴﺎﺳﺎﺕ ﭼﻴﺴﺖ؟ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﻭﻗﺘﯽ ﺍﺣﺴﺎﺳﺎﺕ‌ﺗﺎﻥ ﺗﺤﺮﻳﮏ ﻣﯽ‌ﺷﻮﺩ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ ﭼﻄﻮﺭ ﺑﻪ ﺁﻥ ﺟﻮﺍﺏ ﺩﻫﻴﺪ؟

ﺟﻮﺍﺑﺶ ﺍﺻﻼ ﺁﺳﺎﻥ ﻧﻴﺴﺖ. ﻫﻨﻮﺯ ﺩﺍﻧﺸﻤﻨﺪﺍﻥ ﻭ ﻣﺤﻘﻘﺎﻥ ﻫﻢ ﺳﺮ ﺟﻮﺍﺑﺶ ﺑﻪ ﺗﻮﺍﻓﻖ ﻧﺮﺳﻴﺪﻩ‌ﺍﻧﺪ، ﻭﻟﯽ ﺍﮐﺜﺮﺍ ﻣﻌﺘﻘﺪﻧﺪ:

ﻳﮑﯽ ﺍﺯ ﺧﺼﻮﺻﻴﺎﺕ ﻣﻨﺤﺼﺮ ﺑﻪ ﻓﺮﺩ ﻣﺎ ﺁﺩﻡ‌ﻫﺎ ﻧﺴﺒﺖ ﺑﻪ ﺣﻴﻮﺍﻧﺎﺕ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﻢ ﺩﺭ ﭘﺎﺳﺦ ﺑﻪ ﺍﺗﻔﺎﻕ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﺮﺍﻳﻤﺎﻥ ﻣﯽ‌ﺍﻓﺘﺪ، ﭼﻪ ﮐﺎﺭ ﮐﻨﻴﻢ.

ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﻣﺎ ﺩﺭ ﺻﺪ ﺩﺭ ﺻﺪ ﻣﻮﺍﻗﻊ، ﺑﻪ ﺧﺼﻮﺹ ﻭﻗﺖ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﺗﻔﺎﻗﯽ ﺑﺮﺍﯼ ﺍﻭﻟﻴﻦ ﺑﺎﺭ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺭﻓﺘﺎﺭﻣﺎﻥ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﻢ، ﻧﻪ! ﻭﻗﺘﯽ ﺷﻤﺎ ﺗﺼﻤﻴﻢ ﻣﯽ‌ﮔﻴﺮﻳﺪ ﮎِﯼ ﻭ ﮐﺠﺎ، ﭼﻪ ﮐﺎﺭ ﮐﻨﻴﺪ ﻭ ﭼﮕﻮﻧﻪ ﺧﺸﻢ‌ﺗﺎﻥ ﺭﺍ ﻧﺸﺎﻥ ﺩﻫﻴﺪ، ﺁﻥ ﻭﻗﺖ ﻣﻌﻨﯽ ﺁﻥ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺷﺨﺼﺎ ﻣﺴﺌﻮﻝ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺭﻓﺘﺎﺭﺗﺎﻥ ﺑﺎﺷﻴﺪ.

ﺍﻳﻦ ﻳﻌﻨﯽ ﺍﻳﻨﮑﻪ ﺩﻳﮕﺮ ﻧﮕﻮﻳﻴﺪ: «ﺍﺻﻼ ﺩﺳﺖ ﻣﻦ ﻧﺒﻮﺩ، ﺍﻳﻦ ﺑﭽﻪ‌ﻫﺎ ﺍﻋﺼﺎﺏ ﺑﺮﺍﯼ ﺁﺩﻡ ﻧﻤﯽ‌ﮔﺬﺍﺭﻧﺪ!»
