ﺩﺭﺱ ﻫﻔﺘﻢ- ﺟﺎﺩﻭﯼ ﻣﮑﺎﻟﻤﻪ

ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﻫﻢ ﻧﻮﻋﯽ ﺍﺯ ﻣﮑﺎﻟﻤﻪ ﺍﺳﺖ. ﻣﻌﻤﻮﻻ ﻭﻗﺘﯽ ﺁﺩﻡ‌ﻫﺎ ﻋﺼﺒﯽ ﻣﯽ‌ﺷﻮﻧﺪ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻧﺸﺎﻥ ﺩﻫﻨﺪ ﺍﺯ ﺩﺳﺖ ﮐﺴﯽ ﻳﺎ ﺷﺮﺍﻳﻄﯽ ﺑﻪ ﺗﻨﮓ ﺁﻣﺪﻩ‌ﺍﻧﺪ.

ﺧﻴﻠﯽ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺣﺮﻑ ﺭﺍ ﭼﻄﻮﺭ ﻭ ﭼﮕﻮﻧﻪ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺷﻨﻴﺪﻩ ﺷﻮﺩ ﻭ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻣﻨﻈﻮﺭ ﻭ ﻗﺼﺪ ﻣﺎ ﺑﻮﺩﻩ، ﻓﻬﻤﻴﺪﻩ ﺷﻮﺩ. ﺍﻳﻦ ﻫﻤﺎﻥ ﺷﮕﺮﺩ ﻣﮑﺎﻟﻤﻪ ﻭ ﮔﻔﺘﮕﻮﯼ ﻣﻮﺛﺮ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻴﺸﺘﺮ ﺩﺭ ﻣﻮﺭﺩﺵ ﺑﺪﺍﻧﻴﻢ.

ﺍﻟﺒﺘﻪ ﻫﺮ ﮐﺴﯽ ﺭﻭﺵ‌ﻫﺎﯼ ﺷﺨﺼﯽ ﺧﻮﺩﺵ ﺭﺍ ﺑﺮﺍﯼ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩﻥ ﻭ ﺭﺳﺎﻧﺪﻥ ﺣﺮﻑ ﻭ ﮐﻼﻣﺶ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺩﺍﺭﺩ ﮐﻪ ﺍﺯ ﺗﺠﺮﺑﻪ‌ﻫﺎﯼ ﺑﭽﮕﯽ ﻭ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﻗﺒﻼ ﺩﻳﺪﻩ ﻭ ﺷﻨﻴﺪﻩ ﻣﯽ‌ﺁﻳﺪ. ﻭﻟﯽ ﺩﺭ ﻣﻮﺭﺩ ﺑﻴﺸﺘﺮ ﻣﺎ ﺭﻭﺵ ﺧﻮﺩﻣﺎﻥ، ﮐﺎﺭﻣﺎﻥ ﺭﺍ ﺭﺍﻩ ﻧﻤﯽ‌ﺍﻧﺪﺍﺯﺩ. ﻣﺜﻼ ﺍﮔﺮ ﺷﻤﺎ ﺩﺭ ﻣﺤﻴﻄﯽ ﺑﺰﺭﮒ ﺷﺪﻩ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﻤﻪ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﺑﺎ ﻫﻢ ﺣﺮﻑ ﻣﯽ‌ﺯﺩﻧﺪ، ﺑﻠﻨﺪﯼ ﺁﻥ ﺻﺪﺍ ﺑﺮﺍﯼ ﺷﻤﺎ ﻋﺎﺩﯼ ﺍﺳﺖ ﻭﻟﯽ ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﺪ ﺑﺎ ﻫﻤﺴﺮﺗﺎﻥ ﻫﻢ ﻫﻤﺎﻧﻄﻮﺭ ﺻﺤﺒﺖ ﮐﻨﻴﺪ، ﺑﺎ ﺍﻭ ﮐﻪ ﻋﺎﺩﺕ ﻧﺪﺍﺭﺩ، ﻓﮑﺮ ﻣﯽ‌ﮐﻨﺪ ﺷﻤﺎ ﺩﺍﺭﻳﺪ ﺳﺮﺵ ﺩﺍﺩ ﻣﯽ‌ﺯﻧﻴﺪ!

ﺩﺭ ﺍﺑﺘﺪﺍ ﺑﺪ ﻧﻴﺴﺖ ﮐﻪ ﺑﻪ ﺭﻭﺵ‌ﻫﺎﯼ ﻧﺎﺩﺭﺳﺖ ﻣﮑﺎﻟﻤﻪ ﻭ ﺭﺳﺎﻧﺪﻥ ﺣﺮﻑ‌ﻣﺎﻥ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻧﮕﺎﻫﯽ ﺑﻴﺎﻧﺪﺍﺯﻳﻢ.

 

ﺑﯽ‌ﺗﻮﺟﻬﯽ، ﻧﺎﺩﻳﺪﻩ ﮔﺮﻓﺘﻦ، ﮐﻢ ﻣﺤﻠﯽ
ﺍﺯ ﺩﺳﺖ ﺩﺧﺘﺮﺗﺎﻥ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻴﺪ، ﺍﻭ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﺪ ﺑﺎ ﺷﻤﺎ ﺳﺮ ﺣﺮﻑ ﺭﺍ ﺑﺎﺯ ﮐﻨﺪ، ﺷﺎﻳﺪ ﻫﻢ ﺑﺨﻮﺍﻫﺪ ﺗﻮﺿﻴﺤﺎﺗﯽ ﺑﺪﻫﺪ ﺭﺍﺟﻊ ﺑﻪ ﺁﻧﭽﻪ ﭘﻴﺶ ﺁﻣﺪﻩ. ﺷﻤﺎ ﺑﻪ ﺭﻭﯼ ﺧﻮﺩﺗﺎﻥ ﻧﻤﯽ‌ﺁﻭﺭﻳﺪ ﮐﻪ ﺍﺻﻼ ﺍﻭ ﺭﺍ ﻣﯽ‌ﺑﻴﻨﻴﺪ. ﻫﺮ ﭼﻪ ﺍﻭ ﺑﻴﺸﺘﺮ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﺪ، ﺗﻮﺟﻪ ﺷﻤﺎ ﺭﺍ ﺟﻠﺐ ﮐﻨﺪ ﮐﻪ ﺑﺎ ﻫﻢ ﻣﮑﺎﻟﻤﻪ ﺩﻭ ﻃﺮﻓﻪ‌ﺍﯼ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﺪ، ﺷﻤﺎ ﺑﻴﺸﺘﺮ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺑﻪ ﺁﻥ ﺭﺍﻩ ﻣﯽ‌ﺯﻧﻴﺪ.

ﺩﺧﺘﺮﺗﺎﻥ ﻧﻪ ﻣﻨﻈﻮﺭ ﺍﺻﻠﯽ ﺷﻤﺎ ﻧﻤﯽ‌ﻓﻬﻤﺪ، ﻧﻪ ﺩﻟﻴﻞ ﻋﺼﺒﺎﻧﻴﺖ‌ﺗﺎﻥ ﺭﺍ. ﻧﻪ ﺷﻤﺎ ﻣﯽ‌ﻓﻬﻤﻴﺪ ﺍﻭ ﭼﻪ ﻣﯽ‌ﺧﻮﺍﺳﺖ ﺑﻪ ﺷﻤﺎ ﺑﮕﻮﻳﺪ ﻭ ﺍﻭﺿﺎﻉ ﺑﻬﺘﺮ ﮐﻪ ﻧﻤﯽ‌ﺷﻮﺩ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺪﺗﺮ ﻫﻢ ﺑﺸﻮﺩ.

 

ﺗﺤﻘﻴﺮ، ﺳﺮﺯﻧﺶ، ﻧﻘﺪ ﺩﻧﺪﺍﻥ‌ﺷﮑﻦ
ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺧﺎﻧﻪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺭﻓﺘﺎﺭﻫﺎ ﻭ ﻋﺎﺩﺍﺕ ﺩﺧﺘﺮ ﻭ ﭘﺴﺮ ﻧﻮﺟﻮﺍﻥ‌ﺗﺎﻥ ﺭﺍ ﻧﭙﺴﻨﺪﻳﺪ.

ﺁﻧﻬﺎ ﺍﺯ ﻧﻈﺮ ﻓﮑﺮﯼ، ﺟﺴﻤﯽ، ﺟﻨﺴﯽ ﻣﺴﺘﻘﻞ ﻭ ﺑﺎﻟﻎ ﻫﺴﺘﻨﺪ ﻭﻟﯽ ﻫﻨﻮﺯ ﺍﺯ ﻧﻈﺮ ﺍﻗﺘﺼﺎﺩﯼ، ﻣﺤﻞ ﺧﻮﺍﺏ ﻭ ﻏﺬﺍ ﺑﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻭﺍﺑﺴﺘﻪ‌ﺍﻧﺪ. ﺁﻧﻬﺎ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﻣﺴﺘﻘﻞ‌ﺗﺮ ﺑﺎﺷﻨﺪ ﻭ ﺑﺮﺍﻳﺶ ﺗﻼﺵ ﻣﯽ‌ﮐﻨﻨﺪ، ﻣﯽ‌ﺟﻨﮕﻨﺪ. ﮔﺎﻫﯽ ﮐﺎﺭﻫﺎﯼ ﺷﻤﺎ ﻣﺎﻧﻊ ﺍﺳﺘﻘﻼﻝ ﺁﻧﻬﺎ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻳﺎ ﺣﺪﺍﻗﻞ ﺑﺮﺩﺍﺷﺖ ﻧﻮﺟﻮﺍﻥ ﺷﻤﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺑﺎﺷﺪ. ﺧﻮﺏ ﺍﺳﺖ ﺑﻠﺪ ﺑﺎﺷﻴﺪ ﺭﺍﻩ ﺍﻳﻨﮑﻪ ﭼﻄﻮﺭ، ﮎِﯼ ﻭ ﮐﺠﺎ ﺍﻧﺘﻘﺎﺩ ﮐﻨﻴﺪ ﻭ ﻧﺎﺭﺿﺎﻳﺘﯽ‌ﺗﺎﻥ ﺭﺍ ﺑﮕﻮﻳﻴﺪ.

ﺍﮔﺮ ﺳﺮﺯﻧﺶ‌ﺷﺎﻥ ﮐﻨﻴﺪ ﮐﻪ ﻫﻴﭻ ﭼﻴﺰ ﻧﻤﯽ‌ﺩﺍﻧﻨﺪ ﻳﺎ ﺩﻧﻴﺎ ﺭﺍ ﻫﻨﻮﺯ ﻧﻤﯽ‌ﺷﻨﺎﺳﻨﺪ، ﻳﮏ ﺭﻭﺯ ﮐﺎﺭ ﻧﮑﺮﺩﻩ‌ﺍﻧﺪ ﮐﻪ ﺑﺪﺍﻧﻨﺪ ﺟﺎﻣﻌﻪ ﭼﻪ ﺧﺒﺮ ﺍﺳﺖ ﻭ... ﺩﺍﺭﻳﺪ ﮐﻨﺠﮑﺎﻭﯼ ﻭ ﺗﻼﺵ ﺁﻧﻬﺎ ﺑﺮﺍﯼ ﺍﺳﺘﻘﻼﻝ، ﺗﺠﺮﺑﻪ‌ﯼ ﺷﺨﺼﯽ ﺁﻧﻬﺎ ﺍﺯ ﺩﻧﻴﺎﯼ ﺍﻃﺮﺍﻑ‌ﺷﺎﻥ ﺭﺍ ﺗﺤﻘﻴﺮ ﻣﯽ‌ﮐﻨﻴﺪ. ﻫﻴﭽﮑﺲ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﻣﺪﺕ ﻃﻮﻻﻧﯽ ﺗﺤﻘﻴﺮ ﺭﺍ ﺗﺤﻤﻞ ﮐﻨﺪ ﺣﺘﯽ ﺍﺯ ﻃﺮﻑ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﺵ.

ﻧﺘﻴﺠﻪ ﺍﻳﻦ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﮐﻪ ﻧﻪ ﺁﻧﻬﺎ ﺣﺮﻑ ﻭ ﺍﺣﺴﺎﺳﺎﺕ ﺷﻤﺎ ﺭﺍ ﻣﯽ‌ﻓﻬﻤﻨﺪ ﻭ ﻧﻪ ﻧﺎﺭﺿﺎﻳﺘﯽ ﻭ ﺩﻟﻴﻞ ﻋﺼﺒﺎﻧﻴﺖ ﺷﻤﺎ ﺭﺍ. ﺷﻤﺎ ﻫﻢ ﺳﻌﯽ ﻧﮑﺮﺩﻩ‌ﺍﻳﺪ  ﺁﻧﻬﺎ ﺭﺍ ﺩﺭﮎ ﮐﻨﻴﺪ. ﻫﻴﭻ ﻫﻤﺪﻟﯽ‌ﺍﯼ ﺍﺗﻔﺎﻕ ﻧﻤﯽ‌ﺍﻓﺘﺪ ﻭ ﺍﻭﺿﺎﻉ ﺑﺪﺗﺮ ﻫﻢ ﻣﯽ‌ﺷﻮﺩ.

 

ﭘﺮﺧﺎﺵ، ﺩﺍﺩ ﻭ ﻫﻮﺍﺭ
ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﺍﺯ ﻇﺎﻫﺮ ﻧﻮﺟﻮﺍﻥ‌ﺗﺎﻥ ﺭﺍﺿﯽ ﻧﺒﺎﺷﻴﺪ، ﻣﺜﻼ ﻣﺪﻝ ﻣﻮﻫﺎﻳﺶ ﻳﺎ ﺁﺭﺍﻳﺸﺶ ﺭﺍ ﻧﭙﺴﻨﺪﻳﺪ. ﻫﺮ ﺑﺎﺭ ﻫﻢ ﺍﻋﺘﺮﺍﺽ ﻣﯽ‌ﮐﻨﻴﺪ، ﺍﻭ ﻫﻢ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻫﻤﻪ ﭼﻴﺰ ﺑﻪ ﭘﺮﺧﺎﺵ ﻭ ﺩﺍﺩ ﻭ ﻫﻮﺍﺭ ﻣﯽ‌ﮔﺬﺭﺩ. ﺩﺭ ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ ﻧﻪ ﺷﻤﺎ ﺩﻻﻳﻞ ﻧﺎﺭﺿﺎﻳﺘﯽ‌ﺗﺎﻥ ﺭﺍ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﮔﻔﺘﻪ‌ﺍﻳﺪ ﻭ ﻧﻪ ﺣﺮﻑ‌ﻫﺎﯼ ﺍﻭ ﺭﺍ ﺷﻨﻴﺪﻩ‌ﺍﻳﺪ. 

ﺑﻬﺮﺣﺎﻝ ﺷﻤﺎ ﻫﻢ ﺧﻮﺩﺗﺎﻥ ﺩﻭﺭﻩ‌ﯼ ﻧﻮﺟﻮﺍﻧﯽ ﻭ ﺍﺧﺘﻼﻑ ﺳﻠﻴﻘﻪ ﺑﺎ ﻭﺍﻟﺪﻳﻦ ﺭﺍ ﺗﺠﺮﺑﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﻭ ﺣﺘﻤﺎ ﺣﺮﻑ‌ﻫﺎ ﻭ ﺗﺠﺮﺑﻴﺎﺕ ﻣﺘﻔﺎﻭﺗﯽ ﺩﺍﺭﻳﺪ ﮐﻪ ﺑﺎ ﻧﻮﺟﻮﺍﻥ‌ﺗﺎﻥ ﺩﺭ ﻣﻴﺎﻥ ﺑﮕﺬﺍﺭﻳﺪ. ﺍﻭ ﻫﻢ ﻗﻄﻌﺎ ﺣﺮﻑ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺩﺭ ﺍﻳﻦ ﺑﺎﺭﻩ ﺩﺍﺭﺩ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﺑﮕﻮﻳﺪ. ﻭﻟﯽ ﻓﻀﺎﯼ ﺗﻨﺶ ﻭ ﭘﺮﺧﺎﺵ ﺍﻳﻦ ﻓﺮﺻﺖ ﺭﺍ ﺍﺯ ﺷﻤﺎ ﻭ ﻓﺮﺯﻧﺪ ﻧﻮﺟﻮﺍﻥ‌ﺗﺎﻥ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﺷﺎﻳﺪ ﻣﻨﺠﺮ ﺑﻪ ﻓﺎﺻﻠﻪ ﺑﻴﻦ ﺷﻤﺎ ﻭ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻫﻢ ﺑﺸﻮﺩ.

ﺣﺎﻻ ﺩﻳﮕﺮ ﺣﺘﻤﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﺪﺍﻧﻴﺪ ﺭﺍﻩ ﻣﮑﺎﻟﻤﻪ ﺻﺤﻴﺢ ﻭ ﻣﻮﺛﺮ ﭼﻴﺴﺖ؟

ﻣﮑﺎﻟﻤﻪ ﺳﺎﻟﻢ ﻭ  ﺩﺭﺳﺖ ﮐﻪ ﻣﺎ ﺭﺍ ﺑﻪ ﻫﺪﻓﯽ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﺮﺳﺎﻧﺪ ﭼﻨﺪ ﻣﺸﺨﺼﻪ ﺩﺍﺭﺩ:

- ﭘﻴﺎﻡ ﺭﻭﺷﻦ ﻭ ﻭﺍﺿﺢ: ﻧﻤﯽ‌ﺷﻮﺩ ﺩﺭ ﻟﻔﺎﻓﻪ ﻭ ﺑﺎ ﺍﻳﻤﺎ ﻭ ﺍﺷﺎﺭﻩ ﻭ ﻗﻴﺎﻓﻪ ﮔﺮﻓﺘﻦ ﺣﺮﻑ‌ﺗﺎﻥ ﺭﺍ ﺑﻪ ﺑﻘﻴﻪ ﺑﮕﻮﻳﻴﺪ ﻭ ﺗﻮﻗﻊ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺁﻧﻬﺎ ﺍﺯ ﺭﻓﺘﺎﺭﺗﺎﻥ ﻫﻤﺎﻥ ﺑﺮﺩﺍﺷﺘﯽ ﺭﺍ ﺑﮑﻨﻨﺪ ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ، ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﮐﻼﻡ ﻭﺍﺿﺢ ﻭ ﺭﻭﺷﻦ ﻭ ﮐﺎﺭﻫﺎﻳﯽ ﮐﻪ ﻣﯽ‌ﮐﻨﻴﺪ ﻫﻤﺎﻥ ﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﻭﺍﻗﻌﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ، ﺑﻪ ﺑﻘﻴﻪ ﺑﮕﻮﻳﻴﺪ. ﻓﻘﻂ ﮐﻼﻡ ﻧﻴﺴﺖ، ﺣﺮﮐﺎﺕ‌ﺗﺎﻥ ﻫﻢ ﺩﺭ ﻃﻮﻝ ﻣﮑﺎﻟﻤﻪ ﻣﻬﻢ ﺍﺳﺖ.
