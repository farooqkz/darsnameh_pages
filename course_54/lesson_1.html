ﻣﻘﺪﻣﻪ

ﺍﻳﻦ ﺩﻭﺭﻩ ﺁﻣﻮﺯﺷﯽ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﺪ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻧﺎﻇﺮ ﺑﻴﺮﻭﻧﯽ ﺑﻪ ﻣﺎ ﮐﻤﮏ ﮐﻨﺪ، ﺁﻧﭽﻪ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﻧﺤﻮﻩ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﺍﻳﻦ ﻣﻮﻗﻌﻴﺖ‌ﻫﺎ، ﺗﺎ ﺑﻪ ﺣﺎﻝ ﻧﺪﻳﺪﻩ‌ﺍﻳﻢ ﻭ ﺧﻮﺍﺳﺘﻪ ﻳﺎ ﻧﺎﺧﻮﺍﺳﺘﻪ ﺑﻪ ﺁﻥ ﺑﯽ‌ﺗﻮﺟﻪ ﺑﻮﺩﻩ‌ﺍﻳﻢ، ﺑﻪ ﻣﺎ ﻧﺸﺎﻥ ﺩﻫﺪ. ﺍﻳﻦ ﺁﮔﺎﻫﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﻤﮑﯽ ﺑﺎﺷﺪ ﺑﺮﺍﯼ ﺗﻐﻴﻴﺮ ﺩﺭ ﺭﻓﺘﺎﺭ ﻭ ﻋﺎﺩﺕ‌ﻫﺎﯼ ﻣﺎ ﺩﺭ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﻋﺼﺒﯽ. 

ﺷﺎﻳﺪ ﺍﻭﻟﻴﻦ ﺳﻮﺍﻝ ﺷﻤﺎ ﺍﻳﻦ ﺑﺎﺷﺪ ﮐﻪ ﺩﺭ ﺑﻴﻦ ﺍﻳﻦ ﻫﻤﻪ ﮔﺮﻓﺘﺎﺭﯼ ﻭ ﻣﺸﻐﻠﻪ ﺭﻭﺯﻣﺮﻩ ﭼﺮﺍ ﺑﺎﻳﺪ ﻭﻗﺖ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺭﺍ ﺗﻤﺎﻡ ﮐﻨﻴﺪ؟

ﻭ ﺑﻌﺪ ﻫﻢ ﺳﻮﺍﻻﺕ ﺩﻳﮕﺮ ﻣﺜﻞ:

ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ ﺩﺭ ﺯﻧﺪﮔﯽ ﺷﻤﺎ ﭼﻪ ﺗﻐﻴﻴﺮﯼ ﺍﻳﺠﺎﺩ ﺧﻮﺍﻫﺪ ﮐﺮﺩ؟

ﻧﮑﻨﺪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺍﻳﻦ ﻫﻢ ﺑﺎﺭﯼ ﺷﻮﺩ ﺑﺮ ﺩﻭﺵ ﺷﻤﺎ؟ ﻭﻇﻴﻔﻪ ﺩﻳﮕﺮﯼ؟

ﻣﻦ ﮐﻪ ﺍﺻﻼ ﺁﺩﻡ ﻋﺼﺒﯽ ﻧﻴﺴﺘﻢ! ﻓﮑﺮ ﻧﮑﻨﻢ ﺑﻪ ﺩﺭﺩ ﻣﻦ ﺑﺨﻮﺭﺩ!

ﮔﺬﺭﺍﻧﺪﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﻣﻌﻨﯽ‌ﺍﺵ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﻦ ﺁﺩﻡ ﻋﺼﺒﯽ ﻫﺴﺘﻢ؟

ﻫﻤﻪ ﺍﻳﻦ ﺳﻮﺍﻻﺕ ﻃﺒﻴﻌﯽ ﻭ ﻗﺎﺑﻞ ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺍﺳﺖ. «ﺍﻧﺪﮐﯽ ﺻﺒﺮ» ﻻﺯﻡ ﺍﺳﺖ، ﺗﺎ ﺁﺧﺮ ﺍﻳﻦ ﻣﻘﺪﻣﻪ ﺭﺍ ﺑﺨﻮﺍﻧﻴﺪ.

ﺧﺸﻢ، ﻳﮏ ﺣﺲ ﻃﺒﻴﻌﯽ ﺩﺭ ﺍﻧﺴﺎﻥ‌ﻫﺎﺳﺖ. ﻫﻴﭻ ﺍﻧﺴﺎﻧﯽ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﺍﺩﻋﺎ ﮐﻨﺪ ﮐﻪ ﺣﺘﯽ ﺑﺮﺍﯼ ﻳﮏ ﺑﺎﺭ ﻫﻢ ﻋﺼﺒﺎﻧﯽ ﻧﺸﺪﻩ ﻭ ﺍﺯ ﺧﻮﺩ ﺭﻓﺘﺎﺭﻫﺎﯼ ﻋﺼﺒﯽ ﻭ ﺧﺸﻦ ﻧﺸﺎﻥ ﻧﺪﺍﺩﻩ ﺍﺳﺖ. ﭘﺲ ﻫﻤﻪ ﻣﺎ ﺩﺭ ﺩﺍﺷﺘﻦ ﺍﻳﻦ ﺣﺲ ﻣﺸﺘﺮﮐﻴﻢ. ﺗﻔﺎﻭﺕ ﺩﺭ ﻧﺤﻮﻩ ﺑﺮﺧﻮﺭﺩ ﻭ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺭﻓﺘﺎﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﻣﺎ ﺩﺭ ﺣﺎﻟﺖ ﻋﺼﺒﺎﻧﻴﺖ ﺍﺯ ﺧﻮﺩ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻴﻢ.

ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﺎ ﻫﻢ ﺧﻮﺍﻫﻴﻢ ﺩﻳﺪ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺭﻓﺘﺎﺭﻫﺎﯼ ﻧﺎﺷﯽ ﺍﺯ ﺍﻳﻦ ﺣﺲ ﻃﺒﻴﻌﯽ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﻢ ﻭ ﻣﺪﻳﺮ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﺑﺎﺷﻴﻢ.

ﺑﻪ ﻳﺎﺩ ﺁﻭﺭﺩﻥ ﺁﺧﺮﻳﻦ ﺑﺎﺭﯼ ﮐﻪ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻩ‌ﺍﻳﻢ ﻧﺒﺎﻳﺪ ﺳﺨﺖ ﺑﺎﺷﺪ. ﺑﻪ ﻳﺎﺩ ﺁﻭﺭﺩﻥ ﺁﺧﺮﻳﻦ ﺑﺎﺭﯼ ﮐﻪ ﺩﺭ ﺣﺎﻟﺖ ﻋﺼﺒﺎﻧﻴﺖ ﺣﺮﻓﯽ ﺯﺩﻩ‌ﺍﻳﻢ، ﮐﺎﺭﯼ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﮐﻪ ﺑﻌﺪ ﺍﺯ ﻓﺮﻭﮐﺶ ﮐﺮﺩﻥ ﺧﺸﻢ‌ﻣﺎﻥ ﮐﻠﯽ ﭘﺸﻴﻤﺎﻥ ﺷﺪﻩ‌ﺍﻳﻢ، ﻫﻢ ﻧﺒﺎﻳﺪ ﮐﺎﺭ ﺳﺨﺘﯽ ﺑﺎﺷﺪ. ﻣﻮﺍﺭﺩﯼ ﮐﻪ ﺣﺘﯽ ﺧﻮﺩﻣﺎﻥ ﻫﻢ ﺑﻪ ﺳﺨﺘﯽ ﺑﺎﻭﺭ ﻣﯽ‌ﮐﻨﻴﻢ؛ ﺍﻳﻦ ﻣﺎ ﺑﻮﺩﻩ‌ﺍﻳﻢ ﮐﻪ ﻓﻼﻥ ﮐﺎﺭ ﺭﺍ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﻳﺎ ﺑﻬﻤﺎﻥ ﺣﺮﻑ ﺭﺍ ﺯﺩﻩ‌ﺍﻳﻢ. ﺍﻧﮕﺎﺭ ﺁﺩﻡ ﺩﻳﮕﺮﯼ ﻣﯽ‌ﺷﻮﻳﻢ ﺩﺭ ﻋﺼﺒﺎﻧﻴﺖ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﺩﺳﺖ ﺧﻮﺩﻡ ﻧﺒﻮﺩ، ﻧﻤﯽ‌ﺩﻭﻧﻢ ﻳﮏ ﺩﻓﻌﻪ ﭼﻪ ﺷﺪ!

 ﺍﮔﺮ ﺍﻳﻦ ﺣﺮﻑ‌ﻫﺎ ﺭﺍ ﻗﺒﻮﻝ ﺩﺍﺭﻳﺪ، ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻢ ﺩﻟﻴﻞ ﮐﺎﻓﯽ ﺑﺮﺍﯼ ﺍﺩﺍﻣﻪ ﺍﻳﻦ ﺩﻭﺭﻩ ﺩﺍﺭﻳﺪ.

ﻫﺮ ﺭﻭﺯ ﺩﺭ ﺍﻃﺮﺍﻑ ﻣﺎ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﻋﺼﺒﺎﻧﻴﺖ ﻣﺎ ﺭﺍ ﺗﺤﺮﻳﮏ ﮐﻨﻨﺪ، ﺍﺯ ﻟﺤﻈﻪ ﺑﻴﺪﺍﺭ ﺷﺪﻥ ﺍﺯ ﺧﻮﺍﺏ ﺗﺎ ﺯﻣﺎﻥ ﺑﻪ ﺧﻮﺍﺏ ﺭﻓﺘﻦ. ﺩﺭ ﺧﺎﻧﻮﺍﺩﻩ، ﺩﺭ ﻣﺤﻠﻪ، ﻣﺤﻴﻂ ﮐﺎﺭ، ﺩﺭ ﻣﺴﻴﺮ ﺭﺳﻴﺪﻥ ﺑﻪ ﺳﺮﮐﺎﺭ، ﺩﺭ ﻃﻮﻝ ﺭﺍﻧﻨﺪﮔﯽ ﻳﺎ ﺳﻔﺮ ﺑﺎ ﻭﺳﻴﻠﻪ ﻧﻘﻠﻴﻪ ﻋﻤﻮﻣﯽ، ﺧﻼﺻﻪ ﻫﻤﻪ ﺟﺎ.

ﮐﻨﺘﺮﻝ ﺍﻳﻦ ﻣﺤﺮﮎ‌ﻫﺎ ﻏﻴﺮ ﻣﻤﮑﻦ ﺍﺳﺖ. ﻭﻟﯽ ﮐﻨﺘﺮﻝ ﺭﻓﺘﺎﺭ ﻭ ﭘﺎﺳﺦ ﻣﺎ ﺑﻪ ﺍﻳﻦ ﻣﺤﺮﮎ‌ﻫﺎ ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ ﺩﺭﺑﺎﺭﻩ ﺁﻥ ﺣﺮﻑ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺩﺍﺭﺩ. ﺩﻭﺭﻩ ﻣﺪﻳﺮﻳﺖ ﺑﻪ ﻣﺎ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﭼﻄﻮﺭ ﺍﻧﺴﺎﻥ ﺑﻪ ﻋﻨﻮﺍﻥ ﺗﻨﻬﺎ ﻣﻮﺟﻮﺩ ﻣﺘﻔﮑﺮ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮ ﺭﻓﺘﺎﺭ ﻭ ﭘﺎﺳﺦ‌ﻫﺎﯼ ﺧﻮﺩ ﺑﻪ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﻋﺼﺒﯽ ﻣﺨﺘﺎﺭ ﻭ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﻧﺪﻩ ﺑﺎﺷﺪ.

ﻓﮑﺮ ﮐﻨﻢ ﺗﺎ ﺍﻳﻨﺠﺎ ﺳﺮ ﺩﻟﻴﻞ ﮔﺬﺭﺍﻧﺪﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﻗﺎﻧﻊ ﺷﺪﻩ ﺑﺎﺷﻴﺪ. ﻭﻟﯽ ﺍﮔﺮ ﻫﻨﻮﺯ ﻣﻄﻤﺌﻦ ﻧﻴﺴﺘﻴﺪ، ﺑﺎﺯ ﻫﻢ ﺩﻻﻳﻞ ﺩﻳﮕﺮﯼ ﻫﺴﺖ.

ﺷﻤﺎ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻫﺴﺘﻴﺪ. ﺍﻳﻦ ﻭﻇﻴﻔﻪ ﺑﻪ ﺧﻮﺩﯼ ﺧﻮﺩ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﻋﺼﺒﯽ ﺷﻤﺎ ﺭﺍ ﭼﻨﺪ ﺑﺮﺍﺑﺮ ﻣﯽ‌ﮐﻨﺪ. ﺷﻤﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﻪ ﺧﻮﺑﯽ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﻫﺮ ﺭﻓﺘﺎﺭ ﻭ ﭘﺎﺳﺦ ﺷﻤﺎ ﺩﺭ ﺣﺎﻟﺖ ﻋﺼﺒﺎﻧﻴﺖ ﭼﻄﻮﺭ ﺑﺮ ﻓﺮﺯﻧﺪ ﺷﻤﺎ، ﺁﻳﻨﺪﻩ ﺍﻭ ﻭ ﺭﻭﺍﺑﻂ ﺷﻤﺎ ﺑﺎ ﺍﻭ ﺗﺎﺛﻴﺮ ﻣﯽ‌ﮔﺬﺍﺭﺩ.

ﺍﮔﻪ ﺑﭽﻪ ﮐﻮﭼﮏ‌ﺗﺮ ﺩﺍﺭﻳﺪ، ﺷﻤﺎ ﺩﻧﻴﺎﯼ ﺍﻭ ﻫﺴﺘﻴﺪ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺭﻓﺘﺎﺭﻫﺎ ﻭ ﭘﺎﺳﺦ‌ﻫﺎﯼ ﮐﻨﺘﺮﻝ ﻧﺸﺪﻩ ﺷﻤﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺣﺲ ﺍﻣﻨﻴﺖ ﻭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺍﻭ ﺭﺍ ﺍﺯ ﺍﻭ ﺑﮕﻴﺮﺩ، ﺍﮔﺮ ﻧﻮﺟﻮﺍﻥ ﺩﺍﺭﻳﺪ، ﻓﺎﺻﻠﻪ ﺷﻤﺎ ﺭﺍ ﺑﺎ ﺍﻭ ﺯﻳﺎﺩ ﮐﻨﺪ ﻭ ﺑﻪ ﺍﻭ ﺍﻳﻦ ﺣﺲ ﺭﺍ ﺑﺪﻫﺪ ﮐﻪ ﺍﻭ ﺭﺍ ﺩﻭﺳﺖ ﻧﺪﺍﺭﻳﺪ. ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﻢ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺍﻧﻤﺎﻥ ﺁﺳﺎﻥ ﻧﻴﺴﺖ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺭﻓﺘﺎﺭﻫﺎﯼ ﻋﺼﺒﯽ ﻣﺎ ﺭﺍ ﺷﺨﺼﯽ ﻧﮕﻴﺮﻧﺪ. ﺁﻧﻬﺎ ﺧﻮﺩ ﺭﺍ ﻣﺨﺎﻃﺒﺎﻥ ﺍﺻﻠﯽ ﺭﻓﺘﺎﺭ ﺷﻤﺎ ﻣﯽ‌ﺩﺍﻧﻨﺪ ﻭ ﺍﻭﻟﻴﻦ ﻓﮑﺮﯼ ﻣﯽ‌ﮐﻨﻨﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﭼﻮﻥ ﺩﻳﮕﺮ ﺁﻧﻬﺎ ﺭﺍ ﻧﻤﯽ‌ﺧﻮﺍﻫﻴﺪ ﻳﺎ ﺩﻭﺳﺖ ﻧﺪﺍﺭﻳﺪ ﺍﻳﻨﮕﻮﻧﻪ ﺭﻓﺘﺎﺭ ﻣﯽ‌ﮐﻨﻴﺪ.

ﻳﺎﺩﻣﺎﻥ ﻧﺮﻭﺩ:

  • ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺧﺸﻢ ﺣﺲ‌ﻫﺎﯼ ﻃﺒﻴﻌﯽ ﻫﺴﺘﻨﺪ، ﻳﻌﻨﯽ ﻫﻤﻪ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﻮﻧﺪ.
  • ﺍﮔﻪ ﻗﺮﺍﺭ ﺑﺎﺷﺪ ﺣﻴﻦ ﻋﺼﺒﺎﻧﻴﺖ ﮐﻨﺘﺮﻟﯽ ﺭﻭﯼ ﺧﻮﺩﻣﺎﻥ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ، ﺑﺎﻳﺪ ﻳﮏ ﻓﮑﺮﯼ ﺑﺮﺍﯼ ﮐﻨﺘﺮﻟﺶ ﺑﮑﻨﻴﻢ.
  • ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ، ﻓﺸﺎﺭ ﻋﺼﺒﯽ ﺭﻭﯼ ﺁﺩﻡ ﺭﺍ ﮐﻢ ﻣﯽ‌ﮐﻨﺪ، ﺑﺮﺍﯼ ﺳﻼﻣﺘﯽ ﺟﺴﻢ ﻭ ﺭﻭﺍﻥ ﻫﻢ ﻣﻔﻴﺪ ﺍﺳﺖ.
  • ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﭘﺎﺳﺦ‌ﻫﺎﯼ ﻋﺼﺒﯽ ﻭ ﺧﺸﻦ، ﺑﺮﺍﯼ ﺣﺮﻳﻢ ﺍﻣﻦ ﺧﺎﻧﻮﺍﺩﻩ ﻭ ﻓﺮﺯﻧﺪﺍﻥ ﻣﺎ ﻣﻀﺮ ﺍﺳﺖ.
  • ﮔﺎﻫﯽ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﺑﻬﺎ ﻭ ﺗﺎﻭﺍﻧﯽ ﺩﺍﺭﺩ ﮐﻪ ﺍﺻﻼ ﻧﻤﯽ‌ﺍﺭﺯﺩ. ﮔﺎﻫﯽ ﻫﻢ ﻧﺘﺎﻳﺞ‌ﺍﺵ ﻏﻴﺮﻗﺎﺑﻞ ﺟﺒﺮﺍﻥ ﺍﺳﺖ.

 

ﭼﺮﺍ ﮐﻨﺘﺮﻝ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﺳﺎﺩﻩ ﻧﻴﺴﺖ؟
ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﺍﺻﻼ ﺳﺎﺩﻩ ﻧﻴﺴﺖ. ﺩﻻﻳﻞ ﺯﻳﺎﺩﯼ ﻫﻢ ﺩﺍﺭﺩ، ﺁﺩﻡ ﺑﻪ ﺁﺩﻡ ﻫﻢ ﻓﺮﻕ ﻣﯽ ﮐﻨﺪ ﻭ ﺷﺎﻳﺪ ﺑﺮﺍﯼ ﺑﻌﻀﯽ ﻫﻢ  ﺗﺮﮐﻴﺐ ﭼﻨﺪ ﺗﺎ ﺩﻟﻴﻞ ﺑﺎﺷﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﺴﯽ ﺍﺯ ﮐﻮﺩﮐﯽ ﻳﺎﺩ ﮔﺮﻓﺘﻪ ﺑﺎﺷﺪ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺧﺸﻤﺶ ﺭﺍ ﻧﺸﺎﻥ ﻧﺪﻫﺪ. ﺍﮔﺮ ﮐﺴﯽ ﺩﺭ ﮐﻮﺩﮐﯽ، ﺩﺭ ﺧﺎﻧﻪ، ﺩﺭ ﻣﺪﺭﺳﻪ، ﺩﺭ ﻣﺤﻴﻂ ﺍﻃﺮﺍﻑ ﺧﻮﺩ، ﺭﻓﺘﺎﺭﻫﺎﯼ ﺧﺸﻦ ﻭ ﻋﺼﺒﯽ ﺯﻳﺎﺩ ﺩﻳﺪﻩ ﺑﺎﺷﺪ، ﮐﺎﺭ ﺑﺮﺍﯼ ﺍﻭ ﺳﺨﺖ‌ﺗﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻋﺎﺩﺕ ﺑﻪ ﺩﻳﺪﻥ ﺁﺩﻡ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﺪ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﺪﻧﺪ ﻭ ﺧﺸﻮﻧﺖ ﺑﻪ ﺧﺮﺝ ﻣﯽ‌ﺩﺍﺩﻧﺪ  ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺁﺩﻡ ﻣﺘﻮﺟﻪ ﻧﺸﻮﺩ ﺭﻓﺘﺎﺭﺵ ﮐﺠﺎ ﺩﺍﺭﺩ ﺍﺯ ﮐﻨﺘﺮﻝ ﺧﺎﺭﺝ ﻣﯽ‌ﺷﻮﺩ، ﭼﻮﻥ ﺑﻪ ﺩﻳﺪﻥ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ‌ﻫﺎﯼ ﺍﻳﻦ ﭼﻨﻴﻨﯽ ﻋﺎﺩﺕ ﮐﺮﺩﻩ ﺍﺳﺖ. ﺍﺯ ﻫﻤﻪ ﻣﻬﻤﺘﺮ ﻫﻴﭻ ﻭﻗﺖ ﻫﻢ ﻳﺎﺩ ﻧﮕﺮﻓﺘﻪ ﭼﮕﻮﻧﻪ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺕ ﺩﺭﻭﻧﯽ ﺧﻮﺩ ﮐﻨﺎﺭ ﺑﻴﺎﻳﺪ. ﺗﻤﺎﻡ ﺍﻳﻦ ﻣﺴﺎﺋﻞ ﺭﺍ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ ﺑﻪ ﻭﻇﻴﻔﻪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﯼ ﮐﻪ ﻫﻢ ﺩﺭ ﺍﺭﺗﺒﺎﻁ ﺑﺎ ﻓﺮﺯﻧﺪﺍﻥ ﻭ ﻫﻢ ﺩﺭ ﻣﻘﺎﺑﻞ ﻓﺮﺯﻧﺪﺍﻥ، ﭼﻘﺪﺭ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ ﻣﻬﻢ ﻭ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﻣﺸﮑﻞ ﺑﺎﺷﺪ.

ﻭﻟﯽ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﻪ ﺁﺭﺍﻣﯽ ﺑﺎ ﻫﻢ ﻣﺤﺮﮎ‌ﻫﺎﯼ ﻋﺼﺒﯽ ﺭﺍ ﻣﯽ‌ﺷﻨﺎﺳﻴﻢ  ﻭ ﻣﯽ‌ﺑﻴﻨﻴﻢ ﭼﮕﻮﻧﻪ  ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺧﺘﻴﺎﺭ ﻭ ﻣﺪﻳﺮﻳﺖ ﺧﺸﻢ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺩﺳﺘﺎﻥ ﺧﻮﺩﻣﺎﻥ ﺑﮕﻴﺮﻳﻢ. ﺻﺮﻑ ﻧﻈﺮ ﺍﺯ ﺍﻳﻨﮑﻪ ﭼﺮﺍ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻩ‌ﺍﻳﻢ ﻭ ﭼﻘﺪﺭ ﺑﺮﺍﯼ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻥ ﺩﻟﻴﻞ ﺩﺍﺷﺘﻪ‌ﺍﻳﻢ.

 ﺁﺧﺮﻳﻦ ﻧﮑﺘﻪ ﻣﻬﻢ ﮐﻪ ﻣﻘﺪﻣﻪ ﺭﺍ ﺑﺎ ﺁﻥ ﺑﻪ ﭘﺎﻳﺎﻥ ﻣﯽ‌ﺭﺳﺎﻧﻴﻢ ﺍﻳﻨﮑﻪ، ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﺑﻪ ﺷﻤﺎ ﻧﺨﻮﺍﻫﺪ ﮔﻔﺖ ﻋﺼﺒﺎﻧﯽ ﻧﺸﻮﻳﺪ، ﺑﻠﮑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺭﺍ ﺫﮐﺮ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﭼﮕﻮﻧﻪ ﮐﻨﺘﺮﻝ ﺧﺸﻢ ﺩﺭ ﺩﺳﺖ ﺷﻤﺎ ﺑﺎﺷﺪ ﻧﻪ ﮐﻨﺘﺮﻝ ﺷﻤﺎ ﺩﺭ ﺩﺳﺖ ﺧﺸﻢ. ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻦ ﻫﻤﺎﻧﮕﻮﻧﻪ ﮐﻪ ﺑﺎ ﺧﻮﺍﻧﺪﻥ ﻳﮏ ﮐﺘﺎﺏ ﻣﻬﻨﺪﺳﯽ، ﻣﻬﻨﺪﺱ ﻧﻤﯽ‌ﺷﻮﻳﺪ، ﺑﺎ ﭘﺎﻳﺎﻥ ﺭﺳﺎﻧﺪﻥ ﺍﻳﻦ ﺩﻭﺭﻩ‌ﯼ ﺁﻣﻮﺯﺷﯽ ﻧﻴﺰ ﺷﻤﺎ ﻳﮏ ﺭﻭﺍﻧﺸﻨﺎﺱ ﻧﺨﻮﺍﻫﻴﺪ ﺷﺪ.
