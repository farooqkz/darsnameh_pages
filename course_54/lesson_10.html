ﺗﻮﻗﻌﺎﺕ ﺧﻮﺩ ﺭﺍ ﺑﺸﻨﺎﺳﻴﻢ

ﺧﻴﻠﯽ ﻭﻗﺖ‌ﻫﺎ ﺩﻟﻴﻞ ﻋﺼﺒﺎﻧﻴﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺫﻫﻦ‌ﺗﺎﻥ، ﺭﻓﺘﺎﺭ ﺑﻘﻴﻪ ﺭﺍ ﺑﺎ ﺁﻧﭽﻪ ﺷﻤﺎ ﺍﺯ ﺍﻳﺸﺎﻥ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺭﻳﺪ ﻣﻘﺎﻳﺴﻪ ﻣﯽ‌ﮐﻨﻴﺪ. ﮐﻤﺘﺮ ﺑﻪ ﺍﻳﻦ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﺷﺎﻳﺪ ﺍﻧﺘﻈﺎﺭ ﻣﻦ ﺍﺯ ﺁﻧﻬﺎ ﺑﺎ ﻭﺍﻗﻌﻴﺖ ﻫﻤﺨﻮﺍﻧﯽ ﻧﺪﺍﺷﺘﻪ ﻭ ﮔﺎﻫﯽ ﺧﻴﻠﯽ ﺑﻴﺸﺘﺮ ﻭ ﮔﺎﻫﯽ ﺧﻴﻠﯽ ﮐﻤﺘﺮ ﺑﺎﺷﺪ.

ﺑﻪ ﺯﺑﺎﻥ ﺩﻳﮕﺮ، ﭼﻴﺰﯼ ﮐﻪ ﻋﺼﺒﺎﻧﻴﺖ ﺷﻤﺎ ﺭﺍ ﺗﺤﺮﻳﮏ ﻣﯽ‌ﮐﻨﺪ ﻓﺎﺻﻠﻪ ﺁﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﺗﻮﻗﻊ ﺩﺍﺷﺘﻪ‌ﺍﻳﺪ ﭘﻴﺶ ﺑﻴﺎﻳﺪ ﻭ ﺁﻥ ﭼﻴﺰﯼ ﮐﻪ ﺩﺭ ﺩﻧﻴﺎﯼ ﻭﺍﻗﻌﯽ ﭘﻴﺶ ﺁﻣﺪﻩ ﺍﺳﺖ.

ﺍﻟﺒﺘﻪ ﻣﻌﻨﯽ‌ﺍﺵ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﻣﺎ ﻧﺒﺎﻳﺪ ﺍﺯ ﺩﻳﮕﺮﺍﻥ ﺗﻮﻗﻌﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﭼﻮﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺮﺁﻭﺭﺩﻩ ﻧﺸﻮﺩ ﻭ ﻣﺎ ﻋﺼﺒﺎﻧﯽ ﻭ ﺩﻟﺴﺮﺩ ﺑﺸﻮﻳﻢ. ﻧﻪ!

ﺑﺮﻋﮑﺲ ﺍﻳﻦ ﺧﻴﻠﯽ ﻫﻢ ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﻫﺮ ﮐﺲ ﺑﺎ ﺯﻳﺎﺩﺗﺮ ﺷﺪﻥ ﺳﻨﺶ ﺗﻮﻗﻌﺎﺕ ﻣﺘﻔﺎﻭﺕ ﻭ ﺷﺎﻳﺪ ﺑﻴﺸﺘﺮﯼ ﺍﺯ ﺧﻮﺩﺵ ﻭ ﺩﻳﮕﺮﺍﻥ ﭘﻴﺪﺍ ﮐﻨﺪ.

ﻳﮑﯽ ﺍﺯ ﺷﮕﺮﺩﻫﺎﻳﯽ ﮐﻪ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﺧﺸﻢ ﺷﻤﺎ ﺩﺭ ﮐﻨﺘﺮﻝ ﺷﻤﺎ ﺑﺎﺷﺪ، ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺭﺍﻫﯽ ﭘﻴﺪﺍ ﮐﻨﻴﺪ ﮐﻪ ﺳﻄﺢ ﺗﻮﻗﻊ ﻭ ﻧﻮﻉ ﺗﻮﻗﻊ‌ﺗﺎﻥ ﺍﺯ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺑﻪ ﻭﺍﻗﻌﻴﺖ ﻧﺰﺩﻳﮏ ﮐﻨﺪ.

ﻣﺜﻼ ﺷﻤﺎ ﺍﺯ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺭﻳﺪ ﺳﺮ ﻭ ﻭﺿﻊ ﻣﺮﺗﺐ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭﻟﯽ ﺑﺎ ﻫﻤﺎﻥ ﻣﺪﻝ‌ﻫﺎﯼ ﺯﻣﺎﻥ ﻧﻮﺟﻮﺍﻧﯽ ﺷﻤﺎ! ﺧﻮﺏ ﻓﺮﺯﻧﺪ ﺷﻤﺎ ﻣﺮﺗﺐ ﺍﺳﺖ ﻭ ﺑﻪ ﺳﺮ ﻭ  ﻭﺿﻌﺶ ﻣﯽ‌ﺭﺳﺪ ﻭﻟﯽ ﺑﺎ ﺍﺳﺘﺎﻧﺪﺍﺭﺩ ﻧﻮﺟﻮﺍﻥ ﺍﻣﺮﻭﺯﯼ ﻭ ﻣﺪ ﺭﻭﺯ!

ﻗﺪﻡ ﺍﻭﻝ: ﺁﻳﺎ ﺍﻧﺘﻈﺎﺭﺍﺕ ﺷﻤﺎ ﻣﻨﻄﻘﯽ ﻭ ﻣﻨﺼﻔﺎﻧﻪ ﺍﺳﺖ؟
ﺩﺭ ﺍﻳﻦ ﻗﺪﻡ ﺍﻭﻝ ﺑﺎﻳﺪ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺑﭙﺮﺳﻴﺪ ﺁﻳﺎ ﺗﻮﻗﻌﺎﺕ‌ﺗﺎﻥ ﺍﺯ ﺑﻘﻴﻪ ﻣﻨﻄﻘﯽ ﺍﺳﺖ، ﺑﺎ ﻭﺍﻗﻌﻴﺖ ﺟﻮﺭ ﺩﺭ ﻣﯽ‌ﺁﻳﺪ؟ ﻣﻨﺼﻔﺎﻧﻪ ﺍﺳﺖ؟ ﮐﻤﯽ ﺳﺨﺖ ﺍﺳﺖ ﭼﻮﻥ ﻫﺮ ﮐﺴﯽ ﺍﺯ ﺍﻳﻦ ﻣﻔﺎﻫﻴﻢ ﻣﻤﮑﻦ ﺍﺳﺖ ﻳﮏ ﺗﻌﺮﻳﻔﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﻭﻟﯽ ﻻﺯﻡ ﺍﺳﺖ ﺩﺭ ﺁﺭﺍﻣﺶ ﻭ ﻭﻗﺘﯽ ﻓﮑﺮﺗﺎﻥ ﺁﺯﺍﺩ ﺍﺳﺖ ﺑﻨﺸﻴﻨﻴﺪ ﺑﺎ ﺧﻮﺩﺗﺎﻥ ﺧﻠﻮﺕ ﮐﻨﻴﺪ، ﺑﺎ ﺧﻮﺩﺗﺎﻥ ﺭﻭ ﺭﺍﺳﺖ ﺑﺎﺷﻴﺪ ﻭ ﺑﺒﻴﻨﻴﺪ ﺗﻮﻗﻌﺎﺕ‌ﺗﺎﻥ ﺍﺯ ﺑﻘﻴﻪ ﺗﺎ ﭼﻪ ﺍﻧﺪﺍﺯﻩ ﻣﻨﻄﻘﯽ، ﻣﻨﺼﻔﺎﻧﻪ ﻭ ﻭﺍﻗﻌﯽ ﺍﺳﺖ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺧﺎﻧﻮﺍﺩﻩ ﻭ ﺩﻭﺳﺘﺎﻥ‌ﺗﺎﻥ ﻫﻢ ﮐﻤﮏ ﺑﮕﻴﺮﻳﺪ.

ﻗﺪﻡ ﺩﻭﻡ: ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﮐﺴﯽ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﺪ.
ﺍﻳﻦ ﻳﮏ ﻭﺍﻗﻌﻴﺖ ﺍﺳﺖ ﮐﻪ ﻣﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﻢ ﮐﺴﯽ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﻢ. ﻫﺮ ﮐﺴﯽ ﺑﺎ ﻣﻨﻄﻖ ﻭ ﺩﻻﻳﻞ ﺧﻮﺩﺵ ﺭﻓﺘﺎﺭ ﻣﯽ‌ﮐﻨﺪ. ﻫﻴﭻ ﻭﻗﺖ ﻓﮑﺮ ﻧﮑﻨﻴﺪ ﺁﺩﻡ‌ﻫﺎ ﺑﺎﻳﺪ «ﺍﻳﻨﺠﻮﺭﯼ» ﺭﻓﺘﺎﺭ ﮐﻨﻨﺪ، «ﺩﺭﺳﺖ» ﺭﻓﺘﺎﺭ ﮐﻨﻨﺪ ﻭ ﺍﮔﺮ ﻧﮑﺮﺩﻧﺪ ﺑﺎﻳﺪ ﻧﺎﺭﺍﺣﺖ ﺷﻮﻳﺪ. ﺍﻳﻦ ﺗﻮﻗﻊ ﺷﻤﺎﺳﺖ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﻨﻄﻘﯽ ﻧﺒﺎﺷﺪ ﻭ ﺣﺘﯽ ﺍﺷﺘﺒﺎﻩ ﺑﺎﺷﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﻫﻴﭻ ﻭﻗﺖ ﺍﺯ ﮐﻠﻤﻪ «ﺑﺎﻳﺪ» ﺩﺭ ﻣﻮﺭﺩ ﺩﻳﮕﺮﺍﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻧﮑﻨﻴﺪ، ﮐﻠﻤﺎﺕ‌ﺗﺎﻥ ﺭﺍ ﻋﻮﺽ ﮐﻨﻴﺪ. ﻣﺜﻼ «ﺑﻬﺘﺮ ﺑﻮﺩ ﻧﻈﺮ ﻣﻦ ﺭﺍ ﻫﻢ ﻣﯽ‌ﭘﺮﺳﻴﺪﯼ.» ﺑﻪ ﺟﺎﯼ ﺍﻳﻨﮑﻪ ﺑﮕﻮﻳﻴﺪ «ﺗﻮ ﻧﺒﺎﻳﺪ ﺑﺪﻭﻥ ﭘﺮﺳﻴﺪﻥ ﺍﺯ ﻣﻦ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻣﯽ‌ﮐﺮﺩﯼ.»

ﻗﺪﻡ ﺳﻮﻡ: ﻫﻤﻴﺸﻪ ﺑﻪ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎ ﻭ ﻣﺸﮑﻼﺕ ﻫﻢ ﻓﮑﺮ ﮐﻨﻴﺪ.
ﺧﻴﻠﯽ ﻭﻗﺖ‌ﻫﺎ ﺩﻟﻴﻞ ﺍﻳﻨﮑﻪ ﺁﺩﻡ‌ﻫﺎ ﺑﺎ ﻣﺎ ﺁﻥ ﻃﻮﺭﯼ ﮐﻪ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺭﻳﻢ ﺑﺮﺧﻮﺭﺩ ﻧﻤﯽ‌ﮐﻨﻨﺪ، ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺁﻧﻬﺎ ﻫﻢ ﻣﺸﮑﻼﺕ ﻭ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﯼ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺩﺍﺭﻧﺪ.

ﻗﺪﻡ ﭼﻬﺎﺭﻡ: ﻧﻘﻄﻪ ﻧﻈﺮﺍﺕ ﻣﺘﻔﺎﻭﺕ
ﻫﻴﭻ ﻭﻗﺖ ﻳﺎﺩﺗﺎﻥ ﻧﺮﻭﺩ، ﺁﻧﺠﻮﺭﯼ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﺍﺗﻔﺎﻕ‌ﻫﺎ ﻧﮕﺎﻩ ﻣﯽ‌ﮐﻨﻴﺪ ﺑﺎ ﺟﻮﺭﯼ ﮐﻪ ﻓﺮﺯﻧﺪ ﺷﻤﺎ، ﻫﻤﺴﺮ ﺷﻤﺎ ﻭ ﺩﻭﺳﺘﺎﻥ ﻭ ﻫﻤﮑﺎﺭﺍﻥ‌ﺗﺎﻥ ﻧﮕﺎﻩ ﻣﯽ‌ﮐﻨﻨﺪ ﻳﮑﯽ ﻧﻴﺴﺖ.

ﺑﻪ ﺟﺎﯼ ﺍﻳﻨﮑﻪ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺭﺍﺿﯽ ﮐﻨﻴﺪ ﮐﻪ ﺁﻧﻬﺎ ﺍﺷﺘﺒﺎﻩ ﻣﯽ‌ﮐﻨﻨﺪ، ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﻳﺎﺩﺁﻭﺭﯼ ﮐﻨﻴﺪ ﺁﻧﻬﺎ ﻣﺴﺎﺋﻞ ﺭﺍ ﻳﮏ ﺟﻮﺭ ﺩﻳﮕﺮ ﻣﯽ‌ﺑﻴﻨﻨﺪ. ﺑﺮﺍﯼ ﻫﻤﻴﻦ، ﺩﻟﻴﻠﯽ ﻧﺪﺍﺭﺩ ﺍﺯ ﺩﺳﺖ‌ﺷﺎﻥ ﺣﺮﺹ ﺑﺨﻮﺭﻳﺪ. ﻫﻤﺎﻧﻘﺪﺭ ﮐﻪ ﺷﻤﺎ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﺣﺮﻑ‌ﺗﺎﻥ ﻭ ﻓﮑﺮﺗﺎﻥ ﺩﺭﺳﺖ ﺍﺳﺖ،  ﺁﻧﻬﺎ ﻫﻢ ﻫﻤﻴﻦ ﻓﮑﺮ ﺭﺍ ﺩﺭ ﻣﻮﺭﺩ ﻧﻈﺮﺍﺕ ﻭ ﮐﺎﺭﻫﺎﯼ ﺧﻮﺩﺷﺎﻥ ﻣﯽ‌ﮐﻨﻨﺪ.

ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﺘﻮﺍﻧﻴﺪ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ‌ﺗﺎﻥ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﺪ ﺑﺎﻳﺪ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺍﻳﻦ ﻓﺮﺻﺖ ﺭﺍ ﺑﺪﻫﻴﺪ ﻭ ﻗﺒﻮﻝ ﮐﻨﻴﺪ ﮐﻪ ﺁﻧﻬﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺜﻞ ﺷﻤﺎ ﻓﮑﺮ ﻧﮑﻨﻨﺪ. ﺑﻪ ﺗﻮﻗﻌﺎﺗﯽ ﮐﻪ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﻭ ﺩﻳﮕﺮﺍﻥ ﺩﺍﺭﻳﺪ، ﻫﻢ ﺧﻮﺏ ﻓﮑﺮ ﮐﻨﻴﺪ.

 

ﺑﺒﺨﺸﻴﺪ، ﻭﻟﯽ ﻻﺯﻡ ﻧﻴﺴﺖ ﻓﺮﺍﻣﻮﺵ ﮐﻨﻴﺪ.

ﻭﻗﺘﯽ ﺍﺯ ﮐﺴﯽ ﺭﻧﺠﻴﺪﻳﺪ، ﮐﺴﯽ ﺷﻤﺎ ﺭﺍ ﻧﺎﺭﺍﺣﺖ ﮐﺮﺩﻩ ﻭ ﻳﺎ ﺗﺎ ﺣﺪ ﺟﻨﻮﻥ ﻋﺼﺒﺎﻧﯽ ﮐﺮﺩﻩ، ﻭ ﺍﺯ ﺍﻳﻦ ﺍﺗﻔﺎﻕ ﻣﺪﺕ ﺯﻳﺎﺩﯼ ﮔﺬﺷﺘﻪ، ﺩﻳﮕﺮ ﻣﻬﻢ ﻧﻴﺴﺖ ﺁﻥ ﻓﺮﺩ ﻟﻴﺎﻗﺖ ﺑﺨﺸﻴﺪﻩ ﺷﺪﻥ ﺭﺍ ﺩﺍﺭﺩ ﻳﺎ ﻧﻪ.

ﺑﺨﺸﻴﺪﻥ ﻳﮏ ﺗﺼﻤﻴﻢ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﺮﺍﯼ ﺧﻮﺩﺗﺎﻥ ﻣﯽ‌ﮔﻴﺮﻳﺪ. ﺑﺮﺍﯼ ﺁﺭﺍﻣﺶ ﻭ ﺳﻼﻣﺘﯽ ﺧﻮﺩﺗﺎﻥ. ﻣﺜﻞ ﻫﺮ ﺗﺼﻤﻴﻢ ﺩﺭﺳﺖ ﺩﻳﮕﺮﯼ ﮐﻪ ﻣﯽ‌ﮔﻴﺮﻳﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﺼﻤﻴﻢ ﺑﮕﻴﺮﻳﺪ، ﺑﺒﺨﺸﻴﺪ. ﺷﺎﻳﺪ ﺍﻳﻨﺠﺎ ﺗﻨﻬﺎ ﺟﺎﻳﯽ ﺑﺎﺷﺪ ﮐﻪ ﻫﻴﭻ ﺍﺷﮑﺎﻟﯽ ﻧﺪﺍﺭﺩ ﺧﻮﺩﺧﻮﺍﻩ ﺑﺎﺷﻴﺪ.

ﮐﻴﻨﻪ ﺩﺍﺷﺘﻦ، ﺍﻧﺮﮊﯼ ﺭﻭﺣﯽ ﺯﻳﺎﺩﯼ ﺍﺯ ﺷﻤﺎ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﺍﺣﺴﺎﺱ ﺑﺪﯼ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﺩﻫﺪ.

  • ﻭﻗﺘﯽ ﺍﺯ ﺑﺨﺸﻴﺪﻥ ﺣﺮﻑ ﻣﯽ‌ﺯﻧﻴﻢ، ﻣﻨﻈﻮﺭﻣﺎﻥ ﺍﺻﻼ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﺁﻥ ﻧﺎﺭﺍﺣﺘﯽ ﻭ ﺁﺯﺭﺩﮔﯽ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﮐﺮﺩﻩ‌ﺍﻳﻢ.
  • ﺑﺨﺸﻴﺪﻥ ﺍﺻﻼ ﻣﻌﻨﯽ‌ﺍﺵ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﻗﺒﻮﻝ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﺁﻥ ﻓﺮﺩ ﻳﺎ ﺷﺮﺍﻳﻂ ﺩﺭﺳﺖ ﻭ ﺧﻮﺏ ﺑﻮﺩﻩ ﺍﺳﺖ.
  • ﺍﺻﻼ ﻣﺠﺒﻮﺭ ﻧﻴﺴﺘﻴﺪ ﺑﻪ ﮐﺴﯽ ﺑﮕﻮﻳﻴﺪ ﺁﻥ ﺁﺩﻡ ﺭﺍ ﺑﺨﺸﻴﺪﻩ‌ﺍﻳﺪ.
  • ﺍﻳﻦ ﻳﮏ ﺗﺼﻤﻴﻢ ﺷﺨﺼﯽ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭘﻴﺶ ﺧﻮﺩﺗﺎﻥ ﻧﮕﻪ ﺩﺍﺭﻳﺪ. ﺷﻤﺎ ﺍﻳﻦ ﺗﺼﻤﻴﻢ ﺭﺍ ﺑﺮﺍﯼ ﺧﻮﺩﺗﺎﻥ ﻭ ﺁﺳﺎﻳﺶ ﺧﻴﺎﻝ ﻭ ﻓﮑﺮ ﺧﻮﺩﺗﺎﻥ ﮔﺮﻓﺘﻪ‌ﺍﻳﺪ.
  • ﺑﺨﺸﻴﺪﻥ ﻣﻌﻨﯽ‌ﺍﺵ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﺣﺎﻻ ﺩﻳﮕﺮ ﺍﺯ ﻓﺮﺩﺍ ﻣﺎ ﺑﺎﺯ ﻫﻢ ﻣﺜﻞ ﺳﺎﺑﻖ ﺑﻪ ﺁﻥ ﻓﺮﺩ ﺍﻋﺘﻤﺎﺩ ﻣﯽ‌ﮐﻨﻴﻢ.
  • ﺑﺨﺸﻴﺪﻥ ﻭ ﺍﻋﺘﻤﺎﺩ ﮐﺮﺩﻥ ﺩﻭ ﺑﺤﺚ ﺟﺪﺍ ﻫﺴﺘﻨﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺪﺕ‌ﻫﺎ ﺑﻌﺪ ﺍﺯ ﺑﺨﺸﻴﺪﻥ ﻫﻢ ﺷﻤﺎ ﻧﺘﻮﺍﻧﻴﺪ ﺑﻪ ﺁﻥ ﻓﺮﺩ ﺍﻋﺘﻤﺎﺩ ﮐﻨﻴﺪ. ﺍﺻﻼ ﻣﻨﻄﻘﯽ ﻫﻢ ﻧﻴﺴﺖ ﺷﻤﺎ ﺑﻪ ﮐﺴﯽ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﻟﻄﻤﻪ ﺯﺩﻩ ﻣﺠﺪﺩﺍ ﺍﻋﺘﻤﺎﺩ ﮐﻨﻴﺪ.
  • ﺑﺨﺸﻴﺪﻥ ﺍﺻﻼ ﻣﻌﻨﯽ‌ﺍﺵ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﺁﻥ ﺁﺩﻣﯽ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺭﻧﺠﺎﻧﺪﻩ ﻋﻼﻗﻪ‌ﻣﻨﺪ ﺷﺪﻩ‌ﺍﻳﺪ ﻳﺎ ﺣﺘﯽ ﺍﺻﻼ ﻣﯽ‌‌ﺧﻮﺍﻫﻴﺪ ﺩﻭﺑﺎﺭﻩ ﺑﺎ ﺍﻭ ﺍﺭﺗﺒﺎﻁ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻳﺎ ﻧﻪ. ﻧﺪﺍﺷﺘﻦ ﮐﻴﻨﻪ ﻧﺴﺒﺖ ﺑﻪ ﮐﺴﯽ، ﻣﻌﻨﯽ‌ﺍﺵ ﻋﺸﻖ ﻭ ﻣﺤﺒﺖ ﺑﻪ ﺁﻥ ﺁﺩﻡ ﻧﻴﺴﺖ. ﺍﻳﻦ ﺧﻴﻠﯽ ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺑﮕﻮﻳﻴﺪ، «ﺑﺎﺷﻪ ﺣﺎﻻ ﺑﺒﺨﺸﻤﺶ ﻭﻟﯽ ﺍﺻﻼ ﺩﻳﮕﻪ ﻧﻤﯽ‌ﺧﻮﺍﻡ ﺭﺍﺑﻄﻪ ﺑﺎﻫﺎﺵ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻢ.»
  • ﺍﮔﺮ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﺗﺼﻤﻴﻢ ﺑﮕﻴﺮﻳﺪ ﺑﺒﺨﺸﻴﺪ، ﻣﻨﻈﻮﺭ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﻳﮑﺪﻓﻌﻪ ﮐﻼ ﺑﺒﺨﺸﻴﺪ ﻭ ﺗﻤﺎﻡ ﺷﻮﺩ. ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺮﺣﻠﻪ ﺑﻪ ﻣﺮﺣﻠﻪ ﺑﺎﺷﺪ ﻭ ﺩﺭ ﻃﻮﻝ ﺯﻣﺎﻥ. ﻫﻤﻴﻦ ﮐﻪ ﺷﺮﻭﻉ ﮐﻨﻴﺪ ﺑﻪ ﺑﺨﺸﻴﺪﻥ ، ﻣﻌﻨﯽ‌ﺍﺵ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﺮﻭﻉ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺁﻥ ﺣﺲ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﮐﻴﻨﻪ ﻭ ﻧﺎﺭﺿﺎﻳﺘﯽ ﺭﺍ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺩﻭﺭ ﮐﻨﻴﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺪﺕ‌ﻫﺎ ﻃﻮﻝ ﺑﮑﺸﺪ ﻭ ﺍﺻﻼ ﻫﻢ ﺁﺳﺎﻥ ﻧﺒﺎﺷﺪ ﻭﻟﯽ ﺗﺼﻤﻴﻤﯽ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﺧﺎﻃﺮ ﺧﻮﺩﺗﺎﻥ ﻣﯽ‌ﮔﻴﺮﻳﺪ.

 

ﺍﻭﻝ ﻓﮑﺮ، ﺑﻌﺪ ﻋﻤﻞ

ﺩﺭ ﺷﺮﺍﻳﻄﯽ ﮐﻪ ﺍﺯ ﺩﺳﺖ ﻫﻤﻪ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻴﺪ، ﻳﮏ ﺍﺳﺘﺮﺍﺣﺖ ﮐﻮﺗﺎﻩ ﺑﻪ ﺑﺪﻥ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺁﺭﺍﻡ ﺷﻮﻳﺪ، ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﻣﻐﺰﺗﺎﻥ ﺑﻪ ﺷﺮﺍﻳﻂ ﻋﺎﺩﯼ ﺑﺮﮔﺮﺩﺩ ﻭ ﺑﻬﺘﺮ ﺑﺘﻮﺍﻧﻴﺪ ﻓﮑﺮ ﮐﻨﻴﺪ. ﺗﺎﺯﻩ ﺍﻳﻦ ﻓﺮﺻﺖ ﮐﻮﺗﺎﻩ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺣﺮﻓﯽ ﻧﺰﻧﻴﺪ ﮐﻪ ﺍﺻﻼ ﻧﻪ ﺩﻝ‌ﺗﺎﻥ ﻣﯽ‌ﺧﻮﺍﺳﺘﻪ ﻧﻪ ﻣﻨﻈﻮﺭﺗﺎﻥ ﺑﻮﺩﻩ.  ﺍﮔﺮ ﺍﻳﻦ ﻓﺮﺻﺖ ﺭﺍ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﻭ ﮐﺴﯽ ﮐﻪ ﺑﺎ ﺍﻭ ﻣﺸﮑﻞ ﭘﻴﺪﺍ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﻧﺪﻫﻴﺪ، ﺁﻥ ﻭﻗﺖ ﺁﻥ ﺷﺮﺍﻳﻂ ﺑﺎﻻ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﻳﮏ ﻋﻤﻞ ﺧﺸﻦ ﻓﻴﺰﻳﮑﯽ ﻣﻨﺠﺮ ﺷﻮﺩ.

ﺩﺭ ﺍﻭﺝ ﻋﺼﺒﺎﻧﻴﺖ ﺑﺨﺶ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﺣﺴﺎﺳﺎﺕ ﺩﺭ ﻣﻐﺰ ﺷﻤﺎ، ﮐﻨﺘﺮﻝ ﺑﺨﺶ ﻓﮑﺮ ﮐﺮﺩﻥ ﻣﻐﺰﺗﺎﻥ ﺭﺍ ﺑﻪ ﻋﻬﺪﻩ ﻣﯽ‌ﮔﻴﺮﺩ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺷﻤﺎ ﺧﻮﺩﺗﺎﻥ ﻧﻴﺴﺘﻴﺪ ﻭ ﮐﺎﺭﻫﺎﯼ ﺷﻤﺎ ﺭﻭﯼ ﻓﮑﺮ ﻧﻴﺴﺖ ﻭ ﻧﺘﻴﺠﻪ‌ﺍﺵ ﻫﻢ ﻃﺒﻴﻌﺘﺎ ﺁﻥ ﭼﻴﺰﯼ ﻧﺨﻮﺍﻫﺪ ﺷﺪ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﺳﺘﻴﺪ. ﭘﺲ ﺑﺎﻳﺪ ﻳﮏ ﻓﺮﺻﺖ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﻭ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﺑﺪﻫﻴﺪ، ﻳﮏ ﻓﺮﺻﺖ ﺁﺭﺍﻡ  ﺷﺪﻥ، ﻳﮏ ﻓﺮﺻﺖ ﮐﻪ ﮐﻨﺘﺮﻝ ﺭﻓﺘﺎﺭﺗﺎﻥ ﺑﻴﺎﺩ ﺩﺳﺖ ﺧﻮﺩﺗﺎﻥ ﻭ ﺧﻮﺩﺗﺎﻥ ﺑﺸﻮﻳﺪ ﻭ ﺗﺼﻤﻴﻢ ﺑﮕﻴﺮﻳﺪ ﭼﻪ ﮐﺎﺭﯼ ﺑﻬﺘﺮ ﺍﺳﺖ.

ﺍﻳﻦ ﻓﺮﺻﺖ ﺩﺍﺩﻥ ﻭ ﻓﺎﺻﻠﻪ ﮔﺮﻓﺘﻦ ﺍﺯ ﻣﻌﺮﮐﻪ ﻳﮏ ﻗﺎﻋﺪﻩ‌ﻫﺎﻳﯽ ﺩﺍﺭﺩ.

ﻗﺎﻧﻮﻥ ﺍﻭﻝ: ﺍﻳﻦ ﺷﮕﺮﺩ ﺭﺍ ﺷﻤﺎ ﻧﺒﺎﻳﺪ ﺑﻪ ﮐﺴﯽ ﺗﻮﺻﻴﻪ ﮐﻨﻴﺪ.
ﺍﻳﻦ ﺷﮕﺮﺩ ﻓﻘﻂ ﺑﺮﺍﯼ ﺧﻮﺩﺗﺎﻥ ﺍﺳﺖ. ﺷﻤﺎ ﻧﻤﯽ‌ﺗﻮ‌ﺍﻧﻴﺪ ﺑﻪ ﮐﺲ ﺩﻳﮕﺮﯼ ﺑﮕﻮﻳﻴﺪ «ﺍﻻﻥ ﺧﻮﺏ ﻭﻗﺘﻴﻪ ﻳﻪ ﻓﺮﺻﺘﯽ ﺑﻪ ﺧﻮﺩﺕ ﺑﺪﯼ ﻭ ﻳﻪ ﻓﮑﺮﯼ ﺑﮑﻨﯽ». ﺍﻳﻦ ﺗﻮﺻﻴﻪ ﺷﻤﺎ ﻧﻪ ﺗﻨﻬﺎ ﺑﻪ ﮐﺴﯽ ﮐﻤﮏ ﻧﻤﯽ‌ﮐﻨﺪ، ﺑﻠﮑﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﻃﺮﻑ ﺭﺍ ﺑﻴﺸﺘﺮ ﻫﻢ ﻋﺼﺒﯽ ﮐﻨﺪ.

ﻗﺎﻧﻮﻥ ﺩﻭﻡ: ﺍﻋﻼﻡ ﮐﻨﻴﺪ ﮐﻪ ﻳﮏ ﻓﺮﺻﺘﯽ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ.
ﻭﻗﺘﯽ ﺩﺭ ﺣﺎﻝ ﻣﺸﺎﺟﺮﻩ ﻫﺴﺘﻴﺪ ﻭ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﻴﺪ ﺍﮔﺮ ﺍﻻﻥ ﻳﮏ ﻭﻗﺘﯽ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﻧﺪﻫﻴﺪ، ﺧﺸﻢ‌ﺗﺎﻥ ﺍﺯ ﮐﻨﺘﺮﻝ ﺧﺎﺭﺝ ﻣﯽ‌ﺷﻮﺩ، ﺍﻋﻼﻡ ﮐﻨﻴﺪ ﮐﻪ ﺑﻪ ﻳﮏ ﻓﺎﺻﻠﻪ، ﺯﻧﮓ ﺗﻔﺮﻳﺢ ﻳﺎ ﻓﺮﺻﺖ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﻳﺪ.

ﻣﺜﻼ ﺍﮔﺮ ﺩﺭ ﺧﺎﻧﻪ ﺑﺎ ﭘﺴﺮﺗﺎﻥ ﺳﺮ ﺩﻭﺳﺖ‌ﻫﺎﯼ ﺟﺪﻳﺪ ﺍﻭ ﺑﺤﺚ‌ﺗﺎﻥ ﺷﺪﻩ ﻭ ﺷﻤﺎ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﻴﺪ ﻋﺼﺒﺎﻧﻴﺖ‌ﺗﺎﻥ ﺩﺍﺭﺩ ﺍﺯ ﮐﻨﺘﺮﻝ ﺧﺎﺭﺝ ﻣﯽ‌ﺷﻮﺩ، ﺑﻪ ﭘﺴﺮﺗﺎﻥ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﮐﻤﯽ ﻭﻗﺖ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ، ﻣﺜﻼ ﺍﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﻭﺩ ﻭ ﺩﺭ ﺍﺗﺎﻗﺶ ﻣﻨﺘﻈﺮ ﺷﻤﺎ ﺑﻤﺎﻧﺪ، ﭼﻮﻥ ﺷﻤﺎ ﻧﻤﯽ‌ﺧﻮﺍﻫﻴﺪ ﺣﺮﻑ‌ﻫﺎﻳﯽ ﺑﺰﻧﻴﺪ ﮐﻪ ﺧﻮﺩﺗﺎﻥ ﻭ ﺍﻭ ﺭﺍ ﻧﺎﺭﺍﺣﺖ ﮐﻨﻴﺪ.

ﻗﺎﻧﻮﻥ ﺳﻮﻡ: ﻫﻤﺎﻥ ﻣﻮﻗﻊ ﮐﻪ ﮔﻔﺘﻴﺪ، ﺑﺮﮔﺮﺩﻳﺪ.
ﺍﻳﻦ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺩﻗﻴﻘﺎ ﺑﮕﻮﻳﻴﺪ ﮎِﯼ ﺑﺮﻣﯽ‌ﮔﺮﺩﻳﺪ، ﻣﺜﻼ ﺑﻴﺴﺖ ﺩﻗﻴﻘﻪ ﺩﻳﮕﺮ ﻳﺎ ﻧﻴﻢ ﺳﺎﻋﺖ ﺩﻳﮕﺮ؛ ﻫﻤﺎﻥ ﻣﻮﻗﻊ ﻫﻢ ﮐﻪ ﮔﻔﺘﻴﺪ ﺑﺮﮔﺮﺩﻳﺪ. ﭼﻮﻥ ﺍﮔﺮ ﺍﻳﻨﮑﺎﺭﺍ ﻧﮑﻨﻴﺪ ﻣﻤﮑﻦ ﺍﺳﺖ ﭘﺴﺮﺗﺎﻥ ﻓﮑﺮ ﮐﻨﺪ ﺷﻤﺎ ﺑﻪ ﺍﻭ ﻭ ﻣﺴﺌﻠﻪ ﺍﻭ ﺑﯽ‌ﺗﻮﺟﻬﻴﺪ ﻭ ﺑﻪ ﺣﺮﻑ‌ﺗﺎﻥ ﺩﻳﮕﺮ ﺍﻋﺘﻤﺎﺩ ﻧﮑﻨﺪ.

 

ﺩﺭ ﺍﻳﻦ ﻓﺮﺻﺘﯽ ﮐﻪ ﺑﺮﺍﯼ ﻓﮑﺮ ﮐﺮﺩﻥ ﮔﺮﻓﺘﻪ‌ﺍﻳﺪ، ﭼﻪ ﮐﺎﺭﯼ ﻣﯽ‌ﮐﻨﻴﺪ؟
ﺑﺎﻻﺧﺮﻩ ﺍﺯ ﺧﺎﻧﻪ ﺑﻴﺮﻭﻥ ﺁﻣﺪﻳﺪ. ﻭﻟﯽ ﺧﻮﺏ ﻫﻨﻮﺯ ﻧﺎﺭﺍﺣﺖ ﻭ ﮐﻔﺮﯼ ﻫﺴﺘﻴﺪ. ﺧﻮﺏ ﭼﻪ ﮐﺎﺭ ﺑﺎﻳﺪ ﺑﮑﻨﻴﺪ ﮐﻪ ﺁﺭﺍﻡ ﺷﻮﻳﺪ ﻭ ﺑﺮﻭﻳﺪ ﺑﺎ ﻳﮏ ﺷﮕﺮﺩ ﻗﻀﻴﻪ ﺭﺍ ﺣﻞ ﻭ ﻓﺼﻞ ﮐﻨﻴﺪ؟

ﺧﻮﺩﺗﺎﻥ ﺭﺍ  ﺁﺭﺍﻡ ﮐﻨﻴﺪ، ﻓﮑﺮﺗﺎﻥ ﺭﺍ ﺟﻤﻊ ﮐﻨﻴﺪ. ﺑﻪ ﻣﻮﺳﻴﻘﯽ ﺁﺭﺍﻡ ﻣﻮﺭﺩ ﻋﻼﻗﻪ‌ﻳﺘﺎﻥ ﮔﻮﺵ ﮐﻨﻴﺪ، ﻧﻔﺲ ﻋﻤﻴﻖ ﻳﺎﺩﺗﺎﻥ ﻧﺮﻭﺩ، ﺍﮔﺮ ﺧﺎﻧﻪ ﻣﺎﻧﺪﻩ‌ﺍﻳﺪ، ﺩﻭﺵ ﺑﮕﻴﺮﻳﺪ ﻭ  ﻣﮑﺎﻟﻤﻪ ﺩﻭﺭﻧﯽ‌ﺗﺎﻥ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ.

ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﻳﮑﯽ ﺍﺯ ﺭﺍﻩ‌ﻫﺎﯼ ﺁﺭﺍﻡ ﺷﺪﻥ ﺩﺭ ﺣﺎﻟﺖ ﻋﺼﺒﯽ ﮐﻪ ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﻗﺒﻠﯽ ﺧﻮﺍﻧﺪﻩ‌ﺍﻳﺪ ﮐﻤﮏ ﺑﮕﻴﺮﻳﺪ.

ﺍﻳﻨﻬﺎ ﻧﻤﻮﻧﻪ‌ﻫﺎﻳﯽ ﺍﺯ ﺣﺮﻑ‌ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺍﮔﺮ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺑﮕﻮﻳﻴﺪ، ﺷﻤﺎ ﺭﺍ ﺑﺮﺍﯼ ﻭﻗﺘﯽ ﮐﻪ ﭘﻴﺶ ﭘﺴﺮﺗﺎﻥ ﺑﺮﺍﯼ ﺍﺩﺍﻣﻪ ﺑﺤﺚ ﺑﺮﮔﺸﺘﻴﺪ، ﺑﻬﺘﺮ ﺁﻣﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﻳﻨﻬﺎ ﻧﻤﻮﻧﻪ‌ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﻫﻤﻪ ﺁﻧﻬﺎ ﻣﻨﺎﺳﺐ ﺷﻤﺎ ﻧﺒﺎﺷﺪ:

  • ﻣﻦ ﺧﻮﺩﻡ ﻣﺴﺌﻮﻝ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺍﺣﺴﺎﺱ ﺧﻮﺩﻡ ﻫﺴﺘﻢ.
  • ﻣﻦ ﻫﻢ ﺑﺎﻳﺪ ﺑﻪ ﺳﻬﻢ ﺧﻮﺩﻡ ﺩﺭ ﺍﻳﻦ ﺩﻋﻮﺍ ﻓﮑﺮ ﮐﻨﻢ.
  • ﻫﻴﭽﮑﺲ ﮐﺎﻣﻞ ﻧﻴﺴﺖ، ﭘﺲ ﻣﻦ ﺧﻮﺩﻡ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺭﻓﺘﺎﺭ ﻧﺎﻣﻨﺎﺳﺒﻢ ﻣﯽ‌ﺑﺨﺸﻢ.
  • ﻫﻴﭽﮑﺲ ﮐﺎﻣﻞ ﻧﻴﺴﺖ، ﭘﺲ ﻣﻦ ﭘﺴﺮﻡ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﻟﺤﻦ ﺑﺪﺵ ﻣﯽ‌ﺑﺨﺸﻢ.
  • ﻳﺎﺩﻡ ﻧﺮﻭﺩ ﭼﻘﺪﺭ ﺩﻭﺳﺘﺶ ﺩﺍﺭﻡ.
  • ﺷﺎﻳﺪ ﻣﻦ ﺯﻳﺎﺩﯼ ﺍﺯﺵ ﺗﻮﻗﻊ ﺩﺍﺷﺘﻢ.
  • ﺷﺎﻳﺪ ﺍﺯ ﺧﻮﺩﻡ ﺯﻳﺎﺩﯼ ﺗﻮﻗﻊ ﺩﺍﺭﻡ.
  • ﻣﻬﻢ ﻧﻴﺴﺖ ﺣﻖ ﺑﺎ ﮐﻴﻪ، ﻣﻬﻢ ﺣﻞ ﻣﺸﮑﻠﻪ.
