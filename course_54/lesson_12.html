ﺩﺭﺱ ﺷﺸﻢ- ﺩﻩ ﺷﮕﺮﺩ ﮐﻨﺎﺭ ﺁﻣﺪﻥ ﺑﺎ ﻓﺮﺯﻧﺪﺍﻥ ﺧﺸﻤﮕﻴﻦ

ﺩﺧﺘﺮ ﭼﻬﺎﺭﺩﻩ ﺳﺎﻟﻪ ﺷﻤﺎ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﺩ ﺍﺟﺎﺯﻩ ﻧﺪﺍﺭﺩ ﺑﻪ ﺧﺎﻧﻪ ﺩﻭﺳﺘﺶ ﺑﺮﻭﺩ، ﭼﺮﺍ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﻗﻮﻝ ﺩﺍﺩﻩ ﺑﻮﺩﻩ ﺍﺗﺎﻗﺶ ﺭﺍ ﻣﺮﺗﺐ ﮐﻨﺪ ﻭ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻧﮑﺮﺩﻩ.

ﺩﺧﺘﺮﺗﺎﻥ ﻣﯽ‌ﺁﻳﺪ ﺩﺭ ﺻﻮﺭﺕ‌ﺗﺎﻥ ﻭ ﻓﺮﻳﺎﺩ ﻣﯽ‌ﮐﺸﺪ، «ﺗﻮ ﺧﻮﺩﺧﻮﺍﻩ‌ﺗﺮﻳﻦ ﻣﺎﺩﺭ ﺩﻧﻴﺎﻳﯽ!» ﺑﻌﺪ ﺑﺮﻣﯽ‌ﮔﺮﺩﺩ ﺑﻪ ﺍﺗﺎﻗﺶ ﻭ ﺩﺭ ﺭﺍ ﻫﻢ ﭘﺸﺖ ﺳﺮﺵ ﻣﯽ‌ﮐﻮﺑﺪ.

ﺷﻤﺎ ﻫﻢ ﺷﺎﻳﺪ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﺟﻮﺍﺏ ﺑﺪﻫﻴﺪ «ﺁﺭﻩ ﻫﻤﻴﻨﻪ ﺍﻳﻨﺠﻮﺭﯼ ﻓﮑﺮ ﮐﻦ!». ﺩﺧﺘﺮﺗﺎﻥ ﺣﺴﺎﺑﯽ ﺍﻋﺼﺎﺑﺶ ﺍﺯ ﺩﺳﺖ ﺷﻤﺎ ﻭ ﺧﻮﺩﺵ ﺧﻮﺭﺩ ﺷﺪﻩ، ﺁﺯﺭﺩﻩ ﺷﺪﻩ ﻭ ﻣﻄﻤﺌﻦ ﻧﻴﺴﺖ ﺑﺎﻳﺪ ﭼﻪ ﮐﺎﺭ ﻣﯽ‌ﮐﺮﺩﻩ.

ﻫﻴﭽﻮﻗﺖ ﺑﻪ ﻓﮑﺮ ﺍﺻﻼﺡ ﻓﺮﺯﻧﺪﺍﻧﺘﺎﻥ ﻭﻗﺘﯽ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻨﺪ ﻧﺒﺎﺷﻴﺪ، ﺍﻳﻦ ﮐﺎﺭ  ﻣﺜﻞ ﮐﺒﺮﻳﺖ ﮐﺸﻴﺪﻥ ﺩﺭ ﺍﻧﺒﺎﺭ ﮐﺎﻩ ﺍﺳﺖ.

ﺍﮔﺮ ﺷﻤﺎ ﻫﻢ ﺗﺠﺮﺑﻪ‌ﻫﺎﯼ ﺍﻳﻨﭽﻨﻴﻨﯽ ﺩﺍﺷﺘﻪ‌ﺍﻳﺪ، ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻫﺴﺘﻴﺪ ﻭ ﺩﺭ ﻣﻮﺍﻗﻌﯽ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻋﺼﺒﺎﻧﯽ ﺍﺳﺖ ﮐﺎﺭﯼ ﺟﺰ ﺩﺍﺩ ﺯﺩﻥ ﺩﺭ ﺟﻮﺍﺏ ﺩﺍﺩﻫﺎﯼ ﺍﻭ ﺑﻠﺪ ﻧﻴﺴﺘﻴﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﻫﻴﭽﻮﻗﺖ ﻧﻤﯽ‌ﺩﺍﻧﻴﺪ ﭘﺎﺳﺦ ﺩﺭﺳﺖ ﺩﺭ ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ ﭼﻴﺴﺖ. ﻟﻄﻔﺎ ﺑﺎ ﺩﻗﺖ ﺩﻩ ﺭﻭﺵ ﺍﻳﻦ ﺩﺭﺱ ﺭﺍ ﺑﺨﻮﺍﻧﻴﺪ.

ﻗﺎﻧﻮﻥ ﺍﻭﻝ: ﺩﺍﺩ ﺭﺍ ﺑﺎ ﺩﺍﺩ ﺟﻮﺍﺏ ﻧﺪﻫﻴﺪ ﻭ ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﻭﻗﺘﯽ ﻋﺼﺒﺎﻧﯽ ﺍﺳﺖ، ﺍﺻﻼﺡ ﮐﻨﻴﺪ.
ﭘﺎﺳﺦ ﺑﻴﺸﺘﺮ ﻭﺍﻟﺪﻳﻦ ﺑﻪ ﻋﺼﺒﺎﻧﻴﺖ ﻓﺮﺯﻧﺪﺍﻥ‌ﺷﺎﻥ، ﺑﻠﻨﺪ ﮐﺮﺩﻥ ﺻﺪﺍ ﻭ ﺍﻧﺘﻘﺎﺩ ﮐﺮﺩﻥ ﺍﺯ ﺁﻧﻬﺎﺳﺖ، ﻭﻟﯽ ﻋﻤﻼ ﺍﻳﻦ ﺑﺮﺧﻮﺭﺩ ﻓﻘﻂ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺍﺣﺴﺎﺳﺎﺕ ﺷﻤﺎ ﺍﺯ ﮐﻨﺘﺮﻝ ﺧﺎﺭﺝ ﺷﻮﺩ. ﺑﻬﺘﺮﻳﻦ ﮐﺎﺭ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺁﺭﺍﻣﺶ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺣﻔﻆ ﮐﻨﻴﺪ ﻭ ﺁﺭﺍﻡ ﺑﺎﺷﻴﺪ. ﺩﺭﺳﺖ ﻣﺜﻞ ﺍﻳﻦ ﻣﯽ‌ﻣﺎﻧﺪ ﮐﻪ ﺑﺎ ﻣﺎﺷﻴﻦ ﺩﻳﮕﺮﯼ ﺗﺼﺎﺩﻑ ﮐﺮﺩﻩ‌ﺍﻳﺪ. ﺍﮔﺮ ﺑﺮﺧﻮﺭﺩ ﺷﻤﺎ ﺑﺎ ﺭﺍﻧﻨﺪﻩ ﻋﺼﺒﯽ ﮐﻪ ﺑﻪ ﻃﺮﻑ ﺷﻤﺎ ﻣﯽ‌ﺁﻳﺪ ﺁﺭﺍﻡ ﺑﺎﺷﺪ، ﺍﻭ ﻫﻢ ﺑﻌﺪ ﺍﺯ ﻣﺪﺗﯽ ﺁﺭﺍﻡ ﻣﯽ‌ﺷﻮﺩ؛ ﻭﻟﯽ ﺍﮔﺮ ﺷﻤﺎ ﻫﻢ ﺑﺮﺍﻓﺮﻭﺧﺘﻪ ﺑﻪ ﻃﺮﻑ ﺍﻭ ﺑﺮﻭﻳﺪ ﻭ ﺑﮕﻮﻳﻴﺪ «ﻫﻤﺶ ﺗﻘﺼﻴﺮ ﺗﻮ ﺑﻮﺩ.»، ﮐﺒﺮﻳﺖ ﺭﻭﺷﻦ ﺭﺍ ﺩﺭ ﺍﻧﺒﺎﺭ ﮐﺎﻩ ﺍﻧﺪﺍﺧﺘﻪ‌ﺍﻳﺪ.

 

ﻗﺎﻧﻮﻥ ﺩﻭﻡ: ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻋﺼﺒﯽ ﻭ ﻧﺎﺭﺍﺣﺖ ﺍﺳﺖ، ﺳﻌﯽ ﻧﮑﻨﻴﺪ ﺍﻭ ﺭﺍ ﻣﺘﻘﺎﻋﺪ ﮐﻨﻴﺪ.
ﺧﻴﻠﯽ ﺍﺯ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭﻫﺎ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺑﺎ ﺩﻟﻴﻞ ﻭ ﻣﻨﻄﻖ ﻋﺼﺒﺎﻧﻴﺖ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺭﺍ ﮐﻢ ﮐﻨﻨﺪ. ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﭽﻪ‌ﻫﺎ ﻣﺜﻞ ﺑﺰﺭﮔﺴﺎﻻﻥ ﻗﺎﺩﺭ ﻧﻴﺴﺘﻨﺪ ﺑﺎ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﺩﻻﻳﻞ ﻣﻨﻄﻘﯽ ﺧﺸﻢ‌ﺷﺎﻥ ﺭﺍ ﺁﺭﺍﻡ ﮐﻨﻨﺪ، ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺧﻴﻠﯽ ﺍﺯ ﺑﺰﺭﮔﺴﺎﻻﻥ ﺑﺎ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﻣﻨﻄﻖ ﻭ ﺍﺳﺘﺪﻻﻝ ﻋﻤﻞ‌ﺷﺎﻥ، ﺁﺭﺍﻡ ﺷﻮﻧﺪ. ﺩﺭ ﻣﻮﺭﺩ ﺑﭽﻪ‌ﻫﺎ ﺍﻳﻦ ﺭﻭﺵ ﻓﻘﻂ ﺁﻧﻬﺎ ﺭﺍ ﻋﺼﺒﯽ‌ﺗﺮ ﻣﯽ‌ﮐﻨﺪ. ﺟﻤﻠﻪ‌ﻫﺎﻳﯽ ﻣﺜﻞ «ﺣﺎﻻ ﭼﺮﺍ ﺍﺯ ﺩﺳﺖ ﻣﻦ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﯽ؟، ﺧﻮﺩﺕ ﺗﮑﻠﻴﻒ‌ﺍﺕ ﺭﻭ ﺑﻪ ﻣﻮﻗﻊ ﺍﻧﺠﺎﻡ ﻧﺪﺍﺩﯼ»، ﻓﻘﻂ ﺑﺎﻋﺚ ﺑﺪﺗﺮ ﺷﺪﻥ ﺍﻭﺿﺎﻉ ﻣﯽ‌ﺷﻮﺩ. ﭘﺲ ﮐﻤﯽ ﺻﺒﺮ ﮐﻨﻴﺪ ﻭ ﻭﻗﺘﯽ ﺍﻭ ﺁﺭﺍﻡ ﺷﺪ، ﺑﺎ ﻫﻢ ﺭﺍﺟﻊ ﺑﻪ ﺁﻥ ﺻﺤﺒﺖ ﮐﻨﻴﺪ.

 

ﻗﺎﻧﻮﻥ ﺳﻮﻡ: ﺣﻮﺍﺱ‌ﺗﺎﻥ ﺑﻪ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ‌ﻫﺎﯼ ﺧﻮﺩﺗﺎﻥ ﺑﺎﺷﺪ.
ﻣﺮﺍﻗﺐ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ ﻫﺎﯼ ﺧﻮﺩﺗﺎﻥ ﺑﺎﺷﻴﺪ، ﺑﻬﺮﺣﺎﻝ ﺷﻤﺎ ﺩﺭ ﺷﺮﺍﻳﻄﯽ ﻫﺴﺘﻴﺪ ﮐﻪ ﻃﺮﻑ ﻣﻘﺎﺑﻞ‌ﺗﺎﻥ ﺩﻟﺨﻮﺭ ﻭ ﻋﺼﺒﺎﻧﯽ ﺍﺳﺖ. ﻗﻄﻌﺎ ﺁﺳﺎﻥ ﻧﻴﺴﺖ، ﻭﻟﯽ ﺷﻤﺎ ﺩﺭ ﺷﺮﺍﻳﻄﯽ ﻫﺴﺘﻴﺪ ﮐﻪ ﺑﺎﻳﺪ ﺁﺭﺍﻣﺶ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺣﻔﻆ ﮐﻨﻴﺪ. ﺑﻪ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺷﻤﺎ ﻧﻘﻄﻪ ﺍﺗﮑﺎﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺩﺭ ﺍﻳﻨﮕﻮﻧﻪ ﻟﺤﻈﺎﺕ ﻫﺴﺘﻴﺪ، ﺍﻭ ﺍﺯ ﺷﻤﺎ ﻣﯽ‌ﺁﻣﻮﺯﺩ ﺑﺎ ﺍﻳﻦ ﺷﺮﺍﻳﻂ ﭼﻄﻮﺭ ﮐﻨﺎﺭ ﺑﻴﺎﻳﺪ.

ﺁﺭﺍﻣﺶ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺣﻔﻆ ﮐﻨﻴﺪ، ﺩﺍﺩ ﺭﺍ ﺑﺎ ﺩﺍﺩ ﭘﺎﺳﺦ ﻧﺪﻫﻴﺪ ﻭ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺩﺭ ﺩﺍﻡ ﮐﺸﻤﮑﺶ ﻗﺪﺭﺕ ﻧﻴﻨﺪﺍﺯﻳﺪ ﮐﻪ ﺍﻳﻦ ﻓﻘﻂ ﺗﻨﺶ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﻣﯽ‌ﺩﻫﺪ. ﻭﻗﺘﯽ ﺷﻤﺎ ﺑﻪ ﺧﺸﻢ ﻓﺮﺯﻧﺪﺗﺎﻥ ﭘﺎﺳﺦ ﻋﺼﺒﯽ ﻣﯽ‌ﺩﻫﻴﺪ ﺍﻭ ﺑﻴﺸﺘﺮ ﺗﺤﺮﻳﮏ ﻣﯽ‌ﺷﻮﺩ ﺗﺎ ﺗﻮﺟﻪ ﺷﻤﺎ ﺭﺍ ﺟﻠﺐ ﮐﻨﺪ. ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﻣﺮﺗﺐ ﻳﺎﺩﺁﻭﺭﯼ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻨﺠﺎ ﺟﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺑﺎ ﻣﻬﺎﺭﺕ‌ﻫﺎﻳﯽ ﮐﻪ ﮐﺴﺐ ﻣﯽ‌ﮐﻨﻴﺪ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ، ﭼﻨﻴﻦ ﺷﺮﺍﻳﻄﯽ ﺭﺍ ﺑﻪ ﺩﺭﺳﺘﯽ ﻭ ﺳﺮﻳﻊ ﺟﻤﻊ ﻭ ﺟﻮﺭ ﮐﻨﻴﺪ.

 

ﻗﺎﻧﻮﻥ ﭼﻬﺎﺭﻡ: ﻭﺍﺭﺩ ﺩﺭﮔﻴﺮﯼ ﻓﻴﺰﻳﮑﯽ ﺑﺎ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻧﺸﻮﻳﺪ.
ﺑﻌﻀﯽ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﮐﻨﺘﺮﻝ ﺧﻮﺩﺷﺎﻥ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﻨﺪ ﻭ ﻭﺍﺭﺩ ﺩﺭﮔﻴﺮﯼ ﻓﻴﺰﻳﮑﯽ ﺑﺎ ﻓﺮﺯﻧﺪﺷﺎﻥ ﻣﯽ‌ﺷﻮﻧﺪ. ﺍﺯ ﭼﻨﻴﻦ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ‌ﻫﺎﻳﯽ ﺑﻪ ﺷﺪﺕ ﺑﭙﺮﻫﻴﺰﻳﺪ، ﺷﻤﺎ ﻧﺒﺎﻳﺪ ﻣﺜﻞ ﻓﺮﺯﻧﺪ ﻧﻮﺟﻮﺍﻥ‌ﺗﺎﻥ ﺭﻓﺘﺎﺭ ﮐﻨﻴﺪ. ﺍﮔﺮ ﭼﻨﻴﻦ ﺍﺗﻔﺎﻗﯽ ﻫﻢ ﭘﻴﺶ ﺁﻣﺪ، ﺭﺍﻩ ﺣﻠﺶ ﻓﻘﻂ ﺗﻮﺿﻴﺢ ﺁﻥ ﺷﺮﺍﻳﻂ ﺑﺮﺍﯼ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻭ ﻋﺬﺭﺧﻮﺍﻫﯽ ﺍﺳﺖ. ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺷﻤﺎ ﺧﻮﺩ ﺭﺍ ﻣﺴﺌﻮﻝ ﺭﻓﺘﺎﺭ ﺧﻮﺩ ﻧﺪﺍﻧﺴﺘﻪ ﻭ ﮐﺲ ﺩﻳﮕﺮﯼ ﺭﺍ ﺑﺮﺍﯼ ﻋﻤﻞ‌ﺗﺎﻥ ﺳﺮﺯﻧﺶ ﮐﺮﺩﻩ‌ﺍﻳﺪ. ﺑﺎﺯ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﻳﺎﺩﺁﻭﺭﯼ ﮐﻨﻴﺪ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﻟﺤﻈﻪ‌ﻫﺎﺳﺖ ﮐﻪ ﺑﺎ ﺭﻓﺘﺎﺭﺗﺎﻥ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻣﻮﺍﺟﻬﻪ‌ﯼ ﺩﺭﺳﺖ ﺑﺎ ﭼﺎﻟﺶ‌ﻫﺎ ﻭ ﻟﺤﻈﻪ‌ﻫﺎﯼ ﺳﺨﺖ ﺭﺍ ﻳﺎﺩ ﻣﯽ‌ﺩﻫﻴﺪ. ﺍﮔﺮ ﺷﻤﺎ ﺑﺮﺧﻮﺭﺩ ﻓﻴﺰﻳﮑﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ، ﺍﻭ ﻫﻢ ﺍﺯ ﺷﻤﺎ ﻣﯽ‌ﺁﻣﻮﺯﺩ ﻣﺸﮑﻼﺗﺶ ﺭﺍ ﺑﺎ ﭘﺮﺧﺎﺵ ﻭ ﺧﺸﻮﻧﺖ ﺣﻞ ﮐﻨﺪ.

 

ﻗﺎﻧﻮﻥ ﭘﻨﺠﻢ: ﺑﺎ ﮐﻮﭼﮏ‌ﺗﺮﻫﺎ ﺭﻓﺘﺎﺭ ﻣﺘﻔﺎﻭﺗﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.
ﺍﮔﺮ ﻓﺮﺯﻧﺪ ﺯﻳﺮ ﭼﻬﺎﺭ ﺳﺎﻟﻪ ﺩﺍﺭﻳﺪ، ﺑﺎ ﺍﻭ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ ﻃﻮﺭ ﺩﻳﮕﺮﯼ ﺑﺮﺧﻮﺭﺩ ﮐﻨﻴﺪ. ﺍﮔﺮ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﻓﺮﺯﻧﺪ ﮐﻮﭼﮏ‌ﺗﺎﻥ ﻋﺼﺒﯽ ﺍﺳﺖ، ﺣﺘﻤﺎ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﺑﻪ ﺍﻭ ﮐﻤﮏ ﮐﻨﻴﺪ ﮐﻪ ﺁﺭﺍﻡ ﺷﻮﺩ، ﺑﻪ ﺍﻭ ﭼﻨﺪ ﺭﺍﻩ ﺣﻞ ﭘﻴﺸﻨﻬﺎﺩ ﮐﻨﻴﺪ؛ ﺑﻪ ﺍﺗﺎﻗﺶ ﺑﺮﻭﺩ ﻭ ﮐﻤﯽ ﺩﺭﺍﺯ ﺑﮑﺸﺪ، ﮐﻤﯽ ﺭﻭﯼ ﻣﺒﻞ ﻭﻟﻮ ﺷﻮﺩ ﻭ... . ﺑﻪ ﺍﻭ ﺍﻳﻦ ﺍﻃﻤﻴﻨﺎﻥ ﺭﺍ ﺑﺪﻫﻴﺪ ﮐﻪ ﮐﻤﯽ ﺑﻌﺪ ﺁﺭﺍﻡ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺣﺎﻟﺶ ﺑﻬﺘﺮ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻣﺜﻼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﭘﻴﺶ ﺩﻳﮕﺮﺍﻥ ﺑﺮﮔﺮﺩﺩ.

 

ﻗﺎﻧﻮﻥ ﺷﺸﻢ: ﻣﺴﺦ ﻧﺸﻮﻳﺪ.
ﺑﻌﻀﯽ ﺍﺯ ﻭﺍﻟﺪﻳﻦ ﺩﺭ ﻣﻘﺎﺑﻞ ﻋﺼﺒﺎﻧﻴﺖ ﻓﺮﺯﻧﺪﺍﻧﺸﺎﻥ ﺍﺯ ﺧﻮﺩ ﻧﻘﻄﻪ ﺿﻌﻒ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ. ﮔﺎﻫﯽ ﺑﭽﻪ‌ﻫﺎ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺧﻮﺍﺳﺘﻪ‌ﻫﺎﻳﺸﺎﻥ ﺑﻪ ﺍﻳﻦ ﺭﺍﻩ ﻣﺘﻮﺳﻞ ﻣﯽ‌ﺷﻮﻧﺪ. ﭼﻮﻥ ﻗﺒﻼ ﺩﻳﺪﻩ‌ﺍﻧﺪ ﮐﻪ ﺷﻤﺎ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺁﻧﻬﺎ ﺑﻪ ﺭﻓﺘﺎﺭﻫﺎﯼ ﻋﺼﺒﯽ‌ﺷﺎﻥ ﭘﺎﻳﺎﻥ ﺩﻫﻨﺪ، ﺩﺭ ﻣﻘﺎﺑﻞ ﺧﻮﺍﺳﺘﻪ ﺁﻧﻬﺎ (ﮐﻪ ﺍﻟﺰﺍﻣﺎ ﻫﻢ ﻣﻮﺍﻓﻘﺶ ﻧﻴﺴﺘﻴﺪ) ﮐﻮﺗﺎﻩ ﻣﯽ‌ﺁﻳﻴﺪ. ﺩﺭ ﺍﻳﻦ ﺷﺮﺍﻳﻂ ﮐﺎﺭﯼ ﮐﻪ ﺷﻤﺎ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ ﺑﮑﻨﻴﺪ ﺍﻳﻦ ﺍﺳﺖ: ﻧﻪ ﻋﺼﺒﺎﻧﯽ ﺷﻮﻳﺪ، ﻧﻪ ﮐﻮﺗﺎﻩ ﺑﻴﺎﻳﻴﺪ.

ﺗﻤﺎﻡ ﺳﻌﯽ‌ﺗﺎﻥ ﺭﺍ ﺑﮑﻨﻴﺪ ﻭ ﺁﺭﺍﻡ ﺑﺎﺷﻴﺪ ﺗﺎ ﻭﻗﺘﯽ ﺁﻧﻬﺎ ﻫﻢ ﺁﺭﺍﻡ ﺷﺪﻧﺪ، ﺳﺮﺧﻮﺍﺳﺘﻪ‌ﯼ ﺁﻧﻬﺎ ﺑﺎ ﻫﻢ ﺣﺮﻑ ﺑﺰﻧﻴﺪ.

 

ﻗﺎﻧﻮﻥ ﻫﻔﺘﻢ:ﻓﺮﺯﻧﺪﺗﺎﻥ ﺭﺍ ﺑﺮﺍﯼ ﺭﻓﺘﺎﺭ ﺑﺪ ﺟﺮﻳﻤﻪ ﮐﻨﻴﺪ ﻧﻪ ﺑﺮﺍﯼ ﻋﺼﺒﺎﻧﻴﺖ.
ﻭﻗﺘﯽ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻋﺼﺒﯽ ﺍﺳﺖ ﻣﻤﮑﻦ ﺍﺳﺖ ﺭﻓﺘﺎﺭﻫﺎﻳﯽ ﺍﺯ ﺧﻮﺩﺵ ﺑﺮﻭﺯ ﺩﻫﺪ ﮐﻪ ﺩﺭﺳﺖ ﻧﺒﺎﺷﺪ ﻭ ﺍﺯ ﻧﻈﺮ ﺷﻤﺎ ﻻﺯﻡ ﺑﺎﺷﺪ ﺟﺮﻳﻤﻪ ﺷﻮﺩ. ﻫﻴﭻ ﺍﺷﮑﺎﻟﯽ ﻫﻢ ﻧﺪﺍﺭﺩ ﻭﻟﯽ ﺧﻴﻠﯽ ﺩﻗﺖ ﮐﻨﻴﺪ ﮐﻪ ﺍﻭ ﺭﺍ ﺑﺮﺍﯼ ﺗﺨﻠﻴﻪ‌ﯼ ﺧﺸﻤﺶ ﺟﺮﻳﻤﻪ ﻧﮑﻨﻴﺪ، ﺣﺎﻻ ﺍﮔﺮ ﺭﻓﺖ ﺩﺭ ﺍﺗﺎﻗﺶ ﭼﻬﺎﺭ ﺗﺎ ﺩﺍﺩ ﺯﺩ ﮐﻪ ﭼﻘﺪﺭ ﺑﺪﺷﺎﻧﺴﻪ ﻭ ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﻳﻦ ﭼﻪ ﺯﻧﺪﮔﻴﻪ ﻭ ﺩﻧﻴﺎ ﭼﻘﺪﺭ ﻇﺎﻟﻤﻪ ﻭ...، ﺁﻧﻬﺎ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻧﺎﺷﯽ ﺍﺯ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺗﺨﻠﻴﻪ ﺧﺸﻢ ﺑﺎﺷﺪ. ﻭﻟﯽ ﺍﮔﺮ ﺑﻪ ﺷﻤﺎ ﻳﺎ ﻋﻀﻮ ﺩﻳﮕﺮﯼ ﺍﺯ ﺧﺎﻧﻮﺍﺩﻩ ﻧﺎﺳﺰﺍ ﮔﻔﺖ ﻳﺎ ﺣﺮﮐﺖ ﻓﻴﺰﻳﮑﯽ ﺧﺸﻨﯽ ﺍﺯ ﺧﻮﺩﺵ ﻧﺸﺎﻥ ﺩﺍﺩ، ﺷﺎﻳﺪ ﻻﺯﻡ ﺑﺎﺷﺪ ﺑﻌﺪ ﺍﺯ ﺍﻳﻨﮑﻪ ﺁﺭﺍﻡ ﺷﺪ ﻭ ﻋﺼﺒﺎﻧﻴﺘﺶ ﻓﺮﻭﮐﺶ ﮐﺮﺩ، ﻧﺘﻴﺠﻪ ﻋﻤﻞ ﻧﺎﺷﺎﻳﺴﺘﺶ ﺭﺍ ﺑﺒﻴﻨﺪ.

 

ﻗﺎﻧﻮﻥ ﻫﺸﺘﻢ: ﺗﻬﺪﻳﺪ ﺑﻪ ﺗﻨﺒﻴﻪ ﺳﺨﺖ ﻭ ﻧﺎﻣﺘﻨﺎﺳﺐ ﻧﮑﻨﻴﺪ.
ﺗﻬﺪﻳﺪ ﺑﻪ ﺗﻨﺒﻴﻪ ﺳﺨﺖ ﺩﺭ ﻫﻤﺎﻥ ﺯﻣﺎﻧﯽ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﻋﺼﺒﯽ ﺍﺳﺖ، ﮐﺎﺭ ﻧﻤﯽ‌ﮐﻨﺪ. ﻣﺜﻼ ﺷﻤﺎ ﺧﺴﺘﻪ ﺷﺪﻩ‌ﺍﻳﺪ ﺍﺯ ﺭﻓﺘﺎﺭﻫﺎﯼ ﻋﺼﺒﯽ ﻧﻮﺟﻮﺍﻥ‌ﺗﺎﻥ ﻭ ﺗﻬﺪﻳﺪﺵ ﻣﯽ‌ﮐﻨﻴﺪ: «ﺍﮔﺮ ﻫﻤﻴﻦ ﺟﺎ ﺑﺲ ﻧﮑﻨﯽ ﻣﻮﺑﺎﻳﻠﺖ ﺭﻭ ﻳﮏ ﻫﻔﺘﻪ ﻗﻄﻊ ﻣﯽ‌ﮐﻨﻢ»، ﺧﻮﺏ ﺍﮔﺮ ﺍﺛﺮ ﻧﮑﺮﺩ ﭼﻪ، ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﻫﻤﻴﻨﻄﻮﺭ ﻣﺪﺕ ﺍﻳﻦ ﺗﻨﺒﻴﻪ ﺭﺍ ﺯﻳﺎﺩ ﮐﻨﻴﺪ؟ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﺍﺛﺮ ﺩﺍﺭﺩ؟ ﻓﻘﻂ ﺍﺣﺘﻤﺎﻻ ﺍﻭ ﺭﺍ ﺳﺮ ﻟﺞ ﻣﯽ‌ﺍﻧﺪﺍﺯﺩ ﻭ ﺭﻓﺘﺎﺭ ﻋﺼﺒﯽ‌ﺍﺵ ﺭﺍ ﺗﺸﺪﻳﺪ ﻣﯽ‌ﮐﻨﺪ.

ﻗﺎﺑﻞ ﺩﺭﮎ ﺍﺳﺖ ﮐﻪ ﺗﺤﻤﻞ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﭘﺮﺧﺎﺵ ﺩﺭ ﺧﺎﻧﻪ ﺳﺨﺖ ﺍﺳﺖ، ﻭﻟﯽ ﻣﺮﺗﺐ ﺑﺎﻳﺪ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺑﭙﺮﺳﻴﺪ، ﺑﺎ ﺍﻳﻦ ﮐﺎﺭ ﭼﻪ ﭼﻴﺰﯼ ﺭﺍ ﻣﯽ‌ﺧﻮﺍﻫﻢ ﺑﻪ ﺍﻭ ﻳﺎﺩ ﺑﺪﻫﻢ؟ ﻳﺎﺩ ﺑﺪﻫﻢ ﭼﮕﻮﻧﻪ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﮐﻨﺎﺭ ﺑﻴﺎﻳﺪ؟ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﺎ ﺍﻳﻨﮑﻪ ﻣﺜﻞ ﺍﻭ ﺭﻓﺘﺎﺭ ﮐﻨﻴﺪ ﻭ ﺷﻤﺎ ﻫﻢ ﺍﺯ ﻋﺼﺒﺎﻧﻴﺖ ﺍﻭ ﻋﺼﺒﺎﻧﯽ ﺷﻮﻳﺪ ﻭ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ ﻧﺸﺎﻥ ﺩﻫﻴﺪ، ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻭ ﻳﺎﺩ ﺑﺪﻫﻴﺪ ﻣﺪﻳﺮ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﺑﺎﺷﺪ.

 

ﻗﺎﻧﻮﻥ ﻧﻬﻢ: ﻳﮏ ﻓﺮﺻﺖ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺑﺪﻫﻴﺪ.
ﻫﻤﺎﻧﻄﻮﺭ ﻭﻗﺘﯽ ﮐﻪ ﺑﺎ ﻫﻤﺴﺮﺗﺎﻥ ﺣﺮﻑ‌ﺗﺎﻥ ﻣﯽ‌ﺷﻮﺩ، ﻣﻤﮑﻦ ﺍﺳﺖ ﻳﮏ ﻓﺮﺻﺖ ﮐﻮﺗﺎﻫﯽ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺑﺪﻫﻴﺪ ﻭ ﺍﺯ ﺁﻥ ﻣﺤﻴﻂ ﻭ ﺗﻨﺶ ﺑﻴﺮﻭﻥ ﺑﺮﻭﻳﺪ، ﺩﺭ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﻋﺼﺒﺎﻧﻴﺖ ﻓﺮﺯﻧﺪﺍﻥ‌ﺗﺎﻥ ﻫﻢ ﺍﻳﻦ ﺭﻭﺵ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ. ﻳﮏ ﺯﻧﮓ ﺗﻔﺮﻳﺢ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺑﺪﻫﻴﺪ ﻭ ﻭﻗﺘﯽ ﺑﺮﮔﺸﺘﻴﺪ ﻭ ﺍﻭﺿﺎﻉ ﺁﺭﺍﻡ ﺷﺪ، ﺭﺍﺟﻊ ﺑﻪ ﻣﺴﺌﻠﻪ‌ﻳﺘﺎﻥ ﺑﺎ ﻫﻢ ﺣﺮﻑ ﺑﺰﻧﻴﺪ.

 

ﻗﺎﻧﻮﻥ ﺩﻫﻢ: ﺑﺎ ﻣﺪﻳﺮﻳﺖ ﺻﺤﻴﺢ ﺍﺣﺴﺎﺳﺎﺕ‌ﺗﺎﻥ ﺍﻟﮕﻮﯼ ﻓﺮﺯﻧﺪﺍﻥ‌ﺗﺎﻥ ﺑﺎﺷﻴﺪ.
ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﭘﻴﺸﺘﺮ ﻫﻢ ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﮔﻔﺘﻪ ﺷﺪ، ﺩﺭ ﺍﻳﻦ ﺷﺮﺍﻳﻂ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺑﻴﺎﻣﻮﺯﻳﺪ ﭼﮕﻮﻧﻪ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﮐﻨﺎﺭ ﺑﻴﺎﻳﺪ ﻭ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺩﺭﺳﺘﯽ ﺑﺮﻭﺯ ﺩﻫﺪ.

ﻣﺜﻼ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ «ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺣﺮﻓﯽ ﻧﺰﻧﻢ ﮐﻪ ﺑﻌﺪﺍ ﭘﺸﻴﻤﻮﻥ ﺑﺸﻢ، ﭼﻨﺪ ﺩﻗﻴﻘﻪ ﻣﯽ‌ﺭﻡ ﺑﻴﺮﻭﻥ ﻗﺪﻡ ﺑﺰﻧﻢ»، «ﺩﻟﻢ ﻣﯽ‌‌ﺧﻮﺍﺩ ﺍﻻﻥ ﺑﺎﻫﺎﺕ ﺣﺮﻑ ﺑﺰﻧﻢ، ﻭﻟﯽ ﭼﻮﻥ ﺍﻻﻥ ﮐﻤﯽ ﻋﺼﺒﯽ‌ﺍﻡ ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻡ ﺍﻻﻥ ﺑﺎﻫﺎﺕ ﺻﺤﺒﺖ ﻧﮑﻨﻢ، ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻡ ﻭﻗﺘﯽ ﺁﺭﻭﻡ ﺷﺪﻡ ﺣﺮﻑ ﺑﺰﻧﻴﻢ.»

ﺍﻋﺘﺮﺍﻑ ﺑﻪ ﺍﻳﻨﮑﻪ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻩ‌ﺍﻳﺪ ﻭ ﺑﺮﺍﯼ ﺁﺭﺍﻡ ﺷﺪﻥ ﻧﻴﺎﺯ ﺑﻪ ﺯﻣﺎﻥ ﺩﺍﺭﻳﺪ، ﻧﺸﺎﻧﻪ ﺿﻌﻒ ﻧﻴﺴﺖ. ﺑﻠﮑﻪ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﺷﻤﺎ ﺍﺣﺴﺎﺳﺎﺕ ﺧﻮﺩ ﺭﺍ ﻣﯽ‌ﺷﻨﺎﺳﻴﺪ ﻭ ﻧﺤﻮﻩ ﮐﻨﺎﺭ ﺁﻣﺪﻥ ﺑﺎ ﺁﻧﻬﺎ ﺭﺍ ﻫﻢ ﻣﯽ‌ﺩﺍﻧﻴﺪ. ﺍﻳﻦ ﻫﻤﺎﻥ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﻓﺮﺯﻧﺪﺗﺎﻥ ﺍﺯ ﺷﻤﺎ ﻣﯽ‌ﺁﻣﻮﺯﺩ‌.
