ﺩﺭﺱ ﺍﻭﻝ- ﺍﺯ ﺧﺸﻢ ﭼﻪ ﻣﯽ ﺩﺍﻧﻴﻢ؟

ﺑﺎﺯ ﻫﻢ  ﻻﺯﻡ ﺍﺳﺖ ﻳﺎﺩﺁﻭﺭﯼ ﮐﻨﻴﻢ ﮐﻪ ﺧﺸﻢ ﻳﮏ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ ﺣﺴﯽِ ﻃﺒﻴﻌﯽ ﺑﻪ ﻓﺸﺎﺭﻫﺎ ﻭ ﻧﺎﺭﺿﺎﻳﺘﯽ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ  ﺑﺮﺍﻳﻤﺎﻥ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ. ﺍﻳﻦ ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﻭﻗﺘﯽ ﺑﻪ ﻫﺮ ﺩﻟﻴﻠﯽ ﻣﺎﻧﻌﯽ ﺩﺭ ﺭﺍﻩ ﺭﺳﻴﺪﻥ ﺑﻪ ﺁﻧﭽﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﺮﺍﻳﻤﺎﻥ ﭘﻴﺶ ﻣﯽ‌ﺁﻳﺪ، ﻋﺼﺒﺎﻧﯽ ﺷﻮﻳﻢ.

  • ﺑﻴﺸﺘﺮ ﻭﻗﺖ‌ﻫﺎ ﺍﻳﻦ ﻣﺎ ﻧﻴﺴﺘﻴﻢ ﮐﻪ ﺗﺼﻤﻴﻢ ﻣﯽ‌ﮔﻴﺮﻳﻢ ﻋﺼﺒﯽ ﺷﻮﻳﻢ، ﻭﻟﯽ ﺑﻬﺮﺣﺎﻝ ﺑﻪ ﻧﺎﮔﺎﻩ ﺣﺴﺶ ﻣﯽ‌ﮐﻨﻴﻢ، ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﺣﺘﯽ ﺗﺎ ﻭﻗﺘﯽ ﺣﺲ ﻋﺼﺒﺎﻧﻴﺖ ﻧﻴﺎﻣﺪﻩ، ﻧﻤﯽ‌ﻓﻬﻤﻴﻢ ﻋﺼﺒﯽ ﺷﺪﻩ‌ﺍﻳﻢ.
  • ﺍﺣﺴﺎﺱ ﻋﺼﺒﺎﻧﻴﺖ ﺍﮔﺮ ﺑﺮﺍﯼ ﻃﻮﻻﻧﯽ ﻣﺪﺕ ﺗﮑﺮﺍﺭ  ﺷﻮﺩ، ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺍﺗﻔﺎﻕ ﻳﺎ ﻣﺤﺮﮐﯽ ﺑﺮﺍﯼ ﺍﻳﻦ ﺣﺲ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﺑﻪ ﺁﻥ ﻣﯽ‌ﮔﻮﻳﻴﻢ «ﻣﻮﻭﺩ». ﺣﺎﻟﺖ‌ﻫﺎﯼ (ﻣﻮﻭﺩﻫﺎﯼ) ﻋﺼﺒﯽ ﺳﺮﻳﻊ‌ﺗﺮ ﺑﻪ ﺍﺣﺴﺎﺱ ﻋﺼﺒﺎﻧﻴﺖ، ﺁﻥ ﻫﻢ ﺍﺯ ﻧﻮﻉ ﺷﺪﻳﺪ ﻣﻨﺠﺮ ﻣﯽ‌ﺷﻮﻧﺪ.
  • ﻭﻗﺘﯽ ﺧﺸﻢ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ، ﺭﻭﯼ ﺻﺪﺍ، ﺣﺮﮐﺎﺕ ﻭ ﻭﺿﻌﻴﺖ ﺻﻮﺭﺕ ﻭ ﻃﺮﺯ ﻓﮑﺮ ﮐﺮﺩﻥ ﺗﺎﺛﻴﺮ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﺗﺎﺯﻩ ﻣﺤﺮﮐﯽ ﺑﺮﺍﯼ ﻭﺍﺭﺩ ﻋﻤﻞ ﺷﺪﻥ ﻫﻢ ﻣﯽ‌ﺷﻮﺩ. ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﻓﮑﺮﻫﺎﻳﯽ ﺩﺍﺭﻳﻢ ﮐﻪ ﺑﻪ ﻋﺼﺒﺎﻧﻴﺖ ﻣﯽ‌ﺭﺳﺪ، ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﻫﻢ ﺁﻥ ﻓﮑﺮﻫﺎ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﻳﮑﺠﺎ ﺑﺮﻭﺯ ﻣﯽ‌ﮐﻨﻨﺪ.
  • ﺩﺍﻧﺸﻤﻨﺪﺍﻥ ﺑﺮ ﺍﻳﻦ ﻋﻘﻴﺪﻩ‌ﺍﻧﺪ ﮐﻪ ﻫﺪﻑ ﺣﺲ‌ﻫﺎﻳﯽ ﻣﺜﻞ ﻋﺼﺒﺎﻧﻴﺖ، ﻫﻤﺎﻫﻨﮕﯽ ﻭ ﺑﺴﻴﺞ ﮐﺮﺩﻥ ﮐﻞ ﺳﻴﺴﺘﻢ ﺑﺪﻥ ﻣﺎ ﺩﺭ ﭘﺎﺳﺦ ﺑﻪ ﻣﺤﻴﻂ ﺍﻃﺮﺍﻑ‌ﻣﺎﻥ ﺍﺳﺖ. ﺍﻳﻦ ﺍﺗﻔﺎﻕ ﻭﻗﺘﯽ ﻣﯽ‌ﺍﻓﺘﺪ ﮐﻪ ﻣﺎ ﻫﻴﭻ ﺷﻨﺎﺧﺖ  ﻭ ﺁﮔﺎﻫﯽ ﺑﻪ ﺍﻳﻦ ﻗﻀﻴﻪ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ. ﻭﻟﯽ ﻣﺎ ﺑﻪ ﺣﺲ‌ﻫﺎﻳﻤﺎﻥ ﺁﮔﺎﻫﯽ ﺩﺍﺭﻳﻢ ﻭ ﻭﺟﻮﺩ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﺷﮑﻠﯽ ﺩﺭ ﺑﺪﻧﻤﺎﻥ ﺩﺭﮎ ﻣﯽ‌ﮐﻨﻴﻢ.
  • ﺩﺭ ﺍﺻﻞ، ﺣﺲ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﺎ ﺧﺸﻢ ﺍﺯ ﻳﮏ ﺧﺎﻧﻮﺍﺩﻩ ﻫﺴﺘﻨﺪ، ﺍﺣﺘﻤﺎﻻ ﺑﺨﺸﯽ ﺍﺯ ﺳﻴﺴﺘﻢ ﺩﻓﺎﻋﯽ ﻣﻐﺰ ﻣﺎ ﻫﺴﺘﻨﺪ ﮐﻪ ﺍﺯ ﻣﻴﻠﻴﻮﻥ‌ﻫﺎ ﺳﺎﻝ ﭘﻴﺶ ﻣﺎ ﺭﺍ ﺩﺭ ﺑﺮﺍﺑﺮ ﻣﺤﻴﻂ، ﺑﺮﺍﯼ ﺑﻘﺎ ﻭ ﺯﻧﺪﻩ ﻣﺎﻧﺪﻥ، ﻣﺤﺎﻓﻈﺖ ﮐﺮﺩﻩ‌ﺍﻧﺪ.

ﺑﺮﺍﯼ ﺩﺍﻧﺴﺘﻦ ﺟﺰﺋﻴﺎﺕ ﺑﻴﺸﺘﺮ، ﺑﺎﻳﺪ ﻧﮕﺎﻫﯽ ﺑﻪ ﻋﻤﻠﮑﺮﺩ ﻣﻐﺰ ﺑﻴﻨﺪﺍﺯﻳﻢ. ﺑﻪ ﺯﺑﺎﻥ ﺳﺎﺩﻩ، ﻣﺎ ﺩﺭ ﻣﻐﺰﻣﺎﻥ ﺩﺭ ﮐﻨﺎﺭ ﺑﺨﺶ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﺑﺨﺶ ﻣﺮﺑﻮﻁ ﺑﻪ «ﺍﺣﺴﺎﺳﺎﺕ» ﻭ ﺑﺨﺶ ﻣﺮﺑﻮﻁ ﺑﻪ «ﺗﻔﮑﺮ» ﺩﺍﺭﻳﻢ.

ﺧﺸﻢ ﻣﺎﻧﻨﺪ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺣﺲ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺗﻮﺳﻂ ﺑﺨﺸﯽ ﺍﺯ ﻣﻐﺰ ﺑﻪ ﻧﺎﻡ ﺳﻴﺴﺘﻢ ﻟﻴﻤﺒﻴﮏ ﻳﺎ ﺳﺎﻣﺎﻧﻪ‌ﯼ ﻋﺼﺒﯽ ﺍﺣﺴﺎﺳﯽ، ﮐﻨﺘﺮﻝ ﻣﯽ‌ﺷﻮﺩ. ﺧﺎﻃﺮﺍﺕ ﻋﺎﻃﻔﯽ ﺩﺭ ﺑﺨﺸﯽ ﺑﻪ ﻧﺎﻡ ﻏﺪﻩ ﺑﺎﺩﺍﻣﯽ ﻣﻐﺰ (ﺁﻣﻴﮕﺪﺍﻻ) ﺫﺧﻴﺮﻩ ﻣﯽ‌ﺷﻮﻧﺪ ﮐﻪ ﻣﺤﻞ ﺁﻥ ﺩﺍﺧﻞ ﺳﻴﺴﺘﻢ ﻟﻴﻤﺒﻴﮏ ﻗﺮﺍﺭ ﺩﺍﺭﺩ.

ﺑﻪ ﺩﻟﻴﻞ ﻭﺟﻮﺩ ﺁﻣﻴﮕﺪﺍﻻ ﻭ ﺩﻳﮕﺮ ﺳﺎﺧﺘﺎﺭﻫﺎﯼ ﻣﻐﺰﺗﺎﻥ، ﻣﻤﮑﻦ ﺍﺳﺖ ﭼﻴﺰﯼ ﮐﻪ ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﻣﻮﺟﺐ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﺷﻤﺎ ﺷﺪﻩ، ﺗﺮﮐﻴﺒﯽ ﺍﺯ ﻣﺤﺮﮐﯽ ﺩﺭ ﺯﻣﺎﻥ ﺣﺎﻝ ﻭ ﺗﺠﺮﺑﻪ‌ﺍﯼ ﻋﺎﻃﻔﯽ ﺍﺯ ﮔﺬﺷﺘﻪ ﺑﺎﺷﺪ. ﺍﻳﻦ «ﺧﺸﻢ ﮐﻬﻨﻪ» ﺗﻮﺳﻂ ﻣﻐﺰ ﺷﻤﺎ ﻭ ﺑﻪ ﺩﻟﻴﻞ ﺣﻤﺎﻳﺖ ﻭ ﻣﺤﺎﻓﻈﺖ ﺍﺯ ﺷﻤﺎ ﻓﻌﺎﻝ ﻣﯽ‌ﺷﻮﺩ. ﻋﺼﺐ‌ﺷﻨﺎﺳﺎﻥ ﺍﻳﻦ ﺣﺎﻓﻈﻪ ﺭﺍ، «ﺣﺎﻓﻈﻪ ﺿﻤﻨﯽ» ﻣﯽ‌ﻧﺎﻣﻨﺪ. ﺍﻳﻦ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺎ ﺑﺪﻭﻥ ﺍﻳﻨﮑﻪ ﺣﺘﯽ ﺑﺪﺍﻧﻴﻢ ﭼﻨﻴﻦ ﺧﺎﻃﺮﻩ‌ﺍﯼ ﺩﺍﺭﻳﻢ، ﺗﺎﺛﻴﺮﺍﺕ ﺁﻥ ﺭﺍ ﺗﺠﺮﺑﻪ ﮐﻨﻴﻢ.

ﺍﻳﻦ ﺑﺮﻋﻬﺪﻩ ﺑﺨﺶ ﺗﻔﮑﺮ ﻣﻐﺰ (ﻫﻤﺎﻥ ﻋﻀﻠﻪ‌ﻫﺎﯼ ﭘﻴﺸﺎﻧﯽ/ﺟﻠﻮﻳﯽ ﻣﻐﺰ) ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺍﺣﺴﺎﺱ ﻋﺼﺒﺎﻧﻴﺘﯽ ﮐﻪ ﺁﻣﻴﮕﺪﺍﻻ ﻭ ﺩﻳﮕﺮ ﺳﺎﺧﺘﺎﺭﻫﺎﯼ ﻣﻐﺰ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ‌ﺍﻧﺪ، ﭼﮕﻮﻧﻪ ﺑﺮﺧﻮﺭﺩ ﮐﻨﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﺑﺸﺮ ﻣﺘﻔﮑﺮ، ﻣﺎ ﻗﺎﺑﻠﻴﺖ ﻳﮕﺎﻧﻪ‌ﺍﯼ ﺩﺭ ﻣﻴﺎﻥ ﺳﺎﻳﺮ ﻣﻮﺟﻮﺩﺍﺕ ﺩﺍﺭﻳﻢ. ﺁﻥ ﻫﻢ ﻗﺎﺑﻠﻴﺖ ﺍﻧﺘﺨﺎﺏ ﭼﮕﻮﻧﻪ ﺑﺮﺧﻮﺭﺩ ﮐﺮﺩﻥ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺗﻤﺎﻥ ﺍﺳﺖ.

 

ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﺍﻟﺰﺍﻣﺎ ﻣﻨﻔﯽ ﻧﻴﺴﺖ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺜﺒﺖ ﻫﻢ ﺑﺎﺷﺪ.
ﺧﺸﻢ ﻳﮏ ﺣﺲ ﻃﺒﻴﻌﯽ ﺑﺮﺍﯼ ﻫﻤﻪ ﻣﺎ ﺁﺩﻡ‌ﻫﺎﺳﺖ. ﺧﺸﻢ ﻧﺸﺎﻧﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﺎ ﺁﺳﻴﺐ ﺩﻳﺪ‌ﻩ‌ﺍﻳﻢ ﻳﺎ ﺍﺣﺴﺎﺱ ﺧﻄﺮ ﻭ ﺗﻬﺪﻳﺪ ﻣﯽ‌ﮐﻨﻴﻢ. ﻳﮏ ﺯﻧﮓ ﺧﻄﺮ ﮐﻪ ﻣﯽ‌ﮔﻮﻳﺪ ﺍﻧﮕﺎﺭ ﺟﺎﻳﯽ ﻳﮏ ﺍﻳﺮﺍﺩﯼ ﻫﺴﺖ، ﻳﮏ ﭼﻴﺰﯼ ﺩﺭﺳﺖ ﭘﻴﺶ ﻧﻤﯽ‌ﺭﻭﺩ. ﺷﺎﻳﺪ ﺯﻧﮓ ﺧﻄﺮ ﺧﻮﺑﯽ ﺑﺎﺷﺪ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﺎ ﻳﮏ ﺗﺼﻤﻴﻢ ﺑﻪ ﺟﺎ ﻭ ﺑﻪ ﻣﻮﻗﻊ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﺣﻔﻆ ﮐﻨﻴﻢ.

ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻳﮏ ﺗﺨﻠﻴﻪ ﺳﺎﻟﻢ ﻭ ﺩﺭﺳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﻧﮕﻪ ﺩﺍﺷﺘﻦ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﻗﻮﺭﺕ ﺩﺍﺩﻥ ﺧﺸﻢ ﻭ ﺣﺮﺹ، ﺍﻧﺮﮊﯼ ﺯﻳﺎﺩﯼ ﻣﯽ‌‌ﮔﻴﺮﺩ ﻭ ﺑﺮﺍﯼ ﺳﻼﻣﺘﯽ ﻫﻢ ﻣﻀﺮ ﺍﺳﺖ. ﺍﮔﺮ ﺧﺸﻢ‌ﻣﺎﻥ ﺭﺍ ﺑﻪ ﺩﺭﺳﺘﯽ ﺁﺯﺍﺩ ﮐﻨﻴﻢ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﺭﺗﺒﺎﻃﺎﺕ ﺑﻬﺘﺮ ﻭ ﺳﺎﻟﻢ‌ﺗﺮﯼ ﺩﺭﺳﺖ ﮐﻨﻴﻢ.

ﻳﮏ ﻣﺜﺎﻝ ﺳﺎﺩﻩ: ﻫﻤﺴﺮ ﺷﻤﺎ ﻋﺎﺩﺕ ﺩﺍﺭﺩ ﻫﻤﻴﺸﻪ ﺑﻪ ﻫﻤﻪ ﻗﺮﺍﺭﻫﺎﻳﺶ ﺩﻳﺮ ﺑﺮﺳﺪ، ﺍﺯ ﺟﻤﻠﻪ ﻗﺮﺍﺭﻫﺎﻳﺶ ﺑﺎ ﺷﻤﺎ، ﻭﻗﺘﯽ ﻗﺮﺍﺭ ﺍﺳﺖ ﺟﺎﻳﯽ ﺑﺎ ﻫﻢ ﺑﺮﻭﻳﺪ. ﺧﺐ ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺷﻤﺎ ﺭﺍ ﻋﺼﺒﺎﻧﯽ ﻭ ﺩﻟﺨﻮﺭ ﮐﻨﺪ. ﭘﺎﺳﺦ ﺷﻤﺎ ﭼﻴﺴﺖ؟ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﻴﺪ ﺑﻬﺎﻧﻪ ﺑﻴﺎﻭﺭﻳﺪ ﻭ ﺩﻳﮕﺮ ﺑﻴﺮﻭﻥ ﻫﻴﭻ ﻗﺮﺍﺭﯼ ﻧﮕﺬﺍﺭﻳﺪ ﻭ ﻫﻤﻪ ﺟﺎ ﺟﺪﺍ ﺟﺪﺍ ﺑﺮﻭﻳﺪ؟ 

ﺧﻮﺏ ﺷﺎﻳﺪ ﮐﺎﺭﻫﺎﯼ ﺩﻳﮕﺮﯼ ﻫﻢ ﺑﺘﻮﺍﻥ ﮐﺮﺩ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ ﭼﻘﺪﺭ ﻧﺎﺭﺍﺣﺖ ﻣﯽ‌ﺷﻮﻳﺪ ﻭﻗﺘﯽ ﺍﻭ ﻫﻤﻴﺸﻪ ﺩﻳﺮ ﻣﯽ‌ﮐﻨﺪ، ﺣﺲ ﻣﯽ‌ﮐﻨﻴﺪ ﺍﺻﻼ ﺷﻤﺎ ﺑﺮﺍﻳﺶ ﺍﻫﻤﻴﺖ ﻧﺪﺍﺭﻳﺪ، ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﺑﺮ ﻣﯽ‌ﺧﻮﺭﺩ.

ﻫﻤﺴﺮ ﺷﻤﺎ ﺁﻥ ﻭﻗﺖ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺧﺎﻃﺮ ﺷﻤﺎ ﻭ ﺭﺍﺑﻄﻪ‌ﺗﺎﻥ ﻭ ﺭﻭﺭﺍﺳﺘﯽ ﺷﻤﺎ ﺧﻮﺩﺵ ﺭﺍ ﮐﻤﯽ ﻋﻮﺽ ﮐﻨﺪ، ﺩﺭ ﺭﺍﺑﻄﻪ‌ﺗﺎﻥ ﻫﻢ ﺗﺎﺛﻴﺮ ﺧﻮﺑﯽ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﺍﻳﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺜﺎﻟﯽ ﺑﺮﺍﯼ ﺍﺳﺘﻔﺎﺩﻩ ﻣﺜﺒﺖ ﺍﺯ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﺑﺎﺷﺪ ﮐﻪ ﺗﺎﺯﻩ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﺷﻤﺎ ﺍﺯ ﺩﺳﺖ ﺧﻮﺩﺗﺎﻥ ﺭﺍﺿﯽ ﻭ ﺧﻮﺷﺤﺎﻝ ﻫﻢ ﺑﺸﻮﻳﺪ. ﭼﻮﻥ ﺍﮔﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺭﺍﺟﻊ ﺑﻪ ﺍﺣﺴﺎﺳﺎﺗﺘﺎﻥ ﺑﺎ ﺑﻘﻴﻪ ﺣﺮﻑ ﺑﺰﻧﻴﺪ، ﻳﻌﻨﯽ ﺩﺍﺭﻳﺪ ﺑﻪ ﻫﻤﻪ ﺩﻧﻴﺎ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﻣﯽ‌ﮔﻮﻳﻴﺪ:

«ﻣﻦ ﺑﺮﺍﯼ ﺧﻮﺩﻡ ﺍﻋﺘﺒﺎﺭ ﻭ ﺍﺭﺯﺵ ﻗﺎﺋﻠﻢ ﻭ ﻣﯽ‌ﺧﻮﺍﻫﻢ ﻫﻤﻪ ﻫﻢ ﺑﺎ ﻣﻦ ﺍﻳﻨﻄﻮﺭﯼ ﺭﻓﺘﺎﺭ ﮐﻨﻨﺪ.»

ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﻫﻢ ﺣﺎﻟﺖ‌ﻫﺎﯼ ﺷﺪﻳﺪﯼ ﺍﺯ ﺧﺸﻢ ﻭ ﻧﻔﺮﺕ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻳﮏ ﺗﺼﻤﻴﻢ ﺧﻮﺏ ﺍﺟﺘﻤﺎﻋﯽ ﺗﺒﺪﻳﻞ ﺷﻮﺩ.

ﻣﺜﺎﻝ ﺟﺎﻟﺒﺶ ﻳﮏ ﻣﻮﺳﺴﻪ‌ﯼ ﺧﻴﺮﻳﻪ ﺁﻣﺮﻳﮑﺎﻳﯽ ﺍﺳﺖ ﺑﻪ ﻧﺎﻡ «ﻣﺎﺩﺭﺍﻥ ﻣﺨﺎﻟﻒ ﺭﺍﻧﻨﺪﮔﯽ ﺩﺭ ﻣﺴﺘﯽ» ﮐﻪ ﺭﺍﻩ‌ﺍﻧﺪﺍﺯﯼ‌ﺍﺵ ﺗﻮﺳﻂ ﺩﻭ ﻣﺎﺩﺭ ﺑﻮﺩﻩ ﮐﻪ ﺑﭽﻪ‌ﻫﺎﻳﺸﺎﻥ ﺭﺍ ﺩﻭ ﺭﺍﻧﻨﺪﻩ ﻣﺴﺖ ﺩﺭ ﺗﺼﺎﺩﻑ ﮐﺸﺘﻪ ﺑﻮﺩﻧﺪ. ﺩﻭﺭ ﻭ ﺑﺮﺧﻮﺩﻣﺎﻥ ﻫﻢ ﺯﻳﺎﺩ ﺑﺎﻳﺪ ﺍﺯ ﺍﻳﻦ ﺧﻴﺮﻳﻪ‌ﻫﺎ ﺑﺎﺷﺪ ﮐﻪ ﻧﻘﻄﻪ ﺷﺮﻭﻋﺶ ﺍﺯ ﻳﮏ ﺧﺸﻢ ﻭ ﺣﺲ ﻧﻔﺮﺕ ﺑﻮﺩﻩ ﺍﺳﺖ.

 

ﮎِﯼ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﻣﺸﮑﻞ‌ﺳﺎﺯ ﻣﯽ‌ﺷﻮﺩ؟
ﻣﺎ ﺍﺯ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺧﺸﻢ ﻣﺜﻞ ﻫﺮ ﺭﻓﺘﺎﺭ  ﺩﻳﮕﺮﯼ ﻣﻨﻈﻮﺭﯼ ﺩﺍﺭﻳﻢ (ﻣﺜﻞ ﻫﺮ ﺭﻓﺘﺎﺭ ﺩﻳﮕﺮﯼ ﮐﻪ ﺍﺯ ﻣﺎ ﺳﺮ ﻣﯽ‌ﺯﻧﺪ). ﻳﻌﻨﯽ ﻭﻗﺘﯽ ﻋﺼﺒﺎﻧﻴﺖ‌ﻣﺎﻥ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻴﻢ، ﻳﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﻣﺘﻮﺟﻪ ﺧﻮﺩﻣﺎﻥ ﮐﻨﻴﻢ ﮐﻪ ﺻﺪﺍﯼ ﻣﺎ ﺭﺍ ﺑﺸﻨﻮﻧﺪ، ﺟﻮﺍﺏ‌ﻣﺎﻥ ﺭﺍ ﺑﺪﻫﻨﺪ، ﻣﺎ ﺭﺍ ﺑﺒﻴﻨﻨﺪ ﻳﺎ ﻣﺸﮑﻼﺕ‌ﻣﺎﻥ ﺭﺍ ﺣﻞ ﮐﻨﻨﺪ.

ﻓﻘﻂ، ﻣﺸﮑﻞ ﺍﻳﻨﺠﺎﺳﺖ ﮐﻪ ﺩﺭ ۹۵% ﺍﺯ ﻣﻮﺍﺭﺩ، ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ، ﻭﺿﻊ ﺭﺍ ﺧﺮﺍﺏ‌ﺗﺮ ﻣﯽ ﮐﻨﺪ.

ﺍﺯ ﻃﺮﻑ ﺩﻳﮕﺮ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺧﺸﻢ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﺎﺭﮐﺮﺩ ﻣﺜﺒﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﺎﻭﺍﻥ ﻭ ﺑﻬﺎ ﻫﻢ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ، ﻭ ﺁﻥ ﻭﻗﺖ ﺍﮔﺮ ﻗﻴﻤﺘﺶ ﺍﺯ ﺳﻮﺩﯼ ﮐﻪ ﻣﯽ‌ﺑﺮﻳﻢ ﺧﻴﻠﯽ ﺑﻴﺸﺘﺮ ﺑﺎﺷﺪ ﭼﻪ؟

ﻧﻤﻮﻧﻪ ﺗﺎﻭﺍﻥ ﻭ ﻗﻴﻤﺘﯽ ﮐﻪ ﺑﺮﺍﯼ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﺑﺎﻳﺪ ﭘﺲ ﺑﺪﻫﻴﻢ:

- ﺳﻼﻣﺘﯽ‌ﻣﺎﻥ: ﮐﺴﺎﻧﻴﮑﻪ ﻓﺸﺎﺭ ﺧﻮﻥ‌ﺷﺎﻥ ﺑﻪ ﺩﻟﻴﻞ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﺧﺸﻢ ﺑﺎﻻ ﻣﯽ‌ﺭﻭﺩ، ﺳﻪ ﺑﺮﺍﺑﺮ ﺑﻴﺸﺘﺮ ﺍﺯ ﺁﺩﻡ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺩﺭ ﻣﻌﺮﺽ ﺑﻴﻤﺎﺭﯼ ﻗﻠﺒﯽ‌ﺍﻧﺪ.

- ﺣﺲ ﺭﺿﺎﻳﺖ‌ ﺍﺯ ﺧﻮﺩﻣﺎﻥ ﻭ ﺍﺣﺘﺮﺍﻣﯽ ﮐﻪ ﭘﻴﺶ ﺑﻘﻴﻪ ﺩﺍﺭﻳﻢ: ﺩﺭﺳﺖ ﺍﺳﺖ ﮐﻪ ﺑﻌﻀﯽ ﺁﺩﻡ‌ﻫﺎ ﻫﻤﻴﺸﻪ ﺑﻪ ﺧﻮﺩﺷﺎﻥ ﺣﻖ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﻋﺼﺒﺎﻧﯽ ﺑﺸﻮﻧﺪ ﻭﻟﯽ ﺑﻴﺸﺘﺮ ﺁﺩﻡ‌ﻫﺎ ﺑﻌﺪ ﺍﺯ ﺍﻳﻨﮑﻪ ﻋﺼﺒﺎﻧﯽ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺭﻓﺘﺎﺭﻫﺎﯼ ﻋﺼﺒﯽ ﺍﺯ ﺧﻮﺩﺷﺎﻥ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ، ﭘﺸﻴﻤﺎﻥ ﺷﺪﻩ ﻭ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﻨﺪ ﭘﻴﺶ ﺧﻮﺩﺷﺎﻥ ﮐﻮﭼﮏ ﻭ ﺑﯽ‌ﺍﻋﺘﺒﺎﺭ ﺷﺪﻩ‌ﺍﻧﺪ. ﻃﺮﻑ ﺩﻳﮕﺮ ﻗﻀﻴﻪ ﻫﻢ، ﺍﻋﺘﺒﺎﺭ ﻭ ﺍﺣﺘﺮﺍﻣﯽ ﺍﺳﺖ ﮐﻪ ﭘﻴﺶ ﺑﻘﻴﻪ ﺩﻭﺳﺘﺎﻥ‌ﻣﺎﻥ، ﻓﺮﺯﻧﺪﺍﻥ‌ﻣﺎﻥ، ﺧﺎﻧﻮﺍﺩﻩ، ﻫﻤﮑﺎﺭﺍﻥ ﻭ ﻫﻤﺴﺎﻳﻪ‌ﻫﺎ ﺿﺎﻳﻊ ﻣﯽ‌ﺷﻮﺩ.

- ﺭﻭﺍﺑﻂ‌ﻣﺎﻥ: ﺷﺎﻳﺪ ﻳﮑﯽ ﺍﺯ ﺑﺪﺗﺮﻳﻦ ﻫﺰﻳﻨﻪ‌ﻫﺎﻳﯽ ﺑﺎﺷﻪ ﮐﻪ ﺑﺎﺑﺖ ﺧﺸﻢ ﻭ ﻋﺼﺒﺎﻧﻴﺖ ﮐﻨﺘﺮﻝ ﻧﺸﺪﻩ ﻣﺠﺒﻮﺭ ﺷﻮﻳﻢ ﺑﺪﻫﻴﻢ، ﺩﻭﺳﺘﺎﻥ  ﻭ ﺭﻭﺍﺑﻂ ﺧﻮﺏ ﺧﺎﻧﻮﺍﺩﮔﯽ‌ﻣﺎﻥ.

- ﭘﻮﻝ، ﺁﺯﺍﺩﯼ ﻭ ﻋﻤﺮﻣﺎﻥ: ﺷﺎﻳﺪ ﺑﻪ ﻧﻈﺮ ﺧﻨﺪﻩ‌ﺩﺍﺭ ﺑﻴﺎﻳﺪ، ﻭﻟﯽ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺠﺮﻣﺎﻥ ﮐﻪ ﺩﺭ ﺯﻧﺪﺍﻥ ﻫﺴﺘﻨﺪ ﺩﺭ ﻳﮏ ﻋﺼﺒﺎﻧﻴﺖ ﻭ ﻋﮑﺲ‌ﺍﻟﻌﻤﻞ ﮐﻨﺘﺮﻝ ﻧﺸﺪﻩ، ﻳﮏ ﺧﺸﻢ ﻣﻬﺎﺭ ﻧﺸﺪﻩ، ﻭﺍﺭﺩ ﺩﻋﻮﺍ ﺷﺪﻩ‌ﺍﻧﺪ ﻭ ﺑﻪ ﺧﺸﻮﻧﺖ ﻓﻴﺰﻳﮑﯽ ﺩﺳﺖ ﺯﺩﻧﺪ ﮐﻪ ﺣﺪﺍﻗﻞ ﺑﻬﺎﯼ ﺁﻥ ﺟﺮﻳﻤﻪ ﻧﻘﺪﯼ ﺍﺳﺖ ﺗﺎ ﺣﺒﺲ ﻭ ﺣﺘﯽ ﺍﻋﺪﺍﻡ!

ﺷﺎﻳﺪ ﻻﺯﻡ ﺑﺎﺷﺪ ﮐﻤﯽ ﻓﮑﺮ ﮐﻨﻴﻢ ﺗﺎ ﺑﺒﻴﻨﻴﻢ ﺗﺎ ﺑﻪ ﺣﺎﻻ ﭼﻘﺪﺭ ﻭ ﮐﺠﺎ ﺑﺮﺍﯼ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻥ ﻫﺰﻳﻨﻪ ﺩﺍﺩﻩ‌ﺍﻳﻢ؟ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﺍﺭﺯﺷﺶ ﺭﺍ ﺩﺍﺷﺘﻪ؟ ﺧﻮﺏ ﺑﻪ ﺁﻥ ﻓﮑﺮ ﮐﻨﻴﺪ! ﺍﻳﻦ ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺍﺳﺖ.
