ﻓﻴﻠﺘﺮﻫﺎﯼ ﻣﺮﺑﻮﻁ ﺑﻪ Sharpness
ﻳﮑﯽ ﺍﺯ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻣﻮﻟﻔﻪ‌ﻫﺎﯼ ﻇﺎﻫﺮﯼ ﺩﺭ ﺗﺼﻮﻳﺮ ﮐﻪ ﻫﻢ ﺩﺭ ﮐﻴﻔﻴﺖ ﺩﻳﺪﺍﺭﯼ ﻣﻮﺛﺮ ﺍﺳﺖ ﻭ ﻫﻢ ﺩﺭ ﺧﻮﺍﻧﺶ ﻭ ﺍﻧﺘﻘﺎﻝ ﻣﻔﺎﻫﻴﻢ ﺗﺼﻮﻳﺮ، ﻣﻴﺰﺍﻥ ﻭﺿﻮﺡ ﻳﺎ Sharpness ﻭ ﺑﺨﺶ‌ﻫﺎﯼ ﻭﺍﺿﺢ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﺍﺳﺖ. ﻳﮏ ﻋﮑﺎﺱ ﺑﺮﺍﺳﺎﺱ ﺗﺠﻬﻴﺰﺍﺗﯽ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﻣﯽ‌ﮐﻨﺪ، ﻫﻤﭽﻨﻴﻦ ﺗﮑﻨﻴﮏ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺯﻣﺎﻥ ﻋﮑﺎﺳﯽ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﮔﻴﺮﺩ، ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺩﺭ ﺗﻌﻴﻴﻦ ﺑﺨﺶ‌ﻫﺎﯼ ﻭﺍﺿﺢ ﻭ ﻣﻴﺰﺍﻥ ﻭﺿﻮﺡ ﻋﮑﺲ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﮐﻨﺘﺮﻝ ﺩﺍﺭﺩ. ﺍﻣﺎ ﺩﺭ ﻣﺮﺣﻠﻪ ﺍﺩﻳﺖ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺑﻌﻀﯽ ﺍﺯ ﺍﻣﮑﺎﻧﺎﺕ ﻧﺮﻡ‌ﺍﻓﺰﺍﺭﯼ، ﺩﺭ ﻣﻴﺰﺍﻥ ﻭﺿﻮﺡ ﺗﺼﻮﻳﺮ ﺩﺧﻞ ﻭ ﺗﺼﺮﻑ ﺩﺍﺷﺖ.

ﺩﺭ ﻧﺮﻡ‌ﺍﻓﺰﺍﺭ ﻓﺘﻮﺷﺎﭖ ﻭ ﺩﺭ ﺑﺨﺶ ﻓﻴﻠﺘﺮﻫﺎ، ﻓﻴﻠﺘﺮﯼ ﺑﺎ ﻋﻨﻮﺍﻥ Sharpen ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺍﺯ ﻧﺎﻡ ﺁﻥ ﻧﻴﺰ ﺑﺮ ﻣﯽ‌ﺁﻳﺪ، ﺍﻣﮑﺎﻥ ﺍﻳﺠﺎﺩ ﻳﺎ ﺍﻓﺰﺍﻳﺶ ﻭﺿﻮﺡ ﺭﺍ ﺩﺭ ﺑﺨﺶ‌ﻫﺎﯼ ﻣﻮﺭﺩ ﻧﻈﺮ ﺩﺭ ﺗﺼﻮﻳﺮ ﻓﺮﺍﻫﻢ ﻣﯽ‌ﺳﺎﺯﺩ. ﺍﻣﺎ ﺑﺎﻳﺪ ﺑﻪ ﺍﻳﻦ ﻧﮑﺘﻪ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺗﻮﺟﻪ ﺩﺍﺷﺖ ﮐﻪ ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﻮﺍﺭﺩ ﻋﺪﻡ ﻭﺿﻮﺡ ﻧﺎﺷﯽ ﺍﺯ ﺍﺷﮑﺎﻻﺕ ﻋﮑﺎﺳﯽ ﺑﻮﺩﻩ ﻭ ﺑﻪ ﻫﻴﭻ ﻭﺟﻪ ﻗﺎﺑﻞ ﺭﻓﻊ ﻧﻴﺴﺘﻨﺪ. ﻟﺬﺍ ﻧﺒﺎﻳﺪ ﺗﻮﻗﻊ ﺩﺍﺷﺖ ﻋﮑﺴﯽ ﮐﻪ ﺩﺭ ﺍﺛﺮ ﺗﮑﺎﻥ ﺩﻭﺭﺑﻴﻦ ﻳﺎ ﺳﻮﮊﻩ ﻭ ﻳﺎ ﺗﻨﻈﻴﻢ ﺍﺷﺘﺒﺎﻩ ﻓﮑﻮﺱ ﻭ ﻳﺎ ﻋﻤﻖ ﻣﻴﺪﺍﻥ ﻭﺿﻮﺡ ﺩﭼﺎﺭ ﻋﺪﻡ ﻭﺿﻮﺡ ﺷﺪﻳﺪ ﺍﺳﺖ ﺭﺍ ﺑﺘﻮﺍﻥ ﮐﺎﻣﻼ ﺍﺻﻼﺡ ﮐﺮﺩ. ﻟﺬﺍ ﺩﺭ ﺯﻣﺎﻥ ﻋﮑﺎﺳﯽ ﺑﺎﻳﺪ ﺣﺪﺍﮐﺜﺮ ﺩﻗﺖ ﺩﺭ ﻣﻮﺭﺩ ﻭﺿﻮﺡ ﻋﮑﺲ ﺻﻮﺭﺕ ﺑﭙﺬﻳﺮﺩ.

ﺩﺭ ﺯﻳﺮ ﻣﺠﻤﻮﻋﻪ ﻓﻴﻠﺘﺮ Sharpen ﺩﻭ ﻓﺮﻣﺎﻥ Sharpen ﻭ Sharpen More ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﺧﻮﺩﮐﺎﺭ ﻋﮑﺲ ﺭﺍ ﺷﺎﺭﭖ‌ﺗﺮ ﻣﯽ‌ﮐﻨﻨﺪ. ﻣﻌﻤﻮﻻ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﮐﻨﺘﺮﻝ ﮐﻤﯽ ﮐﻪ ﺭﻭﯼ ﺍﻳﻦ ﺩﻭ ﻓﺮﻣﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﮐﺎﺭﺑﺮﺍﻥ ﺣﺮﻓﻪ‌ﺍﯼ ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻫﻨﺪ ﺍﺯ ﻓﻴﻠﺘﺮﻫﺎﯼ ﺩﻳﮕﺮ ﺟﻬﺖ ﺷﺎﺭﭖ ﮐﺮﺩﻥ ﮐﻞ ﻳﺎ ﺑﺨﺶ‌ﻫﺎﻳﯽ ﺍﺯ ﺗﺼﻮﻳﺮ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﺎﻳﻨﺪ. 

ﻗﺒﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﺩﺭ ﻣﻮﺭﺩ ﻓﻴﻠﺘﺮﻫﺎ ﻭ ﺩﺳﺘﻮﺭﺍﺕ ﺩﻳﮕﺮ ﺩﺭ ﺯﻣﻴﻨﻪ ﻭﺍﺿﺢ‌ﺳﺎﺯﯼ ﺻﺤﺒﺖ ﮐﻨﻴﻢ، ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻓﺮﺍﻳﻨﺪﯼ ﮐﻪ ﻣﻌﻤﻮﻻ ﺟﻬﺖ ﻭﺍﺿﺢ‌ﺳﺎﺯﯼ ﺍﺗﻔﺎﻕ ﻣﯽ‌ﺍﻓﺘﺪ ﺍﺷﺎﺭﻩ ﮐﻨﻴﻢ. ﻣﻌﻤﻮﻻ ﺑﺮﺍﯼ ﻧﺮﻡ‌ﺍﻓﺰﺍﺭﻫﺎ ﻭﺿﻮﺡ ﻋﺒﺎﺭﺕ ﺍﺳﺖ ﺍﺯ ﮐﻨﺘﺮﺍﺳﺖ ﺗﺼﻮﻳﺮ (ﺗﻀﺎﺩ ﻧﻮﺭ ﻭ ﺭﻧﮓ) ﺩﺭ ﻟﺒﻪ‌ﻫﺎﯼ ﺍﺟﺴﺎﻡ، ﻳﻌﻨﯽ ﻫﺮ ﭼﻪ ﻭﺿﻮﺡ ﺑﻴﺸﺘﺮ ﺑﺎﺷﺪ، ﺗﻮﻧﺎﻟﻴﺘﻪ ﺭﻧﮕﯽ ﻳﮏ ﺳﻤﺖ ﻟﺒﻪ ﺑﺎ ﺳﻤﺖ ﺩﻳﮕﺮ ﺗﻔﺎﻭﺕ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﺩ. ﻣﺜﻼ ﺩﺭ ﺗﺼﻮﻳﺮ ﺯﻳﺮ ﺩﺭ ﻟﺒﻪ‌ﻫﺎﯼ ﻣﺴﺘﻄﻴﻠﯽ ﮐﻪ ﻳﮏ ﺳﻮ ﮐﺎﻣﻼ ﺳﻔﻴﺪ ﻭ ﺳﻮﯼ ﺩﻳﮕﺮ ﮐﺎﻣﻼ ﺳﻴﺎﻩ ﺍﺳﺖ، ﻭﺿﻮﺡ ﻟﺒﻪ ﺑﻴﺸﺘﺮ ﺍﺳﺖ ﺗﺎ ﺗﺼﻮﻳﺮﯼ ﮐﻪ ﺩﺭ ﻟﺒﻪ‌ﻫﺎﯼ ﻣﺴﺘﻄﻴﻞ، ﮐﻨﺘﺮﺍﺳﺖ ﮐﻤﺘﺮﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺩﺭ ﻭﺍﻗﻊ ﻟﺒﻪ‌ﻫﺎﯼ ﻣﺴﺘﻄﻴﻞ ﺩﻭﻡ ﮐﺎﻣﻼ ﺩﺭ ﻳﮏ ﺳﻮﯼ ﺳﻔﻴﺪ ﻭ ﺩﺭ ﺳﻮﯼ ﺩﻳﮕﺮ ﺳﻴﺎﻩ ﻧﻴﺴﺘﻨﺪ ﻭ ﺑﺨﺶ‌ﻫﺎﯼ ﺧﺎﮐﺴﺘﺮﯼ ﻧﻴﺰ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﺎﻋﺚ ﭘﺎﻳﻴﻦ ﺁﻣﺪﻥ ﮐﻨﺘﺮﺍﺳﺖ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﻭﺿﻮﺡ ﺩﺭ ﻟﺒﻪ‌ﻫﺎﯼ ﻣﺴﺘﻄﻴﻞ ﺩﻭﻡ ﻣﯽ‌ﺷﻮﺩ.

ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺗﻌﺮﻳﻒ ﻓﻮﻕ، ﺑﻪ ﺩﺭﺳﺘﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﺣﺪﺱ ﺯﺩ ﮐﻪ ﻧﺮﻡ‌ﺍﻓﺰﺍﺭﻫﺎ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﻓﺮﺁﻳﻨﺪ ﻭﺍﺿﺢ‌ﺳﺎﺯﯼ ﺑﺎﻳﺪ ﺍﺑﺘﺪﺍ ﻟﺒﻪ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺗﺼﻮﻳﺮ ﭘﻴﺪﺍ ﮐﻨﻨﺪ ﻭ ﺳﭙﺲ ﮐﻨﺘﺮﺍﺳﺖ ﺭﺍ ﺩﺭ ﺍﻃﺮﺍﻑ ﻟﺒﻪ‌ﻫﺎ ﺍﻓﺰﺍﻳﺶ ﺩﻫﻨﺪ. ﺍﻳﻦ ﺭﻭﺵ ﺍﺻﻞ ﮐﻠﯽ ﺍﻓﺰﺍﻳﺶ ﺷﺎﺭﭘﻨﺲ ﺩﺭ ﻳﮏ ﺗﺼﻮﻳﺮ ﺍﺳﺖ.

ﻳﮑﯽ ﺍﺯ ﺭﺍﻳﺞ‌ﺗﺮﻳﻦ ﺭﻭﺵ‌ﻫﺎ ﺑﺮﺍﯼ ﺷﺎﺭﭖ ﮐﺮﺩﻥ ﺗﺼﻮﻳﺮ ﺩﺭ ﻧﺮﻡ‌ﺍﻓﺰﺍﺭ ﻓﺘﻮﺷﺎﭖ، ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻓﻴﻠﺘﺮ Unsharp Mask ﺩﺭ ﺯﻳﺮ ﻣﺠﻤﻮﻋﻪ ﻓﻴﻠﺘﺮ Sharpen ﺍﺳﺖ. ﺯﻣﺎﻧﯽ ﮐﻪ ﺭﻭﯼ ﺍﻳﻦ ﻓﻴﻠﺘﺮ ﮐﻠﻴﮏ ﮐﻨﻴﻢ، ﭘﻨﺠﺮﻩ‌ﺍﯼ ﻣﺸﺎﺑﻪ ﺗﺼﻮﻳﺮ ﺯﻳﺮ ﺑﺎﺯ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺁﻥ ۳ ﭘﺎﺭﺍﻣﺘﺮ ﺍﺻﻠﯽ ﻗﺎﺑﻞ ﺗﻨﻈﻴﻢ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

ﺩﺭ ﺍﻳﻦ ﭘﻨﺠﺮﻩ، ﺩﺭ ﻣﺮﺑﻌﯽ ﮐﻞ ﻳﺎ ﺑﺨﺸﯽ ﺍﺯ ﺗﺼﻮﻳﺮ ﺩﻳﺪﻩ ﻣﯽ‌ﺷﻮﺩ ﺗﺎ ﺑﺎ ﺗﻐﻴﻴﺮ ﭘﺎﺭﺍﻣﺘﺮﻫﺎ، ﮐﺎﺭﺑﺮ ﺍﻣﮑﺎﻥ ﺩﻳﺪﻥ ﻧﺘﻴﺠﻪ ﺭﺍ ﺩﺭ ﮐﻞ ﻳﺎ ﺑﺨﺶ‌ﻫﺎﻳﯽ ﺍﺯ ﺗﺼﻮﻳﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﮐﺎﺭﺑﺮ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺩﮐﻤﻪ‌ﻫﺎﯼ ﻣﺜﺒﺖ ﻳﺎ ﻣﻨﻔﯽ ﺩﺭ ﺯﻳﺮ ﻣﺮﺑﻊ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺰﺭﮔﻨﻤﺎﻳﯽ ﺗﺼﻮﻳﺮ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﺪ. ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﺷﻮﺩ ﺩﺭ ﺯﻣﺎﻥ ﻋﻤﻠﻴﺎﺕ ﺷﺎﺭﭖ ﮐﺮﺩﻥ، ﺣﺘﻤﺎ ﺗﺼﻮﻳﺮ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺑﺰﺭﮒ، ﻣﺜﻼ ۱۰۰ ﺩﺭﺻﺪ ﻳﺎ ﺑﺰﺭﮒ‌ﺗﺮ ﺑﺮﺭﺳﯽ ﮐﺮﺩ ﺯﻳﺮﺍ ﺷﺎﺭﭖ ﮐﺮﺩﻥ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎ ﺗﺨﺮﻳﺐ‌ﻫﺎﻳﯽ ﺩﺭ ﺗﺼﻮﻳﺮ (ﻣﺜﻼ ﺍﻳﺠﺎﺩ ﻫﺎﻟﻪ ﺳﻔﻴﺪ ﺩﺭ ﺍﻃﺮﺍﻑ ﻟﺒﻪ‌ﻫﺎ ﻭ ﻳﺎ ﺍﻓﺰﺍﻳﺶ ﻧﻮﻳﺰ) ﻫﻤﺮﺍﻩ ﺑﺎﺷﺪ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺗﺼﻮﻳﺮ ﮐﻮﭼﮏ ﺍﻳﻦ ﺗﻐﻴﻴﺮﺍﺕ ﺩﻳﺪﻩ ﻧﺸﻮﻧﺪ.

ﭘﺎﺭﺍﻣﺘﺮ Threshold، ﺩﺭ ﺗﻌﻴﻴﻦ ﻟﺒﻪ‌ﻫﺎ ﻧﻘﺶ ﺩﺍﺭﺩ. ﺑﺮﺍﯼ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻟﺒﻪ‌ﻫﺎ، ﻧﺮﻡ‌ﺍﻓﺰﺍﺭ ﻓﺘﻮﺷﺎﭖ ﺑﻪ ﺻﻮﺭﺕ ﮐﻠﯽ، ﺗﻐﻴﻴﺮﺍﺕ ﮐﻨﺘﺮﺍﺳﺖ ﻧﺎﮔﻬﺎﻧﯽ ﺭﺍ ﺩﺭ ﺗﺼﻮﻳﺮ ﺭﺻﺪ ﻣﯽ‌ﮐﻨﺪ. ﺟﺎﻳﯽ ﺩﺭ ﺗﺼﻮﻳﺮ ﮐﻪ ﺑﺎ ﺗﻐﻴﻴﺮ ﮐﻨﺘﺮﺍﺳﺖ ﻧﺎﮔﻬﺎﻧﯽ ﻫﻤﺮﺍﻩ ﺍﺳﺖ، ﻟﺒﻪ ﻓﺮﺽ ﻣﯽ‌ﺷﻮﺩ.

ﻓﻴﻠﺘﺮ Blur

ﺩﺭ ﺯﻳﺮﻣﺠﻤﻮﻋﻪ ﻓﻴﻠﺘﺮ Blur، ﺍﮔﺮ ﺭﻭﯼ ﻓﺮﻣﺎﻥ Blur ﮐﻠﻴﮏ ﮐﻨﻴﻢ، ﺗﺼﻮﻳﺮ ﺗﺎ ﺣﺪﯼ ﻏﻴﺮ ﻭﺍﺿﺢ ﻳﺎ ﺍﺻﻄﻼﺣﺎ Blur ﺧﻮﺍﻫﺪ ﺷﺪ. ﻓﺮﻣﺎﻥ Blur More ﻧﻴﺰ ﺍﺛﺮﯼ ﻣﺸﺎﺑﻪ Blur ﺩﺍﺭﺩ ﺍﻣﺎ ۵-۶ ﺑﺮﺍﺑﺮ ﻗﻮﯼ‌ﺗﺮ ﺍﺳﺖ. ﻣﻌﻤﻮﻻ ﺍﺯ ﺍﻳﻦ ﺩﻭ ﻓﺮﻣﺎﻥ ﺟﻬﺖ ﻧﺮﻡ‌ﺗﺮ ﮐﺮﺩﻥ ﺧﻴﻠﯽ ﺧﻔﻴﻒ ﻋﮑﺲ ﻭ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩﻥ ﻧﻮﻳﺰ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﻣﺎ ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﮐﺎﺭﺑﺮ ﮐﻨﺘﺮﻟﯽ ﺭﻭﯼ ﻧﺤﻮﻩ ﺍﻋﻤﺎﻝ ﺁﻧﻬﺎ ﻧﺪﺍﺭﺩ، ﺩﺭ ﺩﻧﻴﺎﯼ ﺣﺮﻓﻪ‌ﺍﯼ ﻭ ﺟﻬﺖ ﻏﻴﺮ ﻭﺍﺿﺢ ﮐﺮﺩﻥ ﻳﮏ ﻋﮑﺲ ﻳﺎ ﺑﺨﺶ‌ﻫﺎﻳﯽ ﺍﺯ ﺁﻥ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﭼﻨﺪﺍﻧﯽ ﻧﺪﺍﺭﺩ. 

- Box Blur: ﺑﺮﺍﺳﺎﺱ ﻣﻴﺎﻧﮕﻴﻦ ﭘﻴﮑﺴﻞ‌ﻫﺎﯼ ﻣﺠﺎﻭﺭ، ﺑﺎﻋﺚ ﻏﻴﺮ ﻭﺍﺿﺢ ﺷﺪﻥ ﺗﺼﻮﻳﺮ ﻣﯽ‌ﺷﻮﺩ. ﻣﻌﻤﻮﻻ ﺍﻳﻦ ﻓﻴﻠﺘﺮ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺟﻠﻮﻩ‌ﻫﺎﯼ ﻭﻳﮋﻩ ﮐﺎﺭﺑﺮﺩ ﺩﺍﺭﺩ. ﻣﯽ‌ﺗﻮﺍﻥ ﺳﺎﻳﺰ ﺑﺨﺸﯽ ﺭﺍ ﮐﻪ ﺑﺮﺍﯼ ﻣﻴﺎﻧﮕﻴﻦ ﮔﺮﻓﺘﻦ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ ﺑﻴﺸﺘﺮ ﮐﺮﺩ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺍﺛﺮ ﻧﺎﻭﺍﺿﺢ ﺷﺪﻥ ﺩﺭ ﻋﮑﺲ ﻧﻴﺰ ﺑﻴﺸﺘﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻳﻌﻨﯽ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ، ﺗﺼﻮﻳﺮ ﻏﻴﺮ ﻭﺍﺿﺢ‌ﺗﺮ ﺧﻮﺍﻫﺪ ﺷﺪ.

- Gaussian Blur: ﺍﻳﻦ ﻓﻴﻠﺘﺮ ﻳﮑﯽ ﺍﺯ ﻓﻴﻠﺘﺮﻫﺎﯼ ﺳﺎﺩﻩ‌ﺍﯼ‌ ﺍﺳﺖ ﮐﻪ ﻋﮑﺎﺳﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺮﺍﯼ ﻏﻴﺮ ﻭﺍﺿﺢ ﺳﺎﺧﺘﻦ ﺑﺨﺶ‌ﻫﺎﻳﯽ ﺍﺯ ﻋﮑﺲ ﻭ ﻳﺎ ﻧﺮﻡ‌ﺗﺮ ﮐﺮﺩﻥ ﺣﺎﺷﻴﻪ ﻣﺎﺳﮏ‌ﻫﺎ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﺩﻫﻨﺪ. ﺍﻳﻦ ﻓﻴﻠﺘﺮ ﭘﺲ ﺍﺯ ﻣﻴﺎﻧﮕﻴﻦ ﮔﺮﻓﺘﻦ ﺍﺯ ﭘﻴﮑﺴﻞ‌ﻫﺎﯼ ﻣﺠﺎﻭﺭ، ﺑﺎ ﺷﮑﻠﯽ ﺯﻧﮕﻮﻟﻪ‌ﺍﯼ، ﺗﺼﻮﻳﺮ ﺭﺍ ﻧﺎﻭﺍﺿﺢ ﻣﯽ‌ﮐﻨﺪ. ﻫﺮ ﭼﻪ ﺍﺳﻼﻳﺪﺭ ﻣﺮﺑﻮﻁ ﺑﻪ ﺍﻳﻦ ﻓﻴﻠﺘﺮ ﺭﺍ ﺑﻴﺸﺘﺮ ﺑﻪ ﺳﻤﺖ ﺭﺍﺳﺖ ﺑﮑﺸﻴﻢ (ﻣﻘﺪﺍﺭ ﺁﻥ ﺭﺍ ﺑﻴﺸﺘﺮ ﮐﻨﻴﻢ)، ﻣﻴﺰﺍﻥ ﺗﺎﺛﻴﺮ ﻓﻴﻠﺘﺮ ﻧﻴﺰ ﺑﻴﺸﺘﺮ ﺧﻮﺍﻫﺪ ﺷﺪ. ﮔﺎﻫﯽ ﺍﻳﻦ ﻓﻴﻠﺘﺮ ﻣﻮﺟﺐ ﺍﻳﺠﺎﺩ ﺍﺛﺮ Banding ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺗﺼﻮﻳﺮ ﺑﻪ ﺻﻮﺭﺕ ﻧﻮﺍﺭﯼ ﻭ ﻻﻳﻪ ﻻﻳﻪ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ.

ﻧﮑﺘﻪ: ﺍﮔﺮ ﻓﻴﻠﺘﺮﻫﺎﯼ Gaussian Blur ﻭ Box Blur ﻭ Motion Blur ﻭ Shape Blur ﺭﺍ ﺩﺭ ﺑﺨﺸﯽ ﺍﺯ ﺗﺼﻮﻳﺮ ﮐﻪ ﺍﻧﺘﺨﺎﺏ ﺷﺪﻩ ﺍﻋﻤﺎﻝ ﮐﻨﻴﻢ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺣﺎﺷﻴﻪ‌ﻫﺎ ﺑﺎ ﺗﺎﺛﻴﺮﺍﺕ ﻧﺎﺧﻮﺍﺳﺘﻪ ﻭ ﻧﺎﻣﻄﻠﻮﺏ ﻣﻮﺍﺟﻪ ﺷﻮﻳﻢ. ﺍﻳﻦ ﺑﻪ ﺍﻳﻦ ﺳﺒﺐ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻓﻴﻠﺘﺮﻫﺎ، ﺑﺨﺶ‌ﻫﺎﯼ ﺧﺎﺭﺝ ﺍﺯ ﻗﺴﻤﺖ ﺍﻧﺘﺨﺎﺏ ﺷﺪﻩ ﻧﻴﺰ ﺑﺮﺍﯼ ﻣﻴﺎﻧﮕﻴﻦ ﮔﺮﻓﺘﻦ ﻭ ﻏﻴﺮ ﻭﺍﺿﺢ ﮐﺮﺩﻥ ﺑﺨﺶ ﺍﻧﺘﺨﺎﺏ ﺷﺪﻩ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﺩﺭ ﺻﻮﺭﺕ ﺑﺮﻭﺯ ﭼﻨﻴﻦ ﺍﺷﮑﺎﻻﺗﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﻓﻴﻠﺘﺮﻫﺎﯼ Lens Blur ﻳﺎ Smart Blur ﺑﺮﺍﯼ ﻧﺎﻭﺍﺿﺢ ﮐﺮﺩﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﻮﺩ.

- Lens Blur: ﻳﮑﯽ ﺍﺯ ﻓﻴﻠﺘﺮﻫﺎﯼ ﻗﻮﯼ ﺍﺳﺖ ﮐﻪ ﻋﮑﺎﺳﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺟﻬﺖ ﺷﺒﻴﻪ‌ﺳﺎﺯﯼ ﻗﺴﻤﺖ‌ﻫﺎﯼ ﻧﺎﻭﺍﺿﺢ ﺩﺭ ﻋﮑﺲ ﻧﺎﺷﯽ ﺍﺯ ﻋﻤﻖ ﻣﻴﺪﺍﻥ ﻭﺿﻮﺡ ﮐﻢ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﺩﻫﻨﺪ. ﺑﺎ ﺗﻨﻈﻴﻤﺎﺕ ﻭ ﮐﻨﺘﺮﻟﯽ ﮐﻪ ﮐﺎﺭﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﻓﻴﻠﺘﺮ ﺩﺍﺭﺩ، ﻣﺸﺨﺺ ﻣﯽ‌ﺷﻮﺩ ﮐﺪﺍﻡ ﺑﺨﺶ‌ﻫﺎﯼ ﻋﮑﺲ ﺩﺭ ﺧﺎﺭﺝ ﺍﺯ ﻋﻤﻖ ﻣﻴﺪﺍﻥ ﻭﺿﻮﺡ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﻏﻴﺮ ﻭﺍﺿﺢ ﺷﻮﻧﺪ.
