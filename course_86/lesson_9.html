ﺩﺭﺱ ﻧﻬﻢ- ﻣﻔﺎﻫﻴﻢ ﺍﺻﻠﯽ ﻭ ﺍﺻﻮﻝ ﺍﺑﺘﺪﺍﻳﯽ ﺣﺴﺎﺑﺪﺍﺭﯼ

ﺿﻤﻦ ﺧﺴﺘﻪ ﻧﺒﺎﺷﻴﺪ ﺑﻪ ﺷﻤﺎ ﮐﺎﺭﺁﻣﻮﺯ ﮔﺮﺍﻣﯽ، ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺍﻳﻦ ﺩﺭﺱ، ﺁﺧﺮﻳﻦ ﺩﺭﺱ ﺍﺯ ﺩﻭﺭﻩ‌ﯼ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺩﺭﺳﻨﺎﻣﻪ ﺍﺳﺖ. ﻫﻤﺎﻧﻨﺪ ﺑﻴﺸﺘﺮ ﻋﻠﻮﻡ، ﺣﺴﺎﺑﺪﺍﺭﯼ ﻧﻴﺰ ﺑﺮ ﭘﺎﻳﻪ ﻳﮏ ﺳﺮﯼ ﻣﻔﺮﻭﺿﺎﺕ ﻭ ﺍﺻﻮﻝ ﺑﻨﺎ ﺷﺪﻩ ﺍﺳﺖ.  ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﻫﺪﻑ ﻣﺎ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ، ﺁﻣﻮﺯﺵ ﺍﺻﻮﻝ ﮐﻠﯽ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺑﻪ ﻃﻮﺭ ﻣﻘﺪﻣﺎﺗﯽ ﺑﺮﺍﯼ ﻫﻤﻪ ﺑﻮﺩﻩ، ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺍﻳﻦ ﺍﺻﻮﻝ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ ﻭ ﺑﺎ ﺁﻧﻬﺎ ﺁﺷﻨﺎ ﻣﯽ‌ﺷﻮﻳﻢ.

 

ﻣﻔﺮﻭﺿﺎﺕ ﺣﺴﺎﺑﺪﺍﺭﯼ
ﭘﺎﻳﻪ ﺍﺻﻮﻝ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻣﻔﺮﻭﺿﺎﺕ ﻫﺴﺘﻨﺪ ﻭ ﻣﺒﻨﺎﯼ ﺗﻬﻴﻪ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺎﻟﯽ ﺭﺍ ﺩﺭ ﻣﻮﺳﺴﺎﺕ ﺍﻗﺘﺼﺎﺩﯼ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﻨﺪ. ﺣﺴﺎﺑﺪﺍﺭﯼ ﺩﺍﺭﺍﯼ ۵ ﻓﺮﺽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﺑﻪ ﻃﻮﺭ ﮐﻮﺗﺎﻩ ﻫﻤﻪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺑﺮﺭﺳﯽ ﻣﯽ‌ﮐﻨﻴﻢ.

۱. ﻓﺮﺽ ﺗﻔﮑﻴﮏ ﺷﺨﺼﻴﺖ
ﺑﺮ ﺍﺳﺎﺱ ﺍﻳﻦ ﺍﺻﻞ ﻫﺮ ﻣﻮﺳﺴﻪ ﺩﺍﺭﺍﯼ ﺷﺨﺼﻴﺘﯽ ﻣﺴﺘﻘﻞ ﺍﺯ ﻣﺎﻟﮏ ﻳﺎ ﻣﺎﻟﮑﺎﻥ ﺁﻥ ﺩﺍﺭﺩ ﻭ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﻣﺎﻟﯽ ﺁﻥ ﺑﻪ ﺍﻋﺘﺒﺎﺭ ﺷﺨﺼﻴﺖ ﻣﺴﺘﻘﻞ ﺁﻥ ﺗﺠﺰﻳﻪ ﻭ ﺗﺤﻠﻴﻞ ﻣﯽ‌ﺷﻮﺩ.

ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ ﺍﮔﺮ ﺷﺨﺼﯽ ﺩﺍﺭﺍﯼ ﭼﻨﺪ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺑﺎﺷﺪ، ﺑﺮﺍﯼ ﻫﺮ ﻳﮏ ﺍﺯ ﺍﻳﻦ ﻭﺍﺣﺪ‌ﻫﺎ ﺳﻴﺴﺘﻢ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺟﺪﺍ ﻭ ﻣﺴﺘﻘﻞ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻪ ﻣﯽ‌ﺷﻮﺩ. ﻫﻤﭽﻨﻴﻦ ﻫﻴﭻ ﻳﮏ ﺍﺯ ﺍﻣﻮﺍﻝ ﺷﺨﺼﯽ ﺍﻳﻦ ﻓﺮﺩ ﺩﺭ ﻫﻴﭻ ﮐﺪﺍﻡ ﺍﺯ ﻣﻮﺳﺴﺎﺕ ﺛﺒﺖ ﻧﻤﯽ‌ﺷﻮﻧﺪ.

۲. ﻓﺮﺽ ﺗﺪﺍﻭﻡ ﻓﻌﺎﻟﻴﺖ
ﺑﺮ ﺍﺳﺎﺱ ﺍﻳﻦ ﻓﺮﺽ ﻣﻮﺳﺴﻪ ﺩﺭ ﺁﻳﻨﺪﻩ ﻋﻤﻠﻴﺎﺕ ﺧﻮﺩ ﺭﺍ ﺍﺩﺍﻣﻪ ﺧﻮﺍﻫﺪ ﺩﺍﺩ ﺍﻧﺤﻼﻝ ﻳﺎ ﺗﻮﻗﻒ ﻓﻌﺎﻟﻴﺖ ﻧﺪﺍﺭﺩ. ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﺍﺳﺖ ﮐﻪ ﺑﻌﻀﯽ ﺍﺯ ﺣﺴﺎﺏ‌ﻫﺎ ﺩﺭ ﭘﺎﻳﺎﻥ ﻳﮏ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺑﻪ ﺩﻭﺭﻩ ﺑﻌﺪﯼ ﺍﻧﺘﻘﺎﻝ ﻣﯽ‌ﻳﺎﺑﺪ. ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺩﺭ ﺻﻮﺭﺕ ﻋﺪﻡ ﻭﺟﻮﺩ ﺍﻳﻦ ﻓﺮﺽ، ﺗﻤﺎﻳﺰ ﺑﻴﻦ ﺍﻗﻼﻡ ﺟﺎﺭﯼ ﻭ ﻏﻴﺮ ﺟﺎﺭﯼ ﺍﻫﻤﻴﺖ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﺪ. ﺩﺭ ﺣﺎﻟﺘﯽ ﮐﻪ ﻣﻮﺳﺴﻪ ﺩﺭ ﺣﺎﻝ ﻓﺮﻭﺵ ﻳﺎ ﺗﺼﻔﻴﻪ ﺑﺎﺷﺪ، ﺩﻳﮕﺮ ﺍﻳﻦ ﺍﺻﻞ ﻣﺼﺪﺍﻕ ﻧﺪﺍﺭﺩ.

۳. ﻓﺮﺽ ﺩﻭﺭﻩ ﻣﺎﻟﯽ
ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻧﺘﺎﻳﺞ ﻭﺍﻗﻌﯽ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﻳﮏ ﺷﺮﮐﺖ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﭘﺎﻳﺎﻥ ﺁﻥ ﻭ ﭘﺲ ﺍﺯ ﻭﺻﻮﻝ ﮐﻠﻴﻪ ﻣﻄﺎﻟﺒﺎﺕ، ﻭ ﻓﺮﻭﺵ ﺩﺍﺭﺍﻳﯽ‌ﻫﺎ ﺑﻪ ﻃﻮﺭ ﺩﻗﻴﻖ ﺑﻴﺎﻥ ﮐﺮﺩ. ﻭﻟﯽ ﻣﻌﻤﻮﻻ ﻋﻤﺮ ﻳﮏ ﻣﻮﺳﺴﻪ ﺭﺍ ﺑﻪ ﺩﻭﺭﻩ‌ﻫﺎﯼ ﺯﻣﺎﻧﯽ ﮐﻮﺗﺎﻩ‌ﺗﺮ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﮐﻨﻨﺪ. ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ ﺳﻪ ﻣﺎﻩ ﻳﺎ ﻳﮏ ﺳﺎﻝ ﻭ ﺑﺮﺍﯼ ﻫﺮ ﺩﻭﺭﻩ ﮔﺰﺍﺭﺵ ﻣﺠﺰﺍ ﺗﻬﻴﻪ ﻣﯽ‌ﮐﻨﻨﺪ.

ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﺟﺒﺎﺭﯼ ﺑﺮﺍﯼ ﺍﻧﻄﺒﺎﻕ ﺳﺎﻝ ﻣﺎﻟﯽ ﻣﻮﺳﺴﻪ ﺑﺎ ﺳﺎﻝ ﺗﻘﻮﻳﻤﯽ ﻧﻴﺴﺖ. ﻳﻌﻨﯽ ﺍﻟﺰﺍﻣﯽ ﻧﻴﺴﺖ ﮐﻪ ﻣﺜﻼ ﺩﺭ ﺍﻳﺮﺍﻥ ﺳﺎﻝ ﻣﺎﻟﯽ ﺍﻭﻝ ﻓﺮﻭﺭﺩﻳﻦ ﺷﺮﻭﻉ ﺷﺪﻩ ﻭ ﺁﺧﺮ ﺍﺳﻔﻨﺪ ﭘﺎﻳﺎﻥ ﺑﻴﺎﺑﺪ.

۴. ﻓﺮﺽ ﻳﺎ ﻣﺒﻨﺎﯼ ﺗﻌﻬﺪﯼ
ﺑﺮ ﺍﺳﺎﺱ ﺍﻳﻦ ﺍﺻﻞ ﺩﺭﺁﻣﺪ‌ﻫﺎ ﺑﻪ ﻣﺤﺾ ﺗﺤﻘﻖ ﻭ ﻫﺰﻳﻨﻪ‌ﻫﺎ ﺑﻪ ﻣﺤﺾ ﺗﺤﻤﻴﻞ ﺑﺪﻭﻥ ﺗﻮﺟﻪ ﺑﻪ ﺯﻣﺎﻥ ﺩﺭﻳﺎﻓﺖ ﻳﺎ ﭘﺮﺩﺍﺧﺖ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﻧﺪ.

۵. ﻓﺮﺽ ﻭﺍﺣﺪ ﭘﻮﻝ
ﺍﻳﻦ ﺍﺻﻞ ﻳﮑﯽ ﺍﺯ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﺍﺻﻮﻝ ﭘﺎﻳﻪ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺍﺳﺖ ﻭ ﺑﺮ ﺍﺳﺎﺱ ﺁﻥ ﮐﻠﻴﻪ ﺁﺛﺎﺭ ﻭ ﻧﺘﺎﻳﺞ ﻣﻌﺎﻣﻼﺕ ﻭ ﻋﻤﻠﻴﺎﺕ ﺑﺎﻧﮑﯽ ﻣﻮﺳﺴﻪ ﺑﺎﻳﺪ ﺑﺮ ﺣﺴﺐ ﭘﻮﻝ ﮔﺰﺍﺭﺵ ﺷﻮﺩ.

ﺍﻣﻮﺍﻝ ﻣﺨﺘﻠﻒ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﻣﻘﻴﺎﺱ‌ﻫﺎﯼ ﻣﺘﻔﺎﻭﺕ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ ﮐﺮﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺯﻣﻴﻦ ﺭﺍ ﺑﺎ ﻣﺘﺮ ﻣﺮﺑﻊ، ﺑﻨﺰﻳﻦ ﻭ ﮔﺎﺯﻭﻳﻴﻞ ﺭﺍ ﺑﺎ ﻟﻴﺘﺮ. ﻭﻟﯽ ﺍﮔﺮ ﺩﺭ ﺗﺮﺍﺯﻧﺎﻣﻪ ﻫﻢ ﺍﻳﻨﮕﻮﻧﻪ ﺍﺭﺍﺋﻪ ﺷﻮﻧﺪ ﺩﻳﮕﺮ ﻗﺎﺑﻞ ﻣﻘﺎﻳﺴﻪ ﺑﺎ ﻳﮑﺪﻳﮕﺮ ﻧﻴﺴﺘﻨﺪ. ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﻭﺍﺣﺪ ﭘﻮﻝ ﻣﺒﻨﺎﯼ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ.

 

ﺍﺻﻮﻝ ﺣﺴﺎﺑﺪﺍﺭﯼ
ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺍﺻﻮﻝ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻣﺘﺸﮑﻞ ﺍﺯ ﻗﻮﺍﻋﺪﯼ ﮐﻠﯽ ﺍﺳﺖ ﮐﻪ ﺣﺴﺎﺑﺪﺍﺭﺍﻥ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻣﺒﻨﺎﯼ ﺍﺳﺘﻔﺎﺩﻩ ﺧﻮﺩ ﺩﺭ ﮐﻠﻴﻪ ﻣﺮﺍﺣﻞ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﻨﺪ. ﺍﻳﻦ ﺍﺻﻮﻝ ﻋﺒﺎﺭﺗﻨﺪ ﺍﺯ:

۱. ﺍﺻﻞ ﺑﻬﺎﯼ ﺗﻤﺎﻡ ﺷﺪﻩ
ﺑﻪ ﻣﻮﺟﺐ ﺍﻳﻦ ﺍﺻﻞ، ﺗﻤﺎﻡ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﻣﺎﻟﯽ ﻣﻮﺳﺴﻪ ﺑﻪ ﺑﻬﺎﯼ ﺗﻤﺎﻡ ﺷﺪﻩ ﺩﺭ ﺗﺎﺭﻳﺦ ﻭﻗﻮﻉ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺩﺭ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺎﻟﯽ ﺍﻧﻌﮑﺎﺱ ‌ﻣﯽ ﻳﺎﺑﻨﺪ.

ﺍﻣﮑﺎﻥ ﺩﺍﺭﺩ ﮐﻪ ﻣﻮﺳﺴﻪ ﮐﺎﻻﻳﯽ ﺭﺍ ﮔﺮﺍﻥ‌ﺗﺮ ﻳﺎ ﺍﺭﺯﺍﻥ‌ﺗﺮ ﺍﺯ ﻗﻴﻤﺖ ﻭﺍﻗﻌﯽ ﺁﻥ ﺧﺮﻳﺪﺍﺭﯼ ﮐﻨﺪ ﻳﺎ ﺑﻪ ﻓﺮﻭﺵ ﺑﺮﺳﺎﻧﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻫﻤﻴﺸﻪ ﻗﻴﻤﺖ ﺗﻤﺎﻡ ﺷﺪﻩ ﺩﺭ ﺩﻓﺎ‌ﺗﺮ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﺩ ﻭ ﭼﻨﺎﻧﭽﻪ ﺑﻌﺪﺍ ﺍﺭﺯﺵ ﭘﻮﻟﯽ ﺁﻥ ﮐﺎﻫﺶ ﻳﺎ ﺍﻓﺰﺍﻳﺶ ﻳﺎﺑﺪ ﺍﻳﻦ ﺗﻐﻴﻴﺮﺍﺕ ﺷﻨﺎﺳﺎﻳﯽ ﻭ ﺛﺒﺖ ﻧﻤﯽ‌ﺷﻮﺩ.

۲. ﺍﺻﻞ ﺗﺤﻘﻖ ﺩﺭﺁﻣﺪ
ﺑﺮ ﺍﺳﺎﺱ ﺍﺻﻞ ﺗﺤﻘﻖ ﺩﺭﺁﻣﺪ، ﺩﺭﺁﻣﺪ‌ﻫﺎ ﺑﺪﻭﻥ ﺗﻮﺟﻪ ﺑﻪ ﺯﻣﺎﻥ ﺩﺭﻳﺎﻓﺖ ﻭﺟﻪ ﻧﻘﺪ ﻣﺮﺑﻮﻃﻪ، ﺩﺭ ﺯﻣﺎﻥ ﺗﺤﻘﻖ ﺷﻨﺎﺳﺎﻳﯽ ﻣﯽ‌ﺷﻮﻧﺪ. ﺩﺭﺁﻣﺪ ﺯﻣﺎﻧﯽ ﺗﺤﻘﻖ ﻣﯽ‌ﻳﺎﺑﺪ ﮐﻪ ﻓﺮﺍﻳﻨﺪ ﮐﺴﺐ ﺳﻮﺩ ﮐﺎﻣﻞ ﺷﺪﻩ ﺑﺎﺷﺪ. ﺍﻳﻦ ﻓﺮﺍﻳﻨﺪ ﻋﺒﺎﺭﺕ ﺍﺳﺖ ﺍﺯ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺍﺯ ﺍﻋﻤﺎﻝ ﮐﻪ ﺍﺯ ﺧﺮﻳﺪ ﻣﻮﺍﺩ ﺍﻭﻟﻴﻪ ﺷﺮﻭﻉ ﺷﺪﻩ ﻧﻬﺎﻳﺘﺎ ﻓﺮﻭﺵ ﻣﺤﺼﻮﻝ ﻭ ﺩﺭﻳﺎﻓﺖ ﻭﺟﻪ ﺭﺍ ﺩﺭﺑﺮ ﻣﯽ‌ﮔﻴﺮﺩ.

۳. ﺍﺻﻞ ﺍﻓﺸﺎ
ﺑﺮ ﺍﺳﺎﺱ ﺍﻳﻦ ﺍﺻﻞ ﺑﺎﻳﺪ ﮐﻠﻴﻪ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺩﺭ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪﮔﺎﻥ ﺍﺯ ﺍﻃﻼﻋﺎﺕ ﻣﺎﻟﯽ ﺗﺎﺛﻴﺮ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﺑﻪ ﻃﻮﺭ ﮐﺎﻣﻞ ﺍﻓﺸﺎ ﺷﻮﺩ.

۴. ﺍﺻﻞ ﺗﻄﺎﺑﻖ ﻫﺰﻳﻨﻪ‌ﻫﺎ ﺑﺎ ﺩﺭﺁﻣﺪ‌ﻫﺎ
ﺑﺮ ﺍﺳﺎﺱ ﺍﻳﻦ ﺍﺻﻞ ﺑﺮﺍﯼ ﺍﻧﺪﺍﺯﻩ‌ﮔﻴﺮﯼ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﻫﺮ ﺩﻭﺭﻩ ﺑﺎﻳﺪ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﻫﺮ ﺩﻭﺭﻩ ﺭﺍ ﺑﺎ ﺩﺭﺁﻣﺪﻫﺎﯼ‌‌ ﻫﻤﺎﻥ ﺩﻭﺭﻩ ﻣﻘﺎﻳﺴﻪ ﮐﺮﺩ. ﻳﻌﻨﯽ ﻫﺮﮔﺎﻩ ﺩﺭﺁﻣﺪﯼ ﺩﺭ ﺻﻮﺭﺕ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﻳﮏ ﺩﻭﺭﻩ ﻣﻨﻌﮑﺲ ﻣﯽ‌ﺷﻮﺩ، ﺑﺎﻳﺪ ﻫﺰﻳﻨﻪ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﺁﻥ ﺩﺭﺁﻣﺪ ﺍﺭﺗﺒﺎﻁ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ ﻫﻢ ﺑﺎﻳﺪ ﺩﺭ‌‌ ﻫﻤﺎﻥ ﺻﻮﺭﺕ ﻭ‌‌ ﻫﻤﺎﻥ ﺩﻭﺭﻩ ﻣﻨﻌﮑﺲ ﻣﯽ‌ﺷﻮﺩ.

 

ﻣﻴﺜﺎﻕ‌ﻫﺎ (ﺍﺻﻮﻝ ﻣﺤﺪﻭﺩ ﮐﻨﻨﺪﻩ)
ﺍﻳﻦ ﺍﺻﻮﻝ ﻣﺤﺪﻭﺩ ﮐﻨﻨﺪﻩ ﮐﺎﺭﺑﺮﺩ ﻣﻔﺮﻭﺿﺎﺕ ﻭ ﺍﺻﻮﻝ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺭﺍ ﺩﺭ ﻳﮏ ﭼﺎﺭ ﭼﻮﺏ ﺧﺎﺹ ﻣﺤﺪﻭﺩ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﻳﻦ ﺍﺻﻮﻝ ﺷﺎﻣﻞ ﻣﻮﺍﺭﺩ ﺯﻳﺮ ﺍﺳﺖ:

۱. ﺍﺻﻞ ﻓﺰﻭﻧﯽ ﻣﻨﺎﻓﻊ ﺑﺮ ﻣﺨﺎﺭﺝ
ﻫﺪﻑ ﺍﺯ ﮐﻠﻴﻪ ﻋﻤﻠﻴﺎﺕ ﻭ ﮔﺰﺍﺭﺷﺎﺕ ﻣﺎﻟﯽ ﺁﻣﺎﺩﻩ ﮐﺮﺩﻥ ﺍﻃﻼﻋﺎﺕ ﻻﺯﻡ ﺑﺮﺍﯼ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ ﺻﺤﻴﺢ ﺍﺳﺖ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻣﺨﺎﺭﺝ ﺗﻬﻴﻪ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺑﺎﻳﺪ ﻣﻘﺮﻭﻥ ﺑﻪ ﺻﺮﻓﻪ ﺑﺎﺷﺪ.

۲. ﺍﺻﻞ ﺍﻫﻤﻴﺖ
ﺑﻪ ﻣﻮﺟﺐ ﺍﻳﻦ ﺍﺻﻞ ﺍﮔﺮ ﻣﺒﺎﻟﻐﯽ ﺟﺰﻳﯽ ﻭ ﻧﺎﭼﻴﺰ ﺑﺎﺷﺪ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﻭﺿﻌﻴﺖ ﻭ ﺷﺮﺍﻳﻂ ﻣﻮﺳﺴﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﺍﻋﻤﺎﻝ ﺩﻗﻴﻖ ﺍﺻﻮﻝ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺧﻮﺩﺩﺍﺭﯼ ﮐﺮﺩ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺍﺻﻞ ﺑﻪ ﻣﻌﻨﺎﯼ ﻋﺪﻡ ﺛﺒﺖ ﺍﻗﻼﻡ ﺟﺰﻳﯽ ﻧﻴﺴﺖ.

ﺑﻪ ﻃﻮﺭ ﻣﺜﺎﻝ ﻳﮏ ﻗﻴﭽﯽ ﮐﻪ ﺑﻴﺶ ﺍﺯ ﻳﮏ ﺳﺎﻝ ﻋﻤﺮ ﻣﻔﻴﺪ ﺩﺍﺭﺩ ﺭﺍ ﺍﺯ ﻟﺤﺎﻅ ﻧﻈﺮﯼ ﺑﺎﻳﺪ ﺑﻪ ﻋﻨﻮﺍﻥ ﺩﺍﺭﻳﯽ ﺛﺒﺖ ﮐﺮﺩ ﻭ ﻫﺮ ﺳﺎﻝ ﺁﻥ ﺭﺍ ﻣﺴﺘﻬﻠﮏ ﮐﺮﺩ. ﻭﻟﯽ ﺑﻪ ﻣﻮﺟﺐ ﺍﻳﻦ ﺍﺻﻞ ﺩﺭ ﻣﻮﻗﻊ ﺧﺮﻳﺪ ﻗﻴﭽﯽ ﺁﻥ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻫﺰﻳﻨﻪ ﺛﺒﺖ ﻣﯽ‌ﮐﻨﻨﺪ.

۳. ﺍﺻﻞ ﻣﺤﺎﻓﻈﻪ‌ﮐﺎﺭﯼ
ﺑﺮ ﺍﺳﺎﺱ ﺍﻳﻦ ﺍﺻﻞ ﺩﺭ ﺷﺮﺍﻳﻄﯽ ﮐﻪ ﺍﺑﻬﺎﻡ ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﺍﻋﻤﺎﻝ ﻗﻀﺎﻭﺕ ﺑﺎﻳﺪ ﻃﻮﺭﯼ ﺑﺎﺷﺪ ﮐﻪ ﺩﺭﺁﻣﺪ‌ﻫﺎ ﺑﻴﺸﺘﺮ ﺍﺯ ﻭﺍﻗﻊ ﻭ ﺑﺪﻫﯽ‌‌ﻫﺎ ﮐﻤﺘﺮ ﺍﺯ ﻭﺍﻗﻊ ﺍﺭﺍﺋﻪ ﻧﺸﻮﺩ. ﻳﻌﻨﯽ ﺩﺭ ﺷﺮﺍﻳﻄﯽ ﮐﻪ ﺑﻪ ﺩﻻﻳﻠﯽ ﻧﺎﭼﺎﺭ ﻫﺴﺘﻴﻢ ﺑﺮﺁﻭﺭﺩ ﻣﺎﻟﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﺑﺎﻳﺪ ﺍﻳﻦ ﺑﺮﺁﻭﺭﺩﻫﺎﯼ ﻣﺎﻟﯽ ﺑﺎ ﺍﺣﺘﻴﺎﻁ ﺍﻧﺠﺎﻡ ﺷﻮﺩ.

ﺍﻳﻦ ﺍﺻﻞ ﺑﺮﺍﯼ ﺍﻳﻦ ﺑﻨﺎ ﻧﻬﺎﺩﻩ ﺷﺪﻩ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﺍﺳﺘﻔﺎﺩﻩ‌ﮐﻨﻨﺪﮔﺎﻥ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺎﻟﯽ ﺍﻣﻴﺪ ﻭﺍﻫﯽ ﺩﺭﺳﺖ ﻧﮑﻨﺪ.

۴. ﺍﺻﻞ ﺧﺼﻮﺻﻴﺎﺕ ﺻﻨﻌﺖ
ﺻﻨﺎﻳﻊ ﻣﺨﺘﻠﻒ ﺭﻭﻳﻪ‌ﻫﺎﯼ ﻣﺠﺰﺍ ﻭ ﻣﺘﻔﺎﻭﺗﯽ ﺑﺎ ﻫﻢ ﺩﺍﺭﻧﺪ. ﺑﻪ ﻣﻮﺟﺐ ﺍﺻﻞ ﺧﺼﻮﺻﻴﺎﺕ ﺻﻨﻌﺖ، ﺑﺮﺍﯼ ﻫﺮ ﺻﻨﻌﺘﯽ ﮐﻪ ﺩﺍﺭﺍﯼ ﺷﺮﺍﻳﻂ ﻭﻳﮋﻩ ﺑﺎﺷﺪ ﻣﯽ‌ﺗﻮﺍﻥ ﺭﻭﺷﻬﺎﯼ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺧﺎﺹ ﻭ ﻭﻳﮋﻩ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺖ.

 

ﻭﻳﺰﮔﯽ‌ﻫﺎﯼ ﺍﻃﻼﻋﺎﺕ ﺣﺴﺎﺑﺪﺍﺭﯼ
ﺍﻃﻼﻋﺎﺕ ﺣﺴﺎﺑﺪﺍﺭﯼ ﮐﻪ ﺩﺭ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺎﻟﯽ ﻣﻨﻌﮑﺲ ﻣﯽ‌ﺷﻮﺩ ﺑﺎﻳﺪ ﺩﺍﺭﺍﯼ ﺧﺼﻮﺻﻴﺎﺕ ﺯﻳﺮ ﺑﺎﺷﺪ ﺗﺎ ﺑﺘﻮﺍﻧﺪ ﺩﺭ ﻓﺮﺍﻳﻨﺪ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪﮔﺎﻥ ﻣﻮﺛﺮ ﺑﺎﺷﺪ.

۱. ﻣﺮﺑﻮﻁ ﺑﻮﺩﻥ
ﺍﻃﻼﻋﺎﺕ ﻣﻮﺟﻮﺩ ﺩﺭ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺎﻟﯽ ﺑﺎﻳﺪ ﺩﺭ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ‌ﻫﺎ ﻣﻮﺛﺮ ﺑﺎﺷﺪ ﻭ ﺑﺎﻋﺚ ﺷﻮﺩ ﮐﻪ ﺍﺳﺘﻔﺎﺩﻩ‌ﮐﻨﻨﺪﮔﺎﻥ ﺑﺘﻮﺍﻧﻨﺪ ﺑﺮ ﺍﺳﺎﺱ ﺁﻥ ﺗﺼﻤﻴﻢ ﮔﻴﺮﯼ ﮐﻨﻨﺪ.

۲. ﻗﺎﺑﻠﻴﺖ ﺍﻋﺘﻤﺎﺩ
ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺑﺎﻳﺪ ﻗﺎﺑﻞ ﺍﻋﺘﻤﺎﺩ ﻭ ﻗﺎﺑﻞ ﺍﺗﮑﺎ ﺑﺎﺷﻨﺪ. ﻋﺎﺭﯼ ﺍﺯ ﺍﺷﺘﺒﺎﻩ ﺑﻮﺩﻩ ﻭ ﺟﺎﻧﺒﺪﺍﺭﺍﻧﻪ ﻧﺒﺎﺷﺪ.

۳. ﻗﺎﺑﻠﻴﺖ ﻣﻘﺎﻳﺴﻪ
ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺑﺎﻳﺪ ﺑﻪ ﻧﺤﻮﯼ ﺑﺎﺷﺪ ﮐﻪ ﺍﻣﮑﺎﻥ ﻣﻘﺎﻳﺴﻪ ﻣﻮﺳﺴﺎﺕ ﻭ ﻭﺍﺣﺪ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ ﺑﺎ ﻫﻢ ﻭ ﺑﺎ ﺳﺎﻟﻬﺎﯼ ﻣﺨﺘﻠﻒ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.
