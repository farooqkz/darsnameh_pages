ﺩﺭﺱ ﺍﻭﻝ- ﺣﺴﺎﺑﺪﺍﺭﯼ ﭼﻴﺴﺖ؟

ﺩﺭﻭﺩ ﺑﺮ ﺷﻤﺎ ﮐﺎﺭﺁﻣﻮﺯ ﮔﺮﺍﻣﯽ! ﺍﺯ ﺛﺒﺖ ﻧﺎﻡ ﺷﻤﺎ ﺑﺮﺍﯼ ﺩﻭﺭﻩ‌ﯼ «ﺣﺴﺎﺑﺪﺍﺭﯼ ﺑﺮﺍﯼ ﻫﻤﻪ» ﺩﺭ ﺩﺭﺳﻨﺎﻣﻪ ﺧﺮﺳﻨﺪﻳﻢ ﻭ ﺍﻣﻴﺪﻭﺍﺭﻳﻢ ﮐﻪ ﺑﺘﻮﺍﻧﻴﻢ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ، ﻧﻴﺎﺯﻫﺎﯼ ﺍﺳﺎﺳﯽ ﺷﻤﺎ ﺭﺍ ﺩﺭ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺭﻓﻊ ﮐﻨﻴﻢ.

ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﻣﺪﻳﺮ ﻳﮏ ﺷﺮﮐﺖ ﺑﺎﺷﻴﺪ ﻭ ﺑﺨﻮﺍﻫﻴﺪ ﻣﺴﺎﺋﻞ ﻣﺎﻟﯽ ﺷﺮﮐﺘﺘﺎﻥ ﺭﺍ ﺳﺮ ﻭ ﺳﺎﻣﺎﻥ ﺑﺪﻫﻴﺪ. ﺷﺎﻳﺪ ﻫﻢ ﺻﺎﺣﺐ ﻳﮏ ﻣﻐﺎﺯﻩ ﺑﺎﺷﻴﺪ ﻭ ﺍﺯ ﺭﻭﺵ ﺣﺴﺎﺑﺮﺳﯽ ﺑﻪ ﺷﻴﻮﻩ ﺳﻨﺘﯽ (ﺩﺳﺖ‌ﻧﻮﺷﺘﻪ ﺩﺭ ﺩﻓﺘﺮﻫﺎﯼ ﺣﺴﺎﺑﺮﺳﯽ) ﺑﻪ ﺳﺘﻮﻩ ﺁﻣﺪﻩ ﺑﺎﺷﻴﺪ! ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻤﺎ ﻳﮏ ﺩﺍﻧﺸﺠﻮ، ﻳﮏ ﻓﺮﺩ ﺧﺎﻧﻪ‌ﺩﺍﺭ ﻭ ﻳﺎ ﻳﮏ ﮐﺎﺭﻣﻨﺪ ﺑﺎﺷﻴﺪ ﻭ ﺑﺨﻮﺍﻫﻴﺪ ﺑﻪ ﻣﺎﻝ ﻭ ﺍﻣﻮﺍﻝ ﻭ ﺧﺮﺝ ﻭ ﻣﺨﺎﺭﺝ ﺷﺨﺼﻴﺘﺎﻥ ﻧﻈﺎﺭﺕ ﺩﻗﻴﻖ‌ﺗﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ!

ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺍﻳﻦ ﺩﻭﺭﻩ ﺑﺮﺍﯼ ﻫﻤﻪ‌ﯼ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﺎﺭﺑﺮﺩﯼ ﺑﺎﺷﺪ!

ﻣﺎ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﺗﻼﺵ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺭﺍ ﺑﻪ ﺷﻴﻮﻩ‌ﺍﯼ ﺳﺎﺩﻩ ﻭ ﻫﻤﻪ‌ ﻓﻬﻢ ﺑﻴﺎﻥ ﮐﻨﻴﻢ ﻭ ﺗﻼﺵ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺩﺭ ﺭﺳﻴﺪﻥ ﺑﻪ ﻫﺪﻓﺘﺎﻥ ﻧﺰﺩﻳﮏ‌ﺗﺮ ﮐﻨﻴﻢ.

 

ﺗﻌﺮﻳﻒ ﺣﺴﺎﺑﺪﺍﺭﯼ
ﺣﺴﺎﺑﺪﺍﺭﯼ ﺭﺍ ﺑﻪ ﻣﻌﻨﺎﯼ ﺩﻳﮕﺮ ﮐﻠﻤﻪ، «ﺯﺑﺎﻥ ﺗﺠﺎﺭﺕ» ﻣﯽ‌ﺧﻮﺍﻧﻨﺪ. ﺍﻃﻼﻋﺎﺕ ﺩﻗﻴﻖ ﻭ ﮔﺰﺍﺭﺵ‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﺎ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ ﺩﻗﻴﻖ، ﺳﺮﻳﻊ ﻭ ﺑﻪ ﻣﻮﻗﻊ ﻣﺪﻳﺮﺍﻥ ﻣﻨﺠﺮ ﺷﻮﺩ ﻭ ﺑﺰﺭﮒ‌ﻧﻤﺎﻳﯽ ﻧﻴﺴﺖ ﺍﮔﺮ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﻳﮑﯽ ﺍﺯ ﺭﻣﻮﺯ ﻣﻮﻓﻘﻴﺖ ﺷﺮﮐﺖ‌ﻫﺎ، ﺷﻴﻮﻩ‌ﯼ ﺩﻗﻴﻖ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺁﻥ‌ﻫﺎﺳﺖ.

ﺩﺭ ﮔﺬﺷﺘﻪ، ﺣﺴﺎﺑﺪﺍﺭﯼ ﺭﺍ ﺑﻪ ﻣﻔﻬﻮﻡ «ﺩﻓﺘﺮ ﺩﺍﺭﯼ» ﺗﻌﺒﻴﺮ ﻣﯽ‌ﮐﺮﺩﻧﺪ. ﻣﻔﻬﻮﻣﯽ ﮐﻪ ﻓﻦ ﺛﺒﺖ، ﻃﺒﻘﻪ ﺑﻨﺪﯼ ﻭ ﺧﻼﺻﻪ ﺳﺎﺯﯼ ﻣﺒﺎﺩﻻﺕ ﻭﺍﺣﺪ ﺗﺠﺎﺭﯼ ﺭﺍ ﺩﺭ ﺩﻓﺘﺮﭼﻪ‌ﻫﺎﯼ ﻣﺸﺨﺼﯽ ﮐﻪ ﺑﺎﻳﺪ ﺩﺳﺖ‌ﻧﻮﻳﺲ ﻣﯽ‌ﺷﺪﻧﺪ ﺭﺍ ﺑﻴﺎﻥ ﻣﯽ‌ﮐﺮﺩ.

ﺳﺎﺯﻣﺎﻥ ﺟﻬﺎﻧﯽ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺩﺭ ﺳﺎﻝ ۱۹۶۶ ﺩﺭ ﻳﮏ ﺑﻴﺎﻧﻴﻪ «ﺗﺌﻮﺭﯼ ﺣﺴﺎﺑﺪﺍﺭﯼ» ﺭﺍ ﺍﻳﻨﮕﻮﻧﻪ ﺗﻌﺮﻳﻒ ﮐﺮﺩ: ﺣﺴﺎﺑﺪﺍﺭﯼ ﻋﺒﺎﺭﺕ ﺍﺳﺖ ﺍﺯ ﻓﺮﺍﻳﻨﺪ ﺷﻨﺎﺳﺎﻳﯽ، ﺍﻧﺪﺍﺯﻩ‌ﮔﻴﺮﯼ ﻭ ﮔﺰﺍﺭﺷﮕﺮﯼ ﺍﻃﻼﻋﺎﺕ ﺍﻗﺘﺼﺎﺩﯼ ﺑﺮﺍﯼ ﺗﺼﻤﻴﻢ ﮔﻴﺮﯼ‌ﻫﺎﯼ ﺁﮔﺎﻫﺎﻧﻪ ﺑﻪ ﻭﺳﻴﻠﻪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪﮔﺎﻥ ﺁﻥ ﺍﻃﻼﻋﺎﺕ. ﺍﻳﻦ ﺑﻴﺎﻧﻴﻪ ﺩﺭ ﺗﺸﺮﻳﺢ ﺍﻳﻦ ﺗﻌﺮﻳﻒ ﺁﻭﺭﺩﻩ ﺍﺳﺖ ﮐﻪ ﺑﺮ ﺍﺳﺎﺱ ﺩﻳﺪﮔﺎﻫﯽ ﺟﺎﻣﻊ، «ﺍﻃﻼﻋﺎﺕ ﺣﺴﺎﺑﺪﺍﺭﯼ» ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ «ﺍﻃﻼﻋﺎﺕ ﺍﻗﺘﺼﺎﺩﯼ» ﺗﻠﻘﯽ ﮐﺮﺩ.

ﺍﻣﺮﻭﺯﻩ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺑﻪ ﻣﻌﻨﺎﯼ ﺩﻓﺘﺮﺩﺍﺭﯼ ﺗﻠﻘﯽ ﻧﻤﯽ‌ﺷﻮﺩ ﻭ ﭼﻮﻥ ﺛﺒﺖ ﺍﻃﻼﻋﺎﺕ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻣﻌﻤﻮﻻ ﺑﺎ ﻧﺮﻡ‌ﺍﻓﺰﺍﺭ ﻭ ﺍﺯ ﻃﺮﻳﻖ ﺭﺍﻳﺎﻧﻪ ﺍﺳﺖ، ﺩﻓﺘﺮﺩﺍﺭﯼ ﺧﻮﺩ ﺑﺨﺸﯽ ﺍﺯ ﺳﻴﺴﺘﻢ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﺩ.

ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﻳﮏ ﺳﻴﺴﺘﻢ ﺍﻃﻼﻋﺎﺗﯽ ﺗﺸﺒﻴﻪ ﮐﺮﺩ ﮐﻪ ﺍﺯ ﻃﺮﻳﻖ ﺷﻨﺎﺳﺎﻳﯽ، ﺛﺒﺖ، ﻃﺒﻘﻪ‌ﺑﻨﺪﯼ ﻭ ﺗﻠﺨﻴﺺ ﻭ ﮔﺰﺍﺭﺵ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﻣﺎﻟﯽ ﺍﻗﺘﺼﺎﺩﯼ، ﺍﻃﻼﻋﺎﺕ ﻻﺯﻡ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﮐﺴﺎﻧﯽ ﮐﻪ ﺑﻪ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﻧﻴﺎﺯ ﺩﺍﺭﻧﺪ (ﻣﺜﻞ ﻣﺪﻳﺮﺍﻥ ﻭ ﻳﺎ ﻣﺘﻘﺎﺿﻴﺎﻥ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﺍﻃﻼﻋﺎﺕ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻳﮏ ﺷﺮﮐﺖ) ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﺪ.

ﺍﮔﺮ ﺑﺨﻮﺍﻫﻴﻢ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻭ ﻫﺪﻑ ﻳﺎﺩﮔﻴﺮﯼ ﺍﺯ ﺁﻥ ﺭﺍ ﺩﺭ ﻳﮏ ﺷﮑﻞ ﺗﻌﺮﻳﻒ ﮐﻨﻴﻢ، ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﺯ ﺷﮑﻞ ﺯﻳﺮ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ:

ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺩﺭ ﺗﺼﻮﻳﺮ ﻓﻮﻕ ﻣﯽ‌ﺑﻴﻨﻴﺪ، ﻫﺪﻑ ﺍﺯ ﻳﺎﺩﮔﻴﺮﯼ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﺴﺎﺋﻞ ﻣﺎﻟﯽ ﻳﮏ ﺷﺮﮐﺖ، ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﭘﺮﺩﺍﺯﺵ ﺷﻮﺩ ﮐﻪ ﺩﺭ ﻧﻬﺎﻳﺖ ﮔﺰﺍﺭﺵ‌ﻫﺎﻳﯽ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻣﺪﻳﺮ ﺷﺮﮐﺖ ﻗﺮﺍﺭ ﺩﻫﺪ. ﻣﺪﻳﺮ ﺷﺮﮐﺖ ﺑﺮ ﺍﺳﺎﺱ ﺍﻳﻦ ﮔﺰﺍﺭﺵ‌ﻫﺎ، ﺗﺼﻤﻴﻢ‌ﻫﺎﯼ ﻣﺎﻟﯽ ﻣﯽ‌ﮔﻴﺮﺩ.

 

ﻣﻌﺎﺩﻟﻪ ﺍﺳﺎﺳﯽ ﺩﺭ ﺣﺴﺎﺑﺪﺍﺭﯼ
ﺩﺭ ﻋﻠﻢ ﺣﺴﺎﺑﺪﺍﺭﯼ، ﻳﮏ ﻣﻌﺎﺩﻟﻪ‌ﯼ ﺍﺳﺎﺳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﺎﻳﻪ ﻭ ﻣﺒﻨﺎﯼ ﻳﮏ ﺳﻴﺴﺘﻢ ﺩﻭ ﻃﺮﻓﻪ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻪ ﻣﯽ‌ﺷﻮﺩ.

ﺍﻳﻦ ﻣﻌﺎﺩﻟﻪ ﺑﻪ ﺻﻮﺭﺕ ﺯﻳﺮ ﺍﺳﺖ:

ﺑﻪ ﺯﺑﺎﻥ ﺳﺎﺩﻩ‌ﺗﺮ، ﺩﺍﺭﺍﻳﯽ‌ﻫﺎ، ﺑﺪﻫﯽ‌ﻫﺎ ﻭ ﺳﺮﻣﺎﻳﻪ ﻳﮏ ﺷﺮﮐﺖ، ﻋﻨﺎﺻﺮ ﺍﺻﻠﯽ ﻣﻌﺎﺩﻟﻪ ﺍﺳﺎﺳﯽ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺁﻥ ﺷﺮﮐﺖ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﻨﺪ.

ﺍﮐﻨﻮﻥ ﺑﻪ ﺗﻌﺮﻳﻒ ﻫﺮ ﮐﺪﺍﻡ ﺍﺯ ﻣﻮﺍﺭﺩ ﺑﺎﻻ ﻣﯽ‌ﭘﺮﺩﺍﺯﻳﻢ:

- ﺩﺍﺭﺍﻳﯽ‌ﻫﺎ: ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺍﺯ ﻧﺎﻣﺶ ﭘﻴﺪﺍﺳﺖ، ﻫﺮﮔﻮﻧﻪ ﺍﻣﻮﺍﻝ ﻭ ﻣﻨﺎﺑﻊ ﻣﺎﻟﯽ ﻳﮏ ﺷﺮﮐﺖ ﺑﻪ ﻋﻨﻮﺍﻥ Asset ﻳﺎ ﺩﺍﺭﺍﻳﯽ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻪ ﻣﯽ‌ﺷﻮﺩ.

ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺩﺭ ﺣﺴﺎﺑﺪﺍﺭﯼ ﻫﺮ ﮔﻮﻧﻪ ﺍﻣﻮﺍﻝ، ﻣﻨﺎﺑﻊ ﺍﻗﺘﺼﺎﺩﯼ ﻭ ﻭﺍﺣﺪﻫﺎﯼ ﺗﺠﺎﺭﯼ ﻣﺘﻌﻠﻖ ﺑﻪ ﻳﮏ ﻣﻮﺳﺴﻪ، ﮐﻪ ﺩﺭ ﻧﺘﻴﺠﻪ ﻣﻌﺎﻣﻼﺕ ﻋﻤﻠﻴﺎﺕ ﻣﺎﻟﯽ ﺍﻳﺠﺎﺩ ﮐﻨﺪ ﻭ ﻗﺎﺑﻞ ﺗﺒﺪﻳﻞ ﺑﻪ ﭘﻮﻝ ﻭ ﺩﺍﺭﺍﯼ ﻣﻨﺎﻓﻊ ﺁﺗﯽ ﺑﺎﺷﺪ ﺭﺍ «ﺩﺍﺭﺍﺋﯽ» ﻣﯽ‌ﮔﻮﻳﻨﺪ.

ﺑﻪ ﻋﺒﺎﺭﺕ ﺳﺎﺩﻩ‌ﺗﺮ ﻣﯽ‌ﺗﻮﺍﻥ ﺳﻪ ﺷﺮﻁ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺷﺮﻭﻁ ﭘﺬﻳﺮﺵ ﻳﮏ ﻣﻨﺒﻊ ﺍﻗﺘﺼﺎﺩﯼ ﺑﻪ ﻋﻨﻮﺍﻥ «ﺩﺍﺭﺍﻳﯽ» ﻳﮏ ﻣﻮﺳﺴﻪ ﻗﻠﻤﺪﺍﺩ ﮐﺮﺩ:

  • ﺷﺮﻁ ﻧﺨﺴﺖ:‌ ﺍﻳﻦ «ﺩﺍﺭﺍﻳﯽ»، ﺩﺍﺭﺍﯼ ﻣﻨﺎﻓﻊ ﺍﻗﺘﺼﺎﺩﯼ ﺁﺗﯽ ﺑﺮﺍﯼ ﺷﺮﮐﺖ ﺑﺎﺷﺪ.
  • ﺷﺮﻁ ﺩﻭﻡ: ﺑﻬﺎﯼ ﺗﻤﺎﻡ ﺷﺪﻩ ﺩﺍﺭﺍﻳﯽ ﺑﺮﺍﯼ ﺷﺮﮐﺖ ﻗﺎﺑﻞ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ ﻭ ﻗﺎﺑﻞ ﺗﺒﺪﻳﻞ ﺑﻪ ﭘﻮﻝ ﺑﺎﺷﺪ.
  • ﺷﺮﻁ ﺳﻮﻡ: ﺍﻳﻦ ﺩﺍﺭﺍﻳﯽ ﺩﺭ ﻧﺘﻴﺠﻪ ﻣﻌﺎﻣﻼﺕ ﻳﺎ ﺳﺎﻳﺮ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﮔﺬﺷﺘﻪ ﺑﻪ ﮐﻨﺘﺮﻝ ﺷﺮﮐﺖ ﺩﺭﺁﻣﺪﻩ ﺑﺎﺷﺪ.

ﺑﺮﺍﯼ ﺩﺭﮎ ﺑﻬﺘﺮ ﺷﺮﺍﻳﻂ ﻓﻮﻕ، ﺑﻪ ﻣﺜﺎﻝ ﺯﻳﺮ ﺗﻮﺟﻪ ﮐﻨﻴﺪ:

ﻓﺮﺽ ﮐﻨﻴﺪ ﻳﮏ ﺷﺮﮐﺖ ﺗﺠﺎﺭﯼ ﺍﺗﻮﻣﺒﻴﻠﯽ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺍﻗﺴﺎﻁ ﺧﺮﻳﺪﺍﺭﯼ ﻣﯽ‌ﮐﻨﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺎﻟﮑﻴﺖ ﻗﺎﻧﻮﻧﯽ ﺍﺗﻮﻣﺒﻴﻞ ﺧﺮﻳﺪﺍﺭﯼ ﺷﺪﻩ، ﺗﺎ ﻗﺒﻞ ﺍﺯ ﭘﺮﺩﺍﺧﺖ ﺁﺧﺮﻳﻦ ﻗﺴﻂ ﺑﻪ ﺧﺮﻳﺪﺍﺭ ﻣﻨﺘﻘﻞ ﻧﺸﻮﺩ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﭼﻮﻥ ﺍﻳﻦ ﺍﺗﻮﻣﺒﻴﻞ ﺑﻪ ﺗﺼﺮﻑ ﻭ ﮐﻨﺘﺮﻝ ﮐﺎﻣﻞ ﺧﺮﻳﺪﺍﺭ ﺩﺭ ﻣﯽ‌ﺁﻳﺪ ﻭ ﺍﻭ ﻧﺴﺒﺖ ﺑﻪ ﺁﻥ ﻓﻊ ﻣﺎﻟﮑﺎﻧﻪ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ، ﺍﺯ ﻧﻈﺮ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺍﺗﻮﻣﻮﺑﻴﻞ ﺍﺷﺎﺭﻩ ﺷﺪﻩ ﺟﺰﺀ ﺩﺍﺭﺍﻳﯽ ﺍﻳﻦ ﺷﺮﮐﺖ ﺗﺠﺎﺭﯼ ﻣﺤﺴﻮﺏ ﻣﯽ‌ﺷﻮﺩ.

- ﺑﺪﻫﯽ‌ﻫﺎ: ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺍﺯ ﻧﺎﻣﺶ ﭘﻴﺪﺍﺳﺖ، ﺑﺪﻫﯽ‌ﻫﺎ، ﺗﻌﻬﺪﺍﺗﯽ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻣﻮﺳﺴﻪ ﺩﺭ ﻣﻘﺎﺑﻞ ﺍﺷﺨﺎﺹ ﻭ ﻣﻮﺳﺴﺎﺕ ﺩﻳﮕﺮ ﺩﺍﺭﺩ. ﺍﻳﻦ ﺗﻌﻬﺪﺍﺕ ﻋﻤﻮﻣﺎ ﺍﺯ ﻣﻌﺎﻣﻼﺕ ﻗﺒﻠﯽ ﺻﻮﺭﺕ ﮔﺮﻓﺘﻪ ﻭ ﺑﺎﻳﺪ ﺩﺭ ﻃﯽ ﺯﻣﺎﻥ ﻣﺸﺨﺼﯽ ﺍﺯ ﻃﺮﻳﻖ ﭘﻮﻝ ﻳﺎ ﻋﻤﻠﮑﺮﺩ ﺧﺎﺻﯽ ﺗﺴﻮﻳﻪ ﺷﻮﺩ.

- ﺳﺮﻣﺎﻳﻪ‌: ﺣﻖ ﻣﺎﻟﯽ ﻣﺎﻟﮏ ﻳﺎ ﻣﺎﻟﮑﺎﻥ ﻳﮏ ﺷﺮﮐﺖ ﺗﺠﺎﺭﯼ ﻧﺴﺒﺖ ﺑﻪ ﺩﺍﺭﺍﻳﻴﻬﺎﯼ ﺍٓﻥ ﺭﺍ «ﺳﺮﻣﺎﻳﻪ» ﻣﯽ‌ﮔﻮﻳﻨﺪ. ﺑﺮ ﺍﺳﺎﺱ ﻣﻌﺎﺩﻟﻪ ﺍﺳﺎﺳﯽ ﺣﺴﺎﺑﺪﺍﺭﯼ، ﺩﺭ ﻫﺮ ﺯﻣﺎﻥ ﺑﺎ ﮐﺴﺮ ﮐﺮﺩﻥ ﺑﺪﻫﻴﻬﺎﯼ ﻳﮏ ﻣﻮٔﺳﺴﻪ ﺍﺯ ﺩﺍﺭﺍﻳﻴﻬﺎﯼ ﺍٓﻥ، ﻣﺒﻠﻎ ﺳﺮﻣﺎﻳﻪ ﺑﻪ ﺩﺳﺖ ﻣﯽ‌ﺍٓﻳﺪ.
