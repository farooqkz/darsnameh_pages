ﭼﺮﺧﻪ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺑﻪ ﻳﮏ ﺳﺮﯼ ﺍﺯﻋﻤﻠﻴﺎﺗﯽ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﮐﻪ ﻫﺮ ﺳﺎﻝ ﻭ ﻳﺎ ﺩﺭ ﻫﺮ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺗﮑﺮﺍﺭ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻦ ﭼﺮﺧﻪ ﺑﻪ ﺗﺮﺗﻴﺐ ﺷﺎﻣﻞ ﻋﻤﻠﻴﺎﺕ‌ﻫﺎﯼ ﺯﻳﺮ ﺍﺳﺖ:

۷. ﺁﻣﺎﺩﻩ ﮐﺮﺩﻥ ﺗﺮﺍﺯ ﺁﺯﻣﺎﻳﺸﯽ (ﺍﺻﻼﺡ ﺷﺪﻩ)

۸. ﺁﻣﺎﺩﻩ ﮐﺮﺩﻥ ﺻﻮﺭﺕ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﻭ ﺻﻮﺭﺕ ﺗﻐﻴﻴﺮﺍﺕ ﺳﺮﻣﺎﻳﻪ

۹. ﺑﺴﺘﻦ ﺣﺴﺎﺑﻬﺎﯼ ﻣﻮﻗﺖ

۱۰. ﺁﻣﺎﺩﻩ ﮐﺮﺩﻥ ﺗﺮﺍﺯ ﺁﺯﻣﺎﻳﺸﯽ (ﺍﺧﺘﺘﺎﻣﯽ)

۱۱. ﺗﻬﻴﻪ ﺗﺮﺍﺯﻧﺎﻣﻪ ﻭ ﺑﺴﺘﻦ ﺣﺴﺎﺑﻬﺎﯼ ﺩﺍﺋﻤﯽ

ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻋﻤﻠﻴﺎﺕ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺩﺭ ﺩﻭ ﺩﻭﺭﻩ ﺯﻣﺎﻧﯽ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﻃﻮﻝ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﻭ ﺩﺭ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ ﻣﺎﻟﯽ. ﻗﺴﻤﺖ ﺍﻭﻝ ﮐﻪ ﺩﺭ ﻃﻮﻝ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ ﺷﺎﻣﻞ ﺟﻤﻊ ﺁﻭﺭﯼ ﮐﻠﻴﻪ ﺍﻃﻼﻋﺎﺕ ﻣﺎﻟﯽ ﻣﻮﺳﺴﻪ، ﺗﺠﺰﻳﻪ ﻭ ﺗﺤﻠﻴﻞ ﮐﻠﻴﻪ ﻣﻌﺎﻣﻼﺕ ﻭ ﻋﻤﻠﻴﺎﺕ ﻣﺎﻟﯽ ﻣﻮﺳﺴﻪ، ﺛﺒﺖ ﮐﻠﻴﻪ ﻣﻌﺎﻣﻼﺕ ﻭ ﻋﻤﻠﻴﺎﺕ ﻣﺎﻟﯽ ﻣﻮﺳﺴﻪ ﺩﺭ ﺩﻓﺘﺮ ﻣﺎﻟﯽ، ﺍﻧﺘﻘﺎﻝ ﮐﻠﻴﻪ ﻋﻤﻠﻴﺎﺕ‌ﻫﺎ ﺍﺯ ﺩﻓﺘﺮ ﺭﻭﺯﻧﺎﻣﻪ ﺑﻪ ﺩﻓﺘﺮ ﮐﻞ ﺍﺳﺖ. ﻭ ﺑﻘﻴﻪ ﻋﻤﻠﻴﺎﺕ ﺩﺭ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ.

 

ﺑﺴﺘﻦ ﺣﺴﺎﺏ‌ﻫﺎ ﺑﻪ ﭼﻪ ﻣﻌﻨﯽ ﺍﺳﺖ؟
ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻧﺘﻘﺎﻝ ﻳﺎ ﻋﺪﻡ ﺍﻧﺘﻘﺎﻝ ﻣﺎﻧﺪﻩ ﺑﻪ ﺳﺎﻝ ﺑﻌﺪ، ﺣﺴﺎﺏ‌ﻫﺎ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﺳﻪ ﺩﺳﺘﻪ ﺗﻘﺴﻴﻢ ﮐﺮﺩ:

۱. ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺩﺍﺋﻤﯽ

۲. ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ

۳. ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﺨﺘﻠﻂ

ﺍﻟﺒﺘﻪ ﻧﻬﺎﻳﺘﺎ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﺨﺘﻠﻂ ﻫﻢ ﺑﺎ ﺍﻧﺠﺎﻡ ﺍﺻﻼﺣﺎﺕ ﻭ ﺛﺒﺖ‌ﻫﺎﯼ ﺍﺻﻼﺣﯽ ﻭﺍﺭﺩ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺩﺍﺋﻤﯽ ﻳﺎ ﻣﻮﻗﺖ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﺩﺭ ﭘﺎﻳﺎﻥ ﺳﺎﻝ ﺣﺴﺎﺏ‌ﻫﺎ ﻳﺎ ﺩﺍﺋﻤﯽ ﻫﺴﺘﻨﺪ ﻳﺎ ﻣﻮﻗﺖ. ﻭ ﺑﻌﺪ ﺍﺯ ﺗﻬﻴﻪ ﺻﻮﺭﺕ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ ﻧﻴﺰ ﺑﺴﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﻓﻘﻂ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺩﺍﺋﻤﯽ ﺑﺎﻗﯽ ﻣﯽ‌ﻣﺎﻧﺪ ﮐﻪ ﺍﻳﻦ ﺣﺴﺎﺏ‌ﻫﺎ ﺍﺯ ﻃﺮﻳﻖ ﺻﺪﻭﺭ ﺳﻨﺪ ﺍﺧﺘﺘﺎﻣﻴﻪ ﺑﻪ ﺳﺎﻝ ﺑﻌﺪ ﻣﻨﺘﻘﻞ ﻣﯽ‌ﺷﻮﻧﺪ.

 

ﭼﮕﻮﻧﮕﯽ ﺑﺴﺘﻦ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ ﺩﺭ ﺍﻧﺘﻬﺎﯼ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺩﺭ ﺷﺮﮐﺖ‌ﻫﺎﯼ ﺧﺪﻣﺎﺗﯽ
ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﮔﻔﺘﻪ ﺷﺪ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ ﺑﺎﻳﺪ ﺩﺭ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺑﺴﺘﻪ ﺷﻮﻧﺪ. ﻣﺎﻧﺪﻩ ﻣﺎﻟﯽ ﺍﻳﻦ ﮔﻮﻧﻪ ﺣﺴﺎﺏ‌ﻫﺎ ﺑﻪ ﺳﺎﻝ ﻣﺎﻟﯽ ﺑﻌﺪ ﻣﻨﺘﻘﻞ ﻧﻤﯽ‌ﺷﻮﺩ. ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ ﺷﺎﻣﻞ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺩﺭﺁﻣﺪ، ﻫﺰﻳﻨﻪ‌ﻫﺎ ﻭ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ ﻫﺴﺘﻨﺪ.

ﺟﻬﺖ ﺑﺴﺘﻦ ﺍﻳﻦ ﺣﺴﺎﺏ‌ﻫﺎ ﺑﺎﻳﺪ ﺍﺯ ﻳﮏ ﺣﺴﺎﺏ ﺭﺍﺑﻂ ﻭ ﻭﺍﺳﻄﻪ ﺑﻪ ﻧﺎﻡ ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ. ﻣﺮﺍﺣﻞ ﺑﺴﺘﻦ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ ﺑﻪ ﺻﻮﺭﺕ ﺯﻳﺮ ﺍﺳﺖ:

ﻣﺮﺣﻠﻪ ﺍﻭﻝ: ﺑﺴﺘﻦ ﺣﺴﺎﺏ ﺩﺭﺁﻣﺪ
ﺍﺑﺘﺪﺍ ﺣﺴﺎﺏ ﺩﺭﺁﻣﺪﻫﺎ ﺭﺍ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺭﺍ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ.

ﻣﺮﺣﻠﻪ ﺩﻭﻡ: ﺑﺴﺘﻦ ﺣﺴﺎﺏ ﻫﺰﻳﻨﻪ‌ﻫﺎ
ﺳﭙﺲ ﺣﺴﺎﺏ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺭﺍ ﻣﻌﺎﺩﻝ ﺑﺎ ﺟﻤﻊ ﻫﺰﻳﻨﻪ ﺑﺪﻫﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺳﭙﺲ ﺗﮏ ﺗﮏ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﺎ ﻫﺰﻳﻨﻪ‌ﻫﺎ ﺭﺍ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ.

ﻣﺮﺣﻠﻪ ﺳﻮﻡ: ﺑﺴﺘﻦ ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ
ﻣﺎﻧﺪﻩ ﺧﻼﺻﻪ ﺣﺴﺎﺏ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺭﺍ ﻣﺸﺨﺺ ﻣﯽ‌ﮐﻨﻴﻢ. ﺍﮔﺮ ﻣﺎﻧﺪﻩ ﺁﻥ ﺑﺴﺘﺎﻧﮑﺎﺭ ﺑﺎﺷﺪ ﻣﻌﺎﺩﻝ ﺁﻥ ﺭﺍ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺳﺮﻣﺎﻳﻪ ﺭﺍ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ. ﺍﮔﺮ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺑﺴﺘﺎﻧﮑﺎﺭ ﺑﺎﺷﺪ ﺑﻪ ﺁﻥ ﺳﻮﺩ ﻣﯽ‌ﮔﻮﻳﻴﻢ.

ﻭﻟﯽ ﺍﮔﺮ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺑﺪﻫﮑﺎﺭ ﺑﺎﺷﺪ، ﺍﻳﻦ ﺣﺴﺎﺏ ﺭﺍ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺳﺮﻣﺎﻳﻪ ﺭﺍ ﺑﺪﻫﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﻪ ﺍﻳﻦ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺯﻳﺎﻥ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

ﻣﺮﺣﻠﻪ ﭼﻬﺎﺭﻡ: ﺑﺴﺘﻦ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ
ﺩﺭﺍﻧﺘﻬﺎ ﺑﺮﺍﯼ ﺑﺴﺘﻦ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ، ﺣﺴﺎﺏ ﺳﺮﻣﺎﻳﻪ ﺭﺍ ﻣﻌﺎﺩﻝ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ ﺭﺍ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ.

 

ﺑﺴﺘﻦ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ ﺩﺭ ﺍﻧﺘﻬﺎﯼ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺩﺭ ﺷﺮﮐﺖ‌ﻫﺎﯼ ﺑﺎﺯﺭﮔﺎﻧﯽ
ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺑﺴﺘﻦ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ ﺩﺭ ﻣﻮﺳﺴﺎﺕ ﺑﺎﺯﺭﮔﺎﻧﯽ ﺑﺴﻴﺎﺭ ﺷﺒﻴﻪ ﺑﻪ ﺷﺮﮐﺖ‌ﻫﺎﯼ ﺧﺪﻣﺎﺗﯽ ﺍﺳﺖ. ﺗﻨﻬﺎ ﺗﻔﺎﻭﺕ ﺩﺭ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺣﺴﺎﺏ‌ﻫﺎﻳﯽ ﺩﺭ ﺷﺮﮐﺖ‌ﻫﺎﯼ ﺑﺎﺯﺭﮔﺎﻧﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺷﺮﮐﺖ‌ﻫﺎﯼ ﺧﺪﻣﺎﺗﯽ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺍﻳﻦ ﺣﺴﺎﺏ‌ﻫﺎ ﺷﺎﻣﻞ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺧﺮﻳﺪ، ﻓﺮﻭﺵ، ﺑﺮﮔﺸﺖ ﺍﺯ ﻓﺮﻭﺵ ﻭ ﺗﺨﻔﻴﻔﺎﺕ، ﺑﺮﮔﺸﺖ ﺍﺯ ﺧﺮﻳﺪ ﻭ ﺗﺨﻔﻴﻔﺎﺕ، ﺗﺨﻔﻴﻔﺎﺕ ﻧﻘﺪﯼ ﺧﺮﻳﺪ، ﺗﺨﻔﻴﻔﺎﺕ ﻧﻘﺪﯼ ﻓﺮﻭﺵ ﻭ ﻣﻮﺟﻮﺩﯼ ﮐﺎﻻﯼ ﺍﻭﻝ ﺩﻭﺭﻩ ﻭ ﻣﻮﺟﻮﺩﯼ ﮐﺎﻻﯼ ﺍﻧﺘﻬﺎﯼ ﺩﻭﺭﻩ ﻫﺴﺘﻨﺪ.

ﻣﺮﺣﻠﻪ ﺍﻭﻝ: ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭﺯﻳﺎﻥ ﺑﺪﻫﮑﺎﺭ ﻭ ﻣﻮﺟﻮﺩﯼ ﮐﺎﻻﯼ ﺍﻭﻝ ﺩﻭﺭﻩ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﺷﻮﺩ. ﺳﭙﺲ ﺣﺴﺎﺏ ﻣﻮﺟﻮﺩﯼ ﮐﺎﻻﯼ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﺷﻮﺩ.

ﻣﺮﺣﻠﻪ ﺩﻭﻡ: ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻓﺮﻭﺵ، ﺗﺨﻔﻴﻔﺎﺕ ﻧﻘﺪﯼ ﺧﺮﻳﺪ، ﻭ ﺑﺮﮔﺸﺖ ﺍﺯ ﺧﺮﻳﺪ ﻭ ﺗﺨﻔﻴﻔﺎﺕ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﺷﻮﺩ.

ﻣﺮﺣﻠﻪ ﺳﻮﻡ: ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺧﺮﻳﺪ، ﻫﺰﻳﻨﻪ‌ﻫﺎ، ﺑﺮﮔﺸﺖ ﺍﺯ ﻓﺮﻭﺵ ﻭ ﺗﺨﻔﻴﻔﺎﺕ ﻭ ﺗﺨﻔﻴﻔﺎﺕ ﻧﻘﺪﯼ ﻓﺮﻭﺵ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭﺯﻳﺎﻥ ﺑﺪﻫﮑﺎﺭ ﻣﯽ‌ﺷﻮﺩ.

ﻣﺮﺣﻠﻪ ﭼﻬﺎﺭﻡ: ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﮐﺎﻣﻼ ﺷﺒﻴﻪ ﻣﺮﺣﻠﻪ ﺳﻮﻡ ﺷﺮﮐﺖ‌ﻫﺎﯼ ﺧﺪﻣﺎﺗﯽ ﺍﺳﺖ. ﺍﮔﺮ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺧﻼﺻﻪ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺑﺴﺘﺎﻧﮑﺎﺭ ﺑﺎﺷﺪ ﺁﻥ ﺭﺍ ﺳﻮﺩ ﻭ ﺍﮔﺮ ﺑﺪﻫﮑﺎﺭ ﺑﺎﺷﺪ ﺁﻥ ﺭﺍ ﺯﻳﺎﻥ ﻣﻴﮕﻮﻳﻨﺪ.

ﻣﺮﺣﻠﻪ ﭘﻨﺠﻢ: ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﮐﺎﻣﻼ ﺷﺒﻴﻪ ﻣﺮﺣﻠﻪ ﭼﻬﺎﺭﻡ ﺷﺮﮐﺖ‌ﻫﺎﯼ ﺧﺪﻣﺎﺗﯽ ﺍﺳﺖ. ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﮐﻪ ﺑﺮﺍﯼ ﺑﺴﺘﻦ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ، ﺣﺴﺎﺏ ﺳﺮﻣﺎﻳﻪ ﺭﺍ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ ﺭﺍ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ.

 

ﻧﺤﻮﻩ ﺁﻣﺎﺩﻩ ﮐﺮﺩﻥ ﺗﺮﺍﺯ ﺁﺯﻣﺎﻳﺸﯽ ﺍﺧﺘﺘﺎﻣﯽ

ﻧﺤﻮﻩ ﺍﻧﺘﻘﺎﻝ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺩﺍﺋﻤﯽ ﺑﻪ ﺳﺎﻝ ﺑﻌﺪ
