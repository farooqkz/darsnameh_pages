ﺩﺭﺱ ﺷﺸﻢ- ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ، ﺻﻨﺪﻭﻕ ﻭ ﺑﺎﻧﮏ

ﻣﻬﻢ‌ﺗﺮﻳﻦ ﺩﺍﺭﺍﻳﯽ ﻫﺮ ﻣﻮﺳﺴﻪ ﻭﺟﻪ ﻧﻘﺪ ﺁﻥ ﻣﻮﺳﺴﻪ ﺍﺳﺖ. ﺑﺪﻭﻥ ﺩﺍﺷﺘﻦ ﻭﺟﻪ ﻧﻘﺪ ﻓﻌﺎﻟﻴﺖ ﻭ ﺍﻧﺠﺎﻡ ﺍﻣﻮﺭ ﺍﻗﺘﺼﺎﺩﯼ ﺑﺮﺍﯼ ﻫﻴﭻ ﺷﺮﮐﺘﯽ ﻣﻘﺪﻭﺭ ﻧﻴﺴﺖ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺛﺒﺖ ﺍﻣﻮﺭ ﻣﺎﻟﯽ ﻭ ﮐﻨﺘﺮﻝ ﻣﺎﻟﯽ ﺑﺮ ﻭﺟﻪ ﻧﻘﺪ ﺍﺯ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭ ﺑﺎﻻﻳﯽ ﺑﺮﺧﻮﺭﺩﺍﺭ ﺍﺳﺖ.

 

ﺣﺴﺎﺏ ﺻﻨﺪﻭﻕ
ﺑﺮﺍﯼ ﺍﻣﻮﺭ ﺭﻭﺯﻣﺮﻩ ﻣﺎﻧﻨﺪ ﭘﺮﺩﺍﺧﺖ‌ﻫﺎ ﻭ ﺩﺭﻳﺎﻓﺖ‌ﻫﺎ ﻭﺟﻪ ﻧﻘﺪ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ ﮐﻪ ﺗﺤﺖ ﻋﻨﻮﺍﻥ ﺻﻨﺪﻭﻕ ﺩﺭ ﺩﻓﺘﺮ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﺩ.

ﺑﺎ ﻭﺟﻮﺩ ﺍﻳﻨﮑﻪ ﺩﺭ ﺩﻧﻴﺎﯼ ﭘﻴﺸﺮﻓﺘﻪ ﺍﻣﺮﻭﺯ ﺷﺮﮐﺘﻬﺎ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﻨﺪ ﺍﺯ ﺩﺭﻳﺎﻓﺖ ﻭ ﭘﺮﺩﺍﺧﺖ ﻣﺴﺘﻘﻴﻢ ﭘﻮﻝ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻨﺪ ﻭ ﺍﻳﻦ ﻧﻘﻞ ﻭ ﺍﻧﺘﻘﺎﻝ‌ﻫﺎ ﺭﺍ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻋﻤﻠﻴﺎﺕ ﺑﺎﻧﮑﯽ ﺍﻧﺠﺎﻡ ﺩﻫﻨﺪ، ﻭﻟﯽ ﻧﻮﻉ ﻓﻌﺎﻟﻴﺖ ﺑﻌﻀﯽ ﺍﺯ ﻣﻮﺳﺴﺎﺕ ﮐﺎﺭ ﺑﺎ ﭘﻮﻝ ﻧﻘﺪ ﺭﺍ ﺩﺭ ﻃﯽ ﻋﻤﻠﻴﺎﺕ ﺭﻭﺯﺍﻧﻪ ﺍﺟﺘﻨﺎﺏ ‌ﻧﺎﭘﺬﻳﺮ ﻣﯽ‌ﮐﻨﺪ. ﻣﺎﻧﻨﺪ ﺧﺮﺩﻩ ﻓﺮﻭﺷﯽ‌ﻫﺎ ﮐﻪ ﻣﺠﺒﻮﺭ ﻫﺴﺘﻨﺪ ﺍﺯ ﺣﺴﺎﺏ ﺻﻨﺪﻭﻕ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪ.

 

ﺛﺒﺖ ﺣﺴﺎﺑﺪﺍﺭﯼ ﺻﻨﺪﻭﻕ
ﺣﺴﺎﺏ ﺻﻨﺪﻭﻕ ﺟﺰﻭ ﺩﺍﺭﺍﻳﯽ‌ﻫﺎﯼ ﺷﺮﮐﺖ ﻣﺤﺴﻮﺏ ﻣﯽ‌ﺷﻮﺩ ﺩﺭ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﻓﺰﺍﻳﺶ ﺁﻥ ﺑﺪﻫﮑﺎﺭ ﻭ ﮐﺎﻫﺶ ﺁﻥ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﺷﻮﺩ.

ﻣﺜﺎﻝ: ﺷﺮﮐﺖ ﺳﻌﻴﺪﯼ ﺩﺭ ﺗﺎﺭﻳﺦ ۱۳۸۹/۷/۵ ﻣﺒﻠﻎ ۶۵۰ ﻫﺰﺍﺭ ﺭﻳﺎﻝ ﺍﺯ ﺷﺮﮐﺖ ﻃﻴﻒ ﮔﺴﺘﺮ ﺩﺭﻳﺎﻓﺖ ﮐﺮﺩ. ﺛﺒﺖ ﺍﻳﻦ ﺭﻭﻳﺪﺍﺩ ﺩﺭ ﺩﻓﺘﺮ ﺍﻳﻦ ﺩﻭ ﺷﺮﮐﺖ ﺑﻪ ﺻﻮﺭﺕ ﺯﻳﺮ ﺍﺳﺖ:

ﺷﺮﮐﺖ ﺳﻌﻴﺪﯼ:

ﺷﺮﮐﺖ ﻃﻴﻒ ﮔﺴﺘﺮ:

 

ﮐﻨﺘﺮﻝ ﻣﺎﻧﺪﻩ ﻭﺍﻗﻌﯽ ﺻﻨﺪﻭﻕ ﺑﺎ ﺩﻓﺎﺗﺮ ﻭ ﺛﺒﺖ ﻣﻐﺎﻳﺮﺍﺕ
ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﻣﻮﺍﺭﺩ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭﻳﺎﻓﺖ‌ﻫﺎ ﻭ ﭘﺮﺩﺍﺧﺖ‌ﻫﺎﯼ ﻣﺘﻌﺪﺩ ﺗﻮﺳﻂ ﺻﻨﺪﻭﻕ‌ﺩﺍﺭ ﻣﻨﺠﺮ ﺑﻪ ﺍﺿﺎﻓﻪ ﻳﺎ ﮐﺴﺮ ﺩﺭﻳﺎﻓﺖ ﺷﻮﺩ. ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﺑﺎﻳﺪ ﺑﻪ ﻃﻮﺭ ﻣﺪﺍﻭﻡ ﻣﻮﺟﻮﺩﯼ ﻧﻘﺪ ﺻﻨﺪﻭﻕ ﺑﺎ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺻﻨﺪﻭﻕ ﺩﺭ ﺩﻓﺘﺮ ﮐﻞ ﻣﻘﺎﻳﺴﻪ ﺷﻮﺩ ﺗﺎ ﻫﺮﮔﻮﻧﻪ ﺍﺷﺘﺒﺎﻩ ﻣﻮﺭﺩ ﺑﺮﺭﺳﯽ ﻗﺮﺍﺭ ﮔﻴﺮﺩ.

ﺩﺭ ﺑﻌﻀﯽ ﺍﺯ ﻣﻮﺳﺴﺎﺕ ﮐﺴﺮﯼ ﻳﺎ ﺍﺿﺎﻓﺎﺕ ﺻﻨﺪﻭﻕ ﺭﺍ ﺩﺭ ﺣﺴﺎﺑﯽ ﺑﻪ ﻧﺎﻡ ﺣﺴﺎﺏ ﮐﺴﻮﺭ ﻭ ﺍﺿﺎﻓﺎﺕ ﺻﻨﺪﻭﻕ ﺛﺒﺖ ﻣﯽ‌ﮐﻨﻨﺪ. ﻭﻗﺘﯽ ﮐﻪ ﺩﺭ ﺻﻨﺪﻭﻕ ﮐﺴﺮﯼ ﺍﻳﺠﺎﺩ ﺷﻮﺩ ﺣﺴﺎﺏ ﻣﺬﮐﻮﺭ ﺑﺪﻫﮑﺎﺭ ﻭ ﺯﻣﺎﻧﯽ ﮐﻪ ﺻﻨﺪﻭﻕ ﻭﺟﻪ ﺍﺿﺎﻓﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺣﺴﺎﺏ ﻣﺬﮐﻮﺭ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﺷﻮﺩ.

ﻣﺜﺎﻝ: ﺟﻤﻊ ﻓﺮﻭﺵ ﻓﺮﻭﺷﮕﺎﻩ ﺳﻌﻴﺪﯼ ﺩﺭ ﺗﺎﺭﻳﺦ ۸۸/۷/۲۲ ﻣﺒﻠﻎ ۷۰۰، ۶۵۵ ﺭﻳﺎﻝ ﺍﺳﺖ ﻭ ﻭﺟﻪ ﻧﻘﺪ ﺣﺎﺻﻞ ﺷﺪﻩ ﺩﺭ ﺁﻥ ﺭﻭﺯ ۳۰۰، ۶۵۲ ﺭﻳﺎﻝ ﺍﺳﺖ. ﺍﻳﻦ ﺭﻭﻳﺪﺍﺩ ﻣﺎﻟﯽ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺩﺭ ﺩﻓﺘﺮ ﺭﻭﺯﻧﺎﻣﻪ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﺩ.

 

ﺣﺴﺎﺏ ﺑﺎﻧﮏ
ﺍﻣﺮﻭﺯﻩ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺷﺮﮐﺘﻬﺎ ﺍﺯ ﻧﮕﻬﺪﺍﺭﯼ ﻭﺟﻪ ﻧﻘﺪ ﺩﺭ ﺻﻨﺪﻭﻕ ﺷﺮﮐﺖ ﺧﻮﺩﺩﺍﺭﯼ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺍﺯ ﺣﺴﺎﺏ ﺑﺎﻧﮑﯽ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﮐﻨﺘﺮﻝ ﻣﺎﻟﯽ ﺑﺮ ﻭﺟﻪ ﻧﻘﺪ ﺑﻬﺘﺮ ﻭ ﺩﻗﻴﻖ‌ﺗﺮ ﺻﻮﺭﺕ ﺑﮕﻴﺮﺩ.

 

ﺛﺒﺖ ﻋﻤﻠﻴﺎﺕ ﻣﺎﻟﯽ ﺩﺭ ﺣﺴﺎﺏ ﺑﺎﻧﮑﯽ
ﺣﺴﺎﺏ ﺑﺎﻧﮑﯽ ﻫﻢ ﻣﺎﻧﻨﺪ ﺻﻨﺪﻭﻕ ﻳﮏ ﺩﺍﺭﺍﻳﯽ ﺍﺳﺖ ﻭ ﺑﺪﻫﮑﺎﺭ ﻭ ﺑﺴﺘﺎﻧﮑﺎﺭ ﮐﺮﺩﻥ ﺁﻥ ﺷﺒﻴﻪ ﺻﻨﺪﻭﻕ ﺍﺳﺖ. ﭘﺲ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﭘﻮﻝ ﺑﻪ ﺣﺴﺎﺏ ﺑﺎﻧﮑﯽ ﻭﺍﺭﻳﺰ ﻣﯽ‌ﺷﻮﺩ، ﺣﺴﺎﺏ ﺑﺎﻧﮏ ﺑﺪﻫﮑﺎﺭ ﻭ ﺯﻣﺎﻧﯽ ﮐﻪ ﺍﺯ ﺣﺴﺎﺏ ﺑﺎﻧﮏ ﺑﺮﺩﺍﺷﺖ ﻣﯽ‌ﺷﻮﺩ ﺣﺴﺎﺏ ﺑﺎﻧﮏ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﺷﻮﺩ.

 

ﮐﻨﺘﺮﻝ ﻣﻐﺎﻳﺮﺍﺕ ﺑﺎﻧﮑﯽ ﻭ ﺗﻬﻴﻪ ﺻﻮﺭﺕ ﺁﻥ
ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﺷﺮﮐﺖ‌ﻫﺎ ﻭ ﻣﻮﺳﺴﺎﺕ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﻣﺎﻟﯽ ﺭﺍ ﺩﺭ ﺩﻓﺎﺗﺮ ﺧﻮﺩ ﺛﺒﺖ ﻣﯽ‌ﮐﻨﻨﺪ، ﺑﺎﻧﮏ ﻧﻴﺰ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭ ﺩﻓﺎﺗﺮ ﺧﻮﺩ ﺛﺒﺖ ﻣﯽ‌ﮐﻨﺪ. ﺍﮔﺮ ﺛﺒﺖ‌ﻫﺎ ﺩﺭ ﻫﻤﺎﻥ ﺗﺎﺭﻳﺦ ﻭ ﺑﻪ ﺩﺭﺳﺘﯽ ﺛﺒﺖ ﺷﻮﺩ ﻣﻌﻤﻮﻻ ﻫﻴﭻ ﻣﻐﺎﻳﺮﺗﯽ ﺑﻴﻦ ﺩﻓﺎﺗﺮ ﺑﺎﻧﮏ ﻭ ﺷﺮﮐﺖ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺑﻪ ﻫﺮ ﺣﺎﻝ ﻋﻮﺍﻣﻞ ﻣﺨﺘﻠﻔﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﻮﺟﺐ ﻣﯽ‌ﺷﻮﻧﺪ ﺗﺎ ﺑﻴﻦ ﺩﻓﺎﺗﺮ ﺷﺮﮐﺖ ﻭ ﺑﺎﻧﮏ ﺍﺧﺘﻼﻑ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﮐﻪ ﺩﺭ ﺯﻳﺮ ﺑﻌﻀﯽ ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﺁﻧﻬﺎ ﺷﺮﺡ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﻧﺪ:

- ﺍﻗﻼﻡ ﺑﺎﺯ
ﻣﻌﻤﻮﻻ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﻣﺎﻟﯽ ﮐﻪ ﺣﺴﺎﺑﺪﺍﺭ ﺷﺮﮐﺖ ﺩﺭ ﺩﻓﺎﺗﺮ ﺛﺒﺖ ﻣﯽ‌ﮐﻨﺪ ﺑﻪ ﻣﺤﺾ ﺍﺭﺳﺎﻝ ﺍﻃﻼﻋﺎﺕ ﺑﻪ ﺑﺎﻧﮏ ﺩﺭ ﺩﻓﺎﺗﺮ ﺑﺎﻧﮏ ﻧﻴﺰ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﺩ. ﻭﻟﯽ ﭼﻨﺎﻧﭽﻪ ﺑﻪ ﺧﺎﻃﺮ ﻋﻮﺍﻣﻞ ﻣﺨﺘﻠﻔﯽ ﮐﻪ ﺩﺭ ﺯﻳﺮ ﺑﻪ ﺁﻥ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ ﺭﻭﻳﺪﺍﺩ ﻓﻘﻂ ﺩﺭ ﺩﻓﺎﺗﺮ ﺷﺮﮐﺖ ﻳﺎ ﺑﺎﻧﮏ ﺛﺒﺖ ﺷﻮﺩ ﻭ ﺩﺭ ﺩﻓﺘﺮ ﺩﻳﮕﺮﯼ ﺛﺒﺖ ﻧﺸﻮﺩ، ﺍﺻﻄﻼﺣﺎ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﮐﻪ ﻣﺒﻠﻎ ﺛﺒﺖ ﺷﺪﻩ ﺩﺭ ﺩﻓﺎﺗﺮ ﻃﺮﻓﯽ ﮐﻪ ﺛﺒﺖ ﮐﺮﺩﻩ ﺑﺎﺯ ﺍﺳﺖ.

- ﺍﻗﻼﻡ ﺑﺎﺯ ﺻﻮﺭﺕ ﺣﺴﺎﺏ ﺑﺎﻧﮏ
ﺍﻳﻦ ﻧﻮﻉ ﺍﻗﻼﻡ ﺁﻧﻬﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﺎﻧﮏ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭ ﺩﻓﺎﺗﺮ ﺧﻮﺩ ﺛﺒﺖ ﮐﺮﺩﻩ ﻭﻟﯽ ﺣﺴﺎﺑﺪﺍﺭ ﺷﺮﮐﺖ ﺁﻧﻬﺎ ﺭﺍ ﺗﺎ ﭘﺎﻳﺎﻥ ﻣﺎﻩ ﺩﺭ ﻓﺎﺗﺮ ﺧﻮﺩ ﺛﺒﺖ ﻧﮑﺮﺩﻩ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺁﻧﻬﺎ ﺍﻗﻼﻡ ﺑﺎﺯ ﺻﻮﺭﺕ ﺣﺴﺎﺏ ﺑﺎﻧﮏ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

- ﺍﻗﻼﻡ ﺑﺎﺯ ﺩﻓﺎﺗﺮ ﺷﺮﮐﺖ
ﺑﻪ ﻭﺟﻮﻫﯽ ﮐﻪ ﺷﺮﮐﺖ ﺁﻧﻬﺎ ﺭﺍ ﺩﺭ ﺩﻓﺎﺗﺮ ﺧﻮﺩ ﺛﺒﺖ ﮐﺮﺩﻩ ﻭﻟﯽ ﺑﻪ ﻫﺮ ﻋﻠﺘﯽ ﺗﺎ ﭘﺎﻳﺎﻥ ﻣﺎﻩ ﺩﺭ ﺩﻓﺎﺗﺮ ﺑﺎﻧﮏ ﺛﺒﺖ ﻧﺸﺪﻩ ﺍﺳﺖ ﺭﺍ ﺍﻗﻼﻡ ﺑﺎﺯ ﺩﻓﺎﺗﺮ ﺷﺮﮐﺖ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

ﺑﻪ ﻃﻮﺭ ﺧﻼﺻﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﮔﻔﺖ ﮐﻪ ﻋﺪﻡ ﺗﻄﺎﺑﻖ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺩﻓﺎﺗﺮ ﺷﺮﮐﺖ ﺑﺎ ﺩﻓﺎﺗﺮ ﺑﺎﻧﮏ ﻧﺎﺷﯽ ﺍﺯ ﻣﻮﺍﺭﺩ ﺯﻳﺮ ﺍﺳﺖ:

۱. ﻣﺪﺍﺭﮎ ﻭﺍﺭﻳﺰﯼ‌ﻫﺎ ﻭ ﭼﮏ‌ﻫﺎﯼ ﺻﺎﺩﺭﻩ ﺍﺯ ﺳﻮﯼ ﺷﺮﮐﺖ ﻫﻤﺰﻣﺎﻥ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺑﺎﻧﮏ ﻗﺮﺍﺭ ﻧﻤﯽ‌ﮔﻴﺮﺩ.

۲. ﺍﺷﺘﺒﺎﻩ ﺣﺴﺎﺑﺪﺍﺭ ﺷﺮﮐﺖ ﻳﺎ ﺑﺎﻧﮏ. ﺑﻌﻀﯽ ﻣﻮﺍﻗﻊ ﻣﻤﮑﻦ ﺍﺳﺖ ﺣﺴﺎﺑﺪﺍﺭ ﺷﺮﮐﺖ ﻳﺎ ﺑﺎﻧﮏ ﻣﺒﺎﻟﻎ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺍﺷﺘﺒﺎﻩ ﻭﺍﺭﺩ ﮐﻨﻨﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﺩﺭ ﻃﺮﻑ ﻧﺎﺩﺭﺳﺖ ﺣﺴﺎﺏ ﻋﻤﻞ ﻧﻤﺎﻳﻨﺪ. ﻣﺜﻼ ﻣﺒﻠﻐﯽ ﺭﺍ ﮐﻪ ﺑﺎﻳﺪ ﺩﺭ ﺑﺴﺘﺎﻧﮑﺎﺭ ﺛﺒﺖ ﺷﻮﺩ ﺩﺭ ﺑﺪﻫﮑﺎﺭ ﺛﺒﺖ ﮐﻨﻨﺪ ﻳﺎ ﺑﺮﻋﮑﺲ. ﺩﺭ ﻣﻮﺍﺭﺩﯼ ﻧﻴﺰ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﮐﺎﺭ ﻋﻤﺪﺍ ﺑﺎ ﻫﺪﻑ ﺍﺧﺘﻼﺱ ﺍﻧﺠﺎﻡ ﮔﻴﺮﺩ.

ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﺩﺭ ﺻﻮﺭﺕ ﻣﻐﺎﻳﺮﺕ ﺑﺎﻧﮑﯽ، ﺍﻗﻼﻡ ﺑﺎﺯ ﺑﺪﻫﮑﺎﺭ ﺻﻮﺭﺕ‌ ﺣﺴﺎﺏ ﺑﺎﻧﮏ ﺍﺯ ﻣﺎﻧﺪﻩ ﻃﺒﻖ ﺩﻓﺎﺗﺮ ﺷﺮﮐﺖ ﮐﺴﺮ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺍﻗﻼﻡ ﺑﺎﺯ ﺑﺴﺘﺎﻧﮑﺎﺭ ﺻﻮﺭﺕ ﺣﺴﺎﺏ ﺑﺎﻧﮏ ﺑﻪ ﺁﻥ ﺍﺿﺎﻓﻪ ﻣﯽ‌ﺷﻮﺩ. ﻫﻤﭽﻨﻴﻦ ﺍﻗﻼﻡ ﺑﺎﺯ ﺑﺴﺘﺎﻧﮑﺎﺭ ﺩﻓﺎﺗﺮ ﺷﺮﮐﺖ ﺍﺯ ﻣﺎﻧﺪﻩ ﻃﺒﻖ ﺻﻮﺭﺕ‌ ﺣﺴﺎﺏ ﺑﺎﻧﮏ ﮐﺴﺮ ﻭ ﺍﻗﻼﻡ ﺑﺪﻫﮑﺎﺭ ﺩﻓﺎﺗﺮ ﺷﺮﮐﺖ ﺑﻪ ﺍﻥ ﺍﺿﺎﻓﻪ ﻣﯽ‌ﺷﻮﺩ.

 

ﺣﺴﺎﺏ ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ
ﻣﻌﻤﻮﻻ ﺷﺮﮐﺖ‌ﻫﺎ ﭘﺮﺩﺍﺧﺖ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﭼﮏ ﺍﻧﺠﺎﻡ ﻣﯽ ﺩﻫﻨﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﭘﺮﺩﺍﺧﺖ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﺟﺰﻳﯽ ﺍﺯ ﺣﺴﺎﺏ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﻭ ﻣﻌﻘﻮﻝ ﻧﻴﺴﺖ. ﺑﺮﺍﯼ ﺣﻞ ﺍﻳﻦ ﻣﺸﮑﻞ ﻣﺒﻠﻐﯽ ﭘﻮﻝ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻳﮏ ﻳﺎ ﭼﻨﺪ ﻧﻔﺮ ﺍﺯ ﮐﺎﺭﮐﻨﺎﻥ ﺷﺮﮐﺖ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ ﮐﻪ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﺟﺰﻳﯽ ﺭﺍ ﭘﺮﺩﺍﺧﺖ ﮐﻨﻨﺪ. ﺑﻪ ﺍﻳﻦ ﻣﺒﻠﻎ ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

 

ﺛﺒﺖ ﻋﻤﻠﻴﺎﺕ ﻣﺎﻟﯽ ﺩﺭ ﺣﺴﺎﺏ ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ
ﺣﺴﺎﺏ ﺗﻨﺨﻮﺍﻩ ﺑﻪ ﻋﻨﻮﺍﻥ ﺩﺍﺭﺍﻳﯽ ﻣﺤﺴﻮﺏ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺑﺪﻫﮑﺎﺭ ﻭ ﺑﺴﺘﺎﻧﮑﺎﺭ ﮐﺮﺩﻥ ﺁﻥ ﺑﺴﻴﺎﺭ ﺷﺒﻴﻪ ﺣﺴﺎﺏ ﺻﻨﺪﻭﻕ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﺛﺒﺖ ﻋﻤﻠﻴﺎﺕ ﻣﺎﻟﯽ ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ ﺍﺯ ﺩﻭ ﺭﻭﺵ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﺷﻮﺩ:

- ﺭﻭﺵ ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ ﺛﺎﺑﺖ ﮐﻪ ﺩﺭ ﺁﻥ ﻫﻨﮕﺎﻡ ﺍﺭﺍﺋﻪ ﺍﺳﻨﺎﺩ ﻫﺰﻳﻨﻪ، ﺩﺭ ﺣﺴﺎﺏ ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ ﺛﺒﺘﯽ ﺍﻧﺠﺎﻡ ﻧﻤﯽ‌ﺷﻮﺩ.

- ﺭﻭﺵ ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ ﻣﺘﻐﻴﺮ ﮐﻪ ﺩﺭ ﺁﻥ ﻫﻨﮕﺎﻡ ﺍﺭﺍﺋﻪ ﺍﺳﻨﺎﺩ ﻫﺰﻳﻨﻪ، ﺣﺴﺎﺏ ﻫﺰﻳﻨﻪ ﻣﺮﺑﻮﻁ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺗﻨﺨﻮﺍﻩ ﮔﺮﺩﺍﻥ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﺷﻮﺩ.
