ﺩﺭﺱ ﻫﻔﺘﻢ- ﺍﺻﻼﺡ ﻭ ﺗﻌﺪﻳﻞ ﺣﺴﺎﺏ ﻫﺎ

ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻗﺒﻼ ﮔﻔﺘﻪ ﺷﺪ ﻫﺪﻑ ﺍﺯ ﺗﻬﻴﻪ ﺍﻃﻼﻋﺎﺕ ﻣﺎﻟﯽ ﮐﻤﮏ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﺪﮔﺎﻥ ﺍﺯ ﺍﻳﻦ ﮔﺰﺍﺭﺷﺎﺕ ﺍﺳﺖ ﺗﺎ ﺑﺘﻮﺍﻧﻨﺪ ﺩﺭﺳﺖ ﻭ ﺻﺤﻴﺢ ﺑﺮﺍﯼ ﺁﻳﻨﺪﻩ ﺷﺮﮐﺖ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﮐﻨﻨﺪ. ﺑﺮﺍﯼ ﻫﻤﻴﻦ ﺗﻤﺎﻡ ﺍﻃﻼﻋﺎﺕ ﺑﺎﻳﺪ ﺩﺭﺳﺖ ﻭ ﻭﺍﻗﻌﯽ ﺑﺎﺷﻨﺪ.

ﻳﮑﯽ ﺍﺯ ﻣﺸﮑﻼﺗﯽ ﮐﻪ ﺣﺴﺎﺑﺪﺍﺭﺍﻥ ﺑﺎ ﺁﻥ ﺭﻭﺑﺮﻭ ﻫﺴﺘﻨﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺮﺧﯽ ﺍﺯ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﻣﺎﻟﯽ ﺑﺮ ﺑﻴﺶ ﺍﺯ ﻳﮏ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺍﺛﺮ ﻣﯽ‌ﮔﺬﺍﺭﻧﺪ. ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﺍﺛﺮ ﺍﻳﻦ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﻣﺎﻟﯽ ﺑﺎﻳﺪ ﺑﻪ ﺩﻗﺖ ﺍﻧﺪﺍﺯﻩ‌ﮔﻴﺮﯼ ﺷﻮﺩ. ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﺩﺭ ﭘﺎﻳﺎﻥ ﻫﺮ ﺩﻭﺭﻩ ﺑﺎﻳﺪ ﺍﺻﻼﺣﺎﺕ ﻻﺯﻡ ﺩﺭ ﺣﺴﺎﺏ‌ﻫﺎ ﺍﻧﺠﺎﻡ ﺷﻮﺩ. ﺩﺭ ﺍﻳﻦ ﻓﺼﻞ ﺩﺭ ﻣﻮﺭﺩ ﺍﻳﻦ ﺍﺻﻼﺣﺎﺕ ﻭ ﺛﺒﺖ‌ﻫﺎ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ.

 

ﺍﻧﻮﺍﻉ ﺣﺴﺎﺏ‌ﻫﺎ
ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻧﺘﻘﺎﻝ ﺩﺍﺩﻥ ﻳﺎ ﻧﺪﺍﺩﻥ ﻣﺎﻧﺪﻩ ﺣﺴﺎﺏ ﺑﻪ ﺳﺎﻝ ﺑﻌﺪ ﻣﺎﻟﯽ، ﺣﺴﺎﺏ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺩﺳﺘﻪ‌ﻫﺎﯼ ﺯﻳﺮ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﮐﻨﻨﺪ:

۱- ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺩﺍﺋﻤﯽ
ﺁﻧﻬﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﺎﻧﺪﻩ ﺁﻧﻬﺎ ﺑﻪ ﺩﻭﺭﻩ ﺑﻌﺪ ﻣﻨﺘﻘﻞ ﻣﯽ‌ﺷﻮﺩ. ﻣﺎﻧﻨﺪ ﺩﺍﺭﺍﻳﯽ‌ﻫﺎ، ﺑﺪﻫﯽ‌ﻫﺎ، ﻭ ﺳﺮﻣﺎﻳﻪ. ﺑﻪ ﺍﻳﻦ ﺣﺴﺎﺏ‌ﻫﺎ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺗﺮﺍﺯﻧﺎﻣﻪ‌ﺍﯼ ﻫﻢ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

۲- ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﻮﻗﺖ
ﺁﻧﻬﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﺭ ﭘﺎﻳﺎﻥ ﺳﺎﻝ ﻣﺎﻟﯽ ﺑﺴﺘﻪ ﻣﯽ‌ﺷﻮﻧﺪ. ﻣﺎﻧﻨﺪ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺩﺭﺁﻣﺪ ﻭ ﻫﺰﻳﻨﻪ ﻭ ﺑﺮﺩﺍﺷﺖ. ﺑﻪ ﺍﻳﻦ ﺣﺴﺎﺏ‌ﻫﺎ ﺑﻪ ﺍﺳﺘﺜﻨﺎﯼ ﺣﺴﺎﺏ ﺑﺮﺩﺍﺷﺖ، ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺳﻮﺩ ﻭ ﺯﻳﺎﻧﯽ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

۳- ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﺨﺘﻠﻂ
ﺑﺮﺧﯽ ﺍﺯ ﺣﺴﺎﺏ‌ﻫﺎ ﺩﻭﮔﺎﻧﻪ ﻫﺴﺘﻨﺪ ﻳﻌﻨﯽ ﺑﺨﺸﯽ ﺍﺯ ﺁﻧﻬﺎ ﺩﺍﺋﻤﯽ ﻭ ﺑﺨﺶ ﺩﻳﮕﺮﯼ ﺍﺯ ﺁﻥ ﻣﻮﻗﺖ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻫﻤﻴﻦ ﺧﺎﻃﺮ ﺑﻪ ﺁﻧﻬﺎ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﻣﺨﺘﻠﻂ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

 

ﺍﺻﻼﺡ ﺣﺴﺎﺏ‌ﻫﺎ ﺩﺭ ﻃﻮﻝ ﺩﻭﺭﻩ ﻣﺎﻟﯽ
ﺑﻪ ﺧﺎﻃﺮ ﺗﻄﺎﺑﻖ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﻫﺮ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺍﺯ ﺩﺭﺁﻣﺪﻫﺎﯼ ﻫﻤﺎﻥ ﺩﻭﺭﻩ، ﺑﺎﻳﺪ ﺍﻳﻦ ﺍﻃﻤﻴﻨﺎﻥ ﺣﺎﺻﻞ ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺯﻣﺎﻥ ﺗﻬﻴﻪ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ، ﻫﻤﻪ ﺩﺭﺁﻣﺪﻫﺎ ﻭ ﻫﺰﻳﻨﻪ‌ﻫﺎ ﺛﺒﺖ ﺷﺪﻩ ﺑﺎﺷﺪ. ﺍﮔﺮ ﺑﻌﻀﯽ ﺍﺯ ﺩﺭﺁﻣﺪﻫﺎ ﻳﺎ ﻫﺰﻳﻨﻪ‌ﻫﺎ ﺛﺒﺖ ﻧﺸﺪﻩ ﺑﺎﺷﺪ، ﺣﺴﺎﺑﺪﺍﺭﺍﻥ ﻣﻮﻇﻒ ﻫﺴﺘﻨﺪ ﺁﻧﻬﺎ ﺭﺍ ﺷﻨﺎﺳﺎﻳﯽ ﮐﺮﺩﻩ ﻭ ﺳﭙﺲ ﺣﺴﺎﺏ‌ﻫﺎ ﺭﺍ ﺍﺻﻼﺡ ﮐﻨﻨﺪ. ﺍﺻﻄﻼﺣﺎ ﺑﻪ ﺍﻳﻦ ﮔﻮﻧﻪ ﺛﺒﺖ‌ﻫﺎ ﺛﺒﺖ‌ﻫﺎﯼ ﺍﺻﻼﺣﯽ ﻣﯽ‌ﮔﻮﻳﻨﺪ. ﺍﻳﻦ ﺛﺒﺖ‌ﻫﺎ ﻣﻌﻤﻮﻻ ﺩﺭ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﻭ ﻗﺒﻞ ﺍﺯ ﺗﻬﻴﻪ ﺻﻮﺭﺕ‌ﻫﺎﯼ ﻣﺎﻟﯽ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺑﺮﺧﯽ ﺍﺯ ﺣﺴﺎﺏ‌ﻫﺎ ﺩﺭ ﻃﯽ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﻧﻴﺰ ﺍﺻﻼﺡ ﺷﻮﺩ.

 

ﺍﺻﻼﺡ ﺣﺴﺎﺏ‌ﻫﺎ ﺩﺭ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ ﻣﺎﻟﯽ
ﺩﺭ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺑﻌﺪ ﺍﺯ ﺍﻳﻨﮑﻪ ﺗﻤﺎﻡ ﻣﻌﺎﻣﻼﺕ ﻭ ﺭﻭﻳﺪﺍﺩﻫﺎﯼ ﻣﺎﻟﯽ ﺩﺭ ﺩﻓﺎﺗﺮ ﻣﻮﺳﺴﻪ ﺛﺒﺖ ﺷﺪ ﻣﻌﻤﻮﻻ ﺑﻌﻀﯽ ﺍﺯ ﺣﺴﺎﺏ‌ﻫﺎ ﺩﺭ ﺩﻓﺘﺮ ﮐﻞ ﻣﺎﻧﺪﻩ‌ﻫﺎﯼ ﺻﺤﻴﺢ ﺭﺍ ﺍﺭﺍﺋﻪ ﻧﻤﯽ‌ﺩﻫﻨﺪ. ﻋﻼﻭﻩ ﺑﺮ ﻭﺟﻮﺩ ﺍﻳﻦ ﻣﺸﮑﻞ، ﺑﺮﺧﯽ ﺍﺯ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺩﻓﺘﺮ ﮐﻞ ﺩﺭ ﭘﺎﻳﺎﻥ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﺑﺎﻳﺪ ﺩﺭ ﺟﻬﺖ ﺗﻄﺒﻴﻖ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺗﺮﺍﺯﻧﺎﻣﻪ ﻭ ﺻﻮﺭﺕ ﺳﻮﺩ ﻭ ﺯﻳﺎﻥ ﺍﺻﻼﺡ ﮔﺮﺩﺩ.

ﺑﻪ ﻃﻮﺭ ﻣﻌﻤﻮﻝ ﭘﺒﻪ‌ﻫﺎﯼ ﺍﺻﻼﺣﯽ ﺷﺎﻣﻞ ﻣﻮﺍﺭﺩ ﺯﻳﺮ ﻣﯽ‌ﺷﻮﻧﺪ:
۱. ﺍﺻﻼﺡ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﺗﺤﻘﻖ ﻧﻴﺎﻓﺘﻪ
۲. ﺍﺻﻼﺡ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﺗﺤﻘﻖ ﻳﺎﻓﺘﻪ ﭘﺮﺩﺍﺧﺖ ﻧﺸﺪﻩ ﻣﺎﻧﻨﺪ ﻋﻮﺍﺭﺽ
۳. ﺍﺻﻼﺡ ﺩﺍﺭﻳﯽ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﺒﻠﻐﯽ ﺍﺯ ﺁﻧﻬﺎ ﺩﺭ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﻫﺰﻳﻨﻪ ﺷﺪﻩ
۴. ﺍﺻﻼﺡ ﭘﻴﺶ ﺩﺭﻳﺎﻓﺖ‌ﻫﺎ ﻭ ﺑﺪﻫﯽ ﺟﺎﺭﯼ

ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﺍﺻﻼﺣﺎﺕ ﺑﺎﻻ ﺑﺎﻳﺪ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﺯﻳﺮ ﺍﺻﻼﺡ ﺷﻮﻧﺪ:
- ﺍﺻﻼﺡ ﭘﻴﺶ ﭘﺮﺩﺍﺧﺖ‌ﻫﺎﯼ ﻫﺰﻳﻨﻪ
- ﺛﺒﺖ ﻫﺰﻳﻨﻪ ﺍﺳﺘﻬﻼﮎ ﺩﺍﺭﺍﻳﯽ‌ﻫﺎﯼ ﺍﺳﺘﻬﻼﮎ ‌ﭘﺬﻳﺮ
- ﺛﺒﺖ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﺗﺤﻤﻴﻞ ﺷﺪﻩ
- ﺍﺻﻼﺡ ﭘﻴﺶ ﺩﺭﻳﺎﻓﺖ‌ﻫﺎﯼ ﺩﺭﺁﻣﺪ
- ﺍﺻﻼﺡ ﺣﺴﺎﺏ ﻣﻮﺟﻮﺩﯼ ﻣﻠﺰﻭﻣﺎﺕ
- ﺛﺒﺖ ﺩﺭﺁﻣﺪﻫﺎﯼ ﺗﺤﻘﻖ ﻳﺎﻓﺘﻪ ﺛﺒﺖ ﻧﺸﺪﻩ

 

ﺍﺻﻼﺡ ﭘﻴﺶ ﭘﺮﺩﺍﺧﺖ‌ﻫﺎﯼ ﻫﺰﻳﻨﻪ
ﺍﮔﺮ ﺩﺭ ﻃﻮﻝ ﺩﻭﺭﻩ ﻣﺎﻟﯽ ﻗﺒﻞ ﺍﺯ ﺩﺭﻳﺎﻓﺖ ﮐﺎﻻ ﻳﺎ ﺧﺪﻣﺎﺕ ﻣﺒﻠﻐﯽ ﭘﺮﺩﺍﺧﺖ ﺷﻮﺩ، ﺍﻳﻦ ﻣﺒﻠﻎ ﺩﺭ ﺩﻓﺘﺮ ﻣﻮﺳﺴﻪ ﭘﺮﺩﺍﺧﺖ ﮐﻨﻨﺪﻩ ﺑﻪ ﻋﻨﻮﺍﻥ ﭘﻴﺶ ﭘﺮﺩﺍﺧﺖ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﺩ. ﻫﺮ ﺯﻣﺎﻥ ﮐﻪ ﺍﻳﻦ ﮐﺎﻻ ﻳﺎ ﺧﺪﻣﺎﺕ ﺍﺯ ﺷﺮﮐﺖ ﻃﺮﻑ ﻗﺮﺍﺭﺩﺍﺩ ﺩﺭﻳﺎﻓﺖ ﺷﺪ، ﺑﻪ ﻣﻘﺪﺍﺭ ﮐﺎﻻ ﻳﺎ ﺧﺪﻣﺎﺕ ﺩﺭﻳﺎﻓﺘﻪ ﺷﺪﻩ ﺍﺯ ﺣﺴﺎﺏ ﭘﻴﺶ ﭘﺮﺩﺍﺧﺖ ﺧﺎﺭﺝ ﻭ ﺑﻪ ﺣﺴﺎﺏ ﻣﺮﺑﻮﻃﻪ ﻣﻨﺘﻘﻞ ﻣﯽ‌ﺷﻮﺩ.

 

ﺛﺒﺖ ﻫﺰﻳﻨﻪ ﺍﺳﺘﻬﻼﮎ ﺩﺍﺭﺍﻳﯽ‌ﻫﺎﯼ ﺍﺳﺘﻬﻼﮎ‌ ﭘﺬﻳﺮ
ﮐﻠﻴﻪ ﺩﺍﺭﻳﯽ‌ﻫﺎﯼ ﺷﺮﮐﺖ ﮐﻪ ﻋﻤﺮ ﺁﻧﻬﺎ ﺑﻴﺸﺘﺮ ﺍﺯ ﻳﮏ ﺳﺎﻝ ﺑﺎﺷﺪ، ﺑﻪ ﻋﻨﻮﺍﻥ ﺩﺍﺭﻳﯽ ﺑﻠﻨﺪ ﻣﺪﺕ ﻳﺎ ﺩﺍﺭﺍﻳﯽ ﺛﺎﺑﺖ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﺩ. ﺑﻪ ﻏﻴﺮ ﺍﺯ ﺯﻣﻴﻦ، ﮐﻠﻴﻪ ﺩﺍﺭﺍﻳﯽ‌ﻫﺎﯼ ﺑﻠﻨﺪ ﻣﺪﺕ ﻳﮏ ﻣﻮﺳﺴﻪ ﺩﺍﺭﺍﯼ ﻋﻤﺮ ﻣﺤﺪﻭﺩ ﺍﺳﺖ. ﺑﻪ ﻫﻤﻴﻦ ﺟﻬﺖ ﻗﻴﻤﺖ ﺗﻤﺎﻡ ﺷﺪﻩ ﺍﻳﻦ ﺩﺍﺭﻳﯽ‌ﻫﺎ ﺩﺭ ﻃﯽ ﻣﺪﺕ ﻣﻔﻴﺪﺷﺎﻥ ﺑﻪ ﻫﺰﻳﻨﻪ ﻣﻨﻈﻮﺭ ﻣﯽ‌ﺷﻮﺩ.

ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ ﺩﺭ ﭘﺎﻳﺎﻥ ﻫﺮ ﺳﺎﻝ ﻣﺎﻟﯽ ﻗﺴﻤﺖ ﻣﺸﺨﺼﯽ ﺍﺯ ﺑﻬﺎﯼ ﺗﻤﺎﻡ ﺷﺪﻩ ﺍﻳﻦ ﺩﺍﺭﻳﯽ‌ﻫﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﻫﺰﻳﻨﻪ ﺍﺳﺘﻬﻼﮎ ﺩﺭ ﺩﻓﺘﺮ ﺛﺒﺖ ﻣﯽ‌ﺷﻮﺩ.

ﺩﺭ ﺛﺒﺖ ﻫﺰﻳﻨﻪ ﺍﺳﺘﻬﻼﮎ ﺣﺴﺎﺏ ﻫﺰﻳﻨﻪ ﺍﺳﺘﻬﻼﮎ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ ﺍﺳﺘﻬﻼﮎ ﺍﻧﺒﺎﺷﺘﻪ ﺭﺍ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ.

 

ﺛﺒﺖ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﺗﺤﻤﻴﻞ ﺷﺪﻩ ﺛﺒﺖ ﻧﺸﺪﻩ
ﺑﻌﻀﯽ ﺍﺯ ﻫﺰﻳﻨﻪ ﻫﺎ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﮐﻪ ﻣﻌﻤﻮﻻ ﺩﺭ ﭘﺎﻳﺎﻥ ﺳﺎﻝ ﺛﺒﺖ ﻧﺸﺪﻩ ﺍﻧﺪ. ﺍﺯ ﺟﻤﻠﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﺁﺏ ﻭ ﺑﺮﻕ ﻣﺼﺮﻓﯽ ﺩﺭ ﺭﻭﺯﻫﺎﯼ ﭘﺎﻳﺎﻧﯽ ﺳﺎﻝ ﺍﺷﺎﺭﻩ ﮐﺮﺩ. ﺑﻪ ﺍﻳﻦ ﮔﻮﻧﻪ ﻫﺰﻳﻨﻪ‌ﻫﺎ، ﻫﺰﻳﻨﻪ‌ﻫﺎﯼ ﻣﻌﻮﻕ ﻫﻢ ﻣﯽ‌ﮔﻮﻳﻨﺪ.

ﺟﻬﺖ ﺛﺒﺖ ﺍﻳﻦ ﻫﺰﻳﻨﻪ‌ﻫﺎ ﺣﺴﺎﺏ ﻫﺰﻳﻨﻪ ﻣﺮﺑﻮﻁ ﺭﺍ ﺑﻪ ﻣﻴﺰﺍﻥ ﻣﺒﻠﻎ ﻫﺰﻳﻨﻪ ﺑﺪﻫﮑﺎﺭ ﻭ ﺣﺴﺎﺏ‌ﻫﺎﯼ ﭘﺮﺩﺍﺧﺘﻨﯽ ﺭﺍ ﺑﺴﺘﺎﻧﮑﺎﺭ ﻣﯽ‌ﮐﻨﻴﻢ.

 

ﺍﺻﻼﺡ ﭘﻴﺶ ﺩﺭﻳﺎﻓﺖ‌ﻫﺎﯼ ﺩﺭﺁﻣﺪ
