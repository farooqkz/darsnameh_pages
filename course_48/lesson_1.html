ﺩﺭﺱ ﺍﻭﻝ- ﭼﮕﻮﻧﮕﯽ ﺍﻧﺘﺨﺎﺏ ﻳﮏ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺍﻣﻦ ﻭ ﻣﻨﺎﺳﺐ

ﺷﺎﻳﺪ ﺩﺭ ﻧﮕﺎﻩ ﺍﻭﻝ ﻃﺮﺡ ﭼﻨﻴﻦ ﻣﻮﺿﻮﻋﯽ ﺩﺭ ﻣﺒﺤﺚ ﺍﻣﻨﻴﺖ ﻭﺑﻼﮒ ﻧﻮﻳﺴﯽ ﭼﻨﺪﺍﻥ ﺑﻪ ﺟﺎ ﻭ ﻣﻨﺎﺳﺐ ﺑﻪ ﻧﻈﺮ ﻧﺮﺳﺪ. ﺧﺐ، ﻣﮕﺮ ﺍﻧﺘﺨﺎﺏ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﭼﻪ ﺭﺑﻄﯽ ﺑﻪ ﺍﻣﻨﻴﺖ ﻭﺑﻼﮒ ﺩﺍﺭﺩ؟ ﭼﻪ ﻓﺮﻗﯽ ﻣﯽ ﮐﻨﺪ ﮐﻪ ﻣﻦ ﺍﺯ ﻭﻳﻨﺪﻭﺯ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻢ ﻳﺎ ﻟﻴﻨﻮﮐﺲ؟ ﻣﮕﺮ ﺑﺮﺍﯼ ﻭﺑﻼﮒ ﻧﻮﻳﺴﯽ ﻣﻴﺎﻥ ﻣﮏ ﻭ ﻟﻴﻨﻮﮐﺲ ﺗﻔﺎﻭﺗﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ؟ ﭼﺮﺍ ﺑﺎﻳﺪ ﺧﻮﺩﻡ ﺭﺍ ﺑﻪ ﺩﺭﺩﺳﺮ ﺍﻧﺪﺍﺧﺘﻪ ﻭ ﺍﺯ ﻭﻳﻨﺪﻭﺯ ﺑﻪ ﻟﻴﻨﻮﮐﺲ ﻳﺎ ﻣﮑﻴﻨﺘﺎﺵ ﻣﻬﺎﺟﺮﺕ ﮐﻨﻢ؟ ﻭﻳﻨﺪﻭﺯ، ﻭﻳﻨﺪﻭﺯ ﺍﺳﺖ ﺩﻳﮕﺮ! ﺣﺎﻻ ﭼﻪ ﻧﺴﺨﻪ ﺍﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﺩﻳﮕﺮ ﺑﺴﺘﮕﯽ ﮐﺎﻣﻞ ﺑﻪ ﺳﻠﻴﻘﻪ ﻣﺎﻥ ﺩﺍﺭﺩ!

ﺩﺭ ﺍﻭﻟﻴﻦ ﺩﺭﺱ ﺍﻳﻦ ﺩﻭﺭﻩ ﻣﯽ ﺧﻮﺍﻫﻴﻢ ﮐﻤﯽ ﺩﻗﻴﻖ ﺗﺮ ﺑﻪ ﻣﺴﺎﻟﻪ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺑﭙﺮﺩﺍﺯﻳﻢ، ﺗﺎ ﺑﺒﻴﻨﻴﻢ ﭼﺮﺍ ﺑﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﺧﺼﻮﺹ ﻫﻢ ﻧﮕﺮﺍﻥ ﻭ ﻣﺮﺍﻗﺐ ﺑﺎﺷﻴﻢ. ﺷﺎﻳﺪ ﺍﻭﻟﻴﻦ ﻭ ﻣﻬﻤﺘﺮﻳﻦ ﻧﮑﺘﻪ ﺩﺭ ﺍﻣﻨﻴﺖ ﻳﮏ ﻭﺑﻼﮒ ﻧﻮﻳﺲ ﺣﺮﻓﻪ ﺍﯼ، ﺍﻧﺘﺨﺎﺏ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺑﺎﺷﺪ.

ﻫﻤﺎﻥ ﮔﻮﻧﻪ ﮐﻪ ﻣﯽ ﺩﺍﻧﻴﺪ، ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﺯ ﺳﻄﻮﺡ ﺍﻣﻨﻴﺘﯽ ﻣﺘﻔﺎﻭﺗﯽ ﺑﻬﺮﻩ ﻣﯽ ﺑﺮﻧﺪ ﻭ ﺩﺭ ﮐﻨﺎﺭ ﺑﺪﺍﻓﺰﺍﺭ ﻭ ﻭﻳﺮﻭﺱ ﻫﺎﯼ ﻣﺘﻨﻮﻉ، ﺍﻧﻮﺍﻉ ﮔﻮﻧﺎﮔﻮﻧﯽ ﺍﺯ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﺍﻣﻨﻴﺘﯽ ﻫﻢ ﺑﺮﺍﯼ ﺁﻧﻬﺎ ﺩﺭ ﺑﺎﺯﺍﺭ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﭘﺲ ﻳﮏ ﻭﺑﻼﮒ ﻧﻮﻳﺲ ﮐﻪ ﺍﻫﻤﻴﺖ ﺑﺎﻻﻳﯽ ﺑﺮﺍﯼ ﺍﻣﻨﻴﺖ ﺧﻮﺩ ﻭ ﻭﺑﻼﮔﺶ ﻗﺎﻳﻞ ﺍﺳﺖ، ﺑﺎﻳﺪ ﺩﺭ ﺍﻭﻟﻴﻦ ﻗﺪﻡ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻠﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﺪ ﮐﻪ ﺍﺯ ﺍﻣﻨﻴﺖ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺧﻮﺭﺩﺍﺭ ﺑﻮﺩﻩ ﻭ ﺑﺎ ﮐﻤﺘﺮﻳﻦ ﻣﻴﺰﺍﻥ ﺗﻬﺪﻳﺪﺍﺕ ﻭ ﺍﻳﺮﺍﺩﻫﺎ ﺭﻭﺑﺮﻭ ﺑﺎﺷﺪ. ﺩﺭ ﺍﺩﺍﻣﻪ ﺑﻪ ﺑﺮﺭﺳﯽ ﻧﻘﺎﻁ ﻗﻮﺕ ﻭ ﺿﻌﻒ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﯽ ﭘﺮﺩﺍﺯﻳﻢ ﺗﺎ ﺑﺒﻴﻨﻴﻢ ﮐﺪﺍﻡ ﻳﮏ ﻣﯽ ﺗﻮﺍﻧﺪ ﺍﻧﺘﺨﺎﺏ ﻣﻨﺎﺳﺒﯽ ﺑﺎﺷﺪ.

ﺍﺣﺘﻤﺎﻻ ﺷﻤﺎ ﻗﺒﻞ ﺍﺯ ﺍﻳﻨﮑﻪ ﺗﺼﻤﻴﻢ ﺑﻪ ﻭﺑﻼﮒ ﻧﻮﻳﺲ ﺷﺪﻥ ﺑﮕﻴﺮﻳﺪ، ﮐﺎﻣﭙﻴﻮﺗﺮ ﻳﺎ ﻟﭗ ﺗﺎﭖ ﺷﺨﺼﯽ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺩﺍﺷﺘﻪ ﺍﻳﺪ ﻭ ﮐﻤﺘﺮ ﮐﺴﯽ ﭘﻴﺪﺍ ﻣﯽ ﺷﻮﺩ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﺍﺧﺘﺼﺎﺻﯽ ﺑﺮﺍﯼ ﻭﺑﻼﮒ ﻧﻮﻳﺴﯽ ﺑﻪ ﺩﻧﺒﺎﻝ ﺧﺮﻳﺪ ﻳﮏ ﺳﻴﺴﺘﻢ ﺟﺪﻳﺪ ﺑﺎﺷﺪ. ﺍﻣﺎ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﯽ ﺗﻮﺍﻧﺪ ﺗﺎﺛﻴﺮ ﻓﺮﺍﻭﺍﻧﯽ ﺩﺭ ﮐﺎﺭﺁﻳﯽ، ﺍﻣﻨﻴﺖ ﻭ ﺣﺘﯽ ﺧﻼﻗﻴﺖ ﻳﮏ ﻭﺑﻼﮒ ﻧﻮﻳﺲ ﺣﺮﻓﻪ ﺍﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﻫﻨﮕﺎﻡ ﺍﻧﺘﺨﺎﺏ ﻳﮏ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﻨﺎﺳﺐ ﭼﻨﺪﻳﻦ ﻋﺎﻣﻞ ﻣﯽ ﺗﻮﺍﻧﻨﺪ ﺗﺎﺛﻴﺮ ﮔﺬﺍﺭ ﺑﺎﺷﻨﺪ:

۱- ﺍﻳﻨﮑﻪ ﺷﻤﺎ ﻗﺮﺍﺭ ﺍﺳﺖ ﭼﻪ ﺍﺳﺘﻔﺎﺩﻩ ﻫﺎﻳﯽ ﺍﺯ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻳﺎ ﻟﭗ ﺗﺎﭖ ﺗﺎﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺁﻳﺎ ﺍﻳﻦ ﺩﺳﺘﮕﺎﻩ ﺑﻪ ﻃﻮﺭ ﺍﺧﺘﺼﺎﺻﯽ ﻣﺘﻌﻠﻖ ﺑﻪ ﻣﺪﻳﺮﻳﺖ ﻭﺑﻼﮒ ﻭ ﺳﺎﻳﺖ ﺷﻤﺎ ﺍﺳﺖ؟ ﻳﺎ ﺍﻳﻨﮑﻪ ﮐﺎﺭﻫﺎﯼ ﺍﺩﺍﺭﯼ ﻭ ﺷﻐﻠﯽ ﺗﺎﻥ ﺭﺍ ﻫﻢ ﺑﺎﻳﺪ ﺍﻧﺠﺎﻡ ﺩﻫﺪ؟ ﺷﺎﻳﺪ ﻗﺮﺍﺭ ﺍﺳﺖ ﺍﺑﺰﺍﺭ ﺑﺎﺯﯼ ﻭ ﺳﺮﮔﺮﻣﯽ ﺗﺎﻥ ﻫﻢ ﺑﺎﺷﺪ؟

۲- ﺑﻪ ﭼﻪ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﻳﯽ ﺟﻬﺖ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎﯼ ﺗﺎﻥ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ؟ ﭘﺎﺳﺦ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ، ﺑﺴﺘﮕﯽ ﺯﻳﺎﺩﯼ ﺑﻪ ﺳﻮﺍﻝ ﻗﺒﻞ ﺩﺍﺭﺩ. ﺑﺴﺘﻪ ﺑﻪ ﺍﻳﻨﮑﻪ ﭼﻪ ﻧﻮﻉ ﮐﺎﺭﺑﺮﺩﯼ ﺭﺍ ﺍﺯ ﺩﺳﺘﮕﺎﻩ ﺗﺎﻥ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺭﻳﺪ، ﻧﻮﻉ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﺗﺎﻥ ﻫﻢ ﺗﻌﻴﻴﻦ ﻣﯽ ﮔﺮﺩﺩ. ﺯﻳﺮﺍ ﺩﺭ ﺻﻮﺭﺕ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺑﺮﺧﯽ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﺧﺎﺹ ﮐﻪ ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻭﻳﮋﻩ ﺍﯼ ﺗﻮﻟﻴﺪ ﺷﺪﻩ ﺍﻧﺪ، ﻣﺠﺒﻮﺭ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻫﻤﺎﻥ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺴﺘﻴﺪ.

۳- ﻣﻴﺰﺍﻥ ﺗﺒﺤﺮ ﺷﻤﺎ ﺩﺭ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻫﺮ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻭ ﺑﻪ ﺍﺻﻄﻼﺡ ﮔﻴﮏ ﺑﻮﺩﻥ ﺗﺎﻥ ﺩﺭ ﺯﻣﻴﻨﻪ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻫﻢ ﮔﺰﻳﻨﻪ ﺗﻌﻴﻴﻦ ﮐﻨﻨﺪﻩ ﺩﻳﮕﺮﯼ ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﮐﻪ ﺗﺎ ﭼﻪ ﻣﻴﺰﺍﻥ ﻋﻼﻗﻪ ﻣﻨﺪ ﺑﻪ ﻳﺎﺩﮔﻴﺮﯼ ﻭ ﮐﺴﺐ ﻣﻬﺎﺭﺕ ﺩﺭ ﮐﺎﺭ ﺑﺎ ﺑﺮﻧﺎﻣﻪ ﻫﺎ ﻭ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺎﯼ ﺟﺪﻳﺪ ﻫﻢ ﺑﺎﺷﻴﺪ، ﺩﺭ ﺍﻳﻦ ﺍﻣﺮ ﺩﺧﻴﻞ ﺍﺳﺖ. ﮔﺎﻫﯽ ﺑﻪ ﺩﻟﻴﻞ ﻧﺎﺁﺷﻨﺎ ﺑﻮﺩﻥ ﻳﮏ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻣﻤﮑﻦ ﺍﺳﺖ ﭼﺸﻢ ﺗﺎﻥ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﻫﻤﻪ ﺧﻮﺑﯽ ﻫﺎﯼ ﺁﻥ ﺑﺒﻨﺪﻳﺪ ﻭ ﺳﺮﺍﻍ ﻳﮏ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺭﺍﺣﺖ ﻭ ﺁﺷﻨﺎ، ﻭﻟﯽ ﻣﺸﮑﻞ ﺩﺍﺭ ﺑﺮﻭﻳﺪ. ﻳﺎ ﺍﻳﻨﮑﻪ ﺳﺨﺘﯽ ﻫﺎﯼ ﭼﻨﺪ ﻣﺎﻫﻪ ﻳﺎﺩﮔﻴﺮﯼ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺟﺪﻳﺪ ﺭﺍ ﺑﻪ ﺟﺎﻥ ﺧﺮﻳﺪﻩ ﻭ ﺩﺭ ﺍﺯﺍﯼ ﺁﻥ ﻣﺰﺍﻳﺎ ﻭ ﺍﻣﮑﺎﻧﺎﺕ ﺗﺎﺯﻩ ﻓﺮﺍﻭﺍﻧﯽ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﻳﺪ.

ﻳﮑﯽ ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﮔﺰﻳﻨﻪ ﻫﺎ ﺩﺭ ﺍﻧﺘﺨﺎﺏ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ، ﮐﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﺍﻫﻤﻴﺖ ﺩﻭ ﭼﻨﺪﺍﻧﯽ ﭘﻴﺪﺍ ﻣﯽ ﮐﻨﺪ، ﺍﻣﻨﻴﺖ ﺁﻧﻬﺎ ﺍﺳﺖ. ﻫﻤﻪ ﺷﺮﮐﺖ ﻫﺎﯼ ﺳﺎﺯﻧﺪﻩ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺗﻼﺵ ﻣﯽ ﮐﻨﻨﺪ ﮐﻪ ﺗﺎ ﺣﺪ ﻣﻤﮑﻦ ﺍﻣﻦ ﺗﺮﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺭﺍ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﮐﺎﺭﺑﺮﺍﻥ ﺷﺎﻥ ﻗﺮﺍﺭ ﺩﻫﻨﺪ. ﺍﻣﺎ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﻣﺴﺎﻳﻠﯽ ﻫﻤﭽﻮﻥ ﺗﻌﺪﺍﺩ ﮐﺎﺭﺑﺮﺍﻥ، ﻣﻴﺰﺍﻥ ﺗﻮﺟﻪ ﻫﮑﺮﻫﺎ ﻭ ﺍﻓﺮﺍﺩ ﻧﻔﻮﺫﮔﺮ ﺑﻪ ﺁﻥ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ، ﺑﺎﮒ ﻫﺎ ﻭ ﺍﻳﺮﺍﺩﺍﺕ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻭ ﻫﻤﭽﻨﻴﻦ ﺳﻮﺭﺍﺥ ﻫﺎﯼ ﺍﻣﻨﻴﺘﯽ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﺟﺎﻧﺒﯽ ﻳﮏ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ، ﻣﯽ ﺗﻮﺍﻥ ﺁﻧﻬﺎ ﺭﺍ ﺩﺳﺘﻪ ﺑﻨﺪﯼ ﻧﻤﻮﺩﻩ ﻭ ﺑﺮ ﺣﺴﺐ ﺍﻣﻨﻴﺖ ﻭ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻣﮑﺎﻧﺎﺕ ﻣﻮﺟﻮﺩ، ﺩﺳﺖ ﺑﻪ ﺍﻧﺘﺨﺎﺏ ﺯﺩ. ﺗﻮﺻﻴﻪ ﻣﺎ ﻫﻢ ﺑﻪ ﻭﺑﻼﮒ ﻧﻮﻳﺲ ﻫﺎﯼ ﺣﺮﻓﻪ ﺍﯼ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﻴﺸﺘﺮﻳﻦ ﺗﻮﺟﻪ ﺭﺍ ﺑﻪ ﺍﻳﻦ ﻣﻮﺭﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ.

ﻭﻳﻨﺪﻭﺯ، ﺳﻴﺴﺘﻢ ﻋﺎﻣﻠﯽ ﻫﻤﻪ ﮔﻴﺮ ﺍﺳﺖ ﮐﻪ ﺑﻴﺸﺘﺮﻳﻦ ﮐﺎﺭﺑﺮﺩ ﺭﺍ ﺩﺭ ﺩﻧﻴﺎﯼ ﮐﺎﻣﭙﻴﻮﺗﺮ ﻭ ﻟﭗ ﺗﺎﭖ ﺩﺍﺭﺩ ﻭ ﺗﻘﺮﻳﺒﺎ ﻫﻤﻪ ﺷﻤﺎ ﻫﻢ ﺍﮐﻨﻮﻥ ﺑﺎ ﺁﻥ ﮐﺎﺭ ﻣﯽ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﻗﺒﻼ ﮐﺎﺭ ﮐﺮﺩﻩ ﺍﻳﺪ. ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺳﺎﺯﮔﺎﺭﯼ ﺑﺴﻴﺎﺭ ﺧﻮﺑﯽ ﺑﺎ ﺳﺨﺖ ﺍﻓﺰﺍﺭﻫﺎﯼ ﻣﺘﻨﻮﻉ ﻭ ﻣﺘﻌﺪﺩ ﻣﻮﺟﻮﺩ ﺩﺭ ﺑﺎﺯﺍﺭ ﺩﺍﺷﺘﻪ ﻭ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﯼ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺩﺭ ﺯﻣﻴﻨﻪ ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﺮﺍﯼ ﺁﻥ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺍﺳﺖ. ﺗﻘﺮﻳﺒﺎ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﻫﺮ ﮐﺎﺭﯼ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﺭﺍﻳﮕﺎﻥ ﻭ ﻏﻴﺮﺭﺍﻳﮕﺎﻥ ﻣﺘﻌﺪﺩﯼ ﺑﻴﺎﺑﻴﺪ ﮐﻪ ﺍﻣﮑﺎﻥ ﺍﻧﺘﺨﺎﺏ ﺧﻮﺑﯽ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﻣﯽ ﺩﻫﻨﺪ.

ﮐﺎﺭ ﺑﺎ ﺁﻥ ﻧﺴﺒﺖ ﺑﻪ ﺩﻳﮕﺮ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻫﺎﯼ ﻣﺸﻬﻮﺭ، ﺳﺎﺩﻩ ﺗﺮ ﺍﺳﺖ. ﺩﺭ ﺻﻮﺭﺕ ﺑﺮﻭﺯ ﻫﺮﮔﻮﻧﻪ ﻣﺸﮑﻞ ﻭ ﻳﺎ ﻣﻮﺍﺟﻪ ﺷﺪﻥ ﺑﺎ ﺍﻳﺮﺍﺩﯼ، ﺍﻓﺮﺍﺩ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺩﺭ ﺍﻃﺮﺍﻑ ﺗﺎﻥ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺁﻧﻬﺎ ﮐﻤﮏ ﺑﮕﻴﺮﻳﺪ. ﻧﺮﻡ ﺍﻓﺰﺍﺭ، ﺳﺨﺖ ﺍﻓﺰﺍﺭ، ﻭﺳﺎﻳﻞ ﺟﺎﻧﺒﯽ ﻭ ﻣﺘﺨﺼﺼﺎﻥ ﺁﻥ ﺭﺍ ﺗﻘﺮﻳﺒﺎ ﺩﺭ ﻫﺮ ﺟﺎﻳﯽ ﻣﯽ ﺗﻮﺍﻥ ﻳﺎﻓﺖ ﻭ ﻫﻴﭻ ﮔﺎﻩ ﺍﺯ ﺍﻳﻦ ﻧﻈﺮ ﺑﺎ ﻣﺸﮑﻞ ﺭﻭﺑﺮﻭ ﻧﻤﯽ ﺷﻮﻳﺪ.

ﺑﻪ ﺩﻟﻴﻞ ﮔﺴﺘﺮﻩ ﺑﺴﻴﺎﺭ ﻭﺳﻴﻊ ﮐﺎﺭﺑﺮﺍﻥ ﻭ ﻫﻤﭽﻨﻴﻦ ﺑﺎﮒ ﻫﺎﯼ ﻣﻮﺟﻮﺩ ﺩﺭ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺑﺮﺍﯼ ﺁﻥ، ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺑﺴﻴﺎﺭ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻫﮑﺮﻫﺎ ﻭ ﺍﻓﺮﺍﺩ ﺧﺮﺍﺑﮑﺎﺭ ﺑﻮﺩﻩ ﻭ ﺗﻘﺮﻳﺒﺎ ﻫﺪﻑ ﺍﺻﻠﯽ ﺁﻧﻬﺎ ﺩﺭ ﺩﻧﻴﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺍﺳﺖ. ﺍﻟﺒﺘﻪ ﺑﻪ ﻫﻤﺎﻥ ﻧﺴﺒﺖ ﻫﻢ ﺗﻌﺪﺍﺩ ﺑﺮﻧﺎﻣﻪ ﻫﺎﯼ ﺍﻣﻨﻴﺘﯽ، ﻓﺎﻳﺮﻭﺍﻝ ﻭ ﺁﻧﺘﯽ ﻭﻳﺮﻭﺱ ﺭﺍﻳﮕﺎﻥ ﻭ ﻏﻴﺮ ﺭﺍﻳﮕﺎﻥ ﺧﻮﺏ، ﺑﺮﺍﯼ ﺍﻳﻦ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﻣﺎ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺍﻫﻞ ﺭﺍﺣﺖ ﻃﻠﺒﯽ ﻧﺒﺎﺷﻴﺪ ﻭ ﺑﺨﻮﺍﻫﻴﺪ ﮐﻤﯽ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﺳﺨﺖ ﺑﮕﻴﺮﻳﺪ، ﻭﻳﻨﺪﻭﺯ ﺍﻧﺘﺨﺎﺏ ﺍﻭﻝ ﺩﺭﺳﻨﺎﻣﻪ ﺑﺮﺍﯼ ﻳﮏ ﻭﺑﻼﮒ ﻧﻮﻳﺲ ﺣﺮﻓﻪ ﺍﯼ ﻧﻴﺴﺖ.

ﺍﻟﺒﺘﻪ ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﺍﻣﮑﺎﻥ ﻓﻨﯽ ﻳﺎ ﻭﻗﺖ ﻻﺯﻡ ﺑﺮﺍﯼ ﮐﻮﭺ ﺑﻪ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺩﻳﮕﺮﯼ ﻫﻤﭽﻮﻥ ﻟﻴﻨﻮﮐﺲ ﻳﺎ ﻣﮏ ﺭﺍ ﻧﺪﺍﺭﻳﺪ، ﺑﺎ ﺑﻪ ﮐﺎﺭ ﺑﺴﺘﻦ ﺑﺮﺧﯽ ﻧﮑﺎﺕ ﺍﻣﻨﻴﺘﯽ ﻣﯽ ﺗﻮﺍﻧﻴﺪ ﻭﻳﻨﺪﻭﺯ ﺭﺍ ﻫﻢ ﺗﺒﺪﻳﻞ ﺑﻪ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﻗﺎﺑﻞ ﻗﺒﻮﻟﯽ ﺑﺮﺍﯼ ﻭﺑﻼﮒ ﻧﻮﻳﺴﯽ ﺍﻣﻦ ﮐﻨﻴﺪ:

۱- ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﺍﺯ ﺳﻴﺴﺘﻢ ﺧﻮﺩ ﺗﻨﻬﺎ ﻭ ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﻳﮏ ﮐﺎﺭ، ﻭ ﺁﻥ ﻫﻢ ﻭﺑﻼﮒ ﻧﻮﻳﺴﯽ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﻭﻗﺘﯽ ﻗﺮﺍﺭ ﺑﺎﺷﺪ ﭼﻨﺪ ﮐﺎﺭ ﺑﺎ ﻳﮏ ﻟﭗ ﺗﺎﭖ ﻳﺎ ﮐﺎﻣﭙﻴﻮﺗﺮ ﺍﻧﺠﺎﻡ ﺷﻮﺩ، ﺟﺎﺑﺠﺎﻳﯽ ﺍﻃﻼﻋﺎﺕ ﺑﺎ ﺍﻓﺮﺍﺩ ﻭ ﮔﺮﻭﻩ ﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﺍﺯ ﻃﺮﻳﻖ ﺍﺑﺰﺍﺭﻫﺎﯼ ﮔﻮﻧﺎﮔﻮﻥ ﻣﯽ ﺗﻮﺍﻧﺪ ﻧﻘﻄﻪ ﺿﻌﻒ ﺍﻣﻨﻴﺘﯽ ﺑﺰﺭﮔﯽ ﺑﺮﺍﯼ ﺳﻴﺴﺘﻢ ﺷﻤﺎ ﺑﺎﺷﺪ. ﺷﺎﻳﺪ ﺷﻤﺎ ﺩﺭ ﺍﻳﻤﻨﯽ ﺍﻳﻤﻴﻞ ﻣﺘﻌﻠﻖ ﺑﻪ ﻭﺑﻼﮒ ﻭ ﻳﺎ ﮐﻮﻝ ﺩﻳﺴﮏ ﺣﺎﻭﯼ ﺍﻃﻼﻋﺎﺕ ﺁﻥ ﺩﻗﺖ ﮐﻨﻴﺪ، ﻭﻟﯽ ﮐﻢ ﮐﺎﺭﯼ ﺩﺭ ﺗﻮﺟﻪ ﺑﻪ ﺍﻣﻨﻴﺖ ﻫﺎﺭﺩ ﺩﻳﺴﮏ ﺁﺭﺷﻴﻮ ﻓﻴﻠﻢ، ﻳﺎ ﺍﻳﻤﻴﻞ ﻣﺮﺗﺒﻂ ﺑﺎ ﮐﺎﺭ ﺷﺮﮐﺖ ﻭ ﻳﺎ ﺍﻳﻤﻴﻠﯽ ﮐﻪ ﻣﻄﺎﻟﺐ ﺳﺎﻳﺖ ﻫﺎﯼ ﺳﺮﮔﺮﻣﯽ ﺭﺍ ﺑﺎ ﺁﻥ ﺩﺭﻳﺎﻓﺖ ﻣﯽ ﮐﻨﻴﺪ، ﻣﯽ ﺗﻮﺍﻧﺪ ﺑﻪ ﻧﻘﻄﻪ ﺷﺮﻭﻋﯽ ﺟﻬﺖ ﺣﻤﻠﻪ ﺑﻪ ﻭﺑﻼﮒ ﺗﺎﻥ ﺗﺒﺪﻳﻞ ﺷﻮﺩ.
