ﺩﺭﺱ ﺩﻭﻡ- ﺍﻧﻮﺍﻉ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﻳﻨﺘﺮﻧﺘﯽ

ﺩﺭ ﺩﺭﺱ ﻧﺨﺴﺖ ﺑﺎ ﻣﻔﺎﻫﻴﻢ ﮐﻠﻴﺪﯼ ﺗﺠﺎﺭﺕ ﻭ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺁﺷﻨﺎ ﺷﺪﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺗﻮﺳﻌﻪ ﻣﻔﺎﻫﻴﻤﯽ ﻫﻤﭽﻮﻥ «ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ» ﻭ ﺍﺳﺘﺮﺍﺗﮋﯼ‌ﻫﺎﯼ ﻣﻮﺟﻮﺩ ﺩﺭ ﺍﻳﻨﺘﺮﻧﺖ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.

ﺷﮑﯽ ﻧﻴﺴﺖ ﮐﻪ ﺍﻳﻨﺘﺮﻧﺖ ﺩﻧﻴﺎﻳﯽ ﺭﺍ ﮐﻪ ﺩﺭ ﺁﻥ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﮐﻨﻴﻢ ﺗﻐﻴﻴﺮ ﺩﺍﺩﻩ ﺍﺳﺖ. ﭘﻴﺶ ﺍﺯ ﺍﻳﻦ ﻫﺮﮔﺰ ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﺍﻃﻼﻋﺎﺕ، ﺑﺮﻗﺮﺍﺭﯼ ﺍﺭﺗﺒﺎﻁ ﺑﺎ ﻣﺮﺩﻡ ﺩﺭ ﺳﺮﺗﺎﺳﺮ ﺟﻬﺎﻥ ﻭ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻣﻘﺎﻻﺕ، ﻓﻴﻠﻤﻬﺎﯼ ﻭﻳﺪﺋﻮﻳﯽ، ﻋﮑﺲ‌ﻫﺎ ﻭ ﺩﻳﮕﺮ ﺍﺷﮑﺎﻝ ﺭﺳﺎﻧﻪ ﺑﻪ ﺍﻳﻦ ﺁﺳﺎﻧﯽ ﻧﺒﻮﺩﻩ ﺍﺳﺖ.

ﺍﻓﺰﺍﻳﺶ ﭼﺸﻤﮕﻴﺮ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻳﻨﺘﺮﻧﺖ ﺩﺭ ﻋﺼﺮ ﺟﺪﻳﺪ ﻣﻨﺠﺮ ﺑﻪ ﮐﺎﻫﺶ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺳﻨﺘﯽ ﻣﺎﻧﻨﺪ ﺗﻠﻮﻳﺰﻳﻮﻥ، ﺭﺍﺩﻳﻮ، ﺭﻭﺯﻧﺎﻣﻪ‌ﻫﺎ ﻭ ﻣﺠﻼﺕ ﺷﺪﻩ ﺍﺳﺖ.

ﺩﺭ ﻃﻮﻝ ﺩﻭ ﺩﻫﻪ ﮔﺬﺷﺘﻪ، ﺑﺎﺯﺍﺭﻳﺎﺏ‌ﻫﺎ ﺭﻓﺘﻪ ﺭﻓﺘﻪ ﺑﺎ ﻗﺪﺭﺕ ﺍﻳﻨﺘﺮﻧﺖ، ﻫﻢ ﺑﻪ ﻋﻨﻮﺍﻥ ﺑﺴﺘﺮﯼ ﺑﺮﺍﯼ ﺑﺮﻗﺮﺍﺭﯼ ﺍﺭﺗﺒﺎﻁ ﻭ ﻫﻢ ﺑﻪ ﻋﻨﻮﺍﻥ ﺭﻭﺷﯽ ﺑﺮﺍﯼ ﺭﺩﻳﺎﺑﯽ ﻣﺨﺎﻃﺒﺎﻥ ﺧﻮﺩ، ﺁﺷﻨﺎ ﺷﺪﻧﺪ. ﺍﻳﻨﺘﺮﻧﺖ، ﺩﺭ ﻣﺎﻫﻴﺖ ﺫﺍﺗﯽ ﺧﻮﺩ، ﺷﺒﮑﻪ‌ﺍﯼ ﺍﺯ ﮔﺮﻩ‌ﻫﺎﯼ ﺩﺭ ﻫﻢ ﺁﻣﻴﺨﺘﻪ ﺍﺳﺖ. ﮐﺎﺭ ﻣﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﺑﺎﺯﺍﺭﻳﺎﺏ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﮔﺮﻩ‌ﻫﺎ ﺑﺮﺍﯼ ﭘﻴﮕﻴﺮﯼ ﻣﮑﺎﻟﻤﺎﺕ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺍﻟﮕﻮﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ.

ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺭﺍ ﺩﺭ ﻳﮏ ﺗﻌﺮﻳﻒ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ «ﻣﮑﺎﻟﻤﻪ» ﺗﻌﺒﻴﺮ ﮐﺮﺩ. ﺍﻣﺮﻭﺯﻩ ﺍﻳﻨﺘﺮﻧﺖ ﻧﻴﺰ ﺑﻪ ﺍﺗﺎﻕ ﻣﮑﺎﻟﻤﻪ ﺗﺒﺪﻳﻞ ﺷﺪﻩ ﺍﺳﺖ. ﻣﺎﻫﻴﺖ ﺍﺭﺗﺒﺎﻃﯽ ﺍﻳﻨﺘﺮﻧﺖ ﺑﻪ ﻣﺎ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺍﻳﻦ ﻣﮑﺎﻟﻤﺎﺕ ﺭﺍ ﺩﻧﺒﺎﻝ ﻭ ﭘﻴﮕﻴﺮﯼ ﮐﺮﺩﻩ ﺗﺎ ﺑﺘﻮﺍﻧﻴﻢ ﻣﺸﺘﺮﯼ‌ﻫﺎﯼ ﻣﻮﺭﺩ ﻧﻈﺮ ﺑﺮﺍﯼ ﮐﺎﻻ ﻭ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺗﺠﺎﺭﯼ ﺧﻮﺩ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﻴﻢ. ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﻣﺎ ﺭﻭﺷﻬﺎﯼ ﮔﻔﺘﮕﻮ ﺑﺎ ﻣﺸﺘﺮﻳﺎﻥ ﻣﻮﺟﻮﺩ ﻭ ﺑﺎﻟﻘﻮﻩ ﺍﺯ ﻃﺮﻳﻖ ﺍﻳﻨﺘﺮﻧﺖ ﺭﺍ ﺧﻮﺍﻫﻴﻢ ﺁﻣﻮﺧﺖ.

 

ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ
ﺍﮔﺮ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺑﺎﻋﺚ ﺍﻳﺠﺎﺩ ﺗﻘﺎﺿﺎ ﻣﯽ‌ﺷﻮﺩ، ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻗﺪﺭﺕ ﺍﻳﻨﺘﺮﻧﺖ «ﺑﺴﺘﺮ ﺍﻳﺠﺎﺩ ﺗﻘﺎﺿﺎ» ﺭﺍ ﻓﺮﺍﻫﻢ ﻣﯽ‌ﮐﻨﺪ.

ﺍﻣﺮﻭﺯﻩ ﺩﺭ ﺩﻧﻴﺎﯼ ﺗﺠﺎﺭﺕ، ﻣﺴﺎﻟﻪ‌ﯼ «ﺩﻭﺟﺎﻧﺒﻪ ﺑﻮﺩﻥ ﻣﻌﺎﻣﻠﻪ» ﺑﺮﺍﯼ ﻫﻤﮕﺎﻥ ﺑﻪ ﻳﮏ ﺍﺻﻞ ﺗﺒﺪﻳﻞ ﺷﺪﻩ ﺍﺳﺖ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ، ﻫﺮ ﻧﻮﻉ ﻣﺒﺎﺩﻟﻪ ﻭ ﻣﻌﺎﻣﻠﻪ ﻣﺎﻧﻨﺪ ﺧﻴﺎﺑﺎﻧﯽ ﺩﻭﻃﺮﻓﻪ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﻫﻤﺰﻣﺎﻥ ﺑﺮﺍﯼ ﻫﺮ ﺩﻭ ﻃﺮﻑ ﺳﻮﺩ ﺁﻭﺭ ﺑﺎﺷﺪ.

ﺍﻳﻨﺘﺮﻧﺖ ﺩﻧﻴﺎﯼ ﺗﺠﺎﺭﺕ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﺍﺩﻩ  ﻭ ﺍﻟﮕﻮﯼ ﺟﺪﻳﺪﯼ ﺩﺭ ﺭﻭﺷﯽ ﮐﻪ ﻣﺼﺮﻑ ﮐﻨﻨﺪﮔﺎﻥ ﺑﺎ ﻣﺤﺼﻮﻻﺕ ﺗﺠﺎﺭﯼ ﻭ ﻳﺎ ﺑﺎ ﻳﮑﺪﻳﮕﺮ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌ﮐﻨﻨﺪ، ﻃﺮﺍﺣﯽ ﮐﺮﺩﻩ ﺍﺳﺖ.

ﺭﺳﺎﻧﻪ ﺁﻧﻼﻳﻦ ﺍﻧﺘﺨﺎﺏ، ﻧﻔﻮﺫ ﻭ ﻗﺪﺭﺕ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﻣﺼﺮﻑ ﮐﻨﻨﺪﮔﺎﻥ ﻓﺮﺍﻫﻢ ﮐﺮﺩﻩ ﺍﺳﺖ.

 

ﺍﺳﺘﺮﺍﺗﮋﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ
ﭘﻴﺶ ﺍﺯ ﻓﺮﺍﮔﻴﺮﯼ ﺍﻳﻨﺘﺮﻧﺖ، ﺗﺎﺟﺮﺍﻥ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﯼ ﺳﻨﺘﯽ ﺑﺮﺍﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﻣﺤﺼﻮﻻﺕ ﺧﻮﺩ ﺑﻬﺮﻩ ﻣﯽ‌ﺑﺮﺩﻧﺪ. ﺑﺎﻳﺪ ﮔﻔﺖ ﮐﻪ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﻧﻴﺰ ﺑﺮ ﭘﺎﻳﻪ ﻫﻤﺎﻥ ﺍﺻﻮﻝ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺳﻨﺘﯽ ﺳﺎﺧﺘﻪ ﺷﺪﻩ ﺑﺎ ﺍﻳﻦ ﺗﻔﺎﻭﺕ ﮐﻪ ﺍﺯ ﻓﻨﺎﻭﺭﯼ ﻭ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﻧﻴﺰ ﺑﻬﺮﻩ ﻣﯽ‌ﮔﻴﺮﺩ.

ﺩﺭ ﻫﻨﮕﺎﻡ ﺍﻳﺠﺎﺩ ﻳﮏ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﻣﻮﻓﻖ، ﺗﻔﮑﺮ ﮐﺎﺭﺑﺮ ﻣﺤﻮﺭ، ﮐﻪ ﮐﺎﺭﺑﺮ ﺭﺍ ﺩﺭ ﻣﺮﮐﺰ ﺗﻤﺎﻡ ﺗﺼﻤﻴﻤﺎﺕ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﺪ، ﺍﻣﺮﯼ ﺣﻴﺎﺗﯽ ﺍﺳﺖ.

ﻇﻬﻮﺭ ﻓﻨﺎﻭﺭﻳﻬﺎﯼ ﺟﺪﻳﺪ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﻃﺮﺍﺡ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺍﻣﺮﻭﺯ ﻧﻪ ﺗﻨﻬﺎ ﻣﺠﻤﻮﻋﻪ‌ﺍﯼ ﺍﺯ ﺍﻣﮑﺎﻧﺎﺕ ﺗﺎﮐﺘﻴﮑﯽ ﺟﺪﻳﺪ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺩﺍﺭﺩ، ﺑﻠﮑﻪ ﺭﻭﺷﻬﺎﯼ ﺑﯽ‌ﺳﺎﺑﻘﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺳﻨﺠﺶ ﺍﺛﺮﺑﺨﺸﯽ ﺍﺳﺘﺮﺍﺗﮋﯼ‌ﻫﺎ ﻭ ﺗﺎﮐﺘﻴﮑﻬﺎﯼ ﺍﻧﺘﺨﺎﺏ ﺷﺪﻩ ﺩﺍﺭﺩ. ﺍﻳﻦ ﺣﻘﻴﻘﺖ ﮐﻪ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺍﻣﺮﯼ ﺑﺴﻴﺎﺭﯼ ﺗﺠﺮﺑﯽ ﺍﺳﺖ ﻳﮑﯽ ﺍﺯ ﻧﻘﺎﻁ ﻗﻮﺕ ﺁﻥ ﻣﺤﺴﻮﺏ ﻣﯽ‌ﺷﻮﺩ.

ﺩﺭ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﻫﻤﻪ ﭼﻴﺰ ﻗﺎﺑﻞ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ ﺍﺳﺖ: ﺍﺯ ﺭﻓﺘﺎﺭ‌ﻫﺎ ﮔﺮﻓﺘﻪ ﺗﺎ ﺍﻗﺪﺍﻣﺎﺕ، ﺭﻭﺵ ﺍﻗﺪﺍﻡ ﻭ ﻧﺘﺎﻳﺞ. ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﻃﺮﺍﺡ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺑﺎﻳﺪ ﺑﺎ ﺗﻔﮑﺮ ﺭﻭﯼ ﺑﺎﺯﮔﺸﺖ ﺳﺮﻣﺎﻳﻪ ﮐﺎﺭﺵ ﺭﺍ ﺁﻏﺎﺯ ﮐﻨﺪ. ﻫﻤﺎﻫﻨﮕﯽ ﺑﺎ ﻳﮏ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺑﺎﻳﺪ ﭼﺎﺭﭼﻮﺏ ﺁﺯﻣﺎﻳﺸﯽ ﻭ ﺗﻮﺍﻧﺎﻳﯽ ﺍﻧﻌﻄﺎﻑ ﭘﺬﻳﺮﯼ ﻭ ﭘﻮﻳﺎﻳﯽ ﺩﺭ ﻣﺤﻴﻄﯽ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﮐﻪ ﻫﻤﺰﻣﺎﻥ ﺑﺎ ﺭﻓﺘﺎﺭ ﮐﺎﺭﺑﺮﺍﻥ ﺩﭼﺎﺭ ﺗﻐﻴﻴﺮ ﻭ ﺗﺤﻮﻝ ﻣﯽ‌ﺷﻮﺩ.

 

ﺗﺎﮐﺘﻴﮏ ﺩﺭ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ
ﺍﮔﺮ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺭﺍ ﻳﮏ «ﺑﺮﻧﺎﻣﻪ ﻋﻤﻠﯽ ﻃﺮﺍﺣﯽ ﺷﺪﻩ ﺑﻪ ﻣﻨﻈﻮﺭ ﺭﺳﻴﺪﻥ ﺑﻪ ﻧﺘﻴﺠﻪ‌ﺍﯼ ﺧﺎﺹ» ﺑﺪﺍﻧﻴﻢ، ﺁﻧﮕﺎﻩ ﻧﺘﻴﺠﻪ ﻣﻮﺭﺩ ﻧﻈﺮ ﻳﮏ ﺩﻳﺪﮔﺎﻩ ﺍﺳﺘﺮﺍﺗﮋﻳﮏ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺑﺎﻳﺪ ﻫﻤﺴﻮ ﺑﺎ ﺍﻫﺪﺍﻑ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺳﺎﺯﻣﺎﻥ ﺑﺎﺷﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ، ﺍﮔﺮ ﻳﮑﯽ ﺍﺯ ﺍﻫﺪﺍﻑ ﮐﻠﯽ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺟﺬﺏ ﻣﺸﺘﺮﻳﺎﻥ ﺟﺪﻳﺪ ﺑﺎﺷﺪ، ﺍﻫﺪﺍﻑ ﻣﺤﺘﻤﻞ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﺠﺎﺩ ﺁﮔﺎﻫﯽ ﺁﻧﻼﻳﻦ ﺩﺭ ﻣﻮﺭﺩ ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﺑﺎﺷﺪ.

ﺍﻧﻮﺍﻉ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﻳﻨﺘﺮﻧﺘﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ‌ﯼ «ﺩﺭﺳﻨﺎﻣﻪ» ﻣﻄﺎﻟﻌﻪ ﻣﯽ‌ﮐﻨﻴﻢ:

ﺍﻣﺮﻭﺯﻩ ﺭﻭﺵ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﻳﻨﺘﺮﻧﺘﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﻣﺎ ﺩﺭ ﻳﮏ ﺗﻌﺮﻳﻒ ﮐﻠﯽ، ﻣﯽ‌ﺗﻮﺍﻥ ﺍﻧﻮﺍﻉ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﻳﻨﺘﺮﻧﺘﯽ ﺭﺍ ﺑﻪ ﺑﺨﺶ‌ﻫﺎﯼ ﺯﻳﺮ ﺗﻘﺴﻴﻢ ﮐﺮﺩ:

  • ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﭘﺴﺖ ﺍﻟﮑﺘﺮﻭﻧﻴﮏ (ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﻳﻤﻴﻠﯽ)
  • ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﺗﺒﻠﻴﻐﺎﺕ ﻧﻤﺎﻳﺸﯽ
  • ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ
  • ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﺗﺒﻠﻴﻐﺎﺕ ﭘﺮﺩﺍﺧﺖ ﺑﻪ ﺍﺯﺍﯼ ﻫﺮ ﮐﻠﻴﮏ ﻳﺎ PPC
  • ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ
  • ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﻣﻮﺑﺎﻳﻞ

ﺑﯽ‌ﺗﺮﺩﻳﺪ ﻣﻮﺍﺭﺩ ﺑﻴﺸﺘﺮ ﺍﺯ ﺍﻳﻦ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺮ ﺷﻤﺮﺩ، ﺍﻣﺎ ﻣﺎ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ، ﺑﻪ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﺑﺴﻨﺪﻩ ﻣﯽ‌ﮐﻨﻴﻢ.

 

ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﭘﺴﺖ ﺍﻟﮑﺘﺮﻭﻧﻴﮏ (ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﻳﻤﻴﻠﯽ)
ﺷﮑﻠﯽ ﺍﺯ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﻣﺴﺘﻘﻴﻢ ﮐﻪ ﭘﻴﺎﻡ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ ﺭﺍ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﺑﺰﺍﺭﻫﺎﯼ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﺑﻪ ﻣﺨﺎﻃﺒﺎﻥ ﻣﯽ‌ﺭﺳﺎﻧﺪ. ﺍﻳﻦ ﻧﻮﻉ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺭﻭﺷﯽ ﺑﺴﻴﺎﺭ ﻣﻘﺮﻭﻥ ﺑﻪ ﺻﺮﻓﻪ، ﻫﺪﻓﻤﻨﺪ، ﻗﺎﺑﻞ ﺍﺳﺘﻔﺎﺩﻩ ﺩﺭ ﻣﻘﻴﺎﺱ ﺍﻧﺒﻮﻩ ﻭ ﺑﻪ ﻃﻮﺭ ﮐﺎﻣﻞ ﻗﺎﺑﻞ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ ﺍﺳﺖ ﮐﻪ ﺗﻤﺎﻡ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﺍﻳﻦ ﻧﻮﻉ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﻳﮑﯽ ﺍﺯ ﻗﺪﺭﺗﻤﻨﺪﺗﺮﻳﻦ ﺗﺎﮐﺘﻴﮏ‌ﻫﺎﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﻳﺠﻴﺘﺎﻝ ﺍﺳﺖ.

ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﭘﺴﺖ ﺍﻟﮑﺘﺮﻭﻧﻴﮏ ﺍﺑﺰﺍﺭﯼ ﺑﺮﺍﯼ ﺑﺮﻗﺮﺍﺭﯼ ﺍﺭﺗﺒﺎﻁ، ﻫﻢ ﺑﺎ ﻣﺸﺘﺮﻳﺎﻥ ﺑﺎﻟﻘﻮﻩ ﻭ ﻫﻢ ﻣﺸﺘﺮﻳﺎﻥ ﻓﻌﻠﯽ ﺍﺳﺖ. ﺍﻳﻦ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺑﺎﻳﺪ ﺍﻳﻦ ﻣﺸﺘﺮﻳﺎﻥ ﺭﺍ ﺣﻔﻆ ﮐﺮﺩﻩ ﻭ ﺍﺭﺯﺷﺸﺎﻥ ﺭﺍ ﺑﻪ ﺣﺪﺍﮐﺜﺮ ﺑﺮﺳﺎﻧﺪ ﺗﺎ ﺩﺭ ﻧﻬﺎﻳﺖ ﻣﻨﺠﺮ ﺑﻪ ﺳﻮﺩﺁﻭﺭﯼ ﺑﻴﺸﺘﺮ ﺑﺮﺍﯼ ﮐﻞ ﺳﺎﺯﻣﺎﻥ ﺷﻮﺩ. ﻳﮏ ﭘﺎﻳﮕﺎﻩ ﺩﺍﺩﻩ ﻫﺪﻓﻤﻨﺪ ﻭ ﺗﻘﺴﻴﻢ ﺑﻨﺪﯼ ﺷﺪﻩ ﭘﺴﺖ ﺍﻟﮑﺘﺮﻭﻧﻴﮏ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻣﺤﺼﻮﻝ ﺗﺠﺎﺭﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻣﻨﻈﻮﺭ ﺩﺳﺘﻴﺎﺑﯽ ﺑﻪ ﺑﻬﺘﺮﻳﻦ ﻧﺘﺎﻳﺞ، ﭘﻴﺎﻡ‌ﻫﺎﻳﯽ ﺭﺍ ﺑﻪ ﻋﺪﻩ ﺧﺎﺻﯽ ﺍﺯ ﻣﺸﺘﺮﻳﺎﻥ ﺑﺮﺳﺎﻧﺪ.

 

ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﺗﺒﻠﻴﻐﺎﺕ ﻧﻤﺎﻳﺸﯽ
ﺗﺒﻠﻴﻐﺎﺕ ﺁﻧﻼﻳﻦ ﺷﺎﻣﻞ ﺗﺒﻠﻴﻐﺎﺕ ﺩﺭ ﺳﺮﺍﺳﺮ ﺍﻳﻨﺘﺮﻧﺖ ﻣﯽ‌ﺷﻮﺩ: ﺻﻔﺤﺎﺕ ﻧﺘﺎﻳﺞ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ، ﺗﺒﻠﻴﻐﺎﺕ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺩﺭ ﺍﻳﻤﻴﻞ‌ﻫﺎ، ﺁﮔﻬﯽ‌ﻫﺎﯼ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻭ ﺩﺭ ﺷﺒﮑﻪ ﻧﻤﺎﻳﺶ ﮔﻮﮔﻞ.

ﻫﺪﻑ ﺍﺻﻠﯽ ﺗﺒﻠﻴﻐﺎﺕ ﺍﻓﺰﺍﻳﺶ ﻓﺮﻭﺵ ﺑﺎ ﺑﺎﻻ ﺑﺮﺩﻥ ﺁﮔﺎﻫﯽ ﺁﻧﻼﻳﻦ ﻧﺴﺒﺖ ﺑﻪ ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﺍﺳﺖ. ﺍﻳﻦ ﻧﻮﻉ ﺗﺒﻠﻴﻐﺎﺕ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﻌﺎﻣﻠﯽ‌ﺗﺮ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺑﺎ ﺗﺎﺛﻴﺮ ﻣﺨﺮﺏ ﮐﻤﺘﺮﯼ ﻧﺴﺒﺖ ﺑﻪ ﺗﺒﻠﻴﻐﺎﺕ ﺳﻨﺘﯽ ﻭ ﺗﺒﻠﻴﻐﺎﺕ ﺁﻧﻼﻳﻦ ﻏﻴﺮﺗﻌﺎﻣﻠﯽ ﺻﻮﺭﺕ ﺑﮕﻴﺮﺩ؛ ﺯﻳﺮﺍ ﮐﺎﺭﺑﺮﺍﻥ ﺩﺭ ﺻﻮﺭﺕ ﺩﻟﺨﻮﺍﻩ ﺑﺎ ﺁﮔﻬﯽ ﺗﻌﺎﻣﻞ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌ﮐﻨﻨﺪ. ﺗﺒﻠﻴﻐﺎﺕ ﺁﻧﻼﻳﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﻃﺮﻳﻖ ﻫﺪﻑ ﻗﺮﺍﺭ ﺩﺍﺩﻥ ﻣﻨﺎﻃﻘﯽ ﺧﺎﺹ، ﺑﺎﺯﺍﺭﻫﺎﯼ ﺧﺎﺹ ﻭ ﺍﻳﺠﺎﺩ ﺑﺎﻓﺖ ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺷﻮﺩ.

 

ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ
ﻫﺮ ﺭﻭﺯ ﻣﻴﻠﻴﻮﻥ‌ﻫﺎ ﻧﻔﺮ ﺩﺭ ﺳﺮﺍﺳﺮ ﺟﻬﺎﻥ ﺍﺯ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﺑﺮﺍﯼ ﻳﺎﻓﺘﻦ ﻣﺤﺘﻮﺍ ﺩﺭ ﺍﻳﻨﺘﺮﻧﺖ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ. ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﻳﯽ ﺑﺮ ﭘﺎﻳﻪ ﻭﺏ ﻫﺴﺘﻨﺪ ﮐﻪ ﻭﺏ ﺭﺍ ﻧﻤﺎﻳﻪ ﺳﺎﺯﯼ/ﺍﻳﻨﺪﮐﺲ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﻣﺮﺩﻡ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻨﺪ ﺗﺎ ﺁﻧﭽﻪ ﺑﻪ ﺩﻧﺒﺎﻟﺶ ﻫﺴﺘﻨﺪ ﺭﺍ ﭘﻴﺪﺍ ﮐﻨﻨﺪ. «ﺟﺴﺘﺠﻮ» ﻳﺎ «ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺑﺮ ﺍﺳﺎﺱ ﺟﺴﺘﺠﻮ» ﺍﻏﻠﺐ ﺑﻪ ﺻﻨﻌﺘﯽ ﺍﺷﺎﺭﻩ ﺩﺍﺭﺩ ﮐﻪ ﺩﺭ ﺍﻃﺮﺍﻑ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﺩ. ﺣﺘﻤﺎ ﺑﺎ ﺧﻮﺩ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺤﺼﻮﻝ ﻳﺎ ﻣﺤﺼﻮﻻﺕ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﺑﻪ ﻣﺸﺘﺮﻳﺎﻥ ﺑﻴﺸﺘﺮﯼ ﻧﺸﺎﻥ ﺩﻫﻴﺪ؟! ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ، ﺩﻭ ﺩﺭﺱ ﺭﺍ ﺑﻪ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﺍﺧﺘﺼﺎﺹ ﺩﺍﺩﻩ‌ﺍﻳﻢ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻳﻦ ﻣﻬﻢ، ﺑﭙﺮﺩﺍﺯﻳﺪ!

 

ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﺗﺒﻠﻴﻐﺎﺕ ﭘﺮﺩﺍﺧﺖ ﺑﻪ ﺍﺯﺍﯼ ﻫﺮ ﮐﻠﻴﮏ ﻳﺎ PPC
ﺗﺒﻠﻴﻐﺎﺕ ﭘﺮﺩﺍﺧﺖ ﺑﻪ ﺍﺯﺍﯼ ﻫﺮ ﮐﻠﻴﮏ ﻳﺎ PPC ﻳﮏ ﺳﻴﺴﺘﻢ ﺗﺒﻠﻴﻐﺎﺗﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺁﻥ ﺗﺒﻠﻴﻎ ﮐﻨﻨﺪﻩ ﺗﻨﻬﺎ ﺑﻪ ﺍﺯﺍﯼ ﻫﺮ ﮐﻠﻴﮏ ﺭﻭﯼ ﺁﮔﻬﯽ ﭘﻮﻝ ﭘﺮﺩﺍﺧﺖ ﻣﯽ‌ﮐﻨﺪ؛ ﺍﺯ ﺍﻳﻦ ﺭﻭ ﭘﺮﺩﺍﺧﺖ ﺑﻪ ﺍﺯﺍﯼ ﻫﺮ ﮐﻠﻴﮏ ﻧﺎﻡ ﮔﺮﻓﺘﻪ ﺍﺳﺖ.

ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺍﻳﻦ ﺍﺻﻄﻼﺡ ﺑﻴﺸﺘﺮ ﺑﻪ ﺳﻴﺴﺘﻢ‌ﻫﺎﯼ ﺗﺒﻠﻴﻐﺎﺗﯽ ﺍﺭﺍﺋﻪ ﺷﺪﻩ ﺗﻮﺳﻂ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ، ﻣﺎﻧﻨﺪ ﮔﻮﮔﻞ، ﺑﻴﻨﮓ ﻭ ﻳﺎﻫﻮ! ﺍﻃﻼﻕ ﻣﯽ‌ﺷﻮﺩ، ﺁﮔﻬﯽ‌ﻫﺎﯼ ﭘﺮﺩﺍﺧﺖ ﺑﻪ ﺍﺯﺍﯼ ﻫﺮ ﮐﻠﻴﮏ ﺩﺭ ﺳﺎﻳﺖ‌ﻫﺎﯼ ﻣﺤﺘﻮﺍ ﻭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻧﻴﺰ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﺩ.

ﻓﺮﻣﺖ‌ﻫﺎﯼ ﺭﺍﻳﺞ ﺍﻳﻦ ﻧﻮﻉ ﺗﺒﻠﻴﻐﺎﺕ، ﺍﺯ ﻣﺘﻨﯽ ﺳﺎﺩﻩ ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺑﻪ ﺑﻨﺮﻫﺎﯼ ﻓﻠﺶ ﻭ ﺣﺘﯽ ﺁﮔﻬﯽ‌ﻫﺎﯼ ﻭﻳﺪﺋﻮﻳﯽ ﻧﻴﺰ ﻣﯽ‌ﺭﺳﻨﺪ. ﺗﺒﻠﻴﻐﺎﺕ ﭘﺮﺩﺍﺧﺖ ﺑﻪ ﺍﺯﺍﯼ ﻫﺮ ﮐﻠﻴﮏ ﺑﺎﻋﺚ ﺍﻳﺠﺎﺩ ﺍﻧﻘﻼﺑﯽ ﺩﻭﺑﺎﺭﻩ ﺩﺭ ﺻﻨﻌﺖ ﺗﺒﻠﻴﻐﺎﺕ ﺁﻧﻼﻳﻦ ﺷﺪﻩ ﺍﺳﺖ ﻭ ﺍﻣﺮﻭﺯﻩ، ﺍﻳﻦ ﻧﻮﻉ ﺗﺒﻠﻴﻐﺎﺕ ۹۹ ﺩﺭﺻﺪ ﺍﺯ ﺩﺭﺁﻣﺪ ﮔﻮﮔﻞ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﻨﺪ.

ﺩﺭ ﺩﻭ ﺩﺭﺱ ﺍﺯ ﺍﻳﻦ ﺩﻭﺭﻩ، ﺷﻤﺎ ﺭﺍ ﺑﺎ ﺍﻳﻦ ﺭﻭﺵ ﺗﺒﻠﻴﻎ ﻭ ﭼﮕﻮﻧﮕﯽ ﺍﻧﺠﺎﻡ ﺁﻥ ﺁﺷﻨﺎ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﺩﻟﻴﻞ ﮔﺴﺘﺮﺩﮔﯽ ﺩﺭ ﺩﻧﻴﺎﯼ ﺍﻣﺮﻭﺯ ﺍﻳﻨﺘﺮﻧﺖ ﺍﺯ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﺑﺮﺧﻮﺭﺩﺍﻧﺪ. ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﺍﻳﻦ ﺷﺒﮑﻪ‌ﻫﺎ ﻣﺨﺎﻃﺒﻴﻦ ﺯﻳﺎﺩﯼ ﺩﺍﺭﻧﺪ، ﺑﺴﻴﺎﺭ ﭘﻮﻳﺎ ﻫﺴﺘﻨﺪ. ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﺷﺮﮐﺖ‌ﻫﺎ ﻭ ﻧﺎﻡ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ ﻓﺮﺻﺖ ﺑﻴﺸﺘﺮ ﻭ ﺑﻬﺘﺮﯼ ﺑﺮﺍﯼ ﺗﻌﺎﻣﻞ ﺑﺎ ﻣﺸﺘﺮﯼ ﺍﺯ ﻃﺮﻳﻖ ﺍﺭﺗﺒﺎﻃﺎﺕ ﻫﺪﻓﻤﻨﺪ ﻣﯽ‌ﺩﻫﻨﺪ. ﻳﮑﯽ ﺍﺯ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﺧﺼﻮﺻﻴﺖ‌ﻫﺎﯼ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ، «ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ» ﺍﺳﺖ. ﺍﮔﺮ ﮐﺎﺭﺑﺮﺍﻥ ﻣﺤﺘﻮﺍ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ ﺑﺎ ﺩﻭﺳﺘﺎﻥ ﺧﻮﺩ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺧﻮﺍﻫﻨﺪ ﮔﺬﺍﺷﺖ ﻭ ﺍﻳﻦ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻦ‌ﻫﺎﯼ ﭘﯽ ﺩﺭ ﭘﯽ ﻣﻴﺘﻮﺍﻧﺪ ﺑﺎﻋﺚ ﺷﻮﺩ ﮐﺎﺭﺑﺮﺍﻥ ﺯﻳﺎﺩﯼ ﺍﺯ ﻋﮑﺲ ﻳﺎ ﻭﻳﺪﺋﻮ ﺗﺒﻠﻴﻐﯽ ﺷﻤﺎ ﺑﺎﺯﺩﻳﺪ ﮐﻨﻨﺪ. ﺑﻪ ﺩﻟﻴﻞ ﺍﻫﻤﻴﺖ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺩﺭ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ، ﻣﺎ ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ، ﺩﻭ ﺩﺭﺱ ﺭﺍ ﺑﻪ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﺭ ﺍﻳﻦ ﺷﺒﮑﻪ‌ﻫﺎ ﺍﺧﺘﺼﺎﯼ ﺩﺍﺩﻩ‌ﺍﻳﻢ.

 

ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﻣﻮﺑﺎﻳﻞ
ﺗﻠﻔﻦ ﻫﻤﺮﺍﻩ ﻭﺳﻴﻠﻪ ﻣﻬﻤﯽ ﺩﺭ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺳﺖ. ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻤﺮﺍﻩ ﻫﻤﻴﺸﻪ ﺗﻮﺳﻂ ﻣﺮﺩﻡ ﺣﻤﻞ ﺷﺪﻩ ﻭ ﺭﻭﺷﻦ ﻫﺴﺘﻨﺪ ﮐﻪ ﺍﻳﻦ ﺩﻭ ﻭﻳﮋﮔﯽ، ﺗﺮﮐﻴﺒﯽ ﻗﺪﺭﺗﻤﻨﺪ ﺑﺮﺍﯼ ﻫﺮ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﮐﻨﻨﺪ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﺑﻪ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺗﻠﻔﻦ ﻫﻤﺮﺍﻩ ﺭﺍ ﺩﺭ ﻃﺮﺡ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﮔﻨﺠﺎﻧﺪ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.

ﮐﻠﻴﺪ ﺍﺻﻠﯽ ﻣﻮﻓﻘﻴﺖ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﺭ ﻣﻮﺑﺎﻳﻞ، ﺷﻨﺎﺳﺎﻳﯽ ﻣﺨﺎﻃﺒﺎﻥ ﺍﺳﺖ. ﺑﻪ ﮐﺎﺭﮔﻴﺮﯼ ﻣﻮﺑﺎﻳﻞ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﺍﺭﺗﺒﺎﻃﯽ ﺁﻥ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺑﻪ ﮐﺎﺭﮔﻴﺮﯼ ﻭ ﻳﺎ ﺑﻪ ﺍﻳﺠﺎﺩ ﺗﺮﺍﻓﻴﮏ ﺑﺮﺍﯼ ﻭﺑﺴﺎﻳﺖ‌ﻫﺎ ﻳﺎ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﮐﺎﺭﺑﺮﺩﯼ ﺷﻤﺎ ﺑﺮﺍﯼ ﻣﻮﺑﺎﻳﻞ ﮐﻤﮏ ﮐﻨﺪ. ﺩﺭ ﺍﻳﻦ ﺩﻭﺭﻩ ﻣﺎ ﺩﻭ ﺩﺭﺱ ﺭﺍ ﺑﻪ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺍﺯ ﻃﺮﻳﻖ ﻣﻮﺑﺎﻳﻞ ﺍﺧﺘﺼﺎﺹ ﺩﺍﺩﻩ‌ﺍﻳﻢ ﻭ ﺑﻪ ﺗﺎﮐﺘﻴﮏ‌ﻫﺎﯼ ﺟﺴﺘﺠﻮﯼ ﺑﻬﺘﺮ ﺩﺭ ﻣﻮﺑﺎﻳﻞ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.
