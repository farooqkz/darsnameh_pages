ﻋﺒﺎﺭﺍﺕ ﮐﻠﻴﺪﯼ
ﻋﺒﺎﺭﺍﺕ ﮐﻠﻴﺪﯼ ﭘﺎﻳﻪ ﻭ ﺍﺳﺎﺱ ﺟﺴﺘﺠﻮ ﻫﺴﺘﻨﺪ. ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺻﻔﺤﺎﺗﯽ ﺭﺍ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺍﻋﺘﻘﺎﺩ ﺑﻪ ﺑﻴﺸﺘﺮﻳﻦ ﻣﻴﺰﺍﻥ ﺍﺭﺗﺒﺎﻁ ﺁﻧﻬﺎ ﺑﺎ ﮐﻠﻤﺎﺕ ﻭﺭﻭﺩﯼ ﺩﺍﺭﺩ. ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﺩﺭﮎ ﮐﺎﻣﻠﯽ ﺍﺯ ﻣﻌﻨﺎ ﻭ ﺭﻭﺵ ﺍﺳﺘﻔﺎﺩﻩ ﻣﺎ ﺍﺯ ﺯﺑﺎﻥ ﺩﺍﺭﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺍﮔﺮ ﻳﮏ ﮐﺎﺭﺑﺮ ﻋﺒﺎﺭﺕ «ﺍﺟﺎﺭﻩ ﻣﺎﺷﻴﻦ»‌ ﺭﺍ ﺟﺴﺘﺠﻮ ﮐﻨﺪ، ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺑﻪ ﺩﻧﺒﺎﻝ ﺻﻔﺤﺎﺗﯽ ﻣﯽ‌ﮔﺮﺩﺩ ﮐﻪ ﺑﺎ «ﺍﺟﺎﺭﻩ ﻣﺎﺷﻴﻦ» ﻭ ﺍﺣﺘﻤﺎﻻ ﺑﺎ «ﮐﺮﺍﻳﻪ ﻣﺎﺷﻴﻦ»، «ﺍﺟﺎﺭﻩ ﺧﻮﺩﺭﻭ» ﻭ ﭼﻴﺰﻫﺎﻳﯽ ﺍﺯ ﺍﻳﻦ ﻗﺒﻴﻞ ﺩﺭ ﺍﺭﺗﺒﺎﻁ ﺑﺎﺷﻨﺪ. ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﻫﻤﭽﻨﻴﻦ ﺍﻃﻼﻋﺎﺗﯽ ﺩﺭ ﻣﻮﺭﺩ ﻏﻠﻂ‌ﻫﺎﯼ ﺍﻣﻼﻳﯽ ﺭﺍﻳﺞ، ﮐﻠﻤﺎﺕ ﻣﺘﺮﺍﺩﻑ ﻭ ﺟﺴﺘﺠﻮﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺩﺍﺭﻧﺪ ﻭ ﺑﻨﺎﺑﺮﺍﻳﻦ ﺳﻌﯽ ﻣﯽ‌ﮐﻨﻨﺪ ﺑﻬﺘﺮﻳﻦ ﻧﺘﺎﻳﺞ ﺭﺍ ﺑﻪ ﮐﺎﺭﺑﺮ ﻋﺮﺿﻪ ﮐﻨﻨﺪ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ، ﺿﺮﻭﺭﯼ ﺍﺳﺖ ﮐﻪ ﻭﺑﺴﺎﻳﺖ‌ﻫﺎ ﺍﺯ ﻣﺤﺘﻮﺍﻳﯽ ﺑﺎ ﮐﻠﻤﺎﺕ ﮐﻠﻴﺪﯼ ﺑﻬﺮﻩ ﺑﺒﺮﻧﺪ ﮐﻪ ﺍﺣﺘﻤﺎﻝ ﺩﺍﺭﺩ ﺗﻮﺳﻂ ﻣﺨﺎﻃﺒﻴﻨﺸﺎﻥ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﮔﻴﺮﺩ. ﻭﺑﺴﺎﻳﺖ‌ﻫﺎ ﺑﺎﻳﺪ ﺩﺭ ﺯﻣﺎﻧﯽ ﮐﻪ ﻣﺸﺘﺮﻳﺎﻥ ﺑﺎﻟﻘﻮﻩ‌ﺷﺎﻥ ﺩﺭ ﺟﺴﺘﺠﻮﯼ ﺁﻧﻬﺎ ﻫﺴﺘﻨﺪ، ﺑﺎﻻ ﺑﻴﺎﻳﻨﺪ. ﻣﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﺻﺎﺣﺐ ﻳﮏ ﻭﺑﺴﺎﻳﺖ ﻳﺎ ﻳﮏ ﺑﺎﺯﺍﺭﻳﺎﺏ ﺑﺮﺍﯼ ﻳﮏ ﻭﺑﺴﺎﻳﺖ ﺑﺎﻳﺪ ﻳﮏ ﻟﻴﺴﺖ ﺍﺯ ﺍﺻﻄﻼﺣﺎﺗﯽ ﺗﻬﻴﻪ ﮐﻨﻴﻢ ﮐﻪ ﻣﺸﺘﺮﻳﺎﻥ ﺑﺎﻟﻘﻮﻩ‌ﻣﺎﻥ ﺑﺮﺍﯼ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻣﺤﺼﻮﻻﺕ ﺍﺭﺍﺋﻪ ﺷﺪﻩ ﻣﺎ ﺍﺣﺘﻤﺎﻻ ﺍﺯ ﺁﻧﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ. ﺑﺨﺶ ﺑﺰﺭﮔﯽ ﺍﺯ ﺗﺤﻘﻴﻘﺎﺕ ﺩﺭ ﻣﻮﺭﺩ ﮐﻠﻤﻪ ﮐﻠﻴﺪﯼ ﺷﺎﻣﻞ ﺩﺭﮎ ﺭﻭﺍﻧﺸﻨﺎﺳﯽ ﺟﺴﺘﺠﻮ ﻣﯽ‌ﺷﻮﺩ. ﺯﻣﺎﻧﯽ ﮐﻪ ﻣﺎ ﻟﻴﺴﺘﯽ ﺍﺯ ﻋﺒﺎﺭﺍﺕ ﻳﺎ ﮐﻠﻤﺎﺕ ﮐﻠﻴﺪﯼ ﺗﻬﻴﻪ ﻣﯽ‌ﮐﻨﻴﻢ، ﻭﺍﺭﺩ ﻓﺮﺁﻳﻨﺪ ﺫﻫﻨﯽ ﮐﺎﺭﺑﺮﺍﻥ ﺩﺭ ﺣﺎﻝ ﺟﺴﺘﺠﻮ ﺷﺪﻩ ﻭ ﺗﺮﮐﻴﺐ ﺩﺭﺳﺘﯽ ﺍﺯ ﮐﻠﻤﺎﺕ ﮐﻠﻴﺪﯼ ﻫﺪﻓﻤﻨﺪ ﺭﺍ ﺩﺭ ﮐﻨﺎﺭ ﻳﮑﺪﻳﮕﺮ ﻗﺮﺍﺭ ﺩﺍﺩﻩ ﺍﻳﻢ.

ﺩﺭ ﻫﻨﮕﺎﻡ ﺍﻧﺘﺨﺎﺏ ﻳﮏ ﮐﻠﻤﻪ ﮐﻠﻴﺪﯼ، ﭼﻬﺎﺭ ﭼﻴﺰ ﺭﺍ ﺑﺎﻳﺪ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﻢ:

  • ﺣﺠﻢ ﺟﺴﺘﺠﻮ: ﭼﻨﺪ ﻧﻔﺮ ﺍﺯ ﮐﺎﺭﺑﺮﺍﻥ ﺑﺮﺍﯼ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﺁﻧﭽﻪ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺍﺯ ﺍﻳﻦ ﻋﺒﺎﺭﺕ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ؟ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ، ﻳﮏ ﺑﺮﺁﻭﺭﺩ ﺣﺠﻤﯽ ﻣﺎﻫﺎﻧﻪ، ۳۳۸ ﻣﻴﻠﻴﺎﺭﺩ ﺟﺴﺘﺠﻮ ﺑﺮﺍﯼ ﮐﻠﻤﻪ «Hotel» ‌ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ،‌ ﺍﻣﺎ ﺑﺮﺍﯼ ﮐﻠﻤﻪ ﮐﻠﻴﺪﯼ ﺍﯼ ﻣﺎﻧﻨﺪ «ﻫﺘﻞ ﭼﻬﺎﺭ ﺳﺘﺎﺭﻩ» ﺗﻨﻬﺎ ﺩﺭﺻﺪ ﮐﻤﯽ ﺍﺯ ﺍﻳﻦ ﻣﻘﺪﺍﺭ ﺍﺳﺖ.
  • ﺭﻗﺎﺑﺖ: ﭼﻨﺪ ﻭﺑﺴﺎﻳﺖ ﺩﻳﮕﺮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﻫﻤﻴﻦ ﻋﺒﺎﺭﺕ ﺭﺍ ﻣﻮﺭﺩ ﻫﺪﻑ ﻗﺮﺍﺭ ﺩﺍﺩﻩ‌ﺍﻧﺪ؟ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ، ﮔﻮﮔﻞ ﺑﻴﺶ ﺍﺯ ۲۶۳۰۰۰۰۰۰۰ ﻧﺘﻴﺠﻪ ﺑﺮﺍﯼ ﮐﻠﻤﻪ «Hotel» ﻳﺎﻓﺘﻪ ﺍﻣﺎ ﺑﺮﺍﯼ ﻋﺒﺎﺭﺕ «ﻫﺘﻞ ﭼﻬﺎﺭ ﺳﺘﺎﺭﻩ» ﺗﻨﻬﺎ ۳۳۷۰۰۰۰ ﻧﺘﻴﺠﻪ ﭘﻴﺪﺍ ﮐﺮﺩﻩ ﺍﺳﺖ.
  • ﮔﺮﺍﻳﺶ ﺑﻪ ﺗﺒﺎﺩﻝ: ﺍﺣﺘﻤﺎﻝ ﺍﻳﻨﮑﻪ ﺟﺴﺘﺠﻮﮔﺮ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻳﻦ ﮐﻠﻤﻪ ﮐﻠﻴﺪﯼ ﻭﺍﺭﺩ ﺳﺎﻳﺖ ﺷﻤﺎ ﺷﺪﻩ ﻭ ﺩﺭ ﺍﻧﺘﻬﺎ ﻣﻨﺠﺮ ﺑﻪ ﺧﺮﻳﺪ ﺍﺯ ﺳﺎﻳﺖ ﺷﻤﺎ ﻳﺎ ﺩﺍﻧﻠﻮﺩ ﻳﺎ ﻫﺮﭼﻴﺰ ﺩﻳﮕﺮﯼ ﮐﻪ ﻣﻮﺭﺩ ﻧﻈﺮﺗﺎﻥ ﺍﺳﺖ ﺷﻮﺩ ﭼﻘﺪﺭ ﺍﺳﺖ؟ ﺑﻪ ﺑﻴﺎﻥ ﺩﻳﮕﺮ ﺁﻳﺎ ﺳﺮﻣﺎﻳﻪ‌ﮔﺬﺍﺭﯼ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﮐﻠﻤﻪ ﮐﻠﻴﺪﯼ ﻣﻨﺠﺮ ﺑﻪ ﺁﻣﺪﻥ ﮐﺎﺭﺑﺮﺍﻧﯽ ﺑﻪ ﺳﺎﻳﺖ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺑﺮﺍﯼ ﻣﺎ ﺳﻮﺩﺁﻭﺭﯼ ﺩﺍﺭﺩ ﻳﺎ ﺧﻴﺮ؟
  • ﺍﺭﺯﺵ ﻫﺮ ﻣﺸﺘﺮﯼ ﺟﺬﺏ ﺷﺪﻩ: ﺍﺭﺯﺵ ﻣﺘﻮﺳﻂ ﻫﺮ ﻣﺸﺘﺮﯼ ﺑﺎﻟﻘﻮﻩ ﮐﻪ ﺍﺯ ﻃﺮﻳﻖ ﮐﻠﻤﻪ ﮐﻠﻴﺪﯼ ﺟﺬﺏ ﮐﺮﺩﻩ ﺍﻳﺪ، ﭼﻘﺪﺭ ﺍﺳﺖ؟ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﻫﺰﻳﻨﻪ‌ﺍﯼ ﮐﻪ ﺑﺮﺍﯼ ﻭﺍﺭﺩ ﺷﺪﻥ ﺑﻪ ﻧﺘﺎﻳﺞ ﺟﺴﺘﺠﻮ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺁﻳﺎ ﻣﻨﺠﺮ ﺑﻪ ﺳﻮﺩﺁﻭﺭﯼ ﺷﺪﻩ ﺍﺳﺖ ﻳﺎ ﺧﻴﺮ؟

 

ﭘﮋﻭﻫﺶ ﺩﺭ ﻣﻮﺭﺩ ﮐﻠﻴﺪ ﻭﺍﮊﻩ
ﺁﻳﺎ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﮐﻪ ﺑﺮﺍﯼ ﺷﺮﻭﻉ ﻓﻬﺮﺳﺖ ﮐﻠﻴﺪ ﻭﺍﮊﻩ‌ﻫﺎﻳﺘﺎﻥ ﺑﺎﻳﺪ ﺍﺯ ﮐﺠﺎ ﺷﺮﻭﻉ ﮐﻨﻴﺪ؟ ﺍﻳﻦ ﮐﺎﺭ ﻧﻴﺎﺯﻣﻨﺪ ﮐﻤﯽ ﺗﻔﮑﺮ ﻭ ﻣﻘﺪﺍﺭﯼ ﭘﮋﻭﻫﺶ ﺍﺯ ﻃﺮﻳﻖ ﺍﺑﺰﺍﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺳﺎﺩﮔﯽ ﺩﺭ ﺩﺳﺘﺮﺱ ﺷﻤﺎ ﻗﺮﺍﺭ ﺧﻮﺍﻫﻨﺪ ﮔﺮﻓﺖ ﺗﺎ ﺑﻪ ﺷﻤﺎ ﺩﺭ ﺗﻮﻟﻴﺪ ﻭ ﺑﻬﺒﻮﺩ ﻟﻴﺴﺖ ﮐﻠﻤﺎﺕ ﮐﻠﻴﺪﯼ‌ﺗﺎﻥ ﮐﻤﮏ ﮐﻨﻨﺪ.

ﺑﻌﺪ ﺍﺯ ﺍﻳﻨﮑﻪ ﺑﺎ ﺍﻳﻦ ﺭﻭﺵ‌ﻫﺎ ﺑﻪ ﻟﻴﺴﺘﯽ ﺍﺯ ﮐﻠﻤﺎﺕ ﻣﻮﺭﺩ ﻋﻼﻗﻪ‌ﺍﯼ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺩﺭ ﻫﻨﮕﺎﻡ ﺟﺴﺘﺠﻮﯼ ﺁﻧﻬﺎ ﺩﺭ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ، ﺳﺎﻳﺖ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺭﺗﺒﻪ ﺑﺎﻻ ﻗﺮﺍﺭ ﺩﻫﻴﺪ، ﺩﺳﺖ ﭘﻴﺪﺍ ﮐﺮﺩﻳﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﻣﺘﻦ ﺳﺎﻳﺖ ﺧﻮﺩ ﺭﺍ ﻃﻮﺭﯼ ﺑﻨﻮﻳﺴﻴﺪ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﮐﻠﻤﺎﺕ ﺩﺭ ﺳﺎﻳﺖ ﺷﻤﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺷﺪﻩ ﺑﺎﺷﺪ. ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﺍﺯ ﺍﺳﺘﻔﺎﺩﻩ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﺍﺯ ﮐﻠﻤﺎﺕ ﮐﻠﻴﺪﯼ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ.

ﺍﺯ ﺍﻳﻦ ﮐﻠﻤﺎﺕ ﺩﺭ ﺗﻴﺘﺮ ﺻﻔﺤﺎﺕ، ﺑﺮﭼﺴﺐ ﺗﻮﺻﻴﺤﺎﺕ ﻭﺑﺴﺎﻳﺖ ﺧﻮﺩ ﻭ ﺑﺮﭼﺴﺐ ﮐﻠﻤﺎﺕ ﮐﻠﻴﺪﯼ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ.

 

ﺑﺎﺭﺵ ﺫﻫﻨﯽ (ﻓﮑﺮ ﺑﮑﺮ ﻭ ﻧﺎﮔﻬﺎﻧﯽ)
ﺩﺭ ﻣﻮﺭﺩ ﺳﻮﺍﻻﺕ، ﻧﻴﺎﺯﻫﺎﯼ ﻣﺸﺘﺮﻳﺎﻥ ﻭ ﮐﻠﻤﺎﺗﯽ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﺑﺮﺍﯼ ﺗﻮﺻﻴﻒ ﮐﺴﺐ ﻭ ﮐﺎﺭﺗﺎﻥ ﺍﺯ ﺁﻧﻬﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻴﺪ. ﭼﮕﻮﻧﻪ ﮐﺴﯽ ﺩﺭ ﻣﻮﺭﺩ ﺁﻧﭽﻪ ﺷﻤﺎ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﮐﻨﻴﺪ، ﺍﺯ ﺷﻤﺎ ﺳﻮﺍﻟﯽ ﻣﯽ‌ﭘﺮﺳﺪ؟ ﮐﻠﻤﺎﺕ ﻣﺘﺮﺍﺩﻑ ﻭ ﺍﺷﺘﺒﺎﻫﺎﺕ ﺍﻣﻼﻳﯽ ﺭﺍ ﻧﻴﺰ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ.

 

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﺑﺰﺍﺭ ﭘﮋﻭﻫﺶ ﮐﻠﻴﺪﻭﺍﮊﻩ

۸- ﻧﺎﻡ ﺩﺍﻣﻨﻪ: ﺩﺭ ﺻﻮﺭﺕ ﺍﻣﮑﺎﻥ ﺍﺯ ﻋﺒﺎﺭﺕ ﮐﻠﻴﺪﯼ ﺩﺭ ﻧﺎﻡ ﺩﺍﻣﻨﻪ‌ﺗﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺭﺳﺎﻧﻪ، ﺗﺼﺎﻭﻳﺮ ﻭ ﻭﻳﺪﺋﻮ ﺑﺎﻳﺪ ﺗﻮﺳﻂ ﮐﻠﻴﺪﻭﺍﮊﻩ‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺷﻮﻧﺪ. ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﺗﺼﺎﻭﻳﺮ ﻳﺎ ﻭﻳﺪﺋﻮ‌ﻫﺎ ﺭﺍ ﻣﺸﺎﻫﺪﻩ ﮐﻨﻨﺪ، ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﻪ ﺭﺍﻫﯽ ﺗﮑﻴﻪ ﮐﻨﻴﺪ ﮐﻪ ﺭﺳﺎﻧﻪ ﻗﺎﺩﺭ ﺑﻪ ﻣﻌﺮﻓﯽ ﺧﻮﺩ ﺑﺎﺷﺪ. ﺧﻮﺍﻧﻨﺪﻩ‌ﻫﺎﯼ ﺍﻳﻨﺘﺮﻧﺘﯽ ﻧﻴﺰ ﺍﻳﻦ ﺗﻮﺿﻴﺤﺎﺕ ﺭﺍ ﻣﯽ‌ﺧﻮﺍﻧﻨﺪ ﮐﻪ ﺍﻳﻦ ﺍﻣﺮ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺧﻮﺍﻧﻨﺪﮔﺎﻥ ﺑﺼﺮﯼ ﮐﻤﮏ ﮐﻨﺪ ﺗﺎ ﻣﻔﻬﻮﻡ ﺳﺎﻳﺖ ﺭﺍ ﺩﺭﮎ ﮐﻨﻨﺪ. ﺩﺭ ﻧﻬﺎﻳﺖ، ﺭﺳﺎﻧﻪ‌ﻫﺎ ﻣﺎﻧﻨﺪ ﺗﺼﺎﻭﻳﺮ ﻭ ﻭﻳﺪﺋﻮ ﭼﻴﺰﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺩﺭ ﺻﻔﺤﻪ ﻧﺘﺎﻳﺞ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﻧﻤﺎﻳﺶ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺻﺤﻴﺢ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻳﮏ ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﻓﻀﺎﯼ ﺑﻬﺘﺮﯼ ﺩﺭ ﺻﻔﺤﻪ ﻧﺘﺎﻳﺞ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﮐﺴﺐ ﮐﻨﺪ ﻭ ﻫﻤﭽﻨﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﻧﺤﻮﯼ ﮐﺎﺭﺁﻣﺪ ﺩﺭ ﻫﺪﻓﻤﻨﺪ ﮐﺮﺩﻥ ﺍﺻﻄﻼﺣﺎﺕ ﺭﻗﺎﺑﺘﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺷﻮﺩ. ﺩﺭﺳﺖ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻳﮏ ﻋﮑﺲ ﻳﺎ ﻭﻳﺪﺋﻮ ﺑﺎﻋﺚ ﺗﺎﮐﻴﺪ ﺑﺮ ﻣﺤﺘﻮﺍﯼ ﻳﮏ ﺻﻔﺤﻪ ﻣﯽ‌ﺷﻮﻧﺪ، ﻫﻤﭽﻨﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﺩﺭ ﺻﻔﺤﺎﺕ ﺭﺗﺒﻪ ﺑﻨﺪﯼ ﻧﻴﺰ ﮐﻤﮏ ﮐﻨﻨﺪ ﺍﻟﺒﺘﻪ ﺑﻪ ﺷﺮﻃﯽ ﮐﻪ ﺑﺮﭼﺴﺐ ﺩﺭﺳﺘﯽ ﺑﺮﺍﻳﺸﺎﻥ ﺍﻳﺠﺎﺩ ﺷﻮﺩ. ﺩﺭ ﺍﻳﻨﺠﺎ ﭼﻨﺪ ﺭﺍﻩ ﺑﺮﺍﯼ ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺭﺳﺎﻧﻪ ﺑﻪ ﻫﻤﺮﺍﻩ ﻋﺒﺎﺭﺍﺕ ﮐﻠﻴﺪﯼ ﺑﺮﺍﯼ ﺳﺌﻮ ﻣﻌﺮﻓﯽ ﻣﯽ‌ﺷﻮﻧﺪ:

  • ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻧﺎﻡ‌ﻫﺎﯼ ﺗﻮﺻﻴﻔﯽ ﺑﺮﺍﯼ ﻓﺎﻳﻞ‌ﻫﺎ
  • ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺑﺮﭼﺴﺐ‌ﻫﺎﯼ ﻣﺸﺨﺺ alt ﻭ ﻭﻳﮋﮔﯽ ﻋﻨﻮﺍﻥ ﺑﺮﺍﯼ ﺗﺼﺎﻭﻳﺮ
  • ﺍﻃﻼﻋﺎﺕ ﻣﺘﺎ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﻓﺎﻳﻞ ﻋﮑﺲ ﻳﺎ ﻭﻳﺪﺋﻮ ﻗﺮﺍﺭ ﺩﺍﺩ. ﻣﻄﻤﺌﻦ ﺷﻮﻳﺪ ﮐﻪ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺑﻪ ﻣﺤﺘﻮﺍ ﻣﺮﺗﺒﻂ ﻫﺴﺘﻨﺪ.
  • ﺷﺮﺡ ﺗﻮﺻﻴﻔﯽ ﺑﻨﻮﻳﺴﻴﺪ ﻭ ﻣﻄﻤﺌﻦ ﺷﻮﻳﺪ ﮐﻪ ﻣﻄﻠﺐ ﻭ ﺭﺳﺎﻧﻪ ﺑﻪ ﻳﮑﺪﻳﮕﺮ ﻣﺮﺗﺒﻂ ﻫﺴﺘﻨﺪ.
  • ﺍﻃﻤﻴﻨﺎﻥ ﺣﺎﺻﻞ ﮐﻨﻴﺪ ﮐﻪ ﺑﺮﭼﺴﺐ‌ﻫﺎﯼ ﻋﻨﻮﺍﻥ ﻭ ﺗﺼﺎﻭﻳﺮ ﺑﻪ ﻳﮑﺪﻳﮕﺮ ﻣﺮﺗﺒﻂ ﻫﺴﺘﻨﺪ.
  • ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﺮﺍﯼ ﻓﺎﻳﻞ‌ﻫﺎﯼ ﻭﻳﺪﺋﻮﻳﯽ،‌ ﺍﺳﮑﺮﻳﭙﺖ ﺭﺍ ﺑﻪ ﻣﺘﻦ ﺗﺒﺪﻳﻞ ﮐﺮﺩﻩ ﻭ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﺳﺎﻳﺖ ﻳﻮﺗﻴﻮﺏ ﺑﺮﻧﺎﻣﻪ‌ﺍﯼ ﺗﺤﺖ ﻋﻨﻮﺍﻥ «ﺍﺗﻮ ﮐﭙﺸﻦ» ﻳﺎ ﺯﻳﺮﻧﻮﻳﺲ ﺧﻮﺩﮐﺎﺭ ﺩﺍﺭﺩ ﮐﻪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺁﺳﺎﻥ ﻣﯽ‌ﮐﻨﺪ.

 

ﻡﺣﺒﻮﺑﻴﺖ ﻟﻴﻨﮏ
ﻟﻴﻨﮏ‌ﻫﺎ ﻳﮏ ﺑﺨﺶ ﺣﻴﺎﺗﯽ ﺍﺯ ﻋﻤﻠﮑﺮﺩ ﺍﻳﻨﺘﺮﻧﺖ ﻫﺴﺘﻨﺪ. ﻫﺪﻑ ﻟﻴﻨﮏ، ﺍﺟﺎﺯﻩ ﺩﺍﺩﻥ ﺑﻪ ﮐﺎﺭﺑﺮ ﺑﺮﺍﯼ ﺭﻓﺘﻦ ﺍﺯ ﻳﮏ ﺻﻔﺤﻪ ﻭﺏ ﺑﻪ ﺻﻔﺤﻪ ﺍﯼ ﺩﻳﮕﺮ ﺍﺳﺖ. ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﮐﻪ ﺑﺮﺍﯼ ﺗﻘﻠﻴﺪ ﮐﺎﺭ ﺍﻧﺴﺎﻥ ﻧﻬﺎﻳﺖ ﺗﻼﺵ ﺧﻮﺩ ﺭﺍ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻨﺪ ﻧﻴﺰ ﻟﻴﻨﮏ‌ﻫﺎ ﺭﺍ ﺩﻧﺒﺎﻝ ﻣﯽ‌ﮐﻨﻨﺪ.

ﻟﻴﻨﮏ‌ﻫﺎ، ﻋﻼﻭﻩ ﺑﺮ ﺍﺟﺎﺯﻩ ﺩﺍﺩﻥ ﺑﻪ ﺍﺳﭙﺎﻳﺪﺭﻫﺎﯼ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺑﺮﺍﯼ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻭﺑﺴﺎﻳﺖ‌ﻫﺎ، ﺭﻭﺷﯽ ﺑﺮﺍﯼ ﺍﻋﺘﺒﺎﺭ ﺑﺨﺸﯽ ﺑﻪ ﺍﺭﺗﺒﺎﻁ ﻭ ﻧﺸﺎﻥ ﺩﻫﻨﺪﻩ ﺍﻫﻤﻴﺖ ﻫﺴﺘﻨﺪ. ﺯﻣﺎﻧﯽ ﮐﻪ ﻳﮏ ﺻﻔﺤﻪ ﺑﻪ ﺻﻔﺤﻪ ﺩﻳﮕﺮ ﻟﻴﻨﮏ ﻣﯽ‌ﺷﻮﺩ ﻣﺎﻧﻨﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺁﻥ ﺻﻔﺤﻪ ﺑﻪ ﻧﻔﻊ ﺻﻔﺤﻪ ﻣﻘﺼﺪ ﺭﺍﯼ ﺩﺍﺩﻩ ﻳﺎ ﺁﻥ ﺭﺍ ﺗﺎﻳﻴﺪ ﻣﯽ‌ﮐﻨﺪ. ﺑﻪ ﻃﻮﺭ ﻣﻌﻤﻮﻝ، ﺑﻪ ﺍﺯﺍﯼ ﻫﺮ ﺭﺍﯼ ﺑﻴﺸﺘﺮﯼ ﮐﻪ ﻳﮏ ﺻﻔﺤﻪ ﺩﺭﻳﺎﻓﺖ ﻣﯽ‌ﮐﻨﺪ، ﻗﺎﺑﻞ ﺍﻋﺘﻤﺎﺩﺗﺮ ﺷﺪﻩ ﻭ ﭘﺮ ﺍﻫﻤﻴﺖ‌ﺗﺮ ﻣﯽ‌ﻧﻤﺎﻳﺪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺭﺗﺒﻪ ﺑﻬﺘﺮﯼ ﺩﺭ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﮐﺴﺐ ﻣﯽ‌ﮐﻨﺪ.

 

ﺟﺎﻣﻌﻪ ﻭ ﺟﺴﺘﺠﻮ
ﺍﻃﻼﻋﺎﺕ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻧﻘﺶ ﻓﺰﺍﻳﻨﺪﻩ‌ﺍﯼ ﺩﺭ ﺟﺴﺘﺠﻮ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﻨﺪ. ﻣﺤﺘﻮﺍﯼ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ، ﻣﺎﻧﻨﺪ ﭘﻴﺎﻡ‌ﻫﺎﯼ ﺗﻮﻳﻴﺘﺮ ﻳﺎ ﻭﻳﺪﺋﻮﻫﺎﯼ ﻳﻮﺗﻴﻮﺏ، ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺩﺭ ﺻﻔﺤﻪ ﻧﺘﺎﻳﺞ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﻇﺎﻫﺮ ﺷﻮﻧﺪ ﻭ ﻧﺸﺎﻧﻪ ﻓﺰﺍﻳﻨﺪﻩ ﺍﺯ ﺗﺎﺛﻴﺮ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺮ ﺭﺗﺒﻪ ﺑﻨﺪﯼ ﺟﺴﺘﺠﻮ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺟﺴﺘﺠﻮﯼ ﭘﺮﺳﺮﻋﺖ ﮔﻮﮔﻞ ﺗﻘﺮﻳﺒﺎ ﻧﺘﺎﻳﺠﯽ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺩﻓﻌﺎﺕ ﺑﻴﺸﺘﺮﯼ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ ﺷﺪﻩ ﺍﻧﺪ.

ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺟﺎﻣﻌﻪ ﻭ ﺟﺴﺘﺠﻮ، ﻋﻮﺍﻣﻞ ﺍﺟﺘﻤﺎﻋﯽ ﻣﺘﻌﺪﺩﯼ ﺭﺍ ﺑﺎﻳﺪ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺖ:

۱- ﺍﺯ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺳﻠﻄﻪ ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﺩﺭ ﺻﻔﺤﻪ ﻧﺘﺎﻳﺞ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺯﻣﺎﻧﯽ ﮐﻪ ﺷﺨﺼﯽ ﺩﺭ ﻣﻮﺭﺩ ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﺷﻤﺎ ﺟﺴﺘﺠﻮﻳﯽ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺧﻮﺩ ﺑﺮﺍﯼ ﺗﺼﺎﺣﺐ ﻧﺘﺎﻳﺞ ﺑﻴﺸﺘﺮﯼ ﺩﺭ ﺁﻥ ﺻﻔﺤﻪ ﺑﻬﺮﻩ ﺑﺒﺮﻳﺪ.

۲- ﻟﻴﻨﮏ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﻋﻨﻮﺍﻥ ﻧﺸﺎﻧﻪ‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﻮﺩﻥ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺪﻳﻦ ﻣﻌﻨﯽ ﮐﻪ ﻫﺮﭼﻪ ﺻﻔﺤﻪ ﻭﺑﺴﺎﻳﺖ ﺷﻤﺎ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻣﺮﺗﺒﻂ ﺑﺎ ﻣﻮﺿﻮﻉ ﺧﺎﺹ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ ﺷﻮﺩ ﺩﺍﺭﺍﯼ ﺍﻣﺘﻴﺎﺯ ﺑﻴﺸﺘﺮﯼ ﺍﺳﺖ.

۳- ﻧﺘﺎﻳﺞ ﺷﺨﺼﯽ ﺷﺪﻩ ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﺷﺒﮑﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺁﻧﻼﻳﻦ ﺷﻤﺎ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﺍﮔﺮ ﺷﻤﺎ ﺩﺭ ﺣﺎﻝ ﺟﺴﺘﺠﻮ، ﻭﺍﺭﺩ ﻳﮏ ﺷﺒﮑﻪ ﺍﺟﺘﻤﺎﻋﯽ ﻧﻴﺰ ﺷﺪﻩ ﺑﺎﺷﻴﺪ (ﻓﻴﺲ‌ﺑﻮﮎ ﺑﺮﺍﯼ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮﯼ ﺑﻴﻨﮓ، ﻳﺎ ﻭﺭﻭﺩ ﺑﻪ ﮔﻮﮔﻞ ﺑﺎ ﺣﺴﺎﺏ ﮐﺎﺭﺑﺮﯼ ﺟﯽ‌ﻣﻴﻞ)، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻧﺘﺎﻳﺠﯽ ﺍﺯ ﻳﺎ ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﺩﺍﻳﺮﻩ ﺍﺟﺘﻤﺎﻋﯽ ﺧﻮﺩ ﺭﺍ ﺑﺒﻴﻨﻴﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ، ﻧﺘﺎﻳﺞ ﺩﺭ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮﯼ ﺑﻴﻨﮓ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﺎﻣﻞ ﻧﺸﺎﻧﻪ‌ﻫﺎﻳﯽ ﺍﺯ ﺁﻧﭽﻪ ﺩﻭﺳﺘﺎﻧﺘﺎﻥ ﻗﺒﻼ ﺩﺭ ﻓﻴﺲ‌ﺑﻮﮎ ﭘﺴﻨﺪﻳﺪﻩ ﻳﺎ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ‌ﺍﻧﺪ،‌ ﺑﺎﺷﺪ. ﺩﺭ ﮔﻮﮔﻞ ﺍﺣﺘﻤﺎﻝ ﺑﻴﺸﺘﺮﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﻭﺑﻼﮒ ﺩﻭﺳﺘﺎﻧﺘﺎﻥ ﺭﺍ ﺑﺮﺍﯼ ﻧﺘﺎﻳﺞ ﻣﺮﺗﺒﻂ ﺑﺒﻴﻨﻴﺪ.

۴- ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺑﺮﺍﯼ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ. ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﮔﻮﮔﻞ ﺑﺰﺭﮔﺘﺮﻳﻦ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺩﺭ ﺳﺮﺍﺳﺮ ﺟﻬﺎﻥ ﺍﺳﺖ، ﻳﻮﺗﻴﻮﺏ ﭘﺲ ﺍﺯ ﺁﻥ ﻗﺮﺍﺭ ﺩﺍﺭﺩ. ﮐﺎﺭﺑﺮﺍﻥ، ﺣﺘﯽ ﺑﺎ ﻭﺟﻮﺩ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ، ﻫﻨﻮﺯ ﺍﺯ ﺟﺴﺘﺠﻮ ﺑﺮﺍﯼ ﻳﺎﻓﺘﻦ ﻣﻄﺎﻟﺐ ﻣﻮﺭﺩ ﻧﻈﺮ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ. ﻣﺤﺘﻮﺍﻳﯽ ﮐﻪ ﺑﺮ ﭘﺎﻳﻪ ﺍﻳﻦ ﻭﻳﮋﮔﯽ‌ﻫﺎ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻣﺮﺗﺒﻂ ﻧﻴﺰ ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺷﻮﺩ.

ﺟﺴﺘﺠﻮﯼ ﻣﺤﻠﯽ
ﺟﺴﺘﺠﻮﯼ ﻣﺤﻠﯽ ﺑﻪ ﺟﺴﺘﺠﻮﯼ ﺭﻓﺘﺎﺭ ﻳﺎ ﻧﺘﺎﻳﺞ ﻣﺮﺑﻮﻁ ﺑﻪ ﻣﮑﺎﻥ ﺍﺷﺎﺭﻩ ﺩﺍﺭﺩ. ﻧﺘﺎﻳﺞ ﺑﻪ ﺩﺳﺖ ﺁﻣﺪﻩ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻣﺎﻫﻴﺘﯽ ﻣﮑﺎﻧﯽ ﺩﺍﺷﺘﻪ ﻳﺎ ﺑﺮ ﭘﺎﻳﻪ ﻧﻘﺸﻪ ﺑﺎﺷﻨﺪ.

ﻧﺘﺎﻳﺞ ﺑﺮ ﭘﺎﻳﻪ ﻧﻘﺸﻪ ﺩﺭ ﺻﻔﺤﺎﺕ ﻧﺘﺎﻳﺞ ﺗﺮﮐﻴﺒﯽ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻫﻤﺮﺍﻩ ﺑﺎ ﺩﻳﮕﺮ ﺍﻧﻮﺍﻉ ﻧﺘﺎﻳﺞ ﺣﺎﺻﻞ ﺷﻮﻧﺪ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﻪ ﻧﻮﻉ ﺟﺴﺘﺠﻮ ﺑﺴﺘﮕﯽ ﺩﺍﺭﺩ. ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﺭﻓﺘﻪ ﺭﻓﺘﻪ ﮐﺎﻣﻞ ﻭ ﭘﻴﭽﻴﺪﻩ‌ﺗﺮ ﻣﯽ‌ﺷﻮﻧﺪ، ﻣﯽ‌ﺗﻮﺍﻥ ﻣﮑﺎﻥ ﺭﺍ ﺣﺪﺱ ﺯﺩ ﻭ ﺑﺎﺯ ﻫﻢ ﺍﻧﻮﺍﻉ ﻧﺘﺎﻳﺞ ﺭﺍ ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﻗﺮﺍﺭ ﺩﺍﺩ.

ﺑﻪ ﻣﻨﻈﻮﺭ ﺁﻥ ﮐﻪ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﻧﺘﺎﻳﺠﯽ ﺑﺮ ﺍﺳﺎﺱ ﻣﮑﺎﻥ ﺍﺭﺍﺋﻪ ﺩﻫﻨﺪ، ﺑﺎﻳﺪ ﻣﮑﺎﻥ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﻣﻮﺭﺩ ﺳﻮﺍﻝ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻪ ﺷﺪﻩ‌ﺍﻧﺪ ﺭﺍ ﺑﺪﺍﻧﻨﺪ! ﺍﻳﻦ ﮐﺎﺭ ﻣﻌﻤﻮﻻ ﺗﻮﺳﻂ ﺳﺎﻳﺖ‌ﻫﺎﻳﯽ ﻣﺸﺨﺺ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺷﺎﻣﻞ ﺍﺳﻢ ﻭ ﺁﺩﺭﺱ ﮐﺴﺐ ﻭ ﮐﺎﺭﻫﺎ ﻫﺴﺘﻨﺪ. ﺗﻮﺟﻪ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﻳﻦ ﺳﺎﻳﺖ، ﺳﺎﻳﺖ ﺷﻤﺎ ﻧﺒﺎﺷﺪ. ﻧﺘﺎﻳﺞ ﻣﮑﺎﻧﯽ ﺍﻏﻠﺐ ﺍﺯ ﻃﺮﻳﻖ ﺳﺎﻳﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺑﺮﺭﺳﯽ ﮐﻨﻨﺪﻩ ﻣﺸﺨﺺ ﺷﺪﻩ ﻭ ﻧﺘﺎﻳﺞ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﺎﻣﻞ ﺑﺮﺧﯽ ﺍﺯ ﺁﻥ ﺑﺮﺭﺳﯽ‌ﻫﺎ ﺷﻮﺩ.

ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ ﻫﻤﭽﻨﻴﻦ ﺑﻪ ﺗﺠﺎﺭﺕ‌ﻫﺎ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﺪ ﺗﺎ ﻣﮑﺎﻥ ﺧﻮﺩ ﺭﺍ ﺍﻋﻼﻡ ﮐﻨﻨﺪ. ﻳﮏ ﺗﺠﺎﺭﺕ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﻃﺮﻳﻖ ﺍﻧﺠﺎﻡ ﻳﮏ ﻓﺮﺁﻳﻨﺪ ﺩﺭ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺧﻮﺩ ﺭﺍ ﺛﺎﺑﺖ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺍﻃﻤﻴﻨﺎﻥ ﺑﺪﻫﺪ ﮐﻪ ﺍﻃﻼﻋﺎﺕ ﻣﮑﺎﻧﯽ ﺻﺤﻴﺤﯽ ﺭﺍ ﺍﺭﺍﺋﻪ ﮐﺮﺩﻩ ﺍﺳﺖ.

 

ﭼﻪ ﮐﺎﺭﯼ ﺍﻧﺠﺎﻡ ﻧﺪﻫﻴﻢ؟
ﺳﺌﻮ ﻏﻴﺮﻣﺠﺎﺯ ﺑﻪ ﺷﻴﻮﻩ‌ﻫﺎﻳﯽ ﺍﺷﺎﺭﻩ ﺩﺍﺭﺩ ﮐﻪ ﺗﻼﺵ ﻣﯽ‌ﮐﻨﻨﺪ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺭﺍ ﺑﻪ ﺑﺎﺯﯼ ﺑﮕﻴﺮﻧﺪ. ﺍﮔﺮ ﻳﮏ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﻭﺑﺴﺎﻳﺘﯽ ﺭﺍ ﮐﺸﻒ ﮐﻨﺪ ﮐﻪ ﺍﺯ ﺷﻴﻮﻩ‌ﻫﺎﯼ ﻏﻴﺮ ﺍﺧﻼﻗﯽ ﺑﺮﺍﯼ ﺑﺎﻻ ﺑﺮﺩﻥ ﺭﺗﺒﻪ‌ﺍﺵ ﺩﺭ ﺁﻥ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩﻩ، ﺍﺣﺘﻤﺎﻝ ﺩﺍﺭﺩ ﮐﻪ ﺁﻥ ﻭﺑﺴﺎﻳﺖ ﺭﺍ ﺍﺯ ﻧﻤﺎﻳﻪ/ﺍﻳﻨﺪﮐﺲ ﺧﻮﺩ ﺣﺬﻑ ﮐﻨﺪ. ﮔﻮﮔﻞ ﺩﺳﺘﻮﺭﺍﻟﻌﻤﻠﯽ ﺑﺮﺍﯼ ﻃﺮﺍﺣﺎﻥ ﺳﺎﻳﺖ‌ﻫﺎﯼ ﺍﻳﻨﺘﺮﻧﺘﯽ ﻓﺮﺍﻫﻢ ﮐﺮﺩﻩ ﮐﻪ ﺍﺯ ﻃﺮﻳﻖ «ﮔﻮﮔﻞ ﻭﺏ ﻣﺴﺘﺮ ﻣﺮﮐﺰﯼ» ﻗﺎﺑﻞ ﺩﺳﺘﺮﺳﯽ ﺍﺳﺖ. ﮔﻮﮔﻞ ﻋﻼﻭﻩ ﺑﺮ ﻣﺸﺨﺺ ﮐﺮﺩﻥ ﺑﻬﺘﺮﻳﻦ ﺍﺻﻮﻝ، ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﻣﻨﻊ ﮐﻨﻨﺪﻩ ﺭﺍ ﻧﻴﺰ ﻋﻨﻮﺍﻥ ﮐﺮﺩﻩ ﺍﺳﺖ:

  • ﺍﺯ ﺍﻳﺠﺎﺩ ﻣﺘﻮﻥ ﻳﺎ ﻟﻴﻨﮏ‌ﻫﺎﯼ ﭘﻨﻬﺎﻥ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ.
  • ﺍﺯ ﺍﻳﺠﺎﺩ ﻟﻴﻨﮏ‌ﻫﺎﯼ ﺗﻐﻴﻴﺮ ﻣﺴﻴﺮ ﭘﻨﻬﺎﻧﯽ ﺑﭙﺮﻫﻴﺰﻳﺪ.
  • ﭘﺮﺳﺶ‌ﻫﺎﯼ ﺟﺴﺘﺠﻮﯼ ﺧﻮﺩﮐﺎﺭ ﺭﺍ ﺑﺮﺍﯼ ﮔﻮﮔﻞ ﺍﺭﺳﺎﻝ ﻧﮑﻨﻴﺪ.
  • ﺻﻔﺤﺎﺗﯽ ﺑﺎ ﮐﻠﻤﺎﺕ ﮐﻠﻴﺪﯼ ﺑﯽ ﺭﺑﻂ ﺑﺎﺭﮔﺰﺍﺭﯼ ﻧﮑﻨﻴﺪ.
  • ﺻﻔﺤﺎﺕ ﺩﻭﮔﺎﻧﻪ، ﺯﻳﺮﺩﺍﻣﻨﻪ ﻳﺎ ﺩﺍﻣﻨﻪ‌ﻫﺎﻳﯽ ﺑﺎ ﻣﺤﺘﻮﺍﯼ ﺗﮑﺮﺍﺭﯼ ﺍﻳﺠﺎﺩ ﻧﮑﻨﻴﺪ.
