ﭘﺸﺘﻴﺒﺎﻧﯽ ﻭ ﺧﺪﻣﺎﺕ ﻣﺸﺘﺮﻳﺎﻥ
ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺩﺭ ﺣﺎﻝ ﺗﺒﺪﻳﻞ ﺷﺪﻥ ﺑﻪ ﻳﮏ ﮐﺎﻧﺎﻝ ﺧﺪﻣﺎﺕ ﻣﺸﺘﺮﻳﺎﻥ ﺩﻳﮕﺮ ﺍﺳﺖ. ﻫﻤﺎﻧﮕﻮﻧﻪ ﮐﻪ ﻣﺸﺘﺮﻳﺎﻥ ﺑﻪ ﺻﻮﺭﺕ ﻓﺰﺍﻳﻨﺪﻩ‌ﺍﯼ ﺩﺭ ﺗﻌﺎﻣﻼﺕ ﺁﻧﻼﻳﻦ ﺍﺣﺴﺎﺱ ﺭﺍﺣﺘﯽ ﻣﯽ‌ﮐﻨﻨﺪ، ﺍﻧﺘﻈﺎﺭ ﻣﯽ‌ﺭﻭﺩ ﮐﻪ ﻧﺎﻡ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ ﻧﻴﺰ ﭘﺎﺳﺨﮕﻮﯼ ﻣﺸﺘﺮﻳﺎﻥ ﺧﻮﺩ ﺩﺭ ﻓﻀﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺎﺷﻨﺪ.

ﻧﮑﺘﻪ ﺟﺎﻟﺐ ﺗﻮﺟﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺧﺪﻣﺎﺕ ﻣﺸﺘﺮﻳﺎﻥ ﺩﺭ ﮐﺎﻧﺎﻝ‌ﻫﺎﯼ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺭﻓﺘﻪ ﺭﻓﺘﻪ ﺑﻪ ﺷﮑﻞ ﻫﻤﮑﺎﺭﯼ ﺩﺭ ﻣﯽ‌ﺁﻳﻨﺪ؛‌ ﺑﺪﻳﻨﮕﻮﻧﻪ ﮐﻪ ﺑﺮﺧﯽ ﺍﺯ ﻣﺸﺘﺮﻳﺎﻥ ﭘﺎﺳﺦ ﺩﻳﮕﺮ ﻣﺸﺘﺮﻳﺎﻥ ﺭﺍ ﻣﯽ‌ﺩﻫﻨﺪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺍﺗﮑﺎ ﺑﻪ ﺳﺎﺯﻣﺎﻥ ﺑﺮﺍﯼ ﭘﺸﺘﻴﺒﺎﻧﯽ ﮐﺎﻫﺶ ﻣﯽ‌ﻳﺎﺑﺪ.

 

ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ (ﺳﺌﻮ)
ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﻧﻘﺶ ﻣﻬﻤﯽ ﺩﺭ ﺳﺌﻮ ﺍﻳﻔﺎ ﻣﯽ‌ﮐﻨﺪ. ﺣﻀﻮﺭ ﺩﺭ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺎﻋﺚ ﺍﻳﺠﺎﺩ ﺻﻔﺤﺎﺕ ﺩﻳﮕﺮﯼ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻗﺎﺑﻞ ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﺑﻮﺩﻩ ﻭ ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﺭﺍ ﻗﺎﺩﺭ ﺑﻪ ﺗﺼﺎﺣﺐ ﺻﻔﺤﻪ ﻧﺘﺎﻳﺞ ﺟﺴﺘﺠﻮ ﻣﯽ‌ﮐﻨﺪ. ﻳﮏ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺳﺌﻮﯼ ﻫﻮﺷﻤﻨﺪﺍﻧﻪ ﺍﺯ ﺻﻔﺤﺎﺕ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ،‌ ﻟﻴﻨﮏ‌ﻫﺎ ﻭ ﻣﻮﺍﺭﺩ ﭘﺴﻨﺪﻳﺪﻩ ﺷﺪﻩ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﻣﻮﻗﻌﻴﺖ ﺩﻳﮕﺮ ﺻﻔﺤﺎﺕ ﻭﺏ ﺁﻥ ﺑﺮﻧﺪ ﺩﺭ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮ ﺭﺍ ﺗﻘﻮﻳﺖ ﮐﻨﺪ.

 

ﺍﺭﺗﺒﺎﻃﺎﺕ ﻭ ﺗﻮﺳﻌﻪ
ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ، ﺑﺮﺧﻼﻑ ﺩﻳﮕﺮ ﺭﺳﺎﻧﻪ‌ﻫﺎ، ﮐﺎﻧﺎﻟﯽ ﻣﻮﺛﺮ ﻭ ﺩﻭ ﻃﺮﻓﻪ ﺑﺮﺍﯼ ﺑﺮﻗﺮﺍﺭﯼ ﺍﺭﺗﺒﺎﻁ ﻭ ﮐﺎﻧﺎﻟﯽ ﺑﯽ ﺩﺭﻧﮓ ﺩﺭ ﭘﺨﺶ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﻧﺎﻡ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﻨﺪ. ﺍﻳﻦ ﺍﺭﺗﺒﺎﻁ ﺩﻭ ﻃﺮﻓﻪ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﺟﻮﺍﻣﻊ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﻫﻴﺠﺎﻥ ﺍﻧﮕﻴﺰ (ﻭ ﭼﺎﻟﺶ ﺑﺮ ﺍﻧﮕﻴﺰ) ﮐﺮﺩﻩ ﺍﺳﺖ. ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻣﺼﺮﻑ ﮐﻨﻨﺪﮔﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺎ ﻳﮑﺪﻳﮕﺮ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺷﺮﮐﺖ‌ﻫﺎ ﻭ ﻧﺎﻡ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ ﭘﻴﺎﻡ ﺍﺭﺳﺎﻝ ﮐﻨﻨﺪ، ﺷﺮﮐﺖ‌ﻫﺎ ﻭ ﻧﺎﻡ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺯ ﺍﻳﻦ ﺭﺳﺎﻧﻪ ﺑﺮﺍﯼ ﺍﺭﺗﺒﺎﻁ ﻭ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﻋﻤﻮﻡ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻨﻨﺪ. ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﻃﻮﺭ ﻓﺰﺍﻳﻨﺪﻩ‌ﺍﯼ ﺩﺭ ﺣﺎﻝ ﺗﺒﺪﻳﻞ ﺷﺪﻥ ﺑﻪ ﻳﮏ ﺍﺑﺰﺍﺭ ﺑﺴﻴﺎﺭ ﻣﻮﺛﺮ ﺩﺭ ﺍﺭﺗﺒﺎﻃﺎﺕ ﺍﺟﺘﻤﺎﻋﯽ ﻫﺴﺘﻨﺪ.

 

ﺍﻳﺠﺎﺩ ﻓﺮﻭﺵ ﻭ ﻋﻼﻗﻪ ﻣﺸﺘﺮﯼ ﺑﻪ ﺧﺮﻳﺪ
ﺍﺿﺎﻓﻪ ﮐﺮﺩﻥ ﻳﮏ ﻻﻳﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺮ ﺗﺠﺮﺑﻪ ﺗﺠﺎﺭﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎﻋﺚ ﺍﻳﺠﺎﺩ ﺗﺠﺮﺑﻪ ﺍﯼ ﻏﻨﯽ‌ﺗﺮ ﺑﺮﺍﯼ ﻣﺼﺮﻑ ﮐﻨﻨﺪﮔﺎﻥ ﺁﻧﻼﻳﻦ ﺷﻮﺩ. ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺁﺷﮑﺎﺭﺍ ﺩﺭ ﺭﻭﺍﺑﻂ ﺍﺟﺘﻤﺎﻋﯽ ﻳﺎ ﺩﺭ ﺍﺭﺗﺒﺎﻃﺎﺕ ﺑﻪ ﺩﺳﺖ ﺁﻣﺪﻩ ﺑﺮ ﭘﺎﻳﻪ ﺭﻓﺘﺎﺭ ﺍﻧﺠﺎﻡ ﺩﺍﺩ.

ﻫﻤﭽﻨﻴﻦ ﺟﻮﺍﻣﻊ ﺍﺟﺘﻤﺎﻋﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺎﻋﺚ ﺍﻳﺠﺎﺩ ﻋﻼﻗﻪ ﻣﺸﺘﺮﯼ ﺑﻪ ﺧﺮﻳﺪ ﻳﺎ ﺍﻳﺠﺎﺩ ﻓﺮﻭﺵ ﺷﻮﻧﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﮐﺎﺭﺑﺮﺩﯼ ﺩﺭ ﺻﻔﺤﻪ ﻓﻴﺲ‌ﺑﻮﮎ ﻧﺎﻡ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻨﺪ ﺗﺎ ﺗﺠﺎﺭﺕ ﺍﻟﮑﺘﺮﻭﻧﻴﮏ ﻳﺎ ﺍﻳﺠﺎﺩ ﻋﻼﻗﻪ ﺩﺭ ﻣﺸﺘﺮﯼ ﺩﺭ ﻓﻀﺎﯼ ﺍﻳﻦ ﺭﺳﺎﻧﻪ ﺻﻮﺭﺕ ﮔﻴﺮﺩ.

 

ﺁﻣﺎﺭ ﻭ ﭘﮋﻭﻫﺶ
ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺑﺰﺍﺭﯼ ﺑﺴﻴﺎﺭ ﻗﺪﺭﺗﻤﻨﺪ ﺩﺭ ﺁﻣﺎﺭ ﻭ ﭘﮋﻭﻫﺶ ﺑﺎﺷﻨﺪ ﺍﻣﺎ ﺍﻃﻼﻋﺎﺕ ﺑﺎﻳﺪ ﺩﺭ ﭼﺎﺭﭼﻮﺏ ﻣﻨﺎﺳﺐ ﺧﻮﺩ ﻣﻮﺭﺩ ﻗﻀﺎﻭﺕ ﻗﺮﺍﺭ ﮔﻴﺮﻧﺪ. ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺩﺭ ﺯﻣﺎﻥ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻣﻨﺒﻌﯽ ﻏﻨﯽ ﺍﺯ ﺍﻃﻼﻋﺎﺕ، ﻫﻢ ﺟﻤﻌﻴﺖ ﺷﻨﺎﺧﺘﯽ ﻭ ﻫﻢ ﺑﺮ ﺍﺳﺎﺱ ﺍﻭﻟﻮﻳﺖ، ﻓﺮﺍﻫﻢ ﺁﻭﺭﻧﺪ.

ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﻣﺮﺩﻡ ﺑﻪ ﺭﺍﻳﮕﺎﻥ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﻣﯽ‌ﮔﺬﺍﺭﻧﺪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﺮﺩﻩ ﺗﺎ ﺩﺭ ﻣﻮﺭﺩ ﺑﺎﺯﺍﺭ، ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﻳﺎ ﻣﺤﺼﻮﻝ ﺧﻮﺩ ﺩﺭﮎ ﺑﻬﺘﺮﯼ ﭘﻴﺪﺍ ﮐﻨﻴﺪ. ﺍﺑﺰﺍﺭﻫﺎﯼ ORM ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻨﺪ ﺗﺎ ﺍﺣﺴﺎﺳﺎﺕ ﺭﺍ ﭘﻴﮕﻴﺮﯼ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺷﻤﺎ ﺁﻣﺎﺭﯼ ﺩﺭ ﻣﻮﺭﺩ ﻣﻘﺒﻮﻟﻴﺘﺘﺎﻥ ﻣﯽ‌ﺩﻫﺪ. ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺳﺮﻭﻳﺲ‌ﻫﺎﯼ ﺁﮔﻬﯽ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻣﺎﻧﻨﺪ ﻭﺍﺣﺪ ﻓﻀﺎﯼ ﺁﮔﻬﯽ ﻓﻴﺲ‌ﺑﻮﮎ ﻳﺎ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰ ﺁﮔﻬﯽ ﻳﻮﺗﻴﻮﺏ، ﺍﻃﻼﻋﺎﺗﯽ ﻏﻨﯽ ﺩﺭ ﻣﻮﺭﺩ ﺍﻧﺪﺍﺯﻩ ﺑﺎﺯﺍﺭ ﻭ ﻋﻼﻳﻖ ﻣﺮﺩﻡ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﺑﺨﺸﺪ.

ﺷﻤﺎ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﺑﺰﺍﺭ ORM ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺣﺴﺎﺳﺎﺕ ﻭ ﺗﻐﻴﻴﺮﺍﺕ ﺩﺭ ﻣﻴﺰﺍﻥ ﺑﺤﺚ ﺩﺭ ﻣﻮﺭﺩ ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﺧﻮﺩ ﺭﺍ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ ﮐﺮﺩﻩ ﻭ ﺗﺎﺛﻴﺮ ﺩﻳﮕﺮ ﺳﺘﺎﺩﻫﺎﯼ ﺗﺒﻠﻴﻐﺎﺗﯽ، ﻣﺎﻧﻨﺪ ﺁﮔﻬﯽ‌ﻫﺎﯼ ﺗﻠﻮﻳﺰﻳﻮﻧﯽ ﻳﺎ ﺳﺘﺎﺩﻫﺎﯼ ﺁﻧﻼﻳﻦ، ﺭﺍ ﺩﺭﮎ ﮐﻨﻴﺪ.

ﺳﺎﺧﺖ ﺟﺎﻣﻌﻪ ﺁﻧﻼﻳﻦ ﮔﺮﻭﻫﯽ ﺭﺍ ﺩﺭ ﺩﺳﺘﺮﺱ ﺷﻤﺎ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﺪ ﺗﺎ ﺍﻃﻼﻋﺎﺕ ﻭ ﺑﺎﺯﺧﻮﺭﺩﻫﺎﻳﯽ ﺍﺯ ﺁﻧﻬﺎ ﺑﮕﻴﺮﻳﺪ ﺍﻣﺎ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﻳﻦ ﺑﺎﺯﺧﻮﺭﺩﻫﺎ ﺫﺍﺗﺎ ﺍﺯ ﺭﻭﯼ ﺗﻌﺼﺐ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ ﺯﻳﺮﺍ ﺁﻧﻬﺎ ﺩﺭ ﺟﺎﻣﻌﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺷﻤﺎ ﻗﺮﺍﺭ ﺩﺍﺭﻧﺪ. ﻫﻤﭽﻨﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺟﻮﺍﻣﻌﯽ ﻣﺎﻧﻨﺪ ﻓﻠﻴﮑﺮ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﺗﺎ ﺑﺒﻴﻨﻴﺪ ﮐﻪ ﻣﺮﺩﻡ ﭼﻪ ﭼﻴﺰﯼ ﺩﺭ ﻣﻮﺭﺩ ﺯﻧﺪﮔﻴﺸﺎﻥ، ﺣﺘﯽ ﺑﺪﻭﻥ ﺩﺭﮎ ﺁﻥ، ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﻣﯽ‌ﮔﺬﺍﺭﻧﺪ.

ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﮐﺎﻧﺎﻟﯽ ﺑﺎ ﺳﺮﻋﺖ ﺯﻳﺎﺩ ﺍﺳﺖ؛ ‌ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺩﺭﺳﺖ ﺑﺮﺍﯼ ﻣﻮﻓﻘﻴﺖ ﺿﺮﻭﺭﯼ ﺍﺳﺖ. ﺍﺳﺘﺮﺍﺗﮋﯼ‌ﻫﺎﯼ ﻣﻮﻓﻖ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺍﺯ ﺍﻳﻦ ﺩﺍﻧﺸﯽ ﺯﻳﺮﺑﻨﺎﻳﯽ ﺳﺮﭼﺸﻤﻪ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﮐﻪ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﻳﮏ ﮐﺎﻧﺎﻝ ﺍﺭﺗﺒﺎﻃﯽ ﺩﻭ ﺟﺎﻧﺒﻪ ﺍﺳﺖ:

ﺳﺎﺯﻣﺎﻥ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻓﻀﺎ ﺑﻪ ﺗﻌﺎﻣﻞ ﻣﯽ‌ﭘﺮﺩﺍﺯﻧﺪ ﻧﻴﺎﺯ ﺑﻪ ﻣﻨﺎﺑﻌﯽ ﺩﺍﺭﻧﺪ ﮐﻪ ﻧﻪ ﻓﻘﻂ ﺑﺮﺍﯼ ﺍﺭﺳﺎﻝ ﭘﻴﺎﻡ، ﺑﻠﮑﻪ ﺑﺮﺍﯼ ﺑﻪ ﻋﻬﺪﻩ ﮔﻴﺮﯼ ﭘﻴﺎﻡ‌ﻫﺎﯼ ﺩﺭﻳﺎﻓﺘﯽ ﮐﺎﺭﺳﺎﺯ ﺑﺎﺷﻨﺪ.

 

ﺧﻄﺮﺍﺕ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﻭ ﭼﺎﻟﺶ‌ﻫﺎ
ﻫﺮﮔﻮﻧﻪ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺎﻳﺪ ﺧﻄﺮﺍﺕ ﻭ ﭼﺎﻟﺶ‌ﻫﺎﯼ ﺗﻌﺎﻣﻞ ﺩﺭ ﺍﻳﻦ ﻣﺤﻴﻂ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﺩ ﻭ ﭘﺮﻭﺗﮑﻠﯽ ﺑﺮﺍﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺍﻳﻦ ﺧﻄﺮﺍﺕ ﺍﻳﺠﺎﺩ ﮐﻨﺪ. ﺍﻳﻦ ﻧﮑﺘﻪ ﺑﺮﺍﯼ ﺑﺎﺯﺍﺭﻳﺎﺏ‌ﻫﺎ ﻣﻬﻢ ﺍﺳﺖ ﮐﻪ ﺧﻄﺮﺍﺕ ﻭ ﭼﺎﻟﺶ‌ﻫﺎﯼ ﻣﺮﺑﻮﻁ ﺑﻪ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﺩﺭﮎ ﮐﻨﻨﺪ: ﺑﺮﺍﯼ ﺟﺒﺮﺍﻥ ﺍﺷﺘﺒﺎﻫﺎﺗﯽ ﮐﻪ ﺩﺭ ﻭﺏ ﺭﺥ ﺩﺍﺩﻩ ﺍﻧﺪ، ﺯﻣﺎﻧﯽ ﻃﻮﻻﻧﯽ ﻻﺯﻡ ﺍﺳﺖ.

ﺑﺮﺧﯽ ﺍﺯ ﺧﻄﺮﺍﺕ ﻭ ﭼﺎﻟﺶ‌ﻫﺎﯼ ﺭﺍﻳﺞ ﺩﺭ ﺯﻳﺮ ﺁﻣﺪﻩ‌ﺍﻧﺪ. ﻫﻴﭻ ﮐﺲ ﺍﻫﻤﻴﺖ ﻧﻤﯽ‌ﺩﻫﺪ. ﻣﺨﺼﻮﺻﺎ ﺯﻣﺎﻧﯽ ﮐﻪ ﺟﺎﻣﻌﻪ ﺭﺍ ﺍﺯ ﭘﺎﻳﻪ ﻣﯽ‌ﺳﺎﺯﻳﺪ ﺑﻪ ﺳﺨﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺟﺬﺏ ﻣﻮﺭﺩ ﻧﻈﺮﺗﺎﻥ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺳﺖ ﮐﻪ ﺩﺭﮎ ﭼﺸﻢ ﺍﻧﺪﺍﺯ ﺳﺎﺯﻣﺎﻥ ﺧﻮﺩ ﺑﺴﻴﺎﺭ ﺍﻫﻤﻴﺖ ﺩﺍﺭﺩ.

ﺍﻃﻤﻴﻨﺎﻥ ﺣﺎﺻﻞ ﮐﻨﻴﺪ ﮐﻪ ﺩﺭ ﻓﻀﺎﻳﯽ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﻣﺸﺘﺮﻳﺎﻧﺘﺎﻥ ﺣﻀﻮﺭ ﺩﺍﺷﺘﻪ ﻭ ﺍﺯ ﺧﻮﺍﻧﺪﻥ ﺍﺧﺒﺎﺭﯼ ﺭﺍﺟﻊ ﺑﻪ ﺷﻤﺎ ﺧﻮﺷﺤﺎﻝ ﻣﯽ‌ﺷﻮﻧﺪ.

ﻣﺸﺘﺮﻳﺎﻥ ﻧﺎﺭﺍﺿﯽ ﻧﻴﺰ ﺩﺭ ﻓﻀﺎﯼ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺣﻀﻮﺭ ﺩﺍﺭﻧﺪ (ﮐﻪ ﺁﺯﺍﺩﺍﻧﻪ ﻧﻈﺮﺍﺕ ﻣﺨﺮﺏ ﺍﺭﺳﺎﻝ ﻣﯽ‌ﮐﻨﻨﺪ). ﺣﺘﯽ ﺍﮔﺮ ﺗﻨﻬﺎ ﺑﺎﺯﺧﻮﺭﺩ ﺩﺭﻳﺎﻓﺘﻴﺘﺎﻥ ﻣﻨﻔﯽ ﺑﻮﺩ، ﺑﺎﺯ ﻫﻢ ﺑﺎﺯﺧﻮﺭﺩ ﺧﻮﺑﯽ ﺍﺳﺖ! ﺑﺪﻳﻦ ﮔﻮﻧﻪ ﻓﺮﺻﺘﯽ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﺩﺭﺑﺎﺭﻩ ﺁﻥ ﻣﻮﺿﻮﻉ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ.

ﺍﻳﻦ ﮐﺎﺭ ﻧﻴﺎﺯﻣﻨﺪ ﺗﻮﺟﻪ ﻭ ﻧﻈﺎﺭﺕ ﻣﺪﺍﻭﻡ ﺍﺳﺖ. ﮐﺎﻧﺎﻝ‌ﻫﺎﯼ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺭﺍﻳﮕﺎﻥ ﺑﺎﺷﻨﺪ ﺍﻣﺎ ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻧﻴﺎﺯﻣﻨﺪ ﺳﺮﻣﺎﻳﻪ ﮔﺬﺍﺭﯼ ﺯﻣﺎﻧﯽ ﻭ ﻣﻨﺎﺑﻊ ﺍﺳﺖ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﻣﻮﻓﻘﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺑﺪﺍﻧﻴﺪ ﮐﻪ ﺍﻫﺪﺍﻓﺘﺎﻥ ﺩﺭ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﭼﻴﺴﺖ ﻭ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺁﻥ ﺍﻫﺪﺍﻑ ﺑﺮﻧﺎﻣﻪ ﺭﻳﺰﯼ ﺯﻣﺎﻧﯽ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ.

ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ ﺗﺎﺛﻴﺮ ﺳﺘﺎﺩ ﺗﺒﻠﻴﻐﺎﺗﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺳﺨﺖ ﺑﺎﺷﺪ. ﺷﺎﻳﺪ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ (ﺗﺎﺛﻴﺮ) ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺳﺨﺖ ﺑﺎﺷﺪ ﺍﻣﺎ ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﺳﺘﺎﺩ ﺗﺒﻠﻴﻐﺎﺗﯽ ﺷﻤﺎ ﻣﻮﻓﻖ ﻧﺒﻮﺩﻩ ﺍﺳﺖ. ﺍﻧﺘﻈﺎﺭ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﮐﻪ ﺭﺍﻩ ﺣﻠﯽ (ﻳﺎ ﻣﻮﻓﻘﻴﺘﯽ) ﺩﺭ ﻃﻮﻝ ﻳﮏ ﺭﻭﺯ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﻳﺪ. ﺑﺎ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ ﭼﻴﺰﻫﺎﻳﯽ ﺷﺮﻭﻉ ﮐﻨﻴﺪ ﮐﻪ ﻗﺎﺑﻞ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ‌ﺍﻧﺪ ﻭ ﺑﻪ ﺗﻤﺎﺷﺎﯼ ﻣﻄﺎﻟﻌﺎﺕ ﻣﻮﺭﺩﯼ ﺩﺭ ﺍﻳﻦ ﻓﻀﺎ ﺑﻨﺸﻴﻨﻴﺪ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻨﺪ ﺳﺮﻣﺎﻳﻪ ﮔﺬﺍﺭﻳﺘﺎﻥ ﺩﺭ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﺑﻪ ﺩﺭﺁﻣﺪﯼ ﺑﺮﺍﯼ ﺳﺎﺯﻣﺎﻥ ﺧﻮﺩ ﺑﺪﻝ ﮐﻨﻴﺪ.
