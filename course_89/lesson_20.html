ﺩﺭﺱ ﻳﺎﺯﺩﻫﻢ- ﺗﺒﻠﻴﻐﺎﺕ ﺩﺭ ﺷﺒﮑﻪ ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ- ﺑﺨﺶ ﺍﻭﻝ

ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﺩﻟﻴﻞ ﮔﺴﺘﺮﺩﮔﯽ ﺩﺭ ﺩﻧﻴﺎﯼ ﺍﻣﺮﻭﺯ ﺍﻳﻨﺘﺮﻧﺖ ﺍﺯ ﺍﻫﻤﻴﺖ ﺑﺴﻴﺎﺭﯼ ﺑﺮﺧﻮﺭﺩﺍﻧﺪ. ﺍﺯ ﺁﻧﺠﺎﻳﯽ ﮐﻪ ﺍﻳﻦ ﺷﺒﮑﻪ‌ﻫﺎ ﻣﺨﺎﻃﺒﻴﻦ ﺯﻳﺎﺩﯼ ﺩﺍﺭﻧﺪ، ﺑﺴﻴﺎﺭ ﭘﻮﻳﺎ ﻫﺴﺘﻨﺪ. ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ -ﮐﻪ ﺷﺎﻣﻞ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﻣﺘﻨﯽ- ﺗﺼﻮﻳﺮﯼ- ﺻﻮﺗﯽ- ﻭ ﺗﺮﮐﻴﺒﯽ ﺍﺯ ﺍﻳﻦ‌ﻫﺎ ﺍﺳﺖ- ﺑﻪ ﻣﻨﻈﻮﺭ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﺳﺎﺧﺘﻪ ﺷﺪﻩ‌ﺍﻧﺪ. ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻳﻌﻨﯽ ﻧﻈﺮ ﺩﺍﺩﻥ ﺁﺳﺎﻥ ﺍﺳﺖ، ﻓﺮﺳﺘﺎﺩﻥ ﺁﺳﺎﻥ ﺍﺳﺖ ﻭ ﺑﻪ ﺯﺑﺎﻥ ﺗﺠﺎﺭﺕ، ﻫﺰﻳﻨﻪ ﺯﻳﺎﺩﯼ ﺑﺮﺍﯼ ﺩﻳﺪﻥ ﺍﻳﻦ ﺭﺳﺎﻧﻪ‌ﻫﺎ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺑﻪ ﺩﻟﻴﻞ ﻣﺎﻫﻴﺖ ﺍﺭﺗﺒﺎﻃﯽ ﺍﻳﻨﺘﺮﻧﺖ، ﺍﻳﻦ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ، ﻧﻈﺮ ﺩﺍﺩﻥ ﻭ ﺩﻳﺪﻥ ﻫﻤﮕﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﭘﻴﮕﻴﺮﯼ ﻭ ﺍﻧﺪﺍﺯﻩ ﮔﻴﺮﯼ ﺷﻮﻧﺪ.

ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ «ﻭﺏ ﺩﻭﻡ»، ﺭﺳﺎﻧﻪ ﻣﺼﺮﻑ ﮐﻨﻨﺪﮔﺎﻥ، ﺭﺳﺎﻧﻪ ﺷﻬﺮﻭﻧﺪﺍﻥ ﻭ ﺭﺳﺎﻧﻪ ﺟﺪﻳﺪ ﻧﻴﺰ ﻣﯽ‌ﻧﺎﻣﻨﺪ. ﺩﺭ ﺣﻘﻴﻘﺖ ﻣﻔﻴﺪﺗﺮﻳﻦ ﺭﺍﻩ ﺑﺮﺍﯼ ﺗﻌﺮﻳﻒ ﻣﻌﻨﺎﯼ ﺩﻗﻴﻖ ﺭﺳﺎﻧﻪ ﺍﺟﺘﻤﺎﻋﯽ، ﻣﻘﺎﻳﺴﻪ ﮐﺮﺩﻥ ﺁﻥ ﺑﺎ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺳﻨﺘﯽ ﺍﺳﺖ.

ﺍﻳﻨﺘﺮﻧﺖ ﻭ ﻧﺮﻡ ﺍﻓﺰﺍﺭﻫﺎﯼ ﺳﺎﺧﺘﻪ ﺷﺪﻩ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺁﻥ ﺍﻧﺘﺸﺎﺭ ﻭ ﭘﺨﺶ ﺭﺳﺎﻧﻪ ﺭﺍ ﺑﺮﺍﯼ ﻫﺮ ﮐﺴﯽ ﺁﺳﺎﻥ ﮐﺮﺩﻩ‌ﺍﻧﺪ. ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﻣﺤﺘﻮﺍﯼ ﻣﻨﺘﺸﺮ ﺷﺪﻩ ﻧﻴﺰ ﺑﺮﺍﯼ ﻫﻤﻪ ﺁﺳﺎﻥ ﺷﺪﻩ ﺍﺳﺖ.

ﻣﺎﻫﻴﺖ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻫﻤﮑﺎﺭﯼ، ﻣﺤﺘﻮﺍﯼ ﺗﻮﻟﻴﺪ ﺷﺪﻩ ﺍﺯ ﺳﻮﯼ ﮐﺎﺭﺑﺮﺍﻥ، ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻭ ﺑﻴﺸﺘﺮ ﺍﺯ ﻫﻤﻪ «ﺩﺭ ﺍﺭﺗﺒﺎﻁ ﺑﻮﺩﻥ» ﺍﺳﺖ.

 

ﻧﺸﺎﻧﻪ‌ﮔﺬﺍﺭﯼ ﻭ ﺟﻤﻊ‌ﺁﻭﺭﯼ
ﺍﮔﺮ ﻭﺏ‌ﺳﺎﻳﺖ‌ﻫﺎﻳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﮐﻪ ﺷﻤﺎ ﺍﻏﻠﺐ ﺍﺯ ﺁﻧﻬﺎ ﺑﺎﺯﺩﻳﺪ ﻣﯽ‌ﮐﻨﻴﺪ، ﻳﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺮﺟﻌﯽ ﺑﺮﺍﯼ ﻣﺮﺍﺟﻌﻪ ﺑﻌﺪﯼ ﺣﻔﻈﺸﺎﻥ ﮐﻨﻴﺪ، ﺑﻪ ﺳﺎﺩﮔﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﻣﺮﻭﺭﮔﺮ ﺧﻮﺩ ﺁﻧﻬﺎ ﺭﺍ ﻧﺸﺎﻧﻪ ﮔﺬﺍﺭﯼ ﮐﻨﻴﺪ.

ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺁﺩﺭﺱ ﺍﻳﻨﺘﺮﻧﺘﯽ ﺭﺍ ﺫﺧﻴﺮﻩ ﻣﯽ‌ﮐﻨﻴﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺩﻭﺑﺎﺭﻩ ﺁﻥ ﺭﺍ ﺑﻪ ﺭﺍﺣﺘﯽ ﺑﻴﺎﺑﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﻣﺠﻤﻮﻋﻪ ﺍﯼ ﺷﺨﺼﯽ ﺍﺯ ﻭﺑﺴﺎﻳﺖ ﺑﺮﺍﯼ ﺷﻤﺎ ﺗﻮﻟﻴﺪ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﺭﺍﻳﺎﻧﻪ ﺫﺧﻴﺮﻩ ﮐﻨﻴﺪ. ﺳﺎﻳﺖ‌ﻫﺎﯼ ﻧﺸﺎﻧﻪ ﮔﺬﺍﺭﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻧﻴﺰ ﺑﻪ ﺷﻤﺎ ﺍﺟﺎﺯﻩ ﺫﺧﻴﺮﻩ ﺍﻳﻦ ﻟﻴﻨﮏ‌ﻫﺎ ﻭ ﻫﻤﭽﻨﻴﻦ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺑﺮﭼﺴﺐ ﺑﺮﺍﯼ ﺗﻮﺻﻴﻒ ﻭ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﻳﺸﺎﻥ ﺑﺎ ﺩﻳﮕﺮ ﮐﺎﺭﺑﺮﺍﻥ ﺭﺍ ﻣﯽ‌ﺩﻫﻨﺪ. ﺑﺮﺧﯽ ﺍﺯ ﺍﻳﻦ ﺳﺎﻳﺖ‌ﻫﺎ ﺑﻪ ﺷﻤﺎ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﻟﻴﻨﮏ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﻳﮕﺮ ﮐﺎﺭﺑﺮﺍﻥ ﺑﻪ ﺁﻧﻬﺎ ﺭﺍﯼ ﺩﺍﺩﻩ ﺍﻧﺪ ﺭﺍ ﺛﺒﺖ ﮐﻨﻴﺪ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺳﺎﻳﺖ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﻗﺎﺑﻠﻴﺖ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺑﺮﭼﺴﺐ‌ﻫﺎﻳﯽ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﺑﺨﺸﻨﺪ ﮐﻪ ﺑﻪ ﻫﺪﻑ ﻣﺮﻭﺭ ﻟﻴﺴﺖ‌ﻫﺎ ﻭ ﻣﺠﻤﻮﻋﻪ‌ﻫﺎﯼ ﺗﻮﻟﻴﺪ ﺷﺪﻩ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻩ‌ﺍﻧﺪ.

ﺑﺎ ﻓﺮﺽ ﺍﻳﻨﮑﻪ ﺩﺭ ﺣﺎﻝ ﺣﺎﺿﺮ ﮐﺎﺭﺑﺮﺍﻥ ﺯﻳﺎﺩﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﻭﺑﺴﺎﻳﺖ‌ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻨﺪ ﺑﺮﺍﯼ ﺑﺎﺯﺍﺭﻳﺎﺏ‌ﻫﺎ ﻣﻬﻢ ﻭ ﺣﻴﺎﺗﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻭﺑﺴﺎﻳﺘﻬﺎ ﺣﻀﻮﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺣﺘﻤﺎ ﺍﺯ ﻃﺮﺍﺡ ﻭﺑﺴﺎﻳﺖ ﺧﻮﺩ ﺑﺨﻮﺍﻫﻴﺪ ﺗﺎ ﮐﺪ ﺍﻳﻦ ﻭﺑﺴﺎﻳﺖ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺳﺎﻳﺖ ﺷﻤﺎ ﻗﺮﺍﺭ ﺑﺪﻫﺪ ﺗﺎ ﮐﺎﺭﺑﺮﺍﻥ ﺑﺘﻮﺍﻧﻨﺪ ﺑﻪ ﺁﺳﺎﻧﯽ ﺻﻔﺤﺎﺕ ﻭﺑﺴﺎﻳﺖ ﺷﻤﺎ ﺭﺍ ﺫﺧﻴﺮﻩ ﻭ ﺟﻤﻌﺂﻭﺭﯼ ﮐﻨﻨﺪ.

 

ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻣﺤﺘﻮﺍ
ﺷﺎﻳﺪ ﺍﻭﻟﻴﻦ ﺳﺎﻳﺖ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻣﺤﺘﻮﺍﻳﯽ ﮐﻪ ﺑﻪ ﺫﻫﻦ ﺑﻴﺎﻳﺪ ﻳﻮﺗﻴﻮﺏ ﺑﺎﺷﺪ ﺍﻣﺎ ﮐﺎﺭﺑﺮﺍﻥ ﻋﻼﻭﻩ ﺑﺮ ﻭﻳﺪﺋﻮ، ﻣﺤﺘﻮﺍﻫﺎﯼ ﺩﻳﮕﺮﯼ ﻣﺎﻧﻨﺪ ﻋﮑﺲ، ﺻﺪﺍ ﻭ ﺍﻃﻼﻋﺎﺕ ﻧﻴﺰ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﻣﯽ‌ﻧﻬﻨﺪ. ﭼﻴﺰﯼ ﮐﻪ ﺑﺘﻮﺍﻥ ﺍﻳﺠﺎﺩ ﮐﺮﺩ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﻧﻴﺰ ﮔﺬﺍﺷﺖ. ﺳﺎﻳﺖ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻭﻳﺪﺋﻮ، ﻋﮑﺲ ﻭ ﺻﺪﺍ ﺭﺍ ﺁﺳﺎﻥ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﻓﻮﻕ ﺍﻟﻌﺎﺩﻩ ﻣﺤﺒﻮﺏ ﻫﺴﺘﻨﺪ. ﺳﺎﻳﺖ‌ﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ ﻓﻠﻴﮑﺮ ﻭ ﻳﻮﺗﻴﻮﺏ ﻫﻤﮕﯽ ﺑﻪ ﺍﻳﻦ ﻭﺍﻗﻌﻴﺖ ﺭﺳﻴﺪﻩ‌ﺍﻧﺪ ﮐﻪ ﻫﻤﻪ ﻣﺎ ﺩﻭﺳﺖ ﺩﺍﺭﻳﻢ ﻣﻄﺎﻟﺒﯽ ﺗﻮﻟﻴﺪ ﮐﻨﻴﻢ ﺗﺎ ﺩﻳﮕﺮﺍﻥ ﺑﺒﻴﻨﻨﺪ.

ﮐﻠﻤﻪ ﮐﻠﻴﺪﯼ ﺩﺭ ﺍﻳﻨﺠﺎ «ﺭﺍﻳﮕﺎﻥ» ﺍﺳﺖ: ﻫﻴﭻ ﻫﺰﻳﻨﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﻋﻀﻮ ﺷﺪﻥ، ﭼﻪ ﺑﺮﺍﯼ ﺁﭘﻠﻮﺩ ﻣﺤﺘﻮﺍ ﻭ ﭼﻪ ﺑﺮﺍﯼ ﺩﻳﺪﻥ ﺁﻥ، ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ؛ ﺍﮔﺮﭼﻪ ﺣﻖ ﻋﻀﻮﻳﺖ ﻃﻼﻳﯽ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﺩﻳﮕﺮﯼ ﺩﺭ ﺍﺧﺘﻴﺎﺭ ﺍﻋﻀﺎ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﺍﻳﻨﮕﻮﻧﻪ ﺳﺎﻳﺖ‌ﻫﺎ ﻣﺨﺎﻃﺒﻴﻦ ﻓﻮﻕ ﺍﻟﻌﺎﺩﻩ ﺯﻳﺎﺩﯼ ﺩﺍﺭﺩ. ﺩﺭ ﺣﻘﻴﻘﺖ، ﺑﺮ ﻃﺒﻖ ﺭﺗﺒﻪ ﺑﻨﺪﯼ ﺍﻟﮑﺴﺎ، ﻳﻮﺗﻴﻮﺏ ﺩﻭﻣﻴﻦ ﻣﻮﺗﻮﺭ ﺟﺴﺘﺠﻮﯼ ﺑﺰﺭﮒ ﺩﻧﻴﺎﺳﺖ!

ﻫﻤﭽﻨﻴﻦ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻳﻦ ﺳﺎﻳﺖ‌ﻫﺎ ﮐﺎﺭﺑﺮﺍﻥ ﺭﺍ ﺗﺸﻮﻳﻖ ﺑﻪ ﭘﺨﺶ ﻣﺤﺘﻮﺍ ﻣﯽ‌ﮐﻨﻨﺪ. ﻳﻮﺗﻴﻮﺏ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﮐﺎﺭﺑﺮﺍﻥ ﻭﻳﺪﺋﻮ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺁﺳﺎﻧﯽ ﺩﺭ ﺳﺎﻳﺖ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﻗﺮﺍﺭ ﺩﻫﻨﺪ ﻭ ﻓﻠﻴﮑﺮ ﻧﻴﺰ ﺑﺮﻧﺎﻣﻪ‌ﻫﺎ ﻭ ﻭﻳﺠﺖ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺗﻮﻟﻴﺪ ﮐﺮﺩﻩ ﮐﻪ ﺍﺟﺎﺯﻩ ﻧﻤﺎﻳﺶ ﺗﺼﺎﻭﻳﺮ ﺩﺭ ﺳﺮﺍﺳﺮ ﻭﺏ ﺭﺍ ﻣﯽ‌ﺩﻫﻨﺪ (ﻭ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻥ ﺁﻧﻬﺎ ﺭﺍ ﺑﻪ ﻭﺳﻴﻠﻪ Moo ﺑﺮ ﺭﻭﯼ ﮐﺎﺭﺕ‌ﻫﺎ ﻭ ﺑﺮﭼﺴﺐ‌ﻫﺎ ﭼﺎﭖ ﮐﺮﺩ.)

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻳﻦ ﺳﺎﻳﺖ‌ﻫﺎ ﺑﺮﺍﯼ ﺗﺎﻣﻴﻦ ﺧﺪﻣﺎﺕ ﺭﺍﻳﮕﺎﻧﯽ ﮐﻪ ﻋﺮﺿﻪ ﻣﯽ‌ﮐﻨﻨﺪ ﺑﻪ ﺗﺒﻠﻴﻐﺎﺕ ﻣﺘﮑﯽ ﻫﺴﺘﻨﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺑﺮﺧﯽ ﻧﻴﺰ ﻣﺨﺎﺭﺟﺸﺎﻥ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﻓﺮﻭﺵ ﺣﻖ ﻋﻀﻮﻳﺖ ﻃﻼﻳﯽ ﺗﺎﻣﻴﻦ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺁﮔﻬﯽ ﻧﻤﯽ‌ﭘﺬﻳﺮﻧﺪ.

ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻭﻳﺪﺋﻮﻫﺎ ﺑﻪ ﺻﻮﺭﺕ ﺁﻧﻼﻳﻦ ﺳﺎﻝ ﺑﻪ ﺳﺎﻝ ﻭ ﺑﺎ ﺍﻓﺰﺍﻳﺶ ﺳﺮﻋﺖ ﻭ ﮐﺎﻫﺶ ﻗﻴﻤﺖ ﭘﻬﻨﺎﯼ ﺑﺎﻧﺪ ﺍﻓﺰﺍﻳﺶ ﻣﯽ‌ﻳﺎﺑﺪ. ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻭﻳﺪﺋﻮ ﺑﻪ ﻭﺳﻴﻠﻪ ﺳﺎﻳﺖ‌ﻫﺎﯼ ﻭﻳﺪﺋﻮﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻣﺎﻧﻨﺪ ﻳﻮﺗﻴﻮﺏ ﺑﺴﻴﺎﺭ ﺁﺳﺎﻥ ﺍﺳﺖ. ﻫﺮﮐﺴﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻭﻳﺪﺋﻮﻫﺎﻳﯽ ﮐﻪ ﺑﻪ ﻫﺮ ﻭﺳﻴﻠﻪ‌ﺍﯼ، ﻭﺏ ﮐﻢ، ﻣﻮﺑﺎﻳﻞ ﻳﺎ ﺩﻭﺭﺑﻴﻦ‌ﻫﺎﯼ ﻓﻮﻕ ﭘﻴﺸﺮﻓﺘﻪ ﮔﺮﻓﺘﻪ ﺭﺍ ﺁﭘﻠﻮﺩ ﮐﻨﺪ.

ﻳﻮﺗﻴﻮﺏ ۶۰ ﺩﺭﺻﺪ ﺍﺯ ﺗﻤﺎﻡ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﮔﺎﻥ ﻭﻳﺪﺋﻮﯼ ﺁﻧﻼﻳﻦ ﺭﺍ ﺑﻪ ﺧﻮﺩ ﺍﺧﺘﺼﺎﺹ ﺩﺍﺩﻩ ﮐﻪ ﻃﺒﻖ ﮔﺰﺍﺭﺵ ﻳﻮﺗﻴﻮﺏ ﺩﺭ ﺳﺎﻝ ۲۰۱۱، ﻫﺮ ﺷﺐ ﺩﻭ ﻣﻴﻠﻴﺎﺭﺩ ﺑﻴﻨﻨﺪﻩ ﺍﺯ ﻭﻳﺪﺋﻮﻫﺎ ﺑﺎﺯﺩﻳﺪ ﮐﺮﺩﻩ ﻭ ﻫﺮ ﺭﻭﺯ ۱۳۸۲۴۰ ﻭﻳﺪﺋﻮ ﺁﭘﻠﻮﺩ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺍﻳﻦ ﺳﺎﻳﺖ ﺭﺍ ﺗﺒﺪﻳﻞ ﺑﻪ ﺑﺮﺗﺮﻳﻦ ﺳﺎﻳﺖ ﻭﻳﺪﺋﻮﻳﯽ ﺁﻧﻼﻳﻦ ﻭ ﺳﺎﻳﺖ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺭﯼ ﻭﻳﺪﺋﻮ ﮐﺮﺩﻩ ﺍﺳﺖ.

ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﺑﻴﺸﺘﺮﻳﻦ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻭﻳﺪﺋﻮ ﺩﺭ ﻭﺏ ﺑﺮ ﭘﺎﻳﻪ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺍﺳﺖ ﻭ ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺑﻴﺶ ﺍﺯ ۶ ﻣﻴﻠﻴﺎﺭﺩ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺁﻧﻼﻳﻦ ﻭﻳﺪﺋﻮﻳﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ (ﻳﻮﺗﻴﻮﺏ، ۲۰۱۴). ﻳﻮﺗﻴﻮﺏ ﺩﺭ ﺳﺎﻝ ۲۰۰۶ ﺑﻪ ﻗﻴﻤﺖ ۱.۶۵ ﻣﻴﻠﻴﺎﺭﺩ ﺩﻻﺭ ﺗﻮﺳﻂ ﮔﻮﮔﻞ ﺧﺮﻳﺪﺍﺭﯼ ﺷﺪ.

ﻳﻮﺗﻴﻮﺏ ﺳﺎﻳﺖ ﭘﻴﭽﻴﺪﻩ‌ﺍﯼ ﺍﺳﺖ ﮐﻪ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﻣﺘﻌﺪﺩﯼ ﺭﺍ ﻋﺮﺿﻪ ﻣﯽ‌ﮐﻨﺪ ﺍﻣﺎ ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺁﻥ ﺁﺳﺎﻥ ﺍﺳﺖ. ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺑﺮﺍﯼ ﮐﺎﺭﺑﺮﺍﻧﯽ ﮐﻪ ﺛﺒﺖ ﻧﺎﻡ ﻧﮑﺮﺩﻩ‌ﺍﻧﺪ ﺍﻳﻦ ﻗﺎﺑﻠﻴﺖ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﺘﻮﺍﻧﻨﺪ ﺑﻴﺸﺘﺮ ﻭﻳﺪﺋﻮﻫﺎﯼ ﻋﻤﻮﻣﯽ ﺭﺍ ﺑﺒﻴﻨﻨﺪ، ﻓﺮﺁﻳﻨﺪ ﺛﺒﺖ ﻧﺎﻣﯽ ﺳﺮﻳﻊ ﻭ ﻣﺴﺘﻘﻴﻢ ﺑﻪ ﺍﻋﻀﺎ ﺍﻳﻦ ﺍﺟﺎﺯﻩ ﺭﺍ ﻣﯽ‌ﺩﻫﺪ ﮐﻪ ﺑﯽ ﻧﻬﺎﻳﺖ ﮐﻠﻴﭗ ﺁﭘﻠﻮﺩ ﮐﻨﻨﺪ، ﻧﻈﺮ ﺑﺪﻫﻨﺪ، ﺑﻪ ﻧﻈﺮﺍﺕ ﭘﺎﺳﺦ ﺩﻫﻨﺪ ﻭ ﺩﺭ ﺧﻮﺭﺍﮎ ﻣﺤﺘﻮﺍ ﻋﻀﻮ ﺷﻮﻧﺪ ﮐﻪ ﺑﺎﻋﺚ ﺟﻠﺐ ﺗﻮﺟﻪ ﻭ ﻋﻼﻗﻪ ﺷﺎﻥ ﻣﯽ‌ﺷﻮﺩ.

ﻋﻤﻠﮑﺮﺩ ﭘﻴﺸﺮﻓﺘﻪ ﻭ ﻭﻳﮋﮔﯽ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪﺍﻧﻪ ﻳﻮﺗﻴﻮﺏ ﺑﺎﻋﺚ ﺷﺪﻩ ﮐﻪ ﺍﻳﻦ ﺳﺎﻳﺖ ﺑﻪ ﻃﻮﺭ ﻣﺪﺍﻭﻡ ﺧﺪﻣﺎﺕ ﺑﻴﺸﺘﺮ ﻭ ﺑﻬﺘﺮﯼ ﺭﺍ ﺑﻪ ﭘﺎﻳﮕﺎﻩ ﮐﺎﺭﺑﺮﺍﻥ ﺭﻭﺯ ﺍﻓﺰﻭﻧﺶ ﺍﺭﺍﺋﻪ ﺑﺪﻫﺪ.

 

ﺷﺒﮑﻪ ﺳﺎﺯﯼ ﺍﺟﺘﻤﺎﻋﯽ- ﺍﺗﺼﺎﻝ
ﺷﺒﮑﻪ ﺳﺎﺯﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﺑﺰﺍﺭﻫﺎﯼ ﺍﻳﻨﺘﺮﻧﺘﯽ ﺑﺮﺍﯼ ﺍﺭﺗﺒﺎﻁ ﻭ ﺳﺎﺧﺖ ﺭﻭﺍﺑﻂ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺍﺳﺖ. ﺳﺎﻳﺖ‌ﻫﺎﯼ ﺷﺒﮑﻪ ﺍﺟﺘﻤﺎﻋﯽ ﻣﺎﻧﻨﺪ ﻓﻴﺲ‌ﺑﻮﮎ ﻭ ﻟﻴﻨﮑﺪﻳﻦ ﺑﻪ ﮐﺎﺭﺑﺮﺍﻥ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻨﺪ ﺗﺎ ﭘﺮﻭﻓﺎﻳﻞ‌ﻫﺎﯼ ﺷﺨﺼﯽ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﻭ ﺳﭙﺲ ﺍﺯ ﻃﺮﻳﻖ ﺍﺗﺼﺎﻟﺸﺎﻥ ﺭﺳﺎﻧﻪ‌ﻫﺎﻳﯽ ﺭﺍ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻪ، ﭘﻴﺎﻡ‌ﻫﺎﻳﯽ ﺍﺭﺳﺎﻝ ﮐﺮﺩﻩ ﻭ ﻭﺑﻼﮒ ﻧﻮﻳﺴﯽ ﮐﻨﻨﺪ. ﻧﻪ ﺗﻨﻬﺎ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﺷﻤﺎ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻨﺪ ﺗﺎ ﺑﺎ ﺍﻋﻀﺎﯼ ﻣﺠﺎﺯﯼ ﺑﺨﺶ ﺧﻮﺩ ﺗﻌﺎﻣﻞ ﺑﺮﻗﺮﺍﺭ ﮐﻨﻴﺪ، ﺑﻠﮑﻪ ﺑﻪ ﺷﻤﺎ ﺍﺟﺎﺯﻩ ﮔﺴﺘﺮﺵ ﺷﺒﮑﻪ ﺷﺨﺼﻴﺘﺎﻥ ﺭﺍ ﻧﻴﺰ ﻣﯽ‌ﺩﻫﻨﺪ.

ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻦ، ﺳﺎﺯﻣﺎﻥ‌ﻫﺎﻳﯽ ﻣﺎﻧﻨﺪ ﺳﺎﺯﻣﺎﻥ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ، ﮔﺮﻭﻩ‌ﻫﺎ، ﺧﻴﺮﻳﻪ‌ﻫﺎ ﻭ ﻏﻴﺮﻩ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﭘﺮﻭﻓﺎﻳﻞ‌ﻫﺎﻳﯽ ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻳﻦ ﺷﺒﮑﻪ‌ﻫﺎ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﻭ ﺁﻥ ﺭﺍ ﺣﻔﻆ ﮐﻨﻨﺪ.

ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺎ ﺭﻭﺍﺑﻂ ﺑﺴﻴﺎﺭﯼ ﮐﻪ ﺗﻨﻬﺎ ﺑﻪ ﺻﻮﺭﺕ ﺁﻧﻼﻳﻦ ﺻﻮﺭﺕ ﻣﯽ‌ﮔﻴﺮﻧﺪ، ﻣﻌﻨﺎﯼ ﺗﺎﺯﻩ ﺍﯼ ﺑﺮﺍﯼ ﻭﺍﮊﻩ «ﺩﻭﺳﺖ» ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﺍﻧﺪ. ﺩﺭ ﻗﻠﻤﺮﻭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻻﺯﻡ ﻧﻴﺴﺖ ﮐﻪ ﺣﺘﻤﺎ ﺷﺨﺼﯽ ﺭﺍ ﺩﻳﺪﻩ ﺑﺎﺷﻴﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺑﺎ ﺍﻭ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﮐﻨﻴﺪ.

ﺻﻔﺤﺎﺕ ﻣﺸﺨﺼﺎﺕ ﺷﺨﺼﯽ ﻣﻴﺰﺍﻥ ﺯﻳﺎﺩﯼ ﺍﺯ ﺑﯽ ﻧﺎﻣﯽ ﺩﺭ ﺍﻳﻨﺘﺮﻧﺖ ﺭﺍ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩﻩ‌ﺍﻧﺪ. ﮐﺎﺭﺑﺮﺍﻥ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺍﻃﻼﻋﺎﺕ ﺯﻳﺎﺩﯼ ﺭﺍ ﺩﺭﺑﺎﺭﻩ ﺧﻮﺩ ﺍﻓﺸﺎ ﻣﯽ‌ﮐﻨﻨﺪ؛ ﺍﺯ ﺍﻃﻼﻋﺎﺕ ﺍﺳﺎﺳﯽ ﻭ ﺟﻤﻌﻴﺖ ﺷﻨﺎﺧﺘﯽ ﻣﺎﻧﻨﺪ ﺳﻦ، ﺟﻨﺴﻴﺖ ﻭ ﻣﺤﻞ ﺯﻧﺪﮔﯽ ﮔﺮﻓﺘﻪ ﺗﺎ ﻟﻴﺴﺖ‌ﻫﺎﻳﯽ ﻇﺮﻳﻒ ﻭ ﺩﻗﻴﻖ ﺍﺯ ﻣﻮﺍﺭﺩ ﭘﺴﻨﺪﻳﺪﻩ ﺷﺪﻩ ﻳﺎ ﺑﺮﻋﮑﺲ. ﺍﮔﺮﭼﻪ ﺍﺭﺗﺒﺎﻃﺎﺕ ﮐﺎﺭﺑﺮﺍﻥ ﺑﻪ ﺻﺮﺍﺣﺖ ﺑﻴﺎﻥ ﺷﺪﻩ، ﮐﺎﺭﺑﺮﺍﻥ ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺭﺍ ﺑﻪ ﺷﺒﮑﻪ‌ﻫﺎ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﺑﻪ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺗﺒﻠﻴﻐﺎﺗﯽ ﻟﻮ ﻣﯽ‌ﺩﻫﻨﺪ. ﮐﺎﺭﺑﺮﺍﻥ ﺗﻤﺎﻳﻞ ﺩﺍﺭﻧﺪ ﺗﺎ ﺍﺯ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﭘﺮﻭﻓﺎﻳﻞ ﺁﻧﻼﻳﻦ ﺁﻧﻬﺎ ﮔﺮﺩﺁﻭﺭﯼ ﻣﯽ‌ﺷﻮﺩ، ﺧﺒﺮﺩﺍﺭ ﻧﺸﻮﻧﺪ.

ﺷﺒﮑﻪ‌ﻫﺎﯼ ﻋﻤﻮﻣﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻣﺎﻧﻨﺪ ﻓﻴﺲ‌ﺑﻮﮎ، ﻋﻤﻮﻣﯽ ﻳﺎ ﻣﺎﻧﻨﺪ ﻟﻴﻨﮑﺪﻳﻦ ﻣﺘﻤﺮﮐﺰ ﺑﺎﺷﻨﺪ. ﻟﻴﻨﮑﺪﻳﻦ ﺷﺒﮑﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺣﺮﻓﻪ‌ﺍﯼ ﻫﺎﺳﺖ. ﺍﻋﻀﺎ ﺑﺎ ﺍﻓﺮﺍﺩﯼ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﺣﺮﻓﻪ‌ﺍﯼ ﻣﯽ‌ﺷﻨﺎﺳﻨﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﻋﻀﺎﻳﯽ ﮐﻪ ﻗﺒﻼ ﺑﺎ ﺁﻧﻬﺎ ﮐﺎﺭ ﮐﺮﺩﻩ‌ﺍﻧﺪ ﺭﺍ ﭘﻴﺸﻨﻬﺎﺩ ﺩﻫﻨﺪ.

- ﻓﻴﺲ‌ﺑﻮﮎ

ﭘﻴﮕﻴﺮﯼ ﺳﺘﺎﺩﻫﺎﯼ ﺗﺒﻠﻴﻐﺎﺗﯽ ﺩﺭ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ

ﺩﺭ ﺑﻴﺸﺘﺮ ﮐﺎﻧﺎﻝ‌ﻫﺎﯼ ﺷﺒﮑﻪ ﺍﺟﺘﻤﺎﻋﯽ، ﺣﻀﻮﺭ ﺧﻮﺩ ﺷﻤﺎ ، ﺷﺎﻣﻞ ﻣﻴﺰﺑﺎﻧﯽ ﻭﺏ ﻧﻤﯽ‌ﺷﻮﺩ ﮐﻪ ﺑﺪﺍﻥ ﻣﻌﻨﺎﺳﺖ ﮐﻪ ﭘﻴﮕﻴﺮﯼ ﺳﻔﺎﺭﺷﯽ ﻣﺤﺪﻭﺩ ﺷﺪﻩ ﺍﺳﺖ. ﺩﺭ ﺍﮐﺜﺮ ﻣﻮﺍﺭﺩ، ﺑﻪ ﭘﻴﮕﻴﺮﯼ ﺍﺭﺍﺋﻪ ﺷﺪﻩ ﺗﻮﺳﻂ ﮐﺎﻧﺎﻝ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺍﻋﺘﻤﺎﺩ ﻣﯽ‌ﺷﻮﺩ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﺭﺩ ﮐﻪ ﺗﺮﺍﻓﻴﮏ ﺑﺮﺍﯼ ﻳﮏ ﻭﺏ ﺍﻳﺠﺎﺩ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺷﻤﺎ ﻣﺎﻟﮏ ﺁﻥ ﻫﺴﺘﻴﺪ، ﻣﻤﮑﻦ ﺍﺳﺖ ﺗﺎ ﺣﺪﯼ ﺑﺘﻮﺍﻧﻴﺪ ﭘﻴﮕﻴﺮﯼ ﺭﺍ ﻳﮑﭙﺎﺭﭼﻪ ﺳﺎﺯﻳﺪ.

 

ﺁﻣﺎﺭ ﻓﻴﺲ‌ﺑﻮﮎ
ﺁﻣﺎﺭ ﻓﻴﺲ‌ﺑﻮﮎ ﺩﺭ ﺩﺳﺘﺮﺱ ﻣﺪﻳﺮﺍﻥ ﺻﻔﺤﻪ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ ﻭ ﺍﻃﻼﻋﺎﺗﯽ ﺩﺭ ﻣﻮﺭﺩ ﭼﮕﻮﻧﮕﯽ ﺗﻌﺎﻣﻞ ﻣﺮﺩﻡ ﺑﺎ ﻣﺤﺘﻮﺍ ﻭ ﺻﻔﺤﻪ ﺷﻤﺎ ﺩﺭ ﺍﺧﺘﻴﺎﺭﺗﺎﻥ ﻣﯽ‌ﮔﺬﺍﺭﺩ. ﺍﻳﻦ ﺍﻃﻼﻋﺎﺕ ﺷﺎﻣﻞ ﺍﻃﻼﻋﺎﺕ ﺟﻤﻌﻴﺖ ﺷﻨﺎﺧﺘﯽ ﺩﺭ ﻣﻮﺭﺩ ﻣﺮﺩﻣﯽ ﮐﻪ ﺑﺎ ﻣﺤﺘﻮﺍﯼ ﺷﻤﺎ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩﻩ‌ﺍﻧﺪ (ﺳﻦ، ﻧﮋﺍﺩ ﻭ ﺗﻔﮑﻴﮏ ﻣﺤﻠﯽ)، ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮐﺪﺍﻡ ﺯﺑﺎﻧﻪ‌ﻫﺎ ﻭ ﻣﺤﺘﻮﺍﯼ ﭘﺴﺖ ﺩﻳﺪﻩ ﺷﺪﻩ ﻭ ﺑﺎ ﺁﻧﻬﺎ ﺗﻌﺎﻣﻞ ﺑﺮﻗﺮﺍﺭ ﺷﺪﻩ، ﻭ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﭼﻨﺪ ﻧﻔﺮ ﺍﺯ ﻣﺮﺩﻡ ﻭ ﺩﺭ ﭼﻪ ﺯﻣﺎﻧﯽ ﻣﺤﺘﻮﺍﯼ ﺷﻤﺎ ﺭﺍ ﺍﺯ ﭘﻮﺷﺶ ﺧﺒﺮﯼ ﺧﻮﺩ ﺣﺬﻑ ﮐﺮﺩﻩ‌ﺍﻧﺪ، ﻣﯽ‌ﺷﻮﺩ. ﻫﻤﭽﻨﻴﻦ ﺍﻃﻼﻋﺎﺗﯽ ﻣﺒﻨﯽ ﺑﺮ ﭼﮕﻮﻧﮕﯽ ﺭﺳﻴﺪﻥ ﮐﺎﺭﺑﺮﺍﻥ ﺑﻪ ﺻﻔﺤﻪ ﺷﻤﺎ ﻧﻴﺰ ﻭﺟﻮﺩ ﺩﺍﺭﺩ.

 

ﺁﻣﺎﺭ ﻳﻮﺗﻴﻮﺏ
ﺁﻣﺎﺭ ﻳﻮﺗﻴﻮﺏ ﺩﺭ ﺩﺳﺘﺮﺱ ﺗﻤﺎﻡ ﮐﺎﺭﺑﺮﺍﻥ ﺍﻳﻦ ﺳﺎﻳﺖ ﻭ ﺩﺭ ﻣﻮﺭﺩ ﻭﻳﺪﺋﻮﻫﺎ ﻭ ﮐﺎﻧﺎﻝ‌ﻫﺎﯼ ﺁﻧﻬﺎﺳﺖ. ﻋﻼﻭﻩ ﺑﺮ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺗﻌﺪﺍﺩ ﺩﻓﻌﺎﺕ ﺑﺎﺯﺩﻳﺪ ﻭ ﻣﺤﺒﻮﺑﻴﺖ ﻭﻳﺪﺋﻮﻫﺎ ﺑﺮ ﺍﺳﺎﺱ ﻗﻠﻤﺮﻭ ﺟﻐﺮﺍﻓﻴﺎﻳﯽ، ﺑﺮﺧﯽ ﺍﻃﻼﻋﺎﺕ ﺟﻤﻌﻴﺖ ﺷﻨﺎﺧﺘﯽ ﻧﻴﺰ ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﻣﯽ‌ﺷﻮﻧﺪ. ﺍﻃﻼﻋﺎﺕ ﮐﺸﻒ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﮐﺎﺭﺑﺮﺍﻥ ﭼﮕﻮﻧﻪ ﺑﻪ ﻭﻳﺪﺋﻮﯼ ﺷﻤﺎ ﺭﺳﻴﺪﻩ‌ﺍﻧﺪ.
