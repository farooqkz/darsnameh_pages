function adminnotices($message, $errormsg = false)

{

                        echo '<div id="message" class="error">';

            }

            else {

                        echo '<div id="message" class="updated fade">';

            }

 

            echo "<p><strong>$message</strong></p></div>";

}

function adminnotices($message, $errormsg = false)

{

            if ($errormsg) {

                        echo '<div id="message" class="error">';

            }

            else {

                        echo '<div id="message" class="updated fade">';

            }

            echo "<p><strong>$message</strong></p></div>";

}
