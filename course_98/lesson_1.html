ﺩﺭﺱ ﺍﻭﻝ- ﺁﺷﻨﺎﻳﯽ ﺑﺎ ﻣﻔﻬﻮﻡ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ

ﺗﻘﺮﻳﺒﺎ ﻫﻤﻪ ﻣﺎ ﺩﺭ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﺑﺎ ﺑﻪ ﺷﻴﻮﻩ‌ﺍﯼ ﻧﺪﺍﺷﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﺧﺎﺹ ﺭﺍ ﺗﺤﺮﺑﻪ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﻭ ﺍﻳﻦ ﺣﺲ ﺭﺍ ﺍﻳﻨﮕﻮﻧﻪ ﺩﺭﻳﺎﻓﺘﻴﻢ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﮐﺎﺭﯼ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﻢ ﺍﻣﺎ ﻋﺼﺒﯽ ﻣﯽ‌ﺷﻮﻳﻢ ﻳﺎ ﺍﺣﺴﺎﺱ ﭘﺮﻳﺸﺎﻧﯽ ﺑﻪ ﻣﺎ ﺩﺳﺖ ﺩﺍﺩﻩ ﻭ ﻳﺎ ﺩﺭ ﻫﻨﮕﺎﻡ ﺍﻧﺠﺎﻡ ﺁﻥ ﺍﺣﺴﺎﺱ ﮐﺮﺩﻳﻢ ﮐﻪ ﺗﻮﺍﻧﺎﻳﯽ ﺍﻧﺠﺎﻣﺶ ﺭﺍ ﻧﺪﺍﺭﻳﻢ.  ﺍﻳﻦ ﺍﺣﺴﺎﺱ‌ﻫﺎ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺷﺮﺍﻳﻂ ﻣﺨﺘﻠﻒ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩ ﭘﻴﺶ ﺑﻴﺎﻳﺪ، ﺑﺮﺍﯼ ﻳﮑﯽ ﺩﺭ ﻫﻨﮕﺎﻡ ﻳﮏ ﺳﺨﻨﺮﺍﻧﯽ ﻋﻤﻮﻣﯽ، ﺑﺮﺍﯼ ﺩﻳﮕﺮﯼ ﻫﻨﮕﺎﻡ ﻣﻼﻗﺎﺕ ﻳﮏ ﺩﻭﺳﺖ ﺍﺯ ﺟﻨﺲ ﻣﺨﺎﻟﻒ ﻳﺎ ﺣﺘﯽ ﺗﻌﻤﻴﺮ ﺳﺎﺩﻩ ﻳﮏ ﻭﺳﻴﻠﻪ ﺭﻭﺯﻣﺮﻩ. ﺍﻣﺎ ﺑﺮﺍﯼ ﺭﻫﺎﻳﯽ ﺍﺯ ﺍﻳﻦ ﻭﺿﻌﻴﺖ ﭼﻪ ﺑﺎﻳﺪ ﮐﺮﺩ؟ ﺁﻳﺎ ﻣﯽ‌ﺗﻮﺍﻥ  ﺑﺎ ﺁﮔﺎﻫﯽ ﺑﻪ ﺟﻨﮓ ﮐﻤﺒﻮﺩ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺭﻓﺖ ﻭ ﺁﻥ ﺭﺍ ﺍﺯ ﺑﻴﻦ ﺑﺮﺩ؟ ﭘﺎﺳﺦ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ ﺗﺎ ﺣﺪﻭﺩﯼ ﻣﺜﺒﺖ ﺍﺳﺖ، ﺍﻟﺒﺘﻪ ﺑﻪ ﺧﻮﺩ ﺷﺨﺺ ﻧﻴﺰ ﺑﺴﺘﮕﯽ ﺩﺍﺭﺩ ﮐﻪ ﻗﺒﻮﻝ ﮐﻨﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﺑﻪ ﺭﺍﺣﺘﯽ ﺣﻞ ﮐﻨﺪ ﻭ ﺯﻭﺩ ﺩﺳﺘﭙﺎﭼﻪ ﻧﺸﻮﺩ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺳﺎﺩﻩ‌ﺗﺮ ﺷﺨﺺ ﻣﻮﺭﺩ ﻧﻈﺮ ﺑﺎﻳﺪ ﻗﺒﻮﻝ ﮐﻨﺪ ﮐﻪ ﻫﻴﭻ ﺍﺟﺒﺎﺭﯼ ﺩﺭ ﮐﺎﺭ ﻧﻴﺴﺖ ﮐﻪ ﺍﻭ ﺑﺎﻳﺪ ﺣﺘﻤﺂ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺴﯽ ﭘﺎﻳﻴﻦ ﺯﻧﺪﮔﯽ ﮐﻨﺪ ﻭ ﺩﺭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﻮﺍﺭﺩ ﺍﻭ ﺗﻮﺍﻧﺎﻳﯽ ﺗﻐﻴﻴﺮ ﺷﺮﺍﻳﻂ ﺭﺍ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺩﺍﺭﺍﺳﺖ.

ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺍﮐﺘﺴﺎﺑﯽ ﺍﺳﺖ ﻭ ﺑﻪ ﻣﺮﻭﺭ ﺯﻣﺎﻥ ﻣﯽ‌ﺗﻮﺍﻥ ﺁﻥ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﻳﺎ ﮐﺎﻫﺶ ﺩﺍﺩ. ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﺑﺮﺧﯽ ﺍﻓﺮﺍﺩ ﺑﻪ ﻣﺮﻭﺭ ﺯﻣﺎﻥ، ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﻳﮏ ﺩﺳﺘﻪ ﺍﺯ ﺁﻥ ﺍﻓﺮﺍﺩﯼ ﻫﺴﺘﻨﺪ ﮐﻪ ﺍﺯ ﺍﺳﺘﻘﻼﻝ ﺩﺭ ﮐﺎﺭﻫﺎﯼ ﺧﻮﺩ ﺑﺮﺧﻮﺭﺩﺍﺭ ﻧﻴﺴﺘﻨﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺍﺯ ﺁﻥ ﺟﻬﺖ ﺍﻫﻤﻴﺖ ﺩﺍﺭﺩ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ  ﺧﺎﻧﻮﺍﺩﻩ‌ﻫﺎ ﺭﺥ ﺩﻫﺪ ﻭ ﺍﺯ ﻫﻤﺎﻥ ﮐﻮﺩﮐﯽ ﺯﻣﻴﻨﻪ ﺳﺎﺯ ﮐﺎﻫﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﮐﻮﺩﮐﺎﻥ ﺷﻮﺩ. ﺧﺎﻧﻮﺍﺩﻩ‌ﺍﯼ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ ﮐﻪ ﻓﺮﺯﻧﺪﺍﻥ ﺗﺤﺖ ﮐﻨﺘﺮﻝ ﺷﺪﻳﺪ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﻗﺮﺍﺭ ﺩﺍﺭﻧﺪ ﻭ ﻭﺍﻟﺪﻳﻦ ﺑﺎ ﺍﻳﺠﺎﺩ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎﯼ ﻏﻴﺮ ﺿﺮﻭﺭﯼ ﺭﺍﺑﻄﻪ ﻓﺮﺯﻧﺪﺍﻥ ﺑﺎ ﻣﺤﻴﻂ ﺑﻴﺮﻭﻥ ﺭﺍ ﻣﺤﺪﻭﺩ ﻣﯽ‌ﮐﻨﻨﺪ؛ ﻓﺮﺯﻧﺪﺍﻥ ﺩﺭ ﺍﻳﻦ ﺧﺎﻧﻮﺍﺩﻩ ﺑﺎﻳﺪ ﺗﺴﻠﻴﻢ ﺑﯽ ﭼﻮﻥ ﻭ ﭼﺮﺍﯼ ﭘﺪﺭ ﻭ ﻣﺎﺩﺭ ﺑﺎﺷﻨﺪ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺯﻣﻴﻨﻪ ﺳﺎﺯ ﮐﺎﻫﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺁﻥ‌ﻫﺎ ﺩﺭ ﺑﺰﺭﮔﺴﺎﻟﯽ ﺧﻮﺍﻫﺪ ﺷﺪ، ﭼﺮﺍ ﮐﻪ ﺩﺭ ﮐﻮﺩﮐﯽ ﻧﺘﻮﺍﻧﺴﺘﻪ‌ﺍﻧﺪ ﻣﺴﺌﻮﻟﻴﺖ‌ﭘﺬﻳﺮ ﺷﺪﻩ ﻭ ﺧﻮﺩﺑﺎﻭﺭﯼ ﺭﺍ ﺩﺭ ﺧﻮﺩ ﺗﻘﻮﻳﺖ ﮐﻨﻨﺪ.   

ﺗﺎ ﺍﻳﻨﺠﺎﯼ ﮐﺎﺭ ﺣﺘﻤﺎ ﻣﺘﻮﺟﻪ ﺗﺎﺛﻴﺮﺍﺕ ﺩﻳﮕﺮﺍﻥ ﺑﺮ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺷﺪﻩ‌ﺍﻳﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻧﺎﺷﯽ ﺍﺯ ﺷﻨﺎﺧﺖ ﻣﺎ ﺍﺯ ﺧﻮﺩ، ﻣﺤﻴﻂ ﻭ ﺍﻓﺮﺍﺩ ﺣﺎﺿﺮ ﺩﺭ ﻣﺤﻴﻂ ﺍﺳﺖ. ﺷﻨﺎﺧﺖ ﺑﻴﺸﺘﺮ ﻣﻨﺠﺮ ﺑﻪ ﺍﻓﺰﺍﻳﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻣﯽ‌ﺷﻮﺩ. ﺩﺭ ﺣﻘﻴﻘﺖ ﺷﻨﺎﺧﺖ ﮐﻢ ﺍﺳﺖ ﮐﻪ ﺑﺎﻋﺚ ﮐﻤﺒﻮﺩ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻣﯽ‌ﺷﻮﺩ. ﺍﻣﺎ ﺟﺎﯼ ﻧﮕﺮﺍﻧﯽ ﻧﻴﺴﺖ، ﭼﺮﺍ ﮐﻪ ﻣﯽ‌ﺷﻮﺩ ﺑﺎ ﺍﻳﻦ ﺷﺮﺍﻳﻂ ﻣﻘﺎﺑﻠﻪ ﮐﺮﺩ ﻭ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺑﺎﻻ ﺑﺮﺩﻥ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. 

ﺑﺮﺍﯼ ﺷﺮﻭﻉ ﻭ ﭘﻴﺶ ﺍﺯ ﻫﺮ ﭼﻴﺰﯼ ﺍﺑﺘﺪﺍ ﺑﻬﺘﺮ ﺍﺳﺖ ﺧﻮﺩ ﺭﺍ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻫﺴﺘﻴﺪ ﺑﭙﺬﻳﺮﻳﺪ. ﻫﻤﻪ ﻣﺎ ﺩﺍﺭﺍﯼ ﻧﻘﺎﻁ ﺿﻌﻒ ﻫﺴﺘﻴﻢ ﻭ ﺑﻪ ﺷﻤﺎ ﻗﻮﻝ ﻣﯽ‌ﺩﻫﻴﻢ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺮﺍﯼ ﺗﻤﺎﻣﯽ ﻣﺮﺩﻣﺎﻥ ﮐﺮﻩ ﺯﻣﻴﻦ ﺍﺯ ﺁﺩﻡ‌ﻫﺎﯼ ﭘﻮﻟﺪﺍﺭ ﻭ ﻣﻌﺮﻭﻑ ﮔﺮﻓﺘﻪ ﺗﺎ ﺧﻴﺎﺑﺎﻥ‌ﺧﻮﺍﺏ‌ﻫﺎ ﺻﺪﻕ ﻣﯽ‌ﮐﻨﺪ. ﭘﺬﻳﺮﻓﺘﻦ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﻤﮏ ﺑﺰﺭﮔﯽ ﺑﻪ ﺩﺍﺷﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺷﺮﺍﻳﻂ ﮔﻮﻧﺎﮔﻮﻥ ﺑﮑﻨﺪ.

ﺍﻣﺎ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﭼﻪ ﺑﺎﻳﺪ ﮐﺮﺩ؟ ﭘﻴﺸﻨﻬﺎﺩ ﻣﺎ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻗﺪﻡ ﺍﻭﻝ، ﺩﺭ ﺟﺎﻳﯽ ﺑﻨﺸﻴﻨﻴﺪ ﻭ ﭼﻨﺪ ﺩﻗﻴﻘﻪ‌ﺍﯼ ﺩﺭ ﻣﻮﺭﺩ ﺧﻮﺩﺗﺎﻥ ﻓﮑﺮ ﮐﻨﻴﺪ. ﺍﻳﻨﮑﻪ ﺗﺎ ﮐﻨﻮﻥ ﺷﻤﺎ ﺩﺭﺑﺎﺭﻩ ﺧﻮﺩ ﭼﻪ ﭼﻴﺰﻫﺎﻳﯽ ﻳﺎﺩ ﮔﺮﻓﺘﻪ‌ﺍﻳﺪ ﻭ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻥ ﭼﻪ ﮐﺎﺭﻫﺎﻳﯽ ﺷﻤﺎ ﺭﺍ ﻏﻤﮕﻴﻦ ﻳﺎ ﺷﺎﺩ ﻣﯽ‌ﮐﻨﺪ. ﺑﻪ ﺍﻳﻦ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺩﺭ ﺁﻳﻨﺪﻩ ﭼﻪ ﮐﺴﯽ ﺷﻮﻳﺪ ﻭ ﺍﻳﻦ ﺷﺨﺺ ﭼﻪ ﻧﮑﺎﺕ ﻣﺜﺒﺖ ﻳﺎ ﻣﻨﻔﯽ‌ﺍﯼ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ. ﺍﻧﺘﻈﺎﺭ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺑﺎ ﻳﮏ ﺗﻔﮑﺮ ۱۰ ﺩﻗﻴﻘﻪ‌ﺍﯼ ﺑﻪ ﺟﻮﺍﺏ ﺑﺮﺳﻴﺪ ﻭ ﺍﻳﻦ ﺗﻔﮑﺮ ﮐﻮﺗﺎﻩ ﺗﻨﻬﺎ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺗﻐﻴﻴﺮﯼ ﺑﺰﺭﮒ ﺭﺍ ﺩﺭ ﺁﻳﻨﺪﻩ ﺑﺮﺍﯼ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﭘﯽ‌ﺭﻳﺰﯼ ﮐﻨﻴﺪ.

ﭘﺲ ﺍﺯ ﺗﻔﮑﺮ ﺩﺭﺑﺎﺭﻩ ﺧﻮﺩ، ﺑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﺷﻤﺎ ﺗﻮﺍﻧﺎﻳﯽ ﺗﻐﻴﻴﺮ ﻫﺮ ﭼﻴﺰﯼ ﺩﺭ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﺭﺍ ﻧﺪﺍﺭﻳﺪ ﻭ ﺍﺯ ﺍﻳﻦ ﺑﺎﺑﺖ ﻧﻴﺰ ﻧﺒﺎﻳﺪ ﺍﺣﺴﺎﺱ ﺳﺮﺧﻮﺭﺩﮔﯽ ﮐﻨﻴﺪ. ﺩﺭ ﺣﻘﻴﻘﺖ ﭘﺬﻳﺮﻓﺘﻦ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ ﮐﻨﺪ ﺗﺎ ﺭﺍﺣﺖ‌ﺗﺮ ﺑﭙﺬﻳﺮﻳﺪ ﮐﻪ ﺁﺩﻡ‌ﻫﺎﯼ ﺍﻃﺮﺍﻑ ﺷﻤﺎ ﻭ ﺷﻤﺎ ﺑﻪ ﮐﻤﮏ ﻫﻤﺪﻳﮕﺮ ﻧﻴﺎﺯﻣﻨﺪﻳﺪ ﻭ ﺗﺎ ﮐﻨﻮﻥ ﻧﻴﺰ ﺑﻪ ﻫﻤﺪﻳﮕﺮ ﺩﺭ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻫﺪﺍﻑ ﻣﻮﺭﺩ ﻧﻈﺮﺗﺎﻥ ﮐﻤﮏ ﮐﺮﺩﻩ‌ﺍﻳﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﭼﻴﺰﻫﺎﻳﯽ ﺭﺍ ﻋﻮﺽ ﮐﻨﻴﺪ ﮐﻪ ﻣﺘﻌﻠﻖ ﺑﻪ ﺷﻤﺎ ﺑﺎﺷﻨﺪ، ﭼﻴﺰﻫﺎﯼ ﺳﺎﺩﻩ‌ﺍﯼ ﭼﻮﻥ ﺗﻐﻴﻴﺮ ﺭﻧﮓ ﻣﻮ، ﺗﻐﻴﻴﺮ ﺭﮊﻳﻢ ﻏﺬﺍﻳﯽ ﻭ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ. ﺍﮔﺮ ﻭﺍﻗﻌﺎ ﺍﻳﻦ ﭼﻴﺰﻫﺎ ﻧﺎﺭﺍﺣﺘﺘﺎﻥ ﻣﯽ‌ﮐﻨﺪ، ﺑﻪ ﺩﻧﺒﺎﻝ ﺗﻐﻴﻴﺮ ﺁﻥ‌ﻫﺎ ﺑﺎﺷﻴﺪ. ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺑﭙﺮﺳﺪ ﭼﺮﺍ ﺷﻤﺎ ﺑﺎﻳﺪ ﻟﺒﺎﺱ ﺁﺑﯽ ﻳﺎ ﺻﻮﺭﺗﯽ ﺭﻧﮓ ﺧﻮﺩ ﺭﺍ ﺑﭙﻮﺷﻴﺪ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﻟﺒﺎﺱ ﻳﺎ ﺭﻧﮓ ﺁﻥ ﺧﻮﺷﺘﺎﻥ ﻧﻤﯽ‌ﺁﻳﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺧﻮﺩﺗﺎﻥ ﺑﺎﺷﻴﺪ ، ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﺯ ﻫﺮ ﭼﻴﺰﯼ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﺍﺣﺴﺎﺱ ﺑﺪﯼ ﻣﯽ‌ﺩﻫﺪ ﺩﻭﺭﯼ ﮐﻨﻴﺪ.


ﺣﺘﯽ ﺑﻪ ﺷﻤﺎ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﻴﻢ ﺑﻪ ﺩﻧﺒﺎﻝ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎﻳﯽ ﺑﺮﻭﻳﺪ ﮐﻪ ﺑﺎﺭﻫﺎ ﺩﻭﺳﺖ ﺩﺍﺷﺘﻴﺪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﺍﻣﺎ ﺑﻨﺎ ﺑﻪ ﺩﻻﻳﻠﯽ ﺗﺎ ﮐﻨﻮﻥ ﻣﻮﻓﻖ ﺑﻪ ﺍﻧﺠﺎﻡ ﺁﻥ ﻧﺸﺪﻳﺪ. ﺍﮔﺮ ﺷﻤﺎ ﺑﻪ ﺧﻮﺍﻧﺪﻥ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺗﺎﺭﻳﺨﯽ، ﻧﻮﺍﺧﺘﻦ ﻳﮏ ﺳﺎﺯ ﻣﻮﺳﻴﻘﯽ، ﺭﻓﺘﻦ ﺑﻪ ﮐﻼﺱ ﺭﻗﺺ ﻭ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﻋﻼﻗﻪ ﺩﺍﺭﻳﺪ ﭼﺮﺍ ﺑﻪ ﺩﻧﺒﺎﻝ ﺍﻳﻦ ﻣﻮﺿﻮﻋﺎﺕ ﻧﻤﯽ‌ﺭﻭﻳﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻋﺎﺕ ﺩﺭ ﺣﻘﻴﻘﺖ ﺑﺨﺸﯽ ﺍﺯ ﺗﻤﺎﻳﻼﺕ ﺷﻤﺎ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﻪ ﺷﻨﺎﺳﺎﻧﺪﻥ ﺷﻤﺎ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﻭ ﺑﺎﺯﺧﻮﺭﺩ ﮔﺮﻓﺘﻦ ﺍﺯ ﺁﻥ‌ﻫﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﻢ ﺑﻪ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺩﻳﮕﺮﺍﻥ ﺑﻪ ﺻﻮﺭﺕ ﻣﻨﻔﯽ ﺩﺭﺑﺎﺭﻩ ﺷﻤﺎ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﻧﻴﺰ ﻧﺒﺎﻳﺪ ﺗﻮﺟﻪ ﮐﻨﻴﺪ. ﺁﻥ‌ﻫﺎ ﻣﻌﻤﻮﻟﺂ ﻣﻮﺿﻮﻋﺎﺕ ﺭﺍ ﺍﺯ ﺩﻳﺪﮔﺎﻩ ﺧﻮﺩ ﺑﻴﺎﻥ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﺷﻨﺎﺧﺖ ﮐﺎﻓﯽ ﻧﺴﺒﺖ ﺑﻪ ﻣﺴﺎﺋﻞ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ. ﺑﻬﺘﺮ ﺍﺳﺖ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ﺍﺯ ﻫﺮ ﭼﻴﺰﯼ ﮐﻪ ﺍﻧﺮﮊﯼ ﻣﻨﻔﯽ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﺩﻫﺪ، ﺩﻭﺭﯼ ﮐﻨﻴﺪ.

ﺍﻟﺒﺘﻪ ﺩﺭ ﺭﺍﻩ ﭘﺬﻳﺮﺵ ﺧﻮﺩﺗﺎﻥ ﻧﺒﺎﻳﺪ ﺩﺭﺟﺎ ﺑﺰﻧﻴﺪ ﻭ ﺑﮕﻮﻳﻴﺪ ﻣﻦ ﻫﻤﻴﻦ ﻫﺴﺘﻢ ﻭ ﻫﻤﻴﻦ ﻫﻢ ﺧﻮﺍﻫﻢ ﻣﺎﻧﺪ، ﺑﻠﮑﻪ ﺑﺎﻳﺪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺑﻬﺘﺮ ﺷﺪﻥ ﻧﻴﺰ ﺑﺎﺷﻴﺪ. ﮐﻤﯽ ﺗﻨﺒﻠﯽ ﺭﺍ ﮐﻨﺎﺭ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻫﺪﺍﻑ ﻣﺜﺒﺖ ﺯﻧﺪﮔﻴﺘﺎﻥ ﺗﻼﺵ ﮐﻨﻴﺪ ﺗﺎ ﺍﺣﺴﺎﺱ ﻣﺜﺒﺖ ﺗﻼﺵ ﮐﺮﺩﻥ ﺭﺍ ﺣﺲ ﮐﻨﻴﺪ. ﺍﻳﻦ ﺗﻼﺵ‌ﻫﺎ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺍﺣﺴﺎﺱ ﻣﻔﻴﺪ ﺑﻮﺩﻥ ﮐﻨﻴﺪ ﻭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

ﺑﻪ ﭘﺎﻳﺎﻥ ﺟﻠﺴﻪ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ، ﺟﻠﺴﻪ ﺍﻣﺮﻭﺯ ﺑﻴﺸﺘﺮ ﺣﺎﻟﺖ ﺁﺷﻨﺎﻳﯽ ﺑﺎ ﻣﻮﺿﻮﻉ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺭﺍ ﺩﺍﺷﺖ ﻭ ﺩﺭ ﺟﻠﺴﺎﺕ ﺑﻌﺪ ﻧﺤﻮﻩ ﭘﯽ‌ﺭﻳﺰﯼ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻭ ﺑﻴﺸﺘﺮ ﮐﺮﺩﻥ ﺁﻥ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﺮﻭﺭ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.
