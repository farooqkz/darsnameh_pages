ﺩﺭﺱ ﺩﻭﻡ- ﭼﮕﻮﻧﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺭﺍ ﺑﺴﺎﺯﻳﻢ

ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﺗﻌﺮﻳﻒ‌ﻫﺎﻳﯽ ﻋﻤﻮﻣﯽ ﺍﺯ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺍﺭﺍﺋﻪ ﮐﺮﺩﻳﻢ ﻭ ﮔﻔﺘﻴﻢ ﮐﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺍﮐﺘﺴﺎﺑﯽ ﺍﺳﺖ ﻭ ﺑﻪ ﻣﺮﻭﺭ ﺯﻣﺎﻥ ﻭ ﺑﺎ ﺍﻧﺠﺎﻡ ﺑﺮﺧﯽ ﮐﺎﺭﻫﺎﯼ ﻣﺜﺒﺖ ﻣﯽ‌ﺗﻮﺍﻥ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﺩﺍﺩ. ﻫﻤﭽﻨﻴﻦ ﻳﺎﺩ ﮔﺮﻓﺘﻴﻢ ﮐﻪ ﭘﺎﺭﺍﻣﺘﺮﻫﺎﻳﯽ ﭼﻮﻥ ﺧﻮﺩ ﺷﺨﺺ، ﻣﺤﻴﻂ ﻭ ﺍﻓﺮﺍﺩ ﺣﺎﺿﺮ ﺩﺭ ﻣﺤﻴﻂ ﺩﺭ ﺍﻓﺰﺍﻳﺶ ﻳﺎ ﮐﺎﻫﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺷﺨﺺ ﻧﻘﺶ ﺩﺍﺭﻧﺪ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺍﺑﺘﺪﺍ ﺭﺍﻫﮑﺎﺭﻫﺎﻳﯽ ﺑﺮﺍﯼ ﭘﺬﻳﺮﺵ ﺷﺨﺺ ﺍﺯ ﺳﻮﯼ ﺧﻮﺩ ﺷﺨﺺ ﺍﺭﺍﺋﻪ ﮐﺮﺩﻳﻢ.
 
ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﺎ ﻫﻢ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺭﺍ ﻣﺮﻭﺭ ﮐﻨﻴﻢ ﮐﻪ ﺩﺭ ﺳﺎﺧﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻣﺜﺒﺖ ﺷﺨﺺ ﻧﻘﺶ ﺩﺍﺭﻧﺪ. ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ ﻧﮕﺮﺵ ﻣﺜﺒﺖ ﺑﻪ ﺭﻭﻳﺪﺍﺩﻫﺎ ﻭ ﻣﺤﻴﻂ ﺳﺒﺐ ﺑﺎﻻﺭﻓﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺍﻓﺮﺍﺩ ﻣﯽ‌ﺷﻮﺩ. ﺍﻟﺒﺘﻪ ﻧﺒﺎﻳﺪ ﺍﻳﻦ ﻃﻮﺭ ﺑﺮﺩﺍﺷﺖ ﺷﻮﺩ ﮐﻪ ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻥ ﺍﻳﻦ ﻧﮕﺮﺵ ﺩﺭ ﻫﺮ ﻣﻮﺿﻮﻋﯽ ﺳﺒﺐ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎ ﻭ ﮔﺮﻓﺘﻦ ﺑﻬﺘﺮﻳﻦ ﻧﺘﻴﺠﻪ ﻣﯽ‌ﺷﻮﺩ. ﺑﺪﺑﻴﻨﯽ ﺩﺭ ﺑﻌﻀﯽ ﻣﻮﺍﻗﻊ ﻧﻴﺎﺯ ﺍﺳﺖ ﺍﻣﺎ ﺩﺍﺷﺘﻦ ﺑﺪﺑﻴﻨﯽ ﻧﺴﺒﺖ ﺑﻪ ﺍﮐﺜﺮ ﮐﺎﺭﻫﺎ ﺍﺯ ﺭﻓﺘﺎﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺳﺒﺐ ﮐﺎﻫﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺷﻤﺎ ﻣﯽ‌ﺷﻮﺩ.
 
ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﺍﻳﻨﮕﻮﻧﻪ ﺑﺮﺍﯼ ﺧﻮﺩ ﻣﺠﺴﻢ ﮐﻨﻴﺪ ﮐﻪ ﺍﻓﮑﺎﺭ ﻣﻨﻔﯽ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺑﺎﻳﺪ ﺑﺸﻨﺎﺳﻴﺪ. ﺍﻓﮑﺎﺭ ﻣﻨﻔﯽ ﭼﻴﺰﯼ ﻣﺎﻧﻨﺪ ﺍﻳﻦ ﺟﻤﻠﻪ ﻫﺎﺳﺖ: «ﻣﻦ ﻧﻤﯽ‌ﺗﻮﻧﻢ ﺍﻳﻦ ﮐﺎﺭ ﺭﻭ ﺍﻧﺠﺎﻡ ﺑﺪﻡ»، «ﻣﻦ ﻣﻄﻤﺌﻨﻢ ﺗﻮﯼ ﺍﻧﺠﺎﻡ ﮐﺎﺭ ﺷﮑﺴﺖ ﻣﯽ‌ﺧﻮﺭﻡ» ﻳﺎ ﺣﺘﯽ ﺟﻤﻠﻪ‌ﺍﯼ ﻣﺎﻧﻨﺪ «ﻫﻴﭻ‌ ﮐﯽ ﺩﻭﺳﺖ ﻧﺪﺍﺭﻩ ﺑﻪ ﺣﺮﻑ ﻣﻦ ﮔﻮﺵ ﺑﺪﻩ». ﺩﺭ ﻭﺍﻗﻊ ﺑﻴﺎﻥ ﺍﻳﻦ ﺟﻤﻼﺕ ﻧﺸﺎﻥ‌ﺩﻫﻨﺪﻩ ﻧﻮﻋﯽ ﺑﺪﺑﻴﻨﯽ ﺩﺭ ﺷﺨﺺ ﮔﻮﻳﻨﺪﻩ ﺍﺳﺖ ﮐﻪ ﺳﺒﺐ ﻣﯽ‌ﺷﻮﺩ ﻋﺰﺕ ﻧﻔﺲ ﻭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﻭﯼ ﺩﺭ ﺣﺎﻟﺘﯽ ﭘﺎﻳﻴﻦ ﺑﺎﻗﯽ ﺑﻤﺎﻧﺪ.
 
ﻫﺪﻑ ﺍﺯ ﺍﻳﻨﮑﻪ ﺷﻤﺎ ﺍﻓﮑﺎﺭ ﻣﻨﻔﯽ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺷﻨﺎﺳﺎﻳﯽ ﮐﻨﻴﺪ، ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺁﮔﺎﻫﺎﻧﻪ ﻭ ﺑﺎ ﺷﻨﺎﺧﺖ ﺍﺯ ﺍﻓﮑﺎﺭ ﺧﻮﺩ ﺑﻪ ﺳﻤﺖ ﻣﺜﺒﺖ ﻓﮑﺮ ﮐﺮﺩﻥ ﺩﺭﺑﺎﺭﻩ ﻣﻮﺿﻮﻋﺎﺕ ﺣﺮﮐﺖ ﮐﻨﻴﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺧﻮﺩ ﺭﺍ ﺍﻓﺰﺍﻳﺶ ﺩﻫﻴﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺑﺎﻳﺪ ﺍﻓﮑﺎﺭ ﻣﻨﻔﯽ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺍﻓﮑﺎﺭ ﻣﺜﺒﺖ ﺗﻌﻮﻳﺾ ﮐﻨﻴﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﺟﻤﻼﺕ ﻣﻨﻔﯽ ﺑﺎﻻ ﺭﺍ ﺑﺎ ﺟﻤﻼﺗﯽ ﺍﻳﻦ ﭼﻨﻴﻦ ﻋﻮﺽ ﮐﻨﻴﺪ: «ﻣﻦ ﻣﯽ‌ﺗﻮﻧﻢ ﺍﺯ ﭘﺲ ﺍﻳﻦ ﮐﺎﺭ ﺑﺮ ﺑﻴﺎﻡ»، «ﻣﻦ ﻣﯽ‌ﺗﻮﻧﻢ ﺗﻮﯼ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ ﻣﻮﻓﻖ ﺑﺎﺷﻢ» ﻭ ﻳﺎ «ﻣﺮﺩﻡ ﺑﻪ ﻣﻦ ﮔﻮﺵ ﻣﻴﺪﻥ، ﻣﮕﻪ ﻣﻴﺸﻪ ﺗﻮﯼ ﺍﻳﻦ ﻫﻤﻪ ﺁﺩﻡ ﭼﻨﺪ ﻧﻔﺮ ﭘﻴﺪﺍ ﻧﺸﻦ ﮐﻪ ﺑﻪ ﺣﺮﻓﺎﻡ ﮔﻮﺵ ﺑﺪﻥ؟».
 
ﺑﺪ ﻧﻴﺴﺖ ﮐﻪ ﺍﻳﻦ ﺗﻤﺮﻳﻨﺎﺕ ﺳﺎﺩﻩ ﺭﺍ  ﺑﺎ ﺧﻮﺩ ﺗﮑﺮﺍﺭ ﮐﻨﻴﺪ. ﻫﺮ ﺯﻣﺎﻥ ﮐﻪ ﺑﺎ ﺩﻳﺪ ﻣﻨﻔﯽ ﺑﻪ ﻗﻀﺎﻳﺎ ﻓﮑﺮ ﮐﺮﺩﻳﺪ، ﺁﻥ ﺭﺍ ﺑﺎ ﺍﻓﮑﺎﺭﯼ ﻣﺜﺒﺖ ﻋﻮﺽ ﮐﻨﻴﺪ. ﻣﺜﻼ ﻧﮕﻮﻳﻴﺪ «ﻣﻦ ﺍﺣﺴﺎﺱ ﭼﺎﻗﯽ ﻣﯽ‌ﮐﻨﻢ»، ﺑﮕﻮﻳﻴﺪ «ﺍﻳﻦ ﺭﻭﺯﺍ ﻳﻪ ﻣﻘﺪﺍﺭﯼ ﺗﭙﻞ ﺷﺪﻡ ﺍﻣﺎ ﻣﯽ‌ﺧﻮﺍﻡ ﻳﻪ ﺟﻮﺭﺍﻳﯽ ﺳﺎﻟﻢ‌ﺗﺮ ﻏﺬﺍ ﺑﺨﻮﺭﻡ ﻭ ﻳﻪ ﻣﻘﺪﺍﺭﯼ ﻫﻢ ﻭﺭﺯﺵ ﮐﻨﻢ ﺗﺎ ﺳﻼﻣﺘﯽ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻢ.» ﺑﺎ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﺗﻤﺮﻳﻨﺎﺕ ﭘﺲ ﺍﺯ ﻣﺪﺗﯽ ﺧﻮﺩ ﺑﻪ ﺧﻮﺩ ﺍﻓﮑﺎﺭ ﻣﺜﺒﺖ ﺫﻫﻦ ﺷﻤﺎ ﺭﺍ ﺍﺷﻐﺎﻝ ﮐﺮﺩﻩ ﻭ ﺍﻳﻨﮕﻮﻧﻪ ﻃﺮﻳﻘﻪ ﻓﮑﺮ ﮐﺮﺩﻥ ﺧﻮﺩ ﺭﺍ ﻋﻮﺽ ﺧﻮﺍﻫﻴﺪ ﮐﺮﺩ.
 
ﺍﻟﺒﺘﻪ ﺑﺮﺍﯼ ﻋﻮﺽ ﮐﺮﺩﻥ ﻓﻀﺎﯼ ﺫﻫﻨﯽ ﻣﻨﻔﯽ ﺑﻪ ﻣﺜﺒﺖ، ﺍﻓﺮﺍﺩ ﻭ ﻣﺤﻴﻄﺘﺎﻥ ﻧﻴﺰ ﻧﻘﺶ ﺗﻌﻴﻴﻦ ﮐﻨﻨﺪﻩ‌ﺍﯼ ﺩﺍﺭﻧﺪ. ﺑﻮﺩﻥ ﺩﺭ ﮐﻨﺎﺭ ﺩﻭﺳﺘﺎﻥ ﻭ ﺍﻋﻀﺎﯼ ﺧﺎﻧﻮﺍﺩﻩ‌ﺍﯼ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﺣﺲ ﺧﻮﺑﯽ ﺭﺍ ﻣﻨﺘﻘﻞ ﻣﯽ‌ﮐﻨﻨﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻓﻀﺎﯼ ﺫﻫﻨﯽ ﻣﺜﺒﺘﯽ ﺭﺍ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﻪ‌ﻭﺟﻮﺩ ﺑﻴﺎﻭﺭﺩ ﻳﺎ ﺗﻐﻴﻴﺮ ﻓﻀﺎﯼ ﻣﻨﻔﯽ ﺑﻪ ﻣﺜﺒﺖ ﺍﻓﮑﺎﺭﺗﺎﻥ ﺭﺍ ﺳﺮﻋﺖ ﺑﺒﺨﺸﺪ. ﺍﻟﺒﺘﻪ ﻫﻤﭽﻨﺎﻥ ﺗﻮﺿﻴﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﺍﺯ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺑﻮﺩﻥ ﺩﺭ ﮐﻨﺎﺭﺷﺎﻥ ﺑﺮﺍﻳﺘﺎﻥ ﺧﻮﺷﺎﻳﻨﺪ ﻧﻴﺴﺖ، ﺩﻭﺭﯼ ﮐﻨﻴﺪ؛ ﻣﺨﺼﻮﺻﺂ ﺩﻭﺳﺘﺎﻥ ﻳﺎ ﺍﻗﻮﺍﻣﯽ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺳﺮﺯﻧﺶ ﮐﺮﺩﻩ ﻳﺎ ﻧﮑﺎﺕ ﻣﻨﻔﯽ ﺷﺨﺼﻴﺘﯽ ﺷﻤﺎ ﺭﺍ ﻣﺪﺍﻡ ﺑﻪ ﺷﻤﺎ ﮔﻮﺷﺰﺩ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﻳﺎ ﺣﺘﯽ ﻭﺿﻌﻴﺖ ﺷﻤﺎ ﺭﺍ ﺑﺎ ﺍﻓﮑﺎﺭ ﺧﻮﺩﺷﺎﻥ ﻣﻘﺎﻳﺴﻪ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﮔﻮﻳﻨﺪ «ﺑﺎﻳﺪ» ﺍﻳﻨﮕﻮﻧﻪ ﺑﺎﺷﯽ.
 
ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺩﻭﺭﯼ ﮐﺮﺩﻥ ﺍﺯ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺩﺳﺖ ﮐﻢ ﺗﺎ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺷﻤﺎ ﻣﺸﻐﻮﻝ ﺳﺎﺧﺘﻦ ﻳﮏ ﺩﻧﻴﺎﯼ ﻣﺜﺒﺖ ﺑﺮﺍﯼ ﺧﻮﺩ ﻫﺴﺘﻴﺪ ﺑﺎﻳﺪ ﺍﺩﺍﻣﻪ ﭘﻴﺪﺍ ﮐﻨﺪ. ﺑﻌﻀﯽ ﻭﻗﺖ‌ﻫﺎ ﻫﻢ ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺍﻓﺮﺍﺩﯼ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﺍﺣﺴﺎﺱ ﺑﺴﻴﺎﺭ ﺧﻮﺑﯽ ﻣﯽ‌ﺩﻫﻨﺪ، ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺑﭙﺮﺳﻴﺪ ﭼﺮﺍ ﻣﻦ ﻧﺒﺎﻳﺪ ﻭﻗﺖ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﺑﺎ ﺍﻭ ﺑﮕﺬﺭﺍﻧﻢ؟ ﭘﻴﺸﻨﻬﺎﺩ ﻣﺎ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ!
 
ﺩﻭﺭﯼ ﮐﺮﺩﻥ ﺍﺯ ﺗﻔﮑﺮﺍﺕ ﻣﻨﻔﯽ، ﺗﻨﻬﺎ ﺑﻪ ﺩﻭﺭﯼ ﮐﺮﺩﻥ ﺍﺯ ﺍﻓﺮﺍﺩ ﻣﻨﻔﯽ‌ﻧﮕﺮ ﺧﻼﺻﻪ ﻧﻤﯽ‌ﺷﻮﺩ ﻭ ﺷﻤﺎ ﺑﺎﻳﺪ ﺍﺯ ﺷﺮ ﻭﺳﺎﻳﻠﯽ ﮐﻪ ﺧﺎﻃﺮﺍﺕ ﻳﺎ ﺍﺣﺴﺎﺱ ﻣﻨﻔﯽ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﺩﻫﻨﺪ ﻧﻴﺰ ﺧﻼﺹ ﺷﻮﻳﺪ. ﺍﻳﻦ ﻭﺳﺎﻳﻞ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻟﺒﺎﺳﯽ ﺑﺎﺷﺪ ﮐﻪ ﻣﺪﺕ‌ﻫﺎﺳﺖ ﺑﺮﺍﯼ ﺷﻤﺎ ﺍﻧﺪﺍﺯﻩ ﻧﻴﺴﺖ.
 
ﺩﺭ ﻓﻠﺴﻔﻪ ﻓﻨﮓ ﺷﻮﻳﯽ، ﮐﻪ ﻳﮏ ﻓﻠﺴﻔﻪ ﺑﺎﺳﺘﺎﻧﯽ ﭼﻴﻨﯽ ﺳﺖ ﻭ ﺑﻪ ﺩﻧﺒﺎﻝ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻭ ﻧﻬﺎﺩﻳﻨﻪ ﮐﺮﺩﻥ ﺍﻧﺮﮊﯼ ﻣﺜﺒﺖ ﺩﺭ ﻣﺤﻴﻂ ﺍﺳﺖ، ﻧﻴﺰ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﻴﺎﻥ ﺷﺪﻩ ﺍﺳﺖ. ﺩﺭ ﻓﻨﮓ ﺷﻮﻳﯽ ﺣﺘﯽ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﺷﻮﺩ ﭘﺎ ﺭﺍ ﺍﺯ ﺍﻳﻦ ﻓﺮﺍﺗﺮ ﮔﺬﺍﺷﺘﻪ ﻭ ﻇﺮﻑ‌ﻫﺎﯼ ﻟﭗ ﭘﺮﻳﺪﻩ، ﻭﺳﺎﻳﻞ ﺧﺮﺍﺏ ﻭ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﺑﻴﺮﻭﻥ ﺍﺯ ﻣﺤﻞ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﻣﻨﺘﻘﻞ ﮐﻨﻴﺪ ﺗﺎ ﺍﻧﺮﮊﯼ ﻣﺜﺒﺖ ﺩﺭ ﻣﺤﻴﻂ ﺍﻃﺮﺍﻑ ﺷﻤﺎ ﺑﻮﺟﻮﺩ ﺁﻳﺪ.
 
ﭘﺲ ﺍﺯ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﻣﺮﺍﺣﻞ، ﺩﻳﮕﺮ ﻧﻮﺑﺖ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺍﺳﺘﻌﺪﺍﺩﻫﺎﯼ ﺧﻮﺩ ﺑﭙﺮﺩﺍﺯﻳﺪ ﻭ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺷﻨﺎﺳﺎﻳﯽ ﮐﻨﻴﺪ. ﻫﻤﻴﺸﻪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﺮ ﺁﺩﻣﯽ ﺩﺳﺖ ﮐﻢ ﺩﺭ ﻳﮏ ﺯﻣﻴﻨﻪ ﺍﺳﺘﻌﺪﺍﺩ ﺩﺍﺭﺩ. ﭘﺲ ﺍﺯ ﺷﻨﺎﺧﺘﻦ ﺍﻳﻦ ﺍﺳﺘﻌﺪﺍﺩ ﺑﺎﻳﺪ ﺑﺮ ﺭﻭﯼ ﺁﻥ ﺗﻤﺮﮐﺰ ﮐﺮﺩﻩ ﻭ ﺁﻥ ﺭﺍ ﮔﺴﺘﺮﺵ ﺩﻫﻴﺪ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﻣﻮﺳﻴﻘﯽ، ﻧﻘﺎﺷﯽ، ﻧﻮﺷﺘﻦ ﻳﺎ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﺍﺳﺘﻌﺪﺍﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻳﺎ ﺩﻳﮕﺮﺍﻥ ﺑﻪ ﺷﻤﺎ ﮔﻔﺘﻪ ﺑﺎﺷﻨﺪ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ‌ﻫﺎ ﺧﻮﺏ ﻫﺴﺘﻴﺪ، ﭘﺲ ﺑﺎﻳﺪ ﺑﻪ ﺩﻧﺒﺎﻝ ﺍﻓﺰﺍﻳﺶ ﺗﻮﺍﻧﺎﻳﯽ‌ﻫﺎﯼ ﺧﻮﺩ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ‌ﻫﺎ ﺑﺎﺷﻴﺪ. ﺍﮔﺮ ﮐﻤﯽ ﺷﺎﻧﺲ ﭼﺎﺷﻨﯽ ﮐﺎﺭ ﺷﻮﺩ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﻭﺳﺘﺎﻧﯽ ﭘﻴﺪﺍ ﮐﻨﻴﺪ ﮐﻪ ﺩﺭ ﺯﻣﻴﻨﻪ‌ﻫﺎﯼ ﺍﺳﺘﻌﺪﺍﺩ ﺷﻤﺎ ﺑﺎ ﺷﻤﺎ ﮐﺎﺭ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﻫﻢ‌ﮐﻼﻣﯽ ﺑﺎ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺳﺒﺐ ﺗﻘﻮﻳﺖ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺷﻤﺎ ﻧﻴﺰ ﺷﻮﺩ. ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﭘﻴﺮﻭﯼ ﺍﺯ ﺗﻤﺎﻳﻼﺕ ﻭ ﻫﻢ‌ﭼﻨﻴﻦ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎﻳﯽ ﮐﻪ ﺍﺯ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻥ ﺁﻥ‌ﻫﺎ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﻳﺪ، ﺩﺭ ﺷﻤﺎ ﺍﺣﺴﺎﺱ ﻣﻨﺤﺼﺮ ﺑﻪ ﻓﺮﺩ ﺑﻮﺩﻥ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻭﺭﺩ ﻭ ﺯﻣﻴﻨﻪ‌ﺳﺎﺯ ﺩﺍﺷﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﺎﻻﺗﺮ ﻣﯽ‌ﺷﻮﺩ. 
 
ﺩﺍﺷﺘﻦ ﮐﻤﯽ ﻏﺮﻭﺭ ﻫﻢ  ﺍﻟﺒﺘﻪ ﺑﺪ ﻧﻴﺴﺖ! ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﻗﻊ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﻪ ﮐﻤﮑﺘﺎﻥ ﻣﯽ‌ﺁﻳﺪ ﺍﻣﺎ ﺩﺭ ﺍﻳﻦ ﻣﻮﺭﺩ ﺯﻳﺎﺩﻩ ﺭﻭﯼ ﻧﮑﻨﻴﺪ ﮐﻪ ﻧﺘﻴﺠﻪ ﻋﮑﺲ ﻣﯽ‌ﺩﻫﺪ. ﺗﻤﺮﻳﻨﺎﺕ ﺳﺎﺩﻩ ﺩﻳﮕﺮﯼ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺳﺎﺧﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺷﻤﺎ ﻣﻨﺠﺮ ﺷﻮﺩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻧﮕﺎﻩ ﮐﺮﺩﻥ ﺑﻪ ﺁﻳﻨﻪ ﻭ ﻟﺒﺨﻨﺪ ﺯﺩﻥ ﺑﺎﺷﺪ! ﺩﻳﺪﻥ ﺍﻳﻦ ﻟﺒﺨﻨﺪﻫﺎﯼ ﺳﺎﺩﻩ‌ﯼ ﺧﻮﺩﺗﺎﻥ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﮐﻨﺪ ﺗﺎ ﺩﻧﻴﺎ ﺭﺍ ﺷﺎﺩﺗﺮ ﺍﺣﺴﺎﺱ ﮐﻨﻴﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺍﺯ ﻟﺤﺎﻅ ﻋﻠﻤﯽ ﺩﺭ ﻧﻈﺮﻳﻪ ﺑﺎﺯﺧﻮﺭﺩ ﺻﻮﺭﺕ (facial feedback theory) ﺛﺎﺑﺖ ﺷﺪﻩ ﺍﺳﺖ ﮐﻪ ﺣﺎﻻﺕ ﺻﻮﺭﺕ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻐﺰﺗﺎﻥ ﺭﺍ ﺩﺭ ﺛﺒﺖ ﻳﺎ ﺗﺸﺪﻳﺪ ﺍﺣﺴﺎﺳﺎﺕ ﺧﺎﺹ ﺗﺸﻮﻳﻖ ﮐﻨﺪ. ﭘﺲ ﺍﮔﺮ ﺑﻪ ﺩﻧﺒﺎﻝ ﺳﺎﺧﺖ ﻳﺎ ﺍﺭﺗﻘﺎﯼ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺧﻮﺩﺗﺎﻥ ﻫﺴﺘﻴﺪ، ﺩﻳﺪﻥ ﻟﺒﺨﻨﺪﺗﺎﻥ ﺩﺭ ﺁﻳﻨﻪ ﺭﺍ ﻧﻴﺰ ﻫﻴﭽﮕﺎﻩ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ.
 
ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ، ﺩﺭ ﺟﻠﺴﻪ ﺑﻌﺪ ﺩﺭﺑﺎﺭﻩ ﺑﺮﻗﺮﺍﺭﯼ ﺍﺭﺗﺒﺎﻁ ﺷﺨﺺ ﺑﺎ ﺍﺣﺴﺎﺳﺎﺗﺶ ﻭ ﻣﺮﺗﺒﻂ ﺑﻮﺩﻥ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﻴﺸﺘﺮ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.
