ﺩﺭ ﺩﺭﺱ ﻗﺒﻞ ﻳﺎﺩ ﮔﺮﻓﺘﻴﻢ ﮐﻪ ﭼﮕﻮﻧﻪ ﻧﮕﺮﺵ ﻣﺜﺒﺖ ﺑﻪ ﺭﻭﻳﺪﺍﺩﻫﺎ ﻭ ﻣﺤﻴﻂ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺍﻳﺠﺎﺩ ﻭ ﺍﻓﺰﺍﻳﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﮐﻤﮏ ﮐﻨﺪ. ﻫﻤﭽﻨﻴﻦ ﮔﻔﺘﻴﻢ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ﺑﺎﻳﺪ ﺍﺯ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺁﺯﺍﺭ ﻣﯽ‌ﺩﻫﺪ ﺩﻭﺭﯼ ﮐﻨﻴﺪ ﻭ ﻭﻗﺖ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺁﺩﻡ‌ﻫﺎﻳﯽ ﺑﮕﺬﺭﺍﻧﻴﺪ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ. ﻫﻤﭽﻨﻴﻦ ﮔﻔﺘﻴﻢ ﮐﻪ ﺑﻪ ﺍﺳﺘﻌﺪﺍﺩﻫﺎﯼ ﺧﻮﺩ ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﻭ ﺑﺎ ﺗﻘﻮﻳﺖ ﺁﻥ‌ﻫﺎ ﻭ ﺳﺎﺧﺘﻦ ﺩﻧﻴﺎﯼ ﺟﺪﻳﺪ، ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺧﻮﺩ ﺭﺍ ﺗﻘﻮﻳﺖ ﮐﻨﻴﺪ.
 
ﺍﻣﺮﻭﺯ ﻣﯽ ﺧﻮﺍﻫﻴﻢ ﭘﺎ ﺭﺍ ﮐﻤﯽ ﻓﺮﺍﺗﺮ ﺑﮕﺬﺍﺭﻳﻢ ﻭ ﺍﺯ ﺍﺭﺗﺒﺎﻁ ﺍﺣﺴﺎﺳﺎﺕ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺻﺤﺒﺖ ﮐﻨﻴﻢ. ﺷﺎﻳﺪ ﺷﻤﺎ ﻫﻤﻴﺸﻪ ﺑﺎ ﺧﻮﺩ ﻓﮑﺮ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺍﺯ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﮐﺎﻓﯽ ﺑﺮﺧﻮﺭﺩﺍﺭﻧﺪ ﻫﻴﭽﮕﺎﻩ ﺩﭼﺎﺭ ﺗﺮﺱ ﻧﻤﯽ‌ﺷﻮﻧﺪ ﻭ ﻳﺎ ﻧﺎﺭﺍﺣﺘﯽ‌ﻫﺎﻳﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﻧﺪﺍﺭﻧﺪ. ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺍﻳﻦ ﺗﻔﮑﺮ ﮐﺎﻣﻼ ﺍﺷﺘﺒﺎﻩ ﺍﺳﺖ ﻭ ﺩﺍﺷﺘﻦ ﺗﺮﺱ ﺩﺭ ﻳﮏ ﻣﻮﺿﻮﻉ ﺍﺭﺗﺒﺎﻃﯽ ﺑﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﺎﻻ ﻳﺎ ﭘﺎﻳﻴﻦ ﻧﺪﺍﺭﺩ. ﺷﺎﻳﺪ ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﻳﻨﮕﻮﻧﻪ ﺑﮕﻮﻳﻴﻢ ﮐﻪ ﺍﺭﺗﺒﺎﻁ ﺁﻥ‌ﻫﺎ ﺑﻪ ﺍﻳﻦ ﺻﻮﺭﺕ ﺍﺳﺖ ﮐﻪ ﺁﺩﻡ‌ﻫﺎﻳﯽ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﺎﻻ ﺍﺟﺎﺯﻩ ﻧﻤﯽ‌ﺩﻫﻨﺪ ﺗﺮﺱ ﺩﺭ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎ ﺑﺮ ﺁﻥ‌ﻫﺎ ﻏﻠﺒﻪ ﮐﻨﺪ. ﭘﺲ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﺎ ﺍﺣﺴﺎﺱ ﺗﺮﺱ ﺭﺍﺣﺖ ﺑﺎﺷﻴﺪ!
 
ﺗﺮﺱ ﺭﺍ ﺑﺸﻨﺎﺳﻴﺪ
 
ﻫﺮ ﻭﻗﺖ ﺩﺭ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﺗﺮﺱ ﺭﺍ ﺍﺣﺴﺎﺱ ﮐﺮﺩﻳﺪ، ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺑﭙﺮﺳﻴﺪ ﮐﻪ ﺑﻪ ﺗﺮﺱ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻴﺪ ﻣﺎﻧﻊ ﺭﺳﻴﺪﻥ ﺷﻤﺎ ﺑﻪ ﻫﺪﻓﺘﺎﻥ ﺷﻮﺩ ﻳﺎ ﻧﻪ؟ ﻫﺮﮔﺎﻩ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺷﻤﺎ ﺍﻓﺰﺍﻳﺶ ﻳﺎﺑﺪ، ﮐﻤﺘﺮ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﻴﺪ ﮐﻪ ﺗﺮﺱ ﻣﺎﻧﻊ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎﻳﺘﺎﻥ ﺷﻮﺩ ﻳﺎ ﺷﻤﺎ ﺭﺍ ﻓﻠﺞ ﮐﻨﺪ. 
 
ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻭﺿﻌﻴﺘﯽ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ ﮐﻪ ﺗﺮﺱ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺎﻧﻌﯽ ﺷﻮﺩ ﺗﺎ ﺍﺯ ﺭﺋﻴﺴﺘﺎﻥ ﺩﺭﺑﺎﺭﻩ ﺍﺭﺗﻘﺎﯼ ﺷﻐﻠﯽ ﺧﻮﺩ ﺳﻮﺍﻝ ﺑﭙﺮﺳﻴﺪ؛ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺷﻤﺎ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﻪ ﺍﻳﻦ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﭼﺮﺍ ﺍﺣﺴﺎﺱ ﺗﺮﺱ ﻣﯽ‌ﮐﻨﻴﺪ؟ ﺁﻳﺎ ﺑﺎ ﭘﺮﺳﻴﺪﻥ ﺍﻳﻦ ﺳﻮﺍﻝ ﺍﺯ ﮐﺎﺭ ﺑﺮﮐﻨﺎﺭ ﻣﯽ‌ﺷﻮﻳﺪ؟ ﻳﺎ ﺑﺎ ﭘﺮﺳﻴﺪﻥ ﺍﻳﻦ ﺳﻮﺍﻝ ﻣﻮﺍﺭﺩﯼ ﻫﻤﭽﻮﻥ ﻋﻤﻠﮑﺮﺩ ﺿﻌﻴﻒ ﺷﻤﺎ ﭘﻴﺶ ﺭﺋﻴﺴﺘﺎﻥ ﻣﻄﺮﺡ ﻣﯽ‌ﺷﻮﺩ؟
 
ﺳﭙﺲ ﺑﻪ ﺁﻥ ﭼﻴﺰﻫﺎﻳﯽ ﻓﮑﺮ ﮐﻨﻴﺪ ﮐﻪ ﺑﺎ ﭘﺮﺳﻴﺪﻥ ﺳﻮﺍﻝ ﺑﻪ ﺩﺳﺖ ﻣﯽ‌ﺁﻭﺭﻳﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﺗﺮﻓﻴﻊ ﺑﮕﻴﺮﻳﺪ ﻭ ﭘﻮﻝ ﺑﻴﺸﺘﺮﯼ ﺑﻪ ﺩﺳﺖ ﺑﻴﺎﻭﺭﻳﺪ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﻪ ﺷﻤﺎ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﺪ ﺯﻧﺪﮔﯽ ﺭﺍﺣﺖ‌ﺗﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﻳﺎ ﺣﺘﯽ ﺩﺳﺘﮑﻢ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺣﺴﺎﺱ ﺍﻃﻤﻴﻨﺎﻥ ﮐﻨﻴﺪ ﮐﻪ ﻭﻇﺎﻳﻔﺘﺎﻥ ﺭﺍ ﺩﺭ ﮐﺎﺭ ﺑﻪ ﺧﻮﺑﯽ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ ﮐﻪ ﺧﻮﺩ ﺭﺍ ﺷﺎﻳﺴﺘﻪ ﺍﻳﻦ ﺗﺮﻓﻴﻊ ﻣﯽ‌ﺩﺍﻧﻴﺪ.
 
ﭘﺲ ﺍﺯ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﻧﮑﺎﺕ ﻣﻨﻔﯽ ﻭ ﻣﺜﺒﺖ ﺍﻳﻦ ﻣﻮﺿﻮﻉ، ﻭﻗﺖ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﻬﺘﺮﻳﻦ ﻭ ﺑﺪﺗﺮﻳﻦ ﺳﻨﺎﺭﻳﻮﻫﺎﯼ ﻣﻤﮑﻦ ﺭﺍ ﺍﺭﺯﻳﺎﺑﯽ ﮐﻨﻴﺪ. ﺁﻳﺎ ﮐﺎﺭ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻣﯽ‌ﺩﻫﻴﺪ؟ ﺍﻳﻦ ﺑﺪﺗﺮﻳﻦ ﺳﻨﺎﺭﻳﻮ ﻣﻤﮑﻦ ﺍﺳﺖ. ﺁﻳﺎ ﺭﺋﻴﺲ ﺑﻪ ﺩﺭﺧﻮﺍﺳﺖ ﺗﺮﻓﻴﻊ ﺷﻤﺎ ﺟﻮﺍﺏ ﺭﺩ ﻣﯽ‌ﺩﻫﺪ؟ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺗﻔﺎﻕ ﺑﻴﻔﺘﺪ، ﺍﮔﺮ ﻫﻢ ﺟﻮﺍﺏ ﺍﻭ ﻣﻨﻔﯽ ﺑﻮﺩ ﺷﻤﺎ ﻫﻴﭻ ﭼﻴﺰﯼ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻧﻤﯽ‌ﺩﻫﻴﺪ ﻭ ﺍﻟﺒﺘﻪ ﻣﻬﻤﺘﺮ ﺍﺯ ﺁﻥ، ﺑﺮ ﺗﺮﺱ ﭘﺮﺳﻴﺪﻥ ﺍﻳﻦ ﺳﻮﺍﻝ ﻧﻴﺰ ﻏﻠﺒﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ. ﺍﮔﺮ ﭘﺎﺳﺦ ﺭﺋﻴﺲ ﻣﺜﺒﺖ ﺑﺎﺷﺪ ﮐﻪ ﻫﻤﻪ ﭼﻴﺰ ﺑﺮ ﻭﻓﻖ ﻣﺮﺍﺩ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.
 
ﭘﺲ ﻫﺮﮔﺎﻩ ﺩﺭ ﻣﻮﻗﻌﻴﺘﯽ ﻣﺸﺎﺑﻪ ﮔﺮﻓﺘﺎﺭ ﺷﺪﻳﺪ ﻭ ﻗﺪﺭﺕ ﺗﺼﻤﻴﻢ ﮔﻴﺮﯼ ﻧﺪﺍﺷﺘﻴﺪ، ﺑﻪ ﻧﮑﺎﺕ ﻣﺜﺒﺖ ﻭ ﻣﻨﻔﯽ ﻗﻀﻴﻪ ﻓﮑﺮ ﮐﻨﻴﺪ ﻭ ﺑﺎ ﺍﺭﺯﻳﺎﺑﯽ ﻧﺘﺎﻳﺞ ﺁﻥ، ﺗﺼﻤﻴﻢ ﺩﺭﺳﺖ ﺭﺍ ﺍﺗﺨﺎﺫ ﮐﻨﻴﺪ.
 
ﺑﺎ ﺧﻮﺩﺗﺎﻥ ﻣﻬﺮﺑﺎﻥ ﺑﺎﺷﻴﺪ
 
ﻫﻴﭻ‌ﮔﺎﻩ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﻗﻊ ﺩﺭ ﺯﻧﺪﮔﻴﺘﺎﻥ ﺑﺎﻳﺪ ﺑﺎ ﺧﻮﺩﺗﺎﻥ ﻣﻬﺮﺑﺎﻥ ﺑﺎﺷﻴﺪ! ﭘﻴﺸﺮﻓﺖ‌ ﻫﻤﻮﺍﺭﻩ ﺍﺗﻔﺎﻕ ﻧﻤﯽ‌ﺍﻓﺘﻨﺪ ﻭ ﺑﺮﺧﯽ ﻣﻮﺍﻗﻊ ﺷﻤﺎ ﺷﮑﺴﺖ ﻣﯽ‌ﺧﻮﺭﻳﺪ، ﺍﻣﺎ ﺍﺟﺎﺯﻩ ﻧﺪﻫﻴﺪ ﺍﻳﻦ ﺷﮑﺴﺖ ﺳﺒﺐ ﮐﺎﻫﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺷﻤﺎ ﺷﻮﺩ. ﻫﺮﮔﺎﻩ ﮐﻪ ﺷﮑﺴﺖ ﺧﻮﺭﺩﻳﺪ ﻳﻌﻨﯽ ﭼﻴﺰﯼ ﺑﺮﺍﯼ ﻳﺎﺩﮔﻴﺮﯼ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺷﻤﺎ ﺁﻥ ﻣﻮﺿﻮﻉ ﺭﺍ ﭘﻴﺶ ﺍﺯ ﺍﻳﻦ ﺑﻪ ﺣﺴﺎﺏ ﻧﻴﺎﻭﺭﺩﻩ ﺑﻮﺩﻳﺪ. ﭘﺲ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺧﻮﺩ ﺭﺍ ﺍﺯ ﺩﺳﺖ ﻧﺪﻫﻴﺪ ﻭ ﺍﺟﺎﺯﻩ ﺩﻫﻴﺪ ﮐﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺴﺘﺎﻥ ﺑﻪ ﺻﻮﺭﺕ ﻃﺒﻴﻌﯽ ﺩﺭ ﻣﺮﺍﺣﻞ ﺯﻧﺪﮔﯽ ﺭﺷﺪ ﮐﻨﺪ.
 
ﺗﻌﺎﺩﻝ ﺩﺭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ
 
ﻫﻤﺎﻧﻨﺪ ﻫﺮ ﭼﻴﺰ ﺩﻳﮕﺮﯼ ﺩﺭ ﺯﻧﺪﮔﯽ، ﺳﺎﺧﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻧﻴﺰ ﻧﻴﺎﺯ ﺑﻪ ﺭﻋﺎﻳﺖ ﺗﻌﺎﺩﻝ ﺩﺍﺭﺩ. ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﮐﻢ ﺗﻘﺮﻳﺒﺂ ﻫﻤﻴﺸﻪ ﻣﻨﺠﺮ ﺑﻪ ﺍﻳﻦ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺷﻤﺎ ﻧﺘﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻫﺪﺍﻑ ﺧﻮﺩ ﺑﺮﺳﻴﺪ ﻭ ﻳﺎ ﺍﺣﺴﺎﺱ ﺧﻮﺑﯽ ﺩﺭ ﺭﺍﻩ ﺭﺳﻴﺪﻥ ﺑﻪ ﻫﺪﻑ ﺧﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﺗﻘﺮﻳﺒﺂ ﻫﻤﺎﻥ ﻧﺘﻴﺠﻪ‌ﺍﯼ‌ ﺍﺳﺖ ﮐﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺯﻳﺎﺩ ﺍﺯ ﺧﻮﺩ ﺑﻪ ﺟﺎﯼ ﻣﯽ‌ﮔﺬﺍﺭﺩ ﭼﺮﺍ ﮐﻪ ﺍﻓﺮﺍﺩ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﺎﻻ ﻫﻤﻮﺍﺭﻩ ﻧﻘﺶ ﺗﻼﺵ ﮐﺮﺩﻥ ﻭ ﺯﻣﺎﻥ ﻣﻨﺎﺳﺐ ﺗﻼﺵ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﻫﺪﻑ ﺭﺍ ﺩﺳﺖ ﮐﻢ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﻭ ﺩﺭ ﺭﺳﻴﺪﻥ ﺑﻪ ﺁﻥ ﻧﺎﮐﺎﻡ ﻣﯽ‌ﻣﺎﻧﻨﺪ.

ﺩﺭ ﺟﻠﺴﻪ ﻗﺒﻞ ﮔﻔﺘﻴﻢ ﺑﺎﻳﺪ ﺍﺯ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﻣﻘﺎﻳﺴﻪ ﻣﯽ‌ﮐﻨﻨﺪ، ﺩﻭﺭﯼ ﮐﻨﻴﺪ. ﺍﻳﻨﺠﺎ ﺑﺎﻳﺪ ﺑﮕﻮﻳﻴﻢ ﺍﻳﻦ ﻣﻘﺎﻳﺴﻪ ﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺫﻫﻦ ﺧﻮﺩ ﺷﻤﺎ ﻧﻴﺰ ﺑﻪ ﻭﺟﻮﺩ ﺑﻴﺎﻳﺪ ﮐﻪ ﺑﺎﺯ ﻫﻢ ﺷﻤﺎ ﺑﺎﻳﺪ ﺍﺯ ﺍﻧﺠﺎﻡ ﺁﻥ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ ﭼﺮﺍ ﮐﻪ ﻫﻤﻮﺍﺭﻩ ﺍﻓﺮﺍﺩﯼ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﮐﻪ ﺩﺭ ﺯﻣﻴﻨﻪ‌ﺍﯼ ﺧﺎﺹ ﺑﻬﺘﺮ، ﺑﺎﻫﻮﺵ‌ﺗﺮ ﻳﺎ ﭘﻮﻝ‌ﺩﺍﺭﺗﺮ ﺍﺯ ﺷﻤﺎ ﺑﺎﺷﻨﺪ؛ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻫﻤﻮﺍﺭﻩ ﺍﻓﺮﺍﺩﯼ ﭘﺎﻳﻴﻦ‌ﺗﺮ ﺍﺯ ﺷﻤﺎ ﻧﻴﺰ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ‌ﻫﺎ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ.
 
ﭘﮋﻭﻫﺶ‌ﻫﺎﯼ ﺍﺧﻴﺮ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﮔﺬﺭﺍﻥ ﺯﻣﺎﻥ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻧﻴﺰ ﺍﻓﺮﺍﺩ ﺭﺍ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﻣﻘﺎﻳﺴﻪ ﮐﻨﻨﺪ ﻭ ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﻓﺮﺍﻣﻮﺵ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺩﻭﺳﺘﺎﻥ ﻭ ﺁﺷﻨﺎﻳﺎﻥ ﺁ‌ﻥ‌ﻫﺎ، ﻟﺤﻈﻪ‌ﻫﺎﯼ ﺧﻮﺵ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﻣﯽ‌ﮔﺬﺍﺭﻧﺪ ﻭ ﺍﺯ ﺩﻳﮕﺮ ﻟﺤﻈﻪ‌ﻫﺎﯼ ﺧﻮﺩ ﭼﻴﺰﯼ ﻣﻨﺘﺸﺮ ﻧﻤﯽ‌ﮐﻨﻨﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺁﺳﺎﻧﯽ ﺯﻣﻴﻨﻪ ﺳﺎﺯ ﮐﺎﻫﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺍﻓﺮﺍﺩ ﺷﻮﺩ.
 
ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ، ﺩﺭ ﺩﺭﺱ ﺟﻠﺴﻪ ﺁﻳﻨﺪﻩ ﺑﺎﺯ ﻫﻢ ﺩﺭﺑﺎﺭﻩ ﺍﺣﺴﺎﺳﺎﺕ ﺁﺩﻣﯽ ﻭ ﺭﺍﺑﻄﻪ ﺁﻥ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ ﻭ ﺭﺍﻫﮑﺎﺭﻫﺎﻳﯽ ﺑﺮﺍﯼ ﻣﺮﺍﻗﺒﺖ ﺍﺯ ﺧﻮﺩ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﮐﻪ ﺩﺭ ﺑﺎﻻ ﺑﺮﺩﻥ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻣﻮﺛﺮ ﺍﺳﺖ.
