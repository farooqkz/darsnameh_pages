ﺩﺭﺱ ﭼﻬﺎﺭﻡ- ﺗﺎﺛﻴﺮ ﺍﺣﺴﺎﺳﺎﺕ ﻣﺜﺒﺖ ﺩﺭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ

ﺩﺭ ﺟﻠﺴﻪ ﮔﺬﺷﺘﻪ ﺁﻣﻮﺧﺘﻴﻢ ﮐﻪ ﺗﺮﺱ ﺩﺭ ﻧﻬﺎﺩ ﻫﺮ ﺁﺩﻣﯽ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻭ ﺣﺘﯽ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ‌ﺗﺮﻳﻦ ﺁﺩﻡ‌ﻫﺎ ﻧﻴﺰ ﺗﺮﺱ ﺭﺍ ﺩﺭ ﻣﻮﺍﺭﺩ ﮔﻮﻧﺎﮔﻮﻥ ﺗﺠﺮﺑﻪ ﻣﯽ‌ﮐﻨﻨﺪ. ﺍﻣﺎ ﺗﻔﺎﻭﺕ ﺁﻥ‌ﻫﺎ ﺩﺭ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺍﺟﺎﺯﻩ ﻧﻤﯽ‌ﺩﻫﻨﺪ ﺗﺮﺱ ﻣﺎﻧﻊ ﺍﺯ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻫﺪﺍﻑ‌ﺷﺎﻥ ﺷﻮﺩ. ﻫﻤﭽﻨﻴﻦ ﺁﻣﻮﺧﺘﻴﻢ ﮐﻪ ﺑﺎﻳﺪ ﺷﮑﺴﺖ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺟﺰﻳﯽ ﺍﺯ ﺯﻧﺪﮔﯽ ﭘﺬﻳﺮﻓﺖ ﻭ ﺩﺭ ﺩﺍﺷﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻧﻴﺰ ﺗﻌﺎﺩﻝ ﺭﺍ ﺭﻋﺎﻳﺖ ﮐﺮﺩ ﭼﺮﺍ ﮐﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﺎﻻ ﻫﻤﺎﻧﻨﺪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﭘﺎﻳﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﻀﺮ ﺑﺎﺷﺪ. ﺩﺭ ﺩﺭﺱ ﮔﺬﺷﺘﻪ ﻫﻤﭽﻨﻴﻦ ﮔﻔﺘﻴﻢ ﮐﻪ ﺍﺯ ﺍﺣﺴﺎﺱ ﻣﻘﺎﻳﺴﻪ ﺧﻮﺩﺗﺎﻥ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺑﺎﻳﺪ ﺩﻭﺭﯼ ﮐﻨﻴﺪ ﭼﺮﺍ ﮐﻪ ﺷﻤﺎ ﺍﺯ ﻭﺍﻗﻌﻴﺖ ﺯﻧﺪﮔﯽ ﺩﻳﮕﺮﺍﻥ ﺍﻃﻼﻉ ﻧﺪﺍﺭﻳﺪ ﻭ ﺍﺣﺘﻤﺎﻻ ﺗﻨﻬﺎ ﻟﺤﻈﺎﺕ ﺷﺎﺩ ﺯﻧﺪﮔﯽ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺑﺎ ﺗﻤﺎﻣﯽ ﻟﺤﻈﺎﺕ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﻣﻘﺎﻳﺴﻪ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﻣﻘﺎﻳﺴﻪ‌ﺍﯼ ﻧﺎﺑﻪ‌ﺟﺎﺳﺖ.
 
ﺩﺭ ﺍﺩﺍﻣﻪ، ﺑﺎﺯ ﻫﻢ ﺍﺯ ﺍﺣﺴﺎﺳﺎﺕ ﺩﺭﻭﻧﯽ ﻭ ﺍﺭﺗﺒﺎﻁ ﺁﻥ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺍﺯ ﺍﺭﺗﺒﺎﻁ ﻣﻴﺎﻥ ﺍﻳﻦ ﺩﻭ ﺑﻴﺸﺘﺮ ﺑﺮﺍﻳﺘﺎﻥ ﻣﯽ‌ﮔﻮﻳﻴﻢ.
 
ﺁﺷﻔﺘﮕﯽ‌ﻫﺎﯼ ﺫﻫﻨﯽ ﺧﻮﺩ ﺭﺍ ﺑﺸﻨﺎﺳﻴﺪ
 
ﭼﺮﺍ ﺻﺪﺍﻳﯽ ﮐﻪ ﺩﺭ ﭘﺲ ﺫﻫﻦ ﺷﻤﺎ ﺑﺮﺍﯼ ﻭﺍﻗﻌﻪ‌ﺍﯼ ﺧﺎﺹ ﻗﺮﺍﺭ ﺩﺍﺭﺩ، ﺷﻤﺎ ﺭﺍ ﻧﺎﺭﺍﺣﺖ ﻳﺎ ﺷﺮﻣﮕﻴﻦ ﻣﯽ‌ﮐﻨﺪ؟ ﺍﻳﻦ ﻭﺍﻗﻌﻪ ﻳﺎ ﺻﺪﺍﯼ ﺩﺭﻭﻧﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺮﺑﻮﻁ ﺑﻪ ﺟﻮﺵ ﺻﻮﺭﺕ ﺷﻤﺎ ﺩﺭ ﺩﺑﻴﺮﺳﺘﺎﻥ ﺑﺎﺷﺪ ﮐﻪ ﺳﺒﺐ ﺍﺣﺴﺎﺱ ﺧﺠﺎﻟﺖ ﺩﺭ ﻣﻴﺎﻥ ﺩﻭﺳﺘﺎﻧﺘﺎﻥ ﺷﺪﻩ ﺑﻮﺩ ﻳﺎ ﻣﺮﺑﻮﻁ ﺑﻪ ﭘﻴﺎﻣﺪﻫﺎﯼ ﻳﮏ ﺗﺠﺮﺑﻪ ﻣﻨﻔﯽ. 
 
ﺑﻪ ﺷﻤﺎ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﻴﻢ ﺍﺑﺘﺪﺍ ﻫﺮ ﻭﺍﻗﻌﻪ‌ﺍﯼ ﮐﻪ ﺳﺒﺐ ﻣﯽ‌ﺷﻮﺩ ﺷﻤﺎ ﺍﺣﺴﺎﺱ ﺑﯽ‌ﺍﺭﺯﺵ ﺑﻮﺩﻥ، ﺧﺠﺎﻟﺖ ﻳﺎ  ﺍﺣﺴﺎﺳﺎﺗﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺭﺍ ﺷﻨﺎﺳﺎﻳﯽ ﮐﺮﺩﻩ ﻭ ﺑﺮﺍﯼ ﺁﻥ ﻧﺎﻣﯽ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﺳﭙﺲ ﺍﻳﻦ ﻧﺎﻡ ﺭﺍ ﺑﺮ ﺭﻭﯼ ﻳﮏ ﺗﮑﻪ ﮐﺎﻏﺬ ﺑﻨﻮﻳﺴﻴﺪ ﻭ ﺍﻳﻦ ﺗﮑﻪ‌ﻫﺎﯼ ﮐﺎﻏﺬ ﺭﺍ ﭘﺎﺭﻩ ﮐﺮﺩﻩ ﻳﺎ ﺁﺗﺶ ﺑﺰﻧﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﻳﮏ ﺍﺣﺴﺎﺱ ﺟﺪﻳﺪ ﻭ ﺑﻬﺘﺮ ﻧﺴﺒﺖ ﺑﻪ ﺍﻳﻦ ﻗﻀﺎﻳﺎ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.
 
ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﺗﻤﺮﻳﻦ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﺷﻤﺎ ﭘﺲ ﺍﺯ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﺍﻳﻦ ﻭﻗﺎﻳﻊ ﺩﭼﺎﺭ ﺍﻓﺴﺮﺩﮔﯽ ﺷﻮﻳﺪ ﺑﻠﮑﻪ ﺍﻳﻦ ﺗﻤﺮﻳﻦ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﻪ ﺷﻤﺎ ﺷﻨﺎﺧﺘﯽ ﺍﺯ ﻣﺸﮑﻼﺕ ﺫﻫﻨﯽ‌ﺗﺎﻥ ﺩﺍﺩﻩ ﻭ ﻗﺪﺭﺕ ﻏﻠﺒﻪ ﺑﺮ ﺁﻥ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺷﻤﺎ ﺗﻘﻮﻳﺖ ﮐﻨﺪ.
 
ﻫﻤﻮﺍﺭﻩ ﺑﺎﻳﺪ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻫﻴﭽﮑﺲ ﮐﺎﻣﻞ ﻧﻴﺴﺖ ﻭ ﺑﭙﺬﻳﺮﻳﺪ ﮐﻪ ﺩﺭ ﺑﺮﺧﯽ ﺍﺯ ﻣﻮﺿﻮﻋﺎﺕ ﻧﺴﺒﺖ ﺑﻪ ﺩﻳﮕﺮ ﺍﻓﺮﺍﺩ ﺯﻧﺪﮔﯽ‌ﻣﺎﻥ، ﮐﻤﺒﻮﺩﻫﺎﻳﯽ ﺩﺍﺭﻳﻢ. ﺍﻳﻦ ﺣﻘﻴﻘﺖ ﺭﺍ ﻧﻴﺰ ﺑﺎﻳﺪ ﻫﻤﻮﺍﺭﻩ ﺩﺭ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ ﮐﻪ ﺭﺍﻩ ﺯﻧﺪﮔﯽ ﭘﺮ ﺍﺯ ﭘﺴﺘﯽ‌ﻫﺎ ﻭ ﺑﻠﻨﺪﯼ‌ﻫﺎﺳﺖ ﻭ ﺑﺴﺘﻪ ﺑﻪ ﻣﻮﺿﻮﻋﺎﺗﯽ ﭼﻮﻥ ﺍﻧﺘﺨﺎﺏ ﻣﺴﻴﺮ ﺯﻧﺪﮔﯽ، ﺍﻧﺘﺨﺎﺏ ﺩﻭﺳﺘﺎﻥ ﻭ ﻳﺎ ﺣﺎﻻﺕ ﺭﻭﺣﯽ، ﺁﺷﻔﺘﮕﯽ‌ﻫﺎﻳﯽ ﺭﺍ ﺗﺠﺮﺑﻪ ﻣﯽ‌ﮐﻨﻴﻢ ﻭ ﺍﻳﻦ ﺁﺷﻔﺘﮕﯽ‌ﻫﺎ ﺍﻟﺒﺘﻪ ﭘﺎﻳﺪﺍﺭ ﻧﺒﻮﺩﻩ ﻭ ﻣﯽ‌ﺁﻳﻨﺪ ﻭ ﻣﯽ‌ﺭﻭﻧﺪ.
 
ﭘﺲ ﻫﻤﻪ ﻣﺎ ﺩﺭ ﺯﻧﺪﮔﯽ ﺑﺎﻻ ﻭ ﭘﺎﻳﻴﻦ ﺭﺍ ﺗﺠﺮﺑﻪ ﻣﯽ‌ﮐﻨﻴﻢ، ﺑﻌﻀﯽ ﺍﻭﻗﺎﺕ ﺍﺷﺘﺒﺎﻫﺎﺗﯽ ﺍﺯ ﻣﺎ ﺳﺮ ﻣﯽ‌ﺯﻧﺪ ﮐﻪ ﺑﻬﺘﺮﻳﻦ ﮐﺎﺭ ﺩﺭ ﺍﻳﻦ ﻫﻨﮕﺎﻡ ﻣﻌﺬﺭﺕ ﺧﻮﺍﻫﯽ ﺍﺳﺖ ﻭ ﺍﻟﺒﺘﻪ ﺳﻌﯽ ﮐﻨﻴﺪ ﺩﺭ ﺁﻳﻨﺪﻩ ﺍﻳﻦ ﺍﺷﺘﺒﺎﻫﺎﺕ ﺭﺍ ﺗﮑﺮﺍﺭ ﻧﮑﻨﻴﺪ. 
 
 
ﺍﺯ ﺍﻳﺪﻩ ﺁﻟﻴﺴﻢ ﺩﻭﺭﯼ ﮐﻨﻴﺪ
 
ﺍﻳﺪﻩ ﺁﻟﻴﺴﻢ ﺍﻏﻠﺐ ﺳﺒﺐ ﻣﯽ‌ﺷﻮﺩ ﺩﺭ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎ ﻧﺎﺗﻮﺍﻥ ﺷﻮﻳﺪ ﭼﺮﺍ ﮐﻪ ﺑﻪ ﺟﺎﯼ ﺑﻪ ﭘﺎﻳﺎﻥ ﺭﺳﺎﻧﺪﻥ ﮐﺎﺭﻫﺎ، ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺁﻥ ﺭﺍ ﺑﯽ‌ﻋﻴﺐ ﻭ ﺗﻤﺎﻡ ﻋﻴﺎﺭ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺳﺒﺐ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺻﺎﺩﻗﺎﻧﻪ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﻳﺎ ﻣﻮﻗﻌﻴﺘﯽ ﮐﻪ ﺩﺭ ﺁﻥ ﻫﺴﺘﻴﺪ ﺧﻮﺷﺤﺎﻝ ﻧﺒﺎﺷﻴﺪ. ﺑﻪ ﺟﺎﯼ ﺍﻳﺪﻩ‌ﺁﻝ ﺑﻮﺩﻥ ﻭ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎ ﻳﻪ ﺻﻮﺭﺕ ﺍﻳﺪﻩ‌ﺁﻝ، ﺭﻭﯼ ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎ ﻭ ﺑﻪ ﭘﺎﻳﺎﻥ ﺭﺳﺎﻧﺪﻥ ﺁﻥ ﺗﻤﺮﮐﺰ ﮐﻨﻴﺪ. ﺍﻧﺠﺎﻡ ﮐﺎﺭﻫﺎ ﻭ ﺑﻪ ﭘﺎﻳﺎﻥ ﺭﺳﺎﻧﺪﻥ ﺁﻥ‌ﻫﺎ ﺧﻮﺩ ﺑﻪ ﺧﻮﺩ ﺣﺲ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺭﺍ ﺩﺭ ﺷﻤﺎ ﺗﻘﻮﻳﺖ ﻣﯽ‌ﮐﻨﺪ.
 
ﺗﺸﮑﺮ ﮐﺮﺩﻥ ﺭﺍ ﺗﻤﺮﻳﻦ ﮐﻨﻴﺪ
 
ﻣﻌﻤﻮﻟﺂ ﺁﺷﻔﺘﻪ ﺑﻮﺩﻥ ﻭ ﮐﻤﺒﻮﺩ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻧﺎﺷﯽ ﺍﺯ ﺍﺣﺴﺎﺱ ﮐﻤﺒﻮﺩ ﭼﻴﺰﯼ ﺍﺳﺖ. ﺍﻳﻦ ﮐﻤﺒﻮﺩ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻣﺎﺩﯼ ﺑﺎﺷﺪ ﻣﺎﻧﻨﺪ ﭼﻬﺮﻩ ﺯﻳﺒﺎ ﻳﺎ ﭘﻮﻝ ﻭ ﻳﺎ ﻣﻌﻨﻮﯼ ﺑﺎﺷﺪ ﻣﺎﻧﻨﺪ ﮐﻤﺒﻮﺩ ﺍﺣﺴﺎﺳﺎﺕ. ﺑﺎ ﺷﮑﺮﮔﺰﺍﺭﯼ ﺑﺎﺑﺖ ﭼﻴﺰﻫﺎﻳﯽ ﮐﻪ ﺩﺍﺭﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺍﺣﺴﺎﺱ ﻧﺎﮐﺎﻓﯽ ﻳﺎ ﻧﺎﺭﺍﺿﯽ ﺑﻮﺩﻥ ﺩﺭ ﺧﻮﺩ ﻣﺒﺎﺭﺯﻩ ﮐﻨﻴﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺳﺒﺐ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﻣﺜﺒﺖ‌ﺗﺮ ﻓﮑﺮ ﮐﻨﻴﺪ ﻭ ﭘﺎﻳﻪ‌ﻫﺎﯼ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺭﺍ ﺩﺭ ﺧﻮﺩ ﺳﺎﺧﺘﻪ ﻭ ﺗﻘﻮﻳﺖ ﮐﻨﻴﺪ.
 
 
ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﮐﺎﻓﯽ ﺍﺳﺖ ﻧﺸﺴﺘﻪ ﻭ ﻳﮏ ﻟﻴﺴﺖ ﺳﭙﺎﺳﮕﺰﺍﺭﯼ ﺗﻬﻴﻪ ﮐﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﻟﻴﺴﺖ ﺗﻤﺎﻣﯽ ﺁﻧﭽﻪ ﺭﺍ ﮐﻪ ﺑﺮﺍﯼ ﺁﻥ ﺑﺎﻳﺪ ﻳﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺷﮑﺮﮔﺰﺍﺭ ﺑﺎﺷﻴﺪ ﺭﺍ ﺑﻨﻮﻳﺴﻴﺪ ﻭ ﻫﻔﺘﻪ‌ﺍﯼ ﻳﮏ ﺑﺎﺭ ﺁﻥ ﺭﺍ ﺑﺨﻮﺍﻧﻴﺪ ﻭ ﻣﻮﺿﻮﻋﺎﺕ ﺟﺪﻳﺪ ﺭﺍ ﺑﻪ ﺁﻥ ﺍﺿﺎﻓﻪ ﮐﻨﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﭼﺎﺭﭼﻮﺏ ﺫﻫﻨﯽ ﻗﺪﺭﺗﻤﻨﺪﯼ ﺑﺮﺍﯼ ﺧﻮﺩ ﺍﻳﺠﺎﺩ ﮐﻨﻴﺪ ﻭ ﺍﻟﺒﺘﻪ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﮔﻔﺘﻴﻢ ﭘﺎﻳﻪ‌ﻫﺎﯼ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺭﺍ ﺩﺭ ﺧﻮﺩ ﺗﻘﻮﻳﺖ ﮐﻨﻴﺪ.
 
ﻋﻼﻭﻩ ﺑﺮ ﺍﺣﺴﺎﺳﺎﺕ ﮐﻪ ﻣﻮﺿﻮﻋﺎﺗﯽ ﻣﻌﻨﻮﯼ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ‌ﺭﻭﻧﺪ، ﺗﻮﺟﻪ ﺑﻪ ﺟﺴﻢ ﻭ ﺑﺪﻥ ﻧﻴﺰ ﺩﺭ ﺩﺍﺷﺘﻦ ﻳﺎ ﻧﺪﺍﺷﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻧﻘﺶ ﻣﻬﻤﯽ ﺩﺍﺭﺩ. ﺷﻤﺎ ﻫﻤﻮﺍﺭﻩ ﺑﺎﻳﺪ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﻣﺮﺍﻗﺒﺖ ﮐﻨﻴﺪ ﺗﺎ ﺩﺭ ﻧﻈﺮ ﺩﻳﮕﺮﺍﻥ ﺁﺩﻣﯽ ﻗﺎﺑﻞ ﺍﺣﺘﺮﺍﻡ  ﺑﻪ ﻧﻈﺮ ﺑﻴﺎﻳﻴﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺭﻋﺎﻳﺖ ﻣﻮﺿﻮﻋﺎﺗﯽ ﭼﻮﻥ ﺣﻤﺎﻡ ﻣﻨﻈﻢ، ﻣﺴﻮﺍﮎ ﺯﺩﻥ ﻭ ﺧﻮﺭﺩﻥ ﻏﺬﺍﻫﺎﯼ ﺳﺎﻟﻢ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻋﺎﺕ ﺷﺎﻳﺪ ﺧﻴﻠﯽ ﭘﻴﺶ‌ﭘﺎ ﺍﻓﺘﺎﺩﻩ ﻭ ﺳﺎﺩﻩ ﺑﻪ ﻧﻈﺮ ﺑﻴﺎﻳﻨﺪ ﺍﻣﺎ ﺑﺎ ﺑﺮﺁﻭﺭﺩﻩ ﮐﺮﺩﻥ ﻧﻴﺎﺯﻳﻬﺎﯼ ﺳﺎﺩﻩ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﭘﻴﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ ﮐﻪ ﺳﺰﺍﻭﺍﺭ ﻳﮏ ﺯﻧﺪﮔﯽ ﺷﺎﻳﺴﺘﻪ ﻫﺴﺘﻴﺪ ﻭ ﻫﻤﻴﻦ ﻣﻮﺿﻮﻉ ﺳﺒﺐ ﻣﯽ‌ﺷﻮﺩ ﺧﻮﺩ ﺭﺍ ﺑﺎﻭﺭ ﮐﻨﻴﺪ ﻭ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. 
 
ﻭﺭﺯﺵ ﮐﺮﺩﻥ ﻧﻴﺰ ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺣﻴﻄﻪ ﺗﻮﺟﻪ ﺑﻪ ﺧﻮﺩﺗﺎﻥ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ. ﻣﻨﻈﻮﺭ ﻣﺎ ﺍﺯ ﻭﺭﺯﺵ ﮐﺮﺩﻥ ﺭﮐﺎﺏ ﺯﺩﻥ ﻳﺎ ﭘﺎﺭﻭ ﺯﺩﻥ ﺑﻪ ﺻﻮﺭﺕ ﺣﺮﻓﻪ‌ﺍﯼ ﻧﻴﺴﺖ ﺑﻠﮑﻪ ﻳﮏ ﻗﺪﻡ ﺯﺩﻥ ﺳﺮﻳﻊ ﺑﻪ ﺻﻮﺭﺕ ﺭﻭﺯﺍﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﺷﻤﺎ ﻣﻨﺎﺳﺐ ﺑﺎﺷﺪ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺗﺤﻘﻴﻘﺎﺕ ﺻﻮﺭﺕ ﮔﺮﻓﺘﻪ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﻨﺪ ﮐﻪ ﻭﺭﺯﺵ ﺩﺭ ﺍﻳﻔﺎﯼ ﺫﻫﻨﻴﺘﯽ ﻣﺜﺒﺖ ﺑﺮﺍﯼ ﺯﻧﺪﮔﯽ ﻧﻘﺸﯽ ﺗﺎﺛﻴﺮﮔﺬﺍﺭ ﺩﺍﺭﺩ.
 
ﻋﻼﻭﻩ ﺑﺮ ﻭﺭﺯﺵ، ﺑﺪﻥ ﺷﻤﺎ ﺑﻪ ﺧﻮﺍﺏ ﻣﻨﺎﺳﺐ ﻧﻴﺰ ﻧﻴﺎﺯ ﺩﺍﺭﺩ. ۷ ﺍﻟﯽ ۹ ﺳﺎﻋﺖ ﺧﻮﺍﺏ ﺷﺒﺎﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺷﻤﺎ ﺍﺣﺴﺎﺱ ﺧﻮﺑﯽ ﺑﺪﻫﺪ، ﺍﺯ ﺍﺳﺘﺮﺱ ﺷﻤﺎ ﺑﮑﺎﻫﺪ ﻭ ﺫﻫﻦ‌ﺗﺎﻥ ﺭﺍ ﺁﻣﺎﺩﻩ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﺭﻭﺯﺍﻧﻪ ﮐﻨﺪ. 
 
ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ، ﺩﺭ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺁﻣﻮﺧﺘﻴﻢ ﮐﻪ ﻋﻼﻭﻩ ﺑﺮ ﺍﺣﺴﺎﺳﺎﺕ، ﺭﺳﻴﺪﻥ ﺑﻪ ﺟﺴﻢ ﻭ ﺑﺪﻥ‌ﺗﺎﻥ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺭﺍ ﺩﺭ ﺷﻤﺎ ﺗﻘﻮﻳﺖ ﮐﻨﺪ ﻭ ﺍﺯ ﺍﻳﻦ ﺭﻭ ﺑﺎﻳﺪ ﺑﻪ ﺻﻮﺭﺕ ﻣﻨﻈﻢ ﺧﻮﺍﺏ، ﻭﺭﺯﺵ ﻭ ﻏﺬﺍﻫﺎﯼ ﻣﻨﺎﺳﺐ ﺭﺍ ﺩﺭ ﺑﺮﻧﺎﻣﻪ ﺭﻭﺯﺍﻧﻪ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﺩﺭ ﺟﻠﺴﻪ ﺁﻳﻨﺪﻩ ﻧﻴﺰ ﻣﻮﺿﻮﻋﺎﺕ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﺩﺭﺑﺎﺭﻩ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻣﯽ‌ﮔﻮﻳﻴﻢ ﻭ ﺭﺍﻫﮑﺎﺭﻫﺎﻳﯽ ﺑﺮﺍﯼ ﺩﺍﺷﺘﻦ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﻴﺸﺘﺮ ﺑﻪ ﺷﻤﺎ ﺍﺭﺍﺋﻪ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺩ.
