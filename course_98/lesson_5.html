ﻋﻼﻭﻩ ﺑﺮ ﺗﻮﺟﻪ ﺑﻪ ﺧﻮﺩ ﺩﺭ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩﻥ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ، ﺗﻮﺟﻪ ﺑﻪ ﺍﺭﺗﺒﺎﻃﺎﺕ ﺩﻳﮕﺮﺍﻥ ﻭ ﺣﮏ ﮐﺮﺩﻥ ﭼﻬﺮﻩ‌ﺍﯼ ﻣﺤﺘﺮﻡ ﺩﺭ ﺫﻫﻦ ﺩﻳﮕﺮﺍﻥ ﻧﻴﺰ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺍﻓﺰﺍﻳﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ‌ ﮐﻤﮏ ﮐﻨﺪ. ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﻪ ﺭﻭﺍﺑﻂ ﺍﺟﺘﻤﺎﻋﯽ ﺧﻮﺩ ﺩﻗﺖ ﮐﻨﻴﺪ ﻭ ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﺭﺩ ﺁﻥ ﺭﺍ ﺗﻐﻴﻴﺮ ﺩﻫﻴﺪ.

ﺯﺑﺎﻥ ﺑﺪﻥ (Body language) ﺑﻪ ﺍﺭﺗﺒﺎﻃﯽ ﻏﻴﺮﮐﻼﻣﯽ ﮔﻔﺘﻪ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺍﺯ ﺳﻮﯼ ﺑﺪﻥ ﺷﻤﺎ ﺍﻧﺠﺎﻡ ﺷﺪﻩ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺪﻭﻥ ﻧﻴﺎﺯ ﺑﻪ ﺳﺨﻦ ﮔﻔﺘﻦ ﺑﻪ ﺍﺭﺗﺒﺎﻁ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﻣﻨﺠﺮ ﺷﻮﺩ. ﺯﺑﺎﻥ ﺑﺪﻥ ﺍﻟﺒﺘﻪ ﺩﺭ ﻓﺮﻫﻨﮓ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻓﺮﻕ ﻣﯽ‌ﮐﻨﺪ ﻭ ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺯﺑﺎﻥ ﺑﺪﻥ ﺩﺭ ﻓﺮﻫﻨﮓ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻣﻌﺎﻧﯽ ﻣﺘﻔﺎﻭﺕ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.
 
ﺩﺭ ﻭﺍﻗﻊ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻣﺎ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻴﻢ، ﺑﺨﺶ ﮐﻮﭼﮑﯽ ﺍﺯ ﺍﺭﺗﺒﺎﻁ ﺑﺮﻗﺮﺍﺭ ﮐﺮﺩﻥ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﻢ ﻭ ﺑﺨﺶ ﺑﺰﺭﮒ‌ﺗﺮ ﺍﻳﻦ ﺍﺭﺗﺒﺎﻁ ﺑﻪ ﺗﻦ ﺻﺪﺍ، ﺣﺎﻟﺖ ﭼﻬﺮﻩ ﻭ ﺯﺑﺎﻥ ﺑﺪﻥ ﻣﺎ ﻣﺮﺑﻮﻁ ﻣﯽ‌ﺷﻮﺩ. ﻣﺮﺩﻡ ﺑﻪ ﻃﻮﺭ ﻏﺮﻳﺰﯼ ﻣﻌﻤﻮﻻ ﺍﻳﻦ ﻧﺸﺎﻧﻪ‌ﻫﺎ ﺭﺍ ﺗﻔﺴﻴﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﺮﺍﯼ ﺍﻓﺰﺍﻳﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﮐﻤﯽ ﺣﺎﻟﺖ‌ﻫﺎﯼ ﺍﻓﺮﺍﺩ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﺎﻻ ﺭﺍ ﺗﻤﺮﻳﻦ ﮐﻨﻴﺪ. ﺗﺤﻘﻴﻘﺎﺕ ﺟﺪﻳﺪ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﺩﺭ ﻣﻮﻗﻌﻴﺖ ﺿﻌﻒ ﻳﺎ ﻗﺪﺭﺕ ﺑﺮﺍﯼ ﻣﺪﺗﯽ ﮐﻮﺗﺎﻩ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﻳﻦ ﻣﻮﻗﻌﻴﺖ ﺭﺍ ﺑﺮﺍﯼ ﭼﻨﺪﻳﻦ ﺳﺎﻋﺖ ﺣﻔﻆ ﮐﻨﺪ. 
 
ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺩﺭ ﻓﻀﺎﻳﯽ ﺑﺎﺯ ﻗﺪﻡ ﻣﯽ‌ﺯﻧﻴﺪ، ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﻴﺪ، ﺳﺮﺗﺎﻥ ﺭﺍ ﺑﺎﻻ ﺑﮕﻴﺮﻳﺪ، ﺷﺎﻧﻪ‌ﻫﺎﻳﺘﺎﻥ ﺭﺍ ﮐﻤﯽ ﻋﻘﺐ ﺑﺒﺮﻳﺪ ﻭ ﭘﺸﺘﺘﺎﻥ ﺭﺍ ﺻﺎﻑ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ. ﺩﺭ ﻫﻨﮕﺎﻡ ﻗﺪﻡ ﺯﺩﻥ، ﺑﻪ ﺳﺮﻋﺖ ﻳﺎ ﺑﻪ ﺻﻮﺭﺗﯽ ﻋﺼﺒﯽ ﮔﺎﻡ ﺑﺮ ﻧﺪﺍﺭﻳﺪ. ﺑﺎ ﺭﻋﺎﻳﺖ ﺍﻳﻦ ﻣﻮﺿﻮﻋﺎﺕ ﺳﺎﺩﻩ ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ ﮐﻪ ﭼﮕﻮﻧﻪ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺭﻓﺘﺎﺭﺗﺎﻥ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﺩ!
 
ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﺭﺍﻩ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﮐﻨﺪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻭﺍﺿﺢ ﻭ ﺑﻠﻨﺪ ﺻﺤﺒﺖ ﮐﻨﻴﺪ. ﺑﻠﻨﺪ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺍﻟﺒﺘﻪ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﺷﻤﺎ ﺑﺎﻳﺪ ﺩﺍﺩ ﺑﮑﺸﻴﺪ ﺑﻠﮑﻪ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻧﺒﺎﻳﺪ ﺻﺪﺍﯼ ﻃﺒﻴﻌﯽ ﺧﻮﺩ ﺭﺍ ﻣﺤﺪﻭﺩ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺻﺪﺍﯼ ﺷﻤﺎ ﺑﻠﻨﺪ ﺍﺳﺖ ﺑﻪ ﻫﻤﺎﻥ ﺍﻧﺪﺍﺯﻩ ﺻﺪﺍﯼ ﺧﻮﺩ ﺻﺤﺒﺖ ﮐﻨﻴﺪ  ﺍﺯ ﺍﻳﻦ ﺑﺎﺑﺖ ﺷﺮﻣﺴﺎﺭ ﻳﺎ ﺳﺮﺍﻓﮑﻨﺪﻩ ﻧﺒﺎﺷﻴﺪ. ﺑﺮﺍﯼ ﻭﺍﺿﺢ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﻫﻢ ﮐﺎﻓﯽ ﺍﺳﺖ ﮐﻪ ﺗﻨﺪ ﺣﺮﻑ ﻧﺰﻧﻴﺪ ﻭ ﺑﺎ ﺳﺮﻋﺖ ﻫﻤﻴﺸﮕﯽ ﺧﻮﺩ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﻣﮑﺎﻟﻤﻪ ﮐﻨﻴﺪ.
 
ﺍﺯ ﺭﺍﻩ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﺑﺮﺍﯼ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺣﺮﮐﺎﺕ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﻪ ﺣﺎﻟﺖ‌ﻫﺎﯼ ﻧﺸﺴﺘﻦ ﻭ ﺑﺮﺧﺎﺳﺘﻦ ﺍﺷﺎﺭﻩ ﮐﺮﺩ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﺣﺎﻻﺕ ﺷﻤﺎ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﺪﻧﺘﺎﻥ ﺭﺍ ﻭﻝ ﻧﮑﺮﺩﻩ ﻭ ﻫﻤﺎﻥ ﺣﺎﻟﺖ ﻫﻤﻴﺸﮕﯽ ﺁﻥ ﺭﺍ ﺣﻔﻆ ﮐﻨﻴﺪ. ﺷﺎﻧﻪ‌ﻫﺎ، ﺑﺎﺯﻭﻫﺎ ﻳﺎ ﭘﺎﻫﺎﻳﺘﺎﻥ ﺭﺍ ﺑﻪ ﻫﻢ ﻧﺰﺩﻳﮏ ﻧﮑﻨﻴﺪ ﮐﻪ ﺣﺎﻟﺘﯽ ﻗﻮﺯ ﻣﺎﻧﻨﺪ ﺭﺍ ﺗﺠﺮﺑﻪ ﮐﻨﺪ. ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺳﺘﻮﻥ ﻓﻘﺮﺍﺕ ﺻﺎﻑ ﻭ ﺳﺮ ﺭﻭ ﺑﻪ ﺑﺎﻻ ﻭ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﺷﺎﻧﻪ‌ﻫﺎ، ﭘﺎﻫﺎ ﻭ ﺑﺎﺯﻭﻫﺎ ﺩﺭ ﺣﺎﻟﺖ ﻧﺮﻣﺎﻝ ﺑﺪﻥ، ﺑﻬﺘﺮﻳﻦ ﻧﺸﺎﻧﻪ ﺑﺮﺍﯼ ﺩﻳﮕﺮﺍﻥ ﺍﺳﺖ ﮐﻪ ﻣﺘﻮﺟﻪ ﺷﻮﻧﺪ ﺷﻤﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺍﺭﻳﺪ.
 
ﻭ ﺍﻟﺒﺘﻪ ﻳﮑﯽ ﺍﺯ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻧﮑﺎﺕ ﺩﺭ ﺯﺑﺎﻥ ﺑﺪﻥ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺍﺭﺗﺒﺎﻁ ﭼﺸﻤﯽ ﻓﺮﺍﺭ ﻧﮑﻨﻴﺪ. ﺍﺭﺗﺒﺎﻃﺎﺕ ﭼﺸﻤﯽ ﺑﺮﺍﯼ ﻟﺤﻈﺎﺕ ﮐﻮﺗﺎﻫﯽ ﺭﺥ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺍﮔﺮ ﻓﺮﺩﯼ ﺍﺯ ﺍﺭﺗﺒﺎﻃﺎﺕ ﭼﺸﻤﯽ ﺩﺭ ﮔﻔﺖ ﻭ ﮔﻮ ﺑﺎ ﻳﮏ ﻓﺮﺩ ﺩﻳﮕﺮ ﻓﺮﺍﺭ ﮐﻨﺪ ﺍﻳﻦ ﻣﻌﻨﯽ ﺭﺍ ﻣﯽ‌ﺭﺳﺎﻧﺪ ﮐﻪ ﺍﻭ ﺧﺠﺎﻟﺖ ﺯﺩﻩ ﻳﺎ ﺷﺮﻣﺴﺎﺭ ﺍﺳﺖ ﻭ ﻓﺮﺩﯼ ﮐﻪ ﺑﻪ ﭼﺸﻢ‌ﻫﺎﯼ ﺩﻳﮕﺮﯼ ﺯﻝ ﻣﯽ‌ﺯﻧﺪ ﻧﺸﺎﻥ ﺍﺯ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻭﯼ ﺩﺍﺭﺩ. ﺍﻟﺒﺘﻪ ﺑﻬﺘﺮ ﺍﺳﺖ ﻫﻨﮕﺎﻡ ﺍﺭﺗﺒﺎﻁ ﭼﺸﻤﯽ، ﻟﺒﺨﻨﺪ ﺑﺰﻧﻴﺪ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺍﺭﺗﺒﺎﻁ ﺭﺍ ﺑﻬﺘﺮ ﺍﺯ ﭘﻴﺶ ﻣﻤﮑﻦ ﻣﯽ‌ﺳﺎﺯﺩ.

ﻳﮑﯽ ﺍﺯ ﻧﺸﺎﻧﻪ‌ﻫﺎﯼ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﻳﮏ ﺷﺨﺺ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻭﯼ ﺗﻤﺎﻳﻞ ﺩﺍﺭﺩ ﺩﺭ ﻳﮏ ﻭﺿﻌﻴﺖ ﻳﺎ ﺗﺠﺮﺑﻪ ﺟﺪﻳﺪ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺍﻟﺒﺘﻪ ﺑﻪ ﺍﻳﻦ ﻣﻌﻨﯽ ﻧﻴﺴﺖ ﮐﻪ ﻭﯼ ﻫﻤﻮﺍﺭﻩ ﺑﻪ ﺩﻧﺒﺎﻝ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﭘﺮ ﺍﺳﺘﺮﺱ ﺑﺎﺷﺪ ﺑﻠﮑﻪ ﻓﺮﺩ ﺑﺎ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﺪ ﺗﺎ ﺯﻧﺪﮔﯽ ﻣﺴﻴﺮ ﺗﺎﺯﻩ ﻭ ﭘﺮﻫﻴﺠﺎﻥ ﻳﺎ ﮐﻢ‌ﻫﻴﺠﺎﻥ ﺧﻮﺩ ﺭﺍ ﺍﺩﺍﻣﻪ ﺩﻫﺪ ﻭ ﻭﯼ ﺍﺯ ﺗﺠﺮﺑﻪ ﭼﻴﺰﻫﺎﯼ ﺟﺪﻳﺪ ﻟﺬﺕ ﻣﯽ‌ﺑﺮﺩ. ﻫﻤﻮﺍﺭﻩ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻫﺮ ﭼﻴﺰ ﺟﺪﻳﺪﯼ ﮐﻪ ﺷﻤﺎ ﺗﺠﺮﺑﻪ ﻣﯽ‌ﮐﻨﻴﺪ، ﺍﺭﺯﺵ ﺗﺠﺮﺑﻪ ﻣﺠﺪﺩ ﺭﺍ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺷﺖ ﻭ ﻫﺮ ﮐﺴﯽ ﺭﺍ ﮐﻪ ﻣﻼﻗﺎﺕ ﻣﯽ‌ﮐﻨﻴﺪ، ﺷﻤﺎ ﺭﺍ ﻟﺰﻭﻣﺂ ﺩﻭﺳﺖ ﻧﻤﯽ‌ﺩﺍﺭﺩ ﺍﻣﺎ ﺷﻤﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺮﺍﯼ ﺩﻻﻳﻠﯽ ﺍﻳﻦ‌ﭼﻨﻴﻨﯽ ﺍﺯ ﺗﺠﺮﺑﻪ ﮐﺮﺩﻥ ﻣﻮﺿﻮﻋﺎﺕ ﺟﺪﻳﺪ ﺩﺳﺖ ﺑﮑﺸﻴﺪ.
 
ﺍﮔﺮ ﻣﻬﻤﺎﻧﯽ ﺭﻓﺘﻦ ﻳﺎ ﺩﻳﮕﺮ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻣﺎﻧﻨﺪ ﺷﺮﮐﺖ ﺩﺭ ﮐﺎﺭﻫﺎﯼ ﺧﻴﺮﻳﻪ ﺑﺮﺍﯼ ﺷﻤﺎ ﻫﻴﺠﺎﻥ‌ﺍﻧﮕﻴﺰ ﺍﺳﺖ، ﺣﺴﺎﺳﻴﺖ ﺑﻪ ﺧﺮﺝ ﻧﺪﺍﺩﻩ ﻭ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﺎ ﻓﺮﺩﯼ ﮐﻪ ﻗﺒﻼ ﮐﻤﺘﺮ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﺮﺩﻳﺪ ﻳﺎ ﺣﺘﯽ ﺍﻭ ﺭﺍ ﻣﻼﻗﺎﺕ ﻧﮑﺮﺩﻩ ﺑﻮﺩﻳﺪ، ﺻﺤﺒﺖ ﮐﻨﻴﺪ، ﺍﻳﻦ ﮐﺎﺭ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﺑﺪﺗﺮﻳﻦ ﻧﺘﻴﺠﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺍﺯ ﻭﻗﺖ ﮔﺬﺭﺍﻧﺪﻥ ﺑﺎ ﻭﯼ ﻟﺬﺕ ﻧﻤﯽ‌ﺑﺮﻳﺪ ﻭ ﺑﻪ ﺩﻧﺒﺎﻝ ﺁﺩﻡ ﺟﺪﻳﺪﯼ ﻣﯽ‌ﺭﻭﻳﺪ.
 
ﺍﮔﺮ ﺷﻤﺎ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﺩﺭ ﻣﺤﻴﻂ ﭘﻴﺮﺍﻣﻮﻥ ﺧﻮﺩ ﮔﻴﺮ ﺍﻓﺘﺎﺩﻩ‌ﺍﻳﺪ ﻭ ﺩﻳﮕﺮﺍﻥ ﺗﻮﺍﻧﺎﻳﯽ‌ﻫﺎﯼ ﺷﻤﺎ ﺭﺍ ﺩﺳﺖ ﮐﻢ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﻭ ﻗﻀﺎﻭﺕ ﻫﺎﯼ ﺁﻧﺎﻥ ﻧﺴﺒﺖ ﺑﻪ ﺷﻤﺎ ﻧﺎﻋﺎﺩﻻﻧﻪ ﺍﺳﺖ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ ﮐﻪ ﺑﺎﻳﺪ ﺗﺼﻤﻴﻢ ﻣﻬﻤﯽ ﺑﮕﻴﺮﻳﺪ ﻭ ﺑﻪ ﺟﺎﻳﯽ ﺑﺮﻭﻳﺪ ﮐﻪ ﮐﺴﯽ ﺷﻤﺎ ﺭﺍ ﻧﻤﯽ‌ﺷﻨﺎﺳﺪ ﻭ ﺁﻧﺠﺎ ﺗﻮﺍﻧﺎﻳﯽ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺑﺮﻭﺯ ﺩﻫﻴﺪ. ﺍﻳﻦ ﻣﺤﻴﻂ ﺣﺘﻤﺎ ﺗﻐﻴﻴﺮ ﺷﻬﺮ ﻳﺎ ﮐﺸﻮﺭ ﻣﺤﻞ ﺯﻧﺪﮔﯽ ﻧﻴﺴﺖ ﺑﻠﮑﻪ ﺑﻴﺸﺘﺮ ﻣﻮﺍﻗﻊ ﺑﻪ ﻣﻌﻨﯽ ﺗﻐﻴﻴﺮ ﻣﺤﻞ ﮐﺎﺭ، ﺗﻐﻴﻴﺮ ﺑﺮﺧﯽ ﺍﺯ ﺩﻭﺳﺘﺎﻥ ﻳﺎ ﺍﻧﺠﺎﻡ ﺑﺮﺧﯽ ﺍﺯ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﯼ ﺟﺪﻳﺪ ﺍﺳﺖ.
 
ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﺭﺳﻴﺪﻳﻢ. ﻣﺎ ﺩﺭ ﺍﻳﻦ ﻣﻴﻨﯽ‌ﺩﻭﺭﻩ ﺗﻼﺵ ﮐﺮﺩﻳﻢ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﺧﻼﺻﻪ ﺭﺍﻫﮑﺎﺭﻫﺎﻳﯽ ﺭﺍ ﺑﺮﺍﯼ ﺍﻓﺰﺍﻳﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ  ﺍﺭﺍﺋﻪ ﺩﻫﻴﻢ. ﺑﻪ ﮐﺎﺭ ﺑﺮﺩﻥ ﺍﻳﻦ ﺭﺍﻫﮑﺎﺭﻫﺎ ﺩﺭ ﺷﺮﺍﻳﻂ ﻣﺨﺘﻠﻒ ﺑﻪ ﺗﺼﻤﻴﻢ ﺧﻮﺩ ﻓﺮﺩ ﺑﺴﺘﮕﯽ ﺩﺍﺭﺩ ﺍﻣﺎ ﺍﻳﻦ ﺭﺍﻫﮑﺎﺭﻫﺎ ﺍﺻﻮﻟﯽ ﺷﻨﺎﺧﺘﻪ ﺷﺪﻩ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﻪ ﮐﺎﺭﮔﻴﺮﯼ ﺁﻥ‌ﻫﺎ ﻣﻨﺠﺮ ﺑﻪ ﺳﺎﺧﺖ ﻭ ﺍﻓﺰﺍﻳﺶ ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﺩﺭ ﺷﺨﺺ ﺧﻮﺍﻫﻨﺪ ﺷﺪ.
