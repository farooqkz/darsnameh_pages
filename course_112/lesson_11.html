ﺩﺭﺱ ﻫﺸﺘﻢ- ﮐﺎﻫﺶ ﺍﺳﺘﺮﺱ ﺩﺭ ﺭﺍﺑﻄﻪ ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ

ﺑﻪ ﺩﺭﺱ ﻫﺸﺘﻢ ﺍﻳﻦ ﺩﻭﺭﻩ ﺧﻮﺵ ﺁﻣﺪﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﻪ ﺭﺍﻫﮑﺎﺭﻫﺎﻳﯽ ﺑﭙﺮﺩﺍﺯﻳﻢ ﮐﻪ ﺍﺳﺘﺮﺱ ﺭﺍ ﺩﺭ ﺯﻧﺪﮔﯽ ﺍﺟﺘﻤﺎﻋﯽ ﺷﻤﺎ ﮐﺎﻫﺶ ﺩﺍﺩﻩ ﻭ ﺳﺒﺐ ﺑﻬﺒﻮﺩ ﺭﺍﺑﻄﻪ ﺑﺎ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮ ﺧﻮﺍﻫﺪ ﺷﺪ.

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺷﻤﺎ ﺩﭼﺎﺭ ﺍﺳﺘﺮﺱ ﻣﯽ‌ﺷﻮﻳﺪ، ﺗﻮﺍﻧﺎﻳﯽ‌ﺗﺎﻥ ﺩﺭ ﺑﺮﻗﺮﺍﺭﯼ ﺭﺍﺑﻄﻪ ﻣﻨﺎﺳﺐ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﮐﺎﻫﺶ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺳﺒﺐ ﺍﻓﺴﺮﺩﮔﯽ ﺷﻤﺎ ﻧﻴﺰ ﺷﻮﺩ. ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺩﺭ ﺭﻭﺍﺑﻂ ﮐﻤﺘﺮ ﺩﭼﺎﺭ ﺍﺳﺘﺮﺱ ﺷﻮﻳﺪ ﺑﺎﻳﺪ ﭼﻨﺪ ﺍﺻﻞ ﻣﻬﻢ ﺭﺍ ﻫﻤﻮﺍﺭﻩ ﺩﺭ ﺧﺎﻃﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

ﻳﮑﯽ ﺍﺯ ﺍﻳﻦ ﺍﺻﻮﻝ، ﺧﻮﺏ ﮔﻮﺵ ﺩﺍﺩﻥ ﺑﻪ ﺣﺮﻑ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﺍﺳﺖ. ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﺧﻮﺑﯽ ﺑﻪ ﺣﺮﻑ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﮔﻮﺵ ﻣﯽ‌ﺩﻫﻴﺪ، ﺗﻮﺍﻧﺎﻳﯽ ﺍﻧﺠﺎﻡ ﺑﺎﺯﺧﻮﺭﺩ ﻣﻨﺎﺳﺐ ﺑﻪ ﺍﻳﻦ ﺣﺮﻑ‌ﻫﺎ ﺭﺍ ﻧﻴﺰ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﺍﻳﻦ ﺧﻮﺏ ﮔﻮﺵ ﮐﺮﺩﻥ، ﺗﻨﻬﺎ ﺳﺎﮐﺖ ﻧﺸﺴﺘﻦ ﻧﻴﺴﺖ ﺑﻠﮑﻪ ﻋﻮﺍﻣﻠﯽ ﭼﻮﻥ ﺗﻤﺮﮐﺰ ﺑﺮ ﺣﺮﻑ‌ﻫﺎﯼ ﮔﻮﻳﻨﺪﻩ، ﺧﻤﻴﺎﺯﻩ ﻧﮑﺸﻴﺪﻥ، ﻧﮕﺎﻩ ﻧﮑﺮﺩﻥ ﺑﻪ ﺳﺎﻋﺖ ﻭ ﻋﻮﺍﻣﻠﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﺭﺍ ﻧﻴﺰ ﺷﺎﻣﻞ ﻣﯽ‌ﺷﻮﺩ.

ﺩﺭ ﺣﺎﻟﺘﯽ ﮐﻪ ﺷﻤﺎ ﺑﻪ ﺳﺨﻨﺎﻥ ﻃﺮﻑ ﻣﻘﺎﺑﻠﺘﺎﻥ ﮔﻮﺵ ﻣﯽ‌ﺩﻫﻴﺪ، ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﺻﺤﺒﺖ ﻫﺎﯼ ﺍﻭ ﻭ ﻳﺎ ﭘﺮﺳﻴﺪﻥ ﺳﺌﻮﺍﻝ‌ﻫﺎﻳﯽ ﻣﻨﺎﺳﺐ ﺳﺒﺐ ﺑﺮﻗﺮﺍﺭﯼ ﺍﺭﺗﺒﺎﻃﯽ ﺑﻬﺘﺮ ﻣﯽ‌ﺷﻮﺩ ﻭ ﮔﻮﻳﻨﺪﻩ ﺭﺍ ﺗﺸﻮﻳﻖ ﻣﯽ‌ﮐﻨﺪ ﺗﺎ ﺑﻴﺸﺘﺮ ﺑﺎ ﺷﻤﺎ ﺳﺨﻦ ﺑﮕﻮﻳﺪ. ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭﻫﺎ ﺩﺭ ﻫﻨﮕﺎﻡ ﺍﺳﺘﺮﺱ ﻫﻤﺎﻧﻨﺪ ﺗﮑﻨﻴﮏ ﭘﺮﺕ ﮐﺮﺩﻥ ﺣﻮﺍﺱ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺳﺒﺐ ﻓﺮﺍﻣﻮﺵ ﮐﺮﺩﻥ ﺍﺳﺘﺮﺱ ﺷﻮﺩ ﻭ ﮐﻴﻔﻴﺖ ﺭﺍﺑﻄﻪ ﺷﻤﺎ ﺭﺍ ﻧﻴﺰ ﺍﻓﺰﺍﻳﺶ ﺩﻫﺪ.

ﺭﺍﺑﻄﻪ ﻣﻴﺎﻥ ﺍﻓﺮﺍﺩ ﺗﻨﻬﺎ ﺑﻪ ﮔﻮﺵ ﮐﺮﺩﻥ ﺧﻼﺻﻪ ﻧﻤﯽ‌ﺷﻮﺩ ﻭ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﺘﻮﺍﻧﻴﺪ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺻﺤﺒﺖ ﮐﻨﻴﺪ. ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺑﺎﺷﺪ ﮐﻪ ﺑﺮﺍﯼ ﺷﻤﺎ ﻭ ﺩﻳﮕﺮﺍﻥ ﺍﺳﺘﺮﺳﯽ ﺍﻳﺠﺎﺩ ﻧﮑﻨﺪ. ﺗﺼﻮﺭ ﮐﻨﻴﺪ ﮐﻪ ﻣﺸﻐﻮﻝ ﮐﻨﺎﻳﻪ ﺯﺩﻥ ﺑﻪ ﻃﺮﻑ ﻣﻘﺎﺑﻠﺘﺎﻥ ﻫﺴﺘﻴﺪ، ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺳﺒﺐ ﺍﻓﺰﺍﻳﺶ ﻓﺸﺎﺭ ﻭ ﺍﺳﺘﺮﺱ ﺑﺮ ﺍﻭ ﻭ ﭘﺮﺧﺎﺵ ﻭﯼ ﺧﻮﺍﻫﺪ ﺷﺪ؛ ﭘﺲ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﻣﻨﺎﺳﺐ ﻧﻴﺰ ﻳﮑﯽ ﺍﺯ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺳﺒﺐ ﮐﺎﻫﺶ ﺍﺳﺘﺮﺱ ﺑﻴﻦ ﺍﻓﺮﺍﺩ ﻭ ﺑﺮﻗﺮﺍﺭﯼ ﺭﺍﺑﻄﻪ‌ﺍﯼ ﺑﻬﺘﺮ ﺷﻮﺩ.

ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻧﮑﺘﻪ ﺩﺭ ﻫﻨﮕﺎﻡ ﺻﺤﺒﺖ ﮐﺮﺩﻥ، ﮐﻨﺘﺮﻝ ﺍﺣﺴﺎﺳﺎﺕ ﻓﺮﺩﯼ ﺍﺳﺖ. ﻣﻄﻤﺌﻨﺎ ﻣﺎ ﺩﺭ ﻃﻮﻝ ﺑﺮﺧﯽ ﺭﻭﺯﻫﺎ ﺑﺎ ﻣﻮﺿﻮﻋﺎﺗﯽ ﻣﻮﺍﺟﻪ ﻣﯽ‌ﺷﻮﻳﻢ ﮐﻪ ﺳﺒﺐ ﻋﺼﺒﺎﻧﯽ ﺷﺪﻧﻤﺎﻥ ﻣﯽ‌ﺷﻮﺩ ﻭ ﻫﻤﻪ ﻋﻨﺎﺻﺮ ﺁﻣﺎﺩﻩ ﺍﺳﺖ ﺗﺎ ﺑﺎ ﻣﺨﻠﻮﻁ ﮐﺮﺩﻥ ﺁﻥ‌ﻫﺎ ﺑﻪ ﻃﺮﻑ ﻣﻘﺎﺑﻠﻤﺎﻥ ﭘﺮﺧﺎﺵ ﮐﻨﻴﻢ ﺍﻣﺎ ﺑﻬﺘﺮ ﺍﺳﺖ ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﺧﻮﺩﻣﺎﻥ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﻢ ﻭ ﻳﺎ ﺩﺳﺖ ﮐﻢ ﺑﻪ ﻃﺮﻑ ﻣﻘﺎﺑﻠﻤﺎﻥ ﺑﮕﻮﻳﻴﻢ ﻭﻗﺘﯽ ﺗﻮ ﺍﻳﻦ ﺭﻓﺘﺎﺭﻫﺎ ﻳﺎ ﮐﺎﺭﻫﺎ ﺭﺍ ﺑﻪ ﻓﻼﻥ ﺻﻮﺭﺕ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﯽ، ﺳﺒﺐ ﺁﺯﺭﺩﻩ ﺷﺪﻥ/ﻧﺎﺭﺍﺣﺖ ﺷﺪﻥ/ﻋﺼﺒﺎﻧﯽ ﺷﺪﻥ ﻣﻦ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻨﮕﻮﻧﻪ ﺑﻴﺎﻥ ﻋﻮﺍﻃﻒ ﺳﺒﺐ ﺍﻳﺠﺎﺩ ﺻﺤﺒﺖ ﻭ ﮔﻔﺖ ﻭ ﮔﻮ ﻣﻴﺎﻥ ﺩﻭ ﻃﺮﻑ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩ ﺍﺛﺮﺍﺕ ﻣﻨﻔﯽ ﺑﺴﻴﺎﺭ ﮐﻤﯽ ﺩﺭ ﺍﻧﺘﻈﺎﺭ ﺭﺍﺑﻄﻪ ﻣﻴﺎﻥ ﺩﻭ ﻧﻔﺮ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﺣﺘﯽ ﺍﮔﺮ ﺷﻤﺎ ﺍﺯ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺩﻳﮕﺮ ﻋﺼﺒﺎﻧﯽ ﻫﺴﺘﻴﺪ، ﻣﻮﺿﻮﻋﺎﺗﯽ ﮐﻪ ﺍﺭﺗﺒﺎﻃﯽ ﺑﻪ ﻓﺮﺩﯼ ﮐﻪ ﺑﺎ ﺍﻭ ﮔﻔﺖ ﻭ ﮔﻮ ﻣﯽ ﮐﻨﻴﺪ، ﻧﺪﺍﺭﺩ، ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺍﺯ ﻓﻼﻥ ﻣﻮﺿﻮﻉ ﻧﺎﺭﺍﺣﺖ ﻫﺴﺘﻴﺪ ﭼﺮﺍ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺩﻟﻴﻞ ﻧﺎﺭﺍﺣﺘﯽ ﻭ ﻋﺼﺒﺎﻧﯽ ﺑﻮﺩﻥ ﺍﺯ ﻣﻮﺿﻮﻋﯽ ﺩﻳﮕﺮ، ﺭﻓﺘﺎﺭ ﻫﻤﻴﺸﮕﯽ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻭ ﺣﺮﻑ‌ﻫﺎﻳﯽ ﺭﺍ ﺑﺮ ﺯﺑﺎﻥ ﺑﻴﺎﻭﺭﻳﺪ ﮐﻪ ﺑﺎﻻﺗﺮ ﺍﺯ ﺁﺳﺘﺎﻧﻪ ﺗﺤﻤﻞ ﺩﻳﮕﺮﺍﻥ ﺑﺎﺷﺪ.

ﺩﺭ ﻫﻨﮕﺎﻡ ﻋﺼﺒﺎﻧﯽ ﺑﻮﺩﻥ، ﻣﻌﻤﻮﻻ ﺑﺮﭼﺴﺐ ﺯﺩﻥ ﺑﻪ ﺩﻳﮕﺮﺍﻥ ﺑﺎ ﺍﺣﺘﻤﺎﻝ ﺑﻴﺸﺘﺮﯼ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺷﻮﺩ، ﺍﻳﻨﮑﻪ ﺑﻪ ﻳﮏ ﻧﻔﺮ ﺑﮕﻮﻳﻴﺪ ﮐﻪ ﺗﻮ ﺑﻪ ﻓﻼﻥ ﺩﻟﻴﻞ، ﺍﺣﻤﻖ/ﺗﻨﺒﻞ/ﺧﻮﺩﺧﻮﺍﻩ/ﭘﻮﻝ‌ﭘﺮﺳﺖ/ﺷﻠﺨﺘﻪ ﻫﺴﺘﯽ، ﺍﺳﺘﺮﺱ ﻭ ﻓﺸﺎﺭ ﺭﺍ ﺩﺭ ﺍﻭ ﺍﻓﺰﺍﻳﺶ ﺩﺍﺩﻩ‌ﺍﻳﻢ ﻭ ﺍﻭ ﺭﺍ ﻭﺍﺭﺩ ﻓﺎﺯ ﺩﻓﺎﻋﯽ ﮐﺮﺩﻩ‌ﺍﻳﻢ ﻭ ﺍﺣﺘﻤﺎﻻ ﺑﺎﻳﺪ ﻣﻨﺘﻈﺮ ﺑﺎﺯﺧﻮﺭﺩﻫﺎﯼ ﻣﻨﻔﯽ ﺩﻳﮕﺮﯼ ﺍﺯ ﺍﻭ ﺑﺎﺷﻴﻢ.

ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﻣﻮﺍﺭﺩﯼ ﮐﻪ ﺩﺭ ﻣﻴﺎﻥ ﺭﻭﺍﺑﻂ ﺑﻴﻦ ﻓﺮﺩﯼ ﻣﻬﻢ ﺍﺳﺖ، ﺗﻮﺍﻧﺎﻳﯽ ﻧﻪ ﮔﻔﺘﻦ ﻣﺤﺘﺮﻣﺎﻧﻪ ﻭ ﺧﻮﺩﻣﺎﻧﯽ ﺍﺳﺖ، ﻧﻪ ﮔﻔﺘﻨﯽ ﮐﻪ ﺳﺒﺐ ﺭﻧﺠﺶ ﺩﻳﮕﺮﺍﻥ ﻧﺨﻮﺍﻫﺪ ﺷﺪ. ﻣﺜﻼ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻓﺮﺩﯼ ﺍﺯ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﺧﻮﺍﻫﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﺍﻭ ﺑﮕﻮﻳﻴﺪ: «ﺧﻮﺩﺕ ﺍﻧﺠﺎﻡ ﺑﺪﻩ ﺩﺳﺖ ﻭ ﭘﺎ ﭼﻠﻔﺘﯽ» ﻭ ﺑﺎ ﺑﻴﺎﻥ ﺍﻳﻦ ﺟﻤﻠﻪ ﻫﻢ ﺩﺭﺧﻮﺍﺳﺖ ﮐﻤﮏ ﺍﻭ ﺭﺍ ﺭﺩ ﮐﻨﻴﺪ ﻭ ﻫﻢ ﺑﺎ ﺑﺮﭼﺴﺐ ﺯﺩﻥ ﺑﻪ ﺍﻭ ﻓﺸﺎﺭ ﺑﻴﺎﻭﺭﻳﺪ ﺍﻣﺎ ﮔﻔﺘﻦ ﺟﻤﻠﻪ‌ﺍﯼ ﻣﺎﻧﻨﺪ«ﺑﺒﺨﺸﻴﺪ ﻭﻗﺖ ﻧﺪﺍﺭﻡ» ﻳﺎ «ﻣﯽ ﺧﻮﺍﻫﻢ ﮐﻤﮏ ﮐﻨﻢ ﺍﻣﺎ ﺑﻪ ﻓﻼﻥ ﺩﻟﻴﻞ ﻧﻤﯽ‌ﺗﻮﺍﻧﻢ» ﻋﻼﻭﻩ ﺑﺮ ﺍﻳﻨﮑﻪ ﺩﺭﺧﻮﺍﺳﺖ ﮐﻤﮏ ﺍﻭ ﺭﺍ ﺭﺩ ﮐﺮﺩﻩ‌ﺍﻳﺪ، ﺗﻮﺍﻧﺴﺘﻪ‌ﺍﻳﺪ ﺭﺍﺑﻄﻪ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺩﺭ ﻫﻤﺎﻥ ﻭﺿﻌﻴﺖ ﮔﺬﺷﺘﻪ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ.

ﻳﮑﯽ ﺍﺯ ﻣﻮﺿﻮﻋﺎﺗﯽ ﮐﻪ ﺳﺒﺐ ﺍﻳﺠﺎﺩ ﺍﺳﺘﺮﺱ ﺩﺭ ﺭﻭﺍﺑﻂ ﻣﻴﺎﻥ ﺍﻓﺮﺍﺩ ﻣﯽ‌ﺷﻮﺩ، ﮐﻨﺎﺭ ﺁﻣﺪﻥ ﻭ ﺑﺮﺧﻮﺭﺩ ﮐﺮﺩﻥ ﺑﺎ ﺍﻓﺮﺍﺩﯼ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﺁﻥ‌ﻫﺎ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﻣﺸﮑﻞ ﺩﺍﺭﻳﺪ. ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻃﻴﻒ ﮔﺴﺘﺮﺩﻩ‌ﺍﯼ ﺍﺯ ﺁﺩﻡ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺑﺮ ﺑﮕﻴﺮﻧﺪ، ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺭﻭﺍﺑﻂ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﺑﻪ ﺩﺭﺳﺘﯽ ﻧﻤﯽ‌ﺩﺍﻧﻨﺪ، ﻳﺎ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﻋﺼﺒﺎﻧﯽ/ﺑﺪﻭﻥ ﻣﻨﻄﻖ/ﺑﺪﺍﺧﻼﻕ ﻫﺴﺘﻨﺪ ﻭ ﺩﺭ ﺯﻧﺪﮔﯽ ﻫﻤﻪ ﻣﺎ ﺑﻪ ﻧﻮﻋﯽ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ.

ﺩﺭ ﻫﻨﮕﺎﻡ ﺑﺮﺧﻮﺭﺩ ﺑﺎ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﻭ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺩﭼﺎﺭ ﺍﺳﺘﺮﺱ ﺷﺪﻳﺪ، ﺑﻬﺘﺮ ﺍﺳﺖ ﺗﮑﻨﻴﮏ‌ﻫﺎﯼ ﺳﺎﺩﻩ ﺗﻨﻔﺲ ﮐﺸﻴﺪﻥ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ، ﺗﮑﻨﻴﮏ‌ﻫﺎﻳﯽ ﭼﻮﻥ ﺗﻨﻔﺲ ﺩﺭ ﺧﻨﺪﻕ. ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﻫﻤﭽﻨﻴﻦ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻳﮏ ﺗﻨﻔﺲ ﻋﻤﻴﻖ ﺍﺯ ﺭﺍﻩ ﺑﻴﻨﯽ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﻭ ﺑﺎ ﻓﺸﺮﺩﻥ ﺍﻧﮕﺸﺖ ﺷﺴﺖ ﻭ ﺳﺒﺎﺑﻪ ﻳﮏ ﺩﺳﺖ ﺑﻪ ﻫﻢ ﻭ ﻧﮕﺎﻩ ﺩﺍﺷﺘﻦ ﺗﻨﻔﺲ ﺑﺮﺍﯼ ﭼﻬﺎﺭ ﺗﺎ ﭘﻨﺞ ﺛﺎﻧﻴﻪ، ﺁﻥ ﺭﺍ ﺍﺯ ﻃﺮﻳﻖ ﺩﻫﺎﻥ ﺑﻴﺮﻭﻥ ﺩﻫﻴﺪ. ﺍﻳﻦ ﺗﮑﻨﻴﮏ ﻧﻴﺰ ﺑﻪ ﮐﻤﺘﺮ ﺷﺪﻥ ﺍﺳﺘﺮﺱ ﺷﻤﺎ ﮐﻤﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﺭﺩ ﻭﺍﻗﻌﺎ ﻻﺯﻡ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺭﻓﺘﺎﺭﻫﺎﯼ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺑﻪ ﭼﻨﺪ ﺩﺳﺘﻪ ﺗﻘﺴﻴﻢ ﮐﻨﻴﺪ ﺗﺎ ﺑﻬﺘﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺭﺍﺑﻄﻪ ﺭﺍ ﮐﻨﺘﺮﻝ ﻭ ﺑﺎﺯﺧﻮﺭﺩﻫﺎﯼ ﻣﻨﺎﺳﺐ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ. ﻳﮑﯽ ﺍﺯ ﺗﮑﻨﻴﮏ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﺑﻪ ﮐﻤﮏ ﺷﻤﺎ ﺧﻮﺍﻫﺪ ﺁﻣﺪ، ﺗﮑﻨﻴﮏ ﭼﺮﺍﻍ ﺭﺍﻫﻨﻤﺎﺳﺖ. ﺩﺭ ﺍﻳﻦ ﺗﮑﻨﻴﮏ ﺷﻤﺎ ﺭﻓﺘﺎﺭﻫﺎﯼ ﺍﺷﺨﺎﺹ ﺭﺍ ﺑﻪ ﺳﻪ ﺩﺳﺘﻪ ﺗﻘﺴﻴﻢ ﻣﯽ‌ﮐﻨﻴﺪ. ﺩﺳﺘﻪ ﺍﺯ ﺍﻳﻦ ﺭﻓﺘﺎﺭﻫﺎ ﺩﺭ ﭼﺮﺍﻍ ﺳﺒﺰ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﮐﻪ ﻫﻴﭻ ﻣﺸﮑﻠﯽ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺁﻥ‌ﻫﺎ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ. ﺩﺳﺘﻪ‌ﺍﯼ ﺩﻳﮕﺮ ﺩﺭ ﭼﺮﺍﻍ ﺯﺭﺩ ﮐﻪ ﺑﺎﻋﺚ ﺭﻧﺠﺶ ﺧﺎﻃﺮ ﺷﻤﺎ ﻣﯽ‌ﺷﻮﻧﺪ ﺍﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺪﻭﻥ ﻫﻴﭻ ﮔﻮﻧﻪ ﻣﺸﮑﻠﯽ ﺁﻥ‌ﻫﺎ ﺭﺍ ﭘﺸﺖ ﺳﺮ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺩﺳﺘﻪ ﺳﻮﻡ ﺭﻓﺘﺎﺭﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﭼﺮﺍﻍ ﻗﺮﻣﺰ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ ﮐﻪ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﺎ ﺣﺎﻟﺘﯽ ﻣﻮﺩﺏ ﻭ ﺁﺭﺍﻣﺶ ﺑﻪ ﺁﻥ‌ﻫﺎ ﭘﺎﺳﺦ ﺩﻫﻴﺪ ﺗﺎ ﺍﺳﺘﺮﺱ ﻭ ﻓﺸﺎﺭﯼ ﺑﺮ ﻃﺮﻑ ﻣﻘﺎﺑﻞ ﻗﺮﺍﺭ ﻧﮕﻴﺮﺩ. ﭘﺲ ﻫﻤﺎﻧﻄﻮﺭ ﮐﻪ ﻣﯽ‌ﺑﻴﻨﻴﺪ ﺑﺮﺧﻮﺭﺩﻫﺎ ﺩﺭ ﺭﻭﺍﺑﻂ ﺍﺟﺘﻤﺎﻋﯽ ﻫﻤﻮﺍﺭﻩ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺍﻣﺎ ﺑﺎﻳﺪ ﺑﺎ ﺑﺮﺧﻮﺭﺩﯼ ﻣﻨﻄﻘﯽ ﺍﺳﺘﺮﺱ ﺩﺭ ﺭﺍﺑﻄﻪ ﺭﺍ ﮐﺎﻫﺶ ﺩﺍﺩ.

ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﻫﺸﺘﻢ ﺭﺳﻴﺪﻳﻢ ﺍﻣﺎ ﭘﻴﺶ ﺍﺯ ﭘﺎﻳﺎﻥ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺩﺭﺑﺎﺭﻩ ﺍﺳﺘﺮﺱ ﺑﻴﺸﺘﺮ ﺑﺮ ﻧﻴﻤﯽ ﺍﺯ ﺟﻤﻌﻴﺖ ﺩﻧﻴﺎ ﺻﺤﺒﺖ ﮐﻨﻴﻢ. ﺩﺧﺘﺮﻫﺎ ﻭ ﺯﻧﺎﻥ ﺩﺭ ﺭﻭﺍﺑﻂ ﻓﺮﺩﯼ ﻣﻌﻤﻮﻻ ﺍﺳﺘﺮﺱ ﺑﻴﺶ ﺍﺯ ﺣﺪﯼ ﺭﺍ ﺗﺠﺮﺑﻪ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺍﻳﻦ ﺑﻪ ﺩﻟﻴﻞ ﻓﺮﻫﻨﮓ ﺩﺧﺘﺮ ﺧﻮﺏ ﺍﺳﺖ. ﺩﺧﺘﺮ ﺧﻮﺏ ﺩﺭ ﻓﺮﻫﻨﮓ‌ﻫﺎﯼ ﻣﻠﻞ ﻣﺨﺘﻠﻒ ﻣﻌﻤﻮﻻ ﺑﻪ ﺩﺧﺘﺮﯼ ﮔﻔﺘﻪ ‌ﻣﯽ‌ﺷﻮﺩ ﮐﻪ «ﻧﺒﺎﻳﺪ»ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺭﺍ ﺑﺎﻳﺪ ﺭﻋﺎﻳﺖ ﮐﻨﺪ ﻭ ﺍﺯ ﺍﻳﻦ ﺑﺎﺑﺖ ﺩﺭ ﺧﻮﺭﺩ ﺍﺣﺘﺮﺍﻡ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﻧﺒﺎﻳﺪﻫﺎﻳﯽ ﭼﻮﻥ ﺩﺭ ﻣﺮﮐﺰ ﺗﻮﺟﻪ ﻗﺮﺍﺭ ﻧﮕﺮﻓﺘﻦ، ﻧﺨﻨﺪﻳﺪﻥ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ، ﺷﺮﻡ ﻭ ﺣﻴﺎ ﺩﺍﺷﺘﻦ ﺑﻴﺸﺘﺮ ﺍﺯ ﭘﺴﺮﻫﺎ ﻭ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ. ﺯﻧﺎﻥ ﺑﺎ ﺍﻳﻨﮑﻪ ﺩﺭ ﺍﻳﻦ ﻓﺮﻫﻨﮓ‌ﻫﺎ ﺑﺎﺭ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﺩﺭ ﺯﻧﺪﮔﯽ ﺧﺎﻧﻮﺍﺩﮔﯽ ﺑﻪ ﺩﻭﺵ ﻣﯽ‌ﮐﺸﻨﺪ ﺍﻣﺎ ﻣﺠﺒﻮﺭ ﻫﺴﺘﻨﺪ ﮐﻪ ﺍﺳﺘﺮﺱ ﺑﻴﺸﺘﺮﯼ ﺭﺍ ﻧﻴﺰ ﺗﺤﻤﻞ ﮐﻨﻨﺪ. ﺑﻬﺘﺮ ﺍﺳﺖ ﺩﺭ ﺑﺮﺧﻮﺭﺩﻫﺎﻳﻤﺎﻥ ﺑﺎ ﺩﻳﮕﺮﺍﻥ، ﺑﻪ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﻧﻴﺰ ﺗﻮﺟﻪ ﮐﻨﻴﻢ.
