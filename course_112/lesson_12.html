ﺩﺭﺱ ﭼﻬﺎﺭﻡ- ﻧﻈﻢ ﺭﺍ ﺑﻪ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﺑﺎﺯﮔﺮﺩﺍﻧﻴﺪ

ﺑﻪ ﺩﺭﺱ ﭼﻬﺎﺭﻡ ﺧﻮﺵ ﺁﻣﺪﻳﺪ. ﺩﺭ ﺩﺭﺱ ﺳﻮﻡ ﺑﺎ ﻧﺤﻮﻩ ﺗﻨﻔﺲ ﺩﺭﺳﺖ ﺑﺮﺍﯼ ﮐﺎﻫﺶ ﺍﺳﺘﺮﺱ ﺁﺷﻨﺎ ﺷﺪﻳﺪ ﻭ ﺩﺭ ﺩﺭﺱ ﺩﻭﻡ ﻧﻴﺰ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺫﻫﻨﯽ ﺑﺮﺍﯼ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺍﺳﺘﺮﺱ ﺭﺍ ﻓﺮﺍ ﮔﺮﻓﺘﻴﺪ. ﺑﺎ ﺗﺮﮐﻴﺐ ﺍﻳﻦ ﺭﺍﻫﮑﺎﺭﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺗﺎ ﺍﻧﺪﺍﺯﻩ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﺩﺭ ﺯﻣﺎﻥ‌ﻫﺎﻳﯽ ﺑﺎ ﺍﺳﺘﺮﺱ ﻣﻮﺍﺟﻪ ﻣﯽ‌ﺷﻮﻳﺪ، ﺑﺎ ﺁﻥ ﻣﻘﺎﺑﻠﻪ ﮐﻨﻴﺪ ﻭ ﺍﺯ ﺷﺪﺕ ﺍﺳﺘﺮﺱ ﺩﺭ ﻟﺤﻈﻪ ﺧﻮﺩ ﺑﮑﺎﻫﻴﺪ. ﺍﻣﺎ ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﻗﻊ ﻣﯽ‌ﺗﻮﺍﻥ ﺑﺎ ﺍﻧﺠﺎﻡ ﺭﺍﻫﮑﺎﺭﻫﺎﻳﯽ ﺍﺯ ﻗﺒﻴﻞ ﺭﻋﺎﻳﺖ ﻧﻈﻢ ﺍﺯ ﺍﻳﺠﺎﺩ ﺍﺳﺘﺮﺱ ﺩﺭ ﺁﻳﻨﺪﻩ ﺟﻠﻮﮔﻴﺮﯼ ﮐﺮﺩ.

ﮐﺎﻓﯽ ﺍﺳﺖ ﻟﺤﻈﻪ ﺍﯼ ﺭﺍ ﺗﺼﻮﺭ ﮐﻨﻴﺪ ﮐﻪ ﺩﺭ ﻳﮏ ﻧﻴﻤﻪ ﺷﺐ ﺳﺮﺩ ﺍﺯ ﺧﺎﻧﻪ ﺩﻭﺳﺘﺘﺎﻥ ﺑﻪ ﺧﺎﻧﻪ ﺧﻮﺩ ﺑﺮ ﻣﯽ‌ﮔﺮﺩﻳﺪ ﻭ ﻓﺮﺩﺍ ﺻﺒﺢ ﻧﻴﺰ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﻪ ﻣﺴﺎﻓﺮﺕ ﺑﺮﻭﻳﺪ ﺍﻣﺎ ﻫﻨﮕﺎﻡ ﺑﺎﺯ ﮐﺮﺩﻥ ﺩﺭ ﺧﺎﻧﻪ ﻣﺘﻮﺟﻪ ﻣﯽ‌ﺷﻮﻳﺪ ﮐﻪ ﮐﻠﻴﺪ ﺭﺍ ﺩﺭ ﺧﺎﻧﻪ ﺧﻮﺩﺗﺎﻥ ﺟﺎ ﮔﺬﺍﺷﺘﻪ ﺑﻮﺩﻳﺪ. ﺍﻳﻨﺠﺎﺳﺖ ﮐﻪ ﺍﺳﺘﺮﺱ ﺑﺎﻟﻘﻮﻩ‌ﺍﯼ ﺑﻪ ﺷﻤﺎ ﻭﺍﺭﺩ ﻣﯽ‌ﺷﻮﺩ ﭼﺮﺍ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻟﺤﻈﻪ ﺍﺣﺘﻤﺎﻻ ﮐﺎﺭ ﺧﺎﺻﯽ ﺟﺰ ﺷﮑﺴﺘﻦ ﺷﻴﺸﻪ ﭘﻨﺠﺮﻩ ﭘﻴﺶ ﺭﻭ ﻧﺪﺍﺭﻳﺪ ﻭ ﺩﺭﺩﺳﺮﻫﺎﻳﯽ ﺩﻳﮕﺮ ﻧﻴﺰ ﻫﻤﺮﺍﻩ ﺁﻥ ﺧﻮﺍﻫﺪ ﺁﻣﺪ. ﺩﺍﺷﺘﻦ ﻧﻈﻢ ﻭ ﺑﻪ ﻫﻤﺮﺍﻩ ﺑﺮﺩﻥ ﮐﻠﻴﺪ ﺷﺎﻳﺪ ﺳﺒﺐ ﻣﯽ‌ﺷﺪ ﮐﻪ ﻫﻴﭻ ﮔﺎﻩ ﺍﻳﻦ ﺍﺳﺘﺮﺱ ﺭﺍ ﻟﻤﺲ ﻧﮑﻨﻴﺪ.

ﻣﻌﻤﻮﻻ ﺩﻻﻳﻞ ﺑﯽ‌ﻧﻈﻤﯽ ﮔﻮﻧﺎﮔﻮﻥ ﺍﺳﺖ: ﻭﻗﺖ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﻧﺪﺍﺭﻳﺪ، ﻓﻀﺎﯼ ﺧﺎﻧﻪ‌ﺗﺎﻥ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﮐﺎﻓﯽ ﺑﺰﺭﮒ ﻧﻴﺴﺖ، ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺍﺑﻂ ﭘﻴﭽﻴﺪﻩ‌ﺍﯼ ﺩﺍﺭﻳﺪ ﺍﻣﺎ ﺁﻧﭽﻪ ﮐﻪ ﻣﻬﻢ ﺍﺳﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﺗﺠﺮﺑﻪ ﺍﺳﺘﺮﺱ ﮐﻤﺘﺮ ﺩﺭ ﺯﻧﺪﮔﯽ، ﺑﯽ‌ﻧﻈﻤﯽ ﺭﺍ ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ ﮐﻨﺎﺭ ﮔﺬﺍﺷﺘﻪ ﻭ ﺑﻪ ﻣﻨﻈﻢ ﺷﺪﻥ ﺭﻭﯼ ﺑﻴﺎﻭﺭﻳﺪ.

ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﻗﺼﺪ ﺍﻧﺠﺎﻡ ﮐﺎﺭﯼ ﺭﺍ ﻣﯽ‌ﮐﻨﻴﺪ، ﺗﻨﻬﺎ ﻓﮑﺮ ﮐﺮﺩﻥ ﺑﻪ ﺍﻧﺠﺎﻡ ﻭ ﭘﺎﻳﺎﻥ ﺁﻥ ﻣﻬﻢ ﻧﻴﺴﺖ ﺑﻠﮑﻪ ﺑﺎﻳﺪ ﺗﻤﺎﻣﯽ ﺟﻮﺍﻧﺐ ﮐﺎﺭ ﺭﺍ ﺑﺴﻨﺠﻴﺪ ﻣﺜﻼ ﺑﺎﻳﺪ ﺑﺎ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﭘﻴﺶ ﺑﺮﻭﻳﺪ ﭼﺮﺍ ﮐﻪ ﺑﺎ ﺑﺮﻧﺎﻣﻪ‌ﺭﻳﺰﯼ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﻫﺪﻓﺘﺎﻥ، ﺍﺣﺘﻤﺎﻻ ﻭﻗﺖ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﺩﻳﮕﺮ ﮐﺎﺭﻫﺎ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﻭ ﺍﺯ ﮔﺮﻓﺘﻦ ﺍﺳﺘﺮﺱ ﺭﻫﺎﻳﯽ ﭘﻴﺪﺍ ﻣﯽ‌ﮐﻨﻴﺪ.

ﺍﮔﺮ ﺑﯽ ﻧﻈﻤﯽ ﺩﺭ ﺍﺗﺎﻕ ﻳﺎ ﺧﺎﻧﻪ ﺷﻤﺎ ﺭﺍ ﺍﺫﻳﺖ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺑﻪ ﺩﻧﺒﺎﻝ ﻣﻨﻈﻢ ﮐﺮﺩﻥ ﺁﻥ ﻫﺴﺘﻴﺪ، ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﺎﺭ ﻣﻨﻈﻢ ﮐﺮﺩﻥ ﺭﺍ ﺑﻪ ﭼﻨﺪ ﺑﺨﺶ ﮐﻮﭼﮑﺘﺮ ﺗﻘﺴﻴﻢ ﮐﻨﻴﺪ ﻭ ﺑﺎ ﻣﻨﻈﻢ ﮐﺮﺩﻥ ﺍﻳﻦ ﺑﺨﺶ‌ﻫﺎﯼ ﮐﻮﭼﮏ، ﻧﻈﻢ ﺭﺍ ﺑﻪ ﺧﺎﻧﻪ ﻳﺎ ﺍﺗﺎﻕ ﺧﻮﺩ ﺑﻴﺎﻭﺭﻳﺪ. ﻳﮑﯽ ﺍﺯ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻣﻮﺿﻮﻋﺎﺗﯽ ﮐﻪ ﺩﺭ ﺍﻳﻦ ﻗﺒﻴﻞ ﮐﺎﺭﻫﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﺗﻼﺵ ﮐﺮﺩﻥ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﮐﺎﺭ ﻭ ﺍﻫﺘﻤﺎﻡ ﻭﺭﺯﻳﺪﻥ ﺑﻪ ﺁﻥ ﺍﺳﺖ ﻣﺜﻼ ﺍﻳﻨﮑﻪ ﻭﻗﺘﯽ ﻳﮏ ﺑﺨﺶ ﮐﻮﭼﮏ ﺍﺯ ﺍﺗﺎﻕ ﻳﺎ ﺧﺎﻧﻪ ﺭﺍ ﺗﻤﻴﺰ ﮐﺮﺩﻳﺪ، ﺗﻨﺒﻠﯽ ﻧﮑﻨﻴﺪ ﻭ ﺑﻪ ﺗﻤﻴﺰ ﮐﺮﺩﻥ ﻭ ﻣﻨﻈﻢ ﮐﺮﺩﻥ ﺍﺗﺎﻕ ﺍﺩﺍﻣﻪ ﺑﺪﻫﻴﺪ. ﺍﻳﻦ ﺍﺩﺍﻣﻪ ﺩﺍﺩﻥ ﻭ ﭘﺸﺖ ﮐﺎﺭ ﺩﺍﺷﺘﻦ ﻳﮑﯽ ﺍﺯ ﺧﺼﻠﺖ‌ﻫﺎﯼ ﺧﻮﺏ ﺍﻓﺮﺍﺩ ﻣﻨﻈﻢ ﺍﺳﺖ.

ﺣﺘﻤﺎ ﺩﺭ ﻫﻨﮕﺎﻡ ﺗﻤﻴﺰ ﮐﺮﺩﻥ ﻳﺎ ﻣﻨﻈﻢ ﮐﺮﺩﻥ ﺍﺗﺎﻕ ﺑﻪ ﻭﺳﺎﻳﻠﯽ ﺑﺮ ﻣﯽ‌ﺧﻮﺭﻳﺪ ﮐﻪ ﭘﻴﺶ ﺍﺯ ﺍﻳﻦ ﺗﻬﻴﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﻭ ﺍﮐﻨﻮﻥ ﺑﺮﺍﯼ ﺷﻤﺎ ﻗﺎﺑﻞ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻴﺴﺘﻨﺪ ﻭ ﻳﺎ ﺷﻤﺎ ﺑﻪ ﻧﺪﺭﺕ ﺍﺯ ﺁﻥ ﻫﺎ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﮐﻨﻴﺪ، ﻫﻴﭻ ﻭﻗﺖ ﺍﻳﻦ ﻭﺳﺎﻳﻞ ﺭﺍ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﺷﺎﻳﺪ ﺭﻭﺯﯼ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻢ، ﻧﮕﺎﻩ‌ﺩﺍﺭﯼ ﻧﮑﻨﻴﺪ ﻭ ﺁﻥ ﻫﺎ ﺭﺍ ﺍﺯ ﻣﺤﻞ ﺳﮑﻮﻧﺖ ﺧﻮﺩ ﺧﺎﺭﺝ ﮐﻨﻴﺪ. ﺍﻳﻦ ﻭﺳﺎﻳﻞ ﺍﺣﺘﻤﺎﻻ ﻫﻴﭻ ﮔﺎﻩ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﺷﻤﺎ ﻗﺮﺍﺭ ﻧﺨﻮﺍﻫﻨﺪ ﮔﺮﻓﺖ ﻭ ﻧﮕﺎﻩ‌ﺩﺍﺭﯼ ﺁﻥ ﻫﺎ ﻧﻴﺰ ﺟﺰ ﺩﺭﺩﺳﺮ ﺍﻳﺠﺎﺩ ﺑﯽ‌ﻧﻈﻤﯽ، ﺳﻮﺩ ﺩﻳﮕﺮﯼ ﺑﺮﺍﻳﺘﺎﻥ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺷﺖ! ﺩﺭ ﻭﺍﻗﻊ ﺩﺭ ﺍﻳﻨﺠﻮﺭ ﻣﻮﺍﻗﻊ ﺑﺎﻳﺪ ﺍﻳﻦ ﺗﻔﮑﺮ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﮔﺮ ﺑﺎ ﭼﻨﻴﻦ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺭﻭﺑﺮﻭ ﺷﺪﻳﺪ، ﺑﺎﻳﺪ ﺗﺼﻤﻴﻢ ﺑﮕﻴﺮﻳﺪ ﮐﻪ ﺁﻥ ﺭﺍ ﺩﻭﺭ ﺑﻴﻨﺪﺍﺯﻳﺪ ﻣﮕﺮ ﺍﻳﻨﮑﻪ ﺗﻌﺮﻳﻒ ﻣﺸﺨﺼﯽ ﺑﺮﺍﯼ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺁﻥ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

ﺑﺮﺧﯽ ﻣﻮﺍﻗﻊ ﺷﻤﺎ ﺑﺎ ﺍﻳﻨﮑﻪ ﺍﺯ ﺍﺷﻴﺎﯼ ﺧﺎﺻﯽ ﺧﻮﺷﺘﺎﻥ ﻣﯽ‌ﺁﻳﺪ ﻭ ﺑﺮﺍﻳﺘﺎﻥ ﺑﺎ ﺍﺭﺯﺵ ﻫﺴﺘﻨﺪ ﺍﻣﺎ ﺑﻪ ﺑﺮﺧﯽ ﺩﻻﻳﻞ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﺪ ﺁﻥ ﻫﺎ ﺭﺍ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ. ﻣﺜﻼ ﺑﺮﺧﯽ ﻣﻮﺍﻗﻊ ﻧﻘﺎﺷﯽ‌ﻫﺎﯼ ﮐﻮﺩﮎ ﺷﻤﺎ، ﺍﺭﺯﺵ ﻓﺮﺍﻭﺍﻧﯽ ﺑﺮﺍﯼ ﺷﻤﺎ ﺩﺍﺭﺩ ﺍﻣﺎ ﻧﮕﺎﻩ‌ﺩﺍﺭﯼ ﻫﻤﻪ ﻧﻘﺎﺷﯽ‌ﻫﺎﯼ ﮐﻮﺩﮐﺘﺎﻥ ﺍﻣﮑﺎﻥ‌ﭘﺬﻳﺮ ﻧﻴﺴﺖ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﺮﺧﯽ ﺍﺯ ﻧﻘﺎﺷﯽ‌ﻫﺎﯼ ﮐﻮﺩﮎ ﺧﻮﺩ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ ﻭ ﺑﻘﻴﻪ ﺭﺍ ﺩﻭﺭ ﺑﺮﻳﺰﻳﺪ. ﺍﮔﺮ ﻫﻨﮕﺎﻡ ﺩﻭﺭ ﺭﻳﺨﺘﻦ ﺍﻳﻦ ﻧﻘﺎﺷﯽ‌ﻫﺎ ﻋﺬﺍﺏ ﻭﺟﺪﺍﻥ ﺧﻮﺍﻫﻴﺪ ﮔﺮﻓﺖ، ﺧﻴﻠﯽ ﺳﺎﺩﻩ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﺍﻳﻦ ﻧﻘﺎﺷﯽ‌ﻫﺎ ﻋﮑﺲ ﺑﺮﺩﺍﺭﯼ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺁﻥ ﻫﺎ ﺭﺍ ﺍﺳﮑﻦ ﮐﺮﺩﻩ ﻭ ﺩﺭ ﮐﺎﻣﭙﻴﻮﺗﺮ ﺧﻮﺩ ﻧﮕﺎﻩ ﺩﺍﺭﯼ ﮐﻨﻴﺪ ﺗﺎ ﻓﻀﺎﻳﯽ ﻓﻴﺰﻳﮑﯽ ﺭﺍ ﺑﻪ ﺧﻮﺩ ﺍﺧﺘﺼﺎﺹ ﻧﺪﻫﻨﺪ. ﺍﻳﻦ ﺗﻮﺻﻴﻪ ﺣﺘﯽ ﺑﺮﺍﯼ ﺍﺷﻴﺎﻳﯽ ﺩﻳﮕﺮ ﻧﻴﺰ ﻗﺎﺑﻞ ﺗﻌﻤﻴﻢ ﺍﺳﺖ ﮐﻪ ﺍﺭﺯﺷﯽ ﻣﻌﻨﻮﯼ ﺑﺮﺍﻳﺘﺎﻥ ﺩﺍﺭﻧﺪ ﺍﻣﺎ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﺪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻫﻤﭽﻨﺎﻥ ﺩﺭ ﺍﺗﺎﻕ ﻳﺎ ﺧﺎﻧﻪ ﺧﻮﺩ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ.

ﺩﺭ ﺑﺮﺧﯽ ﻣﻮﺍﻗﻊ ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺗﻤﺎﻣﯽ ﮐﺎﻏﺬﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺍﺗﺎﻗﺘﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﺭﺍ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﻳﻦ ﮐﺎﻏﺬﻫﺎ ﺭﺍ ﻃﺒﻘﻪ‌ﺑﻨﺪﯼ ﮐﻨﻴﺪ. ﻣﺜﻼ ﺍﻳﻦ ﮐﺎﻏﺬﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺍﺳﻨﺎﺩ ﺑﺎﻧﮑﯽ، ﻓﻴﺶ ﺁﺏ ﻭ ﺑﺮﻕ، ﺑﺮﮔﻪ‌ﻫﺎﯼ ﺩﺭﺳﯽ ﻭ ﻳﺎ ﺳﻨﺪ ﻣﻠﮑﯽ ﺑﺎﺷﻨﺪ. ﺩﺭ ﺍﻳﻦ ﻣﻮﺍﻗﻊ ﺑﻬﺘﺮ ﺍﺳﺖ ﺍﺯ ﻳﮏ‌ﺟﺎ ﻗﺮﺍﺭﺩﺍﺩﻥ ﺍﻳﻦ ﮐﺎﻏﺬﻫﺎ ﺧﻮﺩﺩﺍﺭﯼ ﮐﺮﺩﻩ ﻭ ﻫﻤﻮﺍﺭﻩ ﺍﻳﻦ ﻧﮑﺘﻪ ﺭﺍ ﺑﻪ ﻳﺎﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﻳﻦ ﮐﺎﻏﺬﻫﺎ ﺑﺎﻳﺪ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻫﻤﻴﺖ‌ﺷﺎﻥ ﻃﺒﻘﻪ ﺑﻨﺪﯼ ﺷﻮﻧﺪ.

ﺑﺮﺧﯽ ﺍﺯ ﺍﻳﻦ ﺍﺳﻨﺎﺩ ﮐﺎﻏﺬﯼ، ﺍﺳﻨﺎﺩﯼ ﻣﻬﻢ ﻫﺴﺘﻨﺪ ﻣﺎﻧﻨﺪ ﺷﻨﺎﺳﻨﺎﻣﻪ، ﭘﺎﺳﭙﻮﺭﺕ، ﺍﺳﻨﺎﺩ ﻣﻠﮑﯽ ﻭ ﺍﺳﻨﺎﺩﯼ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ. ﺍﻳﻦ ﺍﺳﻨﺎﺩ ﺑﺎﻳﺪ ﺩﺭ ﺟﺎﻫﺎﻳﯽ ﺍﻣﻦ ﻧﮕﺎﻫﺪﺍﺭﯼ ﺷﻮﻧﺪ ﮐﻪ ﻫﻢ ﻫﻤﻴﺸﻪ ﺩﺭ ﺩﺳﺘﺮﺱ ﺷﻤﺎ ﺑﺎﺷﺪ ﻭ ﻫﻢ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮ ﻧﺘﻮﺍﻧﻨﺪ ﺑﻪ ﺁﻥ‌ﻫﺎ ﺑﻪ ﺁﺳﺎﻧﯽ ﺩﺳﺘﺮﺳﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻨﺪ.

ﺷﺎﻳﺪ ﺑﻬﺘﺮﻳﻦ ﭘﻴﺸﻨﻬﺎﺩ ﺑﺮﺍﯼ ﻃﺒﻘﻪ ﺑﻨﺪﯼ ﮐﺎﻏﺬﻫﺎ، ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﭼﻬﺎﺭ ﺳﺒﺪ ﺑﺮﺍﯼ ﻃﺒﻘﻪ ﺑﻨﺪﯼ ﺍﺳﺖ:

ﺩﺭﺑﺎﺭﻩ ﻣﮑﺎﻥ ﺍﻳﻦ ﺳﺒﺪﻫﺎ ﻧﻴﺰ ﺑﺎﻳﺪ ﮔﻔﺖ ﮐﻪ ﺳﺒﺪ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻧﯽ ﺭﺍ ﺭﻭﯼ ﺳﺒﺪ ﭘﺮﺩﺍﺧﺘﻨﯽ ﻭ ﺩﺭ ﺭﻭﯼ ﻣﻴﺰ ﺧﻮﺩ ﺑﮕﺬﺍﺭﻳﺪ. ﺳﺒﺪ ﻓﺎﻳﻞ ﺑﻨﺪﯼ ﺭﺍ ﺩﺭ ﺯﻳﺮ ﻣﻴﺰ ﻭ ﺑﻪ ﺩﻭﺭ ﺍﺯ ﮐﺎﻏﺬﻫﺎﻳﯽ ﮐﻪ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ ﺑﮕﺬﺍﺭﻳﺪ ﻭ ﺳﺒﺪ ﺧﻮﺍﻧﺪﻧﯽ ﺭﺍ ﺩﺭ ﺑﺨﺶ ﺩﻳﮕﺮﯼ ﺍﺯ ﺍﺗﺎﻕ ﺑﮕﺬﺍﺭﻳﺪ ﮐﻪ ﻫﺮﮔﺎﻩ ﻓﺮﺻﺘﯽ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺁﻥ‌ﻫﺎ ﭘﻴﺶ ﺁﻣﺪ، ﺑﺘﻮﺍﻧﻴﺪ ﺁﻥ‌ﻫﺎ ﺭﺍ ﻣﻄﺎﻟﻌﻪ ﮐﻨﻴﺪ.

ﺩﺭ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﺍﻣﺮﻭﺯ ﻗﺮﺍﺭ ﺩﺍﺭﻳﻢ ﺍﻣﺎ ﭘﻴﺶ ﺍﺯ ﺁﻧﮑﻪ ﺩﺭﺱ ﺭﺍ ﺑﻪ ﭘﺎﻳﺎﻥ ﺑﺒﺮﻳﻢ، ﺗﻮﺻﻴﻪ‌ﺍﯼ ﺩﻳﮕﺮ ﺑﺮﺍﻳﺘﺎﻥ ﺩﺍﺭﻳﻢ ﮐﻪ ﺑﯽ‌ﺷﮏ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺩﻭﺭﻩ ﭘﺲ ﺍﺯ ﻣﻨﻈﻢ ﺷﺪﻥ ﺑﻪ ﮐﺎﺭﺗﺎﻥ ﺑﻴﺎﻳﺪ. ﻫﻤﻮﺍﺭﻩ ﺳﻌﯽ ﮐﻨﻴﺪ ﮐﻤﺘﺮ ﺍﺯ ﻧﻴﺎﺯﺗﺎﻥ ﺧﺮﻳﺪ ﮐﻨﻴﺪ ﺗﺎ ﻫﺮﺝ ﻭ ﻣﺮﺝ ﻧﮕﺎﻫﺪﺍﺭﯼ ﻭﺳﺎﻳﻞ ﺍﺿﺎﻓﯽ ﺑﺎﺭ ﺩﻳﮕﺮ ﺑﻪ ﺯﻧﺪﮔﯽ ﺷﻤﺎ ﺑﺎﺯ ﻧﮕﺮﺩﺩ. ﻫﺮ ﻭﻗﺖ ﺧﻮﺍﺳﺘﻴﺪ ﮐﻪ ﺧﺮﻳﺪﯼ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﻫﻤﻮﺍﺭﻩ ﺍﺯ ﺧﻮﺩ ﺑﭙﺮﺳﻴﺪ ﺁﻳﺎ ﻭﺍﻗﻌﺎ ﺑﻪ ﺍﻳﻦ ﻭﺳﻴﻠﻪ ﺍﺣﺘﻴﺎﺝ ﺩﺍﺭﻡ ﻳﺎ ﻋﻤﻠﮑﺮﺩ ﺍﻳﻦ ﻭﺳﻴﻠﻪ ﺑﺮﺍﯼ ﻣﻦ ﻣﺸﺨﺺ ﺍﺳﺖ ﻳﺎ ﺧﻴﺮ. ﭘﺎﺳﺦ ﺑﻪ ﺍﻳﻦ ﺳﻮﺍﻝ‌ﻫﺎ ﺗﺎ ﺣﺪﻭﺩ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺧﻴﺎﻝ ﺷﻤﺎ ﺭﺍ ﺍﺯ ﺧﺮﻳﺪﻥ ﻭﺳﻴﻠﻪ ﺟﺪﻳﺪ ﺭﺍﺣﺖ ﮐﻨﺪ.

ﺑﻪ ﭘﺎﻳﺎﻥ ﺩﺭﺱ ﭼﻬﺎﺭﻡ ﺭﺳﻴﺪﻳﻢ، ﺩﺭ ﺩﺭﺱ ﺁﻳﻨﺪﻩ ﺑﻪ ﺭﺍﻫﮑﺎﺭﻫﺎﻳﯽ ﺑﺮﺍﯼ ﻳﺎﻓﺘﻦ ﻭﻗﺖ ﺑﻴﺸﺘﺮ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.
