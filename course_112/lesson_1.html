ﺩﺭﺱ ﺍﻭﻝ- ﺁﺷﻨﺎﻳﯽ ﺑﺎ ﺍﺳﺘﺮﺱ

ﺣﺘﻤﺎ ﺑﺮﺍﻳﺘﺎﻥ ﭘﻴﺶ ﺁﻣﺪﻩ ﮐﻪ ﭘﺲ ﺍﺯ ﺟﻠﺴﺎﺕ ﺍﺳﺘﺨﺪﺍﻣﯽ ﻳﺎ ﺑﺮﮔﺰﺍﺭﯼ ﺍﻣﺘﺤﺎﻧﺎﺕ ﭘﺎﻳﺎﻥ ﺗﺮﻡ، ﺑﻪ ﺩﻭﺳﺘﺎﻧﺘﺎﻥ ﺑﮕﻮﻳﻴﺪ ﺍﺳﺘﺮﺱ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﺳﺒﺐ ﺷﺪ ﺗﺎ ﺩﺭ ﺁﻥ ﺟﻠﺴﻪ ﻣﻮﻓﻖ ﻧﺒﺎﺷﻢ ﻭ ﻳﺎ ﻫﻨﮕﺎﻡ ﺣﻀﻮﺭ ﺩﺭ ﻳﮏ ﺟﻤﻊ ﻧﺎﺁﺷﻨﺎ ﻳﺎ ﺻﻒ ﺍﻧﺘﻈﺎﺭ ﺍﺗﻮﺑﻮﺱ، ﺍﺳﺘﺮﺱ ﺷﺪﻳﺪﯼ ﺭﺍ ﺍﺣﺴﺎﺱ ﮐﻨﻴﺪ ﮐﻪ ﻋﻤﻠﮑﺮﺩ ﺷﻤﺎ ﺭﺍ ﮐﺎﻫﺶ ﺩﻫﺪ. ﺷﻤﺎ ﺍﺣﺘﻤﺎﻻ ﺑﺎ ﺍﻳﻨﮑﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﺑﺎﺭﻫﺎ ﻭ ﺑﺎﺭﻫﺎ ﺍﺣﺴﺎﺱ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺍﻣﺎ ﺗﺎ ﮐﻨﻮﻥ ﻧﺘﻮﺍﻧﺴﺘﻪ‌ﺍﻳﺪ ﺑﺎ ﺁﻥ ﻣﻘﺎﺑﻠﻪ ﮐﻨﻴﺪ. ﺍﻳﻦ ﺍﺳﺘﺮﺱ ﭼﻴﺴﺖ ﮐﻪ ﺍﻳﻨﮕﻮﻧﻪ ﺑﺎﻋﺚ ﻧﺎﮐﺎﻣﯽ ﺩﺭ ﺷﻤﺎ ﺷﺪﻩ ﺍﺳﺖ؟

ﭘﻴﺶ ﺍﺯ ﺁﻥ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﻣﺴﺘﻘﻴﻢ ﺩﺭﺑﺎﺭﻩ ﺗﻌﺮﻳﻒ ﺍﺳﺘﺮﺱ ﺻﺤﺒﺖ ﮐﻨﻴﻢ ﺍﺟﺎﺯﻩ ﺑﺪﻫﻴﺪ ﺍﺯ ﺩﻧﻴﺎﯼ ﺍﻳﻦ ﺭﻭﺯﻫﺎﯼ‌ﻣﺎﻥ ﺻﺤﺒﺖ ﮐﻨﻴﻢ ﮐﻪ ﺍﺣﺘﻤﺎﻻ ﺳﺒﺐ ﺍﻳﺠﺎﺩ ﺍﺳﺘﺮﺱ ﺑﻴﺸﺘﺮﯼ ﺑﺮﺍﯼ ﻣﺎ ﺷﺪﻩ ﺍﺳﺖ. ﺁﻟﻮﻳﻦ ﺗﺎﻓﻠﺮ، ﻧﻮﻳﺴﻨﺪﻩ ﻭ ﺁﻳﻨﺪﻩ‌ﭘﮋﻭﻩ ﺁﻣﺮﻳﮑﺎﻳﯽ، ﺩﺭ ﮐﺘﺎﺏ ﻣﻌﺮﻭﻓﺶ ﺑﻪ ﻧﺎﻡ ﺷﻮﮎ ﺁﻳﻨﺪﻩ ﻧﻮﺷﺘﻪ ﺍﺳﺖ ﻫﺮﮔﺎﻩ ﺗﻐﻴﻴﺮﺍﺗﯽ ﺯﻳﺎﺩ ﺩﺭ ﺯﻣﺎﻧﯽ ﮐﻮﺗﺎﻩ ﺭﺥ ﺩﻫﺪ، ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻋﺎﻣﻠﯽ ﻣﻬﻢ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺍﺳﺘﺮﺱ ﺩﺭ ﺯﻧﺪﮔﯽ ﺍﻧﺴﺎﻥ‌ﻫﺎ ﺑﺎﺷﺪ.

ﺑﺎﻳﺪ ﻗﺒﻮﻝ ﮐﻨﻴﻢ ﺯﻧﺪﮔﯽ ﺍﻳﻦ ﺭﻭﺯﻫﺎﯼ ﻣﺎ ﻧﺴﺒﺖ ﺑﻪ ﺳﯽ ﺗﺎ ﭼﻬﻞ ﺳﺎﻝ ﮔﺬﺷﺘﻪ ﺳﺨﺖ‌ﺗﺮ ﺷﺪﻩ ﻭ ﺍﻳﻦ ﺭﻭﺯﻫﺎ ﺩﺭ ﮐﻨﺎﺭ ﻣﻔﺎﻫﻴﻤﯽ ﻫﻤﭽﻮﻥ ﺗﺮﺍﻓﻴﮏ، ﺁﭘﺎﺭﺗﻤﺎﻥ‌ﻧﺸﻴﻨﯽ، ﮐﻤﺒﻮﺩ ﻭﻗﺖ، ﺍﺯﺩﺣﺎﻡ ﺟﻤﻌﻴﺖ ﻭ ﺣﺘﯽ ﭘﺪﻳﺪﻩ‌ﺍﯼ ﺑﻪ ﻧﺎﻡ ﻣﻮﺑﺎﻳﻞ، ﺁﺭﺍﻣﺶ ﮐﻤﺘﺮﯼ ﺭﺍ ﻧﺴﺒﺖ ﺑﻪ ﺯﻣﺎﻥ‌ﻫﺎﯼ ﮔﺬﺷﺘﻪ ﺗﺠﺮﺑﻪ ﻣﯽ‌ﮐﻨﻴﻢ. ﺩﺭ ﻭﺍﻗﻊ ﺩﺭ ﺯﻣﺎﻥ ﮐﻮﺗﺎﻩ ﺳﯽ ﺗﺎ ﭼﻬﻞ ﺳﺎﻝ ﮔﺬﺷﺘﻪ، ﺳﺮﻋﺖ ﺗﻐﻴﻴﺮﺍﺕ ﺩﺭ ﺩﻧﻴﺎ ﺍﻓﺰﺍﻳﺶ ﻳﺎﻓﺘﻪ ﺍﺳﺖ ﻭ  ﻣﻔﺎﻫﻴﻤﯽ ﺑﺴﻴﺎﺭ ﺟﺪﻳﺪ ﺑﻪ ﺯﻧﺪﮔﯽ ﻣﺎ ﺍﺿﺎﻓﻪ ﺷﺪﻩ ﮐﻪ ﺯﻣﻴﻨﻪ‌ﺳﺎﺯ ﺍﺳﺘﺮﺱ ﺩﺭ ﺯﻧﺪﮔﯽ ﻣﺎ ﺷﺪﻩ ﺍﺳﺖ.

ﺍﮐﻨﻮﻥ ﺗﻌﺮﻳﻒ ﺁﻗﺎﯼ ﺗﺎﻓﻠﺮ ﺭﺍ ﺑﻪ ﺻﻮﺭﺗﯽ ﺩﻳﮕﺮ ﺑﺮﺍﻳﺘﺎﻥ ﺑﻴﺎﻥ ﻣﯽ‌ﮐﻨﻴﻢ ﺗﺎ ﺁﻥ ﺭﺍ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﺧﻮﺩ ﻟﻤﺲ ﮐﻨﻴﺪ: ﺭﻭﺑﺮﻭ ﺷﺪﻥ ﺑﺎ ﺗﻐﻴﻴﺮﺍﺗﯽ ﻧﺎﺧﻮﺍﺳﺘﻪ ﻭ ﺟﺪﻳﺪ ﺁﻥ ﻫﻢ ﺩﺭ ﻣﻮﻗﻌﻴﺖ‌ﻫﺎﻳﯽ ﮐﻪ ﺍﺣﺘﻤﺎﻻ ﻗﺒﻼ ﺩﺭ ﺁﻥ ﻗﺮﺍﺭ ﻧﮕﺮﻓﺘﻪ‌ﺍﻳﺪ ﻭ ﺑﺮﺍﻳﺘﺎﻥ ﺟﺪﻳﺪ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﺩﺭﮎ ﺍﻳﻦ ﺗﻌﺮﻳﻒ ﮐﺎﻓﯽ ﺍﺳﺖ ﻳﮏ ﻟﺤﻈﻪ ﺍﺯ ﺟﻠﺴﻪ ﺍﻣﺘﺤﺎﻧﯽ ﺭﺍ ﺗﺼﻮﺭ ﮐﻨﻴﺪ ﮐﻪ ﺑﺎ ﺳﻮﺍﻻﺗﯽ ﺟﺪﻳﺪ ﻣﻮﺍﺟﻪ ﺷﺪﻩ‌ﺍﻳﺪ ﮐﻪ ﺗﻮﺍﻧﺎﻳﯽ ﭘﺎﺳﺦ ﺑﻪ ﺁﻥ ﺭﺍ ﻧﺪﺍﺭﻳﺪ ﻭ ﻳﺎ ﺩﺭ ﺟﻠﺴﻪ ﺍﺳﺘﺨﺪﺍﻣﯽ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﮐﻨﻴﺪ ﻓﺮﺩﯼ ﺧﺎﺹ ﺑﻪ ﺩﻧﺒﺎﻝ ﺭﺍﻫﯽ ﺑﺮﺍﯼ ﺭﺩ ﮐﺮﺩﻥ ﺷﻤﺎﺳﺖ ﻭ ﻣﺪﻳﺮﻳﺖ ﺟﻠﺴﻪ ﺭﺍ ﻧﻴﺰ ﺩﺭ ﺩﺳﺖ ﮔﺮﻓﺘﻪ ﺍﺳﺖ، ﺩﺭ ﺍﻳﻦ ﻟﺤﻈﺎﺕ ﺷﻤﺎ ﺩﭼﺎﺭ ﺍﺳﺘﺮﺱ ﻣﯽ‌ﺷﻮﻳﺪ! ﺍﺻﻼ ﺗﻌﺮﻳﻒ ﺍﺳﺘﺮﺱ ﻭ ﺗﺠﺮﺑﻪ ﺁﻥ ﻟﺤﻈﺎﺕ ﻫﻤﻴﻦ ﺍﺳﺖ: ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﻪ ﺩﻭﺭ ﺍﺯ ﺗﺼﻮﺭ ﭘﻴﺸﻴﻦ ﺍﺯ ﺗﻮﺍﻧﺎﻳﯽ‌ﻫﺎﯼ ﺧﻮﺩ، ﺩﺭ ﻳﮏ ﻣﻮﻗﻌﻴﺖ ﺟﺪﻳﺪ، ﻗﺒﻮﻝ ﻣﯽ‌ﮐﻨﻴﺪ ﺗﻮﺍﻧﺎﻳﯽ ﻣﻘﺎﺑﻠﻪ ﺑﺎ ﺗﻬﺪﻳﺪ ﺭﺍ ﻧﺪﺍﺭﻳﺪ ﻭ ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺩﭼﺎﺭ ﺍﺳﺘﺮﺱ ﻣﯽ‌ﺷﻮﻳﺪ.

ﺷﻤﺎ ﺩﺭ ﺯﻧﺪﮔﯽ ﺍﻳﻦ ﺭﻭﺯﻫﺎ، ﺍﺣﺘﻤﺎﻻ ﺩﺭ ﭼﻨﺪﻳﻦ ﻧﻮﺑﺖ ﺍﺯ ﺭﻭﺯ ﺩﭼﺎﺭ ﺍﺳﺘﺮﺱ‌ﻫﺎﯼ ﮐﻮﺗﺎﻩ ﻣﺪﺕ ﻣﯽ‌ﺷﻮﻳﺪ ﺍﻣﺎ ﺁﻳﺎ ﻣﯽ‌ﺩﺍﻧﻴﺪ ﺍﻳﻦ ﺍﺳﺘﺮﺱ‌ﻫﺎ ﭼﻪ ﻋﻮﺍﺭﺿﯽ ﺩﺍﺭﻧﺪ ﻭ ﺟﻤﻊ ﺷﺪﻥ ﺁﻥ‌ﻫﺎ ﭼﻪ ﺑﻼﻳﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮ ﺳﺮ ﺷﻤﺎ ﺑﻴﺎﻭﺭﺩ؟

ﺍﺳﺘﺮﺱ ﻭ ﺑﻴﻤﺎﺭﯼ ﻫﺎﯼ ﺟﺴﻤﯽ
ﺑﻴﺸﺘﺮ ﺷﺪﻥ ﺍﺳﺘﺮﺱ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﻧﻪ ﺗﻨﻬﺎ ﺑﻴﻤﺎﺭﯼ ﺭﻭﺣﯽ ﺑﻠﮑﻪ ﺑﻴﻤﺎﺭﯼ ﺟﺴﻤﯽ ﺭﺍ ﻧﻴﺰ ﺭﻗﻢ ﻣﯽ‌ﺯﻧﺪ. ﭘﮋﻭﻫﺸﮕﺮﺍﻧﯽ ﮐﻪ ﺩﺭ ﺯﻣﻴﻨﻪ ﺍﺳﺘﺮﺱ ﺗﺤﻘﻴﻖ ﻣﯽ‌ﮐﻨﻨﺪ ﺍﺣﺘﻤﺎﻝ ﻣﯽ‌ﺩﻫﻨﺪ ﺗﺎ ۹۰ ﺩﺭﺻﺪ ﻣﺮﺍﺟﻌﻪ ﺑﻴﻤﺎﺭﺍﻥ ﺑﻪ ﻣﺮﺍﮐﺰ ﭘﺰﺷﮑﯽ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺑﺎ ﺍﺳﺘﺮﺱ ﻣﺮﺗﺒﻂ ﺑﺎﺷﺪ. ﭘﮋﻭﻫﺸﮕﺮﺍﻥ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﺍﺳﺘﺮﺱ ﺣﺘﯽ ﺑﻪ ﺭﺍﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻋﻮﺍﺭﺽ ﺑﻴﻤﺎﺭﯼ‌ﻫﺎ ﺭﺍ ﺑﻴﺸﺘﺮ ﮐﺮﺩﻩ ﻭ ﺗﺎﺛﻴﺮﯼ ﻣﻨﻔﯽ ﺑﺮ ﺑﺪﻥ ﺑﻴﻤﺎﺭﺍﻥ ﺑﺮ ﺟﺎﯼ ﺑﮕﺬﺍﺭﺩ.

ﺍﺳﺘﺮﺱ ﺩﺭ ﻧﺨﺴﺘﻴﻦ ﻗﺪﻡ ﺑﺮ ﺭﻭﯼ ﻋﻀﻼﺕ ﺍﻓﺮﺍﺩ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ ﻭ ﺳﺒﺐ ﺍﻧﻘﺒﺎﺽ ﻭ ﻓﺸﺮﺩﻩ ﺷﺪﻥ ﺁﻥ ﻣﯽ‌ﺷﻮﺩ. ﻧﺘﺎﻳﺠﯽ ﻫﻤﭽﻮﻥ ﺳﺮ ﺩﺭﺩ، ﮐﻤﺮ ﺩﺭﺩ ﻳﺎ ﺩﺭﺩ ﺩﺭ ﻧﺎﺣﻴﻪ ﻗﻔﺴﻪ ﺳﻴﻨﻪ ﺟﺰﻭ ﺩﺭﺩﻫﺎﯼ ﺭﺍﻳﺞ ﻧﺎﺷﯽ ﺍﺯ ﺍﺳﺘﺮﺱ ﻫﺴﺘﻨﺪ.

ﺍﺳﺘﺮﺱ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺯﻧﺪﮔﯽ ﺷﻤﺎ ﭘﺎ ﺭﺍ ﻓﺮﺍﺗﺮ ﺑﮕﺬﺍﺭﺩ ﻭ ﺑﺎ ﺍﻓﺰﺍﻳﺶ ﻓﺸﺎﺭ ﺧﻮﻥ، ﺑﺎﺭﻳﮑﺘﺮ ﮐﺮﺩﻥ ﺭﮒ‌ﻫﺎ ﻭ ﻳﺎ ﺿﺮﺑﺎﻥ ﻧﺎﻣﻨﻈﻢ ﻗﻠﺐ ﺳﺒﺐ ﺑﺮﻭﺯ ﺳﮑﺘﻪ ﻗﻠﺒﯽ ﻳﺎ ﻣﻐﺰﯼ ﺷﻮﺩ. ﻧﺘﺎﻳﺞ ﺍﺳﺘﺮﺱ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﺳﻴﺴﺘﻢ ﮔﻮﺍﺭﺵ ﺷﻤﺎ ﻇﺎﻫﺮ ﺷﻮﺩ ﻭ ﺭﻭﻧﺪ ﺣﺮﮐﺖ ﻏﺬﺍ ﺩﺭ ﺭﻭﺩﻩ ﻳﺎ ﻣﻌﺪﻩ ﺷﻤﺎ ﺭﺍ ﮐﻨﺪﺗﺮ ﻳﺎ ﺳﺮﻳﻊ‌ﺗﺮ ﮐﻨﺪ.

ﭘﺲ ﺍﺳﺘﺮﺱ ﻭ ﻧﺘﺎﻳﺞ ﺁﻥ ﺑﺮ ﺯﻧﺪﮔﯽ ﺍﻳﻦ ﺭﻭﺯﻫﺎﯼ ﻣﺎ ﺑﻴﺸﺘﺮ ﺍﺯ ﭼﻴﺰﯼ ﺍﺳﺖ ﮐﻪ ﺗﺼﻮﺭ ﻣﯽ‌ﮐﻨﻴﻢ. ﺍﻣﺎ ﺍﮔﺮ ﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﺴﺘﻴﻢ ﺑﻪ ﺩﺍﺭﻭﻳﯽ ﺑﺮﺍﯼ ﺭﻳﺸﻪ‌ﮐﻨﯽ ﺍﺳﺘﺮﺱ ﺩﺳﺖ ﭘﻴﺪﺍ ﮐﻨﻴﻢ، ﺁﻳﺎ ﺍﻳﻦ ﺩﺍﺭﻭ ﺑﺮﺍﯼ ﻣﺎ ﻣﻔﻴﺪ ﺑﻮﺩ؟

ﺑﺎ ﺗﻤﺎﻡ ﺍﺷﮑﺎﻻﺗﯽ ﮐﻪ ﺩﺭﺑﺎﺭﻩ ﺍﺳﺘﺮﺱ ﮔﻔﺘﻴﻢ ﺑﺎﻳﺪ ﺩﺭ ﺟﻮﺍﺏ ﺍﻳﻦ ﺳﻮﺍﻝ ﺑﮕﻮﻳﻴﻢ ﻧﻪ! ﺑﻠﻪ ﺩﺭﺳﺖ ﺧﻮﺍﻧﺪﻳﺪ، ﻧﻪ! ﺑﺮﺍﯼ ﮐﻨﮑﺎﺵ ﺩﺭﺑﺎﺭﻩ ﭘﺎﺳﺦ ﺍﻳﻦ ﺳﻮﺍﻝ ﻳﮏ ﻟﺤﻈﻪ ﺭﺍ ﺗﺼﻮﺭ ﮐﻨﻴﺪ ﮐﻪ ﺑﻪ ﺩﻭﺭ ﺍﺯ ﺯﻧﺪﮔﯽ ﺍﻳﻦ ﺭﻭﺯﻫﺎﯼ ﺧﻮﺩ، ﺩﺭ ﺯﻣﺎﻥ ﺍﻧﺴﺎﻥ‌ﻫﺎﯼ ﻏﺎﺭ ﻧﺸﻴﻦ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﮐﻨﻴﺪ! ﺩﺭ ﺯﻣﺎﻧﯽ ﮐﻪ ﺍﺣﺘﻤﺎﻻ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻴﺪ ﻫﻴﭽﮕﻮﻧﻪ ﺍﺳﺘﺮﺳﯽ ﻭﺟﻮﺩ ﻧﺪﺍﺭﺩ.

ﺑﻪ ﺗﺼﻮﺭ ﺧﻮﺩ ﺍﺩﺍﻣﻪ ﺑﺪﻫﻴﺪ: ﺷﻤﺎ ﺑﺎ ﻳﮏ ﮔﺮﺯ ﺩﺭ ﺩﺳﺖ ﻭ ﺑﺎ ﻳﮏ ﻟﺒﺎﺳﯽ ﺍﺯ ﺟﻨﺲ ﭘﻮﺳﺖ ﺣﻴﻮﺍﻧﺎﺕ ﺩﺭ ﺣﺎﻝ ﻗﺪﻡ ﺯﺩﻥ ﺩﺭ ﺟﻨﮕﻞ ﻫﺴﺘﻴﺪ ﮐﻪ ﺑﻪ ﻳﮏ ﺑﺎﺭﻩ ﻳﮏ ﺑﺒﺮ ﺩﺭ ﻣﺴﻴﺮ ﺣﺮﮐﺖ ﺷﻤﺎ ﻇﺎﻫﺮ ﻣﯽ‌ﺷﻮﺩ. ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺑﺪﻥ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﻟﺤﻈﻪ ﺁﺩﺭﻧﺎﻟﻴﻦ ﺗﺮﺷﺢ ﻣﯽ‌ﮐﻨﺪ ﻭ ﺩﭼﺎﺭ ﺍﺳﺘﺮﺱ ﻣﯽ‌ﺷﻮﻳﺪ. ﺍﻳﻦ ﺍﺳﺘﺮﺱ ﺍﻟﺒﺘﻪ ﻳﮏ ﺍﺳﺘﺮﺱ ﮐﻮﺗﺎﻩ ﻣﺪﺕ ﻭ ﻣﻔﻴﺪ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﺍﺟﺎﺯﻩ ﻣﯽ‌ﺩﻫﺪ ﻃﯽ ﻳﮑﯽ ﺩﻭ ﺛﺎﻧﻴﻪ ﻓﺮﺍﺭ ﻳﺎ ﺟﻨﮓ ﺑﺎ ﺍﻳﻦ ﺑﺒﺮ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺍﮔﺮ ﺍﻳﻦ ﺁﺩﺭﻧﺎﻟﻴﻦ ﺩﺭ ﺑﺪﻥ ﺷﻤﺎ ﺗﺮﺷﺢ ﻧﻤﯽ‌ﺷﺪ ﻭ ﺷﻤﺎ ﺩﭼﺎﺭ ﺍﺳﺘﺮﺱ ﻧﻤﯽ‌ﺷﺪﻳﺪ، ﺑﺒﺮ ﺩﺍﺳﺘﺎﻥ، ﺷﻤﺎ ﺭﺍ ﻳﮏ ﻟﻘﻤﻪ ﭼﭗ ﻣﯽ‌ﮐﺮﺩ!

ﭘﺲ ﺗﺎ ﮐﻨﻮﻥ ﺑﻪ ﺍﻳﻦ ﻧﺘﻴﺠﻪ ﺭﺳﻴﺪﻩ‌ﺍﻳﻢ ﮐﻪ ﺍﺳﺘﺮﺱ ﺩﺭ ﺗﻤﺎﻡ ﺩﻭﺭﺍﻥ ﺯﻧﺪﮔﯽ ﺍﻧﺴﺎﻥ ﺑﺮ ﺭﻭﯼ ﮐﺮﻩ ﺯﻣﻴﻦ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﻭ ﺩﺭ ﺑﺮﺧﯽ ﺍﺯ ﻣﻮﺍﺭﺩ ﻧﻴﺰ ﺍﺯ ﺿﺮﻭﺭﻳﺎﺕ ﺯﻧﺪﮔﯽ ﺍﻭ ﺑﻮﺩﻩ ﺍﺳﺖ ﺍﻣﺎ ﺍﻳﻦ ﺭﻭﺯﻫﺎ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺳﺒﮏ ﺯﻧﺪﮔﯽ ﺟﺪﻳﺪ ﻣﺎ، ﺑﻴﺸﺘﺮ ﺍﺯ ﭘﻴﺶ ﺍﺣﺴﺎﺱ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﺮﺍﻳﻤﺎﻥ ﺩﺭﺩﺳﺮﺳﺎﺯ ﺷﺪﻩ ﺍﺳﺖ.

ﺩﺭ ﻭﺍﻗﻊ ﺩﺭﺩﺳﺮ ﺍﻳﻦ ﺭﻭﺯﻫﺎﯼ ﻣﺎ ﻧﺎﺷﯽ ﺍﺯ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﺪﻥ ﻣﺎ ﻫﻨﻮﺯ ﻣﺘﻮﺟﻪ ﻧﺸﺪﻩ ﻣﺎ ﺩﺭ ﺯﻧﺪﮔﯽ ﺍﻳﻦ ﺭﻭﺯﻫﺎ ﺩﺭ ﻣﻌﺮﺽ ﺧﻄﺮ ﺩﺭﻳﺪﻩ ﺷﺪﻥ ﺗﻮﺳﻂ ﻳﮏ ﺑﺒﺮ ﻧﻴﺴﺘﻴﻢ ﻭ ﻣﺸﮑﻼﺕ ﺍﻳﻦ ﺭﻭﺯﻫﺎﯼ ﻣﺎ ﻣﺸﮑﻼﺗﯽ ﺍﺟﺘﻤﺎﻋﯽ ﻫﺴﺘﻨﺪ! ﺑﻪ ﻫﻤﻴﻦ ﺩﻟﻴﻞ ﺑﺪﻥ ﻣﺎ ﻧﺘﻮﺍﻧﺴﺘﻪ ﻭﺍﮐﻨﺶ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﺍﻳﻦ ﻣﺸﮑﻼﺕ ﺭﺍ ﻫﻤﺎﻫﻨﮓ ﮐﻨﺪ ﻭ ﺩﺭ ﺍﻳﻦ ﺭﻭﺯﻫﺎ ﺩﺭ ﺑﺮﺍﺑﺮ ﻣﺸﮑﻼﺕ ﺍﺟﺘﻤﺎﻋﯽ ﻫﻤﺎﻧﻨﺪ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﺩﺭ ﺑﺮﺍﺑﺮ ﻳﮏ ﺑﺒﺮ ﺭﻓﺘﺎﺭ ﮐﺮﺩﻩ ﻭ ﻫﻤﺎﻧﮕﻮﻧﻪ ﻭﺍﮐﻨﺶ ﻧﺸﺎﻥ ﻣﯽ‌ﺩﻫﺪ.

ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺭﺍ ﻫﻢ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﻢ ﮐﻪ ﻭﺍﮐﻨﺶ ﺑﺪﻥ ﻣﺎ ﺑﻪ ﻣﺸﮑﻼﺕ ﺍﺟﺘﻤﺎﻋﯽ ﻳﺎ ﮐﺎﺭﯼ، ﺑﻴﺶ ﺍﺯ ﺩﻭﺭﺍﻥ ﻏﺎﺭ ﻧﺸﻴﻨﯽ ﺍﺳﺖ ﻭ ﻫﺮ ﺭﻭﺯﻩ ﺷﺎﻳﺪ ﺩﺭ ﭼﻨﺪﻳﻦ ﻧﻮﺑﺖ ﺑﺎﻳﺪ ﺑﻪ ﺍﻳﻨﮕﻮﻧﻪ ﻣﻮﺿﻮﻋﺎﺕ ﻭﺍﮐﻨﺶ ﻧﺸﺎﻥ ﺑﺪﻫﺪ ﮐﻪ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﻧﻴﺰ ﻗﻄﻌﺎ ﺧﻮﺏ ﻧﻴﺴﺖ.

ﺑﻪ ﭘﺎﻳﺎﻥ ﻧﺨﺴﺘﻴﻦ ﺟﻠﺴﻪ ﺍﻳﻦ ﺩﻭﺭﻩ ﺭﺳﻴﺪﻳﻢ. ﺩﺭ ﺍﻳﻦ ﺟﻠﺴﻪ ﻣﻔﻬﻮﻡ ﺍﺳﺘﺮﺱ ﺭﺍ ﻟﻤﺲ ﮐﺮﺩﻳﻢ ﻭ ﺑﺎ ﺍﺛﺮﺍﺕ ﻣﻨﻔﯽ ﻭ ﻣﺜﺒﺖ ﺁﻥ ﺑﺮ ﺑﺪﻥ ﻭ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﺁﺷﻨﺎ ﺷﺪﻳﻢ. ﺩﺭ ﺟﻠﺴﻪ ﺑﻌﺪ ﻗﺮﺍﺭ ﺍﺳﺖ ﺩﺭﺑﺎﺭﻩ ﻧﺤﻮﻩ ﺍﻳﺠﺎﺩ ﺍﺳﺘﺮﺱ ﺩﺭ ﺍﻓﺮﺍﺩ ﻭ ﺭﺍﺑﻄﻪ ﺁﻥ ﺑﺎ ﻧﺤﻮﻩ ﻧﮕﺮﺵ ﻓﺮﺩ ﺻﺤﺒﺖ ﮐﻨﻴﻢ ﻭ ﺑﺎ ﺭﺍﻫﮑﺎﺭﻫﺎﻳﯽ ﺁﺷﻨﺎ ﺷﻮﻳﻢ ﮐﻪ ﺑﻪ ﺻﻮﺭﺕ ﺫﻫﻨﯽ ﺑﺮﺍﯼ ﮐﺎﻫﺶ ﺍﺳﺘﺮﺱ ﻃﺮﺍﺣﯽ ﺷﺪﻩ‌ﺍﻧﺪ ﻭ ﺳﻌﯽ ﺩﺍﺭﻧﺪ ﺑﺮ ﭘﺎﻳﻪ ﺗﻤﺮﮐﺰ ﺫﻫﻨﯽ ﺍﺳﺘﺮﺱ ﺭﺍ ﮐﺎﻫﺶ ﺑﺪﻫﻨﺪ.
