ﺩﺭﺱ ﻧﻬﻢ- ﺭﻫﺎﻳﯽ ﺍﺯ ﺍﺳﺘﺮﺱ ﻣﺤﻞ ﮐﺎﺭ

ﺑﻪ ﺩﺭﺱ ﻧﻬﻢ ﺧﻮﺵ ﺁﻣﺪﻳﺪ. ﺩﺭ ﺍﻳﻦ ﺩﺭﺱ ﻗﺮﺍﺭ ﺍﺳﺖ ﺑﺎ ﻣﻮﺿﻮﻋﺎﺗﯽ ﺁﺷﻨﺎ ﺷﻮﻳﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺍﺳﺘﺮﺱ ﻣﺤﻞ ﮐﺎﺭ ﺧﻮﺩ ﺭﺍ ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﮐﺎﻫﺶ ﺩﻫﻴﺪ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻓﺮﺍﺩ ﻓﮑﺮ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺷﻐﻞ ﺁﻥ‌ﻫﺎ ﺗﻨﻬﺎ ﺷﻐﻞ ﭘﺮﺍﺳﺘﺮﺱ ﺭﻭﯼ ﺯﻣﻴﻦ ﺍﺳﺖ ﻭ ﺑﻴﺸﺘﺮﻳﻦ ﺍﺳﺘﺮﺱ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﺗﺠﺮﺑﻪ ﻣﯽ‌ﮐﻨﻨﺪ. ﺁﻥ‌ﻫﺎ ﺍﻟﺒﺘﻪ ﺣﻖ ﻫﻢ ﺩﺍﺭﻧﺪ، ﻳﮏ ﻣﺪﻳﺮ ﺳﺨﺖ‌ﮔﻴﺮ، ﻳﮏ ﺍﺭﺑﺎﺏ ﺭﺟﻮﻉ ﺑﺪﺍﺧﻼﻕ ﻭ ﻳﺎ ﺣﺘﯽ ﮐﺎﺭﻣﻨﺪﺍﻧﯽ ﺯﻳﺮﺁﺏ ﺯﻥ ﻫﻤﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﺁﺩﻡ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﺍﺳﺘﺮﺱ ﻭﺍﺭﺩ ﮐﻨﻨﺪ. ﺍﻣﺎ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﭼﮕﻮﻧﻪ ﺑﺎﻳﺪ ﺍﺳﺘﺮﺱ ﺭﺍ ﺑﻪ ﺣﺪﺍﻗﻞ ﺭﺳﺎﻧﺪ.

ﺍﺳﺘﺮﺱ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺎ ﻋﻼﻳﻤﯽ ﻫﻤﭽﻮﻥ ﻧﺪﺍﺷﺘﻦ ﺗﻤﺮﮐﺰ ﺩﺭ ﮐﺎﺭ، ﺧﺴﺘﻪ ﺷﺪﻥ ﺍﺯ ﺍﻧﺠﺎﻡ ﮐﺎﺭ، ﮐﻢ ﺷﺪﻥ ﺑﺎﺯﺩﻫﯽ، ﺑﺪﺧﻠﻘﯽ ﺩﺭ ﻣﺤﻴﻂ ﮐﺎﺭ ﻭ ﻳﺎ ﺟﺪﯼ ﺑﻮﺩﻥ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﺧﻮﺩ ﺭﺍ ﻧﺸﺎﻥ ﺩﻫﺪ. 

ﭘﻴﺶ ﺍﺯ ﺁﻧﮑﻪ ﺩﺭﺑﺎﺭﻩ ﻧﺤﻮﻩ ﺑﺮﺧﺮﻭﺩ ﺑﺎ ﺍﺳﺘﺮﺱ ﻣﺤﻞ ﮐﺎﺭ ﺗﻮﺿﻴﺢ ﺩﻫﻴﻢ ﺑﻬﺘﺮ ﺍﺳﺖ ﺑﻪ ﺩﻧﺒﺎﻝ ﭘﻴﺪﺍ ﮐﺮﺩﻥ ﻣﻮﺿﻮﻋﯽ ﺑﺎﺷﻴﻢ ﮐﻪ ﺑﻴﺶ ﺍﺯ ﻫﻤﻪ ﺳﺒﺐ ﺍﻳﺠﺎﺩ ﺍﺳﺘﺮﺱ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﺭﻭﯼ ﺷﻤﺎ ﺷﺪﻩ ﺍﺳﺖ. ﺍﻳﻦ ﻋﺎﻣﻞ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺳﺎﻋﺎﺕ ﮐﺎﺭﯼ ﻃﻮﻻﻧﯽ، ﻣﺴﺌﻮﻟﻴﺖ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﺯﻳﺎﺩ ﻳﺎ ﮐﻢ، ﺩﺭﺁﻣﺪ ﻧﺎﮐﺎﻓﯽ،  ﻣﺤﻴﻂ ﮐﺎﺭﯼ ﻧﺎﻣﻨﺎﺳﺐ ﻭ ﻳﺎ ﻣﺸﮑﻼﺗﯽ ﺑﺎ ﺭﺋﻴﺲ/ﻣﺸﺘﺮﻳﺎﻥ/ ﮐﺎﺭﻣﻨﺪﺍﻥ ﺷﺮﮐﺖ ﺑﺎﺷﺪ. ﭘﺲ ﺍﺯ ﺷﻨﺎﺧﺖ ﻣﻨﺒﻊ ﺍﺻﻠﯽ ﺍﺳﺘﺮﺱ ﺧﻮﺩ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ، ﺍﺯ ﺧﻮﺩ ﺑﭙﺮﺳﻴﺪ ﮐﻪ ﭼﮕﻮﻧﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻳﻦ ﺍﺳﺘﺮﺱ ﺭﺍ ﮐﺎﻫﺶ ﺩﻫﻴﺪ. ﺷﺎﻳﺪ ﺍﻳﻦ ﺍﺳﺘﺮﺱ ﺑﺎ ﺻﺤﺒﺖ ﮐﺮﺩﻥ ﻣﻨﺎﺳﺐ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﻣﺮﺗﻔﻊ ﺷﻮﺩ ﻭ ﻳﺎ ﺑﻪ ﺷﻴﻮﻩ‌ﻫﺎﯼ ﭘﻴﭽﻴﺪﻩ‌ﺗﺮﯼ ﺑﺮﺍﯼ ﺣﻞ ﺁﻥ ﻧﻴﺎﺯ ﺑﺎﺷﺪ. 

ﺑﻴﺎﻳﻴﺪ ﻳﮏ ﺭﻭﺯ ﮐﺎﺭﯼ ﺭﺍ ﺑﺎ ﻫﻢ ﻣﺮﻭﺭ ﮐﻨﻴﻢ ﻭ ﺍﺯ ﻫﻤﺎﻥ ﻟﺤﻈﻪ ﺍﻭﻝ ﺷﺮﻭﻉ ﮐﻨﻴﻢ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺷﻤﺎ ﺍﺳﺘﺮﺱ ﺑﺪﻫﺪ. ﺷﻤﺎ ﺻﺒﺢ ﺍﺯ ﺧﻮﺍﺏ ﺑﻴﺪﺍﺭ ﺷﺪﻩ ﻭ ﺑﻪ ﻣﺤﻞ ﮐﺎﺭ ﺭﻓﺘﻪ‌ﺍﻳﺪ ﻭ ﺩﺭ ﻫﻤﻴﻦ ﺍﻭﻝ ﮐﺎﺭ، ﺍﺳﺘﺮﺱ ﺩﺍﺭﻳﺪ. ﺍﻳﻦ ﺍﺳﺘﺮﺱ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻪ ﺩﻟﻴﻞ ﺧﻮﺍﺏ ﻧﺎﮐﺎﻓﯽ ﺷﺒﺎﻧﻪ ﻭ ﻳﺎ ﻧﺨﻮﺭﺩﻥ ﺻﺒﺤﺎﻧﻪ ﺑﺎﺷﺪ. ﺍﮔﺮ ﺧﻮﺍﺏ ﮐﺎﻓﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻭ ﺻﺒﺢ‌ﻫﺎ ﻫﻢ ﮐﻤﯽ ﺯﻭﺩﺗﺮ ﺑﻴﺪﺍﺭ ﺷﻮﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻴﺶ‌ﺗﺮ ﺍﺯ ﮔﺬﺷﺘﻪ ﺍﺣﺴﺎﺱ ﺧﻮﺑﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻭ ﺍﺳﺘﺮﺱ ﺷﺮﻭﻉ ﮐﺎﺭ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﮐﻨﻴﺪ. ﺧﻮﺭﺩﻥ ﺻﺒﺤﺎﻧﻪ ﺩﺭ ﺧﺎﻧﻪ ﻧﻴﺰ ﺑﻬﺘﺮ ﺍﺯ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﺻﺒﺤﺎﻧﻪ ﺑﺨﻮﺭﻳﺪ ﻭ ﺳﺒﺐ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺩﺭ ﺍﻭﻝ ﺻﺒﺢ ﺑﺎ ﺗﻤﺮﮐﺰ ﺑﻬﺘﺮﯼ ﮐﺎﺭ ﺧﻮﺩ ﺭﺍ ﺷﺮﻭﻉ ﮐﻨﻴﺪ.

ﺷﻤﺎ ﺣﺘﯽ ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺩﺭ ﻣﺴﻴﺮ ﺭﺳﻴﺪﻥ ﺑﻪ ﻣﺤﻞ ﮐﺎﺭ ﻗﺮﺍﺭ ﺩﺍﺭﻳﺪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﮔﻮﺵ ﺩﺍﺩﻥ ﻳﮏ ﻣﻮﺳﻴﻘﯽ ﺧﻮﺏ ﺧﻮﺩ ﺭﺍ ﺑﺮﺍﯼ ﮐﺎﺭ ﺁﻣﺎﺩﻩ ﮐﻨﻴﺪ.  ﺍﮔﺮ ﺩﺭ ﭘﺸﺖ ﭼﺮﺍﻍ ﻗﺮﻣﺰ ﮔﻴﺮ ﮐﺮﺩﻳﺪ ﻭ ﻳﺎ ﺩﺭ ﺗﺮﺍﻓﻴﮏ ﻣﺎﻧﺪﻳﺪ ﻭ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﻪ ﺷﻤﺎ ﺍﺳﺘﺮﺱ ﻭﺍﺭﺩ ﮐﺮﺩ، ﺍﺯ ﺗﮑﻨﻴﮏ ﺗﻨﻔﺲ ﺩﺭ ﺧﻨﺪﻕ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﻭ ﻳﺎ ﺩﺭ ﭘﺸﺖ ﻓﺮﻣﺎﻥ، ﺳﻌﯽ ﮐﻨﻴﺪ ﺷﺎﻧﻪ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺗﺎ ﮔﻮﺵ‌ﻫﺎﻳﺘﺎﻥ ﺑﺎﻻ ﺑﻴﺎﻭﺭﻳﺪ ﻭ ﺍﻳﻦ ﮐﺸﺶ ﺭﺍ ﺳﻪ ﺗﺎ ﭼﻬﺎﺭ ﺛﺎﻧﻴﻪ ﻧﮕﺎﻩ ﺩﺍﺭﻳﺪ ﻭ ﺑﻌﺪ ﺭﻫﺎ ﮐﻨﻴﺪ. ﮐﻒ ﺩﺳﺖ‌ﻫﺎﻳﺘﺎﻥ ﺩﺭ ﺍﻳﻦ ﺣﺎﻟﺖ ﺑﺎﻳﺪ ﺭﻭﯼ ﻓﺮﻣﺎﻥ ﺑﻮﺩﻩ ﻭ ﻓﺮﻣﺎﻥ ﺍﺗﻮﻣﺒﻴﻞ ﺭﺍ ﻓﺸﺎﺭ ﺩﻫﻨﺪ.

ﺑﺮﺍﯼ ﮐﻨﺘﺮﻝ ﺍﺳﺘﺮﺱ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﻧﻴﺰ ﺗﮑﻨﻴﮏ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﮐﺎﺭ ﮔﺮﻓﺖ. ﺑﺎﻳﺪ ﺗﻮﺟﻪ ﮐﻨﻴﺪ ﮐﻪ ﺍﻳﻦ ﺗﮑﻨﻴﮏ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﻣﻨﻈﻢ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ ﺗﺎ ﺍﺯ ﺗﺠﻤﻊ ﺍﺳﺘﺮﺱ ﻭ ﻓﺸﺎﺭ ﺑﻴﺶ ﺍﺯ ﺣﺪ ﺁﻥ ﺭﻭﯼ ﺷﻤﺎ ﺧﻮﺩﺩﺍﺭﯼ ﮐﻨﻴﺪ. ﺗﮑﻨﻴﮏ ﭘﺮﺕ ﮐﺮﺩﻥ ﺣﻮﺍﺱ ﻳﺎ ﺗﻨﻔﺲ ﺩﺭ ﺧﻨﺪﻕ ﺑﻪ ﻣﺎﻧﻨﺪ ﻫﻤﻴﺸﻪ ﺩﺭ ﺍﻳﻨﺠﺎ ﻧﻴﺰ ﮐﺎﺭﺳﺎﺯ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ. ﺍﻳﻦ ﺗﮑﻨﻴﮏ‌ﻫﺎ ﺭﺍ ﻣﺜﻼ ﻭﻗﺘﯽ ﮐﻪ ﻣﺸﺘﺮﯼ/ﺍﺭﺑﺎﺏ ﺭﺟﻮﻉ ﺩﺭ ﺣﺎﻝ ﺗﺮﮎ ﺍﺗﺎﻕ ﺷﻤﺎﺳﺖ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺟﺮﺍ ﮐﻨﻴﺪ.

ﺍﻳﺴﺘﺎﺩﻥ ﻭ ﺣﺮﮐﺖ ﮐﺮﺩﻥ ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﺑﺮﺍﻳﺘﺎﻥ ﻣﻔﻴﺪ ﺍﺳﺖ. ﺳﻌﻴﺪ ﮐﻨﻴﺪ ﺣﺘﯽ ﺍﮔﺮ ﺭﻭﺯ ﭘﺮ ﮐﺎﺭﯼ ﺭﺍ ﻣﯽ‌ﮔﺬﺭﺍﻧﻴﺪ، ﭼﻨﺪ ﺩﻗﻴﻘﻪ‌ﺍﯼ ﺭﺍ ﺑﻪ ﺍﻳﺴﺘﺎﺩﻥ ﻭ ﺩﻭﺭ ﺷﺪﻥ ﺍﺯ ﻣﻴﺰ ﮐﺎﺭ ﺧﻮﺩ ﺍﺧﺘﺼﺎﺹ ﺩﻫﻴﺪ. ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺑﻪ ﮐﻤﺘﺮ ﺷﺪﻥ ﺍﺳﺘﺮﺱ ﺷﻤﺎ ﮐﻤﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ.

ﮐﺶ ﺩﺍﺩﻥ ﺑﺪﻥ ﺩﺭ ﻫﻨﮕﺎﻡ ﮐﺎﺭ ﻧﻴﺰ ﺍﺯ ﺣﺮﮐﺖ‌ﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﺣﺴﯽ ﺷﺒﻴﻪ ﺑﻪ ﺍﻳﺴﺘﺎﺩﻥ ﻭ ﺩﻭﺭ ﺷﺪﻥ ﺍﺯ ﻣﻴﺰ ﮐﺎﺭ ﺭﺍ ﺗﺠﺮﺑﻪ ﮐﻨﻴﺪ. ﻧﺸﺴﺘﻦ ﻃﻮﻻﻧﯽ ﻣﺪﺕ ﺍﻳﻦ ﺣﺲ ﺭﺍ ﺑﻪ ﺑﺪﻥ ﺷﻤﺎ ﺍﻟﻘﺎ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺟﺎﻳﯽ ﺣﺒﺲ ﻫﺴﺘﻴﺪ ﻭ ﺍﻳﺴﺘﺎﺩﻥ ﻳﺎ ﮐﺶ ﺩﺍﺩﻥ ﺑﺪﻥ ﺑﻪ ﺁﻥ ﻣﯽ‌ﻓﻬﻤﺎﻧﺪ ﮐﻪ ﺍﻳﻨﭽﻨﻴﻦ ﻧﻴﺴﺖ.

ﺑﺮﺧﯽ ﺍﺯ ﮐﺸﺶ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﺮﺍﯼ ﺑﺪﻧﺘﺎﻥ ﻣﻔﻴﺪ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﻋﺒﺎﺭﺗﺴﺖ ﺍﺯ ﻣﺪﻝ ﮔﻴﻼﺱ ﭼﻴﻦ ﻭ ﻣﺪﻝ ﻓﺸﺮﺩﻥ ﻣﺎﻫﻴﭽﻪ‌ﻫﺎﯼ ﺻﺪﺭﯼ. ﺩﺭ ﻣﺪﻝ ﮔﻴﻼﺱ ﭼﻴﻦ، ﺷﻤﺎ ﺑﺎﻳﺪ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺭﻭﯼ ﺻﻨﺪﻟﯽ ﻧﺸﺴﺘﻪ‌ﺍﻳﺪ، ﻳﮑﯽ ﺍﺯ ﺩﺳﺘﺎﻧﺘﺎﻥ ﺭﺍ ﺑﺎﻻ ﺑﺒﺮﻳﺪ ﺑﻪ ﻃﻮﺭﻳﻪ ﮐﻪ ﻧﻮﮎ ﺍﻧﮕﺸﺘﺎﻧﺘﺎﻥ ﺭﻭ ﺑﻪ ﺳﻘﻒ ﺑﺎﺷﺪ. ﺩﺭ ﺍﻳﻦ ﻣﺪﻝ ﺷﻤﺎ ﻓﺮﺽ ﻣﯽ‌ﮐﻨﻴﺪ ﮐﻪ ﺩﺭ ﺣﺎﻝ ﭼﻴﺪﻥ ﮔﻴﻼﺱ ﻫﺴﺘﻴﺪ ﻭ ﮔﻴﻼﺱ ﭼﻨﺪ ﺳﺎﻧﺘﯽ‌ﻣﺘﺮﯼ ﺑﺎ ﺩﺳﺘﺎﻥ ﺷﻤﺎ ﻓﺎﺻﻠﻪ ﺩﺍﺭﺩ ﻭ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﺑﺎﻳﺪ ﺩﺳﺘﺎﻥ ﺧﻮﺩ ﺭﺍ ﺑﺎ ﻓﺸﺎﺭ ﺑﺎﻻﺗﺮ ﺑﺒﺮﻳﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﮔﻴﻼﺱ ﺭﺍ ﺑﭽﻴﻨﻴﺪ. ﺍﻳﻦ ﺣﺎﻟﺖ ﮐﺸﺶ ﺑﺎﻳﺪ ﺩﻭ ﺗﺎ ﺳﻪ ﺛﺎﻧﻴﻪ ﺣﻔﻆ ﺷﻮﺩ ﻭ ﺳﭙﺲ ﺩﺳﺖ ﺭﺍ ﭘﺎﻳﻴﻦ ﺁﻭﺭﺩﻩ ﻭ ﺑﺎ ﺩﺳﺖ ﺩﻳﮕﺮ ﺍﻳﻦ ﺣﺮﮐﺖ ﺭﺍ ﺗﻤﺮﻳﻦ ﮐﻨﻴﺪ.

ﺩﺭ ﻣﺪﻝ ﻓﺸﺮﺩﻥ ﻣﺎﻫﻴﭽﻪ‌ﻫﺎﯼ ﺻﺪﺭﯼ ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﺍﻳﺴﺘﺎﺩﻩ ﻳﺎ ﻧﺸﺴﺘﻪ‌ﺍﻳﺪ، ﺍﻧﮕﺸﺘﺎﻥ ﻫﺮ ﺩﻭ ﺩﺳﺖ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻫﻢ ﻓﺮﻭ ﺑﺮﺩﻩ ﻭ ﭘﺸﺖ ﺳﺮﺗﺎﻥ ﺑﮕﺬﺍﺭﻳﺪ. ﺳﭙﺲ ﺁﺭﻧﺞ‌ﻫﺎﻳﺘﺎﻥ ﺭﺍ ﺗﺎ ﺁﻧﺠﺎ ﮐﻪ ﺑﺮﺍﻳﺘﺎﻥ ﺍﻣﮑﺎﻥ ﺩﺍﺭﺩ، ﺑﻪ ﭘﺸﺖ ﺑﮑﺸﻴﺪ. ﺍﻳﻦ ﺣﺮﮐﺖ ﺳﺒﺐ ﺭﻫﺎ ﮐﺮﺩﻥ ﮔﺮﻓﺘﮕﯽ ﻋﻀﻼﺕ ﺩﺭ ﻗﺴﻤﺖ ﺳﻴﻨﻪ ﻭ ﭘﺸﺖ ﮐﺘﻒ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺣﺲ ﺧﻮﺑﯽ ﺭﺍ ﺑﻪ ﺷﻤﺎ ﺍﻧﺘﻘﺎﻝ ﺧﻮﺍﻫﺪ ﺩﺍﺩ.

ﻓﻀﺎﯼ ﮐﺎﺭﯼ ﺷﻤﺎ ﺩﺭ ﻣﺤﻴﻂ ﮐﺎﺭ ﺑﻬﺘﺮ ﺍﺳﺖ ﻣﻨﻈﻢ ﺑﺎﺷﺪ ﭼﺮﺍ ﮐﻪ ﺩﺭﻫﻢ ﺭﻳﺨﺘﮕﯽ ﻭﺳﺎﻳﻞ ﺷﻤﺎ ﺭﻭﯼ ﻣﻴﺰ ﮐﺎﺭ ﺳﺒﺐ ﺍﻳﺠﺎﺩ ﺷﻠﺨﺘﮕﯽ ﻭ ﺳﺮﺩﺭﮔﻤﯽ ﻭ ﻧﻬﺎﻳﺘﺎ ﺍﺳﺘﺮﺱ ﺩﺭ ﻟﺤﻈﺎﺗﯽ ﺧﻮﺍﻫﺪ ﺷﺪ ﮐﻪ ﻧﻴﺎﺯ ﻣﺒﺮﻡ ﺑﻪ ﭘﺎﺳﺦ ﺳﺮﻳﻊ ﺩﺍﺭﻳﺪ. ﺩﺭ ﻣﺤﻞ ﮐﺎﺭ ﺧﻮﺩ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺩﺍﺷﺘﻦ ﻳﮏ ﺍﺳﭙﺮﯼ ﺍﺭﺯﺍﻥ ﻗﻴﻤﺖ، ﺑﻮﻳﯽ ﺭﺍ ﮐﻪ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﭘﺨﺶ ﮐﻨﻴﺪ ﺗﺎ ﺣﺎﻟﺘﯽ ﺧﺎﺹ ﺑﻪ ﺩﻓﺘﺮ ﺧﻮﺩ ﺑﺒﺨﺸﻴﺪ. (ﺍﻟﺒﺘﻪ ﺑﺎﻳﺪ ﺍﻃﻤﻴﻨﺎﻥ ﺣﺎﺻﻞ ﮐﻨﻴﺪ ﺍﻳﻦ ﺑﻮ ﺩﻳﮕﺮ ﻫﻤﮑﺎﺭﺍﻥ‌ﺗﺎﻥ ﺭﺍ ﺍﺫﻳﺖ ﻧﮑﻨﺪ.)

ﺧﻮﺭﺩﻥ ﻳﮏ ﻣﻴﺎﻥ ﻭﻋﺪﻩ ﻫﻤﺎﻧﻨﺪ ﻣﻐﺰ ﺁﺟﻴﻞ ﻭ ﻳﺎ ﻳﮏ ﮐﻴﮏ ﮐﻢ ﺷﻴﺮﻳﻨﯽ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﺗﺎ ﻋﻼﻭﻩ ﺑﺮ ﮐﻤﯽ ﺍﺳﺘﺮﺍﺣﺖ، ﺍﻧﺮﮊﯼ ﻻﺯﻡ ﺭﻭﺯﺍﻧﻪ‌ﺗﺎﻥ ﺭﺍ ﻧﻴﺰ ﺑﻪ ﺩﺳﺖ ﺑﻴﺎﻭﺭﻳﺪ. ﺩﺭ ﺧﻮﺭﺩﻥ ﻣﻴﺎﻥ ﻭﻋﺪﻩ ﺍﻟﺒﺘﻪ ﻧﺒﺎﻳﺪ ﺯﻳﺎﺩﻩ ﺭﻭﯼ ﮐﻨﻴﺪ ﭼﺮﺍﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺳﺒﺐ ﮐﻢ‌ﺍﺷﺘﻬﺎﻳﯽ ﺷﻤﺎ ﺩﺭ ﻭﻋﺪﻩ ﻧﺎﻫﺎﺭ ﺷﻮﺩ ﻭ ﺍﻳﻦ ﻣﻮﺿﻮﻉ ﺍﺻﻼ ﻣﻨﺎﺳﺐ ﻧﻴﺴﺖ.

ﺩﺭ ﭘﺎﻳﺎﻥ ﮐﺎﺭ ﻫﻢ ﺑﺎﻳﺪ ﺑﻪ ﮔﻮﻧﻪ‌ﺍﯼ ﺑﺎﺷﻴﺪ ﮐﻪ ﺍﺳﺘﺮﺱ ﻣﺤﻞ ﮐﺎﺭ ﺭﺍ ﺑﻪ ﺧﺎﻧﻪ ﻧﺒﺮﻳﺪ، ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺑﻬﺘﺮ ﺍﺳﺖ ﺩﺭ ﭘﺎﻳﺎﻥ ﮐﺎﺭ ﻧﻴﺰ ﺣﺮﮐﺎﺕ ﮐﺸﺸﯽ ﮔﻔﺘﻪ ﺷﺪﻩ ﺭﺍ ﺗﮑﺮﺍﺭ ﮐﻨﻴﺪ ﻭ ﺗﺎ ﺣﺪ ﺍﻣﮑﺎﻥ ﻫﻴﭻ ﮐﺎﺭﯼ ﺭﺍ ﺍﺯ ﻣﺤﻞ ﮐﺎﺭ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻥ ﺑﻪ ﺧﺎﻧﻪ ﻧﺒﺮﻳﺪ. ﻫﻨﮕﺎﻣﯽ ﮐﻪ ﺑﻪ ﺧﺎﻧﻪ ﺭﺳﻴﺪﻳﺪ ﻧﻴﺰ ﺑﻬﺘﺮ ﺍﺳﺖ ﻳﮏ ﺩﻭﺵ ﺁﺏ ﮔﺮﻡ ﺑﮕﻴﺮﻳﺪ ﺗﺎ ﺧﺴﺘﮕﯽ ﻭ ﺍﺳﺘﺮﺱ ﻣﺤﻞ ﮐﺎﺭﺗﺎﻥ ﮐﻢ ﺷﻮﺩ ﻭ ﻳﮏ ﻧﻮﺷﻴﺪﻧﯽ ﺑﺨﻮﺭﻳﺪ.
