ﺩﺭﺱ ﺷﺸﻢ- ﻣﻮﺳﻴﻘﯽ ﻗﺮﻥ ﺑﻴﺴﺘﻢ

ﺩﺭ ﻧﻴﻤﻪ ﺍﻭﻝ ﺳﺪﻩ ﺑﻴﺴﺘﻢ (۱۹۰۰ ﺗﺎ ۱۹۵۰) ﺳﺒﮏ ﻫﺎﯼ ﻣﺘﻨﻮﻋﯽ ﺍﺯ ﻣﻮﺳﻴﻘﯽ ﻭﺟﻮﺩ ﺩﺍﺷﺖ. ﺗﻔﺎﻭﺕ ﻣﻴﺎﻥ ﺳﺒﮏ ﻫﺎ ﭼﻨﺎﻥ ﺷﺪﻳﺪ ﺑﻮﺩﻩ ﺍﺳﺖ ﮐﻪ ﮔﻮﻳﯽ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﺯﺑﺎﻥ ﻫﺎﯼ ﻣﻮﺳﻴﻘﺎﻳﯽ ﺳﺮﺍﺳﺮ ﻣﺘﻔﺎﻭﺗﯽ ﺭﺍ ﺩﺭ ﺁﺛﺎﺭ ﺧﻮﺩ ﺑﻪ ﮐﺎﺭ ﮔﺮﻓﺘﻪ ﺍﻧﺪ. ﺣﺘﯽ ﺩﺭ ﺳﺒﮏ ﻫﺮ ﻳﮏ ﺍﺯ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﻧﻴﺰ ﺩﮔﺮﮔﻮﻧﯽ ﻫﺎﻳﯽ ﺍﺳﺎﺳﯽ ﺭﺥ ﺩﺍﺩﻩ ﺍﺳﺖ.

ﺗﻨﻮﻉ ﻋﻈﻴﻢ ﺳﺒﮏ ﻫﺎﯼ ﻣﻮﺳﻴﻘﯽ، ﺑﺎﺯﺗﺎﺑﯽ ﺍﺯ ﺗﻨﻮﻉ ﻭ ﺭﻧﮕﺎﺭﻧﮕﯽ ﺯﻧﺪﮔﯽ ﺩﺭ ﺁﻏﺎﺯ ﺳﺪﻩ ﺑﻴﺴﺘﻢ ﺍﺳﺖ. ﻭﺟﻮﺩ ﺭﺍﺩﻳﻮ، ﺳﻴﻨﻤﺎ، ﺍﺗﻮﻣﺒﻴﻞ، ﻫﻮﺍﭘﻴﻤﺎ، ﺟﻬﺎﻥ ﺭﺍ ﺑﺮﺍﯼ ﺍﻧﺴﺎﻥ ﺩﺳﺖ ﻳﺎﻓﺘﻨﯽ ﺗﺮ ﻭ ﻗﻠﻤﺮﻭ ﺗﺠﺮﺑﻪ ﻫﺎﯼ ﺍﻭ ﺭﺍ ﭘﻬﻨﺎﻭﺭﺗﺮ ﮐﺮﺩ.

ﻋﻨﺎﺻﺮ ﻣﻮﺳﻴﻘﯽ ﻣﺤﻠﯽ ﻭ ﻣﺮﺩﻡ ﭘﺴﻨﺪ ﺍﻏﻠﺐ ﺑﺎ ﺳﺒﮏ ﺷﺨﺼﯽ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﺳﺪﻩ ﺑﻴﺴﺘﻢ ﺩﺭﻫﻢ ﺁﻣﻴﺨﺘﻪ ﺍﺳﺖ. ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﺑﻪ ﻭﻳﮋﻩ ﻣﺠﺬﻭﺏ ﺭﻳﺘﻢ ﻫﺎ، ﺻﺪﺍﻫﺎ ﻭ ﺍﻟﮕﻮﻫﺎﯼ ﻣﻠﻮﺩﻳﮏ ﻏﻴﺮﺭﺳﻤﯽ ﺑﻮﺩﻧﺪ ﮐﻪ ﺑﺎ ﺳﺒﮏ ﻭ ﺳﻴﺎﻕ ﻣﻌﻤﻮﻝ ﻣﻮﺳﻴﻘﯽ ﻏﺮﺏ ﻣﻐﺎﻳﺮﺕ ﺩﺍﺷﺖ.

ﺩﺭ ﮔﺬﺭ ﺍﻭﺍﻳﻞ ﺳﺪﻩ ﺑﻴﺴﺘﻢ، ﻣﻮﺳﻴﻘﯽ ﻏﻴﺮﺍﺭﻭﭘﺎﻳﯽ ﺗﺎﺛﻴﺮﯼ ﮊﺭﻑ ﺑﺮ ﻣﻮﺳﻴﻘﯽ ﻏﺮﺏ ﮔﺬﺍﺷﺖ. ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﻭ ﻧﻘﺎﺷﺎﻥ، ﺑﻴﺶ ﺍﺯ ﭘﻴﺶ ﭘﺬﻳﺮﺍ ﻭ ﻫﻢ ﻧﻮﺍﯼ ﻓﺮﻫﻨﮓ ﻫﺎﯼ ﺁﺳﻴﺎﻳﯽ ﻭ ﺁﻓﺮﻳﻘﺎﻳﯽ ﺑﻮﺩﻧﺪ. ﺟﺎﺯ ﺁﻣﺮﻳﮑﺎﻳﯽ ﻳﮑﯽ ﺩﻳﮕﺮ ﺍﺯ ﻣﻨﺎﺑﻊ ﻏﻴﺮﺍﺭﻭﭘﺎﻳﯽ ﺗﺎﺛﻴﺮﮔﺬﺍﺭ ﺑﺮ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﺳﺪﻩ ﺑﻴﺴﺘﻢ ﺑﻮﺩ. ﻣﻮﺳﻴﻘﻴﺪﺍﻧﺎﻥ ﻣﺴﺤﻮﺭ ﺭﻳﺘﻢ ﻫﺎﯼ ﺳﻨﮑﻮﭘﺪﺍﺭ، ﺑﺪﺍﻫﻪ ﭘﺮﺩﺍﺯﯼ ﻭ ﻧﻴﺰ ﺭﻧﮓ ﻫﺎﯼ ﺻﻮﺗﯽ ﺑﯽ ﻫﻤﺘﺎﯼ ﮔﺮﻭﻩ ﻫﺎﯼ ﺟﺎﺯ ﺑﻮﺩﻧﺪ.

ﺩﺭ ﻳﮏ ﮔﺮﻭﻩ ﺟﺎﺯ، ﺑﺮﺧﻼﻑ ﺍﺭﮐﺴﺘﺮ ﺳﻤﻔﻮﻧﻴﮏ ﮐﻪ ﺳﺎﺯﻫﺎﯼ ﺯﻫﯽ ﺑﺮ ﺁﻥ ﺳﻴﻄﺮﻩ ﺩﺍﺭﻧﺪ، ﺳﺎﺯﻫﺎﯼ ﺑﺎﺩﯼ ﭼﻮﺑﯽ، ﺑﺎﺩﯼ ﺑﺮﻧﺠﯽ ﻭ ﮐﻮﺑﻪ ﺍﯼ ﺑﺮﺟﺴﺘﻪ ﺗﺮ ﻫﺴﺘﻨﺪ. ﺩﺭ ﺩﺭﺱ ﺁﻳﻨﺪﻩ ﺑﻪ ﺩﺳﺘﻪ ﺑﻨﺪﯼ ﻭ ﺷﻨﺎﺧﺖ ﺳﺎﺯ ﻫﺎ ﺧﻮﺍﻫﻴﻢ ﭘﺮﺩﺍﺧﺖ.

ﺑﺮﺧﯽ ﺁﺛﺎﺭ ﻣﺪﺭﻥ ﺷﺎﻳﺪ ﺍﻟﻬﺎﻡ ﮔﺮﻓﺘﻪ ﺍﺯ ﻣﻮﺳﻴﻘﯽ ﻗﺪﻳﻢ ﺑﺎﺷﻨﺪ، ﺍﻣﺎ ﻣﻌﻨﺎﻳﺶ ﺁﻥ ﻧﻴﺴﺖ ﮐﻪ ﺍﻳﻦ ﺁﺛﺎﺭ ﺗﻘﻠﻴﺪﯼ ﻣﺤﺾ ﺍﺯ ﺳﺒﮏ ﻫﺎﯼ ﭘﻴﺸﻴﻦ ﻫﺴﺘﻨﺪ. ﺩﺭ ﺍﻳﻦ ﮔﻮﻧﻪ ﺁﺛﺎﺭ، ﻳﮏ ﻓﺮﻡ ﻗﺪﻳﻤﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺎ ﭼﻨﺎﻥ ﻫﺎﺭﻣﻮﻧﯽ ﻫﺎ، ﺭﻳﺘﻢ ﻫﺎ، ﻣﻠﻮﺩﯼ ﻫﺎ ﻭ ﺭﻧﮓ ﺁﻣﻴﺰﯼ ﻫﺎﯼ ﺻﻮﺗﯽ ﺍﯼ ﺑﻪ ﮐﺎﺭ ﮔﺮﻓﺘﻪ ﺷﻮﺩ ﮐﻪ ﭘﻴﺶ ﺍﺯ ﺳﺪﻩ ﺑﻴﺴﺘﻢ ﺩﺭ ﺗﺼﻮﺭ ﻧﻤﯽ ﮔﻨﺠﻴﺪﻩ ﺍﺳﺖ.

 

ﺳﺒﮏ ﻫﺎﯼ ﻫﻨﺮﯼ ﻗﺮﻥ ﺑﻴﺴﺘﻢ
ﺍﻣﭙﺮﺳﻴﻮﻧﻴﺴﻢ
ﺁﺳﺘﺎﻧﻪ ﺁﻏﺎﺯ ﺳﺪﻩ ﺑﻴﺴﺘﻢ ﺩﻭﺭﺍﻥ ﻫﻤﺰﻳﺴﺘﯽ ﺳﺒﮏ ﻫﺎﯼ ﮔﻮﻧﺎﮔﻮﻥ ﻣﻮﺳﻴﻘﺎﻳﯽ ﺑﻮﺩ. ﻣﻬﻤﺘﺮﻳﻦ ﺍﻳﻦ ﺳﺒﮏ ﻫﺎ ﺍﻣﭙﺮﺳﻴﻮﻧﻴﺴﻢ ﺑﻮﺩ، ﺳﺒﮑﯽ ﮐﻪ ﺑﻪ ﺭﺳﺎﺗﺮﻳﻦ ﺷﮑﻞ ﺩﺭ ﺁﺛﺎﺭ ﮐﻠﻮﺩ ﺩُﺑﻮﺳﯽ ﺁﻫﻨﮕﺴﺎﺯ ﻓﺮﺍﻧﺴﻮﯼ ﺑﻪ ﺑﻴﺎﻥ ﺩﺭﺁﻣﺪ. ﺍﻳﻦ ﻣﮑﺘﺐ ﺍﺯ ﻧﻘﺎﺷﯽ ﺍﻣﭙﺮﺳﻴﻮﻧﻴﺴﻢ ﻭ ﺑﻪ ﻭﻳﮋﻩ ﺍﺩﺑﻴﺎﺕ ﺳﻤﺒﻮﻟﻴﺴﻢ ﺗﺎﺛﻴﺮ ﭘﺬﻳﺮﻓﺘﻪ ﺍﺳﺖ. ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﺍﻣﭙﺮﺳﻴﻮﻧﻴﺴﻢ ﻫﻤﺎﻧﮕﻮﻧﻪ ﺑﺎ ﻫﺎﺭﻣﻮﻧﯽ ﻭ ﺭﻧﮓ ﻫﺎﯼ ﺻﻮﺗﯽ ﺑﺮﺧﻮﺭﺩ ﻣﯽ ﮐﺮﺩﻧﺪ ﮐﻪ ﻧﻘﺎﺷﺎﻥ ﺑﺎ ﻧﻮﺭ ﻭ ﺭﻧﮓ. ﻓﻀﺎﺳﺎﺯﯼ ﻭ ﺗﺎﮐﻴﺪ ﺑﺮ ﺭﻧﮓ ﺁﻣﻴﺰﯼ ﺻﻮﺗﯽ ﻭﻳﮋﮔﯽ ﺍﻳﻦ ﺳﺒﮏ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺭﻭﺩ.

ﺷﻌﺎﺭ ﺍﻳﻦ ﻧﻬﻀﺖ ﮐﻪ ﻭﺍﮐﻨﺸﯽ ﺑﺮﺍﺑﺮ ﺭﻣﺎﻧﺘﻴﺴﻴﺴﻢ ﻭ ﺍﻣﭙﺮﺳﻴﻮﻧﻴﺴﻢ ﺷﻤﺮﺩﻩ ﻣﯽ ﺷﺪ، ﺑﺎﺯﮔﺸﺖ ﺑﻪ ﺑﺎﺥ ﺑﻮﺩ. ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﻣﻮﺳﻴﻘﯽ ﺑﺎﺥ ﺳﺮﻣﺸﻖ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺁﺛﺎﺭ ﻧﻮﮐﻼﺳﻴﮏ ﺑﻮﺩﻩ ﺍﺳﺖ، ﺷﺎﻳﺪ ﻋﻨﻮﺍﻥ ﻧﻮﺑﺎﺭﻭﮎ (Neobaroque) ﺑﺮﺍﯼ ﺁﻥ ﻣﻨﺎﺳﺐ ﺗﺮ ﻣﯽ ﺑﻮﺩ.

ﺍﻳﮕﻮﺭ ﺍﺳﺘﺮﺍﻭﻳﻨﺴﮑﯽ (Igor Stravinsky) ﺍﺯ ﺳﺎﻝ ۱۸۸۲ ﺗﺎ ۱۹۷۱: ﺁﻫﻨﮕﺴﺎﺯ ﺭﻭﺱ ﮐﻪ ﺁﺛﺎﺭ  ﺍﻭﻟﻴﻪ ﺍﻭ ﺳﺎﺧﺘﻪ ﻫﺎﻳﯽ ﺍﻧﻘﻼﺑﯽ ﻭ ﮐﻼﺳﻴﮏ ﻣﺪﺭﻥ ﻣﺤﺴﻮﺏ ﻣﯽ ﺷﻮﻧﺪ. ﺍﻭ ﺑﺮ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﺑﺴﻴﺎﺭﯼ ﺗﺎﺛﻴﺮ ﮔﺬﺍﺷﺖ ﻭ ﺍﻟﻬﺎﻡ ﺑﺨﺶ ﻧﻘﺎﺷﺎﻥ، ﻧﻮﻳﺴﻨﺪﮔﺎﻥ ﻭ ﻃﺮﺍﺣﺎﻥ ﺭﻗﺺ ﺷﺪ. ﭘﺮﺳﺘﺶ ﺑﻬﺎﺭ ﺍﺯ ﻣﻌﺮﻭﻑ ﺗﺮﻳﻦ ﺑﺎﻟﻪ ﻫﺎﯼ ﺍﻭ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺭﻭﺩ ﮐﻪ ﺩﺭ ﺍﻭﻟﻴﻦ ﺍﺟﺮﺍﯼ ﺧﻮﺩ ﺩﺭ ﭘﺎﺭﻳﺲ ﺳﺮ ﻭ ﺻﺪﺍﻳﯽ ﺑﻪ ﭘﺎ ﮐﺮﺩ .

ﺩﻳﺴﻮﻧﺎﻧﺲ ﻫﺎﯼ (ﻓﻮﺍﺻﻞ ﻧﺎﻣﻄﺒﻮﻉ) ﺑﯽ ﻭﻗﻔﻪ ﻭ ﮔﻮﺵ ﺧﺮﺍﺵ ﻭ ﺭﻳﺘﻢ ﻫﺎﯼ ﭘﺮ ﺿﺮﺑﺎﻥ ﺍﻳﻦ ﺍﺛﺮ ﮐﻪ ﺍﺯ ﺁﻳﻴﻦ ﻫﺎﯼ ﺩﻭﺭﺍﻥ ﺑﺎﺳﺘﺎﻥ ﺍﻟﻬﺎﻡ ﮔﺮﻓﺘﻪ ﺷﺪﻩ ﺑﻮﺩ، ﺑﺎﻋﺚ ﺷﺪ ﮐﻪ ﻋﺪﻩ ﺍﯼ ﺍﺯ ﺣﺎﺿﺮﺍﻥ ﺷﺮﻭﻉ ﺑﻪ ﺳﻮﺕ ﮐﺸﻴﺪﻥ ﻭ ﺍﻇﻬﺎﺭ ﻧﻈﺮ ﺑﺎ ﺻﺪﺍﯼ ﺑﻠﻨﺪ ﮐﻨﻨﺪ. ﺍﻣﺎ ﺍﺟﺮﺍﻫﺎﯼ ﺑﻌﺪﯼ ﺍﻳﻦ ﺍﺛﺮ ﺑﺎ ﻣﻮﻓﻘﻴﺖ ﺯﻳﺎﺩﯼ ﻫﻤﺮﺍﻩ ﺷﺪ ﻭ ﺩﻳﺮﯼ ﻧﭙﺎﻳﻴﺪ ﮐﻪ ﺍﻳﻦ ﺑﺎﻟﻪ ﺭﺍ ﺍﺯ ﺷﺎﻫﮑﺎﺭ ﻫﺎﯼ ﻣﻮﺳﻴﻘﯽ ﺑﻪ ﺣﺴﺎﺏ ﺁﻭﺭﺩﻧﺪ.

ﺍﺯ ﺩﻳﮕﺮ ﺁﺛﺎﺭ ﺍﺳﺘﺮﺍﻭﻳﻨﺴﮑﯽ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﺑﺎﻟﻪ ﭘﺮﻧﺪﻩ ﺁﺗﺸﻴﻦ، ﺑﺎﻟﻪ ﭘﺘﺮﻭﺷﮑﺎ ﻭ ﺑﺎﻟﻪ ﭘﻮﻟﭽﻴﻨﻼ (ﺑﺎﻟﻪ ﺍﯼ ﺑﺮﺍﺳﺎﺱ ﻧﻐﻤﺎﺕ ﺑﺎﺭﻭﮎ ﮐﻪ ﻃﺮﺍﺣﯽ ﺻﺤﻨﻪ ﺁﻥ ﺑﻪ ﻋﻬﺪﻩ ﭘﻴﮑﺎﺳﻮ ﺑﻮﺩ) ﺍﺷﺎﺭﻩ ﮐﺮﺩ. (ﻧﻤﻮﻧﻪ ﺷﻨﻴﺪﺍﺭﯼ- ﺑﺎﻟﻪ ﭘﺮﺳﺘﺶ ﺑﻬﺎﺭ، ﺍﺛﺮ ﺍﺳﺘﺮﺍﻭﻳﻨﺴﮑﯽ)

 

ﺍﮐﺴﭙﺮﺳﻴﻮﻧﻴﺴﻢ
ﺍﮐﺴﭙﺮﺳﻴﻮﻧﻴﺴﻢ، ﺟﻨﺒﺸﯽ ﺍﺳﺖ ﻣﻌﻄﻮﻑ ﺑﻪ ﻧﺎﺭﺿﺎﻳﯽ ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﮐﻪ ﺑﻪ ﮔﻮﻧﻪ ﺍﯼ ﺯﻧﺪﻩ ﻭ ﭘﻮﻳﺎ ﺍﻧﺘﻘﺎﻝ ﺩﻫﻨﺪﻩ ﺧﺸﻢ ﺭﻧﺞ ﺩﻳﺪﮔﺎﻥ ﻭ ﺯﻳﺮﺩﺳﺘﺎﻥ ﺍﺳﺖ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﮐﺴﭙﺮﺳﻴﻮﻧﻴﺴﺖ ﻫﺎ ﺑﻪ ﻣﺨﺎﻟﻔﺖ ﺑﺎ ﺟﻨﮓ ﺟﻬﺎﻧﯽ ﺍﻭﻝ ﺑﺮﺧﺎﺳﺘﻨﺪ ﻭ ﻫﻨﺮﺷﺎﻥ ﺭﺍ ﺑﺮﺍﯼ ﺗﺠﺴﻢ ﻫﺮﺍﺱ ﺧﻮﺩ ﺍﺯ ﮐﺸﺘﺎﺭ ﻭ ﺧﻮﻧﺮﻳﺰﯼ ﺑﻪ ﮐﺎﺭ ﺑﺴﺘﻨﺪ. ﺁﻧﻬﺎ ﺯﻳﺒﺎﻳﯽ ﻭ ﺁﺭﺍﺳﺘﮕﯽ ﺭﺍﻳﺞ ﺭﺍ ﻣﺮﺩﻭﺩ ﺷﻤﺮﺩﻧﺪ. ﺁﺛﺎﺭ ﺁﻧﻬﺎ، ﺑﻪ ﺳﺒﺐ ﺗﻮﺟﻪ ﻓﺮﺍﻭﺍﻥ ﺑﻪ ﻣﻀﻤﻮﻥ ﻫﺎﻳﯽ ﻫﻤﭽﻮﻥ ﺟﻨﻮﻥ ﻭ ﻣﺮﮒ، ﺷﺎﻳﺪ ﮐﺮﻳﻪ ﻭ ﺑﺮﺁﺷﻮﺑﻨﺪﻩ ﺟﻠﻮﻩ ﮐﻨﺪ.

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺁﺛﺎﺭ ﻣﻮﺳﻴﻘﯽ ﺳﺪﻩ ﺑﻴﺴﺘﻢ ﺑﺎﺯﺗﺎﺑﻨﺪﻩ ﺍﮐﺴﭙﺮﺳﻴﻮﻧﻴﺴﻢ ﻫﺴﺘﻨﺪ. ﺟﻨﺒﺸﯽ ﻫﻨﺮﯼ ﮐﻪ ﺗﺎﮐﻴﺪ ﺑﺮ ﺑﻴﺎﻥ ﺣﺲ ﺷﺪﻳﺪ ﻭ ﺩﺭﻭﻧﯽ ﺩﺍﺷﺘﻪ ﺍﺳﺖ. ﮐﺎﻧﻮﻥ ﺍﻳﻦ ﺟﻨﺒﺶ ﺑﻪ ﻃﻮﺭ ﻋﻤﺪﻩ ﺩﺭ ﺳﺎﻝ ﻫﺎﯼ ۱۹۰۵ ﺗﺎ ۱۹۲۵ ﺁﻟﻤﺎﻥ ﻭ ﺍﺗﺮﻳﺶ ﺑﻮﺩ. ﻧﻘﺎﺷﺎﻥ، ﻧﻮﻳﺴﻨﺪﮔﺎﻥ ﻭ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﺍﻳﻦ ﻣﮑﺘﺐ ﺑﻪ ﺟﺎﯼ ﺗﺠﺴﻢ ﻧﻤﻮﺩﻫﺎﯼ ﺟﻬﺎﻥ ﺑﻴﺮﻭﻧﯽ ﺑﻪ ﮐﺎﻭﺵ ﺩﺭ ﻗﻠﻤﺮﻭ ﺍﺣﺴﺎﺳﺎﺕ ﺩﺭﻭﻧﯽ ﭘﺮﺩﺍﺧﺘﻨﺪ. ﺁﻧﻬﺎ ﺑﺮﺍﯼ ﻏﺎﻓﻠﮕﻴﺮ ﻭ ﻣﺒﻬﻮﺕ ﺳﺎﺧﺘﻦ ﻣﺨﺎﻃﺐ ﻭ ﺍﻧﺘﻘﺎﻝ ﺍﺿﻄﺮﺍﺏ ﻭ ﺗﻨﺶ ﻫﺎﯼ ﺭﻭﺍﻥ ﺍﻧﺴﺎﻥ، ﮐﮋﻧﻤﺎﻳﯽ ﻫﺎﻳﯽ ﺗﻌﻤﺪﯼ ﺭﺍ ﺑﻪ ﮐﺎﺭ ﺑﺴﺘﻨﺪ.

ﺧﺎﺳﺘﮕﺎﻩ ﺍﮐﺴﭙﺮﺳﻴﻮﻧﻴﺴﻢ ﻫﻤﺎﻥ ﻗﻠﻤﺮﻭ ﻓﮑﺮﯼ ﭘﮋﻭﻫﺶ ﻫﺎﯼ ﻓﺮﻭﻳﺪ ﺩﺭ ﻫﻴﺠﺎﻥ ﻫﺎ ﻭ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ ﺍﻧﺴﺎﻥ ﺑﻮﺩ. ﻧﻘﺎﺷﺎﻥ ﺍﮐﺴﭙﺮﺳﻴﻮﻧﻴﺴﺖ ﺁﻟﻤﺎﻥ ﺩﺭ ﺑﺮﺍﺑﺮ ﺍﻣﭙﺮﺳﻴﻮﻧﻴﺴﻢ ﻓﺮﺍﻧﺴﻮﯼ- ﮐﻪ ﻣﻀﻤﻮﻥ ﻫﺎﯼ ﺩﻟﻨﺸﻴﻦ، ﺭﻧﮓ ﻫﺎﯼ ﻣﺤﻮ ﻭ ﺩﻝ ﺍﻧﮕﻴﺰ ﻭ ﺳﻄﺢ ﻫﺎﻳﯽ ﻣﻮﺝ ﮔﻮﻥ ﺩﺍﺷﺖ- ﻃﻐﻴﺎﻥ ﮐﺮﺩﻧﺪ. ﺍﮐﺴﭙﺮﺳﻴﻮﻧﻴﺴﻢ ﻣﻮﺳﻴﻘﺎﻳﯽ ﺩﺭ ﺳﺪﻩ ﺑﻴﺴﺘﻢ ﺍﺯ ﻃﻐﻴﺎﻥ ﺣﺴﯽ ﺁﺛﺎﺭ ﻭﺍﭘﺴﻴﻦ ﺭﻣﺎﻧﺘﻴﮏ ﻫﺎ، ﻳﻌﻨﯽ ﺁﻫﻨﮕﺴﺎﺯﺍﻧﯽ ﻫﻤﭽﻮﻥ ﻭﺍﮔﻨﺮ، ﺭﻳﺸﺎﺭﺕ ﺍﺷﺘﺮﺍﻭﺱ ﻭ ﻣﺎﻟﺮ ﺳﺮﺑﺮﺁﻭﺭﺩ.

ﺁﺭﻧﻮﻟﺪ ﺷﻮﺋﻨﺒﺮﮒ (Arnold Schoenberg) ﺍﺯ ﺳﺎﻝ ۱۸۴۷ ﺗﺎ ۱۹۵۱: ﺁﻫﻨﮕﺴﺎﺯ ﺍﺗﺮﻳﺸﯽ ﮐﻪ ﺑﺎ ﺩﺳﺖ ﮐﺸﻴﺪﻥ ﺍﺯ ﻧﻈﺎﻡ ﺳﻨﺘﯽ ﺗﻨﺎﻝ ﮔﺎﻣﯽ ﺍﻧﻘﻼﺑﯽ ﺑﺮﺩﺍﺷﺖ. ﻣﺨﺎﻟﻔﺖ ﻫﺎﯼ ﺑﺴﻴﺎﺭﯼ ﺑﺎ ﺍﻭ ﺷﺪ ﺗﺎ ﺁﻧﺠﺎ ﮐﻪ ﻳﮏ ﻣﻨﺘﻘﺪ ﭘﻴﺸﻨﻬﺎﺩ ﮐﺮﺩ ﮐﻪ ﺷﻮﺋﻨﺒﺮﮒ ﺑﺪﻭﻥ ﮐﺎﻏﺬ ﻧﺖ ﺩﺭ ﺩﺍﺭﺍﻟﻤﺠﺎﻧﻴﻦ ﻧﮕﻬﺪﺍﺭﯼ ﺷﻮﺩ.

ﺍﻭ ﻣﻮﺳﻴﻘﯽ ﺩﻭﺍﺯﺩﻩ ﺗﻨﯽ (ﺩﻭﺩ ﮐﺎﻓﻮﻧﻴﮏ) ﺭﺍ ﺍﺑﺪﺍﻉ ﮐﺮﺩ. ﺩﺭ ﻧﻈﺎﻡ ﺩﻭﺍﺯﺩﻩ ﺻﻮﺗﯽ ﺑﺮ ﺧﻼﻑ ﺳﻴﺴﺘﻢ ﺗﻮﻧﺎﻝ ﮐﻪ ﺗﮑﻴﻪ ﺍﯼ ﻭﻳﮋﻩ ﺑﺮ ﻳﮏ ﺻﺪﺍﯼ ﻣﺮﮐﺰﯼ ﻭﺟﻮﺩ ﺩﺍﺷﺘﻪ ﺍﺳﺖ، ﺑﺎ ﺗﻤﺎﻡ ﺩﻭﺍﺯﺩﻩ ﺻﺪﺍﯼ ﮐﺮﻭﻣﺎﺗﻴﮏ ﻳﮑﺴﺎﻥ ﺭﻓﺘﺎﺭ ﻣﯽ ﺷﻮﺩ. ﻣﻮﺳﻴﻘﯽ ﻭﺍﮔﻨﺮ، ﺑﺮﺍﻣﺲ، ﻭ ﻣﺎﻟﺮ ﻧﻘﻄﻪ ﺁﻏﺎﺯ ﺣﺮﮐﺖ ﺷﻮﺋﻨﺒﺮﮒ ﺑﻮﺩ. ﺍﺯ ﻗﻄﻌﺎﺕ ﺍﻭ ﻣﯽ ﺗﻮﺍﻥ ﺑﻪ ﺍﭘﺮﺍﯼ ﻣﻮﺳﯽ ﻭ ﻫﺎﺭﻭﻥ، ﺑﺎﺯﻣﺎﻧﺪﻩ ﺍﯼ ﺍﺯ ﻭﺭﺷﻮ ﻭ ﭘﻨﺞ ﻗﻄﻌﻪ ﺑﺮﺍﯼ ﺍﺭﮐﺴﺘﺮ ﺍﺷﺎﺭﻩ ﮐﺮﺩ.

ﺍﺯ ﺩﻳﮕﺮ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﻣﻌﺮﻭﻑ ﺍﮐﺴﭙﺮﺳﻴﻮﻧﻴﺰﻡ ﺁﻟﺒﺎﻥ ﺑﺮﮒ ﻭ ﺁﻧﺘﻮﺍﻥ ﻭﺑﺮﻥ، ﺍﺯ ﺷﺎﮔﺮﺩﺍﻥ ﺷﻮﺋﻨﺒﺮﮒ ﺭﺍ ﻣﯽ ﺗﻮﺍﻥ ﻧﺎﻡ ﺑﺮﺩ. (ﻧﻤﻮﻧﻪ ﺷﻨﻴﺪﺍﺭﯼ- ﺑﺎﺯﻣﺎﻧﺪﻩ ﺍﯼ ﺍﺯ ﻭﺭﺷﻮ، ﺍﺛﺮ ﺷﻮﺋﻨﺒﺮﮒ)
