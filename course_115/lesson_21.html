ﺩﺭﺱ ﭼﻬﺎﺭﻡ- ﺧﻠﻖ ﻣﺤﺘﻮﺍﯼ ﻫﺪﻓﻤﻨﺪ

ﻣﺤﺘﻮﺍﻳﯽ ﮐﻪ ﺷﻤﺎ ﺧﻠﻖ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﺩﺭ ﻭﺍﻗﻊ ﭘﻴﺎﻡ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﺭﻭﻧﮕﺮﺍﯼ ﺷﻤﺎﺳﺖ. ﺭﺍﻩ‌ﻫﺎﯼ ﮔﻮﻧﺎﮔﻮﻧﯽ ﺑﺮﺍﯼ ﺭﺳﺎﻧﺪﻥ ﭘﻴﺎﻡ ﺷﻤﺎ ﺑﻪ ﻣﺨﺎﻃﺒﺎﻥ‌ﺗﺎﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﻫﻤﺎﻧﻨﺪ ﻳﮏ ﭘﺴﺖ ﻭﺑﻼﮒ٬ ﺍﻳﻤﻴﻞ٬ ﺻﻔﺤﻪ ﻓﺮﻭﺩ٬ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻭ ﺑﺴﻴﺎﺭﯼ ﺩﻳﮕﺮ. ﻫﻤﻪ ﺍﻳﻦ ﺭﺍﻩ‌ﻫﺎ ﺑﺪﻭﻥ ﭘﻴﺎﻡ ﺷﻤﺎ ﻳﺎ ﻫﻤﺎﻥ ﻣﺤﺘﻮﺍ ﺍﺭﺯﺷﯽ ﻧﺪﺍﺭﻧﺪ. ﺩﺭ ﺣﻘﻴﻘﺖ ﺑﺪﻭﻥ ﻣﺤﺘﻮﺍﯼ ﺗﻮﻟﻴﺪﯼ ﮐﺎﺭﺑﺮﺍﻥ ﺑﺰﺭﮒ ﻭ ﮐﻮﭼﮏ٬ ﺍﻳﻨﺘﺮﻧﺖ ﺧﺎﻟﯽ ﺧﻮﺍﻫﺪ ﺑﻮﺩ.

ﻣﺤﺘﻮﺍﯼ ﺗﻮﻟﻴﺪ ﺷﺪﻩ٬ ﻧﻘﺶ ﺣﻴﺎﺗﯽ ﺩﺭ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﺭﻭﻧﮕﺮﺍﯼ ﺷﻤﺎ ﺩﺍﺭﺩ. ﺍﻣﺎ ﭼﮕﻮﻧﻪ ﻳﮏ ﻣﺤﺘﻮﺍﯼ ﺍﺭﺯﺷﻤﻨﺪ ﺧﻠﻖ ﮐﻨﻴﻢ؟ ﻣﻨﻈﻮﺭ ﺍﺯ ﻣﺤﺘﻮﺍﯼ ﺍﺭﺯﺷﻤﻨﺪ٬ ﻣﺤﺘﻮﺍﻳﯽ ﺍﺳﺖ ﮐﻪ ﺧﻮﺍﻧﻨﺪﻩ ﺗﻤﺎﻳﻞ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﺁﻥ ﺭﺍ ﺗﺎ ﺍﻧﺘﻬﺎ ﺑﺨﻮﺍﻧﺪ (ﻳﺎ ﺑﺒﻴﻨﺪ ﻳﺎ ﮔﻮﺵ ﺩﻫﺪ)٬ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﺩ ﻭ ﻧﻬﺎﻳﺘﺎ ﺩﻭﺑﺎﺭﻩ ﺑﺮﺍﯼ ﺧﻮﺍﻧﺪﻥ ﻣﻄﺎﻟﺐ ﻣﺸﺎﺑﻪ ﻭ ﻫﻢ ﺗﺮﺍﺯ ﺑﻪ ﺳﺎﻳﺖ ﺷﻤﺎ ﺑﺮﮔﺮﺩﺩ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺳﻮﺍﻝ ﺑﻌﺪﯼ ﺍﻳﻦ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﮐﻪ ﭼﻪ ﻋﻮﺍﻣﻠﯽ ﺑﺎﻋﺚ ﺍﺭﺯﺷﻤﻨﺪ ﺷﺪﻥ ﻳﮏ ﻣﺤﺘﻮﺍ ﻣﯽ‌ﺷﻮﻧﺪ؟

ﺍﻳﻦ ﺭﻭﺯﻫﺎ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﻭﺟﻮﺩ ﺍﻳﻨﺘﺮﻧﺖ ﻭ ﻣﻮﺗﻮﺭﻫﺎﯼ ﺟﺴﺘﺠﻮ٬ ﻫﺰﺍﺭﺍﻥ ﻣﻄﻠﺐ ﻣﺮﺗﺒﻂ ﺑﺎ ﻣﻮﺿﻮﻋﺎﺕ ﻣﺨﺘﻠﻒ ﺑﻪ ﺭﺍﺣﺘﯽ ﻗﺎﺑﻞ ﺩﺳﺘﺮﺱ ﺍﺳﺖ. ﺍﻳﻦ ﻣﺎﺟﺮﺍ ﺍﮔﺮ ﭼﻪ ﺑﻪ ﻇﺎﻫﺮ ﺑﺴﻴﺎﺭ ﺭﺍﺿﯽ ﮐﻨﻨﺪﻩ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ٬ ﻭﻟﯽ ﺍﺯ ﺩﻳﮕﺮ ﺳﻮ٬ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﮐﻪ ﺟﻠﺐ ﺗﻮﺟﻪ ﻣﺨﺎﻃﺒﺎﻥ ﺑﻪ ﻣﻄﻠﺐ ﻭ ﻣﺤﺘﻮﺍﯼ ﺗﻮﻟﻴﺪﯼ ﺷﻤﺎ ﺳﺨﺖ‌ﺗﺮ ﺷﻮﺩ. ﭼﻪ ﮐﺴﯽ ﺑﻴﺸﺘﺮﻳﻦ ﺗﻮﺟﻪ ﺭﺍ ﺟﻠﺐ ﻣﯽ‌ﮐﻨﺪ؟ ﻣﺮﺩﻡ ﮐﺪﺍﻡ ﻭﺏ‌ﺳﺎﻳﺖ ﺭﺍ ﺑﻴﺸﺘﺮ ﺧﻮﺍﻫﻨﺪ ﺧﻮﺍﻧﺪ ﻭ ﮐﺪﺍﻡ ﻣﻄﻠﺐ ﺭﺍ ﺑﻴﺸﺘﺮ ﺩﺍﻧﻠﻮﺩ ﺧﻮﺍﻫﻨﺪ ﻧﻤﻮﺩ ﻭ ﺑﻪ ﺭﺍﻫﻨﻤﺎﻳﯽ ﮐﺪﺍﻡ ﻭﺏ‌ﺳﺎﻳﺖ ﺩﺭ ﺧﺮﻳﺪ ﻓﻼﻥ ﮐﺎﻻ ﺗﻮﺟﻪ ﺧﻮﺍﻫﻨﺪ ﮐﺮﺩ؟

ﺍﻳﻦ ﺭﻭﺯﻫﺎ ﺑﻴﺸﺘﺮﻳﻦ ﺟﻨﮓ ﺩﺭ ﻓﻀﺎﯼ ﻣﺠﺎﺯﯼ ﻭ ﺑﻴﻦ ﺳﺎﻳﺖ‌ﻫﺎ ﻭ ﺗﺠﺎﺭﺕ‌ﻫﺎﯼ ﺁﻧﻼﻳﻦ٬ ﺟﻨﮓ ﺟﻠﺐ ﺗﻮﺟﻪ ﺍﺳﺖ!

ﺑﻨﺎﺑﺮﺍﻳﻦ٬ ﺑﺮﺍﯼ ﻭﺍﺭﺩ ﺷﺪﻥ ﺩﺭ ﺍﻳﻦ ﺟﻨﮓ ﻧﺎﭘﻴﺪﺍ ﻭ ﮔﺎﻫﯽ ﻧﺎﺑﺮﺍﺑﺮ٬ ﺩﺍﺷﺘﻦ ﻳﮏ ﺍﺳﺘﺮﺍﺗﮋﯼ ﮐﺎﺭﺁﻣﺪ ﻭ ﺳﻨﺪ ﺷﺪﻩ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﻣﺤﺘﻮﺍ ﺿﺮﻭﺭﯼ ﺍﺳﺖ. ﺍﻳﻦ ﮐﺎﺭ ﺍﮔﺮﭼﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺍﺑﺘﺪﺍ ﻧﺴﺒﺖ ﺑﻪ ﺷﺮﻭﻉ ﺳﺮﻳﻊ ﻭ ﺑﺪﻭﻥ ﺑﺮﻧﺎﻣﻪ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﻣﺤﺘﻮﺍ٬ ﮐﻤﯽ ﻭﻗﺖ‌ﮔﻴﺮ ﻭ ﺳﺨﺖ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﺪ٬ ﺍﻣﺎ ﺩﺭ ﻧﻬﺎﻳﺖ ﺟﻮﺍﺏ ﺧﻮﺍﻫﺪ ﺩﺍﺩ. 

ﺑﻪ ﻃﻮﺭ ﮐﻠﯽ٬ ﻫﺮ ﻣﻄﻠﺐ ﻭ ﻣﺤﺘﻮﺍﯼ ﺗﻮﻟﻴﺪﯼ٬ ﺑﺎ ﺳﻪ ﭘﺎﺭﺍﻣﺘﺮ ﺷﻨﺎﺧﺘﻪ ﻣﯽ‌ﺷﻮﺩ: ﻫﺪﻓﺶ٬ ﻓﺮﻣﺘﺶ ﻭ ﻋﻨﻮﺍﻧﺶ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﻳﮏ ﻣﺤﺘﻮﺍﯼ ﻫﺪﻓﻤﻨﺪ٬ ﺩﺭ ﺍﺑﺘﺪﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺍﺯ ﺧﻮﺩ ﺑﭙﺮﺳﻴﻢ ﮐﻪ ﻫﺪﻑ ﺍﺯ ﺗﻮﻟﻴﺪ ﻣﺤﺘﻮﺍﻳﯽ ﮐﻪ ﺩﺭ ﺫﻫﻦ ﺩﺍﺭﻳﻢ ﭼﻴﺴﺖ؟ ﮐﺸﺎﻧﺪﻥ ﻣﺨﺎﻃﺐ ﺑﻴﺸﺘﺮ ﺑﻪ ﺳﺎﻳﺖ ﻳﺎ ﺗﺒﺪﻳﻞ ﺁﻧﻬﺎ ﺍﺯ ﻳﮏ ﺭﻫﮕﺬﺭ ﺑﻪ ﻳﮏ ﻣﺸﺘﺮﯼ ﺑﺎﻟﻘﻮﻩ ﺷﻨﺎﺳﻨﺎﻣﻪ‌ﺩﺍﺭ (Lead)؟ ﺗﻔﺎﻭﺕ ﻋﻤﺪﻩ‌ﺍﯼ ﺑﻴﻦ ﺍﻳﻦ ﺩﻭ ﻧﻮﻉ ﻣﺤﺘﻮﺍ ﻭﺟﻮﺩ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ ﮐﻪ ﺩﺭ ﺍﺩﺍﻣﻪ ﺑﺮﺭﺳﯽ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

ﺑﻌﺪ ﺍﺯ ﺗﻌﻴﻴﻦ ﻫﺪﻑ‌ﻣﺎﻥ ﺍﺯ ﻧﻮﺷﺘﻦ ﻳﮏ ﻣﻄﻠﺐ٬ ﻧﻮﺑﺖ ﺑﻪ ﺗﻌﻴﻴﻦ ﻓﺮﻣﺖ ﺗﻬﻴﻪ ﺁﻥ ﻣﻄﻠﺐ ﻣﯽ‌ﺭﺳﺪ. ﻣﺤﺘﻮﺍﯼ ﺗﻮﻟﻴﺪﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺩﺭ ﻓﺮﻣﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﻣﺎﻧﻨﺪ ﺑﺮﺭﺳﯽ ﻣﻮﺭﺩﯼ (Case Study)٬ ﻳﮏ ﻭﻳﺪﻳﻮ ﮐﻠﻴﭗ٬ ﺍﻳﻨﻔﻮﮔﺮﺍﻓﻴﮏ٬ ﮔﺰﺍﺭﺵ ﺧﺒﺮﯼ ﻳﺎ ﮔﺰﺍﺭﺵ ﺗﺤﻘﻴﻘﯽ٬ ﮐﺘﺎﺏ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ٬ ﭼﮏ ﻟﻴﺴﺖ٬ ﻣﺠﻤﻮﻋﻪ ﺍﺳﻼﻳﺪﻫﺎ٬ ﻭﺑﻴﻨﺎﺭ ﺁﻧﻼﻳﻦ ﻭ ﺑﺴﻴﺎﺭﯼ ﻣﻮﺍﺭﺩ ﺩﻳﮕﺮ ﺑﺎﺷﺪ. ﺩﺭ ﻳﮏ ﻧﮕﺎﻩ ﮐﻠﯽ٬ ﺗﻤﺎﻡ ﻓﺮﻣﺖ‌ﻫﺎﯼ ﺫﮐﺮ ﺷﺪﻩ ﺗﻔﺎﻭﺕ ﭼﻨﺪﺍﻧﯽ ﺑﺎ ﻫﻢ ﻧﺪﺍﺭﻧﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻥ ﻳﮏ ﺩﺍﺳﺘﺎﻥ ﻭﺍﺣﺪ ﺭﺍ ﺗﻘﺮﻳﺒﺎ ﺑﺎ ﺗﻤﺎﻣﯽ ﻓﺮﻣﺖ‌ﻫﺎﯼ ﺫﮐﺮ ﺷﺪﻩ ﺭﻭﺍﻳﺖ ﮐﺮﺩ. ﺍﻳﻨﮑﻪ ﮐﺪﺍﻡ ﻓﺮﻣﺖ ﺍﺯ ﻣﻮﺍﺭﺩ ﺑﺎﻻ ﺭﺍ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﻢ ﮐﺎﻣﻼ ﺑﻪ ﻣﺨﺎﻃﺒﻴﻨﯽ ﮐﻪ ﻣﺪﻧﻈﺮ ﺩﺍﺭﻳﻢ٬ ﻣﺪﺕ ﺯﻣﺎﻧﯽ ﮐﻪ ﺑﺮﺍﯼ ﺧﻠﻖ ﺍﻳﻦ ﻣﺤﺘﻮﺍ ﺧﻮﺍﻫﻴﻢ ﺩﺍﺷﺖ ﻭ ﻧﻴﺰ ﺍﻃﻼﻋﺎﺕ ﺷﺨﺼﯽ ﺧﻮﺩﻣﺎﻥ ﺑﺴﺘﮕﯽ ﺩﺍﺭﺩ. 

ﺩﺭ ﻧﻬﺎﻳﺖ ﻋﻨﺼﺮ ﭘﺎﻳﺎﻧﯽ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﻳﮏ ﻣﺤﺘﻮﺍﯼ ﺍﺭﺯﺷﻤﻨﺪ٬ ﻋﻨﻮﺍﻧﯽ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﺁﻥ ﺩﺭ ﻧﻈﺮ ﻣﯽ‌ﮔﻴﺮﻳﻢ. ﺑﺮﺍﯼ ﮔﺬﺍﺷﺘﻦ ﻳﮏ ﻋﻨﻮﺍﻥ ﻣﻨﺎﺳﺐ ﺑﺮ ﺭﻭﯼ ﻣﺤﺘﻮﺍﻳﯽ ﮐﻪ ﺗﻮﻟﻴﺪ ﮐﺮﺩﻳﻢ٬ ﻣﺨﺎﻃﺐ ﺧﻮﺩ ﻭ ﺍﻳﻨﮑﻪ ﺑﻪ ﭼﻪ ﻋﻠﺖ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﻪ ﺳﺎﻳﺖ ﺷﻤﺎ ﺑﻴﺎﻳﺪ ﺭﺍ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ. ﻋﻨﻮﺍﻥ ﻣﻨﺎﺳﺐ ﺭﺍ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﺯ ﻃﺮﻳﻖ ﺑﺮﺭﺳﯽ ﮐﻠﻴﺪﻭﺍﮊﻩ‌ﻫﺎ (Keyword Research) ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺁﻧﭽﻪ ﻣﺎﻳﻠﻴﺪ ﺑﺮﺍﻳﺶ ﺭﺗﺒﻪ‌ﺍﯼ ﺑﮕﻴﺮﻳﺪ ﺗﻌﻴﻴﻦ ﮐﻨﻴﺪ. ﺑﺮﺭﺳﯽ ﺗﺎﻻﺭﻫﺎﯼ ﮔﻔﺘﮕﻮﯼ ﻣﺮﺗﺒﻂ٬ ﻣﺮﻭﺭ ﺍﺧﺒﺎﺭ ﻣﻬﻢ ﺻﻨﻌﺖ ﻣﻮﺭﺩ ﻧﻈﺮ٬ ﺑﺮﺭﺳﯽ ﺳﻮﺍﻻﺗﯽ ﮐﻪ ﺍﺯ ﺗﻴﻢ ﺧﺪﻣﺎﺕ ﻣﺸﺘﺮﻳﺎﻥ‌ﺗﺎﻥ ﭘﺮﺳﻴﺪﻩ ﻣﯽ‌ﺷﻮﺩ (ﺩﺭﺻﻮﺭﺕ ﻭﺟﻮﺩ)٬ ﺍﻫﺪﺍﻑ ﺳﺎﻳﺖ ﺧﻮﺩﺗﺎﻥ ﻭ ﻧﻴﺰ ﺑﺴﻴﺎﺭﯼ ﻣﻮﺍﺭﺩ ﺩﻳﮕﺮ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﺷﻤﺎ ﺩﺭ ﺍﻳﻦ ﺍﻣﺮ ﻳﺎﺭﯼ ﺑﺮﺳﺎﻧﻨﺪ.

ﻫﻤﻮﺍﺭﻩ ﺍﻳﻦ ﻧﮑﺘﻪ ﺑﺴﻴﺎﺭ ﻣﻬﻢ ﺭﺍ ﻣﺪ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﮐﻪ ﻳﮏ ﻣﺤﺘﻮﺍﯼ ﻋﺎﻟﯽ٬ ﺁﻣﻮﺯﺵ ﺩﻫﻨﺪﻩ ﺍﺳﺖ ﻧﻪ ﺗﺒﻠﻴﻐﺎﺗﯽ! ﺣﺘﯽ ﺍﮔﺮ ﻫﺪﻑ ﻧﻬﺎﻳﯽ ﺷﻤﺎ ﺩﻳﺪﻩ ﺷﺪﻥ ﺑﻴﺸﺘﺮ ﻭ ﮐﺸﻴﺪﻥ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﮔﺎﻥ ﺑﻴﺸﺘﺮ ﺑﻪ ﺳﺎﻳﺖ‌ﺗﺎﻥ ﻭ ﺗﺒﺪﻳﻞ ﺁﻧﻬﺎ ﺑﻪ ﻣﺸﺘﺮﯼ ﺍﺳﺖ٬ ﻣﻄﻠﺐ ﺗﻬﻴﻪ ﺷﺪﻩ ﻧﺒﺎﻳﺪ ﺗﺒﻠﻴﻎ ﻣﺴﺘﻘﻴﻢ ﮐﺎﺭ ﻭ ﺗﺠﺎﺭﺕ ﺧﻮﺩﺗﺎﻥ ﺑﺎﺷﺪ! ﻣﻄﻠﺐ ﺗﻬﻴﻪ ﺷﺪﻩ ﺷﻤﺎ ﺑﺎﻳﺪ ﻣﺸﮑﻠﯽ ﺍﺯ ﻣﺨﺎﻃﺐ ﺗﻌﺮﻳﻒ ﺷﺪﻩ ﺷﻤﺎ ﺣﻞ ﮐﻨﺪ ﻭ ﺭﺍﻩ ﺣﻞ ﻳﺎ ﺁﻣﻮﺯﺵ ﺧﺎﺻﯽ ﺭﺍ ﺑﻪ ﻫﻤﺮﺍﻩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

ﺩﺭ ﻭﺍﻗﻊ ﺗﺎ ﭘﻴﺶ ﺍﺯ ﺭﺳﻴﺪﻥ ﺑﻪ ﻣﺮﺣﻠﻪ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ٬ ﺯﻣﺎﻥ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺑﻴﺎﻥ ﭘﻴﺸﻨﻬﺎﺩ ﻣﺮﺗﺒﻂ ﺑﺎ ﺗﺠﺎﺭﺕ ﻭ ﮐﺴﺐ‌ﻭﮐﺎﺭ ﺧﻮﺩﺗﺎﻥ ﻧﻴﺴﺖ ﻭ ﺑﻬﺘﺮ ﺍﺳﺖ ﮐﻪ ﮐﺎﺭﺕ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺭﻭ ﻧﮑﻨﻴﺪ! ﺩﺭ ﻣﺮﺣﻠﻪ ﺁﮔﺎﻩ‌ﺳﺎﺯﯼ (Awareness Stage)٬ ﻣﺨﺎﻃﺒﺎﻥ ﺷﻤﺎ ﺣﺘﯽ ﻣﻤﮑﻦ ﺍﺳﺖ ﻧﺪﺍﻧﻨﺪ ﮐﻪ ﺧﺪﻣﺘﯽ ﻣﺎﻧﻨﺪ ﺧﺪﻣﺘﯽ ﮐﻪ ﺷﻤﺎ ﺍﺭﺍﻳﻪ ﻣﯽ‌ﺩﻫﻴﺪ ﻭﺟﻮﺩ ﺩﺍﺭﺩ!

ﺩﺭ ﻣﺮﺣﻠﻪ ﺑﻌﺪﯼ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺁﻥ ﺭﺍ ﻣﺮﺣﻠﻪ «ﺯﻳﺮ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﻳﺎ ﻣﻼﺣﻈﻪ (Consideration Stage)» ﻧﺎﻣﻴﺪ٬ ﻣﺨﺎﻃﺒﺎﻥ‌ﺗﺎﻥ ﺍﺯ ﻭﺟﻮﺩ ﭼﻨﻴﻦ ﺧﺪﻣﺘﯽ ﺁﮔﺎﻩ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ ﻭﻟﯽ ﻧﻤﯽ‌ﺩﺍﻧﻨﺪ ﮐﻪ ﻧﺨﺴﺖ٬ ﺁﻳﺎ ﺍﻳﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻳﮏ ﺭﺍﻩ ﺣﻞ ﺑﺮﺍﯼ ﺁﻧﻬﺎ ﺑﺎﺷﺪ ﻭ ﺩﻭﻡ٬ ﺩﺭ ﮐﺠﺎ ﻣﯽ‌ﺗﻮﺍﻥ ﺁﻥ ﺭﺍ ﻳﺎﻓﺖ؟ ﺗﻨﻬﺎ ﻭﻗﺘﯽ ﮐﻪ ﻣﺨﺎﻃﺒﺎﻥ‌ﺗﺎﻥ ﺍﺯ ﮐﻞ ﺟﻮﺍﻧﺐ ﺍﻣﺮ ﻭ ﺭﺍﻩ‌ﺣﻞ‌ﻫﺎﯼ ﺍﺭﺍﻳﻪ ﺷﺪﻩ ﺷﻤﺎ ﺁﮔﺎﻩ ﺷﺪﻧﺪ، ﻭﻗﺖ ﺭﻭ ﮐﺮﺩﻥ ﺩﺳﺖ‌ﺗﺎﻥ ﻭ ﻧﺸﺎﻥ ﺩﺍﺩﻥ ﭘﻴﺸﻨﻬﺎﺩ ﺧﺮﻳﺪ ﺷﻤﺎﺳﺖ!

ﻣﺮﺣﻠﻪ ﺗﺼﻤﻴﻢ‌ﮔﻴﺮﯼ (Decision Making Stage) ﺩﺭ ﻭﺍﻗﻊ ﺟﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﺎ ﺗﻮﻟﻴﺪ ﻳﮏ ﻣﺤﺘﻮﺍﯼ ﻫﺪﻓﻤﻨﺪ٬ ﺩﻗﻴﻖ٬ ﺟﺎﻣﻊ٬ ﺁﮔﺎﻩ‌ﮐﻨﻨﺪﻩ ﻭ ﺻﺎﺩﻗﺎﻧﻪ٬ ﺍﻋﺘﻤﺎﺩ ﻣﺨﺎﻃﺐ ﺭﺍ ﺟﻠﺐ ﻧﻤﻮﺩﻩ ﻭ ﭘﻴﺸﻨﻬﺎﺩ ﺧﺮﻳﺪ ﺧﻮﺩ ﺭﺍ ﻫﻢ ﺑﻪ ﺍﻭ ﻋﺮﺿﻪ ﻣﯽ‌ﮐﻨﻴﺪ.

ﺩﺭﻣﺠﻤﻮﻉ٬ ﺩﺭ ﻫﻨﮕﺎﻡ ﮐﺎﺭ ﺑﺮﺍﯼ ﺗﻬﻴﻪ ﻣﺤﺘﻮﺍﯼ ﺍﺭﺯﺷﻤﻨﺪ ﺑﺮﺍﯼ ﺳﺎﻳﺖ ﻭ ﺗﺠﺎﺭﺕ ﺧﻮﺩ٬ ﻫﻤﻮﺍﺭﻩ ﻣﻮﺍﺭﺩ ﺯﻳﺮ ﺭﺍ ﻣﺪ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ:

۱- ﻫﻤﻮﺍﺭﻩ ﻣﺨﺎﻃﺐ ﺧﻮﺩ ﻭ ﺍﻳﻨﮑﻪ ﺁﻧﻬﺎ ﺩﺭ ﮐﺪﺍﻣﻴﮏ ﺍﺯ ﻣﺮﺍﺣﻞ ﺳﻪ ﮔﺎﻧﻪ‌ﺍﯼ ﮐﻪ ﺩﺭ ﺑﺎﻻ ﺫﮐﺮ ﺷﺪ ﻗﺮﺍﺭ ﺩﺍﺭﻧﺪ ﺭﺍ ﻣﺪ ﻧﻈﺮ ﻗﺮﺍﺭ ﺩﻫﻴﺪ ﻭ ﺳﻌﯽ ﮐﻨﻴﺪ ﻣﻄﻠﺒﯽ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﻫﻤﺎﻥ ﻣﮑﺎﻥ ﻭ ﺯﻣﺎﻥ ﺗﻬﻴﻪ ﮐﻨﻴﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﺍﺯ ﺗﻬﻴﻪ ﻣﻄﺎﻟﺐ ﺧﻴﻠﯽ ﻋﻤﻮﻣﯽ ﺩﻭﺭﯼ ﮐﻨﻴﺪ ﻭ ﺑﻴﺸﺘﺮ ﻣﺤﺘﻮﺍﯼ ﺧﻮﺩ ﺭﺍ ﻭﺍﺭﺩ ﻣﻮﺍﺭﺩ ﺧﺎﺹ ﻭ ﺩﻗﻴﻖ‌ﺗﺮ ﺑﻨﻤﺎﻳﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺍﮔﺮ ﭼﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﻣﺨﺎﻃﺐ ﮐﻤﺘﺮﯼ ﺭﺍ ﻧﺼﻴﺐ‌ﺗﺎﻥ ﮐﻨﺪ، ﻭﻟﯽ ﺑﻪ ﺟﺬﺏ ﻣﺨﺎﻃﺒﯽ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﮐﻪ ﺑﺎ ﺍﺣﺘﻤﺎﻝ ﺑﺎﻻﺗﺮﯼ ﻣﻤﮑﻦ ﺍﺳﺖ ﺍﺯ ﺷﻤﺎ ﺧﺮﻳﺪ ﮐﻨﺪ.

۲- ﻣﺨﺎﻃﺒﺎﻥ ﺷﻤﺎ ﻫﻢ ﺍﺣﺘﻤﺎﻻ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﺷﻤﺎ ﻣﺸﻐﻠﻪ ﺩﺍﺭﻧﺪ. ﭘﺲ ﺑﺎ ﺩﺭ ﻧﻈﺮ ﮔﺮﻓﺘﻦ ﺍﻳﻦ ﺍﻣﺮ ﺳﻌﯽ ﮐﻨﻴﺪ ﮐﻪ ﺧﻮﺍﻧﺪﻥ، ﻫﻀﻢ ﻭ ﺩﺭﮎ ﻣﻄﻠﺐ‌ﺗﺎﻥ ﺭﺍ ﺑﺮﺍﻳﺸﺎﻥ ﺭﺍﺣﺖ‌ﺗﺮ ﮐﻨﻴﺪ. ﻣﻄﺎﻟﺐ ﺭﺍ ﻣﺨﺘﺼﺮ ﻭ ﻣﻔﻴﺪ ﺍﺭﺍﻳﻪ ﮐﻨﻴﺪ ﻭ ﺍﺯ ﻫﺮ ﻭﺳﻴﻠﻪ ﺁﻣﻮﺯﺷﯽ ﺍﻋﻢ ﺍﺯ ﺗﺼﻮﻳﺮ، ﻭﻳﺪﻳﻮ ﻭ ﻣﺜﺎﻝ‌ﻫﺎﯼ ﺟﺎﻧﺒﯽ ﺑﺮﺍﯼ ﺭﺍﺣﺖ‌ﺗﺮ ﮐﺮﺩﻥ ﻓﻬﻢ ﻣﻄﻠﺐ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﺎﻳﻴﺪ. ﺍﺯ ﻧﻮﺷﺘﻦ ﻣﻄﺎﻟﺐ ﺧﻴﻠﯽ ﻃﻮﻻﻧﯽ ﺍﺟﺘﻨﺎﺏ ﮐﻨﻴﺪ. ﺍﮔﺮ ﻧﺎﭼﺎﺭﻳﺪ ﻣﻄﻠﺐ ﻃﻮﻻﻧﯽ ﺑﻨﮕﺎﺭﻳﺪ٬ ﺁﻥ ﺭﺍ ﺑﻪ ﻗﺴﻤﺖ‌ﻫﺎﯼ ﮐﻮﺗﺎﻩ ﻭ ﺟﺬﺍﺏ ﻗﺴﻤﺖ ﮐﻨﻴﺪ.

۳- ﻣﺤﺘﻮﺍﯼ ﺗﻮﻟﻴﺪﯼ ﺷﻤﺎ ﺑﺎﻳﺪ ﺁﻣﻮﺯﺷﯽ ﺑﺎﺷﺪ ﻧﻪ ﺗﺒﻠﻴﻐﺎﺗﯽ ﺩﺭ ﻏﻴﺮ ﺍﻳﻦ ﺻﻮﺭﺕ ﺑﻼﮒ ﻳﺎ ﻭﺏ‌ﺳﺎﻳﺖ ﺷﻤﺎ ﺩﻳﮕﺮ ﮐﺎﺭﮐﺮﺩ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﺭﻭﻧﮕﺮﺍ ﻧﺨﻮﺍﻫﺪ ﺩﺍﺷﺖ ﻭ ﺑﺮﺍﯼ ﺟﺬﺏ ﻭ ﺗﺒﺪﻳﻞ ﺭﻫﮕﺬﺭﺍﻥ ﺑﻪ ﻣﺸﺘﺮﯼ‌ﻫﺎﯼ ﺑﺎﻟﻘﻮﻩ ﻭ ﻧﻬﺎﻳﺘﺎ ﺧﺮﻳﺪﺍﺭ ﻣﻨﺎﺳﺐ ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ. ﺍﮔﺮ ﻣﻄﻠﺐ ﺷﻤﺎ ﺗﻨﻬﺎ ﺗﺒﻠﻴﻎ ﮐﺎﻻ ﻳﺎ ﺧﺪﻣﺎﺕ ﺷﻤﺎ ﺑﺎﺷﺪ ﺁﻧﮕﺎﻩ ﻭﺑﻼﮒ ﻳﺎ ﻭﺏ‌ﺳﺎﻳﺖ‌ﺗﺎﻥ ﺑﻪ ﺟﺎﯼ ﻳﮏ ﺩﺳﺘﮕﺎﻩ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺑﻪ ﻳﮏ ﺭﻭﺯﻧﺎﻣﻪ ﺩﻳﻮﺍﺭﯼ ﺍﺯ ﺍﺧﺒﺎﺭ ﻭ ﺍﻃﻼﻋﻴﻪ‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﺎ ﺷﺮﮐﺖ ﺷﻤﺎ ﺗﺒﺪﻳﻞ ﺧﻮﺍﻫﺪ ﺷﺪ.

۴- ﺑﻴﺸﺘﺮ ﺗﻤﺮﮐﺰ ﺧﻮﺩ ﺭﺍ ﺑﺮ ﺗﻬﻴﻪ ﻳﮏ ﻣﺤﺘﻮﺍﯼ ﻋﺎﻟﯽ ﻭ ﻣﻨﺤﺼﺮﺑﻔﺮﺩ ﺑﮕﺬﺍﺭﻳﺪ ﺑﻪ ﺟﺎﯼ ﺍﻳﻨﮑﻪ ﺳﻌﯽ ﮐﻨﻴﺪ ﻃﺮﺍﺣﯽ ﻭ ﺩﻳﺰﺍﻳﻦ ﻣﻄﻠﺐ‌ﺗﺎﻥ ﺭﺍ ﺑﺎﻻ ﺑﺒﺮﻳﺪ. ﻣﻨﺤﺼﺮﺑﻔﺮﺩ ﺑﻮﺩﻥ ﻣﺤﺘﻮﺍ ﺍﺯ ﻃﺮﺍﺣﯽ ﻭ ﺯﻳﺒﺎﻳﯽ ﻇﺎﻫﺮﯼ ﺁﻥ ﺍﻫﻤﻴﺖ ﺑﻴﺸﺘﺮﯼ ﺩﺍﺭﺩ.
