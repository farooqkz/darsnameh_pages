ﺩﺭﺱ ﺷﺸﻢ- ﺗﻘﻮﻳﺖ ﻣﺤﺘﻮﺍ ﺑﺎ ﺷﺒﮑﻪ ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ

«ﭼﺮﺍ ﺑﺎﻳﺪ ﺍﺯ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺮﺍﯼ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺧﻮﺩ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻢ؟»

ﺍﻳﻦ ﺳﻮﺍﻟﯽ ﺍﺳﺖ ﮐﻪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺩﺭ ﺍﺑﺘﺪﺍ ﺍﺯ ﺧﻮﺩ ﺑﭙﺮﺳﻴﺪ. ﺟﻮﺍﺏ ﺁﻥ ﺳﺎﺩﻩ ﺍﺳﺖ: «ﺯﻳﺮﺍ ﻣﺸﺘﺮﻳﺎﻥ ﺑﺎﻟﻘﻮﻩ ﮐﺴﺐ ﻭﮐﺎﺭ ﺷﻤﺎ ﺁﻧﺠﺎ ﻫﺴﺘﻨﺪ!». ﺁﻳﺎ ﻣﯽ‌ﺩﺍﻧﺴﺘﻴﺪ ﮐﻪ ﺑﻴﺶ ﺍﺯ ﺩﻭ ﻣﻴﻠﻴﺎﺭﺩ ﻧﻔﺮ ﺍﺯ ﺟﻤﻌﻴﺖ ﺟﻬﺎﻥ ﺩﺭ ﺣﺪﺍﻗﻞ ﻳﮑﯽ ﺍﺯ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺁﻧﻼﻳﻦ ﻋﻀﻮ ﻫﺴﺘﻨﺪ؟ ﺍﻳﻦ ﺟﻤﻌﻴﺖ ﭼﻴﺰﯼ ﺑﻴﺸﺘﺮ ﺍﺯ ۷۰ ﺩﺭﺻﺪ ﮐﻞ ﮐﺎﺭﺑﺮﺍﻥ ﺍﻳﻨﺘﺮﻧﺖ ﺟﻬﺎﻥ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ‌ﺩﻫﻨﺪ! 

ﻫﺮ ﺭﻭﺯﻩ ﻣﻴﻠﻴﻮﻥ‌ﻫﺎ ﻧﻔﺮ ﺍﺯ ﻣﺮﺩﻡ ﺟﻬﺎﻥ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ٬ ﺑﺮﻧﺪﻫﺎﯼ ﺑﺰﺭﮒ ﻭ ﮐﻮﭼﮏ ﺭﺍ ﺩﻧﺒﺎﻝ ﮐﺮﺩﻩ ﻭ ﺑﺎ ﺗﻮﺟﻪ ﺑﻪ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺍﺯ ﺍﻳﻨﺘﺮﻧﺖ ﺑﺪﺳﺖ ﻣﯽ‌ﺁﻭﺭﻧﺪ٬ ﺑﺮﺍﯼ ﺧﺮﻳﺪﻫﺎﯼ ﺑﻌﺪﯼ ﺧﻮﺩ ﺗﺼﻤﻴﻢ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺩﺭ ﻫﻤﻪ ﻣﺮﺍﺣﻞ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﺩﺭﻭﻧﮕﺮﺍ٬ ﺍﺯ ﺟﺬﺏ ﺭﻫﮕﺬﺭﺍﻥ ﺁﺷﻨﺎ ﮔﺮﻓﺘﻪ ﺗﺎ ﺗﺒﺪﻳﻞ ﺁﻧﻬﺎ ﺑﻪ ﻳﮏ ﻣﺸﺘﺮﯼ ﺑﺎﻟﻘﻮﻩ ﻭ ﺣﺘﯽ ﻧﮕﻬﺪﺍﺭﯼ ﻭ ﺭﺿﺎﻳﺘﻤﻨﺪﺳﺎﺯﯼ ﻣﺸﺘﺮﻳﺎﻥ ﮐﻨﻮﻧﯽ، ﺍﺑﺰﺍﺭ ﻣﻬﻤﯽ ﺑﺎﺷﻨﺪ.

ﺑﻪ ﮐﺎﺭﮐﺮﺩ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻫﻤﺎﻧﻨﺪ ﻧﮕﺎﺭﺵ ﻳﮏ ﺗﺤﻘﻴﻖ ﻋﻠﻤﯽ ﺑﻨﮕﺮﻳﺪ. ﺩﺭ ﻧﮕﺎﺭﺵ ﻳﮏ ﻣﻘﺎﻟﻪ ﻋﻠﻤﯽ ﺷﻤﺎ ﺍﺑﺘﺪﺍ ﺗﺤﻘﻴﻖ ﺍﻧﺠﺎﻡ ﺩﺍﺩﻩ ﻭ ﺳﭙﺲ ﻧﺘﺎﻳﺞ ﺗﺤﻘﻴﻖ ﺧﻮﺩ ﺭﺍ ﻣﯽ‌ﻧﮕﺎﺭﻳﺪ ﻭ ﺑﻌﺪ ﺍﺯ ﺍﻧﺘﺸﺎﺭ٬ ﺑﺎﺯﺧﻮﺭﺩﻫﺎﯼ ﺁﻥ ﺩﺭ ﺟﺎﻣﻌﻪ ﻣﻮﺭﺩ ﻧﻈﺮ ﺭﺍ ﺑﺮﺭﺳﯽ ﻣﯽ‌ﮐﻨﻴﺪ. ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻧﻴﺰ ﺑﻪ ﻫﻤﻴﻦ ﺗﺮﺗﻴﺐ ﺍﺳﺖ. ﺩﺭ ﺍﺑﺘﺪﺍ ﺑﺎ ﺑﺮﺭﺳﯽ٬ ﻣﺨﺎﻃﺐ ﺧﻮﺩ ﺭﺍ ﺷﻨﺎﺧﺘﻪ ﻭ ﻣﻄﻠﺒﯽ ﻣﻨﺎﺳﺐ ﺫﺍﺋﻘﻪ ﻭ ﻧﻴﺎﺯ ﺁﻧﻬﺎ ﺗﺪﺍﺭﮎ ﻣﯽ‌ﺑﻴﻨﻴﺪ. ﺑﻌﺪ ﺍﺯ ﺍﻧﺘﺸﺎﺭ ﻣﻄﻠﺐ٬ ﺑﺎﺯﺧﻮﺭﺩ ﺍﺟﺘﻤﺎﻋﯽ ﺁﻥ ﺭﺍ ﺑﺮﺭﺳﯽ ﻧﻤﻮﺩﻩ ﻭ ﺍﺯ ﻧﺘﺎﻳﺞ ﺁﻥ ﺑﺮﺍﯼ ﺑﻬﺒﻮﺩ ﻣﻄﺎﻟﺐ ﺑﻌﺪﯼ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﻧﻤﺎﻳﻴﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻣﻬﻤﺘﺮﻳﻦ ﺑﺨﺶ ﺑﺮﺍﯼ ﻣﻮﻓﻘﻴﺖ ﺷﻤﺎ ﺩﺭ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﻧﻔﻊ ﺗﺠﺎﺭﺕ ﺧﻮﺩ٬ ﺗﻌﻴﻴﻦ ﻣﺨﺎﻃﺒﺎﻥ ﻭ ﺑﺮﺭﺳﯽ ﻧﻴﺎﺯﻣﻨﺪﯼ‌ﻫﺎﯼ ﺁﻧﻬﺎ٬ ﺗﻌﻴﻴﻦ ﺍﻫﺪﺍﻑ ﻭ ﺗﺪﻭﻳﻦ ﻳﮏ ﺍﺳﺘﺮﺍﺗﮋﯼ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻫﺪﺍﻑ ﺗﻌﻴﻴﻦ ﺷﺪﻩ ﺍﺳﺖ. 

ﺍﺳﺘﺮﺍﺗﮋﯼ ﺧﻮﺩ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﺗﻌﻴﻴﻦ ﮐﻨﻴﺪ
ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ٬ ﺍﺑﺘﺪﺍ ﺍﻫﺪﺍﻑ ﺧﻮﺩ ﺭﺍ ﺗﻌﺮﻳﻒ ﮐﻨﻴﺪ. ﺁﻳﺎ ﻫﺪﻑ ﺷﻤﺎ ﺍﺯ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺭﺳﺎﻧﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺻﺮﻓﺎ ﺍﻓﺰﺍﻳﺶ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﻩ ﺳﺎﻳﺖ‌ﺗﺎﻥ ﺍﺳﺖ ﻳﺎ ﺗﺒﺪﻳﻞ ﺭﻫﮕﺬﺭﺍﻥ ﺍﻳﻨﺘﺮﻧﺘﯽ ﻧﺎﺁﺷﻨﺎ ﺑﻪ ﻣﺸﺘﺮﻳﺎﻥ ﺑﺎﻟﻘﻮﻩ ﻳﺎ ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻧﺰﺩﻳﮑﯽ ﺑﻴﺸﺘﺮ ﺑﺎ ﻣﺸﺘﺮﻳﺎﻥ ﮐﻨﻮﻧﯽ ﻭ ﻳﺎ ﺣﺘﯽ ﺍﻓﺰﺍﻳﺶ ﺭﺿﺎﻳﺘﻤﻨﺪﯼ ﺁﻧﻬﺎ؟ ﺗﻌﻴﻴﻦ ﻳﮏ ﻫﺪﻑ ﻣﺸﺨﺺ٬ ﻣﺮﺗﺒﻂ٬ ﻗﺎﺑﻞ ﺍﻧﺪﺍﺯﻩ‌ﮔﻴﺮﯼ٬ ﻗﺎﺑﻞ ﺩﺳﺘﺮﺱ ﻭ ﺩﺍﺭﺍﯼ ﻳﮏ ﺯﻣﺎﻧﺒﺪﯼ ﺩﻗﻴﻖ٬ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ ﻣﻮﻓﻘﻴﺖ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﺗﻌﻴﻴﻦ ﻫﺪﻑ٬ ﺷﻤﺎ ﻧﻴﺎﺯ ﺑﻪ ﺗﻘﺴﻴﻢ‌ﺑﻨﺪﯼ ﻭ ﺍﻧﺘﺨﺎﺏ ﻣﺨﺎﻃﺒﺎﻥ ﺧﻮﺩ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﻳﮑﯽ ﺍﺯ ﺭﺍﻩ‌ﻫﺎﯼ ﺗﻘﺴﻴﻢ‌ﺑﻨﺪﯼ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺗﻘﺴﻴﻢ ﺑﺮﺍﺳﺎﺱ ﻣﮑﺎﻥ ﺟﻐﺮﺍﻓﻴﺎﻳﯽ٬ ﺭﺩﻩ ﺳﻨﯽ٬ ﺯﺑﺎﻥ٬ ﺟﻨﺴﻴﺖ ﻭ ﻋﻼﻗﻪ‌ﻣﻨﺪﯼ‌ﻫﺎﯼ ﻣﺨﺎﻃﺒﺎﻥ ﺑﺎﺷﺪ. ﺑﻪ ﺍﻳﻦ ﺗﺮﺗﻴﺐ ﺷﻤﺎ ﺍﺯ ﺑﻴﻦ ﻫﺰﺍﺭﺍﻥ ﻧﻔﺮﯼ ﮐﻪ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻋﻀﻮ ﻫﺴﺘﻨﺪ٬ ﺗﻨﻬﺎ ﺑﺨﺸﯽ ﮐﻪ ﺩﺭ ﻣﮑﺎﻥ ﻣﺸﺨﺺ ﺯﻧﺪﮔﯽ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺯﺑﺎﻥ (ﻳﺎ ﺯﺑﺎﻥ‌ﻫﺎﯼ) ﻣﻌﻴﻨﯽ ﺻﺤﺒﺖ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺩﺍﺭﺍﯼ ﺟﻨﺴﻴﺖ ﺧﺎﺹ ﻭ ﻋﻼﻗﻤﻨﺪﯼ‌ﻫﺎﯼ ﻣﺸﺨﺼﯽ ﻫﺴﺘﻨﺪ ﺭﺍ ﻫﺪﻑ‌ﮔﺬﺍﺭﯼ ﻣﯽ‌ﮐﻨﻴﺪ. ﺍﻳﻦ ﮐﺎﺭ ﺑﻪ ﻃﺮﺯ ﺷﮕﻔﺖ‌ﺁﻭﺭﯼ ﺑﻪ ﺗﻌﻴﻴﻦ ﺍﺳﺘﺮﺍﺗﮋﯼ ﺷﻤﺎ ﮐﻤﮏ ﺧﻮﺍﻫﺪ ﮐﺮﺩ ﺯﻳﺮﺍ ﺷﻤﺎ ﺗﻌﺮﻳﻔﯽ ﻣﺸﺨﺺ ﺍﺯ ﻣﺨﺎﻃﺐ ﺧﻮﺩ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﮐﻪ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﺪ ﺷﻤﺎ ﺑﺎ ﺯﺑﺎﻥ ﻣﻨﺎﺳﺐ ﺑﺎ ﺁﻧﻬﺎ ﺳﺨﻦ ﺑﮕﻮﻳﻴﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﻓﺮﺽ ﮐﻨﻴﻢ ﮐﻪ ﺗﺠﺎﺭﺕ ﺷﻤﺎ ﺩﺭ ﺯﻣﻴﻨﻪ ﻃﺮﺍﺣﯽ ﺩﺍﺧﻠﯽ ﺳﺎﺧﺘﻤﺎﻥ ﺩﺭ ﻳﮏ ﺷﻬﺮ ﺑﺰﺭﮒ ﺑﻮﺩﻩ ﻭ ﺑﺎﺯﺍﺭ ﻫﺪﻑ ﺷﻤﺎ ﻃﺒﻘﻪ ﻓﺮﻫﻨﮕﯽ ﺗﺤﺼﻴﻞ ﮐﺮﺩﻩ ﻭ ﻣﺘﻮﺳﻂ ﺭﻭ ﺑﻪ ﺑﺎﻻﯼ ﺍﻳﻦ ﺷﻬﺮ ﺍﺳﺖ. ﺯﺑﺎﻥ ﻭ ﻣﺤﺘﻮﺍﻳﯽ ﮐﻪ ﺷﻤﺎ ﺑﺮﺍﯼ ﺟﺬﺏ ﺍﻳﻦ ﺩﺳﺘﻪ ﺍﺯ ﻣﺨﺎﻃﺒﺎﻥ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺑﺮﻳﺪ ﺑﺎ ﺯﺑﺎﻥ ﻭ ﻣﺤﺘﻮﺍﻳﯽ ﮐﻪ ﻳﮏ ﺳﺎﻳﺖ ﻟﻮﺍﺯﻡ ﻳﺪﮐﯽ ﻣﻮﺗﻮﺭﺳﻴﮑﻠﺖ ﺩﺭ ﻳﮏ ﺷﻬﺮﺳﺘﺎﻥ ﮐﻮﭼﮏ ﺑﺮﺍﯼ ﺟﺬﺏ ﻣﺨﺎﻃﺒﺎﻥ ﺟﻮﺍﻥ ﺧﻮﺩ ﺑﻪ ﮐﺎﺭ ﻣﯽ‌ﺑﺮﻳﺪ ﻓﺮﻕ ﺧﻮﺍﻫﺪ ﺩﺍﺷﺖ.

ﭘﺮﻭﻓﺎﻳﻞ ﺧﻮﺩ ﺩﺭ ﺷﺒﮑﻪ ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﺑﻬﻴﻨﻪ ﺳﺎﺯﯼ ﮐﻨﻴﺪ
ﺩﺭ ﻫﺮ ﺷﺒﮑﻪ ﺍﺟﺘﻤﺎﻋﯽ٬ ﻓﺎﺭﻍ ﺍﺯ ﻧﺎﻡ ﻭ ﻇﺎﻫﺮ ﺁﻥ٬ ﺑﻬﻴﻨﻪ‌ﺳﺎﺯﯼ ﭘﺮﻭﻓﺎﻳﻞ ﺷﻤﺎ ﺷﺎﻣﻞ ﻣﻮﺍﺭﺩ ﺯﻳﺮ ﺧﻮﺍﻫﺪ ﺑﻮﺩ:

ﺍﻭﻝ- ﻟﻮﮔﻮ ﻭ ﻧﺎﻡ ﺗﺠﺎﺭﯼ ﺷﻤﺎ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺁﻥ ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﺗﺼﻮﻳﺮ ﭘﺮﻭﻓﺎﻳﻞ ﻭ ﻳﺎ ﺗﺼﻮﻳﺮ ﺑﺎﻻﯼ ﺻﻔﺤﻪ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﺍﻳﻦ ﻟﻮﮔﻮ ﺑﺎﻳﺪ ﻗﺎﺑﻞ ﺗﻤﺎﻳﺰ ﻭ ﺑﺎ ﮐﻴﻔﻴﺖ ﺗﺼﻮﻳﺮﯼ ﺑﺎﻻ ﺑﺎﺷﺪ. ﺍﺯ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻋﮑﺲ‌ﻫﺎﯼ ﻣﺘﻔﺮﻗﻪ ﻭ ﺑﺎ ﮐﻴﻔﻴﺖ ﭘﺎﻳﻴﻦ ﻭ ﺑﺮﺵ‌ﻫﺎﯼ ﻧﺎﺩﺭﺳﺖ ﺗﺼﻮﻳﺮﯼ ﺑﭙﺮﻫﻴﺰﻳﺪ. ﺍﻳﻦ ﺻﻔﺤﻪ ﺑﺎﻳﺪ ﺑﺴﻴﺎﺭ ﺟﺬﺍﺏ ﻭ ﺣﺮﻓﻪ‌ﺍﯼ ﺑﻪ ﻧﻈﺮ ﺑﻴﺎﻳﻴﺪ. ﺍﺯ ﻫﺰﻳﻨﻪ ﮐﺮﺩﻥ ﺑﺮﺍﯼ ﻃﺮﺍﺣﯽ ﺩﺭﺳﺖ ﺩﺭ ﺍﻳﻦ ﺻﻔﺤﻪ ﺍﺑﺎﻳﯽ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ.

ﺩﻭﻡ- ﺗﻮﺿﻴﺤﺎﺕ ﮐﺎﻣﻞ ﻭ ﺩﻗﻴﻘﯽ ﺩﺭ ﻣﻮﺭﺩ ﺗﺠﺎﺭﺕ ﻭ ﺣﺮﻓﻪ ﺧﻮﺩ ﻭ ﻧﻴﺰ ﭘﻴﺸﻨﻬﺎﺩ ﺳﺮﻭﻳﺴﯽ ﮐﻪ ﺍﺭﺍﻳﻪ ﻣﯽ‌ﺩﻫﻴﺪ (ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﮐﺘﺎﺏ ﺍﻟﮑﺘﺮﻭﻧﻴﮏ ﺭﺍﻳﮕﺎﻥ٬ ﻣﺸﺎﻭﺭﻩ ﺭﺍﻳﮕﺎﻥ ﻭ ﻏﻴﺮﻩ...) ﺭﺍ ﺩﺭ ﭘﺮﻭﻓﺎﻳﻞ‌ﺗﺎﻥ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﺍﺯ ﮐﻠﻴﺪﻭﺍﮊﻩ‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺑﺎ ﺗﺠﺎﺭﺕ ﺧﻮﺩ ﺑﻴﺸﺘﺮ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺍﻳﻦ ﻳﮏ ﻓﺮﺻﺖ ﻣﻨﺎﺳﺐ ﺍﺳﺖ ﮐﻪ ﺧﻮﺩ ﺭﺍ ﺑﻪ ﻣﺸﺘﺮﻳﺎﻥ ﻣﺤﺘﻤﻞ ﺁﻳﻨﺪﻩ ﺑﺸﻨﺎﺳﺎﻧﻴﺪ.

ﺳﻮﻡ- ﻳﮏ ﻟﻴﻨﮏ ﻣﺴﺘﻘﻴﻢ ﺑﻪ ﺻﻔﺤﻪ ﺍﻭﻝ ﺳﺎﻳﺖ ﺧﻮﺩ ﺩﺭ ﭘﺮﻭﻓﺎﻳﻞ‌ﺗﺎﻥ ﻗﺮﺍﺭ ﺑﺪﻫﻴﺪ ﻳﺎ ﺣﺘﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻳﮏ ﺻﻔﺤﻪ ﻓﺮﻭﺩ ﺍﺧﺘﺼﺎﺻﯽ ﺑﺮﺍﯼ ﺍﻳﻨﮑﺎﺭ ﺩﺭ ﺳﺎﻳﺖ ﺧﻮﺩ ﺍﻳﺠﺎﺩ ﮐﺮﺩﻩ ﻭ ﻟﻴﻨﮏ ﺁﻥ ﺭﺍ ﺩﺭ ﭘﺮﻭﻓﺎﻳﻞ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ. ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺮﺍﯼ ﻣﺸﺘﺮﻳﺎﻧﯽ ﮐﻪ ﺍﺯ ﺍﻳﻦ ﺷﺒﮑﻪ ﺍﺟﺘﻤﺎﻋﯽ ﺑﻪ ﺳﺎﻳﺖ ﺷﻤﺎ ﻣﯽ‌ﺁﻳﻨﺪ ﭘﻴﺸﻨﻬﺎﺩ ﻭﻳﮋﻩ‌ﺍﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺩﺭ ﺩﺭﺱ‌ﻫﺎﯼ ﺑﻌﺪﯼ ﺩﺭ ﻣﻮﺭﺩ ﺻﻔﺤﻪ ﻓﺮﻭﺩ ﺑﻴﺸﺘﺮ ﺻﺤﺒﺖ ﺧﻮﺍﻫﻴﻢ ﮐﺮﺩ.

ﺳﻌﯽ ﮐﻨﻴﺪ ﺩﺭ ﺍﺭﺳﺎﻝ ﻭ ﺑﻪ ﺍﺷﺘﺮﺍﮎ‌ﮔﺬﺍﺭﯼ ﻣﻄﺎﻟﺐ٬ ﺗﺼﺎﻭﻳﺮ ﻭ ﺳﺎﻳﺮ ﻣﻨﺎﺑﻊ ﻭ ﺣﺘﯽ ﭼﮕﻮﻧﮕﯽ ﭘﺎﺳﺦ ﺩﺍﺩﻥ ﺑﻪ ﺳﻮﺍﻻﺕ ﻣﺸﺘﺮﻳﺎﻥ ﺑﺎ ﺑﺎﺯﺩﻳﺪﮐﻨﻨﺪﮔﺎﻥ ﺩﺭ ﺻﻔﺤﺎﺕ ﺍﺟﺘﻤﺎﻋﯽ ﺍﺯ ﻳﮏ ﺭﻭﻳﻪ ﺛﺎﺑﺖ ﻭ ﻳﮑﺴﺎﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺳﻌﯽ ﮐﻨﻴﺪ ﺑﺴﺘﻪ ﺑﻪ ﻧﻮﻉ ﺗﺠﺎﺭﺕ ﻭ ﺯﻣﻴﻨﻪ ﮐﺎﺭﯼ ﺧﻮﺩ، ﻳﮏ ﺯﺑﺎﻥ ﻭ ﻟﺤﻦ ﻣﺸﺨﺺ ﺳﺨﻦ ﮔﻔﺘﻦ ﺑﺎ ﻣﺨﺎﻃﺒﺎﻥ ﺧﻮﺩ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﺍﻧﺘﺨﺎﺏ ﻧﻤﻮﺩﻩ ﻭ ﻫﻤﻮﺍﺭﻩ ﺍﺯ ﺁﻥ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺩﺭ ﻭﺍﻗﻊ ﺑﺎﻳﺪ ﺑﺮﺍﯼ ﻫﺮ ﮐﺎﺭﯼ ﮐﻪ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺗﺤﺖ ﻋﻨﻮﺍﻥ ﺑﺮﻧﺪ ﺧﻮﺩﺗﺎﻥ ﺍﻧﺠﺎﻡ ﻣﯽ‌ﺩﻫﻴﺪ ﻳﮏ ﺍﺳﺘﺎﻧﺪﺍﺭﺩ ﺗﻌﺮﻳﻒ ﺷﺪﻩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﻭ ﻫﻤﻮﺍﺭﻩ ﺧﻮﺩ ﺭﺍ ﻣﺠﺎﺏ ﺑﻪ ﭘﻴﺮﻭﯼ ﺍﺯ ﺁﻥ ﺍﺳﺘﺎﻧﺪﺍﺭﺩ ﻧﻤﺎﻳﻴﺪ.

ﻳﮑﯽ ﺍﺯ ﺭﺍﻩ‌ﻫﺎﯼ ﻣﻌﻤﻮﻝ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ٬ ﺗﻬﻴﻪ ﻳﮏ ﮐﺘﺎﺑﭽﻪ «ﺭﺍﻫﻨﻤﺎﯼ ﻧﮕﺎﺭﺵ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ» ﺑﺮﺍﯼ ﺷﺮﮐﺖ ﺧﻮﺩﺗﺎﻥ ﺍﺳﺖ. ﺍﻳﻦ ﮐﺘﺎﺑﭽﻪ ﻗﺎﻧﻮﻥ ﺩﺭ ﻭﺍﻗﻊ ﺑﺎﻳﺪ ﻭ ﻧﺒﺎﻳﺪﻫﺎﯼ ﺷﻤﺎ ﺑﺮﺍﯼ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺭﺍ ﺗﻌﺮﻳﻒ ﮐﺮﺩﻩ ﻭ ﮐﺎﺭ ﺷﻤﺎ ﻭ ﻫﺮ ﺷﺨﺺ ﺩﻳﮕﺮﯼ ﮐﻪ ﻣﺴﻮﻝ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﻓﻌﺎﻟﻴﺖ ﺗﺠﺎﺭﯼ ﺷﻤﺎﺳﺖ ﺭﺍ ﺭﺍﺣﺖ ﻣﯽ‌ﻧﻤﺎﻳﺪ. ﺩﺭ ﺍﻳﻦ ﮐﺘﺎﺑﭽﻪ ﺷﻤﺎ ﺑﺎﻳﺪ ﻫﻤﻪ ﭼﻴﺰ ﺍﺯ ﻧﺤﻮﻩ ﭘﺎﺳﺨﮕﻮﻳﯽ ﺑﻪ ﻣﺸﺘﺮﻳﺎﻥ٬ ﻧﺤﻮﻩ ﻭ ﺯﻣﺎﻥ ﻭ ﺗﻌﺪﺍﺩ ﺩﻓﻌﺎﺕ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﮔﺬﺍﺷﺘﻦ ﻣﻄﺎﻟﺐ ﺩﺭ ﺭﻭﺯ٬ ﻗﻮﺍﻧﻴﻦ ﻧﮕﺎﺭﺷﯽ ﮐﻪ ﺑﺎﻳﺪ ﺩﺭ ﻧﮕﺎﺭﺵ ﻣﺤﺘﻮﺍ ﻣﻮﺭﺩ ﺗﻮﺟﻪ ﻗﺮﺍﺭ ﺑﮕﻴﺮﻧﺪ٬ ﺗﻌﺪﺍﺩ ﻫﺸﺘﮓ‌ﻫﺎﯼ ﻻﺯﻡ ﺑﺮﺍﯼ ﻫﺮ ﭘﺴﺖ٬ ﻟﺤﻦ ﻧﻮﺷﺘﻦ ﻣﻄﺎﻟﺐ٬ ﻧﻮﻉ ﺗﺼﺎﻭﻳﺮﯼ ﮐﻪ ﺑﺮﺍﯼ ﻫﺮ ﭘﺴﺖ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ‌ﺷﻮﺩ ﻭ ﺑﺴﻴﺎﺭﯼ ﻣﻮﺍﺭﺩ ﺍﺯ ﺍﻳﻦ ﻗﺒﻴﻞ ﺭﺍ ﺗﻌﺮﻳﻒ ﻧﻤﺎﻳﻴﺪ. 

ﺗﺎ ﺍﻳﻨﺠﺎ ﺷﻤﺎ ﭘﺮﻭﻓﺎﻳﻞ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺳﺎﺧﺘﻴﺪ ﻭ ﺍﻫﺪﺍﻑ ﻭ ﺍﺳﺘﺮﺍﺗﮋﯼ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺗﻌﺮﻳﻒ ﻧﻤﻮﺩﻳﺪ ﻭ ﺑﺮﺍﯼ ﻳﮑﺴﺎﻥ‌ﺳﺎﺯﯼ ﻓﻌﺎﻟﻴﺖ‌ﻫﺎﻳﺘﺎﻥ٬ ﮐﺘﺎﺑﭽﻪ ﺭﺍﻫﻨﻤﺎﯼ ﮐﻮﭼﮑﯽ ﻫﻢ ﺗﻬﻴﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ. ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ ﻣﺤﺘﻮﺍ ﺍﺯ ﺩﺭﺱ‌ﻫﺎﯼ ﻣﺮﺗﺒﻂ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ ﮐﻪ ﭘﻴﺸﺘﺮ ﮔﻔﺘﻪ ﺷﺪ ﺍﺳﺘﻔﺎﺩﻩ ﮐﻨﻴﺪ. ﺩﺭ ﺍﻳﻦ ﻣﺮﺣﻠﻪ ﺑﺎﻳﺪ ﮔﺰﺍﺭﺷﯽ ﺍﺯ ﻋﻤﻠﮑﺮﺩ ﺧﻮﺩ ﺗﻬﻴﻪ ﻧﻤﻮﺩﻩ ﻭ ﮐﺎﺭﮐﺮﺩ ﺧﻮﺩ ﺭﺍ ﺑﺮﺭﺳﯽ ﻧﻤﺎﻳﻴﺪ.
