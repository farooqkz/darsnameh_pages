ﭼﮏ ﻟﻴﺴﺖ: ۷ ﺳﻮﺍﻝ ﺑﺮﺍﯼ ﺗﺸﺨﻴﺺ ﺍﻳﻨﮑﻪ ﺁﻣﺎﺩﻩ ﺍﻧﺘﺸﺎﺭ ﺷﺨﺼﯽ ﮐﺘﺎﺏ ﻫﺴﺘﻴﺪ ﻳﺎ ﺧﻴﺮ
ﺯﻣﺎﻧﯽ ﺑﻮﺩ ﮐﻪ ﺩﺭ ﺍﻳﺮﺍﻥ «ﻣﻮﺝ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻣﻮﻓﻘﻴﺖ» ﺑﻪ ﺭﺍﻩ ﺍﻓﺘﺎﺩﻩ ﺑﻮﺩ. ﺗﻘﺮﻳﺒﺎ ﺍﮐﺜﺮ ﺁﻧﻬﺎ ﺗﺮﺟﻤﻪ‌ﻫﺎﻳﯽ ﺍﺯ ﻧﻮﻳﺴﻨﺪﮔﺎﻥ ﺍﻣﺮﻳﮑﺎﻳﯽ ﺑﻮﺩﻧﺪ. ﺑﺎ ﻓﺎﺻﻠﻪ ﮔﺮﻓﺘﻦ ﺍﺯ ﺟﻮ ﺁﻥ ﺭﻭﺯﮔﺎﺭ ﺑﻪ ﺭﺍﺣﺘﯽ ﻣﯽ‌ﺷﻮﺩ ﻓﻬﻤﻴﺪ ﮐﻪ ﺍﮐﺜﺮ ﺁﻥ ﮐﺘﺎﺏ‌ﻫﺎ، ﮐﻪ ﻧﻮﺷﺘﻪ ﻣﺪﺭﺳﺎﻥ ﺩﻭﺭﻩ‌ﻫﺎﯼ ﻣﻮﻓﻘﻴﺖ ﺩﺭ ﺍﻣﺮﻳﮑﺎ ﺑﻮﺩﻧﺪ، ﺍﺳﺎﺳﺎً ﺁﻣﺎﺩﻩ ﺍﻧﺘﺸﺎﺭ ﻧﺒﻮﺩﻩ‌ﺍﻧﺪ. ﺣﺘﯽ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﺘﺮﺟﻤﺎﻥ ﺁﻥ ﮐﺘﺐ ﻫﻢ ﺁﻣﺎﺩﻩ ﺗﺮﺟﻤﻪ ﻧﺒﻮﺩﻧﺪ.

ﺍﻣﺎ ﺩﺭ ﺭﻭﺯﮔﺎﺭ ﻓﻌﻠﯽ ﻭ ﺑﻪ ﺧﺼﻮﺹ ﺑﺮﺍﯼ ﺍﻧﺘﺸﺎﺭ ﮐﺘﺎﺏ ﺩﻳﺠﻴﺘﺎﻝ، ﻫﻴﭻ ﻣﺤﺪﻭﺩﻳﺘﯽ ﺟﻠﻮﯼ ﺭﻭﯼ ﻣﺎ ﻧﻴﺴﺖ. ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻫﺮ ﺁﻥ ﭼﻴﺰﯼ ﺭﺍ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﺑﻨﻮﻳﺴﻴﻢ ﻭ ﺩﺭ ﺍﻳﻨﺘﺮﻧﺖ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﻢ. ﻭﻟﯽ ﺑﺮﺍﯼ ﺁﻧﻬﺎﻳﯽ ﮐﻪ ﮐﺎﺭ ﺭﺍ ﺟﺪﯼ ﮔﺮﻓﺘﻪ‌ﺍﻧﺪ ﻭ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﻧﻮﻳﺴﻨﺪﮔﯽ -ﻳﺎ ﺗﺮﺟﻤﻪ- ﺭﺍ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮑﯽ ﺍﺯ ﺍﺑﻌﺎﺩ ﻣﻬﻢ ﺯﻧﺪﮔﯽ ﺧﻮﺩ ﮔﺴﺘﺮﺵ ﺩﻫﻨﺪ، ﮐﺎﺭ ﺑﻪ ﺍﻳﻦ ﺳﺎﺩﮔﯽ ﻧﻴﺴﺖ. ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﻭﺳﻮﺍﺱ ﺩﺭﻭﻧﯽ ﺑﺮﺍﯼ ﺍﺭﺍﺋﻪ ﮐﺎﺭﯼ ﺷﺎﻳﺴﺘﻪ ﺩﺍﺭﻧﺪ، ﺑﻪ ﺧﻮﺑﯽ ﻣﯽ‌ﺩﺍﻧﻨﺪ ﮐﻪ ﺧﻠﻖ ﻳﮏ ﺍﺛﺮ ﭼﻪ ﺩﺭ ﺯﻣﻴﻨﻪ ﻧﻮﺷﺘﺎﺭ ﻳﺎ ﻣﻮﺳﻴﻘﯽ ﻳﺎ ﻧﻘﺎﺷﯽ ﻳﺎ... ﭼﻪ ﮐﺎﺭ ﺑﺎﺭﻳﮏ-ﺑﻴﻨﺎﻧﻪ ﻭ ﻣﻬﻤﯽ ﺍﺳﺖ.

ﺍﻳﻦ ﻃﻮﺭ ﺗﺼﻮﺭ ﮐﻨﻴﺪ ﮐﻪ ﺍﺛﺮ ﺷﻤﺎ ﭘﺲ ﺍﺯ ﺍﻧﺘﺸﺎﺭ ﺩﺭ ﻣﻌﺮﺽ ﺩﻳﺪ ﻋﺪﻩ ﺯﻳﺎﺩﯼ ﺍﺯ ﺻﺎﺣﺐ ﻧﻈﺮﺍﻥ ﺩﺭ ﺁﻥ ﺭﺷﺘﻪ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﺩ. ﺁﻥ‌ﻫﺎ ﻫﻢ ﺍﺯ ﻣﻮﻫﺒﺖ ﺍﻳﻨﺘﺮﻧﺖ ﺑﺮﺧﻮﺭﺩﺍﺭﻧﺪ ﻭ ﺑﺪﻭﻥ ﺍﻏﻤﺎﺽ ﻧﻘﺪﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺑﺮ ﺁﻥ ﺍﺛﺮ ﺧﻮﺍﻫﻨﺪ ﻧﻮﺷﺖ. ﻧﻘﺪ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ‌‌ ﻫﻤﺎﻥ ﺳﺮﻋﺖ ﺗﺒﺪﻳﻞ ﺑﻪ ﺭﻭﻧﺪﻫﺎﯼ ﺗﺨﺮﻳﺐ‌ﮔﺮﯼ ﺩﺭ ﻋﺮﺻﻪ ﻭﺏ ﺷﻮﻧﺪ ﻭ ﻓﺎﺗﺤﻪ ﮐﺘﺎﺏ ﺷﻤﺎ ﺭﺍ ﺑﺨﻮﺍﻧﻨﺪ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺁﻣﺎﺩﮔﯽ ﺑﺮﺍﯼ ﺗﺎﻟﻴﻒ ﻳﺎ ﺗﺮﺟﻤﻪ ﻳﮏ ﮐﺘﺎﺏ ﺍﺑﺪﺍ ﺷﻮﺧﯽ ﺑﺮﺩﺍﺭ ﻧﻴﺴﺖ. ﺩﺭ ﺍﻳﻨﺠﺎ ﻣﯽ‌ﺧﻮﺍﻫﻴﻢ ﻋﻼﻭﻩ ﺑﺮ ﻣﻮﺍﺭﺩﯼ ﮐﻪ ﭘﻴﺶ‌ﺗﺮ ﺑﺮﺍﯼ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﺫﻫﻨﯽ -ﻭ ﻧﻪ ﻓﻨﯽ ﻭ ﺣﺮﻓﻪ ﺍﯼ- ﺷﻤﺎ ﺫﮐﺮ ﮐﺮﺩﻳﻢ، ﺑﻪ ﺍﺭﺍﺋﻪ ﻣﻌﻴﺎﺭﻫﺎﻳﯽ ﺑﭙﺮﺩﺍﺯﻳﻢ ﮐﻪ ﻧﺪﺍﺷﺘﻦ ﺁﻧﻬﺎ ﺑﻪ ﻣﻌﻨﺎﯼ ﻋﻼﻣﺖ ﺍﻳﺴﺖ ﺍﺳﺖ.

ﺑﺮﺍﯼ ﻫﺮ ﮐﺴﯽ ﺳﺨﺖ ﺍﺳﺖ ﮐﻪ ﺑﺨﻮﺍﻫﺪ ﺑﻪ ﻃﻮﺭ ﻭﺍﻗﻌﯽ ﭘﻴﺸﺮﻓﺖ ﺧﻮﺩﺵ ﺭﺍ ﺍﺭﺯﻳﺎﺑﯽ ﮐﻨﺪ. ﻧﻔﺲ ﺷﻤﺎ ﻣﺪﺍﻡ ﻧﺠﻮﺍ ﻣﯽ‌ﮐﻨﺪ «ﺍﻟﺒﺘﻪ ﮐﻪ ﺁﻣﺎﺩﻩ‌ﺍﯼ» ﻭ ﺑﺨﺸﯽ ﺍﺯ ﻣﺸﮑﻞ ﺑﻐﺮﻧﺞ ﺁﻣﺎﺩﻩ ﻧﺒﻮﺩﻥ، ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻫﻨﻮﺯ ﺍﺑﺰﺍﺭﻫﺎﯼ ﻻﺯﻡ ﺑﺮﺍﯼ ﻓﻬﻢ ﻣﻴﺰﺍﻥ ﺁﻣﺎﺩﮔﯽ ﺧﻮﺩ ﺭﺍ ﻧﻤﯽ‌ﺷﻨﺎﺳﻴﺪ. ﺍﻟﺒﺘﻪ ﮐﻪ ﺩﺭ ﺩﻧﻴﺎﯼ ﮐﺘﺎﺏ‌ﻫﺎ ﺳﻨﺠﺶ ﻭﺍﻗﻌﯽ ﺑﺴﻴﺎﺭ ﺳﺨﺖ ﻭ ﮔﺎﻫﯽ ﺍﻧﺘﺰﺍﻋﯽ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ ﻭﻟﯽ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﻣﻮﺍﺭﺩﯼ ﺭﺍ ﺑﺮ ﺷﻤﺎﺭﻳﻢ ﮐﻪ ﺍﮐﺜﺮ ﻣﺆﻟﻔﺎﻥ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻣﻮﻓﻖ ﺁﻧﻬﺎ ﺭﺍ ﺩﺍﺭﺍ ﺑﻮﺩﻩ‌ﺍﻧﺪ.

ﺍﻟﺒﺘﻪ ﭼﻨﻴﻦ ﻣﻌﻴﺎﺭﻫﺎﻳﯽ ﺭﺍ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺑﺮ ﻗﺎﻣﺖ ﻫﻤﻪ ﺩﻭﺧﺖ. ﺑﻪ ﺧﺼﻮﺹ ﻧﺎﺑﻐﻪ‌ﻫﺎ ﻭ ﻧﻴﺰ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺑﺎ ﺍﻭﻟﻴﻦ ﮐﺘﺎﺏ ﺧﻮﺩﺷﺎﻥ ﺗﺒﺪﻳﻞ ﺑﻪ ﭘﺮﻓﺮﻭﺵ‌ﺗﺮﻳﻦ ﻧﻮﻳﺴﻨﺪﻩ‌ﻫﺎ ﻣﯽ‌ﺷﻮﻧﺪ ﻳﺎ ﻧﺨﺒﻪ‌ﻫﺎﯼ ﺗﻌﻠﻴﻢ ﻧﺪﻳﺪﻩ‌ﺍﯼ ﮐﻪ ﺑﻪ ﻧﻈﺮ ﻣﯽ‌ﺭﺳﺪ ﺍﺯ ﻣﻮﺍﻫﺐ ﺧﺪﺍ ﺩﺍﺩﯼ ﺑﺮﺧﻮﺭﺩﺍﺭﻧﺪ.

ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ ﮐﻪ ﺍﮐﺜﺮ ﺍﻳﻦ ﺍﻓﺮﺍﺩ ﺑﺪﻭﻥ ﺟﻮ ﺳﺎﺯﯼ ﺑﺎﺯﺍﺭﻳﺎﺑﺎﻥ ﺗﺒﺪﻳﻞ ﺑﻪ ﺳﺘﺎﺭﻩ‌ﻫﺎﯼ ﭘﻴﺸﺨﻮﺍﻥ ﮐﺘﺎﺏ ﻓﺮﻭﺷﯽ‌ﻫﺎ ﻧﻤﯽ‌ﺷﺪﻧﺪ ﻭ ﺑﻴﺸﺘﺮ ﺁﻧﻬﺎ ﻫﻢ ۱۰ ﻳﺎ ۲۰ ﺳﺎﻝ ﺣﺮﻓﻪ ﻋﻮﺽ ﮐﺮﺩﻩ‌ﺍﻧﺪ ﺗﺎ ﻋﺎﻗﺒﺖ ﻟﻨﮕﺮﺷﺎﻥ ﺑﻪ ﮐﻒ ﺩﺭﻳﺎﯼ ﻧﺸﺮ ﻧﺸﺴﺘﻪ.

ﻗﻄﻌﺎً ﺍﻓﺮﺍﺩﯼ ﺩﺭ ﭘﻴﺮﺍﻣﻮﻥ ﺷﻤﺎ ﻳﺎﻓﺖ ﻣﯽ‌ﺷﻮﻧﺪ ﮐﻪ ﺑﮕﻮﻳﻨﺪ ﻧﻴﺎﺯﯼ ﺑﻪ ﺩﺍﺷﺘﻪ‌ﻫﺎﯼ ﺷﺨﺼﯽ، ﻳﻌﻨﯽ ﺍﻧﺒﺎﺷﺖ ﺗﺠﺮﺑﻪ ﻭ ﺩﺍﻧﺶ ﮐﺎﻓﯽ ﻧﺪﺍﺭﻳﺪ ﻭ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺗﮑﻴﻪ ﺑﺮ ﺿﻤﻴﺮ ﻧﺎﺧﻮﺩﺁﮔﺎﻩ‌ﺗﺎﻥ ﻗﻠﻪ‌ﻫﺎﯼ ﻣﻮﻓﻘﻴﺖ ﺭﺍ ﺩﺭ ﻧﻮﺭﺩﻳﺪ. ﻭﻟﯽ ﻭﺍﻗﻌﻴﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻫﻴﭻ ﻓﺮﺩ ﻣﻮﻓﻘﯽ ﺑﺪﻭﻥ ﺗﻼﺵ ﺑﻪ ﺟﺎﻳﯽ ﻧﺮﺳﻴﺪﻩ ﻭ ﺷﻤﺎ ﻫﻢ ﺑﻬﺘﺮ ﺍﺳﺖ ﻋﻤﺮ ﺧﻮﺩ ﺭﺍ ﺑﺎﺑﺖ ﺣﺮﻑ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺑﯽ‌ﭘﺸﺘﻮﺍﻧﻪ ﺧﻮﺷﺒﻴﻦ ﻫﺴﺘﻨﺪ، ﺗﻠﻒ ﻧﮑﻨﻴﺪ.

ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺩﺭ ﺣﺮﻓﻪ‌ﺍﯼ ﺑﻪ ﻣﻮﻓﻘﻴﺖ ﺑﺮﺳﻴﺪ ﺗﻨﻬﺎ ﺭﺍﻩ ﺁﻥ ﮐﺴﺐ ﺩﺍﻧﺶ ﺍﺳﺖ. ﻗﻤﺎﺭ ﮐﺮﺩﻥ ﺑﺮ ﺭﻭﯼ ﺣﺮﻑ‌ﻫﺎﯼ ﭘﻮﭺ ﺩﻳﮕﺮﺍﻥ، ﻓﻘﻂ ﻳﮏ ﻫﺰﻳﻨﻪ ﺑﺮﺍﯼ ﺷﻤﺎ ﺩﺍﺭﺩ: ﻋﻤﺮﺗﺎﻥ! ﻭ ﻫﻴﭻ ﮐﺲ ﻧﻤﯽ‌ﺗﻮﺍﻧﺪ ﻋﻤﺮ ﺭﻓﺘﻪ ﺭﺍ ﺑﺎﺯ ﮔﺮﺩﺍﻧﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﺗﺨﺼﻴﺺ ﻋﻤﺮﺗﺎﻥ ﻫﻮﺷﻤﻨﺪﺍﻧﻪ ﻋﻤﻞ ﮐﻨﻴﺪ. ﺁﻳﺎ ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻫﻴﺪ ﺁﻥ ﺭﺍ ﺑﻪ ﺣﺮﻑ‌ﻫﺎﯼ ﺑﯽ‌ﭘﺸﺘﻮﺍﻧﻪ ﺍﻓﺮﺍﺩﯼ ﺍﺧﺘﺼﺎﺹ ﺩﻫﻴﺪ ﮐﻪ ﺧﻮﺩﺷﺎﻥ ﺩﺭ ﺯﻧﺪﮔﯽ ﺑﻪ ﺟﺎﻳﯽ ﻧﺮﺳﻴﺪﻩ‌ﺍﻧﺪ؟ ﻳﺎ ﺗﺮﺟﻴﺢ ﻣﯽ‌ﺩﻫﻴﺪ ﻣﺜﻞ ﺍﻓﺮﺍﺩ ﻣﻮﻓﻖ، ﺑﻪ ﺩﺍﻧﺶ ﺍﻧﺪﻭﺯﯼ ﻭ ﺣﺮﻓﻪ‌ﺍﯼ ﺷﺪﻥ ﺑﭙﺮﺩﺍﺯﻳﺪ؟

ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﺘﻮﺍﻥ ﮔﻔﺖ ﺁﻣﺎﺩﻩ ﻧﻮﺷﺘﻦ ﻳﮏ ﮐﺘﺎﺏ ﻫﺴﺘﻴﺪ، ۷ ﭘﺮﺳﺶ ﺯﻳﺮ ﺭﺍ ﺍﺯ ﺧﻮﺩ ﺑﭙﺮﺳﻴﺪ:

۱- ﺁﻳﺎ ﺍﻳﺪﻩ ﺑﺴﻴﺎﺭ ﺧﻮﺑﯽ ﺩﺍﺭﻳﺪ؟ ﮔﻔﺘﻦ ﺍﻳﻨﮑﻪ ﮐﺘﺎﺑﯽ ﻣﺜﻞ «ﺭﻣﺰ ﺩﺍﻭﻳﻨﭽﯽ» ﺑﺪ ﻧﻮﺷﺘﻪ ﺷﺪﻩ ﺁﺳﺎﻥ ﺍﺳﺖ. ﺳﻄﻮﺡ ﻣﺨﺘﻠﻔﯽ ﺩﺭ ﺭﺗﺒﻪ ﺑﻨﺪﯼ ﻳﮏ ﺍﺛﺮ ﻭﺟﻮﺩ ﺩﺍﺭﻧﺪ ﻭﻟﯽ ﮐﺘﺎﺏ ﺑﺴﻴﺎﺭ ﭘﺮ ﻓﺮﻭﺵ «ﺩﻥ ﺑﺮﺍﻭﻥ» ﺍﺯ ﻧﻈﺮ ﺍﻳﺪﻩ‌ﭘﺮﺩﺍﺯﯼ، ﺑﺴﻴﺎﺭ ﺩﺭﺧﺸﺎﻥ ﺍﺳﺖ. ﺍﺯ ﺁﻥ ﺩﺳﺖ ﮐﺘﺎﺏ‌ﻫﺎﻳﯽ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻥ ﺩﺭ ﺗﺒﻠﻴﻐﺎﺕ ﺭﻭﯼ ﺁﻧﻬﺎ ﺧﻴﻠﯽ ﻣﺎﻧﻮﺭ ﺩﺍﺩ ﺯﻳﺮﺍ ﺑﺨﺶ ﺑﺰﺭﮔﯽ ﺍﺯ ﺗﻼﺵ ﻧﻮﻳﺴﻨﺪﻩ، ﻣﻌﻄﻮﻑ ﺑﻪ ﺍﻳﺪﻩ ﺩﺍﺳﺘﺎﻥ ﺑﻮﺩﻩ.

ﺍﻳﺪﻩ‌ﺍﯼ ﻋﺎﻟﯽ ﻣﺜﻞ ﺭﻣﺰ ﺩﺍﻭﻳﻨﭽﯽ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺍﺯ ﻋﻨﻮﺍﻥ ﮔﺮﻓﺘﻪ ﺗﺎ ﺟﺰﺋﻴﺎﺕ ﺩﺍﺳﺘﺎﻧﯽ، ﺑﺮ ﺟﺬﺏ ﻣﺨﺎﻃﺐ ﻭ ﻓﺮﻭﺵ ﺑﻬﺘﺮ ﮐﺘﺎﺏ ﺗﺄﺛﻴﺮ ﺑﮕﺬﺍﺭﺩ. ﺁﻳﺎ ﺷﻤﺎ ﺍﻳﺪﻩ ﺟﺬﺍﺏ ﻭ ﭘﺮﺩﺍﺧﺖ ﺷﺪﻩ‌ﺍﯼ ﺩﺍﺭﻳﺪ؟ ﺯﻳﺮﺍ ﺍﮔﺮ ﻧﺪﺍﺭﻳﺪ ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﺪ ﺗﻘﻠﻴﺪ ﮐﺮﺩﻥ ﺍﺯ ﻳﮏ ﺍﻳﺪﻩ ﭘﺮ ﻓﺮﻭﺵ، ﻣﻨﺠﺮ ﺑﻪ ﺧﻠﻖ ﻳﮏ ﮐﺘﺎﺏ ﭘﺮ ﻣﺨﺎﻃﺐ ﻧﻤﯽ‌ﺷﻮﺩ.

۲- ﺁﻳﺎ ﺑﺮﺍﯼ ۱۰.۰۰۰ ﺳﺎﻋﺖ ﺗﻤﺮﻳﻦ ﻭ ﮐﺴﺐ ﺗﺠﺮﺑﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ؟ ﺧﻴﻠﯽ‌ﻫﺎ ۱۰.۰۰۰ ﺳﺎﻋﺖ ﺭﺍ ﻳﮏ ﻣﺮﺯ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﻧﻈﻢ ﺩﺭ ﻫﺮ ﻧﻮﻉ ﮐﺎﺭﯼ ﻣﯽ‌ﺩﺍﻧﻨﺪ. ﺍﻳﻦ ﻣﻌﺎﺩﻝ ﺯﻣﺎﻧﯽ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺩﺭ ﺳﻨﻴﻦ ﻧﻮﺟﻮﺍﻧﻴﺘﺎﻥ ﺩﺭ ﻣﺪﺭﺳﻪ ﮔﺬﺭﺍﻧﺪﻩ‌ﺍﻳﺪ ﻭ ﻫﻤﻴﻦ ﻃﻮﺭ ﻣﺪﺕ ﺯﻣﺎﻧﯽ ﮐﻪ ﮔﺮﻭﻩ ﻣﻮﺳﻴﻘﯽ ﺑﻴﺘﻠﺰ ﻗﺒﻞ ﺍﺯ ﺿﺒﻂ ﻫﺮ ﺁﻟﺒﻮﻡ ﺧﻮﺩ، ﺑﻪ ﻃﻮﺭ ﺯﻧﺪﻩ ﺭﻭﯼ ﺁﻥ ﮐﺎﺭ ﻣﯽ‌ﮐﺮﺩﻧﺪ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺪ ﻧﻴﺴﺖ ﮐﻪ ﺑﮕﻮﻳﻴﻢ ﺑﺮﺍﯼ ﺍﻧﺘﺸﺎﺭ ﻳﮏ ﮐﺘﺎﺏ، ﺷﻤﺎ ﻫﻢ ﻧﻴﺎﺯ ﺑﻪ ﺗﻤﺮﻳﻦ ﻧﻮﻳﺴﻨﺪﮔﯽ ﺑﻪ ﻣﺪﺕ ۱۰.۰۰۰ ﺳﺎﻋﺖ ﺩﺍﺭﻳﺪ. ﺍﮔﺮ ﺑﻪ ﻧﻮﻳﺴﻨﺪﮔﯽ ﺑﻪ ﺩﻳﺪ ﻳﮏ ﮐﺎﺭ ﭘﺎﺭﻩ ﻭﻗﺖ ﻧﮕﺎﻩ ﻣﯽ‌ﮐﻨﻴﺪ ﻭ ﻣﺜﻼً ﻫﻔﺘﻪ‌ﺍﯼ ۲۰ ﺳﺎﻋﺖ ﺯﻣﺎﻥ ﺑﻪ ﺁﻥ ﺍﺧﺘﺼﺎﺹ ﻣﯽ‌ﺩﻫﻴﺪ، ۱۰ ﺳﺎﻝ ﺍﺯ ﺷﻤﺎ ﺯﻣﺎﻥ ﻣﯽ‌ﺑﺮﺩ ﺗﺎ ﺑﻪ ﺩﺭﺟﻪ ﺍﺳﺘﺎﺩﯼ ﺑﺮﺳﻴﺪ. ﺍﻳﻦ ۱۰ ﺳﺎﻝ ﺗﻘﺮﻳﺒﺎً ﺩﺭ ﻣﻮﺭﺩ ﺍﮐﺜﺮ ﻧﻮﻳﺴﻨﺪﮔﺎﻥ ﻣﻮﻓﻖ ﺻﺪﻕ ﻣﯽ‌ﮐﻨﺪ.

ﺍﻟﺒﺘﻪ ﮔﺬﺭﺍﻧﺪﻥ ۱۰.۰۰۰ ﺳﺎﻋﺖ ﺗﻤﺮﻳﻦ ﻭ ﺁﻣﻮﺯﺵ ﺑﺪﺍﻥ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﻧﺒﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﻣﺪﺕ ﮐﺘﺎﺑﯽ ﻣﻨﺘﺸﺮ ﮐﻨﻴﺪ. ﺷﻤﺎ ﺑﺎﻳﺪ ﺩﺭ ﺍﻳﻦ ﻣﺪﺕ ﻃﻮﻻﻧﯽ، ﺑﺴﻴﺎﺭ ﺗﻤﺮﻳﻦ ﮐﻨﻴﺪ ﻭ ﻣﻄﺎﻟﻌﺎﺕ ﻣﺨﺘﻠﻔﯽ ﺭﺍ ﻫﻢ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﺗﺎ ﻣﺮﺣﻠﻪ‌ﺍﯼ ﺑﺮﺳﻴﺪ ﮐﻪ ﺩﺭ ﮐﺎﺭﺗﺎﻥ ﻧﻈﻢ (ﺩﻳﺴﻴﭙﻠﻴﻦ) ﻭ ﺍﺳﺘﻘﻼﻝ ﻗﻠﻢ ﭘﻴﺪﺍ ﮐﻨﻴﺪ. ﺍﮔﺮ ﺩﺭ ﺍﻳﻦ ﻣﺪﺕ ﮐﺘﺎﺑﯽ ﻫﻢ ﻣﻨﺘﺸﺮ ﮐﻨﻴﺪ، ﮐﺎﺭ ﺍﺷﺘﺒﺎﻫﯽ ﻧﻴﺴﺖ ﻭﻟﯽ ﺷﺎﻳﺪ ﺑﻌﺪ‌ﻫﺎ ﺍﺯ ﻭﺟﻮﺩ ﺁﻥ ﺩﺭ ﺳﺎﺑﻘﻪ ﮐﺎﺭﻳﺘﺎﻥ ﺧﻮﺷﺤﺎﻝ ﻧﺸﻮﻳﺪ.

۳- ﺁﻳﺎ ﮐﺎﺭﺗﺎﻥ ﺭﺍ ﺑﻪ ﻳﮏ ﻣﻨﺘﻘﺪ ﺟﺪﯼ ﺍﺭﺍﺋﻪ ﮐﺮﺩﻩ‌ﺍﻳﺪ؟ ﻫﻴﭻ ﮐﺲ ﻣﻮﺍﻓﻖ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﻳﮏ ﻣﺤﺼﻮﻝ ﺭﺍ ﺑﺪﻭﻥ ﺑﺮﺭﺳﯽ ﻫﻤﻪ ﺟﺎﻧﺒﻪ ﺑﻪ ﻣﺨﺎﻃﺐ ﻋﺮﺿﻪ ﮐﻨﻴﺪ. ﺩﺭ ﺻﻮﺭﺕ ﻧﺎﺩﻳﺪﻩ ﮔﺮﻓﺘﻦ ﺍﻳﻦ ﺍﺻﻞ، ﺍﺣﺘﻤﺎﻝ ﺷﮑﺴﺖ ﻭ ﺑﺪﻧﺎﻣﯽ ﺑﺴﻴﺎﺭ ﺯﻳﺎﺩ ﺍﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﻳﮏ ﻣﻨﺘﻘﺪ ﺑﺴﻴﺎﺭ ﺟﺪﯼ ﺑﺮﺍﯼ ﮐﺘﺎﺑﺘﺎﻥ ﭘﻴﺪﺍ ﮐﻨﻴﺪ. ﺗﺮﺟﻴﺤﺎً ﮐﺴﯽ ﻏﻴﺮ ﺍﺯ ﺧﺎﻧﻮﺍﺩﻩ، ﻣﮕﺮ ﺍﻳﻨﮑﻪ ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﺁﻧﻬﺎ ﺑﻪ ﺷﻤﺎ ﺑﺎﺯﺧﻮﺭﺩ ﺻﺎﺩﻗﺎﻧﻪ‌ﺍﯼ ﻣﯽ‌ﺩﻫﻨﺪ.

ﺍﻳﻦ ﻓﺮﺁﻳﻨﺪ، ﻗﻄﻌﺎً ﺩﺭﺩﻧﺎﮎ ﻳﺎ ﻧﺎﺭﺍﺣﺖ ﮐﻨﻨﺪﻩ ﻧﻴﺰ ﺧﻮﺍﻫﺪ ﺑﻮﺩ ﭼﺮﺍ ﮐﻪ ﻫﻴﭻ ﺍﺛﺮﯼ ﺑﺪﻭﻥ ﺍﻳﺮﺍﺩ ﻧﻴﺴﺖ. ﭘﺲ ﺑﺎﻳﺪ ﺍﻧﺘﻈﺎﺭ ﻭﺍﺭﺩ ﺁﻣﺪﻥ ﺟﺪﯼ‌ﺗﺮﻳﻦ ﺍﻧﺘﻘﺎﺩﺍﺕ ﺑﻪ ﺍﺛﺮ ﺧﻮﺩ ﺭﺍ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﻳﮑﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﺭﺍﻩ‌ﻫﺎ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻋﻀﻮ ﻳﮏ ﮔﺮﻭﻩ ﻧﻮﻳﺴﻨﺪﮔﯽ ﺟﻮﺍﻥ ﺑﺎﺷﻴﺪ، ﺟﺎﻳﯽ ﮐﻪ ﺑﺘﻮﺍﻧﻴﺪ ﺩﺭ ﮐﻨﺎﺭ ﻫﻢ ﺭﺷﺪ ﮐﻨﻴﺪ.

ﺩﺭ ﻓﺮﺁﻳﻨﺪ ﻧﻘﺪ ﭘﻴﺶ ﺍﺯ ﺍﺭﺍﺋﻪ ﺍﺛﺮ، ﭼﻴﺰﯼ ﺟﺰ ﻭﻳﺮﺍﻧﯽ ﺍﻧﺘﻈﺎﺭ ﺷﻤﺎ ﺭﺍ ﻧﻤﯽ‌ﮐﺸﺪ. ﺍﻋﺘﻤﺎﺩ ﺑﻪ ﻧﻔﺲ ﻭ ﻏﺮﻭﺭ ﺷﻤﺎ ﺷﮑﺴﺘﻪ ﺧﻮﺍﻫﺪ ﺷﺪ. ﻭﻟﯽ ﺍﻳﻦ ﺧﻮﺏ ﺍﺳﺖ ﺯﻳﺮﺍ ﺳﺎﺧﺘﺎﺭﻫﺎﯼ ﻓﮑﺮﯼ ﺍﺷﺘﺒﺎﻫﯽ ﮐﻪ ﻣﻨﺠﺮ ﺑﻪ ﻧﻮﺷﺘﻦ ﻳﮏ ﺍﺛﺮ ﺑﺪ ﺷﺪﻩ‌ﺍﻧﺪ، ﺍﺯ ﺑﻴﻦ ﺧﻮﺍﻫﻨﺪ ﺭﻓﺖ.

ﺑﻌﺪ ﺍﺯ ﺑﻪ ﺩﺳﺖ ﺭﺳﻴﺪﻥ ﻧﻘﺪ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﺑﺎ ﺧﻮﺩ ﺍﺛﺮ ﺭﻭﺑﺮﻭ ﻣﯽ‌ﺷﻮﻳﺪ، ﺯﻳﺮﺍ ﮐﺴﯽ ﺍﺯ ﺑﻴﺮﻭﻥ ﺑﺪﻭﻥ ﺩﺍﺷﺘﻦ ﭘﻴﺶ ﻓﺮﺽ‌ﻫﺎﯼ ﺫﻫﻨﯽ ﺷﻤﺎ، ﺁﻥ ﺭﺍ ﻣﻄﺎﻟﻌﻪ ﮐﺮﺩﻩ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ ﻧﻘﺎﻁ ﺿﻌﻒ، ﮐﻤﺒﻮﺩ‌ﻫﺎ ﻭ ﻏﻠﻮ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺧﻮﺑﯽ ﺗﺸﺨﻴﺺ ﻣﯽ‌ﺩﻫﺪ. ﻫﻤﻪ ﻣﺴﺎﺋﻞ ﻣﺘﻔﺮﻗﻪ ﺑﻪ ﮐﻨﺎﺭ ﺧﻮﺍﻫﻨﺪ ﺭﻓﺖ ﺗﺎ ﺷﻤﺎ ﺑﻤﺎﻧﻴﺪ ﻭ ﻳﮏ ﺳﺎﺧﺘﺎﺭ ﻣﻌﻴﻮﺏ ﮐﻪ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﻧﻘﺎﻁ ﻗﻮﺗﯽ ﺩﺍﺭﺩ ﻭﻟﯽ ﺑﺎﻳﺪ ﺍﻳﺮﺍﺩﺍﺕ ﺁﻥ ﺭﺍ ﺭﻓﻊ ﮐﺮﺩ ﺗﺎ ﺑﻪ ﺗﻌﺎﺩﻝ ﺳﺎﺧﺘﺎﺭﯼ ﺑﺮﺳﺪ.

ﺍﮔﺮ ﺍﻳﺪﻩ ﺳﭙﺮﺩﻥ ﮐﺘﺎﺏ ﺑﻪ ﻳﮏ ﻣﻨﺘﻘﺪ ﺟﺪﯼ ﺑﻪ ﻧﻈﺮﺗﺎﻥ ﻧﺎﺭﺍﺣﺖ ﮐﻨﻨﺪﻩ ﺍﺳﺖ، ﻫﻨﻮﺯ ﺑﻪ ﺯﻣﺎﻥ ﻧﻴﺎﺯ ﺩﺍﺭﻳﺪ ﺗﺎ ﺑﺘﻮﺍﻧﻴﺪ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺍﺯ ﮐﺘﺎﺏ ﺟﺪﺍ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﺩﺭﮎ ﺍﻳﻦ ﻣﻔﻬﻮﻡ ﻧﺎﻳﻞ ﺁﻳﻴﺪ ﮐﻪ ﺗﺎ ﺯﻣﺎﻧﯽ ﻣﻴﻞ ﺑﻪ ﺑﻬﺒﻮﺩ ﺩﺍﺭﻳﺪ، ﺍﻳﺮﺍﺩﺍﺕ ﮐﺘﺎﺏ، ﺍﻳﺮﺍﺩﺍﺕ ﺷﺨﺺ ﺷﻤﺎ ﻧﻴﺴﺘﻨﺪ. ﺍﻣﺎ ﭼﻨﺎﻧﭽﻪ ﺍﻳﺮﺍﺩﺍﺕ ﺭﺍ ﻭﺍﺭﺩ ﻧﺪﺍﻧﻴﺪ، ﺁﻥ ﻭﻗﺖ ﺍﺣﺘﻤﺎﻻً ﺍﻳﺮﺍﺩﺍﺕ ﺷﻤﺎ ﻭ ﮐﺘﺎﺏ ﻳﮑﯽ ﻣﯽ‌ﺷﻮﻧﺪ ﻭ ﺩﺭ ﻧﺘﻴﺠﻪ، ﺣﻖ ﻫﻢ ﺩﺍﺭﻳﺪ ﮐﻪ ﻧﺨﻮﺍﻫﻴﺪ ﺁﻥ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﻣﻨﺘﻘﺪ ﺑﺴﭙﺎﺭﻳﺪ.

۴- ﺁﻳﺎ ﺩﺭ ﮊﺍﻧﺮ ﻣﻮﺭﺩ ﻋﻼﻗﻪ‌ﺗﺎﻥ (ﻭ ﺗﺮﺟﻴﺤﺎً ﺩﺭ ﻭﺭﺍﯼ ﺁﻥ) ﺑﻪ ﻣﻄﺎﻟﻌﻪ ﺩﺭ ﺳﻄﺢ ﻋﺎﻟﯽ ﭘﺮﺩﺍﺧﺘﻪ‌ﺍﻳﺪ؟ ﮐﺴﯽ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻓﻮﺗﺒﺎﻟﻴﺴﺖ ﺷﻮﺩ، ﻣﺴﺎﺑﻘﺎﺕ ﺯﻳﺎﺩﯼ ﺭﺍ ﺗﻤﺎﺷﺎ ﻣﯽ‌ﮐﻨﺪ. ﻭﻟﯽ ﺩﺭ ﻣﻴﺪﺍﻥ ﻣﺴﺎﺑﻘﻪ ﺑﻮﺩﻥ ﺷﺮﺍﻳﻂ ﻣﺘﻔﺎﻭﺗﯽ ﺭﺍ ﭘﺪﻳﺪ ﻣﯽ‌ﺁﻭﺭﺩ ﮐﻪ ﺗﺎ ﻗﺒﻞ ﺍﺯ ﺁﻥ، ﺩﺭﮐﺸﺎﻥ ﺑﺮﺍﯼ ﺷﻤﺎ ﻣﻤﮑﻦ ﻧﺒﻮﺩﻩ. ﺩﺭ ﺑﺤﺚ ﻧﻮﻳﺴﻨﺪﮔﯽ، ﺷﻤﺎ ﺑﺎ ﺯﺑﺎﻥ، ﮐﻠﻤﺎﺕ، ﺟﻤﻼﺕ ﻭ... ﺳﺮ ﻭ ﮐﺎﺭ ﺩﺍﺭﻳﺪ. ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ ﻧﻮﺷﺘﻦ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺷﻤﺎﺳﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺎﻳﺪ ﺑﺎ ﺟﻮﺍﻧﺐ ﻣﺨﺘﻠﻒ ﺍﻳﻦ ﺣﺮﻓﻪ ﺁﺷﻨﺎ ﺷﻮﻳﺪ ﮐﻪ ﻳﮑﯽ ﺍﺯ ﺁﻧﻬﺎ، ﮊﺍﻧﺮ ﺍﺳﺖ.

ﮊﺍﻧﺮﻫﺎﯼ ﻧﻮﻳﺴﻨﺪﮔﯽ ﺩﺭ ﺍﺻﻞ ﺳﻨﺖ‌ﻫﺎ ﻳﺎ ﻋﺮﻑ‌ﻫﺎ ﻳﺎ ﺭﻭﺍﻳﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﻪ ﻃﻮﺭ ﻃﺒﻴﻌﯽ ﻭ ﺩﺭ ﮔﺬﺭ ﺯﻣﺎﻥ ﻃﺒﻘﻪ ﺑﻨﺪﯼ ﺧﺎﺻﯽ ﺑﺮﺍﯼ ﺁ‌ﻫﺎ ﺷﮑﻞ ﮔﺮﻓﺘﻪ. ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺩﺭ ﻫﺮ ﮊﺍﻧﺮﯼ ﻗﻠﻢ ﺑﺰﻧﻴﺪ. ﺍﺯ ﺧﻴﺎﻝ‌ﭘﺮﺩﺍﺯﯼ ﺣﻤﺎﺳﯽ ﮔﺮﻓﺘﻪ ﺗﺎ ﺍﺩﺑﻴﺎﺕ ﺩﺍﺳﺘﺎﻧﯽ.

ﻧﮑﺘﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻧﻤﯽ‌ﺗﻮﺍﻧﻴﺪ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ ﺑﺪﻭﻥ ﮔﻮﺵ ﮐﺮﺩﻥ ﺑﻪ ﻣﻮﺳﻴﻘﯽ ﺳﻨﺘﯽ ﻳﺎ ﻧﻮﺍﺧﺘﻦ ﺳﺎﺯ ﺳﻨﺘﯽ، ﻳﮏ ﺁﻫﻨﮕﺴﺎﺯ ﺳﻨﺘﯽ ﺷﻮﻳﺪ. ﻏﻴﺮ ﺍﺯ ﺍﻳﻦ ﺍﺳﺖ؟ ﺑﻪ ﻋﺒﺎﺭﺕ ﺩﻳﮕﺮ، ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﺑﺘﻮﺍﻧﻴﺪ ﺩﺭ ﻫﺮ ﻫﻨﺮﯼ ﺑﻪ ﺍﻭﺝ ﺧﻮﺩ ﺑﺮﺳﻴﺪ، ﺑﺎﻳﺪ ﮐﺎﻣﻼً ﺩﺭ ﺁﻥ ﻏﻮﻃﻪ ﻭﺭ ﺷﻮﻳﺪ.

ﺩﺭ ﮔﺎﻡ ﺍﻭﻝ ﺑﺮﺍﯼ ﺗﺒﺪﻳﻞ ﺷﺪﻥ ﺑﻪ ﻳﮏ ﻫﻨﺮﻣﻨﺪ ﺩﺍﺭﺍﯼ ﻫﺪﻑ، ﺑﺎﻳﺪ ﺳﻨﺖ‌ﻫﺎﯼ ﻳﮏ (ﻭ ﻓﻘﻂ ﻳﮏ) ﮊﺍﻧﺮ ﻣﺸﺨﺺ ﺩﺭ ﺷﺎﺧﻪ ﻫﻨﺮﯼ ﺧﻮﺩﺗﺎﻥ ﺭﺍ ﺑﻴﺎﻣﻮﺯﻳﺪ. ﺳﭙﺲ ﺑﺮﺍﯼ ﭘﻴﻤﻮﺩﻥ ﭘﻠﻪ‌ﻫﺎﯼ ﻋﻈﻤﺖ، ﺑﺎﻳﺪ ﭼﻨﺪ ﮊﺍﻧﺮ ﺩﻳﮕﺮ ﺭﺍ ﻧﻴﺰ ﺁﻣﻮﺧﺘﻪ ﻭ ﮐﻢ ﮐﻢ ﺷﺮﻭﻉ ﺑﻪ ﺩﺭ ﻫﻢ ﺑﺎﻓﺘﻦ ﺁﻧﻬﺎ ﺑﺎ ﻳﮑﺪﻳﮕﺮ ﮐﻨﻴﺪ.

ﺑﻪ ﺩﻭﺭﻩ ﺣﺮﻓﻪ‌ﺍﯼ ﻫﻨﺮﻣﻨﺪﯼ ﻣﺜﻞ ﭘﺎﺑﻠﻮ ﭘﻴﮑﺎﺳﻮ ﻧﮕﺎﻩ ﮐﻨﻴﺪ. ﺍﻭ ﻫﺮ ﭼﻨﺪ ﺳﺎﻝ ﻳﮏ ﺑﺎﺭ، ﺳﺒﮏ‌ﻫﺎﯼ ﺟﺪﻳﺪﯼ ﺭﺍ ﻣﯽ‌ﺁﻣﻮﺧﺖ ﻭ ﺩﺭ ﻫﺮ ﮔﺎﻡ، ﻫﻨﺮ ﺧﻮﺩﺵ ﺭﺍ ﺑﻪ ﻃﺮﺯﯼ ﺑﺎﻭﺭ ﻧﮑﺮﺩﻧﯽ ﻏﻨﺎ ﻣﯽ‌ﺑﺨﺸﻴﺪ. ﭘﺲ، ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﻨﻮﻳﺴﻴﺪ، ﺍﺑﺘﺪﺍ ﺧﻮﺩ ﺭﺍ ﻭﻗﻒ ﺧﻮﺍﻧﺪﻥ ﮐﻨﻴﺪ.

۵- ﺁﻳﺎ ﺑﺴﺘﺮ ﻣﻨﺎﺳﺐ ﺭﺍ ﺩﺍﺭﺍ ﻫﺴﺘﻴﺪ؟ ﺑﻪ ﭼﻪ ﻣﻴﺰﺍﻥ ﺷﻨﺎﺧﺘﻪ ﺷﺪﻩ ﻫﺴﺘﻴﺪ؟ ﺁﻳﺎ ﺳﺎﻳﺖ ﻳﺎ ﻭﺑﻼﮔﯽ ﭘﺮ ﺑﺎﺯﺩﻳﺪ ﻳﺎ ﺣﺴﺎﺏ ﺗﻮﻳﻴﺘﺮ ﻭ ﻓﻴﺴﺒﻮﮎ ﭘﺮ ﻃﺮﻓﺪﺍﺭﯼ ﺩﺍﺭﻳﺪ؟ ﺑﺮ ﭼﻪ ﺍﺳﺎﺱ ﺍﻧﺘﻈﺎﺭ ﺩﺍﺭﻳﺪ ﮐﻪ ﮐﺘﺎﺑﺘﺎﻥ ﭘﺮ ﻓﺮﻭﺵ ﺷﻮﺩ؟

ﺣﻘﻴﻘﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻧﺎﺷﺮﺍﻥ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺴﺘﺮ ﻻﺯﻡ ﺑﺮﺍﯼ ﺗﻮﻟﻴﺪ، ﺗﻮﺯﻳﻊ ﻭ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﮐﺘﺎﺏ ﺑﺎﺷﻨﺪ. ﻭﻟﯽ ﺑﺮﺍﯼ ﻧﻮﻳﺴﻨﺪﻩ‌ﺍﯼ ﮐﻪ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﺑﻪ ﺗﻨﻬﺎﻳﯽ ﺁﻥ ﻫﻢ ﺍﺯ ﻃﺮﻳﻖ ﺑﺴﺘﺮ ﺍﻳﻨﺘﺮﻧﺖ ﮐﺘﺎﺏ ﺧﻮﺩﺵ ﺭﺍ ﺍﺭﺍﺋﻪ ﮐﻨﺪ، ﻗﻀﻴﻪ ﻓﺮﻕ ﺩﺍﺭﺩ. ﺭﺳﺎﻧﻪ‌ﻫﺎ ﻭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺴﺘﺮ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﺭﻭﻧﺪ ﺑﺮ ﺭﻭﯼ ﺁﺛﺎﺭ ﻫﺴﺘﻨﺪ. ﺣﺘﯽ ﺩﺭ ﺍﻳﻦ ﺭﺍﻩ، ﻭﺑﻼﮒ‌ﻫﺎﯼ ﭘﺮ ﻃﺮﻓﺪﺍﺭﯼ ﮐﻪ ﺯﻣﻴﻨﻪ ﮐﺎﺭﯼ ﺁﻧﻬﺎ ﺑﺎ ﮐﺘﺎﺏ ﺷﻤﺎ ﻫﻤﺨﻮﺍﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﮐﺎﺭ ﺑﻴﺎﻳﻨﺪ.

ﻧﮑﺘﻪ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﻳﺎﺑﻴﺪ ﺑﺴﺘﺮ ﻣﻨﺎﺳﺐ ﮐﺎﺭ ﺷﻤﺎ ﭼﻴﺴﺖ ﻭ ﺑﺘﻮﺍﻧﻴﺪ ﺍﺯ ﺁﻥ ﺟﻮﺍﺏ ﺑﮕﻴﺮﻳﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﺍﻳﻦ ﺯﻣﻴﻨﻪ ﺧﻮﺩ ﺭﺍ ﻓﺮﻳﺐ ﻧﺪﻫﻴﺪ. ﻫﺪﻑ ﻧﻬﺎﻳﯽ ﺷﻤﺎ، ﺭﺳﻴﺪﻥ ﺑﻪ ﻣﺨﺎﻃﺐ ﻫﺪﻑ ﺍﺳﺖ.

۶- ﺁﻳﺎ ﺣﺎﺿﺮ ﻫﺴﺘﻴﺪ ﺑﺮ ﺭﻭﯼ ﮐﺘﺎﺏ ﺧﻮﺩ ﺳﺮﻣﺎﻳﻪ ﮔﺬﺍﺭﯼ ﮐﻨﻴﺪ؟ ﻧﺎﺷﺮﺍﻥ ﺑﺮﺍﯼ ﺍﻧﺘﺸﺎﺭ ﮐﺘﺎﺏ‌ﻫﺎ ﺳﺮﻣﺎﻳﻪ ﻧﻘﺪﯼ ﻓﺮﺍﻫﻢ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺑﺎ ﺍﺭﺍﺋﻪ ﺧﺪﻣﺎﺗﯽ ﻧﻈﻴﺮ ﻭﻳﺮﺍﻳﺶ، ﻃﺮﺍﺣﯽ ﻭ ﺑﺎﺯﺍﺭﻳﺎﺑﯽ ﮐﺘﺎﺏ ﺗﻮﺳﻂ ﺍﻓﺮﺍﺩ ﺗﺤﺖ ﻗﺮﺍﺭﺩﺍﺩ ﺧﻮﺩ، ﺁﺛﺎﺭ ﺭﺍ ﺑﻪ ﻣﺮﺣﻠﻪ ﭼﺎﭖ ﻣﯽ‌ﺭﺳﺎﻧﻨﺪ. ﻣﺴﻠﻤﺎً ﺷﻤﺎ ﭼﻨﺎﻧﭽﻪ ﺯﻣﺎﻥ ﻭ ﺩﺍﻧﺶ ﮐﺎﻓﯽ ﺑﺮﺍﯼ ﻭﻳﺮﺍﻳﺶ ﻭ ﺗﻮﺯﻳﻊ ﮐﺘﺎﺑﺘﺎﻥ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﻣﺠﺒﻮﺭ ﺑﻪ ﺍﺳﺘﺨﺪﺍﻡ ﺍﻓﺮﺍﺩﯼ ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺎﺭ ﺧﻮﺍﻫﻴﺪ ﺷﺪ. ﺁﻳﺎ ﺳﺮﻣﺎﻳﻪ ﻻﺯﻡ ﺑﺮﺍﯼ ﭼﻨﻴﻦ ﮐﺎﺭﯼ ﺭﺍ ﺩﺍﺭﻳﺪ ﻭ ﺣﺎﺿﺮ ﻫﺴﺘﻴﺪ ﺁﻥ ﺳﺮﻣﺎﻳﻪ ﺭﺍ ﺭﻭﯼ ﺍﺛﺮ ﺧﻮﺩ ﺭﻳﺴﮏ ﮐﻨﻴﺪ؟

ﺑﺪﻭﻥ ﺳﺮﻣﺎﻳﻪ ﮔﺬﺍﺭﯼ ﺣﺮﻓﻪ‌ﺍﯼ، ﮐﺘﺎﺏ ﺷﻤﺎ ﺍﺣﺘﻤﺎﻻً ﻳﮏ ﺍﺛﺮ ﺣﺮﻓﻪ‌ﺍﯼ ﻧﺨﻮﺍﻫﺪ ﺷﺪ. ﺍﻟﺒﺘﻪ ﮐﺘﺎﺏ‌ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﺎ ﻭﺟﻮﺩ ﺍﺭﺍﺋﻪ ﻧﻴﻤﻪ ﺣﺮﻓﻪ‌ﺍﯼ ﺗﺒﺪﻳﻞ ﺑﻪ ﭘﺮ ﻣﺨﺎﻃﺐ‌ﻫﺎﯼ ﭘﻴﺸﺨﻮﺍﻥ ﺍﻳﻨﺘﺮﻧﺖ ﺷﺪﻩ‌ﺍﻧﺪ ﻭﻟﯽ ﺳﺆﺍﻝ ﺍﺻﻠﯽ ﺭﺍ ﻓﺮﺍﻣﻮﺵ ﻧﮑﻨﻴﺪ؛ ﺍﮔﺮ ﺣﺎﺿﺮ ﻧﻴﺴﺘﻴﺪ ﺭﻭﯼ ﮐﺘﺎﺏ ﺧﻮﺩﺗﺎﻥ ﺳﺮﻣﺎﻳﻪ ﮔﺬﺍﺭﯼ ﮐﻨﻴﺪ ﭘﺲ ﭼﺮﺍ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺁﻥ ﺭﺍ ﺑﻪ ﺩﻧﻴﺎ ﻋﺮﺿﻪ ﮐﻨﻴﺪ؟
