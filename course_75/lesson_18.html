ﺣﺘﯽ ﺷﺎﻳﺪ ﻧﻴﺎﺯﯼ ﺍﺳﺎﺳﯽ ﺑﺮﺍﯼ ﮐﺴﺐ ﺍﻃﻼﻋﺎﺕ ﺑﻴﺸﺘﺮ ﺩﺭ ﺯﻣﻴﻨﻪ ﺗﻮﺳﻌﻪ ﻭﺏ ﻧﻴﺰ ﺩﺭ ﺷﻤﺎ ﺍﻳﺠﺎﺩ ﺷﺪﻩ ﺑﺎﺷﺪ، ﻭﻟﯽ ﺁﻥ ﺩﻳﮕﺮ ﺍﺯ ﺩﺍﻣﻨﻪ ﺩﺭﻭﺱ ﺩﻭﺭﻩ ﻓﻌﻠﯽ ﻣﺎ ﺧﺎﺭﺝ ﺍﺳﺖ ﻭ ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﻴﻢ ﺍﺑﺘﺪﺍ ﺍﻳﻦ ﺩﻭﺭﻩ ﺭﺍ ﺑﺎ ﻣﻮﻓﻘﻴﺖ ﭘﺸﺖ ﺳﺮ ﺑﮕﺬﺍﺭﻳﺪ، ﻭ ﺳﭙﺲ ﺩﺭﺑﺎﺭﻩ ﻳﺎﺩﮔﻴﺮﯼ ﭼﻨﻴﻦ ﻣﻬﺎﺭﺕ‌ﻫﺎﻳﯽ ﺍﺯ ﻃﺮﻳﻖ ﺩﺭﺳﻨﺎﻣﻪ ﺗﺼﻤﻴﻢ ﮔﻴﺮﯼ ﻧﻤﺎﻳﻴﺪ.

 

ﻣﺼﺮﻑ ﻣﺤﺘﻮﺍﯼ ﺩﻳﺠﻴﺘﺎﻝ
ﺍﺯ ﺑﺤﺚ ﺍﻧﻮﺍﻉ ﻓﺮﻣﺖ‌ﻫﺎ ﺑﺮﺍﯼ ﻧﺸﺮ ﻣﺤﺘﻮﺍﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﮐﻪ ﺑﮕﺬﺭﻳﻢ، ﺑﻪ ﺩﺳﺘﮕﺎﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﻣﯽ‌ﺭﺳﻴﻢ ﮐﻪ ﺍﻳﻦ ﻣﺘﻮﻥ ﺩﻳﺠﻴﺘﺎﻝ ﺭﻭﯼ ﺁﻧﻬﺎ ﻣﻄﺎﻟﻌﻪ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﻪ ﺍﺣﺘﻤﺎﻝ ﺯﻳﺎﺩ، ﺗﺒﻠﺖ‌ﻫﺎ ﻣﺤﺒﻮﺏ‌ﺗﺮﻳﻦ ﻭ ﭘﺮ ﺍﺳﺘﻔﺎﺩﻩ‌ﺗﺮﻳﻦ ﺩﺳﺘﮕﺎﻩ‌ﻫﺎ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﻫﺴﺘﻨﺪ.

ﺍﺯ‌‌ ﻫﻤﺎﻥ ﺳﺎﻝ ۲۰۱۰ ﮐﻪ ﺍﺳﺘﻴﻮ ﺟﺎﺑﺰ (ﻣﺪﻳﺮ ﻋﺎﻣﻞ ﺳﺎﺑﻖ ﺍﭘﻞ ﮐﻪ ﺩﺭ ﺳﺎﻝ ۲۰۱۱ ﺩﺭ ﮔﺬﺷﺖ) ﺍﻭﻟﻴﻦ ﺁﻳﭙﺪ ﺭﺍ ﻣﻌﺮﻓﯽ ﻭ ﺗﻌﺮﻳﻔﯽ ﻣﺪﺭﻥ ﺍﺯ ﺗﺒﻠﺖ ﺑﻪ ﺟﻬﺎﻧﻴﺎﻥ ﺍﺭﺍﺋﻪ ﮐﺮﺩ، ﺍﻳﻦ ﺩﺳﺘﮕﺎﻩ‌ﻫﺎ ﺗﺒﺪﻳﻞ ﺑﻪ ﺍﺑﺰﺍﺭﻫﺎﻳﯽ ﭘﺮ ﺍﺳﺘﻔﺎﺩﻩ ﺷﺪﻧﺪ. ﺩﺭ ﺍﺻﻞ، ﻃﺮﺍﺣﺎﻥ ﺗﺒﻠﺖ‌ﻫﺎﯼ ﻣﺪﺭﻥ ﺍﻳﻦ ﺍﻳﺪﻩ ﺭﺍ ﺩﺭ ﺳﺮ ﺩﺍﺭﻧﺪ ﮐﻪ ﮐﺎﺭ ﺑﺎ ﺁﻧﻬﺎ، ﺳﺎﺩﻩ ﻭ ﻫﻤﻪ ﻓﻬﻢ ﺑﺎﺷﺪ.

ﺳﺎﺩﮔﯽ ﺍﺳﺘﻔﺎﺩﻩ، ﺩﺭ ﮐﻨﺎﺭ ﺑﺎﺗﺮﯼ‌ﻫﺎﻳﯽ ﺑﺎ ﻋﻤﺮ ﻣﻨﺎﺳﺐ، ﻗﺎﺑﻠﻴﺖ ﺍﺗﺼﺎﻝ ﺑﻪ ﺍﻳﻨﺘﺮﻧﺖ ﻭ ﻣﻬﻢ‌ﺗﺮ ﺍﺯ ﻫﻤﻪ ﻗﺎﺑﻞ ﺣﻤﻞ ﺑﻮﺩﻥ ﺗﺒﻠﺖ‌ﻫﺎ ﺑﻪ ﮐﺎﺭﺑﺮﺍﻥ ﺍﻳﻦ ﺍﻣﮑﺎﻥ ﺭﺍ ﻣﯽ‌ﺩﻫﺪ ﺗﺎ ﺑﺎ ﺍﻧﻮﺍﻉ ﻣﺤﺘﻮﺍﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺑﻪ ﺭﺍﺣﺘﯽ ﮐﺎﺭ ﮐﻨﻨﺪ ﻭ ﺁ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺷﮑﻠﯽ ﮐﻪ ﻣﯽ‌ﭘﺴﻨﺪﻧﺪ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﺩﻫﻨﺪ.

ﺍﻟﺒﺘﻪ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﮐﺘﺎﺏ‌ﻫﺎ ﻭ ﺳﺎﻳﺮ ﻧﺸﺮﻳﺎﺕ ﻧﻴﺎﺯﻣﻨﺪ ﻧﺼﺐ ﺍﭘﻠﻴﮑﻴﺸﻦ‌ﻫﺎﯼ ﻣﺨﺼﻮﺹ ﺑﺮ ﺭﻭﯼ ﺗﺒﻠﺖ‌ﻫﺎ ﻭ ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﺍﻳﻨﺘﺮﻧﺖ ﺑﺮﺍﯼ ﺩﺭﻳﺎﻓﺖ ﮐﺘﺎﺏ‌ﻫﺎ ﺍﺳﺖ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﭘﻠﻴﮑﻴﺸﻦ‌ﻫﺎﯼ «‌ﺍﯼ ﺭﻳﺪﺭ»، ﺑﺮﺍﯼ ﺧﻮﺍﻧﺪﻥ ﻓﺮﻣﺖ‌ﻫﺎﯼ ﺧﺎﺹ ﺳﺎﺧﺘﻪ ﺷﺪﻩ‌ﺍﻧﺪ، ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺑﺮﺧﯽ ﺍﺯ ﺁﻧﻬﺎ ﻣﺜﻞ iBooks ﺍﭘﻞ ﮐﻪ ﺑﺮ ﺭﻭﯼ ﺁﻳﭙﺪ ﻭ ﺁﻳﻔﻮﻥ ﻗﺎﺑﻞ ﻧﺼﺐ ﺍﺳﺖ، ﺍﻣﮑﺎﻥ ﻣﻄﺎﻟﻌﻪ ﺍﻧﻮﺍﻉ ﻓﺮﻣﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺩﻳﺠﻴﺘﺎﻝ ﺭﺍ ﻓﺮﺍﻫﻢ ﻣﯽ‌ﮐﻨﻨﺪ.

ﺟﺎﻟﺐ ﺍﻳﻨﺠﺎ ﺍﺳﺖ ﮐﻪ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺷﺨﺼﯽ (ﻟﭗ ﺗﺎﭖ ﻭ ﺩﺳﮏ ﺗﺎﭖ) ﻧﻪ ﺗﻨﻬﺎ ﺑﺮﺍﯼ ﺳﺎﺧﺖ ﺍﻧﻮﺍﻉ ﻣﺤﺘﻮﺍﻫﺎﯼ ﻣﺘﻨﯽ، ﺑﻠﮑﻪ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﻳﺎ ﻣﺸﺎﻫﺪﻩ ﺁﻧﻬﺎ ﻧﻴﺰ ﻣﻮﺭﺩ ﺍﺳﺘﻔﺎﺩﻩ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﮐﺎﻣﭙﻴﻮﺗﺮ‌ﻫﺎ ﻧﻴﺰ ﺑﺮﺍﯼ ﺍﻳﻦ ﻣﻮﺭﺩ، ﺑﺮﻧﺎﻣﻪ‌ﻫﺎﯼ ﻣﺨﺼﻮﺻﯽ ﺭﺍ ﻣﯽ‌ﻃﻠﺒﻨﺪ ﮐﻪ ﺑﺮﺧﯽ ﺍﺯ ﺁﻧﻬﺎ ﻧﻪ ﻓﻘﻂ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ، ﺑﻠﮑﻪ ﺑﺮﺍﯼ ﻭﻳﺮﺍﻳﺶ ﻣﺘﻮﻥ ﻫﻢ ﻗﺎﺑﻞ ﺍﺳﺘﻔﺎﺩﻩ‌ﺍﻧﺪ.

ﺍﻟﺒﺘﻪ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﯼ ﺷﺨﺼﯽ ﺑﻪ ﺍﻧﺪﺍﺯﻩ ﺗﺒﻠﺖ‌ﻫﺎ ﻗﺎﺑﻠﻴﺖ ﺣﻤﻞ ﺁﺳﺎﻥ ﺭﺍ ﻧﺪﺍﺭﻧﺪ، ﻭﻟﯽ ﺩﺭ ﻋﻮﺽ ﮐﺎﺭﺑﺮﺩ ﺍﺻﻠﯽ ﺁﻧﻬﺎ ﻣﺸﺎﻫﺪﻩ ﻭ ﻣﻄﺎﻟﻌﻪ ﺍﺳﻨﺎﺩ ﺩﻳﺠﻴﺘﺎﻝ ﻣﺜﻞ، ﮔﺰﺍﺭﺵ‌ﻫﺎﯼ ﺗﺠﺎﺭﯼ، ﺩﻓﺘﺮﭼﻪ ﺭﺍﻫﻨﻤﺎﯼ ﻣﺤﺼﻮﻻﺕ ﻣﺨﺘﻠﻒ ﻭ ﻣﻮﺍﺭﺩﯼ ﺍﺯ ﺍﻳﻦ ﺩﺳﺖ ﺍﺳﺖ.

ﺩﺭ ﮐﻨﺎﺭ ﺍﻳﻦ ﻣﻮﺍﺭﺩ، ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪ ﻫﻢ ﺑﻪ ﺧﺎﻃﺮ ﺍﻳﻨﮑﻪ ﮐﻮﭼﮏ‌ﺗﺮ ﺑﻮﺩﻩ ﻭ ﻃﺒﻴﻌﺘﺎً ﺑﻴﺸﺘﺮ ﺍﺯ ﺳﺎﻳﺮ ﻭﺳﺎﻳﻞ ﺩﻳﺠﻴﺘﺎﻝ ﻫﻤﺮﺍﻩ ﻣﺮﺩﻡ‌ﺍﻧﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﺍﺳﺘﻔﺎﺩﻩ ﺷﻮﻧﺪ. ﺍﻟﺒﺘﻪ ﻧﻤﺎﻳﺸﮕﺮ ﮐﻮﭼﮏ‌ﺗﺮ ﺁﻧﻬﺎ ﺩﺭ ﻗﻴﺎﺱ ﺑﺎ ﺗﺒﻠﺖ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﻓﺮﻣﺖ‌ﻫﺎﯼ ﻣﺘﻨﯽ ﮐﻪ ﺍﻧﺪﺍﺯﻩ ﺛﺎﺑﺘﯽ ﺩﺍﺭﻧﺪ (ﺍﺯ ﺟﻤﻠﻪ ﭘﯽ ﺩﯼ ﺍﻑ) ﺩﺭﺩﺳﺮ ﺳﺎﺯ ﺑﺎﺷﺪ. ﺑﺮﺍﯼ ﻣﺜﺎﻝ ﭼﻨﺎﻧﭽﻪ ﺍﻧﺪﺍﺯﻩ ﻓﻮﻧﺖ، ﻃﻮﻝ ﻭ ﻋﺮﺽ، ﻭ ﻧﺴﺒﺖ ﻃﻮﻝ ﺑﻪ ﻋﺮﺽ ﻫﺮ ﺻﻔﺤﻪ ﺍﺯ ﻣﺘﻦ ﺭﺍ ﻃﻮﺭﯼ ﺗﻨﻈﻴﻢ ﻧﻤﺎﻳﻴﺪ ﮐﻪ ﺑﺮﺍﯼ ﻣﺸﺎﻫﺪﻩ ﺩﺭ ﺗﺒﻠﺖ‌ﻫﺎﯼ ۱۰ ﺍﻳﻨﭽﯽ ﻭ ﮐﺎﻣﭙﻴﻮﺗﺮﻫﺎﻳﯽ ﺑﺎ ﻧﻤﺎﻳﺸﮕﺮ ﺑﺰﺭﮒ‌ﺗﺮ ﻣﻨﺎﺳﺐ ﺑﺎﺷﺪ ﻭﻟﯽ ﺁﻥ ﺳﻨﺪ ﺩﺍﺭﺍﯼ ﺍﻧﺪﺍﺯﻩ ﻭ ﻧﺴﺒﺖ‌ﻫﺎﯼ ﺛﺎﺑﺖ ﺑﺎﺷﺪ، ﻣﻄﺎﻟﻌﻪ ﺁﻥ ﺩﺭ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪ ﮐﻪ ﻣﻌﻤﻮﻻ ﻧﻤﺎﻳﺸﮕﺮﻫﺎﻳﯽ ﮐﻮﭼﮏ‌ﺗﺮ ﺍﺯ ۵ ﺍﻳﻨﭻ ﺩﺍﺭﻧﺪ ﺳﺨﺖ ﺧﻮﺍﻫﺪ ﺑﻮﺩ. ﺍﻟﺒﺘﻪ ﮐﺎﺭﺑﺮ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﺮﺍﯼ ﻣﻄﺎﻟﻌﻪ ﭼﻨﻴﻦ ﻣﺘﻦ‌ﻫﺎﻳﯽ ﺗﻠﻔﻦ ﺭﺍ ﺑﻪ ﭼﺸﻢ ﺧﻮﺩ ﻧﺰﺩﻳﮏ ﮐﻨﺪ ﻭ ﻳﺎ ﺍﺯ ﻗﺎﺑﻠﻴﺖ ﺯﻭﻡ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﺎﻳﺪ. ﻭﻟﯽ ﺗﺠﺮﺑﻪ ﺛﺎﺑﺖ ﮐﺮﺩﻩ ﮐﻪ ﻫﻴﭻ ﻳﮏ ﺍﺯ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﺭﺍﻫﮑﺎﺭ ﻗﺎﺑﻞ ﻗﺒﻮﻟﯽ ﻧﻴﺴﺘﻨﺪ ﻭ ﺩﺷﻮﺍﺭﯼ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺭﺍ ﺑﻪ ﻭﺟﻮﺩ ﺁﻭﺭﺩﻩ، ﺩﺭ ‌‌ﻧﻬﺎﻳﺖ ﺑﺎﻋﺚ ﺩﻟﺰﺩﮔﯽ ﻓﺮﺩ ﻭ ﺗﻮﻗﻒ ﻣﻄﺎﻟﻌﻪ ﻣﺘﻦ‌ﻫﺎﯼ ﺑﻠﻨﺪ ﻣﺜﻞ ﮐﺘﺎﺏ‌ﻫﺎ ﻣﯽ‌ﺷﻮﻧﺪ.

ﻟﺬﺍ ﺑﺎﻳﺴﺘﯽ ﺭﻭﺵ‌ﻫﺎﯼ ﺻﺤﻴﺤﯽ ﺑﺮﺍﯼ ﻣﺨﺎﻃﺒﺎﻥ ﭘﻴﺎﺩﻩ ﺳﺎﺯﯼ ﺷﻮﺩ. ﻗﻄﻌﺎً ﭘﻴﺎﺩﻩ ﺳﺎﺯﯼ ﺍﻳﻦ ﺭﻭﺵ‌ﻫﺎ ﺳﺨﺘﯽ‌ﻫﺎﯼ ﺧﺎﺹ ﺧﻮﺩ ﺭﺍ ﺩﺍﺭﺩ، ﻭﻟﯽ ﻳﮏ ﻣﺆﻟﻒ ﻳﺎ ﻣﺘﺮﺟﻢ ﻣﯽ‌ﺑﺎﻳﺴﺘﯽ ﭘﻴﺶ ﺍﺯ ﻫﺮ ﭼﻴﺰ، ﺍﺣﺘﺮﺍﻡ ﻣﺨﺎﻃﺐ ﺭﺍ ﻧﺴﺒﺖ ﺑﻪ ﺧﻮﺩ ﺑﺮ ﺍﻧﮕﻴﺰﺩ ﻭ ﺍﻳﻦ ﻣﻤﮑﻦ ﻧﻴﺴﺖ، ﻣﮕﺮ ﺑﺎ ﺍﺣﺘﺮﺍﻡ ﮔﺬﺍﺷﺘﻦ ﺑﻪ ﻣﺨﺎﻃﺐ.

ﺑﻨﺎﺑﺮﺍﻳﻦ ﺩﺭ ﭘﺎﺳﺦ ﺑﻪ ﻧﻴﺎﺯ ﮐﺎﺭﺑﺮﺍﻥ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪ ﺑﺮﺍﯼ ﺍﺭﺍﺋﻪ ﻣﺘﻮﻧﯽ ﮐﻪ ﺭﺍﺣﺖ ﺑﺮ ﺭﻭﯼ ﺍﻳﻦ ﻧﻤﺎﻳﺸﮕﺮ‌ﻫﺎﯼ ﮐﻮﭼﮏ ﻗﺎﺑﻞ ﻣﻄﺎﻟﻌﻪ ﺑﺎﺷﻨﺪ، ﻣﯽ‌ﺗﻮﺍﻥ ﺩﻭ ﺭﺍﻫﮑﺎﺭ ﺭﺍ ﺑﺮﮔﺰﻳﺪ:

  • ﺗﻨﻈﻴﻢ ﻧﺴﺨﻪ‌ﺍﯼ ﺍﺧﺘﺼﺎﺻﯽ ﺍﺯ ﮐﺘﺎﺏ ﺩﺭ ﻗﺎﻟﺐ ﻓﺮﻣﺘﯽ ﺛﺎﺑﺖ (ﻣﺜﻞ ﭘﯽ ﺩﯼ ﺍﻑ) ﮐﻪ ﺩﺍﺭﺍﯼ ﺍﺑﻌﺎﺩ (ﻓﻮﻧﺖ، ﻃﻮﻝ، ﻋﺮﺽ) ﻭ ﻧﺴﺒﺖ‌ﻫﺎﯼ ﻣﻨﺎﺳﺐ ﺑﺮﺍﯼ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪ ﺑﺎﺷﺪ. ﺍﻳﻦ ﺭﺍﻫﮑﺎﺭ، ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﻣﺸﮑﻞ ﺭﺍ ﺑﺮ ﻃﺮﻑ ﻣﯽ‌ﮐﻨﺪ، ﻭﻟﯽ ﺑﻬﺘﺮﻳﻦ ﺭﻭﺵ ﺑﻪ ﺷﻤﺎﺭ ﻧﻤﯽ‌ﺭﻭﺩ.
  • ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻓﺮﻣﺖ‌ﻫﺎﻳﯽ ﻣﺜﻞ ‌ﺍﯼ ﭘﺎﺏ ﮐﻪ ﻗﺎﺑﻠﻴﺖ ﺗﻄﺒﻴﻖ ﺧﻮﺩﮐﺎﺭ ﻣﺘﻦ ﺑﺎ ﻧﻤﺎﻳﺸﮕﺮﻫﺎﻳﯽ ﺩﺭ ﺍﻧﺪﺍﺯﻩ‌ﻫﺎ ﻭ ﻧﺴﺒﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﻭ ﻧﻴﺰ ﻗﺎﺑﻠﻴﺖ ﺗﻐﻴﻴﺮ ﺩﻟﺨﻮﺍﻩ ﺍﻧﺪﺍﺯﻩ ﻓﻮﻧﺖ ﻣﺘﻦ ﮐﺘﺎﺏ ﺭﺍ ﺩﺍﺭﻧﺪ.

ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺍﺯ ﺍﻳﻦ ﻧﮑﺘﻪ ﮔﺬﺷﺖ ﮐﻪ ﻣﻬﻢ‌ﺗﺮﻳﻦ ﻣﺰﻳﺖ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪ، ﻫﻤﻴﺸﻪ ﻫﻤﺮﺍﻩ ﺑﻮﺩﻥ ﺁﻧﻬﺎ ﺑﺎ ﻣﺨﺎﻃﺐ ﺍﺳﺖ. ﺍﺯ ﺍﻳﻦ ﺭﻭ ﭼﻨﺎﻧﭽﻪ ﻗﺼﺪ ﺍﻧﺘﺸﺎﺭ ﮐﺘﺎﺑﯽ ﺑﺮﺍﯼ ﻣﺨﺎﻃﺒﻴﻦ ﺍﻣﺮﻭﺯﯼ ﺭﺍ ﺩﺍﺭﻳﺪ، ﭘﻴﺸﻨﻬﺎﺩ ﻣﯽ‌ﮐﻨﻢ ﺣﺘﻤﺎ ﻧﺴﺨﻪ‌ﺍﯼ ﻣﻨﺎﺳﺐ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪ ﺭﺍ ﻧﻴﺰ ﺩﺭ ﻧﻈﺮ ﺑﮕﻴﺮﻳﺪ.

ﭼﻪ، ﺭﺍﻫﮑﺎﺭ ﺷﻤﺎ ﺍﺭﺍﺋﻪ ﻳﮏ ﺍﭘﻠﻴﮑﻴﺸﻦ ﺑﺮﺍﯼ ﻧﻤﺎﻳﺸﮕﺮ ﮐﻮﭼﮏ‌ﺗﺮ ﺑﺎﺷﺪ ﻭ ﭼﻪ، ﺍﺭﺍﺋﻪ ﻓﺮﻣﺖ EPUB، ﻣﻄﻤﺌﻦ ﺑﺎﺷﻴﺪ ﮐﺎﺭﺑﺮﺍﻥ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪ ﺑﻪ ﮔﺮﻣﯽ ﺍﺯ ﺁﻥ ﺍﺳﺘﻘﺒﺎﻝ ﺧﻮﺍﻫﻨﺪ ﮐﺮﺩ. ﺁﻧﻬﺎ ﻋﻼﻗﻤﻨﺪ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﺤﺘﻮﺍ ﺭﺍ ﺑﺴﺘﻪ ﺑﻪ ﺣﺎﻟﺖ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺯﻧﺪﮔﯽ (ﺩﺭ ﺧﺎﻧﻪ، ﺩﺭ ﻣﺴﻴﺮ ﮐﺎﺭ ﻳﺎ ﺗﺤﺼﻴﻞ، ﺩﺭ ﺳﺎﺣﻞ ﺩﺭﻳﺎ، ﺩﺭ ﮐﺎﻓﯽ ﺷﺎﭖ ﻳﺎ...) ﺑﺮ ﺭﻭﯼ ﺩﺳﺘﮕﺎﻩ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ ﺧﻮﺩ ﻣﻄﺎﻟﻌﻪ ﮐﻨﻨﺪ. ﺗﺼﻤﻴﻢ ﺩﺭﺳﺖ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﻧﻴﺰ ﺣﻖ ﺍﻧﺘﺨﺎﺏ ﺁﻧﻬﺎ ﺭﺍ ﻣﺤﺘﺮﻡ ﺑﺸﻤﺎﺭﻳﺪ.

 

ﭼﺮﺍ ﺑﻪ ﺳﻤﺖ ﻧﺸﺮ ﺩﻳﺠﻴﺘﺎﻝ ﺑﺮﻭﻳﻢ؟
ﺩﺭ ﺣﺎﻟﯽ ﮐﻪ ﻫﻨﻮﺯ ﻧﺸﺮﻳﺎﺕ ﭼﺎﭘﯽ ﺯﻳﺎﺩﯼ ﻣﺸﺎﻫﺪﻩ ﻣﯽ‌ﺷﻮﻧﺪ، ﺟﺮﻳﺎﻥ ﻏﺎﻟﺐ ﺍﻣﺮﻭﺯﯼ ﺍﺭﺍﺋﻪ ﻧﺴﺨﻪ ﺩﻳﺠﻴﺘﺎﻝ ﻧﺸﺮﻳﺎﺕ ﻭ ﮐﺘﺎﺏ‌ﻫﺎ ﺩﺭ ﮐﻨﺎﺭ ﻧﺴﺨﻪ ﭼﺎﭘﯽ ﺍﺳﺖ. ﭼﺮﺍ؟ ﺩﺭ ﻭﻫﻠﻪ ﺍﻭﻝ ﺑﻪ ﺍﻳﻦ ﺧﺎﻃﺮ ﮐﻪ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﺩﻥ ﻣﺤﺘﻮﺍﻫﺎﯼ ﻣﺨﺘﻠﻒ ﻫﺮﮔﺰ ﺁﺳﺎﻥ ﻧﺒﻮﺩﻩ ﻭﻟﯽ ﺍﻣﺮﻭﺯﻩ ﺑﺎ ﻭﺟﻮﺩ ﻓﺮﻭﺷﮕﺎﻩ‌ﻫﺎﯼ ﺁﻧﻼﻳﻦ ﻣﺨﺘﻠﻒ ﻣﺎﻧﻨﺪ ﺁﻳﺘﻮﻧﺰ ﺍﭘﻞ، ﮐﻴﻨﺪﻝ ﺍﺳﺘﻮﺭ ﺁﻣﺎﺯﻭﻥ، ﺑﻮﮎ ﺍﺳﺘﻮﺭ ﺑﺎﺭﻧﺰ-ﺍﻧﺪ-ﻧﻮﺑﻠﺰ، ﻓﺮﻭﺷﮕﺎﻩ ﮔﻮﮔﻞ ﭘﻠﯽ ﻣﺘﻌﻠﻖ ﺑﻪ ﺳﻴﺴﺘﻢ ﻋﺎﻣﻞ ﺍﻧﺪﺭﻭﻳﺪ، ﻭ ﻓﺮﻭﺷﮕﺎﻩ‌ﻫﺎﯼ ﻣﻨﻄﻘﻪ‌ﺍﯼ، ﮐﺸﻮﺭﯼ ﻭ ﻣﺤﻠﯽ، ﺣﻘﻴﻘﺘﺎً ﺍﻧﻮﺍﻉ ﻣﺤﺘﻮﺍ ﺗﻨﻬﺎ ﻳﮏ ﻟﻤﺲ ﺍﻧﮕﺸﺖ ﺑﺎ ﻣﺨﺎﻃﺒﺎﻥ ﻓﺎﺻﻠﻪ ﺩﺍﺭﻧﺪ. ﻧﺎﺷﺮﺍﻥ ﻫﻤﺰﻣﺎﻥ ﺑﺎ ﺍﺭﺍﺋﻪ ﻧﺴﺨﻪ ﭼﺎﭘﯽ ﮐﺘﺎﺏ‌ﻫﺎ ﻳﺎ ﺣﺘﯽ ﺯﻭﺩ‌ﺗﺮ، ﻧﺴﺨﻪ‌ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺭﺍ ﺍﺭﺍﺋﻪ ﻣﯽ‌ﮐﻨﻨﺪ. ﻣﺠﻼﺕ ﻭ ﻧﺸﺮﻳﺎﺕ ﺩﻳﮕﺮ ﻧﻴﺰ ﺑﻪ ﺳﻬﻮﻟﺖ ﻭ ﺑﻪ ﻣﺤﺾ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﺩﺭ ﻓﺮﻭﺷﮕﺎﻩ، ﺑﺮﺍﯼ ﻣﺨﺎﻃﺒﻴﻦ ﭘﻴﺎﻡ ﻳﺎﺩﺁﻭﺭﯼ ﻣﯽ‌ﻓﺮﺳﺘﻨﺪ ﻭ ﺑﻪ ﺍﻳﻦ ﺷﮑﻞ ﻣﯽ‌ﺗﻮﺍﻥ ﻧﺴﺨﻪ‌ﻫﺎﯼ ﺟﺪﻳﺪ ﺭﺍ ﺩﺭﻳﺎﻓﺖ ﮐﺮﺩ.

ﺩﻻﻳﻞ ﺩﻳﮕﺮﯼ ﻧﻴﺰ ﺑﺮﺍﯼ ﺑﻪ ﺭﺍﻩ ﺍﻓﺘﺎﺩﻥ ﻣﻮﺝ ﻧﺸﺮ ﺩﻳﺠﻴﺘﺎﻝ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺗﻌﺎﻣﻠﯽ ﺑﺎﺷﻨﺪ ﻭ ﺷﻤﺎ ﺑﺎ ﻟﻤﺲ ﻟﻴﻨﮏ‌ﻫﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﻪ ﻣﻨﺎﺑﻊ ﺧﺎﺭﺝ ﺍﺯ ﮐﺘﺎﺏ، ﻣﺜﻞ ﻭﻳﮑﻴﭙﺪﻳﺎ ﻣﺘﺼﻞ ﺷﻮﻳﺪ ﻭ ﺍﻃﻼﻋﺎﺕ ﭘﺎﻭﺭﻗﯽ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺩﺳﺖ ﺁﻭﺭﻳﺪ. ﻫﻤﭽﻨﻴﻦ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺨﺶ‌ﻫﺎﯼ ﻣﻔﻴﺪ ﮐﺘﺎﺏ ﻳﺎ ﻧﺸﺮﻳﻪ‌ﺍﯼ ﺭﺍ ﮐﻪ ﭘﺴﻨﺪﻳﺪﻩ‌ﺍﻳﺪ، ﺩﺭ ﺷﺒﮑﻪ‌ﻫﺎﯼ ﺍﺟﺘﻤﺎﻋﯽ ﺑﺎ ﺩﻳﮕﺮﺍﻥ ﺑﻪ ﺍﺷﺘﺮﺍﮎ ﺑﮕﺬﺍﺭﻳﺪ. ﺣﺘﯽ ﺑﺮﺧﯽ ﻣﺠﻼﺕ ﺩﻳﺠﻴﺘﺎﻝ ﺍﻣﺮﻭﺯﯼ ﺩﺭ ﺩﺍﺧﻞ ﺧﻮﺩ ﻣﺤﺘﻮﺍ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺷﮑﻞ ﻓﻴﻠﻢ ﻳﺎ ﺍﻧﻴﻤﻴﺸﻦ ﻣﻴﺰﺑﺎﻧﯽ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺑﻪ ﺍﻳﻦ ﺷﮑﻞ ﺗﺠﺮﺑﻪ ﺷﻤﺎ ﻋﻼﻭﻩ ﺑﺮ ﻣﻄﺎﻟﻌﻪ ﻭ ﺗﻤﺎﺷﺎﯼ ﻋﮑﺲ‌ﻫﺎ، ﺑﻪ ﺗﺼﺎﻭﻳﺮ ﻣﺘﺤﺮﮎ ﻧﻴﺰ ﮔﺴﺘﺮﺵ ﻳﺎﻓﺘﻪ ﻭ ﻟﺬﺕ ﺑﻴﺸﺘﺮﯼ ﻣﯽ‌ﺑﺮﻳﺪ. ﺗﻤﺎﻡ ﺍﻳﻦ‌ﻫﺎ ﺩﻻﻳﻠﯽ ﻏﻴﺮ ﻗﺎﺑﻞ ﭼﺸﻢ ﭘﻮﺷﯽ ﺑﺮﺍﯼ ﺣﺮﮐﺖ ﺑﻪ ﺳﻤﺖ ﺍﻧﻮﺍﻉ ﻧﺸﺮﻳﻪ‌ﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻟﯽ ﻫﺴﺘﻨﺪ؛ ﭼﻪ ﮐﺘﺎﺏ ﻭ ﭼﻪ ﺳﺎﻳﺮ ﻣﺤﺘﻮﺍ‌ﻫﺎ.

ﺩﺭ ﮐﻨﺎﺭ ﺍﻳﻦ ﻣﻮﺍﺭﺩ، ﺭﺍﺣﺘﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺗﺒﻠﺖ ﻳﺎ ﮐﺘﺎﺑﺨﻮﺍﻥ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﺭﺍ ﻧﻴﺰ ﻣﻮﺭﺩ ﺗﺄﮐﻴﺪ ﻣﺠﺪﺩ ﻗﺮﺍﺭ ﻣﯽ‌ﺩﻫﻴﻢ. ﮐﻪ ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺧﺮﻳﺪ ﺁﻧﻬﺎ، ﮐﻞ ﮐﺘﺎﺑﺨﺎﻧﻪ ﺧﻮﺩ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺷﺎﻣﻞ ﺩﻩ‌ﻫﺎ ﻳﺎ ﺻﺪ‌ﻫﺎ ﮐﺘﺎﺏ ﺑﺎﺷﺪ ﺭﺍ ﺩﺭ ﻭﺳﻴﻠﻪ‌ﺍﯼ ﺑﻪ ﻭﺯﻥ ۳۰۰ ﺗﺎ ۶۰۰ ﮔﺮﻡ، ﻫﻤﻴﺸﻪ ﻭ ﺩﺭ ﺍﮐﺜﺮ ﻣﮑﺎﻥ‌ﻫﺎ ﺩﺭ ﮐﻨﺎﺭ ﺧﻮﺩ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺑﺮﻧﺎﻣﻪ‌ﺍﯼ ﺭﻳﺪﺭ ﺩﺳﺘﮕﺎﻩ ﺧﻮﺩ، ﻳﺎﺩﺩﺍﺷﺖ ﺑﺮﺩﺍﺭﯼ ﮐﺮﺩﻩ ﻳﺎ ﺑﻪ ﺍﺻﻄﻼﺡ ﺑﺮﺍﯼ ﺻﻔﺤﺎﺕ ﺣﺎﺷﻴﻪ ﺑﻨﻮﻳﺴﻴﺪ، ﺩﺭﺳﺖ‌‌ ﻫﻤﺎﻥ ﻃﻮﺭ ﮐﻪ ﺍﮐﺜﺮ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻣﺎ ﺩﺍﺭﺍﯼ ﺣﺎﺷﻴﻪ ﻧﻮﻳﺴﯽ ﻫﺴﺘﻨﺪ.
