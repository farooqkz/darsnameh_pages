ﺩﺭﺱ ﺳﻮﻡ- ﭼﺮﺍ ﻣﺮﺩﻡ ﮐﺘﺎﺏ ﻣﯽ ﺧﻮﺍﻧﻨﺪ؟

ﻫﺪﻑ ﻣﺎ ﺩﺭ ﺩﺭﺱ ﺣﺎﺿﺮ، ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﮐﻨﻴﻢ ﺗﺎ ﺑﺎ ﺩﻳﺪﮔﺎﻩ‌ﻫﺎﯼ ﺻﺤﻴﺢ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﻧﮕﺎﺭﺵ ﮐﺘﺎﺏ ﺁﺷﻨﺎ ﺷﻮﻳﺪ ﻭ ﻫﻤﭽﻨﻴﻦ ﺩﺭ ﻳﺎﺑﻴﺪ ﺁﻳﺎ ﮐﺘﺎﺑﯽ ﮐﻪ ﺑﺮﺍﯼ ﺗﺮﺟﻤﻪ ﻳﺎ ﻣﻮﺿﻮﻋﯽ ﮐﻪ ﺑﺮﺍﯼ ﻧﮕﺎﺭﺵ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ‌ﺍﻳﺪ ﻣﻨﺎﺳﺐ ﺍﺳﺖ ﻭ ﻣﺨﺎﻃﺐ ﺧﻮﺩ ﺭﺍ ﺧﻮﺍﻫﺪ ﻳﺎﻓﺖ ﻳﺎ ﺧﻴﺮ.

ﺍﺑﺘﺪﺍ ﺑﻴﺎﻳﻴﺪ ﻳﮏ ﭼﺎﻟﺶ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﻢ. ﺗﺼﻮﺭ ﮐﻨﻴﺪ ﺩﺍﺧﻞ ﻳﮏ ﮐﺘﺎﺑﻔﺮﻭﺷﯽ ﻫﺴﺘﻴﺪ؛ ﻓﺮﻗﯽ ﻧﻤﯽ‌ﮐﻨﺪ ﻳﮏ ﻓﺮﻭﺷﮕﺎﻩ ﻓﻴﺰﻳﮑﯽ ﻳﺎ ﺍﻳﻨﺘﺮﻧﺘﯽ، ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺧﻮﺍﻫﻴﺪ ﺩﻳﺪ. ﺭﻣﺎﻥ‌ﻫﺎﻳﯽ ﻣﺜﻞ ﺍﺭﺑﺎﺏ ﺣﻠﻘﻪ‌ﻫﺎ ﺍﺯ ﺗﺎﻟﮑﻴﻦ، ﻫﺮﯼ ﭘﺎ‌ﺗﺮ ﺍﺯ ﺟﯽ‌ﮐﯽ ﺭﻭﻟﻴﻨﮓ، ﻳﺎ ﮐﺘﺎﺏ‌ﻫﺎﻳﯽ ﻣﺜﻞ ﺳﻴﻨﻮﻫﻪ ﺗﺮﺟﻤﻪ ﺫﺑﻴﺢ ﺍﻟﻠﻪ ﻣﻨﺼﻮﺭﯼ ﻭ ﻏﻴﺮﻩ. ﻫﻤﻪ ﺍﻳﻦ ﮐﺘﺎﺏ‌ﻫﺎ ﺧﺮﻳﺪﻧﯽ ﻫﺴﺘﻨﺪ ﻭ ﻧﻪ ﺭﺍﻳﮕﺎﻥ! ﻣﻌﻤﻮﻻً ﺩﺭ ﮐﺘﺎﺑﻔﺮﻭﺷﯽ‌ﻫﺎ ﺩﺭﻳﺎﻳﯽ ﺍﺯ ﺍﻧﺘﺨﺎﺏ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺩﺭ ﺍﻳﻦ ﺑﻴﻦ ﭼﺮﺍ ﻳﮏ ﻧﻔﺮ -ﻫﺮ ﮐﺴﯽ ﺍﺯ ﻫﺮ ﻗﺸﺮﯼ- ﺑﺎﻳﺪ ﺑﺨﻮﺍﻫﺪ ﮐﺘﺎﺏ ﺷﻤﺎ ﺭﺍ ﺑﺮﺩﺍﺭﺩ ﻭ ﻭﺭﻕ ﺑﺰﻧﺪ؟ ﺍﻳﻦ ﺳﻮﺍﻝ ﺩﺭﺳﺘﯽ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﺍﺯ ﺧﻮﺩﺗﺎﻥ ﺑﭙﺮﺳﻴﺪ!

ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻧﻮﻳﺴﻨﺪﻩ‌ﻫﺎ ﻳﺎ ﺣﺘﯽ ﻧﺎﺷﺮﺍﻥ ﻧﻤﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﺍﻳﻦ ﭘﺮﺳﺶ، ﺟﻮﺍﺏ ﺑﺪﻫﻨﺪ ﺯﻳﺮﺍ ﺳﻮﺍﻝ ﺩﻳﮕﺮﯼ ﺟﻠﻮﯼ ﺩﻳﺪﺷﺎﻥ ﺭﺍ ﺳﺪ ﮐﺮﺩﻩ: ﭼﻄﻮﺭ ﻣﯽ‌ﺗﻮﺍﻧﻢ ﺍﺯ ﻧﻮﺷﺘﻦ ﻳﮏ ﮐﺘﺎﺏ ﭘﻮﻝ ﺑﻪ ﺟﻴﺐ ﺑﺰﻧﻢ؟

ﺷﺎﻳﺪ ﺩﺭ ﺍﺑﺘﺪﺍﯼ ﺭﺳﻴﺪﻥ ﺑﻪ ﺍﻳﺪﻩ ﻧﻮﺷﺘﻦ ﮐﺘﺎﺏ، ﺑﺎ ﺧﻮﺩﺗﺎﻥ ﺑﮕﻮﻳﻴﺪ: «ﻣﯽ‌ﺧﻮﺍﻫﻢ ﺑﻴﻦ ﻣﺮﺩﻡ ﺩﻳﺪﻩ ﺷﻮﻡ.» ﻳﺎ «ﻣﯽ‌ﺧﻮﺍﻫﻢ ﭘﻮﻟﺪﺍﺭ ﺷﻮﻡ.» ﻳﺎ «ﮐﺘﺎﺏ ﺑﺎﻋﺚ ﻣﯽ‌ﺷﻮﺩ ﻧﺎﻣﻢ ﺳﺮ ﺯﺑﺎﻥ‌ﻫﺎ ﺑﻴﻔﺘﺪ ﻭ ﻣﻌﺮﻭﻑ ﺷﻮﻡ.» ﻳﺎ «ﺑﺮﺍﯼ ﻭﺟﻬﻪ‌ﺍﻡ ﺧﻮﺏ ﺍﺳﺖ.» ﻳﺎ ﻫﺮ ﭼﻴﺰ ﺩﻳﮕﺮﯼ.

ﺍﺯ ﺷﻤﺎ ﻣﯽ‌ﺧﻮﺍﻫﻢ ﺑﻪ ﻳﮏ ﺳﻮﺍﻝ ﺟﻮﺍﺏ ﺩﻫﻴﺪ ﺗﺎ ﺑﺮﺍﻳﺘﺎﻥ ﺑﻬﺘﺮ ﻣﻌﻠﻮﻡ ﺷﻮﺩ ﮐﻪ ﭼﻘﺪﺭ ﺍﻳﻦ ﺩﻻﻳﻞ ﺑﺮﺍﯼ ﻣﺨﺎﻃﺒﻴﻦ ﺑﯽ‌ﺭﺑﻂ ﻫﺴﺘﻨﺪ، ﻫﺮ ﭼﻨﺪ ﻣﻤﮑﻦ ﺍﺳﺖ ﺑﺮﺍﯼ ﺧﻮﺩ ﻣﻮﻟﻒ ﻭﺍﻗﻌﺎً ﻣﻬﻢ ﺑﺎﺷﻨﺪ! ﺳﻮﺍﻝ ﺍﻳﻦ ﺍﺳﺖ: ﭼﻨﺪ ﺑﺎﺭ ﺷﺪﻩ ﺑﻪ ﺧﺮﻳﺪ ﮐﺘﺎﺏ ﻳﺎ ﮔﺸﺖ ﺯﺩﻥ ﺩﺭ ﻳﮏ ﮐﺘﺎﺏ ﻓﺮﻭﺷﯽ ﺑﺮﻭﻳﺪ ﻭ ﺍﻳﻦ ﻓﮑﺮ ﺩﺭ ﺳﺮﺗﺎﻥ ﺑﺎﺷﺪ ﮐﻪ «ﭼﻄﻮﺭ ﻣﯽ‌ﺗﻮﺍﻧﻢ ﺑﻪ ﻣﺆﻟﻒ ﮐﺘﺎﺏ ﮐﻤﮏ ﮐﻨﻢ ﺑﻪ ﺍﻫﺪﺍﻓﺶ ﺑﺮﺳﺪ؟» ﻣﻨﺼﻒ ﺑﺎﺷﻴﺪ، ﭘﺎﺳﺦ ﺗﻘﺮﻳﺒﺎً ﺑﺮﺍﯼ ﻫﻤﻪ ﺍﻓﺮﺍﺩ ﺍﻳﻦ ﺍﺳﺖ: «ﻫﺮﮔﺰ!» ﺑﻠﻪ ﻫﺴﺘﻨﺪ ﻧﻮﻳﺴﻨﺪﮔﺎﻧﯽ ﮐﻪ ﺩﺭﺁﻣﺪﻫﺎﯼ ﺁﻧﭽﻨﺎﻧﯽ ﺩﺍﺭﻧﺪ، ﻭﻟﯽ ﺍﻳﻦ ﺩﻟﻴﻞ ﺧﺮﻳﺪ ﮐﺘﺎﺑﺸﺎﻥ ﺗﻮﺳﻂ ﻣﺮﺩﻡ ﻧﻴﺴﺖ. ﭼﻨﻴﻦ ﻗﻀﻴﻪ‌ﺍﯼ ﺑﺮﺍﯼ ﺷﻤﺎ ﻫﻢ ﺻﺪﻕ ﻣﯽ‌ﮐﻨﺪ.

 

۴ ﺩﻟﻴﻞ ﺩﺭﺳﺖ ﺑﺮﺍﯼ ﻧﻮﺷﺘﻦ ﻳﮏ ﮐﺘﺎﺏ ﻳﺎ ﺍﻧﺘﺨﺎﺏ ﺁﻥ ﺑﺮﺍﯼ ﺗﺮﺟﻤﻪ
ﻳﮑﻢ:

  • ﻏﻨﯽ ﮐﺮﺩﻥ ﺯﻧﺪﮔﯽ ﻣﺮﺩﻡ؛ ﻣﺮﺩﻡ ﺑﺎ ﻋﻼﻳﻖ ﻭ ﺳﻠﻴﻘﻪ‌ﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﺑﺮﺍﯼ ﺍﻳﻦ ﮐﺘﺎﺏ ﻧﻤﯽ‌ﺧﺮﻧﺪ ﮐﻪ ﺁﻥ ﺭﺍ ﺑﺎﻳﮕﺎﻧﯽ ﮐﻨﻨﺪ. ﺑﻠﮑﻪ ﻣﯽ‌ﺧﻮﺍﻫﻨﺪ ﺍﺭﺯﺷﯽ ﺭﺍ ﺑﻪ ﺯﻧﺪﮔﯽ ﺧﻮﺩﺷﺎﻥ ﺍﺿﺎﻓﻪ ﮐﻨﻨﺪ ﻭ ﺷﻤﺎ ﺑﺎﻳﺪ ﺑﺎ ﻳﮏ ﭼﻨﻴﻦ ﻫﺪﻓﯽ ﺑﻪ ﻧﮕﺎﺭﺵ ﻳﺎ ﺗﺮﺟﻤﻪ ﮐﺘﺎﺏ ﺑﭙﺮﺩﺍﺯﻳﺪ. ﮐﺘﺎﺏ‌ﻫﺎ ﺍﺯ ﺍﻳﻦ ﺭﺍﻩ‌ﻫﺎ ﺯﻧﺪﮔﯽ ﻣﺮﺩﻡ ﺭﺍ ﻏﻨﯽ ﻣﯽ‌ﮐﻨﻨﺪ:
  • ﺩﺍﻧﺶ‌ﻫﺎﯼ ﺗﺨﺼﺼﯽ؛ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻋﻠﻤﯽ ﺩﺭﺑﺎﺭﻩ ﭼﮕﻮﻧﮕﯽ ﮐﺎﺭﮐﺮﺩ ﺩﻧﻴﺎﯼ ﭘﻴﺮﺍﻣﻮﻧﻤﺎﻥ ﺗﻮﺿﻴﺢ ﻣﯽ‌ﺩﻫﻨﺪ. ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻣﺮﺑﻮﻁ ﺑﻪ ﮐﺴﺐ ﻭ ﮐﺎﺭ ﺩﺭﺑﺎﺭﻩ ﻓﻨﻮﻥ ﻣﺪﻳﺮﻳﺘﯽ ﺣﺮﻑ ﻣﯽ‌ﺯﻧﻨﺪ. ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺗﺎﺭﻳﺨﯽ ﺑﻪ ﺷﺮﺡ ﺍﺗﻔﺎﻗﺎﺕ ﮔﺬﺷﺘﻪ ﻣﯽ‌ﭘﺮﺩﺍﺯﻧﺪ. ﺍﻳﻦ ﻧﻮﻉ ﮐﺘﺎﺏ‌ﻫﺎ، ﺑﻪ ﺍﺷﺎﻋﻪ ﺩﺍﻧﺶ ﻭ ﺗﺨﺼﺺ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻨﺪ.
  • ﺍﺩﺭﺍﮎ ﻭ ﺁﮔﺎﻫﯽ؛ ﺑﺮﺧﯽ ﺭﻣﺎﻥ‌ﻫﺎ ﺩﺭ ﻳﮏ ﺳﻮ ﻭ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﻏﻴﺮ ﺩﺍﺳﺘﺎﻧﯽ ﺩﺭ ﺳﻮﯼ ﺩﻳﮕﺮ ﺑﻪ ﻣﺮﺩﻡ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻨﺪ ﮐﻪ ﺧﻮﺩﺷﺎﻥ ﻭ ﺩﻳﮕﺮﺍﻥ ﺭﺍ ﺑﻬﺘﺮ ﺑﺸﻨﺎﺳﻨﺪ. ﺩﺭ ﺍﻳﻦ ﮐﺘﺎﺏ‌ﻫﺎ، ﺍﺑﺰﺍﺭ‌ﻫﺎ ﻭ ﺭﻭﺵ‌ﻫﺎﻳﯽ ﺑﺮﺍﯼ ﭘﺮﻭﺭﺵ ﺩﺍﺩﻥ ﻫﻮﺷﻴﺎﺭﯼ ﺑﻬﺘﺮ ﻧﺴﺒﺖ ﺑﻪ ﺩﻧﻴﺎﯼ ﭘﻴﺮﺍﻣﻮﻥ، ﺍﺭﺍﺋﻪ ﻣﯽ‌ﺷﻮﺩ. ﺑﻪ ﻋﻨﻮﺍﻥ ﻣﺜﺎﻝ ﮐﺘﺎﺏ‌ﻫﺎﻳﯽ ﻧﻈﻴﺮ «ﺯﻧﺪﮔﻴﻨﺎﻣﻪ ﺍﺳﺘﻴﻮ ﺟﺎﺑﺰ» ﺍﺯ ﺁﻥ ﺩﺳﺖ ﮐﺘﺎﺏ‌ﻫﺎﻳﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺷﻤﺎ ﺭﺍ ﻧﻪ ﺗﻨﻬﺎ ﺑﺎ ﺩﺍﻧﺶ ﻣﺪﻳﺮﻳﺖ (ﺩﺍﻧﺶ‌ﻫﺎﯼ ﺗﺨﺼﺼﯽ) ﺁﺷﻨﺎ ﻣﯽ‌ﮐﻨﻨﺪ ﺑﻠﮑﻪ ﺩﺭﮐﯽ ﺑﺴﻴﺎﺭ ﺧﻮﺏ ﻧﺴﺒﺖ ﺑﻪ ﮐﺎﺭ ﮐﺮﺩﻥ ﺩﺭ ﺑﻴﻦ ﻧﻮﺁﻭﺭﺍﻥ، ﻓﻨﺎﻭﺭﺍﻥ ﻭ ﻫﻨﺮﻣﻨﺪﺍﻥ ﺭﺍ ﻣﯽ‌ﺩﻫﻨﺪ.
  • ﺳﺮﮔﺮﻡ ﺳﺎﺯﯼ؛ ﻣﺴﻠﻤﺎً ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺩﺍﺳﺘﺎﻧﯽ ﺑﺎ ﺗﺼﻮﻳﺮ ﮐﺮﺩﻥ ﺍﻓﺴﺎﻧﻪ، ﺧﻴﺎﻝ‌ﭘﺮﺩﺍﺯﯼ، ﻣﺎﺟﺮﺍﺟﻮﻳﯽ ﻭ ﻣﻌﺮﻓﯽ ﺷﺨﺼﻴﺖ‌ﻫﺎﻳﯽ ﮐﻪ ﺩﺭ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ ﻧﻤﯽ‌ﺑﻴﻨﻴﻢ، ﻣﺎ ﺭﺍ ﺳﺮﮔﺮﻡ ﻣﯽ‌ﮐﻨﻨﺪ. ﻳﮏ ﻧﻔﺮ ﻣﯽ‌ﺧﻮﺍﻫﺪ ﻗﻬﺮﻣﺎﻥ ﺑﺎﺷﺪ، ﺩﻳﮕﺮﯼ ﺟﺎﺳﻮﺱ، ﺷﺨﺼﯽ ﻣﯽ‌ﺧﻮﺍﻫﻮ ﻣﺜﻞ ﺑﺘﻤﻦ ﺑﻪ ﮐﻤﮏ ﺟﺎﻣﻌﻪ ﺑﻴﺎﻳﺪ ﻭ ﺁﻥ ﺭﺍ ﺍﺯ ﭘﻠﺸﺘﯽ ﻧﺠﺎﺕ ﺩﻫﺪ. ﻫﺮ ﮐﺴﯽ ﺑﺮﺍﯼ ﺧﻮﺩﺵ ﺩﺍﺳﺘﺎﻧﯽ ﺷﺨﺼﯽ ﺩﺍﺭﺩ ﻭ ﺩﺭ ﺁﻥ ﺑﻪ ﻧﻘﺸﯽ ﺟﺪﺍﯼ ﺍﺯ ﺯﻧﺪﮔﯽ ﺭﻭﺯﻣﺮﻩ‌ﺍﺵ ﻣﯽ‌ﭘﺮﺩﺍﺯﺩ. ﺷﺎﻳﺪ «ﻓﺮﻭﺩﻭ ﺑﮕﻴﻨﺰ» ﺩﺭ ﺳﻪ ﮔﺎﻧﻪ ﺍﺭﺑﺎﺏ ﺣﻠﻘﻪ‌ﻫﺎ، ﻳﮑﯽ ﺍﺯ ﺑﻬﺘﺮﻳﻦ ﻧﻤﻮﻧﻪ ﺷﺨﺼﻴﺖ‌ﻫﺎ ﺑﺎﺷﺪ ﮐﻪ ﻋﺪﻩ ﺯﻳﺎﺩﯼ ﺍﺯ ﻣﺮﺩﻡ ﺩﻧﻴﺎ ﺑﺎ ﺁﻥ ﻫﻤﺬﺍﺕ ﭘﻨﺪﺍﺭﯼ ﮐﺮﺩﻩ‌ﺍﻧﺪ، ﺯﻳﺮﺍ ﻫﻴﭻ ﻗﺪﺭﺕ ﻣﺎﻭﺭﺍﻳﯽ ﻳﺎ ﺟﺴﻤﺎﻧﯽ ﺧﺎﺻﯽ ﻧﺪﺍﺭﺩ ﺟﺰ ﺍﺭﺍﺩﻩ ﻗﻮﯼ ﻭ ﻗﻠﺐ ﻣﻬﺮﺑﺎﻧﯽ ﮐﻪ ﺍﻭ ﺭﺍ ﺍﺯ ﻏﻠﺘﻴﺪﻥ ﺑﻪ ﺗﺎﺭﻳﮑﯽ ﺩﺭ ﺍﻣﺎﻥ ﻧﮕﺎﻩ ﻣﯽ‌ﺩﺍﺭﺩ.

ﮔﻔﺘﻢ ﮐﻪ ﻫﺪﻑ ﺍﺻﻠﯽ ﺍﺯ ﻧﮕﺎﺭﺵ ﻳﮏ ﮐﺘﺎﺏ ﻧﺒﺎﻳﺪ ﺻﺮﻓﺎً ﮐﺴﺐ ﺛﺮﻭﺕ ﺑﺎﺷﺪ. ﻭﻟﯽ ﺍﻳﻦ ﺑﻪ ﺁﻥ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﻧﺒﺎﻳﺪ ﺩﺭ ﺯﻣﻴﻨﻪ ﭘﺮ ﻓﺮﻭﺵ ﺑﻮﺩﻥ ﮐﺘﺎﺏ، ﻫﻮﺷﻤﻨﺪﺍﻧﻪ ﻋﻤﻞ ﮐﻨﻴﺪ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺣﺘﻤﺎً ﻭﻗﺖ ﺧﻮﺩ ﺭﺍ ﺻﺮﻑ ﮐﺘﺎﺑﯽ ﮐﻨﻴﺪ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﻓﺮﻭﺵ ﺧﻮﺑﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺑﺮﺍﯼ ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ، ﻣﻮﺍﺭﺩ ﺯﻳﺮ ﺭﺍ ﻣﺪ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ:

  • ﺣﻞ ﮐﺮﺩﻥ ﻳﮏ ﻣﺸﮑﻞ؛ ﻳﮑﯽ ﺍﺯ ﺟﺬﺍﺏ‌ﺗﺮﻳﻦ ﻣﻮﺿﻮﻋﺎﺕ ﺑﺮﺍﯼ ﻧﻮﺷﺘﻦ ﮐﺘﺎﺏ، ﻣﻮﺿﻮﻋﯽ ﺳﺖ ﮐﻪ ﺑﺎ «ﭼﻄﻮﺭ...» ﺷﺮﻭﻉ ﻣﯽ‌ﺷﻮﺩ: «ﭼﻄﻮﺭ ﻳﮏ ﻧﺠﺎﺭ ﻣﺎﻫﺮ ﺷﻮﻳﻢ؟» ﻳﺎ «ﭼﻄﻮﺭ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺲ ﻣﺎﻫﺮ ﺷﻮﻳﻢ؟» ﻭ ﻏﻴﺮﻩ. ﺧﻮﺑﯽ ﭼﻨﻴﻦ ﮐﺘﺎﺏ‌ﻫﺎﻳﯽ ﺩﺭ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺁﻧﻬﺎ ﺭﺍ ﺳﻄﺢ ﺑﻨﺪﯼ ﻫﻢ ﺑﮑﻨﻴﺪ ﻭ ﺑﻪ ﺟﺎﯼ ﻳﮏ ﮐﺘﺎﺏ، ﻳﮏ ﻣﺠﻤﻮﻋﻪ ﮐﺘﺎﺏ ﺑﻴﺮﻭﻥ ﺩﻫﻴﺪ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺭﻭﻳﮑﺮﺩ ﺑﺮﺍﯼ ﻣﺒﺎﺣﺚ ﺑﻪ ﻫﻢ ﭘﻴﻮﺳﺘﻪ ﺑﻴﺸﺘﺮ ﺍﺯ ﺳﺎﻳﺮ ﻣﺒﺎﺣﺚ ﺟﻮﺍﺏ ﻣﯽ‌ﺩﻫﺪ.
  • ﭘﺮﺩﺍﺧﺘﻦ ﺑﻪ ﻳﮏ ﺗﺮﺱ؛ ﺍﮐﺜﺮ ﻣﺎ ﺗﺮﺱ‌ﻫﺎﻳﯽ ﺩﺭ ﺩﺭﻭﻥ ﺧﻮﺩ ﺩﺍﺭﻳﻢ ﻭ ﺗﻤﺎﻡ ﺗﻼﺷﻤﺎﻥ ﺭﺍ ﻣﯽ‌ﮐﻨﻴﻢ ﺗﺎ ﺍﺯ ﺁﻧﻬﺎ ﺩﻭﺭﯼ ﺑﺠﻮﻳﻴﻢ ﻳﺎ ﺩﺭ ﺍﻋﻤﺎﻕ ﻭﺟﻮﺩﻣﺎﻥ ﺩﻓﻨﺸﺎﻥ ﮐﻨﻴﻢ. ﺗﺮﺱ‌ﻫﺎﻳﯽ ﺍﺯ «ﻣﺮﺩﻡ ﺩﺭﺑﺎﺭﻩ ﻣﻦ ﭼﻪ ﻓﮑﺮﯼ ﻣﯽ‌ﮐﻨﻨﺪ؟» ﮔﺮﻓﺘﻪ ﺗﺎ «ﺍﮔﺮ ﻧﺎﻣﺰﺩﻡ ﺍﺯ ﺳﺮﻃﺎﻥ ﺑﻤﻴﺮﺩ ﭼﻪ؟» ﻳﺎ «ﺍﮔﺮ ﺑﯽ‌ﭘﻮﻝ ﻭ ﻓﻘﻴﺮ ﺷﻮﻡ ﭼﻪ؟» ﻳﺎ ﺍﮔﺮ «ﺑﻤﻴﺮﻡ ﭼﻪ؟» ﻳﺎ «ﺍﮔﺮ ﺑﭽﻪ‌ﻫﺎﻳﻢ ﺑﺰﻫﮑﺎﺭ ﺷﻮﻧﺪ ﭼﻪ؟» ﻳﺎ «ﺍﮔﺮ ﺯﻧﺪﮔﯽ ﻣﺸﺘﺮﮐﻢ ﺩﻭﺍﻡ ﻧﻴﺎﻭﺭﺩ ﭼﻪ؟»
    ﺍﮔﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺩﺭ ﻗﺎﻟﺐ ﻳﮏ ﮐﺘﺎﺏ ﺑﻪ ﺍﻳﻦ ﺗﺮﺱ‌ﻫﺎ ﺑﭙﺮﺩﺍﺯﻳﺪ، ﺑﻪ ﺗﻌﺪﺍﺩ ﺁﺩﻡ‌ﻫﺎﻳﯽ ﮐﻪ ﺑﺎ ﺍﻳﻦ ﺗﺮﺱ‌ﻫﺎ ﺯﻧﺪﮔﯽ ﻣﯽ‌ﮐﻨﻨﺪ ﻣﺨﺎﻃﺐ ﺑﺎﻟﻘﻮﻩ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﺷﻤﺎ ﻓﻘﻂ ﺑﺎﻳﺪ ﺑﺎ ﺟﻮﺍﺏ‌ﻫﺎﯼ ﺩﺭﺳﺖ ﺍﺯ ﻋﻤﻖ ﺩﺭﻳﺎﯼ ﺗﺮﺱ ﺁﺩﻡ‌ﻫﺎ ﺑﻴﺮﻭﻥ ﺑﻴﺎﻳﻴﺪ ﻭ ﺑﻄﺮﯼ ﻧﺠﺎﺕ ﺭﺍ ﺗﻘﺪﻳﻤﺸﺎﻥ ﮐﻨﻴﺪ. ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺑﻪ ﺁﻥ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻫﺮ ﻣﺰﺧﺮﻓﯽ ﺭﺍ ﺑﻪ ﺧﻮﺭﺩ ﻣﺨﺎﻃﺐ ﺩﻫﻴﺪ. ﻳﺎﺩﺗﺎﻥ ﻧﺮﻭﺩ ﮐﻪ ﺍﺣﺘﺮﺍﻡ ﺑﻪ ﻣﺨﺎﻃﺐ، ﺑﻪ ﻣﻌﻨﺎﯼ ﺭﺍﺳﺘﮕﻮﻳﯽ ﺑﻪ ﺍﻭ ﻧﻴﺰ ﻫﺴﺖ. ﺑﻨﺎﺑﺮﺍﻳﻦ ﺑﺎﻳﺪ ﺭﺍﻩ ﺣﻞ‌ﻫﺎﯼ ﻭﺍﻗﻌﯽ ﺁﻥ ﺗﺮﺱ ﺭﺍ ﮐﻪ ﺑﺮﺍﯼ ﺧﻮﺩﺗﺎﻥ ﻣﻔﻴﺪ ﺑﻮﺩﻩ‌ﺍﻧﺪ، ﺑﻪ ﻣﺨﺎﻃﺐ ﺍﺭﺍﺋﻪ ﮐﻨﻴﺪ.
  • ﺍﺭﺿﺎﺀ ﻳﮏ ﮐﻨﺠﮑﺎﻭﯼ؛ ﻣﺮﺩﻡ ﺩﺭﺑﺎﺭﻩ ﭼﻴﺰﻫﺎﯼ ﻣﺨﺘﻠﻔﯽ ﺷﮕﻔﺖ ﺯﺩﻩ ﻣﯽ‌ﺷﻮﻧﺪ. ﺑﺮﺍﯼ ﻫﻤﻴﻦ ﺍﺳﺖ ﮐﻪ ﺳﺘﺎﺭﻩ‌ﻫﺎﯼ ﻋﺎﻟﻢ ﻫﻨﺮ ﻭ ﻣﺴﺎﺑﻘﻪ‌ﻫﺎﯼ ﺗﻠﻮﻳﺰﻳﻮﻧﯽ ﻏﻴﺮ ﻗﺎﺑﻞ ﭘﻴﺶ ﺑﻴﻨﯽ، ﺍﻳﻨﻘﺪﺭ ﭼﺸﻢ‌ﻫﺎ ﺭﺍ ﺑﻪ ﺳﻤﺖ ﺧﻮﺩ ﺧﻴﺮﻩ ﻣﯽ‌ﮐﻨﻨﺪ. «ﺯﻧﺪﮔﯽ ﺩﺭ ﻳﮏ ﺧﺎﻧﻮﺍﺩﻩ ۱۹ ﻧﻔﺮﻩ ﭼﻄﻮﺭ ﺍﺳﺖ؟» ﻳﺎ «ﭼﻄﻮﺭ ﺑﻴﻞ ﮔﻴﺘﺲ ﺛﺮﻭﺗﻤﻨﺪ‌ﺗﺮﻳﻦ ﻓﺮﺩ ﺯﻣﻴﻦ ﺷﺪ؟» ﻳﺎ «ﭼﻄﻮﺭ ﺩﻧﻴﻞ ﺩِﯼ ﻟﻮﺋﻴﺲ ﺑﺎ ﻫﺮ ﻓﻴﻠﻤﯽ ﮐﻪ ﺑﺎﺯﯼ ﻣﯽ‌ﮐﻨﺪ ﻳﮏ ﺍﺳﮑﺎﺭ ﻣﯽ‌ﺑﺮﺩ؟»
    ﺧﻮﺩﺗﺎﻥ ﺗﺼﻮﺭ ﮐﻨﻴﺪ ﮐﻪ ﭘﺮﺩﺍﺧﺘﻦ ﺑﻪ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﭼﻘﺪﺭ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺟﺬﺍﺏ ﺑﺎﺷﺪ. ﺩﺭ ﺍﺻﻞ ﺍﺭﺍﺋﻪ ﺍﻃﻼﻋﺎﺗﯽ ﮐﻪ ﺩﺭ ﻫﻤﻪ ﺟﺎ ﻣﻮﺟﻮﺩ ﻧﻴﺴﺘﻨﺪ ﺍﺳﺖ ﮐﻪ ﺑﺎﻋﺚ ﺑﻪ ﻭﺟﻮﺩ ﺁﻣﺪﻥ ﺍﻳﻦ ﺟﺬﺍﺑﻴﺖ ﻣﯽ‌ﺷﻮﺩ. ﺯﻧﺪﮔﻴﻨﺎﻣﻪ ﺍﺳﺘﻴﻮ ﺟﺎﺑﺰ ﺭﺍ ﺑﻪ ﺧﺎﻃﺮ ﺑﻴﺎﻭﺭﻳﺪ ﮐﻪ ﭼﻄﻮﺭ ﭘﺮﻓﺮﻭﺵ‌ﺗﺮﻳﻦ ﮐﺘﺎﺏ ﺳﺎﻝ ﺁﻣﺎﺯﻭﻥ ﺷﺪ. ﺯﻳﺮﺍ ﻫﻤﻪ ﮐﻨﺠﮑﺎﻭ ﺑﻮﺩﻧﺪ ﮐﻪ ﺑﺪﺍﻧﻨﺪ ﭘﺸﺖ ﺻﺤﻨﻪ ﺯﻧﺪﮔﯽ ﺍﻳﻦ ﻧﺎﺑﻐﻪ ﺩﻧﻴﺎﯼ ﻓﻨﺎﻭﺭﯼ، ﭼﻄﻮﺭ ﺑﻮﺩﻩ.
  • ﻳﮏ ﺗﻴﺮ ﻭ ﺩﻭ ﻧﺸﺎﻥ؛ ﺣﺘﻤﺎ ﺍﻳﻦ ﺟﻤﻠﻪ ﺭﺍ ﺷﻨﻴﺪﻩ‌ﺍﻳﺪ. ﺩﺭ ﺍﻳﻨﺠﺎ ﻫﻢ ﻣﻔﻬﻮﻡ ﺁﻥ ﺻﺪﻕ ﻣﯽ‌ﮐﻨﺪ. ﺍﮔﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺩﻭ ﺗﺎ ﺍﺯ ﻣﻮﺍﺭﺩ ﻓﻮﻕ ﺭﺍ ﺩﺭ ﮐﺘﺎﺏ ﺧﻮﺩ ﺑﮕﻨﺠﺎﻧﻴﺪ ﻃﻮﺭﯼ ﮐﻪ ﺗﺄﺛﻴﺮ ﮔﺬﺍﺭﯼ ﺧﻮﺩ ﺭﺍ ﻧﻴﺰ ﺍﺯ ﺩﺳﺖ ﻧﺪﻫﻨﺪ، ﺁﻥ ﻭﻗﺖ ﺍﺳﺖ ﮐﻪ ﺑﺮﺩ ﮐﺮﺩﻩ‌ﺍﻳﺪ. ﺣﺘﯽ ﺍﮔﺮ ﺑﺘﻮﺍﻧﻴﺪ ﺭﺍﻫﯽ ﺑﻴﺎﺑﻴﺪ ﺑﺮﺍﯼ ﺍﻳﻨﮑﻪ ﻫﻤﺰﻣﺎﻥ ﻫﻢ ﺭﺍﻩ ﺣﻠﯽ ﺑﺮﺍﯼ ﻳﮏ ﻣﺸﮑﻞ ﺍﺭﺍﺋﻪ ﮐﻨﻴﺪ، ﻫﻢ ﺑﻪ ﻳﮏ ﺗﺮﺱ ﺑﭙﺮﺩﺍﺯﻳﺪ ﻭ ﻫﻢ ﻳﮏ ﮐﻨﺠﮑﺎﻭﯼ ﺭﺍ ﻫﻢ ﺍﺭﺿﺎﺀ ﮐﻨﻴﺪ، ﺁﻥ ﻭﻗﺖ ﺩﻳﮕﺮ ﮐﺘﺎﺏ ﺷﻤﺎ ﺟﺰﻭ ﺳﺘﺎﺭﻩ‌ﻫﺎ ﺧﻮﺍﻫﺪ ﺑﻮﺩ!

 

۵ ﭘﻨﺪ ﺍﺭﺯﺷﻤﻨﺪ ﺩﺭ ﺭﺍﺑﻄﻪ ﺑﺎ ﺍﻧﺘﺨﺎﺏ ﻣﻮﺿﻮﻉ
۱. ﺩﺭﺑﺎﺭﻩ ﻣﻮﺿﻮﻋﯽ ﮐﺘﺎﺏ ﺑﻨﻮﻳﺴﻴﺪ ﮐﻪ ﻧﺴﺒﺖ ﺑﻪ ﺁﻥ ﺍﺣﺴﺎﺱ ﺍﺷﺘﻴﺎﻕ ﻭ ﻋﻼﻗﻪ ﺷﺪﻳﺪﯼ ﺩﺍﺭﻳﺪ؛ ﻭ ﺍﮔﺮ ﭼﻨﻴﻦ ﻧﻴﺴﺖ ﻻ ﺍﻗﻞ ﺩﺭﺑﺎﺭﻩ ﭼﻴﺰﯼ ﺑﻨﻮﻳﺴﻴﺪ ﮐﻪ ﺍﺯ ﺁﻥ ﺍﻃﻼﻋﺎﺕ ﺯﻳﺎﺩﯼ ﺩﺍﺭﻳﺪ.

۲. ﺩﺭﺑﺎﺭﻩ ﻣﻮﺿﻮﻋﯽ ﺑﻨﻮﻳﺴﻴﺪ ﮐﻪ ﻣﺎﺩﻩ ﺍﻭﻟﻴﻪ ﮐﺎﻓﯽ ﺭﺍ ﺩﺍﺭﺩ ﻳﺎ ﺑﻪ ﻋﺒﺎﺭﺕ ﺑﻬﺘﺮ، ﻳﮏ ﺷﮑﺎﺭ ﭘﺮﻭﺍﺭ ﺍﻧﺘﺨﺎﺏ ﮐﻨﻴﺪ. ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﺣﺪﺍﻗﻞ ﺑﺎﻳﺪ ۲۵ ﺻﻔﺤﻪ ﺑﺎﺷﻨﺪ. ﻣﻮﺿﻮﻋﯽ ﺍﻧﺘﺨﺎﺏ ﻧﮑﻨﻴﺪ ﮐﻪ ﺩﺭ ﺻﻔﺤﻪ ﺩﻫﻢ، ﻧﻮﺷﺘﻦ ﺁﻥ ﻣﺘﻮﻗﻒ ﺷﻮﺩ.

۳. ﺩﺭﺑﺎﺭﻩ ﻣﻮﺿﻮﻋﯽ ﺑﻨﻮﻳﺴﻴﺪ ﮐﻪ ﺑﺎ ﻫﻮﻳﺖ ﺷﻤﺎ (ﭼﻪ ﺩﺭ ﺩﻧﻴﺎﯼ ﻭﺍﻗﻌﯽ، ﭼﻪ ﺩﺭ ﺩﻧﻴﺎﯼ ﺍﻳﻨﺘﺮﻧﺖ) ﻫﻤﺨﻮﺍﻧﯽ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ.

۴. ﻣﺨﺎﻃﺐ ﻫﺪﻑ ﺭﺍ ﭘﻴﺶ ﺍﺯ ﺷﺮﻭﻉ ﺑﻪ ﻧﻮﺷﺘﻦ، ﺩﺭ ﻧﻈﺮ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ. ﺩﺭ ﺩﻧﻴﺎﯼ ﺍﻳﻨﺘﺮﻧﺖ ﺑﺨﺶ ﺑﺰﺭﮔﯽ ﺍﺯ ﻣﺨﺎﻃﺒﻴﻦ ﺟﺪﻳﺪ ﻫﺮ ﮐﺘﺎﺑﯽ، ﺑﺎﺯﺍﺭﻳﺎﺑﺎﻥ ﺭﺍﻳﮕﺎﻥ ﻭ ﺳﻴﻨﻪ ﺑﻪ ﺳﻴﻨﻪ ﺑﺮﺍﯼ ﺳﺎﻳﺮ ﻣﺨﺎﻃﺒﻴﻦ ﺁﻥ ﻫﺴﺘﻨﺪ. ﺁﻳﺎ ﺣﻮﺯﻩ‌ﺍﯼ ﮐﻪ ﺑﺮﺍﯼ ﻧﻮﻳﺴﻨﺪﮔﯽ ﺍﻧﺘﺨﺎﺏ ﮐﺮﺩﻩ‌ﺍﻳﺪ، ﻣﺨﺎﻃﺒﻴﻦ ﻓﻌﺎﻟﯽ ﺩﺍﺭﺩ؟
