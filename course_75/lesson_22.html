ﺩﺭﺱ ﭼﻬﺎﺭﻡ- ﺩﮐﺘﺮﻳﻦ ﺍﻧﺘﺸﺎﺭ ﺷﺨﺼﯽ- ﺑﺨﺶ ﺍﻭﻝ

ﺗﻨﻬﺎ ﭼﻴﺰ ﺑﺪ‌ﺗﺮ ﺍﺯ ﻧﺎﺑﻮﺩ ﮐﺮﺩﻥ ﺭﺅﻳﺎﻫﺎﯼ ﺍﻓﺮﺍﺩ، ﮔﻤﺮﺍﻩ ﮐﺮﺩﻥ ﺁﻧﻬﺎﺳﺖ. ﻣﺸﮑﻞ ﺍﻳﻦ ﻧﻴﺴﺖ ﮐﻪ ﻧﺎﺷﺮﺍﻥ ﺳﻨﺘﯽ ﺍﻓﺮﺍﺩ ﺷﻴﻄﺎﻥ ﺻﻔﺖ ﻳﺎ ﮐﺎﺳﺒﮑﺎﺭﯼ ﻫﺴﺘﻨﺪ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻓﺮﺍﺩ ﺍﻳﻦ ﺻﻨﻌﺖ، ﺑﺎﻫﻮﺵ ﻭ ﺩﺍﺭﺍﯼ ﺫﻫﻦ‌ﻫﺎﯼ ﺑﺎﺯ ﻫﺴﺘﻨﺪ، ﺷﺎﻳﺪ ﺣﺘﯽ ﺑﻴﺸﺘﺮ ﺍﺯ ﻫﺮ ﺻﻨﻌﺖ ﺩﻳﮕﺮﯼ؛ ﺯﻳﺮﺍ ﻋﺸﻖ ﺑﻪ ﮐﺘﺎﺏ‌ﻫﺎ ﺍﺳﺖ ﮐﻪ ﺍﻳﻦ ﺁﺩﻡ‌ﻫﺎ ﺭﺍ ﺑﻪ ﮐﺎﺭ ﻭﺍ ﻣﯽ‌ﺩﺍﺭﺩ. ﺍﻟﺒﺘﻪ ﺁﺩﻡ ﺧﻮﺏ ﻭ ﺑﺪ، ﺩﺭ ﻫﻤﻪ ﺟﺎ ﭘﻴﺪﺍ ﻣﯽ‌ﺷﻮﺩ.

ﻣﺸﮑﻞ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻧﺸﺮ ﺳﻨﺘﯽ ﺑﺎ ﺩﻧﻴﺎﻳﯽ ﺍﺯ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎ ﺭﺷﺪ ﻳﺎﻓﺘﻪ ﻭ ﺷﮑﻞ ﮔﺮﻓﺘﻪ ﺍﺳﺖ. ﺍﻣﺎ ﺩﺭ ﺯﻣﺎﻥ ﺣﺎﺿﺮ، ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺍﻳﻦ ﻣﺤﺪﻭﺩﻳﺖ‌ﻫﺎ ﺩﻳﮕﺮ ﻭﺟﻮﺩ ﺧﺎﺭﺟﯽ ﻧﺪﺍﺭﻧﺪ. ﻣﻮﺍﺭﺩﯼ ﻣﺜﻞ، ﭘﻴﺸﺨﻮﺍﻥ ﮐﺘﺎﺏ ﻓﺮﻭﺷﯽ‌ﻫﺎ، ﺩﺳﺘﺮﺳﯽ ﺑﻪ ﻣﺎﺷﻴﻦ‌ﻫﺎﯼ ﭼﺎﭖ، ﻣﻬﺎﺭﺕ ﻭﻳﺮﺍﻳﺶ ﻭ ﺁﻣﺎﺩﻩ ﺳﺎﺯﯼ ﮐﺘﺎﺏ ﻭ... ﺩﻳﮕﺮ ﺍﻓﺮﺍﺩ ﺭﺍ ﻣﺤﺪﻭﺩ ﻧﻤﯽ‌ﮐﻨﻨﺪ.

ﺩﺭ ﺩﻧﻴﺎﯼ ﺟﺒﺮ ﻣﺴﻠﮏِ ﮔﺬﺷﺘﻪ، ﺑﺎﻳﺪ ﻫﺮ ﭼﻴﺰﯼ ﻣﻮﺭﺩ ﺑﺮﺭﺳﯽ ﻗﺮﺍﺭ ﻣﯽ‌ﮔﺮﻓﺖ ﺗﺎ ﻣﺸﺨﺺ ﺷﻮﺩ ﮐﻪ ﺁﻳﺎ ﺍﺭﺯﺵ ﻗﺮﺍﺭ ﮔﺮﻓﺘﻦ ﺩﺭ ﭘﻴﺸﺨﻮﺍﻥ ﮐﺘﺎﺏ ﻓﺮﻭﺷﯽ‌ﻫﺎ ﻭ ﺍﻋﻄﺎﯼ ﺗﻮﺟﻪ ﺭﺍ ﺩﺍﺭﺩ ﻳﺎ ﺧﻴﺮ. ﻋﻬﺪﻩ ﺩﺍﺭ ﺍﻳﻦ ﺍﻣﺮ ﺧﻄﻴﺮ، ﻳﻌﻨﯽ ﺗﺼﻤﻴﻢ ﮔﻴﺮﯼ ﺑﺮﺍﯼ ﺁﻳﻨﺪﻩ ﺁﺛﺎﺭ، ﻧﺎﺷﺮﺍﻥ ﺑﻮﺩﻧﺪ؛ ﺁﻧﻬﺎ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻓﻴﻠﺘﺮ، ﺗﺼﻔﻴﻪ ﮐﻨﻨﺪﻩ، ﻃﻌﻢ ﺩﻫﻨﺪﻩ ﻭ ﺩﺍﻭﺭ ﻧﻬﺎﻳﯽ ﻋﻤﻞ ﻣﯽ‌ﮐﺮﺩﻧﺪ. ﭼﻨﺪﻳﻦ ﻫﺰﺍﺭ ﻧﺎﺷﺮ ﺩﺭ ﻃﻮﻝ ﺻﺪ‌ﻫﺎ ﺳﺎﻝ، ﺑﺎ ﺍﻳﻦ ﺭﻭﺵ ﺑﺎﻋﺚ ﺷﺪﻧﺪ ﺍﺭﺯﺵ‌ﻫﺎﯼ ﺯﻳﺎﺩﯼ ﺩﺭ ﻗﺎﻟﺐ ﮐﺘﺎﺏ‌ﻫﺎ ﻣﺎﻧﺪﮔﺎﺭ ﺷﻮﻧﺪ.

ﻭﻟﯽ ﺣﺎﻻ، ﺩﻳﮕﺮ ﭘﻴﺸﺨﻮﺍﻥ ﮐﺘﺎﺑﻔﺮﻭﺷﯽ‌ﻫﺎ ﺍﺯ ﺻﻔﺤﻪ ﺷﻄﺮﻧﺞ ﺑﻴﺮﻭﻥ ﺭﻓﺘﻪ‌ﺍﻧﺪ؛ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﻭ ﭼﺎﭖ ﺁﻧﻬﺎ ﺑﻪ ﺗﻘﺎﺿﺎﯼ ﻣﺨﺎﻃﺐ، ﺑﻪ ﺳﺮﻋﺖ ﺩﺭ ﺳﺮﺗﺎﺳﺮ ﺩﻧﻴﺎ ﺩﺭ ﺣﺎﻝ ﮔﺴﺘﺮﺵ ﻫﺴﺘﻨﺪ ﻭ ﻫﺮ ﮐﺴﯽ ﺑﺪﻭﻥ ﻧﻴﺎﺯ ﺑﻪ ﻧﺎﺷﺮﺍﻥ ﻗﺎﺩﺭ ﺑﻪ ﭼﺎﭘﺸﺎﻥ ﺍﺳﺖ؛ ﺍﻣﺮﻭﺯﻩ ﻫﻤﻪ ﻣﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﻢ ﺑﺎ ﻳﮏ ﺑﺮﻧﺎﻣﻪ ﭘﺮﺩﺍﺯﺵ ﻣﺘﻦ، ﺑﺮﺍﯼ ﺧﻮﺩﻣﺎﻥ ﮐﺘﺎﺑﯽ ﺑﻨﻮﻳﺴﻴﻢ ﻭ ﻣﻨﺘﺸﺮ ﮐﻨﻴﻢ.

ﺍﻟﺒﺘﻪ ﺍﻳﻦ ﺗﻐﻴﻴﺮﺍﺕ ﺑﻪ ﻣﻌﻨﺎﯼ ﺑﻬﺘﺮ ﺑﻮﺩﻥ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺍﻣﺮﻭﺯﯼ ﻧﺴﺒﺖ ﺑﻪ ﮐﺘﺐ ﺳﻨﺘﯽ ﻧﻴﺴﺘﻨﺪ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﺗﮑﻴﻪ ﮐﻤﺘﺮ ﺑﻪ ﻧﺎﺷﺮﺍﻥ، ﻣﺼﺮﻑ ﮐﻤﺘﺮ ﮐﺎﻏﺬ ﻭ ﻗﺘﻞ ﮐﻤﺘﺮ ﺩﺭﺧﺘﺎﻥ، ﺍﺯ ﺩﺳﺘﺎﻭﺭﺩﻫﺎﯼ ﺳﻴﺴﺘﻢ ﻧﺸﺮ ﺁﺯﺍﺩ ﮐﺘﺎﺏ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﺍﻳﻨﺘﺮﻧﺖ ﺍﺳﺖ. ﻣﺮﺩﻡ ﭼﻬﺎﺭ ﮔﻮﺷﻪ ﺩﻧﻴﺎ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻣﻴﻠﻴﻮﻥ‌ﻫﺎ ﻋﻨﻮﺍﻥ ﮐﺘﺎﺏ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﺭﺍ ﺁﻣﺎﺩﻩ ﮐﺮﺩﻩ ﻭ ﺑﺮﺍﯼ ﮐﺎﻣﭙﻴﻮﺗﺮ‌ﻫﺎ، ﮐﺘﺎﺑﺨﻮﺍﻥ‌ﻫﺎﯼ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ، ﺗﺒﻠﺖ‌ﻫﺎ ﻭ ﺗﻠﻔﻦ‌ﻫﺎﯼ ﻫﻮﺷﻤﻨﺪ ﺩﺭ ﺩﺳﺘﺮﺱ ﻗﺮﺍﺭ ﺩﻫﻨﺪ. ﻧﺴﺨﻪ‌ﻫﺎﯼ ﭼﺎﭘﯽ ﺍﻳﻦ ﮐﺘﺎﺏ‌ﻫﺎ ﻫﻢ ﻣﻌﻤﻮﻻً ﻗﺎﺑﻞ ﺳﻔﺎﺭﺵ ﻫﺴﺘﻨﺪ.

 

ﺗﻐﻴﻴﺮﺍﺕ ﺑﻨﻴﺎﺩﯼ ﺑﻪ ﻭﺍﺳﻄﻪ ﺍﻣﮑﺎﻥ ﻧﺸﺮ ﺷﺨﺼﯽ
ﺣﺘﯽ ﺍﮔﺮ ﺣﺲ ﻭ ﺗﺠﺮﺑﻪ ﺩﺭ ﺩﺳﺖ ﮔﺮﻓﺘﻦ ﻭ ﻣﻄﺎﻟﻌﻪ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺳﻨﺘﯽ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ، ﺑﺎﻳﺪ ﻗﺒﻮﻝ ﮐﻨﻴﺪ ﮐﻪ ﺩﻧﻴﺎﯼ ﺍﻣﺮﻭﺯ، ﻋﺮﺻﻪ ﺭﻭﻧﻖ ﻣﺤﺘﻮﺍﻫﺎﯼ ﺩﻳﺠﻴﺘﺎﻝ ﺍﺳﺖ. ﺷﺎﻳﺪ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﭼﺎﭘﯽ ﻫﺮﮔﺰ ﻧﻤﻴﺮﻧﺪ، ﻭﻟﯽ ﺯﻣﺎﻥ ﻫﻢ ﺭﻭ ﺑﻪ ﻋﻘﺐ -ﺑﻪ ﺟﺎﻳﯽ ﮐﻪ ﻓﻘﻂ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﭼﺎﭘﯽ ﺣﻀﻮﺭ ﺩﺍﺷﺘﻨﺪ- ﺑﺎﺯ ﻧﺨﻮﺍﻫﺪ ﮔﺸﺖ. ﭘﺮﻭﮊﻩ ﮔﻮﺗﻨﺒﺮﮒ، ﺁﻣﺎﺯﻭﻥ، ﺍﭘﻞ ﻭ ﻫﺰﺍﺭﺍﻥ ﺷﺮﮐﺖ، ﺑﺮﻧﺎﻣﻪ ﻧﻮﻳﺲ ﻭ ﮔﻴﮏ ﺩﻧﻴﺎﯼ ﻣﺪﺭﻥ ﺗﻮﺍﻧﺴﺘﻪ‌ﺍﻧﺪ ﺳﻪ ﺗﻐﻴﻴﺮ ﺑﻨﻴﺎﺩﯼ ﺭﺍ ﺑﻪ ﻭﺟﻮﺩ ﺁﻭﺭﻧﺪ:

ﺩﻣﮑﺮﺍﺗﻴﺰﻩ ﮐﺮﺩﻥ ﻧﺸﺮ؛ ﻫﺮ ﮐﺴﯽ ﺑﺎ ﻳﮏ ﮐﺎﻣﭙﻴﻮ‌ﺗﺮ ﻭ ﻧﺮﻡ ﺍﻓﺰﺍﺭ ﭘﺮﺩﺍﺯﺵ ﻣﺘﻦ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﺘﺎﺏ ﻣﻨﺘﺸﺮ ﮐﻨﺪ ﻭ ﻫﺮ ﮐﺴﯽ ﺑﺎ ﻳﮏ ﮐﺎﻣﭙﻴﻮ‌ﺗﺮ، ﺗﺒﻠﺖ ﻳﺎ ﺗﻠﻔﻦ ﻫﻮﺷﻤﻨﺪ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺁﻥ ﮐﺘﺎﺏ ﺭﺍ ﺑﺨﻮﺍﻧﺪ. ﻧﻮﺷﺘﻦ ﻭ ﺧﻮﺍﻧﺪﻥ ﺩﻳﮕﺮ ﺗﺤﺖ ﺳﻠﻄﻪ ﺛﺮﻭﺕ ﻳﺎ ﻗﺪﺭﺕ ﻧﻴﺴﺘﻨﺪ.

ﺣﺬﻑ ﻭﺍﺳﻄﻪ‌ﻫﺎﯼ ﺑﯽ‌ﻓﺎﻳﺪﻩ؛ ﻧﻬﺎﺩ‌ﻫﺎ ﻭ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﺑﻪ ﺍﺭﺯﺵ ﻭ ﻏﻨﺎﯼ ﮐﺘﺎﺏ‌ﻫﺎ ﻣﯽ‌ﺍﻓﺰﻭﺩﻧﺪ ﺩﺭ ﺟﺮﻳﺎﻥ ﻋﺮﺿﻪ ﺑﻪ ﺗﻘﺎﺿﺎ ﺑﺎﻗﯽ ﻣﯽ‌ﻣﺎﻧﻨﺪ. ﻭﻟﯽ ﻧﻬﺎﺩ‌ﻫﺎ ﻭ ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﻧﻘﺸﯽ ﺍﺯ ﺍﻳﻦ ﺟﻨﺲ ﻧﺪﺍﺭﻧﺪ، ﺣﺬﻑ ﻣﯽ‌ﺷﻮﻧﺪ. ﺗﻮﺯﻳﻊ ﮐﺘﺎﺏ‌ﻫﺎ ﺍﺯ ﺻﺎﺣﺐ ﺍﺛﺮ ﺑﻪ ﻣﺨﺎﻃﺒﻴﻦ «ﺩﻗﻴﻖ، ﻣﺴﺘﻘﻴﻢ، ﺳﺮﻳﻊ ﻭ ﺍﺭﺯﺍﻥ» ﻣﯽ‌ﺷﻮﺩ.

ﺍﺭﺍﺩﻩ ﻭ ﻗﺪﺭﺕ ﺗﻌﻴﻴﻦ؛ ﺍﻳﻦ ﺧﻮﺩ ﻧﻮﻳﺴﻨﺪﮔﺎﻥ ﻫﺴﺘﻨﺪ ﮐﻪ ﻣﻮﻓﻘﻴﺖ ﮐﺘﺎﺑﺸﺎﻥ ﺭﺍ ﺗﻌﻴﻴﻦ ﻣﯽ‌ﮐﻨﻨﺪ ﻭ ﺍﻳﻦ ﺧﻮﺩ ﻣﺨﺎﻃﺒﻴﻦ ﻫﺴﺘﻨﺪ ﮐﻪ ﮐﻴﻔﻴﺖ ﺍﺛﺮﯼ ﮐﻪ ﻗﺼﺪ ﺧﺮﻳﺪﺵ ﺭﺍ ﺩﺍﺭﻧﺪ، ﺗﺸﺨﻴﺺ ﺩﺍﺩﻩ ﻭ ﺳﭙﺲ ﺗﺼﻤﻴﻢ ﺑﻪ ﺧﺮﻳﺪ ﻳﺎ ﺍﻧﺼﺮﺍﻑ ﺍﺯ ﺁﻥ ﻣﯽ‌ﮔﻴﺮﻧﺪ. ﺍﻳﻦ ﺑﺪﺍﻥ ﻣﻌﻨﺎ ﻧﻴﺴﺖ ﮐﻪ ﻫﻤﻪ ﮐﺘﺎﺏ‌ﻫﺎ ﺧﻮﺏ ﺧﻮﺍﻫﻨﺪ ﺑﻮﺩ، ﻭﻟﯽ ﺯﻣﺎﻧﯽ ﮐﻪ ﺷﻤﺎ ﻣﺸﻐﻮﻝ «ﺍﻧﺘﺸﺎﺭ ﺷﺨﺼﯽ» ﮐﺘﺎﺏ ﺧﻮﺩ ﻫﺴﺘﻴﺪ، ﮐﺎﺳﻪ ﻭ ﮐﻮﺯﻩ‌ﻫﺎ ﺭﺍ ﺑﺮ ﺳﺮ ﮐﺲ ﺩﻳﮕﺮﯼ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺷﮑﺴﺖ. ﻣﺴﺌﻮﻟﻴﺖ ﮐﺘﺎﺏ ﺑﺎ ﺷﻤﺎ ﺍﺳﺖ ﻭ ﻣﻘﺒﻮﻟﻴﺖ ﺁﻥ ﺑﻴﻦ ﻣﺮﺩﻡ ﺍﺳﺖ ﮐﻪ ﺍﺯ ﺑﻴﻦ ﻣﻴﻠﻴﻮﻥ‌ﻫﺎ ﮐﺘﺎﺏ، ﭼﻨﺪ ﮐﺘﺎﺏ ﻣﻌﺪﻭﺩ ﺭﺍ ﺑﺮﺟﺴﺘﻪ ﻣﯽ‌ﮐﻨﺪ.

ﺍﮔﺮ ﻣﯽ‌ﺧﻮﺍﻫﻴﺪ ﺑﻪ ﻧﺸﺮ ﺷﺨﺼﯽ ﺭﻭﯼ ﺑﻴﺎﻭﺭﻳﺪ، ﺑﺎﻳﺪ ﺍﻳﻦ ﻣﻮﺍﺭﺩ ﺭﺍ ﺩﺭ ﻣﺮﮐﺰ ﺑﺎﻭﺭﻫﺎﯼ ﺧﻮﺩ ﻗﺮﺍﺭ ﺩﻫﻴﺪ.

 

ﻣﺰﺍﻳﺎﯼ ﺍﻧﺘﺸﺎﺭ ﺷﺨﺼﯽ ﮐﺘﺎﺏ
ﺣﺎﻻ ﮐﻪ ﺑﻪ ﺗﺮﺳﻴﻢ ﺗﺎﺑﻠﻮﯼ ﺑﺰﺭﮒ ﭼﺸﻢ ﺍﻧﺪﺍﺯ ﭘﺮﺩﺍﺧﺘﻴﻢ، ﺑﻪ ﺭﻳﺰ ﻣﺰﺍﻳﺎﯼ ﺍﻧﺘﺸﺎﺭ ﮐﺘﺎﺏ ﺑﻪ ﺻﻮﺭﺕ ﺩﻳﺠﻴﺘﺎﻝ ﻫﻢ ﺍﺷﺎﺭﻩ ﻣﯽ‌ﮐﻨﻴﻢ:

ﺩﺭ ﺩﺳﺖ ﮔﺮﻓﺘﻦ ﻃﺮﺍﺣﯽ ﻭ ﻣﺤﺘﻮﺍ؛ ﻭﻗﺘﯽ ﺷﺨﺼﺎً ﻳﮏ ﮐﺘﺎﺏ ﺭﺍ ﻣﻨﺘﺸﺮ ﮐﻨﻴﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻣﺤﺘﻮﺍ ﻭ ﻇﺎﻫﺮ ﺁﻥ ﺭﺍ ﮐﻨﺘﺮﻝ ﮐﻨﻴﺪ. ﺍﻳﻦ ﺳﻄﺢ ﺍﺯ ﮐﻨﺘﺮﻝ، ﻣﺎﻳﻪ ﻧﺸﺎﻁ ﻫﺮ ﺻﺎﺣﺐ ﺍﺛﺮﯼ ﺍﺳﺖ ﻭ ﺑﺎ ﻓﺮﺽ ﺍﻳﻨﮑﻪ ﺷﻤﺎ ﺩﺭﮎ ﺩﺭﺳﺘﯽ ﺍﺯ ﺁﻧﭽﻪ ﮐﻪ ﻣﺮﺩﻡ ﻣﺸﺘﺎﻕ ﺑﻪ ﺧﻮﺍﻧﺪﺵ ﻫﺴﺘﻨﺪ ﺩﺍﺭﻳﺪ ﻭ ﻧﻴﺰ ﺵَﻡّ ﻃﺮﺍﺣﻴﺘﺎﻥ ﺧﻮﺏ ﺍﺳﺖ، ﻣﯽ‌ﺗﻮﺍﻥ ﻧﺘﻴﺠﻪ ﮔﺮﻓﺖ ﮐﻪ ﮐﺘﺎﺏ ﭘﺮ ﻣﺤﺘﻮﺍ ﻭ ﺯﻳﺒﺎﻳﯽ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﻳﺎ ﻻ ﺍﻗﻞ ﮐﺘﺎﺑﯽ ﺧﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ ﮐﻪ ﺧﻮﺩﺗﺎﻥ ﺁﻥ ﺭﺍ ﺩﻭﺳﺖ ﺩﺍﺭﻳﺪ ﻭ ﺍﮔﺮ ﻫﻢ ﺩﻭﺳﺘﺶ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﮐﺴﯽ ﺭﺍ ﺟﺰ ﺧﻮﺩﺗﺎﻥ ﺳﺮﺯﻧﺶ ﻧﺨﻮﺍﻫﻴﺪ ﮐﺮﺩ.

ﺣﻖ ﺗﺄﻟﻴﻒ؛ ﺩﺭﻳﺎﻓﺖ ﺣﻖ ﺗﺄﻟﻴﻒ ﺍﺯ ﻧﺎﺷﺮﺍﻥ، ﺑﺮﺍﯼ ﺧﻮﺩ ﺩﺍﺳﺘﺎﻧﯽ ﺩﺍﺭﺩ. ﺍﮔﺮ ﻳﮏ ﻣﺘﺮﺟﻢ ﺑﺎﺷﻴﺪ، ﺣﺪﺍﮐﺜﺮ ۱۰ ﺩﺭﺻﺪ ﺍﺯ ﻗﻴﻤﺖ ﭘﺸﺖ ﺟﻠﺪ ﺍﻧﺘﻈﺎﺭ ﺷﻤﺎ ﺭﺍ ﻣﯽ‌ﮐﺸﺪ ﻭ ﺍﮔﺮ ﻣﺆﻟﻒ ﺑﺎﺷﻴﺪ، ﺍﻳﻦ ﺭﻗﻢ ﺑﻪ ۱۵ ﻳﺎ ﺩﺭ ﻣﻮﺭﺩ ﺍﻓﺮﺍﺩ ﺻﺎﺣﺐ ﻗﻠﻢ ﺑﻪ ۲۰ ﺩﺭﺻﺪ ﺑﺎﻟﻎ ﻣﯽ‌ﺷﻮﺩ. ﺍﻳﻦ ﺩﺭ ﺣﺎﻟﯽ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﺩﻧﻴﺎ، ۳۵ ﺗﺎ ۷۰ ﺩﺭﺻﺪ ﺍﺯ ﻗﻴﻤﺖ ﺍﻳﺒﻮﮎ‌ﻫﺎﯼ ﻓﺮﻭﺵ ﺭﻓﺘﻪ ﺩﺭ ﺁﻣﺎﺯﻭﻥ ﺑﻪ ﺻﺎﺣﺐ ﺍﺛﺮ ﻣﯽ‌ﺭﺳﺪ. ﺍﭘﻞ ﻫﻢ ۵۵ ﺩﺭﺻﺪ ﺍﺯ ﻗﻴﻤﺖ ﮐﺘﺎﺏ ﺭﺍ ﺑﻪ ﻣﺆﻟﻒ ﭘﺮﺩﺍﺧﺖ ﻣﯽ‌ﮐﻨﺪ.

ﺯﻣﺎﻥ ﺭﺳﻴﺪﻥ ﺑﻪ ﺑﺎﺯﺍﺭ؛ ﻳﮑﯽ ﺍﺯ ﺑﺰﺭﮒ‌ﺗﺮﻳﻦ ﮐﺎﺑﻮﺱ‌ﻫﺎﯼ ﻣﺘﺮﺟﻤﺎﻥ ﻳﺎ ﻣﺆﻟﻔﺎﻥ -ﺑﻪ ﺧﺼﻮﺹ ﺁﻧﻬﺎﻳﯽ ﮐﻪ ﺩﺍﺭﺍﯼ ﺭﻭﺍﺑﻂ ﺣﺴﻨﻪ ﺑﺎ ﻧﺎﺷﺮﺍﻥ ﻧﻴﺴﺘﻨﺪ- ﺭﺳﺎﻧﺪﻥ ﻫﺮ ﭼﻪ ﺳﺮﻳﻊ‌ﺗﺮ ﮐﺘﺎﺏ ﺑﻪ ﻣﺨﺎﻃﺒﺎﻥ ﺑﺎﻟﻘﻮﻩ ﺍﺳﺖ. ﺍﺯ ﻃﺮﻳﻖ ﻳﮏ ﻧﺎﺷﺮ ﺳﻨﺘﯽ، ۶ ﺗﺎ ۹ ﻣﺎﻩ ﺯﻣﺎﻥ ﺑﺮﺍﯼ ﭼﺎﭖ ﮐﺘﺎﺏ ﻭ ﺭﺳﺎﻧﺪﻥ ﺁﻥ ﺑﻪ ﭘﻴﺸﺨﻮﺍﻥ ﮐﺘﺎﺑﻔﺮﻭﺷﯽ‌ﻫﺎ ﻻﺯﻡ ﺍﺳﺖ ﻭ ﺩﺭ ﺿﻤﻦ، ﺁﻧﻬﺎ ﺣﺎﺿﺮ ﺑﻪ ﻋﺮﺿﻪ ﻧﺴﺨﻪ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﻗﺒﻞ ﺍﺯ ﻧﺴﺨﻪ ﭼﺎﭘﯽ ﻫﻢ ﻧﻴﺴﺘﻨﺪ. ﭼﻨﺎﻧﭽﻪ ﺗﺎ ﺑﻪ ﺣﺎﻝ ﮐﺘﺎﺑﯽ ﻣﻨﺘﺸﺮ ﻧﮑﺮﺩﻩ‌ﺍﻳﺪ، ﺑﺎﻳﺪ ﺑﺪﺍﻧﻴﺪ ﮐﻪ ﺍﺯ ﺳﺮ ﮔﺬﺭﺍﻧﺪﻥ ﺍﻳﻦ ﺩﻭﺭﻩ ﺯﻣﺎﻧﯽ، ﺍﺻﻼً ﺭﺍﺣﺖ ﻧﻴﺴﺖ.

ﺍﻧﻌﻄﺎﻑ ﮐﺎﺭﯼ؛ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﻧﺎﺷﺮ ﺷﺨﺼﯽ، ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﻗﺮﺍﺭﺩﺍﺩﻫﺎﯼ ﺁﺯﺍﺩﺍﻧﻪ‌ﺗﺮﯼ ﺑﺎ ﺷﺮﮐﺖ ﻳﺎ ﺳﺎﻳﺖ ﺗﻮﺯﻳﻊ ﮐﻨﻨﺪﻩ ﺍﺛﺮﺗﺎﻥ ﺑﺒﻨﺪﻳﺪ. ﺷﺎﻳﺪ ﻧﺪﺍﻧﻴﺪ ﻭﻟﯽ ﻧﺎﺷﺮﺍﻥ ﺳﻨﺘﯽ ﺩﺭ ﻗﺎﻟﺐ ﻓﺮﻭﺵ ﺍﻧﺒﻮﻩ، ﮐﺘﺎﺏ‌ﻫﺎﯼ ﭼﺎﭘﯽ ﺭﺍ ﺑﻪ ﺳﺎﺯﻣﺎﻥ‌ﻫﺎ، ﮐﺘﺎﺏ ﻓﺮﻭﺷﯽ‌ﻫﺎ ﻭ... ﻣﯽ‌ﻓﺮﻭﺷﻨﺪ. ﺷﻤﺎ ﻫﻢ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﮔﻮﺍﻫﯽ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﮐﺘﺎﺑﺘﺎﻥ ﺭﺍ ﺑﻪ ﻃﻮﺭ ﺍﻧﺒﻮﻩ ﺑﻪ ﻓﺮﻭﺵ ﺑﮕﺬﺍﺭﻳﺪ ﻳﺎ ﮐﺎﺭﻫﺎﯼ ﻣﺸﺎﺑﻪ. ﺑﻪ ﻫﺮ ﺻﻮﺭﺕ ﺁﺯﺍﺩﯼ ﺑﻴﺸﺘﺮﯼ ﻧﺴﺒﺖ ﺑﻪ ﻧﺸﺮ ﺳﻨﺘﯽ ﺩﺍﺭﻳﺪ.

ﺣﻘﻮﻕ ﺧﺎﺭﺟﯽ؛ ﺍﮔﺮ ﮐﺘﺎﺏ ﺷﻤﺎ ﻣﻮﻓﻖ ﺑﺎﺷﺪ، ﺍﻣﮑﺎﻥ ﻋﺮﺿﻪ ﺁﻥ ﺩﺭ ﮐﺸﻮﺭﻫﺎﯼ ﺧﺎﺭﺟﯽ ﻫﻢ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺍﻟﺒﺘﻪ ﺑﺎ ﻋﺒﻮﺭ ﺍﺯ ﻣﻮﺍﻧﻌﯽ ﮐﻪ ﺑﺮﺍﯼ ﻧﻘﻞ ﺍﻧﺘﻘﺎﻝ ﭘﻮﻝ ﺑﻪ ﺍﻳﺮﺍﻥ ﻭﺟﻮﺩ ﺩﺍﺭﺩ، ﺷﻤﺎ ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﮐﺸﻮﺭ‌ﻫﺎﯼ ﺩﻳﮕﺮ ﻳﺎ ﺍﺯ ﻃﺮﻳﻖ ﻧﺎﺷﺮﺍﻥ ﻓﺎﺭﺳﯽ ﺯﺑﺎﻥ ﮐﺸﻮﺭﻫﺎﯼ ﺩﻳﮕﺮ ﻧﻴﺰ ﺑﻪ ﺻﻮﺭﺕ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﻣﻨﺘﺸﺮ ﮐﻨﻴﺪ.

 

ﺍِﺷﮑﺎﻻﺕ ﺍﻧﺘﺸﺎﺭ ﺷﺨﺼﯽ ﮐﺘﺎﺏ
ﻫﺮ ﺭﻭﺷﯽ ﻣﺰﺍﻳﺎﻳﯽ ﺩﺍﺭﺩ ﻭ ﺍِﺷﮑﺎﻻﺗﯽ. ﺭﻭﺵ ﻣﻌﻘﻮﻝ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺭﺍﺣﺘﯽ ﺍﺯ ﮐﻨﺎﺭ ﺍِﺷﮑﺎﻻﺕ ﻧﮕﺬﺭﻳﻢ ﻭ ﺩﻳﺪ ﺩﺭﺳﺘﯽ ﻧﺴﺒﺖ ﺑﻪ ﺣﻴﻄﻪ‌ﺍﯼ ﮐﻪ ﺩﺭ ﺣﺎﻝ ﻭﺭﻭﺩ ﺑﻪ ﺁﻥ ﻫﺴﺘﻴﻢ ﮐﺴﺐ ﮐﻨﻴﻢ. ﺍﻧﺘﺸﺎﺭ ﺷﺨﺼﯽ ﮐﺘﺎﺏ ﺑﻪ ﺻﻮﺭﺕ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ، ﺍِﺷﮑﺎﻻﺗﯽ ﺭﺍ ﺩﺭ ﺯﻣﻴﻨﻪ‌ﻫﺎﯼ ﺯﻳﺮ ﺑﻪ ﻭﺟﻮﺩ ﻣﯽ‌ﺁﻭﺭﺩ:

ﭘﻴﺸﺮﻓﺖ؛ ﻓﻘﻂ ﺳﻪ ﻧﻮﻉ ﻣﺆﻟﻒ ﺩﺭ ﺩﻧﻴﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ: ﺁﻧﻬﺎﻳﯽ ﮐﻪ ﭘﻴﺸﺮﻓﺖ‌ﻫﺎﯼ ﺑﺰﺭﮒ ﮐﺮﺩﻩ‌ﺍﻧﺪ، ﺁﻧﻬﺎﻳﯽ ﮐﻪ ﺁﺭﺯﻭﯼ ﭘﻴﺸﺮﻓﺖ‌ﻫﺎﯼ ﺑﺰﺭﮒ ﺭﺍ ﺩﺍﺭﻧﺪ، ﻭ ﺁﻧﻬﺎﻳﯽ ﮐﻪ ﺑﻪ ﺩﺭﻭﻍ ﻣﯽ‌ﮔﻮﻳﻨﺪ ﺑﻪ ﺍﻳﻦ ﻃﻮﺭ ﭼﻴﺰ‌ﻫﺎ ﺍﻫﻤﻴﺖ ﻧﻤﯽ‌ﺩﻫﻨﺪ. ﺩﺍﺷﺘﻦ ﻧﺎﺷﺮ ﺍﻳﻦ ﺧﻮﺑﯽ ﺭﺍ ﺩﺍﺭﺩ ﮐﻪ ﺍﮔﺮ ﻭﺍﻗﻌﺎً ﺍﺛﺮ ﺩﺭﺧﻮﺭﯼ ﺩﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﻧﺎﺷﺮ ﮐﺎﺭ ﺷﻤﺎ ﺭﺍ ﭘﻴﺶ ﻣﯽ‌ﺑﺮﺩ، ﭼﻮﻥ ﭘﺎﯼ ﺳﻮﺩ ﺧﻮﺩﺵ ﻫﻢ ﺩﺭ ﻣﻴﺎﻥ ﺍﺳﺖ. ﺩﺭ ﻧﺘﻴﺠﻪ ﺍﮔﺮ ﮐﺎﺭ ﻧﺎﺷﺮ ﺭﻭﯼ ﮐﺘﺎﺏ ﻳﺎ ﮐﺘﺎﺏ‌ﻫﺎﯼ ﺷﻤﺎ ﺟﺪﯼ ﺑﺎﺷﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻴﺪ ﺑﺎ ﺧﻴﺎﻝ ﺭﺍﺣﺖ ﺭﻭﯼ ﺁﺛﺎﺭ ﺑﻌﺪﯼ ﮐﺎﺭ ﮐﻨﻴﺪ ﻭ ﻣﻮﺍﺭﺩﯼ ﺍﺯ ﻗﺒﻴﻞ ﻭﻳﺮﺍﺳﺘﺎﺭﯼ، ﺻﻔﺤﻪ ﺑﻨﺪﯼ، ﺣﺮﻭﻑ ﭼﻴﻨﯽ ﻭ ﻃﺮﺍﺣﯽ ﮐﻠﯽ ﮐﺘﺎﺏ ﺭﺍ ﺑﻪ ﺍﻓﺮﺍﺩ ﻣﺘﺨﺼﺺ ﺑﺴﭙﺎﺭﻳﺪ ﻭ ﺧﻮﺩﺗﺎﻥ ﺭﻭﯼ ﺁﺛﺎﺭ ﺑﻌﺪﯼ ﺗﻤﺮﮐﺰ ﮐﻨﻴﺪ. ﺣﺎﻝ ﺍﮔﺮ ﭘﻴﺸﺮﻓﺖ ﺷﻤﺎ ﺩﺭ ﮐﺎﺭ ﺑﺎ ﻧﺎﺷﺮ ﺑﻪ ﺩﻓﻌﺎﺕ ﺗﮑﺮﺍﺭ ﺷﻮﺩ، ﺁﻳﺎ ﺍﺭﺯﺵ ﻧﺪﺍﺭﺩ ﮐﻪ ﺍﺯ ﺧﻴﺮ ﻧﺎﺗﻮﺍﻧﯽ ﻧﺎﺷﺮ ﺩﺭ ﻓﺮﻭﺵ ﻧﺴﺨﻪ ﺍﻟﮑﺘﺮﻭﻧﻴﮏ ﮐﺘﺎﺏ ﺑﮕﺬﺭﻳﺪ؟

ﻭﻳﺮﺍﻳﺶ؛ ﺍﻧﺘﺸﺎﺭ ﺷﺨﺼﯽ ﮐﺘﺎﺏ، ﻳﮏ ﺭﻭﺵ ﺍﻧﻔﺮﺍﺩﯼ ﺍﺳﺖ. ﺑﻪ ﺧﺼﻮﺹ ﮐﻪ ﺷﻤﺎ ﺑﺎ ﻭﻳﺮﺍﺳﺘﺎﺭ ﺍﺭﺗﺒﺎﻁ ﻧﺨﻮﺍﻫﻴﺪ ﺩﺍﺷﺖ. ﻭﻳﺮﺍﺳﺘﺎﺭﻫﺎﯼ ﺧﻮﺏ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺭﺍﻫﻨﻤﺎ، ﻣﺸﺎﻭﺭ، ﺍﺳﺘﺎﺩ ﻭ ﻳﺎ ﺣﺘﯽ ﺭﻭﺍﻥ ﭘﺰﺷﮏ‌ﻫﺎﯼ ﺧﻮﺑﯽ ﺑﺮﺍﯼ ﻣﺆﻟﻔﺎﻥ ﺑﺎﺷﻨﺪ. ﺑﺎ ﺧﻮﺩﺗﺎﻥ ﺭﻭ ﺭﺍﺳﺖ ﺑﺎﺷﻴﺪ: ﻳﮏ ﻭﻳﺮﺍﺳﺘﺎﺭ ﺧﻮﺏ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﮐﺎﺭﯼ ﮐﻨﺪ ﮐﻪ ﮐﺘﺎﺏ ﺷﻤﺎ ﺑﻬﺘﺮ ﺑﻪ ﻧﻈﺮ ﺑﺮﺳﺪ. ﮔﺮﻓﺘﻦ ﺑﺎﺯﺧﻮﺭﺩ ﺍﺯ ﺩﻭﺳﺘﺎﻥ ﻭ ﺁﺷﻨﺎﻳﺎﻥ ﻫﻢ ﻣﻤﮑﻦ ﺍﺳﺖ، ﻭﻟﯽ ﺑﻪ ﺧﻮﺑﯽ ﺑﺎﺯﺧﻮﺭﺩﯼ ﮐﻪ ﻳﮏ ﻭﻳﺮﺍﺳﺘﺎﺭ ﺧﻮﺏ ﺑﻪ ﺷﻤﺎ ﻣﯽ‌ﺩﻫﺪ، ﻧﺨﻮﺍﻫﺪ ﺑﻮﺩ.

ﮔﺮﻭﻩ؛ ﻋﻼﻭﻩ ﺑﺮ ﻭﻳﺮﺍﺳﺘﺎﺭ، ﻧﺎﺷﺮ ﺍﻓﺮﺍﺩ ﺩﻳﮕﺮﯼ ﺭﺍ ﻫﻢ ﺑﺎ ﺷﻤﺎ ﻣﺮﺗﺒﻂ ﻣﯽ‌ﮐﻨﺪ. ﻣﻤﮑﻦ ﺍﺳﺖ ﮐﻪ ﺩﺭ ﮐﻞ ﺩﻭﺭﻩ ﮐﺎﺭ ﺗﻮﺟﻪ ﺻﺪ ﺩﺭ ﺻﺪﯼ ﺁﻧﻬﺎ ﺭﺍ ﻧﺪﺍﺷﺘﻪ ﺑﺎﺷﻴﺪ، ﻭﻟﯽ ﺑﻪ ﻫﺮ ﺣﺎﻝ ﺍﻓﺮﺍﺩ ﮐﺎﺭﺩﺍﻧﯽ ﻫﺴﺘﻨﺪ ﮐﻪ ﺑﻪ ﺷﻤﺎ ﮐﻤﮏ ﻣﯽ‌ﮐﻨﻨﺪ. ﺩﺳﺘﻴﺎﺭ ﻭﻳﺮﺍﺳﺘﺎﺭ، ﻭﻳﺮﺍﻳﺸﮕﺮ ﻣﺘﻦ، ﻃﺮﺍﺡ ﺟﻠﺪ، ﻣﺪﻳﺮ ﭼﺎﭖ ﻭ ﻋﻮﺍﻣﻞ ﻓﺮﻭﺵ ﻫﻤﻪ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﻧﻈﺮﺍﺕ ﺧﻮﺑﯽ ﺭﺍ ﺑﻪ ﮐﺎﺭ ﺗﺰﺭﻳﻖ ﻭ ﻣﻮﺟﺒﺎﺕ ﻣﻮﻓﻘﻴﺖ ﺑﻴﺸﺘﺮ ﮐﺘﺎﺏ ﺭﺍ ﻓﺮﺍﻫﻢ ﮐﻨﻨﺪ. ﺑﺪﻭﻥ ﻭﺟﻮﺩ ﺁﻧﻬﺎ، ﺩﻭ ﺭﺍﻩ ﺩﺭ ﭘﻴﺶ ﺩﺍﺭﻳﺪ: ﺧﻮﺩﺗﺎﻥ ﮐﺎﺭ ﺁﻧﻬﺎ ﺭﺍ ﺍﻧﺠﺎﻡ ﺩﻫﻴﺪ ﻳﺎ ﺍﻳﻨﮑﻪ ﺍﻓﺮﺍﺩ ﻣﺴﺘﻘﻠﯽ ﺭﺍ ﺑﺮﺍﯼ ﺍﻧﺠﺎﻡ ﺍﻳﻦ ﮐﺎﺭ‌ﻫﺎ ﺍﺳﺘﺨﺪﺍﻡ ﮐﻨﻴﺪ.

ﺍﻋﺘﺒﺎﺭ؛ ﺑﻌﺪ ﺍﺯ ﺧﻠﻖ ﺍﺑﺰﺍﺭﻫﺎﯼ ﻻﺯﻡ ﺑﺮﺍﯼ ﻧﺸﺮ ﺭﻭﻣﻴﺰﯼ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﮐﺎﻣﭙﻴﻮ‌ﺗﺮ ﮐﻪ ﺑﻪ ﻣﻴﺎﻧﻪ ﺩﻫﻪ ۸۰ ﻣﻴﻼﺩﯼ ﺑﺮ ﻣﯽ‌ﮔﺮﺩﺩ، ﺳﻴﻠﯽ ﺍﺯ ﻣﺤﺘﻮﺍ ﺑﻪ ﺭﺍﻩ ﺍﻓﺘﺎﺩ. ﺁﻥ ﺭﻭﺯ‌ﻫﺎ ﺑﻪ ﮐﺮﺍﺕ ﺗﺮﮐﻴﺐ ﺍﻳﻦ ﺩﻭ ﻋﺒﺎﺭﺕ ﺩﺭ ﮐﻨﺎﺭ ﻫﻢ ﺷﻨﻴﺪﻩ ﻣﯽ‌ﺷﺪ: «ﻧﺘﻮﺍﻧﺴﺘﻢ ﻧﺎﺷﺮ ﭘﻴﺪﺍ ﮐﻨﻢ.» ﻭ «ﺑﻪ ﺍﻧﺘﺸﺎﺭ ﺷﺨﺼﯽ ﺭﻭ ﺁﻭﺭﺩﻡ.» ﻫﺮ ﭼﻨﺪ ﺍﻣﺮﻭﺯﻩ ﻓﻘﺪﺍﻥ ﺍﻋﺘﺒﺎﺭ ﻭ ﻧﮕﺎﻩ ﺑﺪﺑﻴﻨﺎﻧﻪ ﺑﻪ ﻧﺸﺮ ﺷﺨﺼﯽ ﮐﻤﺘﺮ ﺷﺪﻩ ﻭﻟﯽ ﺑﻪ ﻫﺮ ﺣﺎﻝ ﺻﻔﺮ ﻫﻢ ﻧﻴﺴﺖ، ﺑﻪ ﺧﺼﻮﺹ ﺩﺭ ﺟﻮﺍﻣﻊ ﺩﺭ ﺣﺎﻝ ﺗﻮﺳﻌﻪ ﺍﻳﻦ ﺍﻣﺮ ﺑﻴﺸﺘﺮ ﺑﻪ ﭼﺸﻢ ﻣﯽ‌ﺁﻳﺪ.

ﺗﻮﺯﻳﻊ؛ ﺗﻌﺪﺍﺩ ﮐﺘﺎﺏ ﻓﺮﻭﺷﯽ‌ﻫﺎ ﻭ ﻧﺎﺷﺮﺍﻥ ﻫﺮ ﭼﻨﺪ ﺭﻭ ﺑﻪ ﻧﺰﻭﻝ ﺍﺳﺖ ﻭﻟﯽ ﻫﻨﻮﺯ ﺻﻔﺮ ﻧﺸﺪﻩ. ﻫﻨﻮﺯ ﺑﺨﺖ ﺩﻳﺪﻩ ﻭ ﭘﺴﻨﺪﻳﺪﻩ ﺷﺪﻥ ﮐﺘﺎﺏ ﺑﻪ ﻃﻮﺭ ﺍﺗﻔﺎﻗﯽ ﺩﺭ ﮐﺘﺎﺏ ﻓﺮﻭﺷﯽ‌ﻫﺎ ﻭﺟﻮﺩ ﺩﺍﺭﺩ. ﺗﻮﻳﻴﺘﺮ، ﻓﻴﺴﺒﻮﮎ ﻭ ﮔﻮﮔﻞ ﭘﻼﺱ ﺟﺎﻫﺎﯼ ﻣﻨﺎﺳﺒﯽ ﺑﺮﺍﯼ ﺑﻪ ﺭﺍﻩ ﺍﻧﺪﺍﺧﺘﻦ ﻫﻴﺎﻫﻮ ﭘﻴﺮﺍﻣﻮﻥ ﮐﺘﺎﺏ‌ﻫﺎ ﻫﺴﺘﻨﺪ ﻭﻟﯽ ﻧﺎﺷﺮ‌ﻫﺎ ﺭﺍ ﻫﻢ ﻧﻤﯽ‌ﺗﻮﺍﻥ ﺩﺳﺖ ﮐﻢ ﮔﺮﻓﺖ. ﺁﻧﻬﺎ ﻧﻴﺮﻭﻫﺎﯼ ﻓﺮﻭﺵ ﺧﻮﺩ ﺭﺍ ﺩﺍﺭﻧﺪ ﻭ ﺍﺯ ﺩﺭﻳﭽﻪ ﻣﺬﺍﮐﺮﻩ ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﮐﺘﺎﺏ‌ﻫﺎ ﺭﺍ ﺩﺭ ﺗﻌﺪﺍﺩ ﺍﻧﺒﻮﻩ ﺑﻪ ﻓﺮﻭﺵ ﺑﺮﺳﺎﻧﻨﺪ.

 

ﻧﻘﺸﻪ ﺳﻮﻡ
ﻧﻘﺸﻪ ﺍﻭﻝ ﺑﺮﺍﯼ ﺍﮐﺜﺮ ﻧﻮﻳﺴﻨﺪﮔﺎﻥ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﻳﮏ ﻃﺮﺡ ﺍﻭﻟﻴﻪ ﺑﻨﻮﻳﺴﻨﺪ، ﺍﺯ ﻃﺮﻳﻖ ﻣﺬﺍﮐﺮﻩ ﺷﺨﺼﯽ ﻳﺎ ﺑﻪ ﮐﻤﮏ ﻳﮏ ﻭﺍﺳﻄﻪ، ﺑﺎ ﻧﺎﺷﺮ ﺗﻤﺎﺱ ﺑﺮ ﻗﺮﺍﺭ ﮐﺮﺩﻩ ﻭ ﭘﺲ ﺍﺯ ﻗﺮﺍﺭﺩﺍﺩ ﺑﺴﺘﻦ، ﮐﺘﺎﺏ ﺭﺍ ﺑﻪ ﺍﺗﻤﺎﻡ ﺭﺳﺎﻧﺪﻩ ﻭ ﺑﻪ ﻓﺮﻭﺵ ﺑﮕﺬﺍﺭﻧﺪ.

ﺩﺭ ﺻﻮﺭﺗﯽ ﮐﻪ ﻧﻘﺸﻪ ﺍﻭﻝ ﺑﻪ ﮐﺎﺭ ﻧﻴﺎﻳﺪ، ﻧﻘﺸﻪ ﺩﻭﻡ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺳﺮﺍﻍ ﻧﺸﺮ ﺷﺨﺼﯽ ﮐﺘﺎﺏ ﺑﺮﻭﻳﺪ. ﻭﻟﯽ ﻧﻘﺸﻪ ﺳﻮﻣﯽ ﻫﻢ ﻭﺟﻮﺩ ﺩﺍﺭﺩ: ﺍﻳﻨﮑﻪ ﺑﺎ ﻧﺸﺮ ﻧﺴﺨﻪ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﺑﻪ ﻧﺎﺷﺮ ﺛﺎﺑﺖ ﮐﻨﻴﺪ ﮐﻪ ﮐﺘﺎﺏ ﺷﻤﺎ ﭘﺮ ﻣﺨﺎﻃﺐ ﺍﺳﺖ ﻭ ﺳﭙﺲ ﺁﻥ ﺭﺍ ﺑﻪ ﺻﻮﺭﺕ ﺣﺮﻓﻪ‌ﺍﯼ ﻣﻨﺘﺸﺮ ﮐﻨﻴﺪ.

ﺍﻳﻦ ﺭﻭﺵ ﻣﺰﺍﻳﺎﻳﯽ ﺩﺍﺭﺩ، ﺍﻓﺮﺍﺩﯼ ﮐﻪ ﮐﺘﺎﺏ ﺷﻤﺎ ﺭﺍ ﺧﻮﺍﻧﺪﻩ ﻭ ﭘﺴﻨﺪﻳﺪﻩ‌ﺍﻧﺪ، ﻣﯽ‌ﺗﻮﺍﻧﻨﺪ ﺑﻪ ﺳﺒﺐ ﺗﻤﺎﻳﻞ ﺑﺮﺍﯼ ﺩﺍﺷﺘﻦ ﻧﺴﺨﻪ ﭼﺎﭘﯽ، ﻣﺘﻘﺎﺿﻴﺎﻥ ﺑﺎﻟﻘﻮﻩ ﺁﻥ ﺑﺎﺷﻨﺪ. ﺿﻤﻦ ﺍﻳﻨﮑﻪ ﺍﻓﺮﺍﺩ ﺟﺪﻳﺪﯼ ﻫﻢ ﺍﺯ ﻣﺴﻴﺮﻫﺎﯼ ﺳﻨﺘﯽ ﻧﺎﺷﺮ ﺑﻪ ﺍﻳﻦ ﺟﻤﻊ ﺍﺿﺎﻓﻪ ﺧﻮﺍﻫﻨﺪ ﺷﺪ ﻭﻟﯽ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﻣﻤﮑﻦ ﺍﺳﺖ ﻧﺸﺮ ﺷﺨﺼﯽ ﮐﺘﺎﺏ ﺩﺭ ﮔﺎﻡ ﺍﻭﻝ، ﻣﺸﮑﻼﺗﯽ ﺭﺍ ﻫﻢ ﺑﺮﺍﯼ ﺷﻤﺎ ﺑﻪ ﻭﺟﻮﺩ ﺑﻴﺎﻭﺭﺩ ﮐﻪ ﻣﻮﻓﻖ ﺑﻪ ﺍﺟﺮﺍﯼ ﻧﻘﺸﻪ ﺳﻮﻡ ﻧﺸﻮﻳﺪ.

ﻧﺸﺮ ﺷﺨﺼﯽ ﺩﺭ ﺍﻳﺮﺍﻥ ﺑﻪ ﻣﻌﻨﺎﯼ ﺍﻳﻦ ﺍﺳﺖ ﮐﻪ ﺷﻤﺎ ﮐﺎﻧﺎﻝ ﺭﺳﻤﯽ ﺍﻧﺘﺸﺎﺭ ﮐﺘﺎﺏ ‏ﺭﺍ ﻧﺎﺩﻳﺪﻩ ﮔﺮﻓﺘﻪ‌ﺍﻳﺪ. ﭼﻨﺎﻧﭽﻪ ﻧﺴﺨﻪ ﺍﻟﮑﺘﺮﻭﻧﻴﮑﯽ ﮐﺘﺎﺏ ﺷﻤﺎ ﭘﺮ ﻣﺨﺎﻃﺐ ﺍﺯ ﮐﺎﺭ ﺩﺭ ﺁﻳﺪ ﻭ ﺑﻪ ﻗﻮﻝ ﻣﻌﺮﻭﻑ، ﻧﺎﻡ ﺷﻤﺎ ﻭ ﻋﻨﻮﺍﻥ ﮐﺘﺎﺑﺘﺎﻥ ﺑﺮ ﺳﺮ ﺯﺑﺎﻥ‌ﻫﺎ ﺑﻴﻔﺘﺪ، ﺍﻳﻦ ﻣﻌﺮﻭﻓﻴﺖ ﻣﯽ‌ﺗﻮﺍﻧﺪ ﺑﻌﺪ‌ﻫﺎ ﺑﻪ ﺳﺮﺍﻍ ﺷﻤﺎ ﺁﻣﺪﻩ ﻣﺎﻧﻊ ﺍﺯ ﺁﻥ ﺷﻮﺩ ﮐﻪ ﮐﺘﺎﺑﺘﺎﻥ ﺗﻮﺳﻂ ﺍﺭﮔﺎﻥ ﺩﻭﻟﺘﯽ ﻣﺮﺑﻮﻃﻪ ﺑﺮﺍﯼ ﭼﺎﭖ ﺭﺳﻤﯽ ﻭ ﺩﺭﻳﺎﻓﺖ ﺷﻨﺎﺳﻨﺎﻣﻪ ﺍﺯ ﮐﺘﺎﺑﺨﺎﻧﻪ ﻣﻠﯽ، ﻣﻮﺭﺩ ﺗﺄﻳﻴﺪ ﻗﺮﺍﺭ ﺑﮕﻴﺮﺩ.
