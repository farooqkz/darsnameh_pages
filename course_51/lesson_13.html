ﺩﺭﺱ ﺩﻫﻢ- ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ

ﻫﻨﺮ ﺳﻴﻨﻤﺎ، ﻫﻨﺮﯼ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﻫﻨﺮﻫﺎﯼ ﺩﻳﮕﺮ ﺍﺯ ﺟﻤﻠﻪ ﻧﻘﺎﺷﯽ، ﺭﻗﺺ ﻭ ﻣﻮﺳﻴﻘﯽ ﺍﺭﺗﺒﺎﻁ ﺗﻨﮕﺎﺗﻨﮕﯽ ﺩﺍﺭﺩ. ﻣﻮﺳﻴﻘﯽ، ﻳﮑﯽ ﺍﺯ ﻋﻨﺎﺻﺮ ﻣﻬﻢ ﻓﻴﻠﻢ ﻫﺎ ﺑﻪ ﻭﻳﮋﻩ ﻓﻴﻠﻢ ﻫﺎﯼ ﺩﺍﺳﺘﺎﻧﯽ ﺍﺳﺖ ﻭ ﻏﺎﻟﺒﺎ ﻓﻴﻠﻤﺴﺎﺯﺍﻥ ﺍﺯ ﺁﻥ ﺑﺮﺍﯼ ﺍﻧﺘﻘﺎﻝ ﺑﻬﺘﺮ ﺍﻳﺪﻩ ﻫﺎﯼ ﺧﻮﺩ ﻭ ﺍﻓﺰﺍﻳﺶ ﺗﺎﺛﻴﺮﺍﺕ ﺩﺭﺍﻣﺎﺗﻴﮏ ﻭ ﺣﺴﯽ ﻓﻴﻠﻢ ﺍﺳﺘﻔﺎﺩﻩ ﻣﯽ ﮐﻨﻨﺪ. 

ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﺍﺯ ﺩﻭﺭﺍﻥ ﺻﺎﻣﺖ ﺗﺎ ﺍﻣﺮﻭﺯ ﺭﺍﻩ ﺩﺭﺍﺯﯼ ﺭﺍ ﻃﯽ ﮐﺮﺩﻩ ﻭ ﺑﻪ ﻋﻨﻮﺍﻥ ﻋﻨﺼﺮ ﻣﮑﻤﻞ ﻭ ﺟﺪﺍﻳﯽ ﻧﺎﭘﺬﻳﺮ ﻳﮏ ﻓﻴﻠﻢ، ﻓﺮﻡ ﻫﺎ ﻭ ﻗﺎﻟﺐ ﻫﺎﯼ ﻣﺘﻨﻮﻉ ﺯﻳﺎﺩﯼ ﺭﺍ ﺑﻪ ﺧﻮﺩ ﺩﻳﺪﻩ ﺍﺳﺖ. ﺗﻐﻴﻴﺮﺍﺕ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﺩﺭ ﻃﯽ ﻳﮏ ﻗﺮﻥ، ﻧﺎﺷﯽ ﺍﺯ ﺗﻐﻴﻴﺮ ﺳﺒﮏ ﻫﺎ، ﻣﮑﺎﺗﺐ ﻭ ﮊﺍﻧﺮﻫﺎﯼ ﺳﻴﻨﻤﺎﻳﯽ ﺑﻮﺩﻩ ﺍﺳﺖ. ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﺑﺰﺭﮔﯽ ﺩﺭ ﺍﻳﻦ ﺣﻮﺯﻩ ﻭﺍﺭﺩ ﺷﺪﻧﺪ ﻭ ﺑﺎ ﺑﻪ ﮐﺎﺭﮔﻴﺮﯼ ﺧﻼﻗﻴﺖ، ﺍﺳﺘﻌﺪﺍﺩ ﻭ ﻣﻬﺎﺭﺕ ﻫﺎﯼ ﻣﻨﺤﺼﺮ ﺑﻪ ﻓﺮﺩﺷﺎﻥ، ﻧﻪ ﺗﻨﻬﺎ ﺑﻪ ﻏﻨﺎﯼ ﺁﺛﺎﺭ ﺳﻴﻨﻤﺎﻳﯽ ﻳﺎﺭﯼ ﺭﺳﺎﻧﺪﻧﺪ، ﺑﻠﮑﻪ ﺁﺛﺎﺭ ﻓﺮﺍﻣﻮﺵ ﻧﺸﺪﻧﯽ ﻭ ﻣﺎﻧﺪﮔﺎﺭﯼ ﺍﺯ ﺧﻮﺩ ﺑﻪ ﺟﺎ ﮔﺬﺍﺷﺘﻨﺪ.

ﺍﻣﺮﻭﺯ، ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﺗﻔﺎﻭﺕ ﻫﺎﯼ ﭼﺸﻤﮕﻴﺮﯼ ﺑﺎ ﮔﺬﺷﺘﻪ ﺩﺍﺭﺩ ﻭ ﺑﺎ ﭘﻴﭽﻴﺪﻩ ﺗﺮ ﺷﺪﻥ ﺯﺑﺎﻥ ﺳﻴﻨﻤﺎ ﻭ ﻓﺮﻡ ﻫﺎﯼ ﺑﺼﺮﯼ ﻭ ﺭﻭﺍﻳﺖ ﭘﺮﺩﺍﺯﯼ، ﻓﺮﻡ ﻫﺎﯼ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﻧﻴﺰ ﺗﻐﻴﻴﺮ ﮐﺮﺩﻩ ﻭ ﺷﮑﻞ ﻫﺎﯼ ﭘﻴﭽﻴﺪﻩ ﺗﺮﯼ ﺑﻪ ﺧﻮﺩ ﮔﺮﻓﺘﻪ ﺍﺳﺖ. ﺩﺭ ﺍﻳﻦ ﻓﺎﺻﻠﻪ، ﻧﻪ ﺗﻨﻬﺎ ﺩﺭﮎ ﺗﻤﺎﺷﺎﮔﺮﺍﻥ ﺳﻴﻨﻤﺎ ﺍﺯ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ، ﺗﻐﻴﻴﺮ ﮐﺮﺩﻩ ﺑﻠﮑﻪ ﺑﻪ ﻫﻤﺎﻥ ﻧﺴﺒﺖ، ﺧﻮﺍﺳﺖ ﻫﺎ ﻭ ﺍﻧﺘﻈﺎﺭﻫﺎﯼ ﺁﻧﻬﺎ ﺍﺯ ﺍﻳﻦ ﻫﻨﺮ ﻧﻴﺰ ﺩﮔﺮﮔﻮﻥ ﺷﺪﻩ ﺍﺳﺖ. ﺑﺎ ﺍﻳﻦ ﺣﺎﻝ ﻫﻨﻮﺯ ﻫﺴﺘﻨﺪ ﮐﺴﺎﻧﯽ ﮐﻪ ﻫﻤﺎﻧﻨﺪ ﺍﻳﮕﻮﺭ ﺍﺳﺘﺮﺍﻭﻳﻨﺴﮑﯽ، ﺷﺎﻥ ﻣﻮﺳﻴﻘﯽ ﺭﺍ ﺑﺎﻻﺗﺮ ﺍﺯ ﺳﻴﻨﻤﺎ ﻣﯽ ﺩﺍﻧﻨﺪ ﻭ ﻣﺜﻞ ﺍﻭ ﺑﺮ ﺍﻳﻦ ﺑﺎﻭﺭﻧﺪ ﮐﻪ « ﻣﻮﺳﻴﻘﯽ ﻫﻨﺮﯼ ﭘﺮ ﺍﺭﺝ ﺗﺮ ﺍﺯ ﺁﻥ ﺍﺳﺖ ﮐﻪ ﺑﻪ ﺧﺪﻣﺖ ﻫﻨﺮﻫﺎﯼ ﺩﻳﮕﺮ ﺩﺭﺁﻳﺪ.»

ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ، ﺳﻬﻢ ﺑﻪ ﺳﺰﺍﻳﯽ ﺩﺭ ﺗﻘﻮﻳﺖ ﺟﻨﺒﻪ ﻫﺎﯼ ﺣﺴﯽ، ﺑﻴﺎﻧﯽ ﻭ ﺩﺭﺍﻣﺎﺗﻴﮏ ﻓﻴﻠﻢ ﺩﺍﺭﺩ. ﺍﺯ ﺁﻧﺠﺎ ﮐﻪ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﺍﺳﺎﺳﺎ ﺑﺮﺍﯼ ﻳﮏ ﻓﻴﻠﻢ ﺧﺎﺹ ﺳﺎﺧﺘﻪ ﻣﯽ ﺷﻮﺩ، ﻃﺒﻴﻌﯽ ﺍﺳﺖ ﮐﻪ ﺑﺎﻳﺪ ﭘﻴﻮﻧﺪ ﺍﺭﮔﺎﻧﻴﮑﯽ ﺑﺎ ﺩﻧﻴﺎﯼ ﺁﻥ ﻓﻴﻠﻢ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ. ﺑﺎ ﺍﻳﻨﮑﻪ ﻗﺎﻋﺪﺗﺎ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﺭﺍ ﺑﺎﻳﺪ ﻫﻤﺮﺍﻩ ﻭ ﻫﻤﺰﻣﺎﻥ ﺑﺎ ﺗﻤﺎﺷﺎﯼ ﻓﻴﻠﻢ ﺷﻨﻴﺪ ﻭ ﻭﺍﺑﺴﺘﮕﯽ ﺁﻥ ﺑﻪ ﺩﺍﺳﺘﺎﻥ ﻭ ﺗﺼﺎﻭﻳﺮ ﻓﻴﻠﻢ، ﺍﺳﺘﻘﻼﻝ ﻫﻨﺮﯼ ﺁﻥ ﺭﺍ ﺧﺪﺷﻪ ﺩﺍﺭ ﺳﺎﺧﺘﻪ ﺍﺳﺖ، ﺍﻣﺎ ﻫﺴﺘﻨﺪ ﻗﻄﻌﺎﺕ ﺑﺴﻴﺎﺭﯼ ﮐﻪ ﺑﺮﺍﯼ ﻓﻴﻠﻢ ﻫﺎ ﻧﻮﺷﺘﻪ ﻭ ﺍﺟﺮﺍ ﺷﺪﻩ ﺍﻧﺪ ﻭﻟﯽ ﺑﻪ ﺧﺎﻃﺮ ﺯﻳﺒﺎﻳﯽ ﻭ ﻭﻳﮋﮔﯽ ﻫﺎﯼ ﺧﺎﺹ ﺧﻮﺩ، ﺑﻪ ﻃﻮﺭ ﻣﺴﺘﻘﻞ ﻭ ﺟﺪﺍ ﺍﺯ ﺁﻥ ﻓﻴﻠﻢ ﻫﺎ ﻧﻴﺰ ﻣﻄﺮﺡ ﻭ ﻗﺎﺑﻞ ﺷﻨﻴﺪﻥ ﺍﻧﺪ.

ﺗﺤﻠﻴﻞ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ، ﺍﺯ ﭘﻴﭽﻴﺪﻩ ﺗﺮﻳﻦ ﻣﺒﺎﺣﺚ ﻣﺮﺑﻮﻁ ﺑﻪ ﺳﻴﻨﻤﺎﺳﺖ ﻭ ﻧﻴﺎﺯﻣﻨﺪ ﺩﺍﻧﺶ ﻭﻳﮋﻩ ﺍﯼ ﺍﺳﺖ؛ ﺑﻪ ﺍﻳﻦ ﺩﻟﻴﻞ ﮐﻪ ﺧﻮﺩ ﻣﻮﺳﻴﻘﯽ ﻫﻤﺎﻧﻨﺪ ﻓﻴﻠﻢ، ﺯﺑﺎﻧﯽ ﭘﻴﭽﻴﺪﻩ ﻭ ﺩﺷﻮﺍﺭ ﺍﺳﺖ ﻭ ﺩﺭﮎ ﺩﺭﺳﺖ ﻭ ﻋﻠﻤﯽ ﺁﻥ ﺑﺮﺍﯼ ﺍﻓﺮﺍﺩ ﻏﻴﺮ ﻣﺘﺨﺼﺺ ﺗﺎ ﺣﺪ ﺯﻳﺎﺩﯼ ﺍﻣﮑﺎﻥ ﭘﺬﻳﺮ ﻧﻴﺴﺖ. ﺍﺯ ﺳﻮﯼ ﺩﻳﮕﺮ، ﻭﺍﮊﻩ ﻫﺎﯼ ﺗﺨﺼﺼﯽ ﻣﻮﺳﻴﻘﯽ ﻭ ﻓﻴﻠﻢ ﻧﻴﺰ ﻓﺮﺁﻳﻨﺪ ﺍﻳﻦ ﺩﺭﮎ ﺭﺍ ﺩﺷﻮﺍﺭﺗﺮ ﻣﯽ ﺳﺎﺯﻧﺪ.

ﺑﻪ ﮔﻔﺘﻪ ﺟﺮﺝ ﺑﺮﺕ، ﺁﻫﻨﮕﺴﺎﺯ ﻭ ﻣﺪﺭﺱ ﺑﺮﺟﺴﺘﻪ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ: «ﺩﺭ ﻫﺮ ﮔﻮﻧﻪ ﺑﺤﺚ ﺩﺭﺑﺎﺭﻩ ﭼﮕﻮﻧﮕﯽ ﮐﺎﺭﮐﺮﺩ ﻣﻮﺳﻴﻘﯽ ﺩﺭ ﻓﻴﻠﻢ، ﺍﺭﺟﺎﻉ ﻣﺪﺍﻭﻡ ﺑﻪ ﺩﺍﺳﺘﺎﻥ، ﻳﮏ ﺍﻣﺮ ﻧﺎﮔﺰﻳﺮ ﺍﺳﺖ.»

ﺩﺭ ﺑﺮﺭﺳﯽ ﻣﻮﺳﻴﻘﯽ ﻳﮏ ﻓﻴﻠﻢ ﻭ ﻗﻄﻌﺎﺕ ﻣﻬﻢ ﺁﻥ، ﺑﺎﻳﺪ ﭘﻴﻮﻧﺪﯼ ﺩﺭﻭﻧﯽ ﻭ ﺗﻨﮕﺎﺗﻨﮓ ﺁﻥ ﺑﺎ ﺗﻢ ﻫﺎ، ﻓﻀﺎﻫﺎ، ﻣﻮﻗﻌﻴﺖ ﻫﺎﯼ ﺯﻣﺎﻧﯽ، ﻣﮑﺎﻧﯽ ﻭ ﺩﺭﺍﻣﺎﺗﻴﮏ ﻭ ﺷﺨﺼﻴﺖ ﻫﺎﯼ ﺁﻥ ﻓﻴﻠﻢ ﺑﺮﺭﺳﯽ ﺷﻮﺩ.

ﻣﮑﺲ ﺍﺷﺘﺎﻳﻨﺮ، ﻣﻴﮑﻠﻮﺱ ﺭﻭﮊﺍ، ﻣﻮﺭﻳﺲ ﮊﺍﺭ، ﻣﺎﻳﮑﻞ ﻧﺎﻳﻤﻦ، ﻓﻴﻠﻴﭗ ﮔﻠﺲ ﻟﺌﻮﻧﺎﺭﺩ ﺭﻭﺯﻧﺘﺎﻝ، ﺟﺎﻥ ﻭﻳﻠﻴﺎﻣﺰ، ﺭﻳﻮﭼﯽ ﺳﺎﮐﻮﻣﺎﺗﻮ، ﺑﺮﻧﺎﺭﺩ ﻫﺮﻣﻦ، ﻧﻴﻨﻮ ﺭﻭﺗﺎ ﻭ ﺟﺮﯼ ﮔﻠﺪ ﺍﺳﻤﻴﺖ ﺍﺯ ﺑﺮﺟﺴﺘﻪ ﺗﺮﻳﻦ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﻓﻴﻠﻢ ﺍﻧﺪ. 

ﻣﻴﮑﻠﻮﺵ ﺭﻭﮊﺍ ﺩﺭ ﺯﻣﻴﻨﻪ ﻫﺎ ﻭ ﮊﺍﻧﺮﻫﺎﯼ ﻣﺨﺘﻠﻒ، ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﺳﺎﺧﺘﻪ ﺍﺳﺖ، ﺍﻣﺎ ﺷﻬﺮﺕ ﺍﻭ ﺑﻴﺸﺘﺮ ﺑﻪ ﺧﺎﻃﺮ ﺳﺎﺧﺘﻦ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﻫﺎﯼ ﺣﻤﺎﺳﯽ ﻭ ﺗﺎﺭﻳﺨﯽ ﻭ ﻏﺎﻟﺒﺎ ﺑﺎ ﺩﺭﻭﻧﻤﺎﻳﻪ ﻫﺎﯼ ﻣﺬﻫﺒﯽ ﻣﺜﻞ «ﺑﻦ ﻫﻮﺭ»، «ﺷﺎﻩ ﺷﺎﻫﺎﻥ» ﻭ «ﺍﻝ ﺳﻴﺪ» ﺍﺳﺖ.

ﺭﻭﮊﺍ ﻳﮑﯽ ﺍﺯ ﻣﻌﺪﻭﺩ ﺁﻫﻨﮕﺴﺎﺯﺍﻧﯽ ﺍﺳﺖ ﮐﻪ ﻣﻔﻬﻮﻡ ﻣﻮﺳﻴﻘﯽ ﺑﺮﺍﯼ ﻓﻴﻠﻢ ﻫﺎﯼ ﺣﻤﺎﺳﯽ ﻭ ﺗﺎﺭﻳﺨﯽ ﺭﺍ ﺩﺭﻳﺎﻓﺖ ﻭ ﺗﺸﺨﻴﺺ ﺩﺍﺩ ﮐﻪ ﻣﻮﺳﻴﻘﯽ ﻳﮏ ﻓﻴﻠﻢ ﺣﻤﺎﺳﯽ ﻳﺎ ﻣﺬﻫﺒﯽ ﭼﮕﻮﻧﻪ ﺑﺎﻳﺪ ﺑﺎﺷﺪ.

ﻣﻮﺭﻳﺲ ﮊﺍﺭ ﺁﻫﻨﮕﺴﺎﺯ ﻓﺮﺍﻧﺴﻮﯼ ﻧﻴﺰ ﻳﮑﯽ ﺍﺯ ﺑﺰﺭﮔﺘﺮﻳﻦ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﻓﻴﻠﻢ ﺍﺳﺖ. ﮐﺴﯽ ﮐﻪ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﻫﺎﻳﯽ ﭼﻮﻥ «ﻟﻮﺭﻧﺲ ﻋﺮﺑﺴﺘﺎﻥ»، «ﺩﮐﺘﺮ ﮊﻳﻮﺍﮔﻮ»، «ﮔﺬﺭﮔﺎﻫﯽ ﺑﻪ ﻫﻨﺪ»، «ﻣﺤﻤﺪ ﺭﺳﻮﻝ ﺍﻟﻠﻪ»، «ﺍﻧﺠﻤﻦ ﺷﺎﻋﺮﺍﻥ ﻣﺮﺩﻩ» ﻭ «ﺩﺧﺘﺮ ﺭﺍﻳﺎﻥ» ﺭﺍ ﺳﺎﺧﺘﻪ ﻭ ﻣﻬﺎﺭﺕ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﺳﺎﺧﺘﻦ ﻣﻮﺳﻴﻘﯽ ﺑﺮﺍﯼ ﻓﻴﻠﻢ ﻫﺎﯼ ﻋﻈﻴﻢ ﺣﻤﺎﺳﯽ ﻭ ﺩﺍﺳﺘﺎﻥ ﻫﺎﯼ ﺩﺭﺍﻣﺎﺗﻴﮏ ﭘﺮﻗﺪﺭﺕ ﻣﺜﻞ «ﺍﺳﺐ ﮐﻬﺮ ﺭﺍ ﺑﻨﮕﺮ» ﻭ «ﺷﺎﻫﺪ» ﻧﺸﺎﻥ ﺩﺍﺩﻩ ﺍﺳﺖ.

 

ﻋﻨﺎﺻﺮ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ
ﺭﻳﺘﻢ، ﻣﻠﻮﺩﯼ، ﻫﺎﺭﻣﻮﻧﯽ ﻭ ﺳﺎﺯﺑﻨﺪﯼ ﺩﺭ ﻣﻮﺳﻴﻘﯽ ﻳﮏ ﻓﻴﻠﻢ، ﻣﯽ ﺗﻮﺍﻧﺪ ﻭﺍﮐﻨﺶ ﻫﺎﯼ ﺍﺣﺴﺎﺳﯽ ﺑﻴﻨﻨﺪﻩ ﺭﺍ ﺑﻪ ﺷﺪﺕ ﺗﺤﺖ ﺗﺎﺛﻴﺮ ﻗﺮﺍﺭ ﺩﻫﺪ. ﻳﮏ ﻣﻠﻮﺩﯼ ﻳﺎ ﻋﺒﺎﺭﺕ ﻣﻮﺳﻴﻘﺎﻳﯽ ﻣﯽ ﺗﻮﺍﻧﺪ ﺗﺪﺍﻋﯽ ﮔﺮ ﻳﮏ ﮐﺎﺭﺍﮐﺘﺮ، ﻓﻀﺎ، ﻣﻮﻗﻌﻴﺖ ﻳﺎ ﻣﻔﻬﻮﻡ ﺧﺎﺻﯽ ﺩﺭ ﻓﻴﻠﻢ ﺑﺎﺷﺪ. ﺑﺎ ﺩﺧﻞ ﻭ ﺗﺼﺮﻑ ﺩﺭ ﭼﻨﻴﻦ ﻣﻮﺗﻴﻒ ﻫﺎﻳﯽ، ﻓﻴﻠﻤﺴﺎﺯ ﻣﯽ ﺗﻮﺍﻧﺪ ﺻﺤﻨﻪ ﻫﺎ ﺭﺍ ﺩﺭ ﻗﻴﺎﺱ ﺑﺎ ﻫﻢ ﻗﺮﺍﺭ ﺩﺍﺩﻩ، ﺍﻟﮕﻮﻫﺎﯼ ﺑﺴﻂ ﺭﺍ ﺗﺮﺳﻴﻢ ﮐﺮﺩﻩ ﻭ ﻣﻌﺎﻧﯽ ﺿﻤﻨﯽ ﺭﺍ ﺍﻟﻘﺎ ﮐﻨﺪ.

ﻣﺜﻼ ﻣﻮﺳﻴﻘﯽ ﮊﺭﮊ ﺩﻟﺮﻭ ﺑﺮﺍﯼ ﻓﻴﻠﻢ «ﮊﻭﻝ ﻭ ﮊﻳﻢ» ﺳﺎﺧﺘﻪ ﻓﺮﺍﻧﺴﻮﺍ ﺗﺮﻭﻓﻮ، ﭘﺎﺭﻳﺲ ﺳﺎﻝ ﻫﺎﯼ ۱۹۱۲ ﺗﺎ ۱۹۳۳ ﺭﺍ ﺑﺎﺯﺗﺎﺏ ﻣﯽ ﺩﻫﺪ ﮐﻪ ﻧﺸﺎﻥ ﺩﻫﻨﺪﻩ ﻭﻗﺎﻳﻊ ﻓﻴﻠﻢ ﺍﺳﺖ. ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﻣﻠﻮﺩﯼ ﻫﺎﻳﯽ ﮐﻪ ﺑﺮﺍﯼ ﺍﻳﻦ ﻓﻴﻠﻢ ﺍﺳﺘﻔﺎﺩﻩ ﺷﺪﻩ، ﺷﺒﻴﻪ ﺁﺛﺎﺭ ﺩﺑﻮﺳﯽ ﻭ ﺍﺭﻳﮏ ﺳﺎﺗﯽ، ﺩﻭ ﺗﻦ ﺍﺯ ﺑﺮﺟﺴﺘﻪ ﺗﺮﻳﻦ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﻓﺮﺍﻧﺴﻮﯼ ﺩﺭ ﺁﻥ ﺩﻭﺭﻩ ﺍﻧﺪ. ﺩﺭ ﻭﺍﻗﻊ ﮐﻞ ﻣﻮﺳﻴﻘﯽ ﻣﺘﺸﮑﻞ ﺍﺯ ﻣﻠﻮﺩﯼ ﻫﺎﻳﯽ ﺩﺭ ﻣﻴﺰﺍﻥ ۴/۳ ﻭ ﺑﺴﻴﺎﺭﯼ ﺍﺯ ﺁﻧﻬﺎ ﺩﺭ ﺿﺮﺏ ﻭﺍﻟﺲ ﻭ ﻫﻤﻪ ﺗﻢ ﻫﺎﯼ ﺍﺻﻠﯽ ﺩﺭ ﻻﻣﺎﮊﻭﺭ ﺍﺳﺖ ﻭ ﺍﻧﺘﺨﺎﺏ ﺍﻳﻦ ﺭﻳﺘﻢ ﻫﺎ ﻭ ﻣﻠﻮﺩﯼ ﻫﺎ، ﻭﺣﺪﺕ ﻓﻴﻠﻢ ﺭﺍ ﺗﻀﻤﻴﻦ ﻣﯽ ﮐﻨﺪ.

ﺗﻢ ﻫﺎﯼ ﻣﻮﺳﻴﻘﯽ ﺍﻳﻦ ﻓﻴﻠﻢ ﺩﺭ ﺍﺭﺗﺒﺎﻃﯽ ﺍﺭﮔﺎﻧﻴﮏ ﺑﺎ ﺭﻭﺍﻳﺖ ﻓﻴﻠﻢ ﺍﻧﺪ. ﻣﺜﻼ ﺗﺮﺍﻧﻪ «ﮔﺮﺩﺑﺎﺩ» ﮐﻪ ﺳﺎﺧﺘﻪ ﺳﺮﮊ ﺭﺿﻮﺍﻧﯽ، ﺁﻫﻨﮕﺴﺎﺯ ﺍﻳﺮﺍﻧﯽ ﺗﺒﺎﺭ ﻓﺮﺍﻧﺴﻮﯼ ﺍﺳﺖ، ﺩﺭ ﺍﻳﻦ ﻓﻴﻠﻢ، ﻋﻄﺶ ﮐﺎﺗﺮﻳﻦ (ﺑﺎ ﺑﺎﺯﯼ ﮊﺍﻥ ﻣﻮﺭﻭ)، ﺷﺨﺼﻴﺖ ﻣﺤﻮﺭﯼ ﺯﻥ ﻓﻴﻠﻢ ﺭﺍ ﺑﺮﺍﯼ ﺷﺎﺩﯼ ﻭ ﺁﺯﺍﺩﯼ ﺧﺎﺭﺝ ﺍﺯ ﻗﻴﺪ ﻭ ﺑﻨﺪﻫﺎﯼ ﺍﺧﻼﻗﯽ ﻭ ﺍﺟﺘﻤﺎﻋﯽ ﭘﺬﻳﺮﻓﺘﻪ ﺷﺪﻩ ﺟﺎﻣﻌﻪ ﻓﺮﺍﻧﺴﻪ ﺁﻥ ﺯﻣﺎﻥ ﻣﯽ ﺭﺳﺎﻧﺪ. ﻟﺤﻈﻪ ﻫﺎﻳﯽ ﮐﻪ ﺷﺨﺼﻴﺖ ﻫﺎ ﺩﺭ ﻳﮏ ﮐﺎﻓﻪ ﻧﺸﺴﺘﻪ ﺍﻧﺪ، ﺑﺎ ﻳﮏ ﻣﻮﺳﻴﻘﯽ ﺛﺎﺑﺖ ﻫﻤﺮﺍﻩ ﺍﺳﺖ ﮐﻪ ﺑﻌﺪ ﺍﺯ ﮔﺬﺷﺖ ﺳﺎﻝ ﻫﺎ، ﺍﺯ ﺣﺎﻟﺖ ﺍﺟﺮﺍﯼ ﺧﺸﮏ ﺁﻥ ﺑﻪ ﻭﺳﻴﻠﻪ ﭘﻴﺎﻧﻮﯼ ﮐﻮﮐﯽ ﺩﺭﺁﻣﺪﻩ ﻭ ﺑﻪ ﻧﺴﺨﻪ ﺟﺎﺯﯼ ﮐﻪ ﺑﻪ ﻭﺳﻴﻠﻪ ﻳﮏ ﭘﻴﺎﻧﻴﺴﺖ ﺳﻴﺎﻩ ﭘﻮﺳﺖ ﻧﻮﺍﺧﺘﻪ ﻣﯽ ﺷﻮﺩ، ﺗﻐﻴﻴﺮ ﻣﯽ ﮐﻨﺪ.

 

ﻣﻮﺳﻴﻘﯽ ﻣﻠﻮﺩﻳﮏ
ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﻣﯽ ﺗﻮﺍﻧﺪ ﻣﻠﻮﺩﻳﮏ ﺑﺎﺷﺪ، ﻳﻌﻨﯽ ﺁﻫﻨﮕﺴﺎﺯ ﺑﺮ ﻣﺒﻨﺎﯼ ﻣﻠﻮﺩﯼ ﺧﺎﺻﯽ ﮐﻪ ﺑﻪ ﺗﻢ ﻭ ﻓﻀﺎﯼ ﻓﻴﻠﻢ ﻳﺎ ﺭﻭﺣﻴﺎﺕ ﺷﺨﺼﻴﺖ ﻫﺎﯼ ﺍﺻﻠﯽ ﺁﻥ ﻧﺰﺩﻳﮏ ﺍﺳﺖ، ﺑﺮﺍﯼ ﻓﻴﻠﻢ ﻣﻮﺳﻴﻘﯽ ﻣﯽ ﺳﺎﺯﺩ.

ﮐﺎﺭﻫﺎﯼ ﻧﻴﻨﻮ ﺭﻭﺗﺎ، ﺁﻫﻨﮕﺴﺎﺯ ﺑﺰﺭﮒ ﺍﻳﺘﺎﻟﻴﺎﻳﯽ ﮐﻪ ﺑﺮﺍﯼ ﻓﻴﻠﻢ ﻫﺎﯼ ﻓﻴﻠﻤﺴﺎﺯﺍﻥ ﺑﺮﺟﺴﺘﻪ ﺍﯼ ﭼﻮﻥ ﻓﺪﺭﻳﮑﻮ ﻓﻠﻴﻨﯽ، ﻟﻮﮐﻴﻨﻮ ﻭﻳﺴﮑﻮﻧﺘﯽ ﻭ ﻓﺮﺍﻧﺴﻴﺲ ﻓﻮﺭﺩ ﮐﺎﭘﻮﻻ، ﻣﻮﺳﻴﻘﯽ ﺳﺎﺧﺘﻪ ﺍﺳﺖ، ﻧﻤﻮﻧﻪ ﻫﺎﯼ ﺩﺭﺧﺸﺎﻧﯽ ﺍﺯ ﺑﻪ ﮐﺎﺭﮔﻴﺮﯼ ﻣﻮﺳﻴﻘﯽ ﻣﻠﻮﺩﻳﮏ ﺩﺭ ﺳﻴﻨﻤﺎﺳﺖ. ﻧﻴﻨﻮ ﺭﻭﺗﺎ ﺩﺭ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ «ﭘﺪﺭﺧﻮﺍﻧﺪﻩ»، ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻠﻮﺩﯼ ﻫﺎﯼ ﻓﻮﻟﮑﻠﻮﺭﻳﮏ ﺍﻳﺘﺎﻟﻴﺎﻳﯽ، ﺗﻢ ﻫﺎﯼ ﺯﻳﺒﺎﻳﯽ ﺧﻠﻖ ﮐﺮﺩﻩ ﺍﺳﺖ ﮐﻪ ﺑﺎ ﻓﻀﺎﯼ ﮔﺎﻧﮕﺴﺘﺮﯼ ﻭ ﺩﺭ ﻋﻴﻦ ﺣﺎﻝ ﺧﺎﻧﻮﺍﺩﮔﯽ ﻭ ﮔﺮﻡ ﻓﻴﻠﻢ ﭘﺪﺭﺧﻮﺍﻧﺪﻩ ﻫﻤﺨﻮﺍﻧﯽ ﺩﻗﻴﻘﯽ ﺩﺍﺭﺩ.

ﻣﻮﺳﻴﻘﯽ ﺍﻟﻨﯽ ﮐﺎﺭﻳﻨﺪﺭﻭ، ﺁﻫﻨﮕﺴﺎﺯ ﻳﻮﻧﺎﻧﯽ ﮐﻪ ﺑﺎ ﺗﺌﻮ ﺁﻧﺠﻠﻮ ﭘﻮﻟﻮﺱ، ﻓﻴﻠﻤﺴﺎﺯ ﻓﻘﻴﺪ ﻳﻮﻧﺎﻧﯽ ﻫﻤﮑﺎﺭﯼ ﺩﺍﺷﺖ ﻧﻴﺰ ﺑﻪ ﺷﺪﺕ ﻣﻠﻮﺩﻳﮏ ﺍﺳﺖ. ﻣﻮﺳﻴﻘﯽ ﺍﻭ ﺩﺭ ﻋﻴﻦ ﻣﺪﺭﻥ ﺑﻮﺩﻥ، ﻭﺍﻣﺪﺍﺭ ﻣﻮﺳﻴﻘﯽ ﺳﻨﺘﯽ ﻭ ﻓﻮﻟﮑﻠﻮﺭﻳﮏ ﻳﻮﻧﺎﻧﯽ ﺍﺳﺖ. ﮐﺎﺭﻳﻨﺪﺭﻭ، ﺁﻻﺕ ﻣﻮﺳﻴﻘﯽ ﺳﻨﺘﯽ ﻳﻮﻧﺎﻥ ﻣﺜﻞ ﺳﻨﺘﻮﺭ، ﺁﮐﺎﺭﺩﺋﻮﻥ ﻭ ﻣﻨﺪﻭﻟﻴﻦ ﺭﺍ ﺑﻪ ﺷﮑﻠﯽ ﻣﺪﺭﻥ ﺩﺭ ﻣﻮﺳﻴﻘﻴﺎﺵ ﺑﻪ ﮐﺎﺭ ﻣﯽ ﮔﻴﺮﺩ ﻭ ﺩﺳﺖ ﺑﻪ ﺑﺎﺯﺁﻓﺮﻳﻨﯽ ﻣﺪﺭﻥ ﺭﻳﺘﻢ ﻫﺎ ﻭ ﻣﻠﻮﺩﯼ ﻫﺎﯼ ﺑﻮﻣﯽ ﻣﯽ ﺯﻧﺪ.

 

ﻣﻮﺳﻴﻘﯽ ﻏﻴﺮﻣﻠﻮﺩﻳﮏ
ﺍﻣﺎ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﻣﯽ ﺗﻮﺍﻧﺪ ﻏﻴﺮﻣﻠﻮﺩﻳﮏ ﺑﺎﺷﺪ ﻭ ﮐﻴﻔﻴﺖ ﺍﺗﻤﺴﻔﺮﻳﮏ ﺩﺍﺷﺘﻪ ﺑﺎﺷﺪ ﻭ ﺑﺮﺍﯼ ﺍﻳﺠﺎﺩ ﻳﮏ ﻓﻀﺎ ﻳﺎ ﺣﺲ ﺧﺎﺹ ﻣﺎﻧﻨﺪ ﻓﻀﺎﯼ ﻭﺣﺸﺖ، ﺗﻌﻠﻴﻖ ﻭ ﻫﻴﺠﺎﻥ، ﺍﺳﺘﻔﺎﺩﻩ ﺷﻮﺩ.

ﻣﻮﺳﻴﻘﯽ ﺟﺮﯼ ﮔﻠﺪ ﺍﺳﻤﻴﺖ ﺑﺮﺍﯼ ﻓﻴﻠﻢ «ﻃﺎﻟﻊ ﻧﺤﺲ» ﺳﺎﺧﺘﻪ ﺭﻳﭽﺎﺭﺩ ﺩﺍﻧﺮ ﮐﻪ ﻓﻴﻠﻤﯽ ﺩﺭ ﮊﺍﻧﺮ ﻭﺣﺸﺖ ﺍﺳﺖ، ﻣﻮﺳﻴﻘﯽ ﺍﯼ ﺍﺗﻤﺴﻔﺮﻳﮏ ﺍﺳﺖ ﻭ ﮔﻠﺪ ﺍﺳﻤﻴﺖ، ﻓﻀﺎﯼ ﮔﻮﺗﻴﮏ ﻭ ﺷﻴﻄﺎﻧﯽ ﻓﻴﻠﻢ ﺭﺍ ﺑﺎ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﻮﺳﻴﻘﯽ ﮐﺮﺍﻝ ﻭ ﺍﺭﮐﺴﺘﺮﺍﻝ ﺳﺎﺧﺖ. ﺻﺪﺍﯼ ﺧﺸﻦ ﻭ ﻧﺎﻫﻨﺠﺎﺭ ﺳﺎﺯﻫﺎﯼ ﺑﺮﻧﺠﯽ ﺩﺭ ﺍﻳﻦ ﻓﻴﻠﻢ، ﺣﻀﻮﺭ ﻧﻴﺮﻭﻫﺎﯼ ﺷﻴﻄﺎﻧﯽ ﺭﺍ ﺍﻋﻼﻡ ﻣﯽ ﮐﻨﺪ.

ﮐﺮﻳﺴﺘﻮﻑ ﮐﻮﻣﺪﺍ، ﺁﻫﻨﮕﺴﺎﺯ ﻟﻬﺴﺘﺎﻧﯽ ﻧﻴﺰ ﺑﺎ ﺗﺮﮐﻴﺐ ﻣﻮﺳﻴﻘﯽ ﮔﻮﺗﻴﮏ، ﺟﺎﺯ ﻭ ﭘﺎﭖ ﺩﻫﻪ ﺷﺼﺖ ﺩﺭ ﻓﻴﻠﻢ «ﺑﭽﻪ ﺭﺯﻣﺮﯼ» ﺳﺎﺧﺘﻪ ﺭﻭﻣﻦ ﭘﻮﻻﻧﺴﮑﯽ، ﺑﻪ ﻧﺘﻴﺠﻪ ﺷﮕﻔﺖ ﺍﻧﮕﻴﺰﯼ ﺩﺭ ﻓﻀﺎﺳﺎﺯﯼ ﺭﺳﻴﺪﻩ ﺍﺳﺖ.

ﺩﺭ ﻳﮑﯽ ﺍﺯ ﺻﺤﻨﻪ ﻫﺎﯼ ﻓﻴﻠﻢ ﮐﻪ ﻣﺮﺑﻮﻁ ﺑﻪ ﮐﺎﺑﻮﺱ ﺭﺯﻣﺮﯼ ﺍﺳﺖ ﻭ ﺷﮑﻠﯽ ﺳﻮﺭﺭﺋﺎﻟﻴﺴﺘﯽ ﺩﺍﺭﺩ، ﺍﻭ ﺭﺍ ﻣﯽ ﺑﻴﻨﻴﻢ ﮐﻪ ﺑﺮ ﺗﺨﺖ ﺩﺭﺍﺯ ﮐﺸﻴﺪﻩ ﻭ ﺍﺗﺎﻗﺶ ﺑﻪ ﺗﺪﺭﻳﺞ ﺩﺭ ﺁﺏ ﻏﺮﻕ ﻣﯽ ﺷﻮﺩ. ﻧﺎﮔﻬﺎﻥ ﺭﺯﻣﺮﯼ ﺧﻮﺩ ﺭﺍ ﺩﺭ ﻣﺤﺎﺻﺮﻩ ﺷﻴﻄﺎﻥ ﭘﺮﺳﺘﺎﻥ ﻭ ﺩﺭ ﻳﮏ ﻣﺮﺍﺳﻢ ﺁﺋﻴﻨﯽ ﺷﻴﻄﺎﻧﯽ ﻣﯽ ﺑﻴﻨﺪ ﮐﻪ ﺩﺭ ﺁﻥ ﺷﻮﻫﺮﺵ ﺑﻪ ﻳﮏ ﺣﻴﻮﺍﻥ ﺍﻫﺮﻳﻤﻨﯽ ﺗﺒﺪﻳﻞ ﺷﺪﻩ ﻭ ﺑﻪ ﺍﻭ ﺗﺠﺎﻭﺯ ﻣﯽ ﮐﻨﺪ. ﻗﻄﻌﻪ «ﺭﻭﻳﺎ (Dream)» ﮐﻪ ﺑﺮﺍﯼ ﺍﻳﻦ ﺻﺤﻨﻪ ﺳﺎﺧﺘﻪ ﺷﺪﻩ، ﺗﺮﮐﻴﺒﯽ ﺍﺯ ﻣﻮﺳﻴﻘﯽ ﺟﺎﺯ، ﺳﻴﻨﺘﯽ ﺳﺎﻳﺰﺭ ﻭ ﻧﺠﻮﺍﻫﺎﻳﯽ ﺍﺳﺖ ﮐﻪ ﮐﻴﻔﻴﺘﯽ ﮔﻮﺗﻴﮏ، ﺷﻮﻡ ﻭ ﺩﻟﻬﺮﻩ ﺁﻭﺭ ﺩﺍﺭﺩ ﻭ ﻓﻀﺎﯼ ﺷﻴﻄﺎﻧﯽ ﺣﺎﮐﻢ ﺑﺮ ﺻﺤﻨﻪ ﺭﺍ ﺗﻘﻮﻳﺖ ﻣﯽ ﮐﻨﺪ.

 

ﻣﻮﺳﻴﻘﯽ ﺗﻤﺎﺗﻴﮏ
ﺑﺮﺧﯽ ﻣﻮﺳﻴﻘﯽ ﻫﺎﯼ ﻓﻴﻠﻢ ﺑﺮ ﻣﺒﻨﺎﯼ ﺗﻢ ﺧﺎﺻﯽ ﺳﺎﺧﺘﻪ ﻣﯽ ﺷﻮﻧﺪ ﮐﻪ ﻳﺎ ﺩﺭﻭﻧﻤﺎﻳﻪ ﺍﺻﻠﯽ ﻓﻴﻠﻢ ﺭﺍ ﺗﺸﮑﻴﻞ ﻣﯽ ﺩﻫﻨﺪ ﻳﺎ ﻳﮏ ﺷﺨﺼﻴﺖ ﮐﻠﻴﺪﯼ ﺩﺭ ﻓﻴﻠﻢ ﺭﺍ ﻧﻤﺎﻳﻨﺪﮔﯽ ﻣﯽ ﮐﻨﻨﺪ. ﻳﻌﻨﯽ ﻫﺮ ﮔﺎﻩ ﺁﻥ ﻗﻄﻌﻪ ﺷﻨﻴﺪﻩ ﻣﯽ ﺷﻮﺩ، ﺗﻤﺎﺷﺎﮔﺮ ﻓﻮﺭﺍ ﻳﺎﺩ ﺁﻥ ﺷﺨﺼﻴﺖ ﻣﯽ ﺍﻓﺘﺪ.

ﻣﺜﻼ ﺩﺍﺭﻳﻮ ﻣﺎﺭﻳﺎﻧﻠﯽ ﺩﺭ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ «ﺗﺎﻭﺍﻥ» (ﺳﺎﺧﺘﻪ ﺟﻮ ﺭﺍﻳﺖ)، ﺗﻢ ﺍﺻﻠﯽ ﺭﺍ ﺑﺮﺍﺳﺎﺱ ﺭﻳﺘﻢ ﻭ ﻣﻮﺳﻴﻘﯽ ﻧﻬﻔﺘﻪ ﺩﺭ ﺻﺪﺍﯼ ﻣﺎﺷﻴﻦ ﺗﺤﺮﻳﺮ ﺁﻓﺮﻳﺪﻩ. ﺍﺑﺘﮑﺎﺭﯼ ﺯﻳﺒﺎ ﻭ ﺧﻼﻗﺎﻧﻪ ﮐﻪ ﺟﺪﺍ ﺍﺯ ﻣﻔﻬﻮﻡ ﺗﻤﺎﺗﻴﮏ ﺁﻥ، ﺭﻳﺘﻢ ﻭ ﺿﺮﺏ ﺁﻫﻨﮓ ﺯﻳﺒﺎﻳﯽ ﺑﻪ ﻓﻴﻠﻢ ﺑﺨﺸﻴﺪﻩ ﺍﺳﺖ. ﺻﺪﺍﯼ ﻣﺎﺷﻴﻦ ﺗﺤﺮﻳﺮ ﺑﻪ ﻋﻨﻮﺍﻥ ﻳﮏ ﺳﺎﺯ ﻭ ﺁﻟﺖ ﻣﻮﺳﻴﻘﯽ ﺑﻪ ﮐﺎﺭ ﮔﺮﻓﺘﻪ ﺷﺪﻩ ﻭ ﻣﺜﻞ ﺿﺮﺑﻪ ﺳﺎﺯﻫﺎﯼ ﮐﻮﺑﻪ ﺍﯼ ﻋﻤﻞ ﻣﯽ ﮐﻨﺪ. ﻫﺮ ﺑﺎﺭ ﮐﻪ ﺻﺪﺍﯼ ﻣﺎﺷﻴﻦ ﺗﺤﺮﻳﺮ ﺑﻪ ﮔﻮﺵ ﻣﯽ ﺭﺳﺪ، ﻣﻮﺳﻴﻘﯽ ﺍﻧﺮﮊﯼ ﻭ ﺗﺤﺮﮎ ﻓﻮﻕ ﺍﻟﻌﺎﺩﻩ ﺍﯼ ﻣﯽ ﮔﻴﺮﺩ. ﺗﻢ ﺍﺻﻠﯽ ﻣﻮﺳﻴﻘﯽ ﺍﻳﻦ ﻓﻴﻠﻢ ﺑﺎ ﻋﻨﻮﺍﻥ «ﺑﺮﺍﻳﻨﯽ» ﮐﻪ ﺑﺮﺍﺳﺎﺱ ﺷﺨﺼﻴﺖ ﺭﺍﻭﯼ ﻓﻴﻠﻢ ﻧﻮﺷﺘﻪ ﺷﺪﻩ، ﺍﺭﺗﺒﺎﻁ ﺍﺭﮔﺎﻧﻴﮑﯽ ﺑﺎ ﺩﺭﻭﻧﻤﺎﻳﻪ ﻓﻴﻠﻢ ﺩﺍﺭﺩ. ﺍﻓﮑﺖ ﺻﻮﺗﯽ ﻣﺎﺷﻴﻦ ﺗﺤﺮﻳﺮ، ﻳﮏ ﻋﻨﺼﺮ ﺍﺩﺑﯽ ﺍﺳﺖ ﻭ ﺗﺎﮐﻴﺪﯼ ﺍﺳﺖ ﺑﺮ ﺣﻀﻮﺭ ﻧﻮﻳﺴﻨﺪﻩ ﺑﻪ ﻋﻨﻮﺍﻥ ﺭﺍﻭﯼ ﺍﺻﻠﯽ ﻓﻴﻠﻢ.

ﻣﺎﺭﻳﺎﻧﻠﯽ ﺩﺭ ﻭﺍﻗﻊ ﺍﻳﺪﻩ ﺍﺳﺘﻔﺎﺩﻩ ﺍﺯ ﻣﺎﺷﻴﻦ ﺗﺤﺮﻳﺮ ﺭﺍ ﺍﺯ ﺷﺨﺼﻴﺖ ﻫﺎﯼ ﺭﻭﺑﯽ ﻭ ﺑﺮﺍﻳﻨﯽ ﮔﺮﻓﺘﻪ ﮐﻪ ﻫﺮ ﺩﻭ ﺩﺭ ﻓﻴﻠﻢ ﺑﺎ ﻣﺎﺷﻴﻦ ﺗﺤﺮﻳﺮ ﮐﺎﺭ ﻣﯽ ﮐﻨﻨﺪ. ﺑﺮﺍﻳﻨﯽ ﺑﺎ ﺁﻥ ﻧﻤﺎﻳﺸﻨﺎﻣﻪ ﻭ ﺩﺍﺳﺘﺎﻥ ﻫﺎﻳﺶ ﺭﺍ ﻣﯽ ﻧﻮﻳﺴﺪ ﻭ ﺭﻭﺑﯽ ﻧﺎﻣﻪ ﻫﺎﯼ ﻋﺎﺷﻘﺎﻧﻪ ﺍﺵ ﺭﺍ ﺑﻪ ﺳﻴﺴﻴﻠﻴﺎ. ﺗﻢ ﺍﺻﻠﯽ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﺑﺎ ﺗﻠﻖ ﺗﻮﻟﻮﻕ ﺿﺮﺑﻪ ﻫﺎﯼ ﻣﺎﺷﻴﻦ ﺗﺤﺮﻳﺮ ﺷﺮﻭﻉ ﻣﯽ ﺷﻮﺩ، ﺑﻌﺪ ﺑﺎ ﺿﺮﺑﻪ ﻫﺎﯼ ﺗﻨﺪ ﭘﻴﺎﻧﻮ ﺩﻧﺒﺎﻝ ﺷﺪﻩ ﻭ ﺑﻪ ﺗﺪﺭﻳﺞ ﺑﻪ ﻳﮏ ﻣﻠﻮﺩﯼ ﺯﻳﺒﺎ ﺗﺒﺪﻳﻞ ﻣﯽ ﺷﻮﺩ

ﺍﻣﺎ ﺑﺮﺧﯽ ﺍﺯ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ، ﺍﺯ ﻣﻮﺳﻴﻘﯽ ﺗﻤﺎﺗﻴﮏ ﺍﺳﺘﻔﺎﺩﻩ ﻧﻤﯽ ﮐﻨﻨﺪ. ﺟﺎﻧﯽ ﮔﺮﻳﻦ ﻭﻭﺩ ﺑﺮﺍﯼ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ «ﺧﻮﻥ ﺑﻪ ﭘﺎ ﻣﯽ ﺷﻮﺩ»، ﺍﺯ ﺳﺎﺧﺘﻦ ﻣﻮﺳﻴﻘﯽ ﺗﻤﺎﺗﻴﮏ ﻭ ﻣﻠﻮﺩﻳﮏ ﭘﺮﻫﻴﺰ ﮐﺮﺩﻩ ﺍﺳﺖ. ﻳﻌﻨﯽ ﻣﻮﺳﻴﻘﯽ ﺍﻳﻦ ﻓﻴﻠﻢ ﺑﺮﺍﺳﺎﺱ ﺗﻢ ﺧﺎﺻﯽ ﺳﺎﺧﺘﻪ ﻧﺸﺪﻩ، ﺑﻠﮑﻪ ﺗﻨﻬﺎ ﺑﺎ ﻫﺪﻑ ﻓﻀﺎﺳﺎﺯﯼ ﻭ ﺍﻟﻘﺎﯼ ﺣﺲ ﻭ ﺣﺎﻝ ﻣﺸﺨﺼﯽ ﺧﻠﻖ ﺷﺪﻩ ﺍﺳﺖ.

 

ﻣﻮﺳﻴﻘﯽ ﻣﻴﻨﯽ ﻣﺎﻟﻴﺴﺘﯽ
ﻧﻮﻉ ﺩﻳﮕﺮﯼ ﺍﺯ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﻭﺟﻮﺩ ﺩﺍﺭﺩ ﮐﻪ ﺑﻪ ﻣﻮﺳﻴﻘﯽ ﻣﻴﻨﯽ ﻣﺎﻟﻴﺴﺘﯽ ﻣﺸﻬﻮﺭ ﺍﺳﺖ ﻭ ﻓﻴﻠﻴﭗ ﮔﻼﺱ ﻭ ﻣﺎﻳﮑﻞ ﻧﺎﻳﻤﻦ ﻧﻤﺎﻳﻨﺪﮔﺎﻥ ﺍﻳﻦ ﻧﻮﻉ ﻣﻮﺳﻴﻘﯽ ﺩﺭ ﺳﻴﻨﻤﺎ ﺑﻪ ﺷﻤﺎﺭ ﻣﯽ ﺭﻭﻧﺪ.

ﻫﺎﺭﻣﻮﻧﯽ ﺛﺎﺑﺖ، ﺗﮑﺮﺍﺭ ﻣﻮﺗﻴﻒ ﻫﺎﯼ ﮐﻮﺗﺎﻩ ﻭ ﺗﺎﮐﻴﺪ ﺑﺮ ﺭﻳﺘﻢ ﻫﺎﯼ ﻣﺘﻨﺎﻭﺏ ﻭ ﻣﻨﻈﻢ، ﻭ ﺗﺎﺛﻴﺮ ﻣﻮﺳﻴﻘﯽ ﻣﻠﻞ ﺩﻳﮕﺮ (ﺑﻪ ﻭﻳﮋﻩ ﻣﻮﺳﻴﻘﯽ ﺷﺮﻕ، ﺧﺎﻭﺭﻣﻴﺎﻧﻪ ﻭ ﺁﻓﺮﻳﻘﺎ) ﺍﺯ ﻣﻬﻤﺘﺮﻳﻦ ﻋﻨﺎﺻﺮ ﻣﻮﺳﻴﻘﯽ ﻣﻴﻨﻴﻤﺎﻟﻴﺴﺘﯽ ﺍﺳﺖ.

ﻣﺎﻳﮑﻞ ﻧﺎﻳﻤﻦ، ﺁﻫﻨﮕﺴﺎﺯ ﺑﺮﻳﺘﺎﻧﻴﺎﻳﯽ ﻭ ﺳﺎﺯﻧﺪﻩ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﻫﺎﻳﯽ ﭼﻮﻥ «ﻏﺮﻕ ﺷﺪﻥ ﺑﻪ ﺗﺮﺗﻴﺐ ﺷﻤﺎﺭﻩ»، «ﻗﺮﺍﺭﺩﺍﺩ ﻃﺮﺍﺡ»، «ﮐﺘﺎﺏ ﭘﺮﺍﺳﭙﺮﻭ» ﻭ ﻓﻴﻠﻢ ﻣﺸﻬﻮﺭ «ﭘﻴﺎﻧﻮ»، ﻳﮑﯽ ﺍﺯ ﭼﻬﺮﻩ ﻫﺎﯼ ﻣﻄﺮﺡ ﻣﻮﺳﻴﻘﯽ ﻓﻴﻠﻢ ﻣﻴﻨﯽ ﻣﺎﻟﻴﺴﺘﯽ ﺍﺳﺖ. ﺩﺭ ﻭﺍﻗﻊ ﺍﻭ ﮐﺴﯽ ﺍﺳﺖ ﮐﻪ ﺑﺮﺍﯼ ﻧﺨﺴﺘﻴﻦ ﺑﺎﺭ ﻭﺍﮊﻩ ﻣﻴﻨﻴﻤﺎﻟﻴﺴﻢ ﺭﺍ ﻭﺍﺭﺩ ﺩﻧﻴﺎﯼ ﻣﻮﺳﻴﻘﯽ ﮐﺮﺩ.

ﺍﺯ ﻣﻴﺎﻥ ﺁﻫﻨﮕﺴﺎﺯﺍﻥ ﻓﻴﻠﻢ ﺑﺮﺟﺴﺘﻪ ﺍﻳﺮﺍﻧﯽ ﻧﻴﺰ ﺑﺎﻳﺪ ﺍﺯ ﺍﺳﻔﻨﺪﻳﺎﺭ ﻣﻨﻔﺮﺩﺯﺍﺩﻩ، ﻭﺍﺭﻭﮊﺍﻥ، ﻓﺮﻳﺪﻭﻥ ﻧﺎﺻﺮﯼ، ﺷﻴﺪﺍ ﻗﺮﻫﭽﻪ ﺩﺍﻏﯽ، ﺍﺣﻤﺪ ﭘﮋﻣﺎﻥ، ﺑﺎﺑﮏ ﺑﻴﺎﺕ ﻭ ﻣﺠﻴﺪ ﺍﻧﺘﻈﺎﻣﯽ ﻳﺎﺩ ﮐﺮﺩ.
